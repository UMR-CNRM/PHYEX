24 !!**** *ICE_ADJUST* - compute the ajustment of water vapor in mixed-phase
25 !! clouds
26 !!
27 !! PURPOSE
28 !! -------
29 !! The purpose of this routine is to compute the fast microphysical sources
30 !! through a saturation ajustement procedure in case of mixed-phase clouds.
31 !!
32 !!
33 !!** METHOD
34 !! ------
35 !! Langlois, Tellus, 1973 for the cloudless version.
36 !! When cloud water is taken into account, refer to book 1 of the
37 !! documentation.
38 !!
39 !!
40 !!

47 !! ------------------

50 !! XMD,XMV ! Molar mass of dry air and molar mass of vapor
51 !! XRD,XRV ! Gaz constant for dry air, gaz constant for vapor
52 !! XCPD,XCPV ! Cpd (dry air), Cpv (vapor)
53 !! XCL ! Cl (liquid)
54 !! XCI ! Ci (ice)
55 !! XTT ! Triple point temperature
56 !! XLVTT ! Vaporization heat constant
57 !! XLSTT ! Sublimation heat constant
58 !! XALPW,XBETAW,XGAMW ! Constants for saturation vapor over liquid
59 !! ! pressure function
60 !! XALPI,XBETAI,XGAMI ! Constants for saturation vapor over ice
61 !! ! pressure function

63 !! CCONF
64 !! Module MODD_BUDGET:
65 !! NBUMOD
66 !! CBUTYPE
67 !! LBU_RTH
68 !! LBU_RRV
69 !! LBU_RRC
70 !! LBU_RRI
71 !!
72 !!
73 !! REFERENCE
74 !! ---------
75 !! Book 1 and Book2 of documentation ( routine ICE_ADJUST )

150 LOGICAL, INTENT(IN) :: LMFCONV ! =SIZE(PMFCONV)!=0
151 REAL, DIMENSION(MERGE(D%NIJT,0,LMFCONV),& MERGE(D%NKT,0,LMFCONV)), INTENT(IN) :: PMFCONV ! convective mass flux
152 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PPABST ! Absolute Pressure at t

155 !
156 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PCF_MF ! Convective Mass Flux Cloud fraction
157 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRC_MF ! Convective Mass Flux liquid mixing ratio
158 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRI_MF ! Convective Mass Flux ice mixing ratio

166 LOGICAL, INTENT(IN) :: OCOMPUTE_SRC
167 REAL, DIMENSION(MERGE(D%NIJT,0,OCOMPUTE_SRC),& MERGE(D%NKT,0,OCOMPUTE_SRC)), INTENT(OUT) :: PSRCS ! Second-order flux
168  ! s'rc'/2Sigma_s2 at time t+1
169  ! multiplied by Lambda_3
170 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PCLDFR ! Cloud fraction
171 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PICLDFR ! ice cloud fraction
172 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PWCLDFR ! water or mixed-phase cloud fraction
173 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PSSIO ! Super-saturation with respect to ice in the
174  ! supersaturated fraction
175 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PSSIU ! Sub-saturation with respect to ice in the
176  ! subsaturated fraction
177 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PIFR ! Ratio cloud ice moist part to dry part
178 !
179 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT):: PRIS ! Cloud ice m.r. at t+1
180 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRR ! Rain water m.r. to adjust
181 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRI ! Cloud ice m.r. to adjust
182 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRS ! Aggregate m.r. to adjust
183 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRG ! Graupel m.r. to adjust
184 TYPE(tbudgetdata), DIMENSION(KBUDGETS), INTENT(INOUT):: TBUDGETS
185 INTEGER, INTENT(IN) :: KBUDGETS
186 REAL, DIMENSION(D%NIJT), OPTIONAL, INTENT(IN) :: PICE_CLD_WGT
187 REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(IN) :: PRH ! Hail m.r. to adjust
188 REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(OUT) :: POUT_RV ! Adjusted value
189 REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(OUT) :: POUT_RC ! Adjusted value
190 REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(OUT) :: POUT_RI ! Adjusted value
191 REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(OUT) :: POUT_TH ! Adjusted value
192 REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(OUT) :: PHLC_HRC
193 REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(OUT) :: PHLC_HCF
194 REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(OUT) :: PHLI_HRI
195 REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(OUT) :: PHLI_HCF
196 !
197 !
198 !* 0.2 Declarations of local variables :
199 !
200 !
201 REAL :: ZW1,ZW2 ! intermediate fields
202 REAL, DIMENSION(D%NIJT,D%NKT) &
203  :: ZT, & ! adjusted temperature
204  ZRV, ZRC, ZRI, & ! adjusted state
205  ZCPH, & ! guess of the CPh for the mixing
206  ZLV, & ! guess of the Lv at t+1
207  ZLS ! guess of the Ls at t+1
208 REAL :: ZCRIAUT, & ! Autoconversion thresholds
209  ZHCF, ZHR
210 !
211 INTEGER :: JITER,ITERMAX ! iterative loop for first order adjustment
212 INTEGER :: JIJ, JK
213 INTEGER :: IKTB, IKTE, IIJB, IIJE
214 !
215 REAL, DIMENSION(D%NIJT,D%NKT) :: ZSIGS, ZSRCS
216 REAL, DIMENSION(D%NIJT) :: ZSIGQSAT
217 LOGICAL :: LLNONE, LLTRIANGLE, LLHLC_H, LLHLI_H
218 
219 REAL(KIND=jphook) :: zhook_handle
220 !
221 !-------------------------------------------------------------------------------
222 !
223 !* 1. PRELIMINARIES
224 ! -------------
225 !
226 IF (lhook) CALL dr_hook('ICE_ADJUST',0,zhook_handle)
227 !
228 iktb=d%NKTB
229 ikte=d%NKTE
230 iijb=d%NIJB
231 iije=d%NIJE
232 !
233 itermax=1
234 !
235 IF(buconf%LBUDGET_TH) CALL budget_store_init_phy(d, tbudgets(nbudget_th), trim(hbuname), pths(:, :)*prhodj(:, :))
236 IF(buconf%LBUDGET_RV) CALL budget_store_init_phy(d, tbudgets(nbudget_rv), trim(hbuname), prvs(:, :)*prhodj(:, :))
237 IF(buconf%LBUDGET_RC) CALL budget_store_init_phy(d, tbudgets(nbudget_rc), trim(hbuname), prcs(:, :)*prhodj(:, :))
238 IF(buconf%LBUDGET_RI) CALL budget_store_init_phy(d, tbudgets(nbudget_ri), trim(hbuname), pris(:, :)*prhodj(:, :))
239 !-------------------------------------------------------------------------------
240 !
241 !* 2. COMPUTE QUANTITIES WITH THE GUESS OF THE FUTURE INSTANT
242 ! -------------------------------------------------------
243 !
244 !
245 ! beginning of the iterative loop (to compute the adjusted state)
246 !
247 DO jiter =1,itermax
248  !
249  !* 2.3 compute the latent heat of vaporization Lv(T*) at t+1
250  ! and the latent heat of sublimation Ls(T*) at t+1
251  !
252  DO jk=iktb,ikte
253  DO jij=iijb,iije
254  IF (jiter==1) zt(jij,jk) = pth(jij,jk) * pexn(jij,jk)
255  zlv(jij,jk) = cst%XLVTT + ( cst%XCPV - cst%XCL ) * ( zt(jij,jk) -cst%XTT )
256  zls(jij,jk) = cst%XLSTT + ( cst%XCPV - cst%XCI ) * ( zt(jij,jk) -cst%XTT )
257  ENDDO
258  ENDDO
259  !
260  !* 2.4 Iterate
261  !
262  IF (jiter==1) THEN
263  ! compute with input values
264  CALL iteration(prv,prc,pri,zrv,zrc,zri)
265  ELSE
266  ! compute with updated values
267  CALL iteration(zrv,zrc,zri,zrv,zrc,zri)
268  ENDIF
269 ENDDO ! end of the iterative loop
270 !
271 !* 5. COMPUTE THE SOURCES AND STORES THE CLOUD FRACTION
272 ! -------------------------------------------------
273 !
274 !
275 DO jk=iktb,ikte
276  DO jij=iijb,iije
277  !
278  !* 5.0 compute the variation of mixing ratio
279  !
280  ! Rc - Rc*
281  zw1 = (zrc(jij,jk) - prc(jij,jk)) / ptstep ! Pcon = ----------
282  ! 2 Delta t
283  zw2 = (zri(jij,jk) - pri(jij,jk)) / ptstep ! idem ZW1 but for Ri
284  !
285  !* 5.1 compute the sources
286  !
287  IF( zw1 < 0.0 ) THEN
288  zw1 = max( zw1, -prcs(jij,jk) )
289  ELSE
290  zw1 = min( zw1, prvs(jij,jk) )
291  ENDIF
292  prvs(jij,jk) = prvs(jij,jk) - zw1
293  prcs(jij,jk) = prcs(jij,jk) + zw1
294  pths(jij,jk) = pths(jij,jk) + &
295  zw1 * zlv(jij,jk) / (zcph(jij,jk) * pexnref(jij,jk))
296  !
297  IF( zw2 < 0.0 ) THEN
298  zw2 = max( zw2, -pris(jij,jk) )
299  ELSE
300  zw2 = min( zw2, prvs(jij,jk) )
301  ENDIF
302  prvs(jij,jk) = prvs(jij,jk) - zw2
303  pris(jij,jk) = pris(jij,jk) + zw2
304  pths(jij,jk) = pths(jij,jk) + &
305  zw2 * zls(jij,jk) / (zcph(jij,jk) * pexnref(jij,jk))
306  ENDDO
307  !
308  !* 5.2 compute the cloud fraction PCLDFR
309  !
310  IF ( .NOT. nebn%LSUBG_COND ) THEN
311  DO jij=iijb,iije
312  IF (prcs(jij,jk) + pris(jij,jk) > 1.e-12 / ptstep) THEN
313  pcldfr(jij,jk) = 1.
314  ELSE
315  pcldfr(jij,jk) = 0.
316  ENDIF
317  IF (ocompute_src) THEN
318  psrcs(jij,jk) = pcldfr(jij,jk)
319  END IF
320  ENDDO
321  ELSE !NEBN%LSUBG_COND case
322  ! Tests on characters strings can break the vectorization, or at least they would
323  ! slow down considerably the performance of a vector loop. One should use tests on
324  ! reals, integers or booleans only. REK.
325  llnone=parami%CSUBG_MF_PDF=='NONE'
326  lltriangle=parami%CSUBG_MF_PDF=='TRIANGLE'
327  llhlc_h=PRESENT(phlc_hrc).AND.PRESENT(phlc_hcf)
328  llhli_h=PRESENT(phli_hri).AND.PRESENT(phli_hcf)
329  DO jij=iijb,iije
330  !We limit PRC_MF+PRI_MF to PRVS*PTSTEP to avoid negative humidity
331  zw1=prc_mf(jij,jk)/ptstep
332  zw2=pri_mf(jij,jk)/ptstep
333  IF(zw1+zw2>prvs(jij,jk)) THEN
334  zw1=zw1*prvs(jij,jk)/(zw1+zw2)
335  zw2=prvs(jij,jk)-zw1
336  ENDIF
337  pcldfr(jij,jk)=min(1.,pcldfr(jij,jk)+pcf_mf(jij,jk))
338  prcs(jij,jk)=prcs(jij,jk)+zw1
339  pris(jij,jk)=pris(jij,jk)+zw2
340  prvs(jij,jk)=prvs(jij,jk)-(zw1+zw2)
341  pths(jij,jk) = pths(jij,jk) + &
342  (zw1 * zlv(jij,jk) + zw2 * zls(jij,jk)) / zcph(jij,jk) / pexnref(jij,jk)
343  !
344  IF(llhlc_h) THEN
345  zcriaut=icep%XCRIAUTC/prhodref(jij,jk)
346  IF(llnone)THEN
347  IF(zw1*ptstep>pcf_mf(jij,jk) * zcriaut) THEN
348  phlc_hrc(jij,jk)=phlc_hrc(jij,jk)+zw1*ptstep
349  phlc_hcf(jij,jk)=min(1.,phlc_hcf(jij,jk)+pcf_mf(jij,jk))
350  ENDIF
351  ELSEIF(lltriangle)THEN
352  !ZHCF is the precipitating part of the *cloud* and not of the grid cell
353  IF(zw1*ptstep>pcf_mf(jij,jk)*zcriaut) THEN
354  zhcf=1.-.5*(zcriaut*pcf_mf(jij,jk) / max(1.e-20, zw1*ptstep))**2
355  zhr=zw1*ptstep-(zcriaut*pcf_mf(jij,jk))**3 / &
356  &(3*max(1.e-20, zw1*ptstep)**2)
357  ELSEIF(2.*zw1*ptstep<=pcf_mf(jij,jk) * zcriaut) THEN
358  zhcf=0.
359  zhr=0.
360  ELSE
361  zhcf=(2.*zw1*ptstep-zcriaut*pcf_mf(jij,jk))**2 / &
362  &(2.*max(1.e-20, zw1*ptstep)**2)
363  zhr=(4.*(zw1*ptstep)**3-3.*zw1*ptstep*(zcriaut*pcf_mf(jij,jk))**2+&
364  (zcriaut*pcf_mf(jij,jk))**3) / &
365  &(3*max(1.e-20, zw1*ptstep)**2)
366  ENDIF
367  zhcf=zhcf*pcf_mf(jij,jk) !to retrieve the part of the grid cell
368  phlc_hcf(jij,jk)=min(1.,phlc_hcf(jij,jk)+zhcf) !total part of the grid cell that is precipitating
369  phlc_hrc(jij,jk)=phlc_hrc(jij,jk)+zhr
370  ENDIF
371  ENDIF
372  IF(llhli_h) THEN
373  zcriaut=min(icep%XCRIAUTI,10**(icep%XACRIAUTI*(zt(jij,jk)-cst%XTT)+icep%XBCRIAUTI))
374  IF(llnone)THEN
375  IF(zw2*ptstep>pcf_mf(jij,jk) * zcriaut) THEN
376  phli_hri(jij,jk)=phli_hri(jij,jk)+zw2*ptstep
377  phli_hcf(jij,jk)=min(1.,phli_hcf(jij,jk)+pcf_mf(jij,jk))
378  ENDIF
379  ELSEIF(lltriangle)THEN
380  !ZHCF is the precipitating part of the *cloud* and not of the grid cell
381  IF(zw2*ptstep>pcf_mf(jij,jk)*zcriaut) THEN
382  zhcf=1.-.5*(zcriaut*pcf_mf(jij,jk) / (zw2*ptstep))**2
383  zhr=zw2*ptstep-(zcriaut*pcf_mf(jij,jk))**3/(3*(zw2*ptstep)**2)
384  ELSEIF(2.*zw2*ptstep<=pcf_mf(jij,jk) * zcriaut) THEN
385  zhcf=0.
386  zhr=0.
387  ELSE
388  zhcf=(2.*zw2*ptstep-zcriaut*pcf_mf(jij,jk))**2 / (2.*(zw2*ptstep)**2)
389  zhr=(4.*(zw2*ptstep)**3-3.*zw2*ptstep*(zcriaut*pcf_mf(jij,jk))**2+&
390  (zcriaut*pcf_mf(jij,jk))**3)/(3*(zw2*ptstep)**2)
391  ENDIF
392  zhcf=zhcf*pcf_mf(jij,jk) !to retrieve the part of the grid cell
393  phli_hcf(jij,jk)=min(1.,phli_hcf(jij,jk)+zhcf) !total part of the grid cell that is precipitating
394  phli_hri(jij,jk)=phli_hri(jij,jk)+zhr
395  ENDIF
396  ENDIF
397  ENDDO
398  !
399  IF(PRESENT(pout_rv) .OR. PRESENT(pout_rc) .OR. &
400  &PRESENT(pout_ri) .OR. PRESENT(pout_th)) THEN
401  DO jij=iijb,iije
402  zw1=prc_mf(jij,jk)
403  zw2=pri_mf(jij,jk)
404  IF(zw1+zw2>zrv(jij,jk)) THEN
405  zw1=zw1*zrv(jij,jk)/(zw1+zw2)
406  zw2=zrv(jij,jk)-zw1
407  ENDIF
408  zrc(jij,jk)=zrc(jij,jk)+zw1
409  zri(jij,jk)=zri(jij,jk)+zw2
410  zrv(jij,jk)=zrv(jij,jk)-(zw1+zw2)
411  zt(jij,jk) = zt(jij,jk) + &
412  (zw1 * zlv(jij,jk) + zw2 * zls(jij,jk)) / zcph(jij,jk)
413  ENDDO
414  ENDIF
415  ENDIF !NEBN%LSUBG_COND
416 ENDDO
417 !
418 IF(PRESENT(pout_rv)) pout_rv=zrv
419 IF(PRESENT(pout_rc)) pout_rc=zrc
420 IF(PRESENT(pout_ri)) pout_ri=zri
421 IF(PRESENT(pout_th)) pout_th=zt / pexn(:,:)
422 !
423 !
424 !* 6. STORE THE BUDGET TERMS
