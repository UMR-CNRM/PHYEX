37 !!**** *TURB* - computes the turbulent source terms for the prognostic
38 !! variables.
39 !!
40 !! PURPOSE
41 !! -------
42 !!**** The purpose of this routine is to compute the source terms in
43 !! the evolution equations due to the turbulent mixing.
44 !! The source term is computed as the divergence of the turbulent fluxes.
45 !! The cartesian fluxes are obtained by a one and a half order closure, based
46 !! on a prognostic equation for the Turbulence Kinetic Energy( TKE ). The
47 !! system is closed by prescribing a turbulent mixing length. Different
48 !! choices are available for this length.
49 !
50 !!** METHOD
51 !! ------
52 !!
53 !! The dimensionality of the turbulence parameterization can be chosen by
54 !! means of the parameter TURBN%CTURBDIM:
55 !! * TURBN%CTURBDIM='1DIM' the parameterization is 1D but can be used in
56 !! 3D , 2D or 1D simulations. Only the sources associated to the vertical
57 !! turbulent fluxes are taken into account.
58 !! * TURBN%CTURBDIM='3DIM' the parameterization is fully 2D or 3D depending
59 !! on the model dimensionality. Of course, it does not make any sense to
60 !! activate this option with a 1D model.
61 !!
62 !! The following steps are made:
63 !! 1- Preliminary computations.
64 !! 2- The metric coefficients are recovered from the grid knowledge.
65 !! 3- The mixing length is computed according to its choice:
66 !! * TURBN%CTURBLEN='BL89' the Bougeault and Lacarrere algorithm is used.
67 !! The mixing length is given by the vertical displacement from its
68 !! original level of an air particule having an initial internal
69 !! energy equal to its TKE and stopped by the buoyancy forces.
70 !! The discrete formulation is second order accurate.
71 !! * TURBN%CTURBLEN='DELT' the mixing length is given by the mesh size
72 !! depending on the model dimensionality, this length is limited
73 !! with the ground distance.
74 !! * TURBN%CTURBLEN='DEAR' the mixing length is given by the mesh size
75 !! depending on the model dimensionality, this length is limited
76 !! with the ground distance and also by the Deardorff mixing length
77 !! pertinent in the stable cases.
78 !! * TURBN%CTURBLEN='KEPS' the mixing length is deduced from the TKE
79 !! dissipation, which becomes a prognostic variable of the model (
80 !! Duynkerke formulation).
81 !! 3'- The cloud mixing length is computed according to HTURBLEN_CLOUD
82 !! and emphasized following the CEI index
83 !! 4- The conservative variables are computed along with Lv/Cp.
84 !! 5- The turbulent Prandtl numbers are computed from the resolved fields
85 !! and TKE
86 !! 6- The sources associated to the vertical turbulent fluxes are computed
87 !! with a temporal scheme allowing a degree of implicitness given by
88 !! TURBN%XIMPL, varying from TURBN%XIMPL=0. ( purely explicit scheme) to TURBN%XIMPL=1.
89 !! ( purely implicit scheme)
90 !! The sources associated to the horizontal fluxes are computed with a
91 !! purely explicit temporal scheme. These sources are only computed when
92 !! the turbulence parameterization is 2D or 3D( TURBN%CTURBDIM='3DIM' ).
93 !! 7- The sources for TKE are computed, along with the dissipation of TKE
94 !! if TURBN%CTURBLEN='KEPS'.
95 !! 8- Some turbulence-related quantities are stored in the synchronous
96 !! FM-file.
97 !! 9- The non-conservative variables are retrieved.
98 !!
99 !!
100 !! The saving of the fields in the synchronous FM-file is controlled by:
101 !! * TURBN%LTURB_FLX => saves all the turbulent fluxes and correlations
102 !! * TURBN%LTURB_DIAG=> saves the turbulent Prandtl and Schmidt numbers, the
103 !! source terms of TKE and dissipation of TKE
104 !!
105 !! EXTERNAL
106 !! --------
107 !! SUBROUTINE PRANDTL : computes the turbulent Prandtl number
108 !! SUBROUTINE TURB_VER : computes the sources from the vertical fluxes
109 !! SUBROUTINE TURB_HOR : computes the sources from the horizontal fluxes
110 !! SUBROUTINE TKE_EPS_SOURCES : computes the sources for TKE and its
111 !! dissipation
112 !! SUBROUTINE BUDGET : computes and stores the budgets
113 !!

115 !! ------------------

1464 ! ########################################################################
1465 !!
1466 !!**** *COMPUTE_FUNCTION_THERMO* routine to compute several thermo functions
1467 !
1468 !! AUTHOR
1469 !! ------
1470 !!
1471 !! JP Pinty *LA*
1472 !!
1473 !! MODIFICATIONS
1474 !! -------------
1475 !! Original 24/02/03
1476 !!
1477 !-------------------------------------------------------------------------------
1478 !
1479 !* 0. DECLARATIONS
1480 ! ------------
1481 !
1482 IMPLICIT NONE
1483 !
1484 !* 0.1 Declarations of dummy arguments
1485 !
1486 REAL, INTENT(IN) :: PALP,PBETA,PGAM,PLTT,PC
1487 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PT,PEXN,PCP
1488 !
1489 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PLOCPEXN
1490 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PAMOIST,PATHETA
1491 !
1492 !-------------------------------------------------------------------------------
1493 !
1494  IF (lhook) CALL dr_hook('TURB:COMPUTE_FUNCTION_THERMO',0,zhook_handle2)
1495  zeps = cst%XMV / cst%XMD
1496  !
1497  !* 1.1 Lv/Cph at t
1498  !
1499  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
1500  plocpexn(iijb:iije,1:ikt) = ( pltt + (cst%XCPV-pc) * (pt(iijb:iije,1:ikt)-cst%XTT) ) &
1501  / pcp(iijb:iije,1:ikt)
1502  !
1503  !* 1.2 Saturation vapor pressure at t
1504  !
1505  zrvsat(iijb:iije,1:ikt) = exp( palp - pbeta/pt(iijb:iije,1:ikt) - pgam*alog( pt(iijb:iije,1:ikt) ) )
1506  !
1507  !* 1.3 saturation mixing ratio at t
1508  !
1509  zrvsat(iijb:iije,1:ikt) = zrvsat(iijb:iije,1:ikt) &
1510  * zeps / ( ppabst(iijb:iije,1:ikt) - zrvsat(iijb:iije,1:ikt) )
1511  !
1512  !* 1.4 compute the saturation mixing ratio derivative (rvs')
1513  !
1514  zdrvsatdt(iijb:iije,1:ikt) = ( pbeta / pt(iijb:iije,1:ikt) - pgam ) / pt(iijb:iije,1:ikt) &
1515  * zrvsat(iijb:iije,1:ikt) * ( 1. + zrvsat(iijb:iije,1:ikt) / zeps )
1516  !
1517  !* 1.5 compute Amoist
1518  !
1519  pamoist(iijb:iije,1:ikt)= 0.5 / ( 1.0 + zdrvsatdt(iijb:iije,1:ikt) * plocpexn(iijb:iije,1:ikt) )
1520  !
1521  !* 1.6 compute Atheta
1522  !
1523  patheta(iijb:iije,1:ikt)= pamoist(iijb:iije,1:ikt) * pexn(iijb:iije,1:ikt) * &
1524  ( ( zrvsat(iijb:iije,1:ikt) - prt(iijb:iije,1:ikt,1) ) * plocpexn(iijb:iije,1:ikt) / &
1525  ( 1. + zdrvsatdt(iijb:iije,1:ikt) * plocpexn(iijb:iije,1:ikt) ) * &
1526  ( &
1527  zrvsat(iijb:iije,1:ikt) * (1. + zrvsat(iijb:iije,1:ikt)/zeps) &
1528  * ( -2.*pbeta/pt(iijb:iije,1:ikt) + pgam ) / pt(iijb:iije,1:ikt)**2 &
1529  +zdrvsatdt(iijb:iije,1:ikt) * (1. + 2. * zrvsat(iijb:iije,1:ikt)/zeps) &
1530  * ( pbeta/pt(iijb:iije,1:ikt) - pgam ) / pt(iijb:iije,1:ikt) &
1531  ) &
1532  - zdrvsatdt(iijb:iije,1:ikt) &
1533  )
1534  !
1535  !* 1.7 Lv/Cph/Exner at t-1
1536  !
1537  plocpexn(iijb:iije,1:ikt) = plocpexn(iijb:iije,1:ikt) / pexn(iijb:iije,1:ikt)
1538  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
1539 !
1540 IF (lhook) CALL dr_hook('TURB:COMPUTE_FUNCTION_THERMO',1,zhook_handle2)

1699  ELSE
1700  EXIT
1701  ENDIF

1726  SUBROUTINE dear(PLM)
1727 ! ####################
1728 !!
1729 !!**** *DEAR* routine to compute mixing length for DEARdorff case
1730 !
1731 !! AUTHOR
1732 !! ------
1733 !!
1734 !! M Tomasini *Meteo-France
1735 !!
1736 !! MODIFICATIONS
1737 !! -------------
1738 !! Original 01/05
1739 !! I.Sandu (Sept.2006) : Modification of the stability criterion
1740 !! (theta_v -> theta_l)
1741 !!
1742 !-------------------------------------------------------------------------------
1743 !
1744 !* 0. DECLARATIONS
1745 ! ------------
1746 !
1747 !* 0.1 Declarations of dummy arguments
1748 !
1749 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PLM
1750 !
1751 !-------------------------------------------------------------------------------
1752 !
1753 ! initialize the mixing length with the mesh grid
1754 IF (lhook) CALL dr_hook('TURB:DEAR',0,zhook_handle2)
1755 IF ( turbn%CTURBDIM /= '1DIM' ) THEN
1756  CALL mxf_phy(d,pdxx,zwork1)
1757  IF (.NOT. o2d) THEN
1758  CALL myf_phy(d,pdyy,zwork2)
1759  END IF
1760 END IF
1761 ! 1D turbulence scheme
1762 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
1763 plm(iijb:iije,iktb:ikte) = pzz(iijb:iije,iktb+ikl:ikte+ikl) - pzz(iijb:iije,iktb:ikte)
1764 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
1765 !$mnh_expand_array(JIJ=IIJB:IIJE)
1766 plm(iijb:iije,iku) = plm(iijb:iije,ike)
1767 plm(iijb:iije,ika) = pzz(iijb:iije,ikb) - pzz(iijb:iije,ika)
1768 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
1769 !
1770 IF ( turbn%CTURBDIM /= '1DIM' ) THEN ! 3D turbulence scheme
1771  IF ( o2d) THEN
1772  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
1773  plm(iijb:iije,1:ikt) = sqrt( plm(iijb:iije,1:ikt)*zwork1(iijb:iije,1:ikt) )
1774  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
1775  ELSE
1776  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
1777  plm(iijb:iije,1:ikt) = (plm(iijb:iije,1:ikt)*zwork1(iijb:iije,1:ikt) &
1778  * zwork2(iijb:iije,1:ikt) ) ** (1./3.)
1779  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
1780  END IF
1781 END IF

1783 !
1784 CALL etheta(d,cst,krr,krri,pthlt,prt,zlocpexnm,zatheta,psrct,oocean,ocompute_src,zetheta)
1785 CALL emoist(d,cst,krr,krri,pthlt,prt,zlocpexnm,zamoist,psrct,oocean,zemoist)
1786 !
1787 IF (krr>0) THEN
1788  DO jk = iktb+1,ikte-1
1789  DO jij=iijb,iije
1790  zdthldz(jij,jk)= 0.5*((pthlt(jij,jk+ikl)-pthlt(jij,jk ))/pdzz(jij,jk+ikl)+ &
1791  (pthlt(jij,jk )-pthlt(jij,jk-ikl))/pdzz(jij,jk ))
1792  zdrtdz(jij,jk) = 0.5*((prt(jij,jk+ikl,1)-prt(jij,jk ,1))/pdzz(jij,jk+ikl)+ &
1793  (prt(jij,jk ,1)-prt(jij,jk-ikl,1))/pdzz(jij,jk ))

1795  zvar=cst%XG*(cst%XALPHAOC*zdthldz(jij,jk)-cst%XBETAOC*zdrtdz(jij,jk))
1796  ELSE
1797  zvar=cst%XG/pthvref(jij,jk)* &
1798  (zetheta(jij,jk)*zdthldz(jij,jk)+zemoist(jij,jk)*zdrtdz(jij,jk))
1799  END IF
1800  !
1801  IF (zvar>0.) THEN
1802  plm(jij,jk)=max(cst%XMNH_EPSILON,min(plm(jij,jk), &
1803  0.76* sqrt(ptket(jij,jk)/zvar)))
1804  END IF
1805  END DO
1806  END DO
1807 else! For dry atmos or unsalted ocean runs
1808  DO jk = iktb+1,ikte-1
1809  DO jij=iijb,iije
1810  zdthldz(jij,jk)= 0.5*((pthlt(jij,jk+ikl)-pthlt(jij,jk ))/pdzz(jij,jk+ikl)+ &
1811  (pthlt(jij,jk )-pthlt(jij,jk-ikl))/pdzz(jij,jk ))
1812  IF (oocean) THEN
1813  zvar= cst%XG*cst%XALPHAOC*zdthldz(jij,jk)
1814  ELSE
1815  zvar= cst%XG/pthvref(jij,jk)*zetheta(jij,jk)*zdthldz(jij,jk)
1816  END IF
1817 !
1818  IF (zvar>0.) THEN
1819  plm(jij,jk)=max(cst%XMNH_EPSILON,min(plm(jij,jk), &
1820  0.76* sqrt(ptket(jij,jk)/zvar)))
1821  END IF
1822  END DO
1823  END DO
1824 END IF
1825 ! special case near the surface
1826 !$mnh_expand_array(JIJ=IIJB:IIJE)
1827 zdthldz(iijb:iije,ikb)=(pthlt(iijb:iije,ikb+ikl)-pthlt(iijb:iije,ikb))/pdzz(iijb:iije,ikb+ikl)
1828 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
1829 ! For dry simulations
1830 IF (krr>0) THEN
1831  !$mnh_expand_array(JIJ=IIJB:IIJE)
1832  zdrtdz(iijb:iije,ikb)=(prt(iijb:iije,ikb+ikl,1)-prt(iijb:iije,ikb,1))/pdzz(iijb:iije,ikb+ikl)
1833  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
1834 ELSE
1835  zdrtdz(:,ikb)=0
1836 ENDIF
1837 !
1838 IF (oocean) THEN
1839  !$mnh_expand_array(JIJ=IIJB:IIJE)
1840  zwork2d(iijb:iije)=cst%XG*(cst%XALPHAOC*zdthldz(iijb:iije,ikb)-cst%XBETAOC*zdrtdz(iijb:iije,ikb))
1841  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
1842 ELSE
1843  !$mnh_expand_array(JIJ=IIJB:IIJE)
1844  zwork2d(iijb:iije)=cst%XG/pthvref(iijb:iije,ikb)* &
1845  (zetheta(iijb:iije,ikb)*zdthldz(iijb:iije,ikb)+zemoist(iijb:iije,ikb)*zdrtdz(iijb:iije,ikb))
1846  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
1847 END IF
1848 !$mnh_expand_where(JIJ=IIJB:IIJE)
1849 WHERE(zwork2d(iijb:iije)>0.)
1850  plm(iijb:iije,ikb)=max(cst%XMNH_EPSILON,min( plm(iijb:iije,ikb), &

1852 END WHERE
1853 !$mnh_end_expand_where(JIJ=IIJB:IIJE)
1854 !
1855 ! mixing length limited by the distance normal to the surface (with the same factor as for BL89)
1856 !
1857 IF (.NOT. turbn%LRMC01) THEN
1858  zalpha=0.5**(-1.5)
1859  !
1860  DO jij=iijb,iije
1861  IF (oocean) THEN
1862  DO jk=ikte,iktb,-1
1863  zd=zalpha*(pzz(jij,ikte+1)-pzz(jij,jk))
1864  IF ( plm(jij,jk)>zd) THEN
1865  plm(jij,jk)=zd
1866  ELSE
1867  EXIT
1868  ENDIF
1869  END DO
1870  ELSE
1871  DO jk=iktb,ikte
1872  zd=zalpha*(0.5*(pzz(jij,jk)+pzz(jij,jk+ikl))-pzz(jij,ikb)) &
1873  *pdircoszw(jij)
1874  IF ( plm(jij,jk)>zd) THEN
1875  plm(jij,jk)=zd
1876  ELSE
1877  EXIT
1878  ENDIF
1879  END DO
1880  ENDIF
1881  END DO
1882 END IF
1883 !
1884 !$mnh_expand_array(JIJ=IIJB:IIJE)
1885 plm(iijb:iije,ika) = plm(iijb:iije,ikb)

1887 plm(iijb:iije,iku) = plm(iijb:iije,iku-ikl)

1899 !! 2/ emphasize the mixing length in the cloud
1900 !! by the coefficient ZCOEF_AMPL calculated here
1901 !! when the CEI index is above ZCEI_MIN.
1902 !!
1903 !!
1904 !! ZCOEF_AMPL ^
1905 !! |
1906 !! |
1907 !! ZCOEF_AMPL_SAT - ---------- Saturation
1908 !! (XDUMMY1) | -
1909 !! | -
1910 !! | -
1911 !! | -
1912 !! | - Amplification
1913 !! | - straight
1914 !! | - line
1915 !! | -
1916 !! | -
1917 !! | -
1918 !! | -
1919 !! | -
1920 !! 1 ------------
1921 !! |
1922 !! |
1923 !! 0 -----------|------------|----------> PCEI
1924 !! 0 ZCEI_MIN ZCEI_MAX
1925 !! (XDUMMY2) (XDUMMY3)
1926 !!
1927 !!
1928 !!
1929 !! AUTHOR
1930 !! ------
1931 !! M. Tomasini *CNRM METEO-FRANCE
1932 !!
1933 !! MODIFICATIONS
1934 !! -------------
1935 !! Original 09/07/04
1936 !!
1937 !-------------------------------------------------------------------------------
1938 !
1939 !* 0. DECLARATIONS
1940 ! ------------
1941 !

1943 !
1944 !-------------------------------------------------------------------------------
1945 !

1947 ! --------------
1948 !

1951 zcoef_ampl_cei_nul = 1. - zpente * pcei_min
1952 !
1953 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)

1956 !

1958 ! --------------------------------------------

1961 !

1965 END WHERE
1966 !$mnh_end_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)

1969 ! amplification coefficient ZCOEF_AMPL as a function of CEI
1970 !
1971 !$mnh_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
1972 WHERE ( pcei(iijb:iije,1:ikt) < pcei_max .AND. pcei(iijb:iije,1:ikt) > pcei_min)
1973  zcoef_ampl(iijb:iije,1:ikt) = zpente * pcei(iijb:iije,1:ikt) + zcoef_ampl_cei_nul
1974 END WHERE

1991  !
1992  !* 3.2 Delta mixing length
1993  ! -------------------
1994  CASE ('DELT')

1996  !
1997  !* 3.3 Deardorff mixing length
1998  ! -----------------------
1999  CASE ('DEAR')
2000  CALL dear(zlm_cloud)
2001  !
2002  END SELECT
2003 ENDIF
2004 !
2005 !* 4. MODIFICATION OF THE MIXING LENGTH IN THE CLOUDS
2006 ! -----------------------------------------------
2007 !
2008 ! Impression before modification of the mixing length
2009 IF ( turbn%LTURB_DIAG .AND. tpfile%LOPENED ) THEN

2011  cmnhname = 'LM_CLEAR_SKY', &
2012  cstdname = '', &

2014  cunits = 'm', &
2015  cdir = 'XY', &
2016  ccomment = 'X_Y_Z_LM CLEAR SKY', &
2017  ngrid = 1, &
2018  ntype = typereal, &
2019  ndims = 3, &
2020  ltimedep = .true. )
2021  CALL io_field_write_phy(d,tpfile,tzfield,zlm)
2022 ENDIF

2024 ! Amplification of the mixing length when the criteria are verified
2025 !

2027 WHERE (zcoef_ampl(iijb:iije,1:ikt) /= 1.)
2028  zlm(iijb:iije,1:ikt) = zcoef_ampl(iijb:iije,1:ikt)*zlm_cloud(iijb:iije,1:ikt)
2029 END WHERE
