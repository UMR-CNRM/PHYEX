90 zlsfact(:) = (xlstt+(xcpv-xci)*ztcelsius(:))/zw(:) ! L_s/(Pi_ref*C_ph)
91 zlvfact(:) = (xlvtt+(xcpv-xcl)*ztcelsius(:))/zw(:) ! L_v/(Pi_ref*C_ph)

93 zw(:) = 0.0

95 WHERE( (zt(:)<xtt-35.0) .AND. (prrt(:)>xrtmin(3)) .AND. ldcompute(:) )
96  p_th_honr(:) = prrt(:)*(zlsfact(:)-zlvfact(:))
97  p_rr_honr(:) = - prrt(:)
98  p_cr_honr(:) = - pcrt(:)

100  pb_rr(:) = pb_rr(:) - prrt(:)

102  pb_rg(:) = pb_rg(:) + prrt(:)
