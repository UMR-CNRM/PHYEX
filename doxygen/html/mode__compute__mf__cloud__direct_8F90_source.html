15 !!
16 !!**** *COMPUTE_MF_CLOUD_DIRECT* -
17 !! compute diagnostic subgrid cumulus cloud caracteristics with a direct scheme
18 !!
19 !! PURPOSE
20 !! -------
21 !!**** The purpose of this routine is to compute the cloud fraction and
22 !! the mean cloud content associated with clouds described by the
23 !! mass flux scheme
24 !!
25 !
26 !!** METHOD
27 !! ------
28 !! Updraft variables are used directly to diagnose subgrid clouds
29 !! This scheme may be activated only if the selected updraft model
30 !! gives the updraft fraction as an output
31 !!
32 !! EXTERNAL
33 !! --------
34 !!
35 !! IMPLICIT ARGUMENTS
36 !! ------------------
37 !!
38 !! REFERENCE
39 !! ---------
40 !!
41 !!
42 !! AUTHOR
43 !! ------

45 !!
46 !! MODIFICATIONS
47 !! -------------
48 !! Original 25 Aug 2011
49 !! S. Riette Jan 2012: support for both order of vertical levels
50 !! S. Riette Apr 2013: computation begins one level lower (to be able to have a cloud
51 !! on mass level just below the first saturated flux level)
52 !! --------------------------------------------------------------------------
53 !
54 !* 0. DECLARATIONS
55 ! ------------
56 USE modd_dimphyex, ONLY: dimphyex_t
57 USE modd_param_mfshall_n, ONLY : param_mfshall_t
58 USE yomhook , ONLY : lhook, dr_hook, jphook
59 !

61 !
62 !* 0.1 Declaration of Arguments
63 !
64 TYPE(dimphyex_t), INTENT(IN) :: D
65 TYPE(param_mfshall_t), INTENT(IN) :: PARAMMF
66 INTEGER, DIMENSION(D%NIJT), INTENT(IN) :: KKLCL ! index of updraft condensation level
67 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PFRAC_UP ! Updraft Fraction
68 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRC_UP,PRI_UP ! updraft characteritics
69 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PRC_MF, PRI_MF ! cloud content
70 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PCF_MF ! and cloud fraction for MF scheme
71 !
72 !* 0.1 Declaration of local variables
73 !
74 REAL(KIND=jphook) :: zhook_handle
75 INTEGER :: JI,JK, JK0, IKB,IKE,IKL,IIJB,IIJE
76 !
77 !* 0.2 Initialisation
78 !
79 IF (lhook) CALL dr_hook('COMPUTE_MF_CLOUD_DIRECT',0,zhook_handle)
80 !
81 ikb=d%NKB
82 ike=d%NKE
83 ikl=d%NKL
84 iijb=d%NIJB
85 iije=d%NIJE
86 !* 1. COMPUTATION OF SUBGRID CLOUD
87 ! ----------------------------
88 
89 !
90 ! Warning: updraft variables are on flux levels
91 ! and PRC_MF, PRI_MF and PCF_MF are on mass levels
92 prc_mf(:,:)=0.
93 pri_mf(:,:)=0.
94 pcf_mf(:,:)=0.
95 
96 DO ji=iijb,iije
97  jk0=kklcl(ji)-ikl ! first mass level with cloud
98  jk0=max(jk0, min(ikb,ike)) !protection if KKL=1
99  jk0=min(jk0, max(ikb,ike)) !protection if KKL=-1
100  DO jk=jk0,ike-ikl,ikl
101  pcf_mf(ji,jk ) = max( 0., min(1.,parammf%XKCF_MF *0.5* ( &
102  & pfrac_up(ji,jk) + pfrac_up(ji,jk+ikl) ) ))
103  prc_mf(ji,jk) = 0.5* parammf%XKCF_MF * ( pfrac_up(ji,jk)*prc_up(ji,jk) &
104  + pfrac_up(ji,jk+ikl)*prc_up(ji,jk+ikl) )
105  pri_mf(ji,jk) = 0.5* parammf%XKCF_MF * ( pfrac_up(ji,jk)*pri_up(ji,jk) &
106  + pfrac_up(ji,jk+ikl)*pri_up(ji,jk+ikl) )
107  END DO
108 END DO
109 
110 IF (lhook) CALL dr_hook('COMPUTE_MF_CLOUD_DIRECT',1,zhook_handle)
111 
112 END SUBROUTINE compute_mf_cloud_direct
