108!* 0.2 declaration of local variables
109!
110REAL, DIMENSION(KPROMA,0:KRR) :: ZVART
111REAL, DIMENSION(KPROMA) :: ZT, &
112 & ZKA, ZDV, ZAI, ZCJ, &
113 & ZLBDAR, ZLBDAS, ZLBDAG, ZLBDAH, ZLBDAR_RF, &
114 & ZRGSI, ZRGSI_MR, ZRAINFR

123IF (lhook) CALL dr_hook('ICE4_TENDENCIES', 0, zhook_handle)

125
126!
127!$acc kernels
128zt(:)=pt(:)
129DO jv=0,krr
130 zvart(:,jv)=pvart(:,jv)
131 pa(:,jv)=0.
132 pb(:,jv)=0.
133ENDDO
134!$acc end kernels
135!
136IF(odsoft) THEN
137!$acc kernels
138 pbu_inst(:, irvheni_mr)=0.
139 pbu_inst(:, irrhong_mr)=0.
140 pbu_inst(:, irimltc_mr)=0.
141 pbu_inst(:, irsrimcg_mr)=0.
142!$acc end kernels
143ELSE
144 !
145 !* 2. COMPUTES THE SLOW COLD PROCESS SOURCES
146 ! --------------------------------------
147 DO jl=1, ksize
148 CALL ice4_nucleation(cst, parami, icep, iced, ldcompute(jl), &
149 zvart(jl,ith), ppres(jl), prhodref(jl), pexn(jl), plsfact(jl), zt(jl), &
150 zvart(jl,irv), &
151 pcit(jl), pbu_inst(jl, irvheni_mr))
152 ENDDO
153 DO jl=1, ksize
154 zvart(jl,ith)=zvart(jl,ith) + pbu_inst(jl, irvheni_mr)*plsfact(jl)
155 zt(jl) = zvart(jl,ith) * pexn(jl)
156 zvart(jl,irv)=zvart(jl,irv) - pbu_inst(jl, irvheni_mr)
157 zvart(jl,iri)=zvart(jl,iri) + pbu_inst(jl, irvheni_mr)
158 ENDDO
159!$acc end kernels
160 !
161 !* 3.3 compute the spontaneous freezing source: RRHONG
162 !
163 CALL ice4_rrhong(cst, parami, iced, kproma, ksize, ldcompute, &
164 &pexn, plvfact, plsfact, &
165 &zt, zvart(:,irr), &
166 &zvart(:,ith), &
167 &pbu_inst(:, irrhong_mr))
168!$acc kernels
169!$acc loop independent
170 DO jl=1, ksize
171 zvart(jl,ith) = zvart(jl,ith) + pbu_inst(jl, irrhong_mr)*(plsfact(jl)-plvfact(jl)) ! f(L_f*(RRHONG))
172 zt(jl) = zvart(jl,ith) * pexn(jl)
173 zvart(jl,irr) = zvart(jl,irr) - pbu_inst(jl, irrhong_mr)
174 zvart(jl,irg) = zvart(jl,irg) + pbu_inst(jl, irrhong_mr)
175 ENDDO
176!$acc end kernels
177 !
178 !* 7.1 cloud ice melting
179 !
180 CALL ice4_rimltc(cst, parami, kproma, ksize, ldcompute, &
181 &pexn, plvfact, plsfact, &
182 &zt, &
183 &zvart(:,ith), zvart(:,iri), &
184 &pbu_inst(:, irimltc_mr))
185!$acc kernels
186!$acc loop independent
187 DO jl=1, ksize
188 zvart(jl,ith) = zvart(jl,ith) - pbu_inst(jl, irimltc_mr)*(plsfact(jl)-plvfact(jl)) ! f(L_f*(-RIMLTC))
189 zt(jl) = zvart(jl,ith) * pexn(jl)
190 zvart(jl,irc) = zvart(jl,irc) + pbu_inst(jl, irimltc_mr)
191 zvart(jl,iri) = zvart(jl,iri) - pbu_inst(jl, irimltc_mr)
192 ENDDO
193!$acc end kernels
194 !
195 ! 5.1.6 riming-conversion of the large sized aggregates into graupel (old parametrisation)
196 !
197 IF(parami%CSNOWRIMING=='OLD ') THEN
198!$acc kernels
199 !$mnh_expand_where(JL=1:KSIZE)
200 IF (parami%LSNOW_T) THEN
201 zlbdas(1:ksize)=0.
202 WHERE (zvart(1:ksize,irs)>0. .AND. zt(1:ksize)>263.15)
203 zlbdas(1:ksize) = max(min(iced%XLBDAS_MAX, 10**(14.554-0.0423*zt(1:ksize))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
204 END WHERE
205 WHERE (zvart(1:ksize,irs)>0. .AND. zt(1:ksize)<=263.15)
206 zlbdas(1:ksize) = max(min(iced%XLBDAS_MAX, 10**(6.226-0.0106*zt(1:ksize))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
207 END WHERE
208 ELSE
209 WHERE(zvart(1:ksize,irs)>0.)
210 zlbdas(1:ksize) = min(iced%XLBDAS_MAX, iced%XLBS*(prhodref(1:ksize)*max(zvart(1:ksize,irs), iced%XRTMIN(5)))**iced%XLBEXS)
211 ELSEWHERE
212 zlbdas(1:ksize)=0.
213 END WHERE
214 END IF
215 !$mnh_end_expand_where(JL=1:KSIZE)
216!$acc end kernels
217 CALL ice4_rsrimcg_old(cst, parami, icep, iced, kproma, ksize, odsoft, ldcompute, &
218 &prhodref, &
219 &zlbdas, &
220 &zt, zvart(:,irc), zvart(:,irs), &
221 &pbu_inst(:, irsrimcg_mr))
222!$acc kernels
223!$acc loop independent
224 DO jl=1, ksize
225 zvart(jl,irs) = zvart(jl,irs) - pbu_inst(jl, irsrimcg_mr)
226 zvart(jl,irg) = zvart(jl,irg) + pbu_inst(jl, irsrimcg_mr)
227 ENDDO
228!$acc end kernels
229 ELSE
230!$acc kernels
231 pbu_inst(:, irsrimcg_mr) = 0.
232!$acc end kernels
233 ENDIF
234!
235!$acc kernels
236!$acc loop independent
237 DO jl=1, ksize
238 pb(jl, ith)=pb(jl, ith) + pbu_inst(jl, irvheni_mr)*plsfact(jl)
239 pb(jl, ith)=pb(jl, ith) + pbu_inst(jl, irrhong_mr)*(plsfact(jl)-plvfact(jl))
240 pb(jl, ith)=pb(jl, ith) - pbu_inst(jl, irimltc_mr)*(plsfact(jl)-plvfact(jl))
241
242 pb(jl, irv)=pb(jl, irv) - pbu_inst(jl, irvheni_mr)
243
244 pb(jl, irc)=pb(jl, irc) + pbu_inst(jl, irimltc_mr)
245
246 pb(jl, irr)=pb(jl, irr) - pbu_inst(jl, irrhong_mr)
247
248 pb(jl, iri)=pb(jl, iri) + pbu_inst(jl, irvheni_mr)
249 pb(jl, iri)=pb(jl, iri) - pbu_inst(jl, irimltc_mr)
250
251 pb(jl, irs)=pb(jl, irs) - pbu_inst(jl, irsrimcg_mr)
252
253 pb(jl, irg)=pb(jl, irg) + pbu_inst(jl, irrhong_mr)
254 pb(jl, irg)=pb(jl, irg) + pbu_inst(jl, irsrimcg_mr)
255 ENDDO
256!$acc end kernels
257 !
258 !* Derived fields
259 !
260!$acc kernels
261!$acc loop independent
262 DO jl=1, ksize
263 zzw = exp(cst%XALPI-cst%XBETAI/zt(jl)-cst%XGAMI*log(zt(jl)))
264 pssi(jl) = zvart(jl,irv)*( ppres(jl)-zzw ) / ( cst%XEPSILO * zzw ) - 1.0
265 ! Supersaturation over ice
266 zka(jl) = 2.38e-2 + 0.0071e-2*(zt(jl)-cst%XTT) ! k_a
267 zdv(jl) = 0.211e-4*(zt(jl)/cst%XTT)**1.94 * (cst%XP00/ppres(jl)) ! D_v
268 zai(jl) = (cst%XLSTT+(cst%XCPV-cst%XCI)*(zt(jl)-cst%XTT))**2 / (zka(jl)*cst%XRV*zt(jl)**2) &
269 + ( cst%XRV*zt(jl) ) / (zdv(jl)*zzw)
270 zcj(jl) = icep%XSCFAC*prhodref(jl)**0.3 / sqrt(1.718e-5+0.0049e-5*(zt(jl)-cst%XTT))
271 ENDDO
272!$acc end kernels
273ENDIF ! ODSOFT
274!
275!Cloud water split between high and low content part is done here
276CALL ice4_compute_pdf(cst, icep, iced, ksize, parami%CSUBG_AUCV_RC, parami%CSUBG_AUCV_RI, parami%CSUBG_PR_PDF,&
277 ldcompute, prhodref, zvart(:,irc), zvart(:,iri), pcf, zt, psigma_rc, &
278 phlc_hcf, phlc_lcf, phlc_hrc, phlc_lrc, &
279 phli_hcf, phli_lcf, phli_hri, phli_lri, zrainfr)
280llrfr=parami%CSUBG_RC_RR_ACCR=='PRFR' .OR. parami%CSUBG_RR_EVAP=='PRFR'
281IF (llrfr) THEN
282 !To be exact, ICE4_RAINFR_VERT should be called here with the updated PRAINFR
283 !But this call would require the full 3D arrays for rain, snow and graupel which
284 !are not available here (due to separation between 1D and 3D computation for GPU).
285 !
286 !We replace the full computation by a small update to ensure consistency
287!$acc kernels
288!$acc loop independent
289 DO jl=1, ksize
290 prainfr(jl)=max(prainfr(jl), zrainfr(jl))
291 IF(krr==7) THEN
292 llmask=zvart(jl,irr) .GT. iced%XRTMIN(3) .OR. zvart(jl,irs) .GT. iced%XRTMIN(5) .OR. &
293 &zvart(jl,irg) .GT. iced%XRTMIN(6) .OR. zvart(jl,irh) .GT. iced%XRTMIN(7)
294 ELSE
295 llmask=zvart(jl,irr) .GT. iced%XRTMIN(3) .OR. zvart(jl,irs) .GT. iced%XRTMIN(5) .OR. &
296 &zvart(jl,irg) .GT. iced%XRTMIN(6)
297 ENDIF
298 IF(llmask .AND. prainfr(jl)==0.) THEN
299 prainfr(jl)=1.
300 ENDIF
301 ENDDO
302!$acc end kernels
303ELSE
304!$acc kernels
305 prainfr(:)=1.
306!$acc end kernels
307ENDIF
308!
309!* compute the slope parameters
310!
311!$acc kernels
312!$acc loop independent
313DO jl=1, ksize
314 !ZLBDAR will be used when we consider rain diluted over the grid box
315 IF(zvart(jl,irr)>0.) THEN
316 zlbdar(jl)=iced%XLBR*(prhodref(jl)*max(zvart(jl,irr), iced%XRTMIN(3)))**iced%XLBEXR
317 ELSE
318 zlbdar(jl)=0.
319 ENDIF
320 !ZLBDAR_RF is used when we consider rain concentrated in its fraction
321 IF(llrfr) THEN
322 IF(zvart(jl,irr)>0. .AND. prainfr(jl)>0.) THEN
323 zlbdar_rf(jl)=iced%XLBR*(prhodref(jl)*max(zvart(jl,irr)/prainfr(jl), iced%XRTMIN(3)))**iced%XLBEXR
324 ELSE
325 zlbdar_rf(jl)=0.
326 ENDIF
327 ELSE
328 zlbdar_rf(jl)=zlbdar(jl)
329 ENDIF
330 IF (parami%LSNOW_T) THEN
331 IF (zvart(jl,irs)>0. .AND. zt(jl)>263.15) THEN
332 zlbdas(jl) = max(min(iced%XLBDAS_MAX, 10**(14.554-0.0423*zt(jl))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
333 ELSE IF (zvart(jl,irs)>0. .AND. zt(jl)<=263.15) THEN
334 zlbdas(jl) = max(min(iced%XLBDAS_MAX, 10**(6.226-0.0106*zt(jl))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
335 ELSE
336 zlbdas(jl)=0.
337 END IF
338 ELSE
339 IF(zvart(jl,irs)>0.) THEN
340 zlbdas(jl)=min(iced%XLBDAS_MAX, iced%XLBS*(prhodref(jl)*max(zvart(jl,irs), iced%XRTMIN(5)))**iced%XLBEXS)
341 ELSE
342 zlbdas(jl)=0.
343 ENDIF
344 END IF
345 IF(zvart(jl,irg)>0.) THEN
346 zlbdag(jl)=iced%XLBG*(prhodref(jl)*max(zvart(jl,irg), iced%XRTMIN(6)))**iced%XLBEXG
347 ELSE
348 zlbdag(jl)=0.
349 ENDIF
350 IF(krr==7) THEN
351 IF(zvart(jl,irh)>0.) THEN
352 zlbdah(jl)=iced%XLBH*(prhodref(jl)*max(zvart(jl,irh), iced%XRTMIN(7)))**iced%XLBEXH
353 ELSE
354 zlbdah(jl)=0.
355 ENDIF
356 ENDIF
357ENDDO
358!$acc end kernels
359!
360!
361CALL ice4_slow(cst, icep, iced, kproma, ksize, odsoft, oelec, ldcompute, prhodref, zt, &
362 &pssi, plvfact, plsfact, &
363 &zvart(:,irv), zvart(:,irc), zvart(:,iri), zvart(:,irs), zvart(:,irg), &
364 &zlbdas, zlbdag, &
365 &zai, zcj, phli_hcf, phli_hri, &
366 &platham_iaggs, &
367 &pbu_inst(:, irchoni), pbu_inst(:, irvdeps), pbu_inst(:, iriaggs), pbu_inst(:, iriauts), pbu_inst(:, irvdepg))
368!
369!-------------------------------------------------------------------------------
370!
371!
372!* 3. COMPUTES THE SLOW WARM PROCESS SOURCES
373! --------------------------------------
374!
375!
376IF(parami%LWARM) THEN ! Check if the formation of the raindrops by the slow
377 ! warm processes is allowed
378 CALL ice4_warm(cst, icep, iced, kproma, ksize, odsoft,ldcompute, &
379 &parami%CSUBG_RC_RR_ACCR, parami%CSUBG_RR_EVAP, &
380 &prhodref, plvfact, zt, ppres, zvart(:,ith),&
381 &zlbdar, zlbdar_rf, zka, zdv, zcj, &
382 &phlc_lcf, phlc_hcf, phlc_lrc, phlc_hrc, &
383 &pcf, prainfr, &
384 &zvart(:,irv), zvart(:,irc), zvart(:,irr), &
385 &pbu_inst(:, ircautr), pbu_inst(:, ircaccr), pbu_inst(:, irrevav))
386ELSE
387!$acc kernels
388 pbu_inst(:, ircautr)=0.
389 pbu_inst(:, ircaccr)=0.
390 pbu_inst(:, irrevav)=0.
391!$acc end kernels
392END IF
393!
394!-------------------------------------------------------------------------------
395!
396!
397!* 4. COMPUTES THE FAST COLD PROCESS SOURCES FOR r_s
398! ----------------------------------------------
399!
400CALL ice4_fast_rs(cst, parami, icep, iced, kproma, ksize, odsoft, ldcompute, &
401 &prhodref, plvfact, plsfact, ppres, &
402 &zdv, zka, zcj, &
403 &zlbdar, zlbdas, &
404 &zt, zvart(:,irv), zvart(:,irc), zvart(:,irr), zvart(:,irs), &
405 &pbu_inst(:, iriaggs), &
406 &pbu_inst(:, ircrimss), pbu_inst(:, ircrimsg), pbu_inst(:, irsrimcg), &
407 &pbu_inst(:, irraccss), pbu_inst(:, irraccsg), pbu_inst(:, irsaccrg), pbu_inst(:, irsmltg), &
408 &pbu_inst(:, ircmltsr), &
409 &prs_tend)
410!
411!-------------------------------------------------------------------------------
412!
413!
414!* 5. COMPUTES THE FAST COLD PROCESS SOURCES FOR r_g
415! ------------------------------------------------------
416!
417!$acc kernels
418!$acc loop independent
419DO jl=1, ksize
420 zrgsi(jl) = pbu_inst(jl, irvdepg) + pbu_inst(jl, irsmltg) + pbu_inst(jl, irraccsg) + &
421 & pbu_inst(jl, irsaccrg) + pbu_inst(jl, ircrimsg) + pbu_inst(jl, irsrimcg)
422 zrgsi_mr(jl) = pbu_inst(jl, irrhong_mr) + pbu_inst(jl, irsrimcg_mr)
423ENDDO
424!$acc end kernels
425CALL ice4_fast_rg(cst, parami, icep, iced, kproma, ksize, odsoft, ldcompute, krr, &
426 &prhodref, plvfact, plsfact, ppres, &
427 &zdv, zka, zcj, pcit, &
428 &zlbdar, zlbdas, zlbdag, &
429 &zt, zvart(:,irv), zvart(:,irc), zvart(:,irr), zvart(:,iri), zvart(:,irs), zvart(:,irg), &
430 &zrgsi, zrgsi_mr(:), &
431 &llwetg, &
432 &pbu_inst(:, iricfrrg), pbu_inst(:, irrcfrig), pbu_inst(:, iricfrr), pbu_inst(:, ircwetg), &
433 &pbu_inst(:, iriwetg), pbu_inst(:, irrwetg), pbu_inst(:, irswetg), &
434 &pbu_inst(:, ircdryg), pbu_inst(:, iridryg), pbu_inst(:, irrdryg), pbu_inst(:, irsdryg), &
435 &pbu_inst(:, irwetgh), pbu_inst(:, irwetgh_mr), pbu_inst(:, irgmltr), &
436 &prg_tend)
437!
438!-------------------------------------------------------------------------------
439!
440!
441!* 6. COMPUTES THE FAST COLD PROCESS SOURCES FOR r_h
442! ----------------------------------------------
443!
444IF (krr==7) THEN
445 CALL ice4_fast_rh(cst, parami, icep, iced, kproma, ksize, odsoft, ldcompute, llwetg, &
446 &prhodref, plvfact, plsfact, ppres, &
447 &zdv, zka, zcj, &
448 &zlbdas, zlbdag, zlbdar, zlbdah, &
449 &zt, zvart(:,irv), zvart(:,irc), zvart(:,irr), zvart(:,iri), zvart(:,irs), zvart(:,irg), zvart(:,irh), &
450 &pbu_inst(:, ircweth), pbu_inst(:, iriweth), pbu_inst(:, irsweth), pbu_inst(:, irgweth), pbu_inst(:, irrweth), &
