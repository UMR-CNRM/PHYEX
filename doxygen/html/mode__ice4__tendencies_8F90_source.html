142  !
143  !* 2. COMPUTES THE SLOW COLD PROCESS SOURCES
144  ! --------------------------------------
145  !ZBUF is sized in (NPROMA,4) when it should be in (KSIZE,4), but this is of no consequence
146  !since the values are not used.
147  CALL ice4_nucleation(cst, parami, icep, iced, ksize, ldcompute(1:ksize), &
148  zvart(1:ksize,ith), ppres(1:ksize), prhodref(1:ksize), pexn(1:ksize), plsfact(1:ksize), zt(1:ksize), &
149  zvart(1:ksize,irv), &
150  pcit(1:ksize), pbu_inst(1:ksize, irvheni_mr), zbuf(:, :), llbuf(1:ksize))
151 
152  DO jl=1, ksize
153  zvart(jl,ith)=zvart(jl,ith) + pbu_inst(jl, irvheni_mr)*plsfact(jl)
154  zt(jl) = zvart(jl,ith) * pexn(jl)
155  zvart(jl,irv)=zvart(jl,irv) - pbu_inst(jl, irvheni_mr)
156  zvart(jl,iri)=zvart(jl,iri) + pbu_inst(jl, irvheni_mr)
157  ENDDO
158  !
159  !* 3.3 compute the spontaneous freezing source: RRHONG
160  !
161  CALL ice4_rrhong(cst, parami, iced, kproma, ksize, ldcompute, &
162  &pexn, plvfact, plsfact, &
163  &zt, zvart(:,irr), &
164  &zvart(:,ith), &
165  &pbu_inst(:, irrhong_mr))
166  DO jl=1, ksize
167  zvart(jl,ith) = zvart(jl,ith) + pbu_inst(jl, irrhong_mr)*(plsfact(jl)-plvfact(jl)) ! f(L_f*(RRHONG))
168  zt(jl) = zvart(jl,ith) * pexn(jl)
169  zvart(jl,irr) = zvart(jl,irr) - pbu_inst(jl, irrhong_mr)
170  zvart(jl,irg) = zvart(jl,irg) + pbu_inst(jl, irrhong_mr)
171  ENDDO
172  !
173  !* 7.1 cloud ice melting
174  !
175  CALL ice4_rimltc(cst, parami, kproma, ksize, ldcompute, &
176  &pexn, plvfact, plsfact, &
177  &zt, &
178  &zvart(:,ith), zvart(:,iri), &
179  &pbu_inst(:, irimltc_mr))
180  DO jl=1, ksize
181  zvart(jl,ith) = zvart(jl,ith) - pbu_inst(jl, irimltc_mr)*(plsfact(jl)-plvfact(jl)) ! f(L_f*(-RIMLTC))
182  zt(jl) = zvart(jl,ith) * pexn(jl)
183  zvart(jl,irc) = zvart(jl,irc) + pbu_inst(jl, irimltc_mr)
184  zvart(jl,iri) = zvart(jl,iri) - pbu_inst(jl, irimltc_mr)
185  ENDDO
186  !
187  ! 5.1.6 riming-conversion of the large sized aggregates into graupel (old parametrisation)
188  !
189  IF(parami%CSNOWRIMING=='OLD ') THEN
190  !$mnh_expand_where(JL=1:KSIZE)
191  IF (parami%LSNOW_T) THEN
192  zlbdas(1:ksize)=0.
193  WHERE (zvart(1:ksize,irs)>0. .AND. zt(1:ksize)>263.15)
194  zlbdas(1:ksize) = max(min(iced%XLBDAS_MAX, 10**(14.554-0.0423*zt(1:ksize))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
195  END WHERE
196  WHERE (zvart(1:ksize,irs)>0. .AND. zt(1:ksize)<=263.15)
197  zlbdas(1:ksize) = max(min(iced%XLBDAS_MAX, 10**(6.226-0.0106*zt(1:ksize))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
198  END WHERE
199  ELSE
200  WHERE(zvart(1:ksize,irs)>0.)
201  zlbdas(1:ksize) = min(iced%XLBDAS_MAX, iced%XLBS*(prhodref(1:ksize)*max(zvart(1:ksize,irs), iced%XRTMIN(5)))**iced%XLBEXS)
202  ELSEWHERE
203  zlbdas(1:ksize)=0.
204  END WHERE
205  END IF
206  !$mnh_end_expand_where(JL=1:KSIZE)
207  CALL ice4_rsrimcg_old(cst, parami, icep, iced, kproma, ksize, odsoft, ldcompute, &
208  &prhodref, &
209  &zlbdas, &
210  &zt, zvart(:,irc), zvart(:,irs), &
211  &pbu_inst(:, irsrimcg_mr))
212  DO jl=1, ksize
213  zvart(jl,irs) = zvart(jl,irs) - pbu_inst(jl, irsrimcg_mr)
214  zvart(jl,irg) = zvart(jl,irg) + pbu_inst(jl, irsrimcg_mr)
215  ENDDO
216  ELSE
217  pbu_inst(:, irsrimcg_mr) = 0.
218  ENDIF
219 
220  DO jl=1, ksize
221  pb(jl, ith)=pb(jl, ith) + pbu_inst(jl, irvheni_mr)*plsfact(jl)
222  pb(jl, ith)=pb(jl, ith) + pbu_inst(jl, irrhong_mr)*(plsfact(jl)-plvfact(jl))
223  pb(jl, ith)=pb(jl, ith) - pbu_inst(jl, irimltc_mr)*(plsfact(jl)-plvfact(jl))
224 
225  pb(jl, irv)=pb(jl, irv) - pbu_inst(jl, irvheni_mr)
226 
227  pb(jl, irc)=pb(jl, irc) + pbu_inst(jl, irimltc_mr)
228 
229  pb(jl, irr)=pb(jl, irr) - pbu_inst(jl, irrhong_mr)
230 
231  pb(jl, iri)=pb(jl, iri) + pbu_inst(jl, irvheni_mr)
232  pb(jl, iri)=pb(jl, iri) - pbu_inst(jl, irimltc_mr)
233 
234  pb(jl, irs)=pb(jl, irs) - pbu_inst(jl, irsrimcg_mr)
235 
236  pb(jl, irg)=pb(jl, irg) + pbu_inst(jl, irrhong_mr)
237  pb(jl, irg)=pb(jl, irg) + pbu_inst(jl, irsrimcg_mr)
238  ENDDO
239  !
240  !* Derived fields
241  !
242  DO jl=1, ksize
243  zzw = exp(cst%XALPI-cst%XBETAI/zt(jl)-cst%XGAMI*alog(zt(jl)))
244  pssi(jl) = zvart(jl,irv)*( ppres(jl)-zzw ) / ( cst%XEPSILO * zzw ) - 1.0
245  ! Supersaturation over ice
246  zka(jl) = 2.38e-2 + 0.0071e-2*(zt(jl)-cst%XTT) ! k_a
247  zdv(jl) = 0.211e-4*(zt(jl)/cst%XTT)**1.94 * (cst%XP00/ppres(jl)) ! D_v
248  zai(jl) = (cst%XLSTT+(cst%XCPV-cst%XCI)*(zt(jl)-cst%XTT))**2 / (zka(jl)*cst%XRV*zt(jl)**2) &
249  + ( cst%XRV*zt(jl) ) / (zdv(jl)*zzw)
250  zcj(jl) = icep%XSCFAC*prhodref(jl)**0.3 / sqrt(1.718e-5+0.0049e-5*(zt(jl)-cst%XTT))
251  ENDDO
252 ENDIF ! ODSOFT
253 !
254 !Cloud water split between high and low content part is done here
255 CALL ice4_compute_pdf(cst, icep, iced, ksize, parami%CSUBG_AUCV_RC, parami%CSUBG_AUCV_RI, parami%CSUBG_PR_PDF,&
256  ldcompute, prhodref, zvart(:,irc), zvart(:,iri), pcf, zt, psigma_rc, &
257  phlc_hcf, phlc_lcf, phlc_hrc, phlc_lrc, &
258  phli_hcf, phli_lcf, phli_hri, phli_lri, zrainfr)
259 llrfr=parami%CSUBG_RC_RR_ACCR=='PRFR' .OR. parami%CSUBG_RR_EVAP=='PRFR'
260 IF (llrfr) THEN
261  !To be exact, ICE4_RAINFR_VERT should be called here with the updated PRAINFR
262  !But this call would require the full 3D arrays for rain, snow and graupel which
263  !are not available here (due to separation between 1D and 3D computation for GPU).
264  !
265  !We replace the full computation by a small update to ensure consistency
266  DO jl=1, ksize
267  prainfr(jl)=max(prainfr(jl), zrainfr(jl))
268  IF(krr==7) THEN
269  llmask=zvart(jl,irr) .GT. iced%XRTMIN(3) .OR. zvart(jl,irs) .GT. iced%XRTMIN(5) .OR. &
270  &zvart(jl,irg) .GT. iced%XRTMIN(6) .OR. zvart(jl,irh) .GT. iced%XRTMIN(7)
271  ELSE
272  llmask=zvart(jl,irr) .GT. iced%XRTMIN(3) .OR. zvart(jl,irs) .GT. iced%XRTMIN(5) .OR. &
273  &zvart(jl,irg) .GT. iced%XRTMIN(6)
274  ENDIF
275  IF(llmask .AND. prainfr(jl)==0.) THEN
276  prainfr(jl)=1.
277  ENDIF
278  ENDDO
279 ELSE
280  prainfr(:)=1.
281 ENDIF
282 !
283 !* compute the slope parameters
284 !
285 DO jl=1, ksize
286  !ZLBDAR will be used when we consider rain diluted over the grid box
287  IF(zvart(jl,irr)>0.) THEN
288  zlbdar(jl)=iced%XLBR*(prhodref(jl)*max(zvart(jl,irr), iced%XRTMIN(3)))**iced%XLBEXR
289  ELSE
290  zlbdar(jl)=0.
291  ENDIF
292  !ZLBDAR_RF is used when we consider rain concentrated in its fraction
293  IF(llrfr) THEN
294  IF(zvart(jl,irr)>0. .AND. prainfr(jl)>0.) THEN
295  zlbdar_rf(jl)=iced%XLBR*(prhodref(jl)*max(zvart(jl,irr)/prainfr(jl), iced%XRTMIN(3)))**iced%XLBEXR
296  ELSE
297  zlbdar_rf(jl)=0.
298  ENDIF
299  ELSE
300  zlbdar_rf(jl)=zlbdar(jl)
301  ENDIF
302  IF (parami%LSNOW_T) THEN
303  IF (zvart(jl,irs)>0. .AND. zt(jl)>263.15) THEN
304  zlbdas(jl) = max(min(iced%XLBDAS_MAX, 10**(14.554-0.0423*zt(jl))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
305  ELSE IF (zvart(jl,irs)>0. .AND. zt(jl)<=263.15) THEN
306  zlbdas(jl) = max(min(iced%XLBDAS_MAX, 10**(6.226-0.0106*zt(jl))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
307  ELSE
308  zlbdas(jl)=0.
309  END IF
310  ELSE
311  IF(zvart(jl,irs)>0.) THEN
312  zlbdas(jl)=min(iced%XLBDAS_MAX, iced%XLBS*(prhodref(jl)*max(zvart(jl,irs), iced%XRTMIN(5)))**iced%XLBEXS)
313  ELSE
314  zlbdas(jl)=0.
315  ENDIF
316  END IF
317  IF(zvart(jl,irg)>0.) THEN
318  zlbdag(jl)=iced%XLBG*(prhodref(jl)*max(zvart(jl,irg), iced%XRTMIN(6)))**iced%XLBEXG
319  ELSE
320  zlbdag(jl)=0.
321  ENDIF
322  IF(krr==7) THEN
323  IF(zvart(jl,irh)>0.) THEN
324  zlbdah(jl)=iced%XLBH*(prhodref(jl)*max(zvart(jl,irh), iced%XRTMIN(7)))**iced%XLBEXH
325  ELSE
326  zlbdah(jl)=0.
327  ENDIF
328  ENDIF
329 ENDDO

333  &pssi, plvfact, plsfact, &

335  &zlbdas, zlbdag, &
336  &zai, zcj, phli_hcf, phli_hri, &
337  &platham_iaggs, &
338  &pbu_inst(:, irchoni), pbu_inst(:, irvdeps), pbu_inst(:, iriaggs), pbu_inst(:, iriauts), pbu_inst(:, irvdepg))

353  &phlc_lcf, phlc_hcf, phlc_lrc, phlc_hrc, &

368 !

371  &zdv, zka, zcj, &
372  &zlbdar, zlbdas, &

392  &prhodref, plvfact, plsfact, ppres, &
393  &zdv, zka, zcj, pcit, &
394  &zlbdar, zlbdas, zlbdag, &
395  &zt, zvart(:,irv), zvart(:,irc), zvart(:,irr), zvart(:,iri), zvart(:,irs), zvart(:,irg), &
396  &zrgsi, zrgsi_mr(:), &
397  &llwetg, &

401  &pbu_inst(:, irwetgh), pbu_inst(:, irwetgh_mr), pbu_inst(:, irgmltr), &
402  &prg_tend)
