420  zw_up2(iijb:iije,ikb)=(pemf(iijb:iije,ikb)/(pfrac_up(iijb:iije,ikb)*zrho_f(iijb:iije,ikb)))**2
421  gtest(iijb:iije)=.true.
422  ELSEWHERE
423  pemf(iijb:iije,ikb) =0.
424  gtest(iijb:iije)=.false.
425  ENDWHERE
426  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
427 ELSE
428  !$mnh_expand_array(JIJ=IIJB:IIJE)
429  gtest(iijb:iije)=pemf(iijb:iije,ikb+ikl)>0.
430  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
431 END IF
432 
433 !--------------------------------------------------------------------------
434 
435 ! 3. Vertical ascending loop
436 ! -----------------------
437 !
438 ! If GTEST = T the updraft starts from the KKB level and stops when GTEST becomes F
439 !
440 !
441 gtestlcl(:)=.false.
442 gtestetl(:)=.false.
443 
444 ! Loop on vertical level
445 
446 DO jk=ikb,ike-ikl,ikl
447 
448  ! IF the updraft top is reached for all column, stop the loop on levels
449  itest=0
450  DO jij=iijb,iije
451  IF(gtest(jij)) itest = itest + 1
452  END DO
453  IF (itest==0) cycle
454 
455  ! Computation of entrainment and detrainment with KF90
456  ! parameterization in clouds and LR01 in subcloud layer
457 
458 
459  ! to find the LCL (check if JK is LCL or not)
460  !$mnh_expand_where(JIJ=IIJB:IIJE)
461  WHERE ((prc_up(iijb:iije,jk)+pri_up(iijb:iije,jk)>0.).AND.(.NOT.(gtestlcl(iijb:iije))))
462  kklcl(iijb:iije) = jk
463  gtestlcl(iijb:iije)=.true.
464  ENDWHERE
465  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
466 
467  ! COMPUTE PENTR and PDETR at mass level JK
468  IF (oentr_detr) THEN
469  IF(jk/=ikb) THEN
470  !$mnh_expand_array(JIJ=IIJB:IIJE)
471  zrc_mix(iijb:iije,jk) = zrc_mix(iijb:iije,jk-ikl) ! guess of Rc of mixture
472  zri_mix(iijb:iije,jk) = zri_mix(iijb:iije,jk-ikl) ! guess of Ri of mixture
473  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
474  ENDIF
475  CALL compute_entr_detr(d, cst, nebn, parammf, jk,ikb,ike,ikl,gtest,gtestlcl,pfrac_ice_up(:,jk),&
476  prhodref(:,jk),zpres_f(:,jk),zpres_f(:,jk+ikl),&
477  pzz(:,:),pdzz(:,:),zthvm(:,:), &
478  pthlm(:,:),prtm(:,:),zw_up2(:,:),zth_up(:,jk), &
479  pthl_up(:,jk),prt_up(:,jk),zlup(:), &
480  prc_up(:,jk),pri_up(:,jk),pthv_up(:,jk),&
481  prsat_up(:,jk),zrc_mix(:,jk),zri_mix(:,jk), &
482  pentr(:,jk),pdetr(:,jk),zentr_cld(:,jk),zdetr_cld(:,jk),&
483  zbuo_integ_dry(:,jk), zbuo_integ_cld(:,jk), &
484  zpart_dry(:) )
485  !$mnh_expand_where(JIJ=IIJB:IIJE)
486  pbuo_integ(iijb:iije,jk)=zbuo_integ_dry(iijb:iije,jk)+zbuo_integ_cld(iijb:iije,jk)
487 
488  IF (jk==ikb) THEN
489  pdetr(iijb:iije,jk)=0.
490  zdetr_cld(iijb:iije,jk)=0.
491  ENDIF
492 
493  ! Computation of updraft characteristics at level JK+KKL
494  WHERE(gtest(iijb:iije))
495  zmix1(iijb:iije)=0.5*(pzz(iijb:iije,jk+ikl)-pzz(iijb:iije,jk))*&
496  &(pentr(iijb:iije,jk)-pdetr(iijb:iije,jk))
497  pemf(iijb:iije,jk+ikl)=pemf(iijb:iije,jk)*exp(2*zmix1(iijb:iije))
498  ENDWHERE
499  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
500  ELSE !OENTR_DETR
501  !$mnh_expand_array(JIJ=IIJB:IIJE)
502  gtest(iijb:iije) = (pemf(iijb:iije,jk+ikl)>0.)
503  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
504  END IF !OENTR_DETR
505 
506  ! stop the updraft if MF becomes negative
507  !$mnh_expand_where(JIJ=IIJB:IIJE)
508  WHERE (gtest(iijb:iije).AND.(pemf(iijb:iije,jk+ikl)<=0.))
509  pemf(iijb:iije,jk+ikl)=0.
510  kkctl(iijb:iije) = jk+ikl
511  gtest(iijb:iije)=.false.
512  pfrac_ice_up(iijb:iije,jk+ikl)=pfrac_ice_up(iijb:iije,jk)
513  prsat_up(iijb:iije,jk+ikl)=prsat_up(iijb:iije,jk)
514  ENDWHERE
515  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
516 
517  ! If the updraft did not stop, compute cons updraft characteritics at jk+KKL
518  DO jij=iijb,iije
519  IF(gtest(jij)) THEN
520  zmix2(jij) = (pzz(jij,jk+ikl)-pzz(jij,jk))*pentr(jij,jk) !&
521  zmix3_cld(jij) = (pzz(jij,jk+ikl)-pzz(jij,jk))*(1.-zpart_dry(jij))*zdetr_cld(jij,jk) !&
522  zmix2_cld(jij) = (pzz(jij,jk+ikl)-pzz(jij,jk))*(1.-zpart_dry(jij))*zentr_cld(jij,jk)
523  pthl_up(jij,jk+ikl)=(pthl_up(jij,jk)*(1.-0.5*zmix2(jij)) + pthlm(jij,jk)*zmix2(jij)) &
524  /(1.+0.5*zmix2(jij))
525  prt_up(jij,jk+ikl) =(prt_up(jij,jk)*(1.-0.5*zmix2(jij)) + prtm(jij,jk)*zmix2(jij)) &
526  /(1.+0.5*zmix2(jij))
527  ENDIF
528  ENDDO
529 
530  IF(parammf%LMIXUV) THEN
531  IF(jk/=ikb) THEN
532  !$mnh_expand_where(JIJ=IIJB:IIJE)
533  WHERE(gtest(iijb:iije))
534  pu_up(iijb:iije,jk+ikl) = (pu_up(iijb:iije,jk)*(1-0.5*zmix2(iijb:iije)) + &
535  &pum(iijb:iije,jk)*zmix2(iijb:iije)+ &
536  0.5*parammf%XPRES_UV*(pzz(iijb:iije,jk+ikl)-pzz(iijb:iije,jk))*&
537  ((pum(iijb:iije,jk+ikl)-pum(iijb:iije,jk))/pdzz(iijb:iije,jk+ikl)+&
538  (pum(iijb:iije,jk)-pum(iijb:iije,jk-ikl))/pdzz(iijb:iije,jk)) ) &
539  /(1+0.5*zmix2(iijb:iije))
540  pv_up(iijb:iije,jk+ikl) = (pv_up(iijb:iije,jk)*(1-0.5*zmix2(iijb:iije)) + &
541  &pvm(iijb:iije,jk)*zmix2(iijb:iije)+ &
542  0.5*parammf%XPRES_UV*(pzz(iijb:iije,jk+ikl)-pzz(iijb:iije,jk))*&
543  ((pvm(iijb:iije,jk+ikl)-pvm(iijb:iije,jk))/pdzz(iijb:iije,jk+ikl)+&
544  (pvm(iijb:iije,jk)-pvm(iijb:iije,jk-ikl))/pdzz(iijb:iije,jk)) ) &
545  /(1+0.5*zmix2(iijb:iije))
546  ENDWHERE
547  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
548  ELSE
549  !$mnh_expand_where(JIJ=IIJB:IIJE)
550  WHERE(gtest(iijb:iije))
551  pu_up(iijb:iije,jk+ikl) = (pu_up(iijb:iije,jk)*(1-0.5*zmix2(iijb:iije)) + &
552  &pum(iijb:iije,jk)*zmix2(iijb:iije)+ &
553  0.5*parammf%XPRES_UV*(pzz(iijb:iije,jk+ikl)-pzz(iijb:iije,jk))*&
554  ((pum(iijb:iije,jk+ikl)-pum(iijb:iije,jk))/pdzz(iijb:iije,jk+ikl)) ) &
555  /(1+0.5*zmix2(iijb:iije))
556  pv_up(iijb:iije,jk+ikl) = (pv_up(iijb:iije,jk)*(1-0.5*zmix2(iijb:iije)) + &
557  &pvm(iijb:iije,jk)*zmix2(iijb:iije)+ &
558  0.5*parammf%XPRES_UV*(pzz(iijb:iije,jk+ikl)-pzz(iijb:iije,jk))*&
559  ((pvm(iijb:iije,jk+ikl)-pvm(iijb:iije,jk))/pdzz(iijb:iije,jk+ikl)) ) &
560  /(1+0.5*zmix2(iijb:iije))
561  ENDWHERE
562  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
563  ENDIF
564  ENDIF !PARAMMF%LMIXUV
565  DO jsv=1,ksv
566  IF (onomixlg .AND. jsv >= ksv_lgbeg .AND. jsv<= ksv_lgend) cycle
567  !$mnh_expand_where(JIJ=IIJB:IIJE)
568  WHERE(gtest(iijb:iije))
569  psv_up(iijb:iije,jk+ikl,jsv) = (psv_up(iijb:iije,jk,jsv)*(1-0.5*zmix2(iijb:iije)) + &
570  psvm(iijb:iije,jk,jsv)*zmix2(iijb:iije)) /(1+0.5*zmix2(iijb:iije))
571  ENDWHERE
572  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
573  END DO
574 
575  IF (oentr_detr) THEN
576 
577  ! Compute non cons. var. at level JK+KKL
578  !$mnh_expand_array(JIJ=IIJB:IIJE)
579  zrc_up(iijb:iije)=prc_up(iijb:iije,jk) ! guess = level just below
580  zri_up(iijb:iije)=pri_up(iijb:iije,jk) ! guess = level just below
581  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
582  CALL th_r_from_thl_rt(d, cst, nebn, nebn%CFRAC_ICE_SHALLOW_MF, pfrac_ice_up(:,jk+ikl), zpres_f(:,jk+ikl), &
583  pthl_up(:,jk+ikl),prt_up(:,jk+ikl),zth_up(:,jk+ikl), &
584  zrv_up(:),zrc_up(:),zri_up(:),zrsatw(:),zrsati(:), oocean=.false., &
585  pbuf=zbuf(:,:))
586  !$mnh_expand_where(JIJ=IIJB:IIJE)
587  WHERE(gtest(iijb:iije))
588  prc_up(iijb:iije,jk+ikl)=zrc_up(iijb:iije)
589  prv_up(iijb:iije,jk+ikl)=zrv_up(iijb:iije)
590  pri_up(iijb:iije,jk+ikl)=zri_up(iijb:iije)
591  prsat_up(iijb:iije,jk+ikl) = zrsatw(iijb:iije)*(1-pfrac_ice_up(iijb:iije,jk+ikl)) + &
592  & zrsati(iijb:iije)*pfrac_ice_up(iijb:iije,jk+ikl)
593  ENDWHERE
594  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
595  ! Compute the updraft theta_v, buoyancy and w**2 for level JK+KKL
596  DO jij=iijb,iije
597  IF(gtest(jij)) THEN
598  pthv_up(jij,jk+ikl) = zth_up(jij,jk+ikl)* &
599  & ((1+zrvord*prv_up(jij,jk+ikl))/(1+prt_up(jij,jk+ikl)))
600  IF (zbuo_integ_dry(jij,jk)>0.) THEN
601  zw_up2(jij,jk+ikl) = zw_up2(jij,jk) + 2.*(parammf%XABUO-parammf%XBENTR*parammf%XENTR_DRY)* &
602  &zbuo_integ_dry(jij,jk)
603  ELSE
604  zw_up2(jij,jk+ikl) = zw_up2(jij,jk) + 2.*parammf%XABUO* zbuo_integ_dry(jij,jk)
605  END IF
606  zw_up2(jij,jk+ikl) = zw_up2(jij,jk+ikl)*(1.-(parammf%XBDETR*zmix3_cld(jij)+ &
607  &parammf%XBENTR*zmix2_cld(jij)))&
608  /(1.+(parammf%XBDETR*zmix3_cld(jij)+parammf%XBENTR*zmix2_cld(jij))) &
609  +2.*(parammf%XABUO)*zbuo_integ_cld(jij,jk)/ &
610  &(1.+(parammf%XBDETR*zmix3_cld(jij)+parammf%XBENTR*zmix2_cld(jij)))
611  END IF
612  END DO
613 
614  ! Test if the updraft has reach the ETL
615  !$mnh_expand_where(JIJ=IIJB:IIJE)
616  WHERE (gtest(iijb:iije).AND.(pbuo_integ(iijb:iije,jk)<=0.))
617  kketl(iijb:iije) = jk+ikl
618  gtestetl(iijb:iije)=.true.
619  ELSEWHERE
620  gtestetl(iijb:iije)=.false.
621  ENDWHERE
622  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
623 
624  ! Test is we have reached the top of the updraft
625  !$mnh_expand_where(JIJ=IIJB:IIJE)
626  WHERE (gtest(iijb:iije).AND.((zw_up2(iijb:iije,jk+ikl)<=0.).OR.(pemf(iijb:iije,jk+ikl)<=0.)))
627  zw_up2(iijb:iije,jk+ikl)=0.
628  pemf(iijb:iije,jk+ikl)=0.
629  gtest(iijb:iije)=.false.
630  pthl_up(iijb:iije,jk+ikl)=zthlm_f(iijb:iije,jk+ikl)
631  prt_up(iijb:iije,jk+ikl)=zrtm_f(iijb:iije,jk+ikl)
632  prc_up(iijb:iije,jk+ikl)=0.
633  pri_up(iijb:iije,jk+ikl)=0.
634  prv_up(iijb:iije,jk+ikl)=0.
635  pthv_up(iijb:iije,jk+ikl)=zthvm_f(iijb:iije,jk+ikl)
636  pfrac_up(iijb:iije,jk+ikl)=0.
637  kkctl(iijb:iije)=jk+ikl
638  ENDWHERE
639  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
640 
641  ! compute frac_up at JK+KKL
642  !$mnh_expand_where(JIJ=IIJB:IIJE)
643  WHERE (gtest(iijb:iije))
644  pfrac_up(iijb:iije,jk+ikl)=pemf(iijb:iije,jk+ikl)/&
645  &(sqrt(zw_up2(iijb:iije,jk+ikl))*zrho_f(iijb:iije,jk+ikl))
646  ENDWHERE
647  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
648 
649  ! Updraft fraction must be smaller than XFRAC_UP_MAX
650  !$mnh_expand_where(JIJ=IIJB:IIJE)
651  WHERE (gtest(iijb:iije))
652  pfrac_up(iijb:iije,jk+ikl)=min(parammf%XFRAC_UP_MAX,pfrac_up(iijb:iije,jk+ikl))
653  ENDWHERE
654  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
655 
656  ! When cloudy and non-buoyant, updraft fraction must decrease
657  !$mnh_expand_where(JIJ=IIJB:IIJE)
658  WHERE ((gtest(iijb:iije).AND.gtestetl(iijb:iije)).AND.gtestlcl(iijb:iije))
659  pfrac_up(iijb:iije,jk+ikl)=min(pfrac_up(iijb:iije,jk+ikl),pfrac_up(iijb:iije,jk))
660  ENDWHERE
661  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
662 
663  ! Mass flux is updated with the new updraft fraction
664 
665  !$mnh_expand_array(JIJ=IIJB:IIJE)
666  IF (oentr_detr) pemf(iijb:iije,jk+ikl)=pfrac_up(iijb:iije,jk+ikl)*sqrt(zw_up2(iijb:iije,jk+ikl))* &
667  &zrho_f(iijb:iije,jk+ikl)
668  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
669  END IF !OENTR_DETR
670 ENDDO
671 
672 IF(oentr_detr) THEN
673 
674  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
675  pw_up(iijb:iije,1:ikt)=sqrt(zw_up2(iijb:iije,1:ikt))
676  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
677 
678  !$mnh_expand_array(JIJ=IIJB:IIJE)
679  pemf(iijb:iije,ikb) =0.
680  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
681 
682  ! Limits the shallow convection scheme when cloud heigth is higher than 3000m.
683  ! To do this, mass flux is multiplied by a coefficient decreasing linearly
684  ! from 1 (for clouds of ZDEPTH_MAX1 m of depth) to 0 (for clouds of ZDEPTH_MAX2 m of depth).
685  ! This way, all MF fluxes are diminished by this amount.
686  ! Diagnosed cloud fraction is also multiplied by the same coefficient.
687  !
688  DO jij=iijb,iije
689  pdepth(jij) = max(0., pzz(jij,kkctl(jij)) - pzz(jij,kklcl(jij)) )
690  END DO
691  IF(parammf%LVERLIMUP) THEN
692  DO jk=1,ikt
693  DO jij=iijb,iije
694  kdepth(jij,jk) = min(max(0., pzz(jij,jk) - pzz(jij,kklcl(jij)) ), pdepth(jij))
695  END DO
696  END DO
697  END IF
698  !$mnh_expand_array(JIJ=IIJB:IIJE)
699  gwork1(iijb:iije)= (gtestlcl(iijb:iije) .AND. (pdepth(iijb:iije) > zdepth_max1) )
700  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
701  DO jk=1,ikt
702  !$mnh_expand_array(JIJ=IIJB:IIJE)
703  gwork2(iijb:iije,jk) = gwork1(iijb:iije)
704  IF(parammf%LVERLIMUP) THEN
705  zcoef(iijb:iije,jk) = (1.-(kdepth(iijb:iije,jk)-zdepth_max1)/(zdepth_max2-zdepth_max1))
706  ELSE
707  zcoef(iijb:iije,jk) = (1.-(pdepth(iijb:iije)-zdepth_max1)/(zdepth_max2-zdepth_max1))
708  END IF
709  zcoef(iijb:iije,jk)=min(max(zcoef(iijb:iije,jk),0.),1.)
710  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
711  ENDDO
712  !$mnh_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
713  WHERE (gwork2(iijb:iije,1:ikt))
714  pemf(iijb:iije,1:ikt) = pemf(iijb:iije,1:ikt) * zcoef(iijb:iije,1:ikt)
715  pfrac_up(iijb:iije,1:ikt) = pfrac_up(iijb:iije,1:ikt) * zcoef(iijb:iije,1:ikt)
716  ENDWHERE
717  !$mnh_end_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
718 ENDIF
719 
720 IF (lhook) CALL dr_hook('COMPUTE_UPDRAFT',1,zhook_handle)
