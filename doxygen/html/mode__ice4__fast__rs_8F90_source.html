9SUBROUTINE ice4_fast_rs(CST, PARAMI, ICEP, ICED, KPROMA, KSIZE, LDSOFT, LDCOMPUTE, &
10 &PRHODREF, PLVFACT, PLSFACT, PPRES, &
11 &PDV, PKA, PCJ, &
12 &PLBDAR, PLBDAS, &
13 &PT, PRVT, PRCT, PRRT, PRST, &
14 &PRIAGGS, &
15 &PRCRIMSS, PRCRIMSG, PRSRIMCG, &
16 &PRRACCSS, PRRACCSG, PRSACCRG, PRSMLTG, &
17 &PRCMLTSR, &
18 &PRS_TEND)
19!!
20!!** PURPOSE
21!! -------
22!! Computes the fast rs processes
23!!
24!! AUTHOR
25!! ------
26!! S. Riette from the splitting of rain_ice source code (nov. 2014)
27!!
28!! MODIFICATIONS
29!! -------------
30!!
31! P. Wautelet 26/04/2019: replace non-standard FLOAT function by REAL function
32! P. Wautelet 29/05/2019: remove PACK/UNPACK intrinsics (to get more performance and better OpenACC support)
33!! R. El Khatib 24-Aug-2021 Optimizations
34! J. Wurtz 03/2022: New snow characteristics with LSNOW_T
35!
36!
37!* 0. DECLARATIONS
38! ------------
39!
40USE modd_cst, ONLY: cst_t
41USE modd_param_ice_n, ONLY: param_ice_t
42USE modd_rain_ice_descr_n, ONLY: rain_ice_descr_t
43USE modd_rain_ice_param_n, ONLY: rain_ice_param_t
44USE yomhook , ONLY : lhook, dr_hook, jphook
45!
46IMPLICIT NONE
47!
48!* 0.1 Declarations of dummy arguments :
49!
50TYPE(cst_t), INTENT(IN) :: CST
51TYPE(param_ice_t), INTENT(IN) :: PARAMI
52TYPE(rain_ice_param_t), INTENT(IN) :: ICEP
53TYPE(rain_ice_descr_t), INTENT(IN) :: ICED
54INTEGER, INTENT(IN) :: KPROMA, KSIZE
55LOGICAL, INTENT(IN) :: LDSOFT
56LOGICAL, DIMENSION(KPROMA), INTENT(IN) :: LDCOMPUTE
57REAL, DIMENSION(KPROMA), INTENT(IN) :: PRHODREF ! Reference density
58REAL, DIMENSION(KPROMA), INTENT(IN) :: PLVFACT
59REAL, DIMENSION(KPROMA), INTENT(IN) :: PLSFACT
60REAL, DIMENSION(KPROMA), INTENT(IN) :: PPRES ! absolute pressure at t
61REAL, DIMENSION(KPROMA), INTENT(IN) :: PDV ! Diffusivity of water vapor in the air
62REAL, DIMENSION(KPROMA), INTENT(IN) :: PKA ! Thermal conductivity of the air
63REAL, DIMENSION(KPROMA), INTENT(IN) :: PCJ ! Function to compute the ventilation coefficient
64REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAR ! Slope parameter of the raindrop distribution
65REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAS ! Slope parameter of the aggregate distribution
66REAL, DIMENSION(KPROMA), INTENT(IN) :: PT ! Temperature
67REAL, DIMENSION(KPROMA), INTENT(IN) :: PRVT ! Water vapor m.r. at t
68REAL, DIMENSION(KPROMA), INTENT(IN) :: PRCT ! Cloud water m.r. at t
69REAL, DIMENSION(KPROMA), INTENT(IN) :: PRRT ! Rain water m.r. at t
70REAL, DIMENSION(KPROMA), INTENT(IN) :: PRST ! Snow/aggregate m.r. at t
71REAL, DIMENSION(KPROMA), INTENT(IN) :: PRIAGGS ! r_i aggregation on r_s
72REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRCRIMSS ! Cloud droplet riming of the aggregates
73REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRCRIMSG ! Cloud droplet riming of the aggregates
74REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSRIMCG ! Cloud droplet riming of the aggregates
75REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRACCSS ! Rain accretion onto the aggregates
76REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRACCSG ! Rain accretion onto the aggregates
77REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSACCRG ! Rain accretion onto the aggregates
78REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRSMLTG ! Conversion-Melting of the aggregates
79REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRCMLTSR ! Cloud droplet collection onto aggregates by positive temperature
80REAL, DIMENSION(KPROMA, 8), INTENT(INOUT) :: PRS_TEND ! Individual tendencies
81!
82!* 0.2 declaration of local variables
83!
84INTEGER, PARAMETER :: IRCRIMS=1, ircrimss=2, irsrimcg=3, irraccs=4, irraccss=5, irsaccrg=6, &
85 & ifreez1=7, ifreez2=8
86LOGICAL, DIMENSION(KPROMA) :: GRIM, GACC
87INTEGER :: IGRIM, IGACC
88INTEGER, DIMENSION(KPROMA) :: IBUF1, IBUF2, IBUF3
89REAL, DIMENSION(KPROMA) :: ZBUF1, ZBUF2, ZBUF3
90REAL, DIMENSION(KPROMA) :: ZZW, ZZW1, ZZW2, ZZW3, ZFREEZ_RATE
91INTEGER :: JL
92REAL :: ZZW0D
93REAL(KIND=jphook) :: zhook_handle
94!-------------------------------------------------------------------------------
95!
96IF (lhook) CALL dr_hook('ICE4_FAST_RS', 0, zhook_handle)
97!
98!
99!-------------------------------------------------------------------------------
100!
101!
102!* 5.0 maximum freezing rate
103!
104!$acc kernels
105!$acc loop independent
106DO jl=1, ksize
107 IF(prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
108 IF(.NOT. ldsoft) THEN
109 prs_tend(jl, ifreez1)=prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
110 IF(parami%LEVLIMIT) THEN
111 prs_tend(jl, ifreez1)=min(prs_tend(jl, ifreez1), exp(cst%XALPI-cst%XBETAI/pt(jl)-cst%XGAMI*log(pt(jl)))) ! min(ev, es_i(T))
112 ENDIF
113 prs_tend(jl, ifreez1)=pka(jl)*(cst%XTT-pt(jl)) + &
114 &(pdv(jl)*(cst%XLVTT+(cst%XCPV-cst%XCL)*(pt(jl)-cst%XTT)) &
115 &*(cst%XESTT-prs_tend(jl, ifreez1))/(cst%XRV*pt(jl)) )
116 IF(.NOT. icep%LNEWCOEFF) THEN
117 prs_tend(jl, ifreez1)=prs_tend(jl, ifreez1)* (icep%X0DEPS* plbdas(jl)**icep%XEX0DEPS + &
118 & icep%X1DEPS*pcj(jl)*plbdas(jl)**icep%XEX1DEPS )/ &
119 &(prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
120 ELSE
121 prs_tend(jl, ifreez1)=prs_tend(jl, ifreez1)* prst(jl) *(icep%X0DEPS* plbdas(jl)**icep%XEX0DEPS + &
122 & icep%X1DEPS*pcj(jl)*plbdas(jl)**(iced%XBS+icep%XEX1DEPS )* &
123 (1+0.5*(iced%XFVELOS/plbdas(jl))**iced%XALPHAS)**(-iced%XNUS+icep%XEX1DEPS/iced%XALPHAS))/ &
124 &(prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
125 ENDIF
126 prs_tend(jl, ifreez2)=(prhodref(jl)*(cst%XLMTT+(cst%XCI-cst%XCL)*(cst%XTT-pt(jl))) ) / &
127 &(prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
128 ENDIF
129 !We must agregate, at least, the cold species
130 !And we are only interested by the freezing rate of liquid species
131 zfreez_rate(jl)=max(0., max(0., prs_tend(jl, ifreez1) + &
132 &prs_tend(jl, ifreez2) * priaggs(jl)) - &
133 priaggs(jl))
134 ELSE
135 prs_tend(jl, ifreez1)=0.
136 prs_tend(jl, ifreez2)=0.
137 zfreez_rate(jl)=0.
138 ENDIF
139ENDDO
140!$acc end kernels
141!
142!* 5.1 cloud droplet riming of the aggregates
143!
144!$acc kernels
145!$acc loop independent
146DO jl=1, ksize
147 IF (prct(jl)>iced%XRTMIN(2) .AND. prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
148 IF(.NOT. icep%LNEWCOEFF) THEN
149 zzw(jl) = plbdas(jl)
150 ELSE
151 zzw(jl) = (plbdas(jl)**iced%XALPHAS + iced%XFVELOS**iced%XALPHAS)**(1./iced%XALPHAS)
152 ENDIF
153 grim(jl) = .true.
154 ELSE
155 grim(jl) = .false.
156 prs_tend(jl, ircrims)=0.
157 prs_tend(jl, ircrimss)=0.
158 prs_tend(jl, irsrimcg)=0.
159 ENDIF
160ENDDO
161!$acc end kernels
162!
163! Collection of cloud droplets by snow: this rate is used for riming (T<0) and for conversion/melting (T>0)
164IF(.NOT. ldsoft) THEN
165 CALL interp_micro_1d(kproma, ksize, zzw, icep%NGAMINC, icep%XRIMINTP1, icep%XRIMINTP2, &
166 parami%LPACK_INTERP, grim(:), ibuf1, ibuf2, zbuf1, zbuf2, &
167 igrim, &
168 icep%XGAMINC_RIM1(:), zzw1(:), icep%XGAMINC_RIM2(:), zzw2(:), icep%XGAMINC_RIM4(:), zzw3(:))
169 IF(igrim>0) THEN
170 !
171 ! 5.1.4 riming of the small sized aggregates
172 !
173!$acc kernels
174 !$mnh_expand_where(JL=1:KSIZE)
175 IF(.NOT. icep%LNEWCOEFF) THEN
176 WHERE (grim(1:ksize))
177 prs_tend(1:ksize, ircrimss) = icep%XCRIMSS * zzw1(1:ksize) * prct(1:ksize) & ! RCRIMSS
178 * plbdas(1:ksize)**icep%XEXCRIMSS &
179 * prhodref(1:ksize)**(-iced%XCEXVT)
180 END WHERE
181 ELSE
182 WHERE (grim(1:ksize))
183 prs_tend(1:ksize, ircrimss) = icep%XCRIMSS * zzw1(1:ksize) * prct(1:ksize) & ! RCRIMSS
184 * prst(1:ksize)*(1+(iced%XFVELOS/plbdas(1:ksize))**iced%XALPHAS) &
185 **(-iced%XNUS+icep%XEXCRIMSS/iced%XALPHAS) &
186 * prhodref(1:ksize)**(-iced%XCEXVT+1.) &
187 * (plbdas(1:ksize)) ** (icep%XEXCRIMSS+iced%XBS)
188 END WHERE
189 ENDIF
190 !$mnh_end_expand_where(JL=1:KSIZE)
191!$acc end kernels
192 !
193 ! 5.1.6 riming-conversion of the large sized aggregates into graupeln
194 !
195 !
196!$acc kernels
197 !$mnh_expand_where(JL=1:KSIZE)
198 IF(.NOT. icep%LNEWCOEFF) THEN
199 WHERE(grim(1:ksize))
200 prs_tend(1:ksize, ircrims)=icep%XCRIMSG * prct(1:ksize) & ! RCRIMS
201 * plbdas(1:ksize)**icep%XEXCRIMSG &
202 * prhodref(1:ksize)**(-iced%XCEXVT)
203 END WHERE
204 ELSE
205 WHERE(grim(1:ksize))
206 prs_tend(1:ksize, ircrims)=icep%XCRIMSG * prct(1:ksize) & ! RCRIMS
207 * prst(1:ksize)*(1+(iced%XFVELOS/plbdas(1:ksize))**(iced%XALPHAS)) &
208 **(-iced%XNUS+icep%XEXCRIMSG/iced%XALPHAS) &
209 * prhodref(1:ksize)**(-iced%XCEXVT+1.) &
210 * plbdas(1:ksize)**(iced%XBS+icep%XEXCRIMSG)
211 END WHERE
212 ENDIF
213 !$mnh_end_expand_where(JL=1:KSIZE)
214!$acc end kernels
215 IF(parami%CSNOWRIMING=='M90 ')THEN
216 !Murakami 1990
217!$acc kernels
218 !$mnh_expand_where(JL=1:KSIZE)
219 WHERE(grim(1:ksize))
220 zzw(1:ksize) = prs_tend(1:ksize, ircrims) - prs_tend(1:ksize, ircrimss) ! RCRIMSG
221 END WHERE
222 IF(.NOT. icep%LNEWCOEFF) THEN
223 WHERE(grim(1:ksize))
224 prs_tend(1:ksize, irsrimcg)=icep%XSRIMCG * plbdas(1:ksize)**icep%XEXSRIMCG*(1.0-zzw2(1:ksize))
225 prs_tend(1:ksize, irsrimcg)=zzw(1:ksize)*prs_tend(1:ksize, irsrimcg)/ &
226 max(1.e-20, &
227 icep%XSRIMCG3*icep%XSRIMCG2*plbdas(1:ksize)**icep%XEXSRIMCG2*(1.-zzw3(1:ksize)) - &
228 icep%XSRIMCG3*prs_tend(1:ksize, irsrimcg))
229 END WHERE
230 ELSE
231 WHERE(grim(1:ksize))
232 prs_tend(1:ksize, irsrimcg)=icep%XSRIMCG * prst(1:ksize)*prhodref(1:ksize) &
233 * plbdas(1:ksize)**(icep%XEXSRIMCG+iced%XBS)*(1.0-zzw2(1:ksize))
234 prs_tend(1:ksize, irsrimcg)=zzw(1:ksize)*prs_tend(1:ksize, irsrimcg)/ &
235 max(1.e-20, &
236 icep%XSRIMCG3*icep%XSRIMCG2*prst(1:ksize)*prhodref(1:ksize) &
237 *plbdas(1:ksize)**icep%XEXSRIMCG2*(1.-zzw3(1:ksize)) - &
238 icep%XSRIMCG3*prs_tend(1:ksize, irsrimcg))
239
240 END WHERE
241 ENDIF
242 !$mnh_end_expand_where(JL=1:KSIZE)
243!$acc end kernels
244 ELSE
245!$acc kernels
246 prs_tend(:, irsrimcg)=0.
247!$acc end kernels
248 END IF
249 ENDIF
250ENDIF
251!
252!$acc kernels
253!$acc loop independent
254DO jl=1, ksize
255 ! More restrictive RIM mask to be used for riming by negative temperature only
256 IF(grim(jl) .AND. pt(jl)<cst%XTT) THEN
257 prcrimss(jl)=min(zfreez_rate(jl), prs_tend(jl, ircrimss))
258 zfreez_rate(jl)=max(0., zfreez_rate(jl)-prcrimss(jl))
259 zzw0d = min(1., zfreez_rate(jl) / max(1.e-20, prs_tend(jl, ircrims) - prcrimss(jl))) ! proportion we are able to freeze
260 prcrimsg(jl) = zzw0d * max(0., prs_tend(jl, ircrims) - prcrimss(jl)) ! RCRIMSG
261 zfreez_rate(jl)=max(0., zfreez_rate(jl)-prcrimsg(jl))
262 prsrimcg(jl) = zzw0d * prs_tend(jl, irsrimcg)
263
264 prsrimcg(jl) = prsrimcg(jl) * max(0., -sign(1., -prcrimsg(jl)))
265 prcrimsg(jl)=max(0., prcrimsg(jl))
266 ELSE
267 prcrimss(jl)=0.
268 prcrimsg(jl)=0.
269 prsrimcg(jl)=0.
270 ENDIF
271ENDDO
272!$acc end kernels
273!
274!* 5.2 rain accretion onto the aggregates
275!
276!$acc kernels
277!$acc loop independent
278DO jl = 1, ksize
279 IF (prrt(jl)>iced%XRTMIN(3) .AND. prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
280 gacc(jl) = .true.
281 ELSE
282 gacc(jl) = .false.
283 prs_tend(jl, irraccs)=0.
284 prs_tend(jl, irraccss)=0.
285 prs_tend(jl, irsaccrg)=0.
286 END IF
287ENDDO
288!$acc end kernels
289IF(.NOT. ldsoft) THEN
290!$acc kernels
291 prs_tend(:, irraccs)=0.
292 prs_tend(:, irraccss)=0.
293 prs_tend(:, irsaccrg)=0.
294!$acc end kernels
295 CALL interp_micro_2d(kproma, ksize, plbdas, plbdar, icep%NACCLBDAS, icep%NACCLBDAR, &
296 &icep%XACCINTP1S, icep%XACCINTP2S, icep%XACCINTP1R, icep%XACCINTP2R,&
297 &parami%LPACK_INTERP, gacc(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
298 &igacc, &
299 &icep%XKER_RACCSS(:,:), zzw1(:), icep%XKER_RACCS(:,:), zzw2(:), icep%XKER_SACCRG(:,:), zzw3(:))
300 IF(igacc>0)THEN
301 ! 5.2.4 raindrop accretion on the small sized aggregates
302 !
303!$acc kernels
304 !$mnh_expand_where(JL=1:KSIZE)
305 IF(.NOT. icep%LNEWCOEFF) THEN
306 WHERE(gacc(1:ksize))
307 zzw(1:ksize) = & !! coef of RRACCS
308 icep%XFRACCSS*( plbdas(1:ksize)**iced%XCXS )*( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
309 *( icep%XLBRACCS1/((plbdas(1:ksize)**2) ) + &
310 icep%XLBRACCS2/( plbdas(1:ksize) * plbdar(1:ksize) ) + &
311 icep%XLBRACCS3/( (plbdar(1:ksize)**2)) )/plbdar(1:ksize)**4
312 END WHERE
313 ELSE
314 WHERE(gacc(1:ksize))
315 zzw(1:ksize) = & !! coef of RRACCS
316 icep%XFRACCSS*( prst(1:ksize)*plbdas(1:ksize)**iced%XBS )*( prhodref(1:ksize)**(-iced%XCEXVT) ) &
317 *( icep%XLBRACCS1/((plbdas(1:ksize)**2) ) + &
318 icep%XLBRACCS2/( plbdas(1:ksize) * plbdar(1:ksize) ) + &
319 icep%XLBRACCS3/( (plbdar(1:ksize)**2)) )/plbdar(1:ksize)**4
320 END WHERE
321 ENDIF
322 WHERE(gacc(1:ksize))
323 prs_tend(1:ksize, irraccss) =zzw1(1:ksize)*zzw(1:ksize)
324 END WHERE
325 !$mnh_end_expand_where(JL=1:KSIZE)
326!$acc end kernels
327 !
328!$acc kernels
329 !$mnh_expand_where(JL=1:KSIZE)
330 WHERE(gacc(1:ksize))
331 prs_tend(1:ksize, irraccs) = zzw2(1:ksize)*zzw(1:ksize)
332 END WHERE
333 !$mnh_end_expand_where(JL=1:KSIZE)
334!$acc end kernels
335 !
336 ! 5.2.6 raindrop accretion-conversion of the large sized aggregates
337 ! into graupeln
338 !
339!$acc kernels
340 !$mnh_expand_where(JL=1:KSIZE)
341 IF(.NOT. icep%LNEWCOEFF) THEN
342 WHERE(gacc(1:ksize))
343 prs_tend(1:ksize, irsaccrg) = icep%XFSACCRG*zzw3(1:ksize)* & ! RSACCRG
344 ( plbdas(1:ksize)**(iced%XCXS-iced%XBS) )*( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
345 *( icep%XLBSACCR1/((plbdar(1:ksize)**2) ) + &
346 icep%XLBSACCR2/( plbdar(1:ksize) * plbdas(1:ksize) ) + &
347 icep%XLBSACCR3/( (plbdas(1:ksize)**2)) )/plbdar(1:ksize)
348 END WHERE
349 ELSE
350 WHERE(gacc(1:ksize))
351 prs_tend(1:ksize, irsaccrg) = icep%XFSACCRG*zzw3(1:ksize)* & ! RSACCRG
352 ( prst(1:ksize))*( prhodref(1:ksize)**(-iced%XCEXVT) ) &
353 *( icep%XLBSACCR1/((plbdar(1:ksize)**2) ) + &
354 icep%XLBSACCR2/( plbdar(1:ksize) * plbdas(1:ksize) ) + &
355 icep%XLBSACCR3/( (plbdas(1:ksize)**2)) )/plbdar(1:ksize)
356 END WHERE
357 ENDIF
358 !$mnh_end_expand_where(JL=1:KSIZE)
359!$acc end kernels
360 ENDIF
361ENDIF
362!
363!$acc kernels
364!$acc loop independent
365DO jl=1, ksize
366 ! More restrictive ACC mask to be used for accretion by negative temperature only
367 IF(gacc(jl) .AND. pt(jl)<cst%XTT) THEN
368 prraccss(jl)=min(zfreez_rate(jl), prs_tend(jl, irraccss))
369 zfreez_rate(jl)=max(0., zfreez_rate(jl)-prraccss(jl))
370 zzw(jl) = min(1., zfreez_rate(jl) / max(1.e-20, prs_tend(jl, irraccs)-prraccss(jl))) ! proportion we are able to freeze
371 prraccsg(jl)=zzw(jl) * max(0., prs_tend(jl, irraccs)-prraccss(jl))
372 zfreez_rate(jl) = max(0., zfreez_rate(jl)-prraccsg(jl))
373 prsaccrg(jl)=zzw(jl) * prs_tend(jl, irsaccrg)
374
375 prsaccrg(jl) = prsaccrg(jl) * max(0., -sign(1., -prraccsg(jl)))
376 prraccsg(jl)=max(0., prraccsg(jl))
377 ELSE
378 prraccss(jl)=0.
379 prraccsg(jl)=0.
380 prsaccrg(jl)=0.
381 ENDIF
382ENDDO
383!$acc end kernels
384!
385!
386!* 5.3 Conversion-Melting of the aggregates
387!
388!$acc kernels
389!$acc loop independent
390DO jl=1, ksize
391 IF(prst(jl)>iced%XRTMIN(5) .AND. pt(jl)>cst%XTT .AND. ldcompute(jl)) THEN
392 IF(.NOT. ldsoft) THEN
393 prsmltg(jl)=prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
394 IF(parami%LEVLIMIT) THEN
395 prsmltg(jl)=min(prsmltg(jl), exp(cst%XALPW-cst%XBETAW/pt(jl)-cst%XGAMW*log(pt(jl)))) ! min(ev, es_w(T))
396 ENDIF
397 prsmltg(jl)= pka(jl)*(cst%XTT-pt(jl)) + &
398 &(pdv(jl)*(cst%XLVTT + ( cst%XCPV - cst%XCL ) * ( pt(jl) - cst%XTT )) &
399 & *(cst%XESTT-prsmltg(jl))/(cst%XRV*pt(jl)) )
400 !
401 ! compute RSMLT
402 !
403 IF(.NOT. icep%LNEWCOEFF) THEN
404 prsmltg(jl) = icep%XFSCVMG*max(0., (-prsmltg(jl) * &
405 (icep%X0DEPS* plbdas(jl)**icep%XEX0DEPS + &
406 icep%X1DEPS*pcj(jl)*plbdas(jl)**icep%XEX1DEPS) &
407 -(prs_tend(jl, ircrims) + prs_tend(jl, irraccs)) * &
408 (prhodref(jl)*cst%XCL*(cst%XTT-pt(jl))) &
409 ) / (prhodref(jl)*cst%XLMTT))
410 ELSE
411 prsmltg(jl) = icep%XFSCVMG*max(0., (-prsmltg(jl) * &
412 prst(jl)*prhodref(jl) * &
413 (icep%X0DEPS* plbdas(jl)**(iced%XBS+icep%XEX0DEPS) + &
414 icep%X1DEPS*pcj(jl)* &
415 (1+0.5*(iced%XFVELOS/plbdas(jl))**iced%XALPHAS)**(-iced%XNUS+icep%XEX1DEPS/iced%XALPHAS) &
416 *plbdas(jl)**(iced%XBS+icep%XEX1DEPS)) &
417 -(prs_tend(jl, ircrims) + prs_tend(jl, irraccs)) * &
418 (prhodref(jl)*cst%XCL*(cst%XTT-pt(jl))) &
419 ) / (prhodref(jl)*cst%XLMTT))
420 ENDIF
421 !
