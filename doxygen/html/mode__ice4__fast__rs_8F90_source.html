49 !
50 TYPE(cst_t), INTENT(IN) :: CST
51 TYPE(param_ice_t), INTENT(IN) :: PARAMI
52 TYPE(rain_ice_param_t), INTENT(IN) :: ICEP
53 TYPE(rain_ice_descr_t), INTENT(IN) :: ICED
54 INTEGER, INTENT(IN) :: KPROMA, KSIZE
55 LOGICAL, INTENT(IN) :: LDSOFT
56 LOGICAL, DIMENSION(KPROMA), INTENT(IN) :: LDCOMPUTE
57 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRHODREF ! Reference density
58 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLVFACT
59 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLSFACT
60 REAL, DIMENSION(KPROMA), INTENT(IN) :: PPRES ! absolute pressure at t
61 REAL, DIMENSION(KPROMA), INTENT(IN) :: PDV ! Diffusivity of water vapor in the air
62 REAL, DIMENSION(KPROMA), INTENT(IN) :: PKA ! Thermal conductivity of the air
63 REAL, DIMENSION(KPROMA), INTENT(IN) :: PCJ ! Function to compute the ventilation coefficient
64 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAR ! Slope parameter of the raindrop distribution
65 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAS ! Slope parameter of the aggregate distribution

67 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRVT ! Water vapor m.r. at t
68 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRCT ! Cloud water m.r. at t
69 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRRT ! Rain water m.r. at t
70 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRST ! Snow/aggregate m.r. at t
71 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRIAGGS ! r_i aggregation on r_s
72 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRCRIMSS ! Cloud droplet riming of the aggregates
73 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRCRIMSG ! Cloud droplet riming of the aggregates
74 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSRIMCG ! Cloud droplet riming of the aggregates
75 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRACCSS ! Rain accretion onto the aggregates
76 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRACCSG ! Rain accretion onto the aggregates
77 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSACCRG ! Rain accretion onto the aggregates
78 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRSMLTG ! Conversion-Melting of the aggregates
79 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRCMLTSR ! Cloud droplet collection onto aggregates by positive temperature
80 REAL, DIMENSION(KPROMA, 8), INTENT(INOUT) :: PRS_TEND ! Individual tendencies
81 !

83 !
84 INTEGER, PARAMETER :: IRCRIMS=1, ircrimss=2, irsrimcg=3, irraccs=4, irraccss=5, irsaccrg=6, &
85  & ifreez1=7, ifreez2=8
86 LOGICAL, DIMENSION(KPROMA) :: GRIM, GACC
87 INTEGER :: IGRIM, IGACC
88 INTEGER, DIMENSION(KPROMA) :: IBUF1, IBUF2, IBUF3
89 REAL, DIMENSION(KPROMA) :: ZBUF1, ZBUF2, ZBUF3
90 REAL, DIMENSION(KPROMA) :: ZZW, ZZW1, ZZW2, ZZW3, ZFREEZ_RATE
91 INTEGER :: JL
92 REAL :: ZZW0D
93 REAL(KIND=jphook) :: zhook_handle
94 !-------------------------------------------------------------------------------
95 !
96 IF (lhook) CALL dr_hook('ICE4_FAST_RS', 0, zhook_handle)
97 !
98 !
99 !-------------------------------------------------------------------------------
100 !
101 !
102 !* 5.0 maximum freezing rate
103 !
104 DO jl=1, ksize
105  IF(prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
106  IF(.NOT. ldsoft) THEN
107  prs_tend(jl, ifreez1)=prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
108  IF(parami%LEVLIMIT) THEN
109  prs_tend(jl, ifreez1)=min(prs_tend(jl, ifreez1), exp(cst%XALPI-cst%XBETAI/pt(jl)-cst%XGAMI*alog(pt(jl)))) ! min(ev, es_i(T))
110  ENDIF
111  prs_tend(jl, ifreez1)=pka(jl)*(cst%XTT-pt(jl)) + &
112  &(pdv(jl)*(cst%XLVTT+(cst%XCPV-cst%XCL)*(pt(jl)-cst%XTT)) &
113  &*(cst%XESTT-prs_tend(jl, ifreez1))/(cst%XRV*pt(jl)) )
114 #ifdef REPRO48
115  prs_tend(jl, ifreez1)=prs_tend(jl, ifreez1)* (icep%X0DEPS* plbdas(jl)**icep%XEX0DEPS + &
116  & icep%X1DEPS*pcj(jl)*plbdas(jl)**icep%XEX1DEPS )/ &
117  &(prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
118 #else
119  prs_tend(jl, ifreez1)=prs_tend(jl, ifreez1)* prst(jl) *(icep%X0DEPS* plbdas(jl)**icep%XEX0DEPS + &
120  & icep%X1DEPS*pcj(jl)*plbdas(jl)**(iced%XBS+icep%XEX1DEPS )* &
121  (1+0.5*(iced%XFVELOS/plbdas(jl))**iced%XALPHAS)**(-iced%XNUS+icep%XEX1DEPS/iced%XALPHAS))/ &
122  &(prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
123 #endif
124  prs_tend(jl, ifreez2)=(prhodref(jl)*(cst%XLMTT+(cst%XCI-cst%XCL)*(cst%XTT-pt(jl))) ) / &
125  &(prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
126  ENDIF
127  !We must agregate, at least, the cold species
128  !And we are only interested by the freezing rate of liquid species
129  zfreez_rate(jl)=max(0., max(0., prs_tend(jl, ifreez1) + &
130  &prs_tend(jl, ifreez2) * priaggs(jl)) - &
131  priaggs(jl))
132  ELSE
133  prs_tend(jl, ifreez1)=0.
134  prs_tend(jl, ifreez2)=0.
135  zfreez_rate(jl)=0.
136  ENDIF
137 ENDDO
138 !
139 !* 5.1 cloud droplet riming of the aggregates
140 !
141 DO jl=1, ksize
142  IF (prct(jl)>iced%XRTMIN(2) .AND. prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
143 #ifdef REPRO48
144  zzw(jl) = plbdas(jl)
145 #else
146  zzw(jl) = (plbdas(jl)**iced%XALPHAS + iced%XFVELOS**iced%XALPHAS)**(1./iced%XALPHAS)
147 #endif
148  grim(jl) = .true.
149  ELSE
150  grim(jl) = .false.
151  prs_tend(jl, ircrims)=0.
152  prs_tend(jl, ircrimss)=0.
153  prs_tend(jl, irsrimcg)=0.
154  ENDIF
155 ENDDO
156 !
157 ! Collection of cloud droplets by snow: this rate is used for riming (T<0) and for conversion/melting (T>0)
158 IF(.NOT. ldsoft) THEN
159  CALL interp_micro_1d(kproma, ksize, zzw, icep%NGAMINC, icep%XRIMINTP1, icep%XRIMINTP2, &
160  parami%LPACK_INTERP, grim(:), ibuf1, ibuf2, zbuf1, zbuf2, &
161  igrim, &
162  icep%XGAMINC_RIM1(:), zzw1(:), icep%XGAMINC_RIM2(:), zzw2(:), icep%XGAMINC_RIM4(:), zzw3(:))
163  IF(igrim>0) THEN
164  !
165  ! 5.1.4 riming of the small sized aggregates
166  !
167  !$mnh_expand_where(JL=1:KSIZE)
168  WHERE (grim(1:ksize))
169 #ifdef REPRO48
170  prs_tend(1:ksize, ircrimss) = icep%XCRIMSS * zzw1(1:ksize) * prct(1:ksize) & ! RCRIMSS
171  * plbdas(1:ksize)**icep%XEXCRIMSS &
172  * prhodref(1:ksize)**(-iced%XCEXVT)
173 #else
174  prs_tend(1:ksize, ircrimss) = icep%XCRIMSS * zzw1(1:ksize) * prct(1:ksize) & ! RCRIMSS
175  * prst(1:ksize)*(1+(iced%XFVELOS/plbdas(1:ksize))**iced%XALPHAS) &
176  **(-iced%XNUS+icep%XEXCRIMSS/iced%XALPHAS) &
177  * prhodref(1:ksize)**(-iced%XCEXVT+1.) &
178  * (plbdas(1:ksize)) ** (icep%XEXCRIMSS+iced%XBS)
179 #endif
180  END WHERE
181  !$mnh_end_expand_where(JL=1:KSIZE)
182  !
183  ! 5.1.6 riming-conversion of the large sized aggregates into graupeln
184  !
185  !
186  !$mnh_expand_where(JL=1:KSIZE)
187  WHERE(grim(1:ksize))
188 #ifdef REPRO48
189  prs_tend(1:ksize, ircrims)=icep%XCRIMSG * prct(1:ksize) & ! RCRIMS
190  * plbdas(1:ksize)**icep%XEXCRIMSG &
191  * prhodref(1:ksize)**(-iced%XCEXVT)
192 #else
193  prs_tend(1:ksize, ircrims)=icep%XCRIMSG * prct(1:ksize) & ! RCRIMS
194  * prst(1:ksize)*(1+(iced%XFVELOS/plbdas(1:ksize))**(iced%XALPHAS)) &
195  **(-iced%XNUS+icep%XEXCRIMSG/iced%XALPHAS) &
196  * prhodref(1:ksize)**(-iced%XCEXVT+1.) &
197  * plbdas(1:ksize)**(iced%XBS+icep%XEXCRIMSG)
198 #endif
199  END WHERE
200  !$mnh_end_expand_where(JL=1:KSIZE)
201 
202  IF(parami%CSNOWRIMING=='M90 ')THEN
203  !Murakami 1990
204  !$mnh_expand_where(JL=1:KSIZE)
205  WHERE(grim(1:ksize))
206  zzw(1:ksize) = prs_tend(1:ksize, ircrims) - prs_tend(1:ksize, ircrimss) ! RCRIMSG
207 #ifdef REPRO48
208  prs_tend(1:ksize, irsrimcg)=icep%XSRIMCG * plbdas(1:ksize)**icep%XEXSRIMCG*(1.0-zzw2(1:ksize))
209 #else
210  prs_tend(1:ksize, irsrimcg)=icep%XSRIMCG * prst(1:ksize)*prhodref(1:ksize) &
211  * plbdas(1:ksize)**(icep%XEXSRIMCG+iced%XBS)*(1.0-zzw2(1:ksize))
212 #endif
213 #ifdef REPRO48
214  prs_tend(1:ksize, irsrimcg)=zzw(1:ksize)*prs_tend(1:ksize, irsrimcg)/ &
215  max(1.e-20, &
216  icep%XSRIMCG3*icep%XSRIMCG2*plbdas(1:ksize)**icep%XEXSRIMCG2*(1.-zzw3(1:ksize)) - &
217  icep%XSRIMCG3*prs_tend(1:ksize, irsrimcg))
218 #else
219  prs_tend(1:ksize, irsrimcg)=zzw(1:ksize)*prs_tend(1:ksize, irsrimcg)/ &
220  max(1.e-20, &
221  icep%XSRIMCG3*icep%XSRIMCG2*prst(1:ksize)*prhodref(1:ksize) &
222  *plbdas(1:ksize)**icep%XEXSRIMCG2*(1.-zzw3(1:ksize)) - &
223  icep%XSRIMCG3*prs_tend(1:ksize, irsrimcg))
224 #endif
225  END WHERE
226  !$mnh_end_expand_where(JL=1:KSIZE)
227  ELSE
228  prs_tend(:, irsrimcg)=0.
229  END IF
230  ENDIF
231 ENDIF
232 !
233 DO jl=1, ksize
234  ! More restrictive RIM mask to be used for riming by negative temperature only
235  IF(grim(jl) .AND. pt(jl)<cst%XTT) THEN
236  prcrimss(jl)=min(zfreez_rate(jl), prs_tend(jl, ircrimss))
237  zfreez_rate(jl)=max(0., zfreez_rate(jl)-prcrimss(jl))
238  zzw0d = min(1., zfreez_rate(jl) / max(1.e-20, prs_tend(jl, ircrims) - prcrimss(jl))) ! proportion we are able to freeze
239  prcrimsg(jl) = zzw0d * max(0., prs_tend(jl, ircrims) - prcrimss(jl)) ! RCRIMSG
240  zfreez_rate(jl)=max(0., zfreez_rate(jl)-prcrimsg(jl))
241  prsrimcg(jl) = zzw0d * prs_tend(jl, irsrimcg)
242 
243  prsrimcg(jl) = prsrimcg(jl) * max(0., -sign(1., -prcrimsg(jl)))
244  prcrimsg(jl)=max(0., prcrimsg(jl))
245  ELSE
246  prcrimss(jl)=0.
247  prcrimsg(jl)=0.
248  prsrimcg(jl)=0.
249  ENDIF
250 ENDDO
251 !
252 !* 5.2 rain accretion onto the aggregates
253 !
254 DO jl = 1, ksize
255  IF (prrt(jl)>iced%XRTMIN(3) .AND. prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
256  gacc(jl) = .true.
257  ELSE
258  gacc(jl) = .false.
259  prs_tend(jl, irraccs)=0.
260  prs_tend(jl, irraccss)=0.
261  prs_tend(jl, irsaccrg)=0.
262  END IF
263 ENDDO
264 IF(.NOT. ldsoft) THEN
265  prs_tend(:, irraccs)=0.
266  prs_tend(:, irraccss)=0.
267  prs_tend(:, irsaccrg)=0.
268  CALL interp_micro_2d(kproma, ksize, plbdas, plbdar, icep%NACCLBDAS, icep%NACCLBDAR, &
269  &icep%XACCINTP1S, icep%XACCINTP2S, icep%XACCINTP1R, icep%XACCINTP2R,&
270  &parami%LPACK_INTERP, gacc(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
271  &igacc, &
272  &icep%XKER_RACCSS(:,:), zzw1(:), icep%XKER_RACCS(:,:), zzw2(:), icep%XKER_SACCRG(:,:), zzw3(:))
273  IF(igacc>0)THEN
274  ! 5.2.4 raindrop accretion on the small sized aggregates
275  !
276  !$mnh_expand_where(JL=1:KSIZE)
277  WHERE(gacc(1:ksize))
278 #ifdef REPRO48
279  zzw(1:ksize) = & !! coef of RRACCS
280  icep%XFRACCSS*( plbdas(1:ksize)**iced%XCXS )*( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
281  *( icep%XLBRACCS1/((plbdas(1:ksize)**2) ) + &
282  icep%XLBRACCS2/( plbdas(1:ksize) * plbdar(1:ksize) ) + &
283  icep%XLBRACCS3/( (plbdar(1:ksize)**2)) )/plbdar(1:ksize)**4
284 #else
285  zzw(1:ksize) = & !! coef of RRACCS
286  icep%XFRACCSS*( prst(1:ksize)*plbdas(1:ksize)**iced%XBS )*( prhodref(1:ksize)**(-iced%XCEXVT) ) &
287  *( icep%XLBRACCS1/((plbdas(1:ksize)**2) ) + &
288  icep%XLBRACCS2/( plbdas(1:ksize) * plbdar(1:ksize) ) + &
289  icep%XLBRACCS3/( (plbdar(1:ksize)**2)) )/plbdar(1:ksize)**4
290 #endif
291  prs_tend(1:ksize, irraccss) =zzw1(1:ksize)*zzw(1:ksize)
292  END WHERE
293  !$mnh_end_expand_where(JL=1:KSIZE)
294  !
295  !$mnh_expand_where(JL=1:KSIZE)
296  WHERE(gacc(1:ksize))
297  prs_tend(1:ksize, irraccs) = zzw2(1:ksize)*zzw(1:ksize)
298  END WHERE
299  !$mnh_end_expand_where(JL=1:KSIZE)
300  !
301  ! 5.2.6 raindrop accretion-conversion of the large sized aggregates
302  ! into graupeln
303  !
304  !$mnh_expand_where(JL=1:KSIZE)
305  WHERE(gacc(1:ksize))
306 #ifdef REPRO48
307  prs_tend(1:ksize, irsaccrg) = icep%XFSACCRG*zzw3(1:ksize)* & ! RSACCRG
308  ( plbdas(1:ksize)**(iced%XCXS-iced%XBS) )*( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
309  *( icep%XLBSACCR1/((plbdar(1:ksize)**2) ) + &
310  icep%XLBSACCR2/( plbdar(1:ksize) * plbdas(1:ksize) ) + &
311  icep%XLBSACCR3/( (plbdas(1:ksize)**2)) )/plbdar(1:ksize)
312 #else
313  prs_tend(1:ksize, irsaccrg) = icep%XFSACCRG*zzw3(1:ksize)* & ! RSACCRG
314  ( prst(1:ksize))*( prhodref(1:ksize)**(-iced%XCEXVT) ) &
315  *( icep%XLBSACCR1/((plbdar(1:ksize)**2) ) + &
316  icep%XLBSACCR2/( plbdar(1:ksize) * plbdas(1:ksize) ) + &
317  icep%XLBSACCR3/( (plbdas(1:ksize)**2)) )/plbdar(1:ksize)
318 #endif
319  END WHERE
320  !$mnh_end_expand_where(JL=1:KSIZE)
321  ENDIF
322 ENDIF
323 !
324 DO jl=1, ksize
325  ! More restrictive ACC mask to be used for accretion by negative temperature only
326  IF(gacc(jl) .AND. pt(jl)<cst%XTT) THEN
327  prraccss(jl)=min(zfreez_rate(jl), prs_tend(jl, irraccss))
328  zfreez_rate(jl)=max(0., zfreez_rate(jl)-prraccss(jl))
329  zzw(jl) = min(1., zfreez_rate(jl) / max(1.e-20, prs_tend(jl, irraccs)-prraccss(jl))) ! proportion we are able to freeze
330  prraccsg(jl)=zzw(jl) * max(0., prs_tend(jl, irraccs)-prraccss(jl))
331  zfreez_rate(jl) = max(0., zfreez_rate(jl)-prraccsg(jl))
332  prsaccrg(jl)=zzw(jl) * prs_tend(jl, irsaccrg)
333 
334  prsaccrg(jl) = prsaccrg(jl) * max(0., -sign(1., -prraccsg(jl)))
335  prraccsg(jl)=max(0., prraccsg(jl))
336  ELSE
337  prraccss(jl)=0.
338  prraccsg(jl)=0.
339  prsaccrg(jl)=0.
340  ENDIF
341 ENDDO
342 !
343 !
344 !* 5.3 Conversion-Melting of the aggregates
345 !
346 DO jl=1, ksize
347  IF(prst(jl)>iced%XRTMIN(5) .AND. pt(jl)>cst%XTT .AND. ldcompute(jl)) THEN
348  IF(.NOT. ldsoft) THEN
349  prsmltg(jl)=prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
350  IF(parami%LEVLIMIT) THEN
351  prsmltg(jl)=min(prsmltg(jl), exp(cst%XALPW-cst%XBETAW/pt(jl)-cst%XGAMW*alog(pt(jl)))) ! min(ev, es_w(T))
352  ENDIF
353  prsmltg(jl)= pka(jl)*(cst%XTT-pt(jl)) + &
354  &(pdv(jl)*(cst%XLVTT + ( cst%XCPV - cst%XCL ) * ( pt(jl) - cst%XTT )) &
355  & *(cst%XESTT-prsmltg(jl))/(cst%XRV*pt(jl)) )
356  !
357  ! compute RSMLT
358  !
359 #ifdef REPRO48
360  prsmltg(jl) = icep%XFSCVMG*max(0., (-prsmltg(jl) * &
361  (icep%X0DEPS* plbdas(jl)**icep%XEX0DEPS + &
362  icep%X1DEPS*pcj(jl)*plbdas(jl)**icep%XEX1DEPS) &
363  -(prs_tend(jl, ircrims) + prs_tend(jl, irraccs)) * &
364  (prhodref(jl)*cst%XCL*(cst%XTT-pt(jl))) &
365  ) / (prhodref(jl)*cst%XLMTT))
366 #else
367  prsmltg(jl) = icep%XFSCVMG*max(0., (-prsmltg(jl) * &
368  prst(jl)*prhodref(jl) * &
369  (icep%X0DEPS* plbdas(jl)**(iced%XBS+icep%XEX0DEPS) + &
370  icep%X1DEPS*pcj(jl)*(1+0.5*(iced%XFVELOS/plbdas(jl))**iced%XALPHAS)**(-iced%XNUS+icep%XEX1DEPS/iced%XALPHAS) &
371  *plbdas(jl)**(iced%XBS+icep%XEX1DEPS)) &
372  -(prs_tend(jl, ircrims) + prs_tend(jl, irraccs)) * &
373  (prhodref(jl)*cst%XCL*(cst%XTT-pt(jl))) &
374  ) / (prhodref(jl)*cst%XLMTT))
375 #endif
376  !
377  ! note that RSCVMG = RSMLT*XFSCVMG but no heat is exchanged (at the rate RSMLT)
378  ! because the graupeln produced by this process are still icy!!!
379  !
380  ! When T < XTT, rc is collected by snow (riming) to produce snow and graupel
