29 TYPE(cst_t), INTENT(IN) :: cst
30 CHARACTER(LEN=1), INTENT(IN) :: hfrac_ice ! scheme to use
31 TYPE(neb_t), INTENT(IN) :: nebn
32 REAL, INTENT(IN) :: pt ! temperature
33 REAL, INTENT(INOUT) :: pfrac_ice ! Ice fraction (1 for ice only, 0 for liquid only)
34 INTEGER, OPTIONAL, INTENT(OUT) :: kerr ! Error code in return
35 !
36 !------------------------------------------------------------------------
37 
38 ! 1. Compute FRAC_ICE
39 !
40 IF (PRESENT(kerr)) kerr=0
41 SELECT CASE(hfrac_ice)
42  CASE ('T') !using Temperature
43  pfrac_ice = max( 0., min(1., (( nebn%XTMAXMIX - pt ) / ( nebn%XTMAXMIX - nebn%XTMINMIX )) ) ) ! freezing interval
44  CASE ('O') !using Temperature with old formulae
45  pfrac_ice = max( 0., min(1., (( cst%XTT - pt ) / 40.) ) ) ! freezing interval
46  CASE ('N') !No ice
47  pfrac_ice = 0.
48  CASE ('S') !Same as previous
49  ! (almost) nothing to do
50  pfrac_ice = max( 0., min(1., pfrac_ice ) )
51  CASE DEFAULT
52  IF (PRESENT(kerr)) kerr=1
53 END SELECT
54 
55 END SUBROUTINE compute_frac_ice
