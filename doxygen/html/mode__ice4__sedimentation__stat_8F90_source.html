15 &PSEA, PTOWN, &
16 &PINPRH, PRHT, PRHS, PFPR)
17
18!!
19!!** PURPOSE
20!! -------
21!! Computes the sedimentation
22!!
23!! AUTHOR
24!! ------
25!! S. Riette from the plitting of rain_ice source code (nov. 2014)
26!!
27!! MODIFICATIONS
28!! -------------
29!!
30! P. Wautelet 10/04/2019: replace ABORT and STOP calls by Print_msg
31! P. Wautelet 28/05/2019: move COUNTJV function to tools.f90
32!! Ryad El Khatib 09-Oct-2019 Substantial re-write for optimization
33!! (outerunrolling, vectorization, memory cache saving, unrolling)
34! P. Wautelet 21/01/2021: initialize untouched part of PFPR
35! J. Wurtz 03/2022: New snow characteristics with LSNOW_T
36!
37!
38!* 0. DECLARATIONS
39! ------------
40!
41USE yomhook , ONLY : lhook, dr_hook, jphook
42USE modd_dimphyex, ONLY: dimphyex_t
43USE modd_cst, ONLY: cst_t
44USE modd_rain_ice_descr_n, ONLY: rain_ice_descr_t
45USE modd_rain_ice_param_n, ONLY: rain_ice_param_t
46USE modd_param_ice_n, ONLY: param_ice_t
47USE modi_gamma, ONLY: gamma
48!
49IMPLICIT NONE
50!
51!* 0.1 Declarations of dummy arguments :
52!
53TYPE(dimphyex_t), INTENT(IN) :: D !array dimensions
54TYPE(cst_t), INTENT(IN) :: CST
55TYPE(rain_ice_param_t), INTENT(IN) :: ICEP
56TYPE(rain_ice_descr_t), INTENT(IN) :: ICED
57TYPE(param_ice_t), INTENT(IN) :: PARAMI
58REAL, INTENT(IN) :: PTSTEP ! Double Time step (single if cold start)
59INTEGER, INTENT(IN) :: KRR ! Number of moist variable
60REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDZZ ! Layer thikness (m)
61REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODREF! Reference density
62REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PPABST ! absolute pressure at t
63REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTHT ! Theta at time t
64REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PT ! Temperature
65REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODJ ! Dry density * Jacobian
66REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRCS ! Cloud water m.r. source
67REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRCT ! Cloud water m.r. at t
68REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRRS ! Rain water m.r. source
69REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRRT ! Rain water m.r. at t
70REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRIS ! Pristine ice m.r. source
71REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRIT ! Pristine ice m.r. at t
72REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRSS ! Snow/aggregate m.r. source
73REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRST ! Snow/aggregate m.r. at t
74REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRGS ! Graupel m.r. source
75REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRGT ! Graupel/hail m.r. at t
76REAL, DIMENSION(D%NIJT), INTENT(OUT) :: PINPRC ! Cloud instant precip
77REAL, DIMENSION(D%NIJT), INTENT(OUT) :: PINPRR ! Rain instant precip
78REAL, DIMENSION(D%NIJT), INTENT(OUT) :: PINPRI ! Pristine ice instant precip
79REAL, DIMENSION(D%NIJT), INTENT(OUT) :: PINPRS ! Snow instant precip
80REAL, DIMENSION(D%NIJT), INTENT(OUT) :: PINPRG ! Graupel instant precip
81REAL, DIMENSION(D%NIJT), OPTIONAL, INTENT(IN) :: PSEA ! Sea Mask
82REAL, DIMENSION(D%NIJT), OPTIONAL, INTENT(IN) :: PTOWN ! Fraction that is town
83REAL, DIMENSION(D%NIJT), OPTIONAL, INTENT(OUT) :: PINPRH ! Hail instant precip
84REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(IN) :: PRHT ! Hail m.r. at t
85REAL, DIMENSION(D%NIJT,D%NKT), OPTIONAL, INTENT(INOUT) :: PRHS ! Hail m.r. source
86REAL, DIMENSION(D%NIJT,D%NKT,KRR), OPTIONAL, INTENT(OUT) :: PFPR ! upper-air precipitation fluxes
87!
88!* 0.2 declaration of local variables
89!
90LOGICAL :: LLSEA_AND_TOWN
91INTEGER :: JRR, JIJ, JK, IKB, IKE,IKL, IIJB, IIJE, IKTB, IKTE
92INTEGER :: ISHIFT, IK, IKPLUS
93REAL :: ZINVTSTEP, ZGAC, ZGC, ZGAC2, ZGC2, ZRAYDEFO
94REAL, DIMENSION(D%NIJT) :: ZTSORHODZ ! TimeStep Over (Rhodref times delta Z)
95REAL, DIMENSION(D%NIJT,0:1,2:KRR) :: ZSED ! sedimentation flux array for each species and for above and current levels
96!
97REAL(KIND=jphook) :: zhook_handle
98!
99!-------------------------------------------------------------------------------
100IF (lhook) CALL dr_hook('ICE4_SEDIMENTATION_STAT',0,zhook_handle)
101!
102ikb=d%NKB
103ike=d%NKE
104ikl=d%NKL
105iijb=d%NIJB
106iije=d%NIJE
107iktb=d%NKTB
108ikte=d%NKTE
109!
110IF ( PRESENT( pfpr ) ) THEN
111 !Set to 0. to avoid undefined values (in files)
112 pfpr(:, : iktb, :) = 0.
113 pfpr(:, ikte :, :) = 0.
114END IF
115
116!-------------------------------------------------------------------------------
117!
118!* 1. compute the fluxes
119!
120zinvtstep = 1./ptstep
121zgac=gamma(iced%XNUC+1.0/iced%XALPHAC)
122zgc=gamma(iced%XNUC)
123zgac2=gamma(iced%XNUC2+1.0/iced%XALPHAC2)
124zgc2=gamma(iced%XNUC2)
125zraydefo=max(1.,0.5*(zgac/zgc))
126llsea_and_town=PRESENT(psea).AND.PRESENT(ptown)
127
128!
129!* 2. compute the fluxes
130!
131! Start shift mechanism:
132ishift=0
133CALL shift
134
135! Initialize vertical loop
136DO jrr=2,krr
137 zsed(:,ikplus,jrr) = 0.
138ENDDO
139
140! calculation sedimentation flux
141DO jk = ike , ikb, -1*ikl
142
143 DO jij = iijb, iije
144 ztsorhodz(jij) =ptstep/(prhodref(jij,jk)*pdzz(jij,jk))
145 ENDDO
146!
147 DO jrr=2,krr
148
149 IF (jrr==2) THEN
150
151 !******* for cloud
152 IF (parami%LSEDIC) THEN
153 CALL cloud(prct(:,jk))
154 ELSE
155 zsed(:,ik,jrr)=0.
156 ENDIF
157
158 ELSEIF (jrr==3) THEN
159
160 !* 2.2 for rain
161 CALL other_species(icep%XFSEDR,icep%XEXSEDR,prrt(:,jk))
162
163 ELSEIF (jrr==4) THEN
164
165 CALL pristine_ice(prit(:,jk))
166
167 ELSEIF (jrr==5) THEN
168
169 !* 2.4 for aggregates/snow
170 IF(.NOT. icep%LNEWCOEFF) THEN
171 CALL other_species(icep%XFSEDS,icep%XEXSEDS,prst(:,jk))
172 ELSE
173 CALL snow(prst(:,jk))
174 ENDIF
175
176 ELSEIF (jrr==6) THEN
177
178 !* 2.5 for graupeln
179 CALL other_species(icep%XFSEDG,icep%XEXSEDG,prgt(:,jk))
180
181 ELSEIF (jrr==7) THEN
182
183 !* 2.6 for hail
184 IF (PRESENT(prht)) THEN
185 CALL other_species(icep%XFSEDH,icep%XEXSEDH,prht(:,jk))
186 ENDIF
187
188 ENDIF
189
190 ENDDO ! JRR
191
192 ! Wrap-up
193
194 IF(PRESENT(pfpr)) THEN
195 DO jrr=2,krr
196 pfpr(:,jk,jrr)=zsed(:,ik,jrr)
197 ENDDO
198 ENDIF
199
200 DO jij = iijb, iije
201 prcs(jij,jk) = prcs(jij,jk)+ztsorhodz(jij)*(zsed(jij,ikplus,2)-zsed(jij,ik,2))*zinvtstep
202 prrs(jij,jk) = prrs(jij,jk)+ztsorhodz(jij)*(zsed(jij,ikplus,3)-zsed(jij,ik,3))*zinvtstep
203 pris(jij,jk) = pris(jij,jk)+ztsorhodz(jij)*(zsed(jij,ikplus,4)-zsed(jij,ik,4))*zinvtstep
204 prss(jij,jk) = prss(jij,jk)+ztsorhodz(jij)*(zsed(jij,ikplus,5)-zsed(jij,ik,5))*zinvtstep
205 prgs(jij,jk) = prgs(jij,jk)+ztsorhodz(jij)*(zsed(jij,ikplus,6)-zsed(jij,ik,6))*zinvtstep
206 IF (PRESENT(prhs) .AND. krr==7) THEN
207 prhs(jij,jk) = prhs(jij,jk)+ztsorhodz(jij)*(zsed(jij,ikplus,7)-zsed(jij,ik,7))*zinvtstep
208 ENDIF
209 ENDDO
210
211 IF (jk==ikb) THEN
212 DO jij = iijb, iije
213 IF(parami%LSEDIC) pinprc(jij) = zsed(jij,ik,2)/cst%XRHOLW
214 pinprr(jij) = zsed(jij,ik,3)/cst%XRHOLW
215 pinpri(jij) = zsed(jij,ik,4)/cst%XRHOLW
216 pinprs(jij) = zsed(jij,ik,5)/cst%XRHOLW
217 pinprg(jij) = zsed(jij,ik,6)/cst%XRHOLW
218 IF (PRESENT(pinprh) .AND. krr==7) THEN
219 pinprh(jij) = zsed(jij,ik,7)/cst%XRHOLW
220 ENDIF
221 ENDDO
222 ENDIF
223
224 ! shift mechanism : current level now takes the place of previous one
225 CALL shift
226
227ENDDO ! JK
228
229IF (lhook) CALL dr_hook('ICE4_SEDIMENTATION_STAT',1,zhook_handle)
230
231CONTAINS
232
233 SUBROUTINE cloud(PRXT)
234
235 REAL, INTENT(IN) :: PRXT(D%NIJT) ! mr of specy X
236
237 REAL :: ZLBC ! XLBC weighted by sea fraction
238 REAL :: ZFSEDC
239 REAL :: ZCONC3D ! droplet condensation
240 REAL :: ZRAY ! Cloud Mean radius
241 REAL :: ZZWLBDA, ZZWLBDC, ZZCC
242 INTEGER :: JIJ
243 REAL :: ZQP
244 REAL :: ZWSEDW1, ZWSEDW2
245
246 !!REAL(KIND=JPHOOK) :: ZHOOK_HANDLE
247
248 !!IF (LHOOK) CALL DR_HOOK('ICE4_SEDIMENTATION_STAT:CLOUD',0,ZHOOK_HANDLE)
249
250 DO jij = iijb, iije
251 !estimation of q' taking into account incoming ZWSED from previous vertical level
252 zqp=zsed(jij,ikplus,jrr)*ztsorhodz(jij)
253 IF ((prxt(jij) > iced%XRTMIN(jrr)) .OR. (zqp > iced%XRTMIN(jrr))) THEN
254 IF (llsea_and_town) THEN
255 zray = max(1.,0.5*((1.-psea(jij))*zgac/zgc+psea(jij)*zgac2/zgc2))
256 zlbc = max(min(iced%XLBC(1),iced%XLBC(2)),(psea(jij)*iced%XLBC(2)+(1.-psea(jij))*iced%XLBC(1)) )
257 zfsedc = max(min(icep%XFSEDC(1),icep%XFSEDC(2)), (psea(jij)*icep%XFSEDC(2)+(1.-psea(jij))*icep%XFSEDC(1)) )
258 zconc3d= (1.-ptown(jij))*(psea(jij)*iced%XCONC_SEA+(1.-psea(jij))*iced%XCONC_LAND) + &
259 ptown(jij) *iced%XCONC_URBAN
260 ELSE
261 zray = zraydefo
262 zlbc = iced%XLBC(1)
263 zfsedc = icep%XFSEDC(1)
264 zconc3d= iced%XCONC_LAND
265 ENDIF
266 !calculation of w
267 IF(prxt(jij) > iced%XRTMIN(jrr)) THEN
268 zzwlbda=6.6e-8*(101325./ppabst(jij,jk))*(ptht(jij,jk)/293.15)
269 zzwlbdc=(zlbc*zconc3d/(prhodref(jij,jk)*prxt(jij)))**iced%XLBEXC
270 zzcc=iced%XCC*(1.+1.26*zzwlbda*zzwlbdc/zray) !! ZCC : Fall speed
271 zwsedw1=prhodref(jij,jk)**(-iced%XCEXVT ) * zzwlbdc**(-iced%XDC)*zzcc*zfsedc
272 ELSE
273 zwsedw1=0.
274 ENDIF
275 IF ( zqp > iced%XRTMIN(jrr) ) THEN
276 zzwlbda=6.6e-8*(101325./ppabst(jij,jk))*(ptht(jij,jk)/293.15)
277 zzwlbdc=(zlbc*zconc3d/(prhodref(jij,jk)*zqp))**iced%XLBEXC
278 zzcc=iced%XCC*(1.+1.26*zzwlbda*zzwlbdc/zray) !! ZCC : Fall speed
279 zwsedw2=prhodref(jij,jk)**(-iced%XCEXVT ) * zzwlbdc**(-iced%XDC)*zzcc*zfsedc
280 ELSE
281 zwsedw2=0.
282 ENDIF
283 ELSE
284 zwsedw1=0.
285 zwsedw2=0.
286 ENDIF
287!- duplicated code -------------------------------------------------------------------------
288 IF (zwsedw2 /= 0.) THEN
289 zsed(jij,ik,jrr)=fwsed1(zwsedw1,ptstep,pdzz(jij,jk),prhodref(jij,jk),prxt(jij),zinvtstep) &
290 & + fwsed2(zwsedw2,ptstep,pdzz(jij,jk),zsed(jij,ikplus,jrr))
291 ELSE
292 zsed(jij,ik,jrr)=fwsed1(zwsedw1,ptstep,pdzz(jij,jk),prhodref(jij,jk),prxt(jij),zinvtstep)
293 ENDIF
294!-------------------------------------------------------------------------------------------
295 ENDDO
296
297 !!IF (LHOOK) CALL DR_HOOK('ICE4_SEDIMENTATION_STAT:CLOUD',1,ZHOOK_HANDLE)
298
299 END SUBROUTINE cloud

301 SUBROUTINE pristine_ice(PRXT)
302
303 REAL, INTENT(IN) :: PRXT(D%NIJT) ! mr of specy X
304 INTEGER :: JIJ
305 REAL :: ZQP
306 REAL :: ZWSEDW1, ZWSEDW2
307
308 !!REAL(KIND=JPHOOK) :: ZHOOK_HANDLE
309
310 !!IF (LHOOK) CALL DR_HOOK('ICE4_SEDIMENTATION_STAT:PRISTINE_ICE',0,ZHOOK_HANDLE)
311
312 ! ******* for pristine ice
313 DO jij = iijb, iije
314 zqp=zsed(jij,ikplus,jrr)*ztsorhodz(jij)
315 IF ((prxt(jij) > iced%XRTMIN(jrr)) .OR. (zqp > iced%XRTMIN(jrr))) THEN
316 !calculation of w
317 IF ( prxt(jij) > max(iced%XRTMIN(jrr),1.0e-7 ) ) THEN
318 zwsedw1= icep%XFSEDI * &
319 & prhodref(jij,jk)**(-iced%XCEXVT) * & ! McF&H
320 & max( 0.05e6,-0.15319e6-0.021454e6* &
321 & log(prhodref(jij,jk)*prxt(jij)) )**icep%XEXCSEDI
322 ELSE
323 zwsedw1=0.
324 ENDIF
325 IF ( zqp > max(iced%XRTMIN(jrr),1.0e-7 ) ) THEN
326 zwsedw2= icep%XFSEDI * &
327 & prhodref(jij,jk)**(-iced%XCEXVT) * & ! McF&H
328 & max( 0.05e6,-0.15319e6-0.021454e6* &
329 & log(prhodref(jij,jk)*zqp) )**icep%XEXCSEDI
330 ELSE
331 zwsedw2=0.
332 ENDIF
333 ELSE
334 zwsedw1=0.
335 zwsedw2=0.
336 ENDIF
337!- duplicated code -------------------------------------------------------------------------
338 IF (zwsedw2 /= 0.) THEN
339 zsed(jij,ik,jrr)=fwsed1(zwsedw1,ptstep,pdzz(jij,jk),prhodref(jij,jk),prxt(jij),zinvtstep) &
340 & + fwsed2(zwsedw2,ptstep,pdzz(jij,jk),zsed(jij,ikplus,jrr))
341 ELSE
342 zsed(jij,ik,jrr)=fwsed1(zwsedw1,ptstep,pdzz(jij,jk),prhodref(jij,jk),prxt(jij),zinvtstep)
343 ENDIF
344!-------------------------------------------------------------------------------------------
345 ENDDO
346
347 !!IF (LHOOK) CALL DR_HOOK('ICE4_SEDIMENTATION_STAT:PRISTINE_ICE',1,ZHOOK_HANDLE)
348
349 END SUBROUTINE pristine_ice

351 SUBROUTINE snow(PRXT)
352
353 REAL, INTENT(IN) :: PRXT(D%NIJT) ! mr of specy X
354 INTEGER :: JIJ
355 REAL :: ZQP, ZLBDAS
356 REAL :: ZWSEDW1, ZWSEDW2
357
358 !!REAL(KIND=JPHOOK) :: ZHOOK_HANDLE
359
360 !!IF (LHOOK) CALL DR_HOOK('ICE4_SEDIMENTATION_STAT:SNOW',0,ZHOOK_HANDLE)
361
362 ! ******* for snow
363 DO jij = iijb, iije
364 zqp=zsed(jij,ikplus,jrr)*ztsorhodz(jij)
365 IF ((prxt(jij) > iced%XRTMIN(jrr)) ) THEN
366 !Compute lambda_snow parameter
367 IF (parami%LSNOW_T) THEN
368 IF(pt(jij,jk)>cst%XTT-10.0) THEN
369 zlbdas = max(min(iced%XLBDAS_MAX, 10**(14.554-0.0423*pt(jij,jk))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
370 ELSE
371 zlbdas = max(min(iced%XLBDAS_MAX, 10**(6.226-0.0106*pt(jij,jk))),iced%XLBDAS_MIN)*iced%XTRANS_MP_GAMMAS
372 END IF
373 ELSE
374 zlbdas = max(min(iced%XLBDAS_MAX,iced%XLBS*(prhodref(jij,jk)*prxt(jij))**iced%XLBEXS),iced%XLBDAS_MIN)
375 END IF
376 !calculation of w
377 IF ( prxt(jij) > iced%XRTMIN(jrr) ) THEN
378 zwsedw1= icep%XFSEDS * &
379 & prhodref(jij,jk)**(-iced%XCEXVT) * &
380 & (1+(iced%XFVELOS/zlbdas)**iced%XALPHAS)**(-iced%XNUS+icep%XEXSEDS/iced%XALPHAS)* &
381 & zlbdas**(iced%XBS+icep%XEXSEDS)
382 ELSE
383 zwsedw1=0.
384 ENDIF
385 IF ( zqp > iced%XRTMIN(jrr) ) THEN
386 zwsedw2= icep%XFSEDS * &
387 & prhodref(jij,jk)**(-iced%XCEXVT) * &
388 & (1+(iced%XFVELOS/zlbdas)**iced%XALPHAS)**(-iced%XNUS+icep%XEXSEDS/iced%XALPHAS)* &
389 & zlbdas**(iced%XBS+icep%XEXSEDS)
390 ELSE
391 zwsedw2=0.
392 ENDIF
393 ELSE
394 zwsedw1=0.
395 zwsedw2=0.
396 ENDIF
397!- duplicated code -------------------------------------------------------------------------
398 IF (zwsedw2 /= 0.) THEN
399 zsed(jij,ik,jrr)=fwsed1(zwsedw1,ptstep,pdzz(jij,jk),prhodref(jij,jk),prxt(jij),zinvtstep) &
400 & + fwsed2(zwsedw2,ptstep,pdzz(jij,jk),zsed(jij,ikplus,jrr))
401 ELSE
402 zsed(jij,ik,jrr)=fwsed1(zwsedw1,ptstep,pdzz(jij,jk),prhodref(jij,jk),prxt(jij),zinvtstep)
403 ENDIF
404!-------------------------------------------------------------------------------------------
405 ENDDO
406
407 !!IF (LHOOK) CALL DR_HOOK('ICE4_SEDIMENTATION_STAT:SNOW',1,ZHOOK_HANDLE)
408
409 END SUBROUTINE snow

411 SUBROUTINE other_species(PFSED,PEXSED,PRXT)
412
413 REAL, INTENT(IN) :: PFSED
414 REAL, INTENT(IN) :: PEXSED
415 REAL, INTENT(IN) :: PRXT(D%NIJT) ! mr of specy X
416 INTEGER :: JIJ
417 REAL :: ZQP
418 REAL :: ZWSEDW1, ZWSEDW2
419
420 !!REAL(KIND=JPHOOK) :: ZHOOK_HANDLE
421
422 !!IF (LHOOK) CALL DR_HOOK('ICE4_SEDIMENTATION_STAT:OTHER_SPECIES',0,ZHOOK_HANDLE)
423
424 ! for all but cloud and pristine ice :
425 DO jij = iijb, iije
426 zqp=zsed(jij,ikplus,jrr)*ztsorhodz(jij)
427 IF ((prxt(jij) > iced%XRTMIN(jrr)) .OR. (zqp > iced%XRTMIN(jrr))) THEN
428 !calculation of w
429 IF ( prxt(jij) > iced%XRTMIN(jrr) ) THEN
430 zwsedw1= pfsed *prxt(jij)**(pexsed-1)*prhodref(jij,jk)**(pexsed-iced%XCEXVT-1)
431 ELSE
432 zwsedw1=0.
433 ENDIF
434 IF ( zqp > iced%XRTMIN(jrr) ) THEN
435 zwsedw2= pfsed *zqp**(pexsed-1)*prhodref(jij,jk)**(pexsed-iced%XCEXVT-1)
436 ELSE
437 zwsedw2=0.
438 ENDIF
439 ELSE
440 zwsedw1=0.
441 zwsedw2=0.
442 ENDIF
443!- duplicated code -------------------------------------------------------------------------
444 IF (zwsedw2 /= 0.) THEN
445 zsed(jij,ik,jrr)=fwsed1(zwsedw1,ptstep,pdzz(jij,jk),prhodref(jij,jk),prxt(jij),zinvtstep) &
446 & + fwsed2(zwsedw2,ptstep,pdzz(jij,jk),zsed(jij,ikplus,jrr))
447 ELSE
448 zsed(jij,ik,jrr)=fwsed1(zwsedw1,ptstep,pdzz(jij,jk),prhodref(jij,jk),prxt(jij),zinvtstep)
449 ENDIF
450!-------------------------------------------------------------------------------------------
451 ENDDO
452
453 !!IF (LHOOK) CALL DR_HOOK('ICE4_SEDIMENTATION_STAT:OTHER_SPECIES',1,ZHOOK_HANDLE)
454
455 END SUBROUTINE other_species

457 SUBROUTINE shift
458
459 ikplus=ishift
460 ik=1-ishift
461 ishift=1-ishift
462
463 END SUBROUTINE shift
464!
465!
466ELEMENTAL FUNCTION fwsed1(PWSEDW,PTSTEP1,PDZZ1,PRHODREF1,PRXT1,PINVTSTEP) RESULT(PVAR)
467 REAL, INTENT(IN) :: PWSEDW,PTSTEP1,PDZZ1,PRHODREF1,PRXT1,PINVTSTEP
468 REAL :: PVAR
469! 5 multiplications only => cost = 5X
470 pvar = min(prhodref1*pdzz1*prxt1*pinvtstep,pwsedw*prhodref1*prxt1)
471END FUNCTION fwsed1
472!
473ELEMENTAL FUNCTION fwsed2(PWSEDW,PTSTEP1,PDZZ1,PWSEDWSUP) RESULT(PVAR)
474 REAL, INTENT(IN) :: PWSEDW,PTSTEP1,PDZZ1,PWSEDWSUP
475 REAL :: PVAR
