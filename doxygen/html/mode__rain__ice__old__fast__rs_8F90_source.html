103 IF (buconf%LBUDGET_TH) CALL budget_store_init_phy(d, tbudgets(nbudget_th), 'RIM', unpack(zths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
104 IF (buconf%LBUDGET_RC) CALL budget_store_init_phy(d, tbudgets(nbudget_rc), 'RIM', unpack(zrcs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
105 IF (buconf%LBUDGET_RS) CALL budget_store_init_phy(d, tbudgets(nbudget_rs), 'RIM', unpack(zrss(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
106 IF (buconf%LBUDGET_RG) CALL budget_store_init_phy(d, tbudgets(nbudget_rg), 'RIM', unpack(zrgs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))

108 zzw1(:,:) = 0.0
109!
110 igrim=0
111 DO jk=1, ksize
112 IF((zrct(jk)>iced%XRTMIN(2)) .AND. (zrst(jk)>iced%XRTMIN(5)) .AND. &
113 (zrcs(jk)>0.0) .AND. (zzt(jk)<cst%XTT)) THEN
114 igrim=igrim+1
115 gmask(jk)=.true.
116 ! 5.1.1 select the ZLBDAS
117 i1(igrim)=jk
118 zvec1(igrim)=zlbdas(jk)
119 ELSE
120 gmask(jk)=.false.
121 ENDIF
122 ENDDO
123
124 IF( igrim>0 ) THEN
125!
126! 5.1.2 find the next lower indice for the ZLBDAS in the geometrical
127! set of Lbda_s used to tabulate some moments of the incomplete
128! gamma function
129!
130 zvec2(1:igrim) = max(1.00001, min(float(icep%NGAMINC) - 0.00001, &
131 icep%XRIMINTP1 * log(zvec1(1:igrim)) + icep%XRIMINTP2))
132 ivec2(1:igrim) = int(zvec2(1:igrim))
133 zvec2(1:igrim) = zvec2(1:igrim) - float(ivec2(1:igrim))
134!
135! 5.1.3 perform the linear interpolation of the normalized
136! "2+XDS"-moment of the incomplete gamma function
137!
138 zvec1(1:igrim) = icep%XGAMINC_RIM1(ivec2(1:igrim)+1)* zvec2(1:igrim) &
139 - icep%XGAMINC_RIM1(ivec2(1:igrim) )*(zvec2(1:igrim) - 1.0)
140 zzw(:) = 0.
141 DO jk=1, igrim
142 zzw(i1(jk))=zvec1(jk)
143 ENDDO
144!
145! 5.1.4 riming of the small sized aggregates
146!
147 DO jk = 1, ksize
148 IF (gmask(jk)) THEN
149 zzw1(jk,1) = min( zrcs(jk), &
150 icep%XCRIMSS * zzw(jk) * zrct(jk)*zcolf(jk) & ! RCRIMSS
151 * zlbdas(jk)**icep%XEXCRIMSS &
152 * zrhodref(jk)**(-iced%XCEXVT) )
153 zrcs(jk) = zrcs(jk) - zzw1(jk,1)
154 zrss(jk) = zrss(jk) + zzw1(jk,1)
155 zths(jk) = zths(jk) + zzw1(jk,1)*(zlsfact(jk)-zlvfact(jk)) ! f(L_f*(RCRIMSS))
156 END IF
157 END DO
158!
159! 5.1.5 perform the linear interpolation of the normalized
160! "XBS"-moment of the incomplete gamma function
161!
162 zvec1(1:igrim) = icep%XGAMINC_RIM2( ivec2(1:igrim)+1 )* zvec2(1:igrim) &
163 - icep%XGAMINC_RIM2( ivec2(1:igrim) )*(zvec2(1:igrim) - 1.0)
164 zzw(:) = 0.
165 DO jk=1, igrim
166 zzw(i1(jk))=zvec1(jk)
167 ENDDO
168!
169! 5.1.6 riming-conversion of the large sized aggregates into graupeln
170!
171!
172 DO jk = 1, ksize
173 IF (gmask(jk) .AND. (zrss(jk) > 0.0)) THEN
174 zzw1(jk,2) = min(zrcs(jk), &
175 icep%XCRIMSG * zrct(jk)*zcolf(jk) & ! RCRIMSG
176 * zlbdas(jk)**icep%XEXCRIMSG &
177 * zrhodref(jk)**(-iced%XCEXVT) &
178 - zzw1(jk,1))
179
180 zzw1(jk,3) = min(zrss(jk), &
181 icep%XSRIMCG * zlbdas(jk)**icep%XEXSRIMCG & ! RSRIMCG
182 * (1.0 - zzw(jk))/(ptstep*zrhodref(jk)))
183
184 zrcs(jk) = zrcs(jk) - zzw1(jk,2)
185 zrss(jk) = zrss(jk) - zzw1(jk,3)
186 zrgs(jk) = zrgs(jk) + zzw1(jk,2)+zzw1(jk,3)
187 zths(jk) = zths(jk) + zzw1(jk,2)*(zlsfact(jk)-zlvfact(jk)) ! f(L_f*(RCRIMSG))
188 END IF
189 END DO
190 END IF
191
192 IF (buconf%LBUDGET_TH) CALL budget_store_end_phy(d, tbudgets(nbudget_th), 'RIM', unpack(zths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
193 IF (buconf%LBUDGET_RC) CALL budget_store_end_phy(d, tbudgets(nbudget_rc), 'RIM', unpack(zrcs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
194 IF (buconf%LBUDGET_RS) CALL budget_store_end_phy(d, tbudgets(nbudget_rs), 'RIM', unpack(zrss(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
195 IF (buconf%LBUDGET_RG) CALL budget_store_end_phy(d, tbudgets(nbudget_rg), 'RIM', unpack(zrgs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
196
197!* 5.2 rain accretion onto the aggregates
198
199 IF (buconf%LBUDGET_TH) CALL budget_store_init_phy(d, tbudgets(nbudget_th), 'ACC', unpack(zths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
200 IF (buconf%LBUDGET_RR) CALL budget_store_init_phy(d, tbudgets(nbudget_rr), 'ACC', unpack(zrrs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
201 IF (buconf%LBUDGET_RS) CALL budget_store_init_phy(d, tbudgets(nbudget_rs), 'ACC', unpack(zrss(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
202 IF (buconf%LBUDGET_RG) CALL budget_store_init_phy(d, tbudgets(nbudget_rg), 'ACC', unpack(zrgs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
203
204 zzw1(:,2:3) = 0.0
205 igacc=0
206 DO jk=1, ksize
207 IF((zrrt(jk) > iced%XRTMIN(3)) .AND. &
208 (zrst(jk) > iced%XRTMIN(5)) .AND. &
209 (zrrs(jk) > 0.0) .AND. &
210 (zzt(jk) < cst%XTT)) THEN
211 igacc=igacc+1
212 gmask(jk)=.true.
213 ! 5.1.1 select the (ZLBDAS,ZLBDAR) couplet
214 i1(igacc)=jk
215 zvec1(igacc)=zlbdas(jk)
216 zvec2(igacc)=zlbdar(jk)
217 ELSE
218 gmask(jk)=.false.
219 ENDIF
220 ENDDO
221
222 IF( igacc>0 ) THEN
223!
224! 5.2.2 find the next lower indice for the ZLBDAS and for the ZLBDAR
225! in the geometrical set of (Lbda_s,Lbda_r) couplet use to
226! tabulate the RACCSS-kernel
227!
228 zvec1(1:igacc) = max( 1.00001, min( float(icep%NACCLBDAS)-0.00001, &
229 icep%XACCINTP1S * log( zvec1(1:igacc) ) + icep%XACCINTP2S ) )
230 ivec1(1:igacc) = int( zvec1(1:igacc) )
231 zvec1(1:igacc) = zvec1(1:igacc) - float( ivec1(1:igacc) )
232
233 zvec2(1:igacc) = max( 1.00001, min( float(icep%NACCLBDAR)-0.00001, &
234 icep%XACCINTP1R * log( zvec2(1:igacc) ) + icep%XACCINTP2R ) )
235 ivec2(1:igacc) = int( zvec2(1:igacc) )
236 zvec2(1:igacc) = zvec2(1:igacc) - float( ivec2(1:igacc) )
237
238! 5.2.3 perform the bilinear interpolation of the normalized
239! RACCSS-kernel
240
241 DO jl = 1,igacc
242 zvec3(jl) = ( icep%XKER_RACCSS(ivec1(jl)+1,ivec2(jl)+1)* zvec2(jl) &
243 - icep%XKER_RACCSS(ivec1(jl)+1,ivec2(jl) )*(zvec2(jl) - 1.0)) &
244 * zvec1(jl) &
245 - ( icep%XKER_RACCSS(ivec1(jl) ,ivec2(jl)+1)* zvec2(jl) &
246 - icep%XKER_RACCSS(ivec1(jl) ,ivec2(jl) )*(zvec2(jl) - 1.0)) &
247 *(zvec1(jl) - 1.0)
248 END DO
249 zzw(:) = 0.
250 DO jk=1, igacc
251 zzw(i1(jk))=zvec3(jk)
252 ENDDO
253!
254! 5.2.4 raindrop accretion on the small sized aggregates
255!
256 DO jk = 1, ksize
257 IF (gmask(jk)) THEN
258 zzw1(jk,2) = & !! coef of RRACCS
259 icep%XFRACCSS*( zlbdas(jk)**iced%XCXS )*( zrhodref(jk)**(-iced%XCEXVT-1.) ) &
260 *(icep%XLBRACCS1/((zlbdas(jk)**2) ) + &
261 icep%XLBRACCS2/( zlbdas(jk) * zlbdar(jk) ) + &
262 icep%XLBRACCS3/( (zlbdar(jk)**2)) )/zlbdar(jk)**4
263 zzw1(jk,4) = min( zrrs(jk),zzw1(jk,2)*zzw(jk) ) ! RRACCSS
264 zrrs(jk) = zrrs(jk) - zzw1(jk,4)*icep%XFRMIN(7)
265 zrss(jk) = zrss(jk) + zzw1(jk,4)*icep%XFRMIN(7)
266 zths(jk) = zths(jk) + zzw1(jk,4)*(zlsfact(jk)-zlvfact(jk))*icep%XFRMIN(7) ! f(L_f*(RRACCSS))
267 END IF
268 END DO
269!
270! 5.2.4b perform the bilinear interpolation of the normalized
271! RACCS-kernel
272!
273 DO jl = 1,igacc
274 zvec3(jl) = (icep%XKER_RACCS(ivec2(jl)+1,ivec1(jl)+1)* zvec1(jl) &
275 - icep%XKER_RACCS(ivec2(jl)+1,ivec1(jl) )*(zvec1(jl) - 1.0) ) &
276 * zvec2(jl) &
277 - (icep%XKER_RACCS(ivec2(jl) ,ivec1(jl)+1)* zvec1(jl) &
278 - icep%XKER_RACCS(ivec2(jl) ,ivec1(jl) )*(zvec1(jl) - 1.0) ) &
279 *(zvec2(jl) - 1.0)
280 END DO
281 zzw(:) = 0.
282 DO jk=1, igacc
283 zzw(i1(jk)) = zvec3(jk)
284 ENDDO
285 zzw1(:,2) = zzw1(:,2)*zzw(:)
286
287 !! RRACCS!
288! 5.2.5 perform the bilinear interpolation of the normalized
289! SACCRG-kernel
290!
291 DO jl = 1,igacc
292 zvec3(jl) = ( icep%XKER_SACCRG(ivec2(jl)+1,ivec1(jl)+1)* zvec1(jl) &
293 - icep%XKER_SACCRG(ivec2(jl)+1,ivec1(jl) )*(zvec1(jl) - 1.0) ) &
294 * zvec2(jl) &
295 - ( icep%XKER_SACCRG(ivec2(jl) ,ivec1(jl)+1)* zvec1(jl) &
296 - icep%XKER_SACCRG(ivec2(jl) ,ivec1(jl) )*(zvec1(jl) - 1.0) ) &
297 * (zvec2(jl) - 1.0)
298 END DO
299 zzw(:) = 0.
300 DO jk=1, igacc
301 zzw(i1(jk)) = zvec3(jk)
302 ENDDO
303!
304! 5.2.6 raindrop accretion-conversion of the large sized aggregates
305! into graupeln
306!
307 DO jk = 1, ksize
308 IF (gmask(jk) .AND. (zrss(jk) > 0.0)) THEN
309 zzw1(jk,2) = max( min( zrrs(jk),zzw1(jk,2)-zzw1(jk,4) ),0.0 ) ! RRACCSG
310 END IF
311 END DO
312
313 DO jk = 1, ksize
314 IF (gmask(jk) .AND. (zrss(jk)>0.0) .AND. zzw1(jk,2) > 0.0 .AND. zrss(jk) > icep%XFRMIN(1)/ptstep) THEN
315 zzw1(jk,3) = min( zrss(jk),icep%XFSACCRG*zzw(jk)* & ! RSACCRG
316 ( zlbdas(jk)**(iced%XCXS-iced%XBS) )*( zrhodref(jk)**(-iced%XCEXVT-1.) ) &
317 *( icep%XLBSACCR1/((zlbdar(jk)**2) ) + &
318 icep%XLBSACCR2/( zlbdar(jk) * zlbdas(jk) ) + &
319 icep%XLBSACCR3/( (zlbdas(jk)**2)) )/zlbdar(jk) )
320 zrrs(jk) = zrrs(jk) - zzw1(jk,2)
321 zrss(jk) = zrss(jk) - zzw1(jk,3)
322 zrgs(jk) = zrgs(jk) + zzw1(jk,2)+zzw1(jk,3)
323 zths(jk) = zths(jk) + zzw1(jk,2)*(zlsfact(jk)-zlvfact(jk)) !
324 ! f(L_f*(RRACCSG))
325 END IF
326 END DO
327 END IF
328
329 IF (buconf%LBUDGET_TH) CALL budget_store_end_phy(d, tbudgets(nbudget_th), 'ACC', unpack(zths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
330 IF (buconf%LBUDGET_RR) CALL budget_store_end_phy(d, tbudgets(nbudget_rr), 'ACC', unpack(zrrs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
331 IF (buconf%LBUDGET_RS) CALL budget_store_end_phy(d, tbudgets(nbudget_rs), 'ACC', unpack(zrss(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
332 IF (buconf%LBUDGET_RG) CALL budget_store_end_phy(d, tbudgets(nbudget_rg), 'ACC', unpack(zrgs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
333
334!* 5.3 Conversion-Melting of the aggregates
335
336 IF (buconf%LBUDGET_RS) CALL budget_store_init_phy(d, tbudgets(nbudget_rs), 'CMEL', unpack(zrss(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
337 IF (buconf%LBUDGET_RG) CALL budget_store_init_phy(d, tbudgets(nbudget_rg), 'CMEL', unpack(zrgs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
338
339 zzw(:) = 0.0
340 DO jk = 1, ksize
341 IF ((zrst(jk)>iced%XRTMIN(5)) .AND. (zrss(jk)>0.0) .AND. (zzt(jk)>cst%XTT)) THEN
342 zzw(jk) = zrvt(jk)*zpres(jk)/(cst%XEPSILO+zrvt(jk)) ! Vapor pressure
343 zzw(jk) = zka(jk)*(cst%XTT-zzt(jk)) + &
344 ( zdv(jk)*(cst%XLVTT + ( cst%XCPV - cst%XCL ) * ( zzt(jk) - cst%XTT )) &
345 *(cst%XESTT-zzw(jk))/(cst%XRV*zzt(jk)) )
346!
347! compute RSMLT
348!
349 zzw(jk) = min( zrss(jk), icep%XFSCVMG*max( 0.0,( -zzw(jk) * &
350 ( icep%X0DEPS* zlbdas(jk)**icep%XEX0DEPS + &
351 icep%X1DEPS*zcj(jk)*zlbdas(jk)**icep%XEX1DEPS ) - &
352 ( zzw1(jk,1)+zzw1(jk,4) ) * &
353 (zrhodref(jk)*cst%XCL*(cst%XTT-zzt(jk)))) / &
354 ( zrhodref(jk)*cst%XLMTT ) ) )
355!
356! note that RSCVMG = RSMLT*ICEP%XFSCVMG but no heat is exchanged (at the rate RSMLT)
357! because the graupeln produced by this process are still icy!!!
358!
359 zrss(jk) = zrss(jk) - zzw(jk)
360 zrgs(jk) = zrgs(jk) + zzw(jk)
361 END IF
362 END DO
363
364 IF (buconf%LBUDGET_RS) CALL budget_store_end_phy(d, tbudgets(nbudget_rs), 'CMEL', unpack(zrss(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
365 IF (buconf%LBUDGET_RG) CALL budget_store_end_phy(d, tbudgets(nbudget_rg), 'CMEL', unpack(zrgs(:)*zrhodj(:),mask=gmicro(:,:),field=0.0))
366
367 IF (lhook) CALL dr_hook('RAIN_ICE_OLD:RAIN_ICE_FAST_RS',1,zhook_handle)
368
369 END SUBROUTINE rain_ice_old_fast_rs
