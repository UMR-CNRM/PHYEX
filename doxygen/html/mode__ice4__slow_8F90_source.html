53 REAL, DIMENSION(KPROMA), INTENT(IN) :: PSSI ! Supersaturation over ice
54 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLVFACT

57 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRCT ! Cloud water m.r. at t
58 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRIT ! Pristine ice m.r. at t
59 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRST ! Snow/aggregate m.r. at t

64 REAL, DIMENSION(KPROMA), INTENT(IN) :: PCJ ! Function to compute the ventilation coefficient
65 REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLI_HCF !
66 REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLI_HRI !
67 REAL, DIMENSION(MERGE(KPROMA,0,OELEC)), INTENT(IN) :: PLATHAM_IAGGS ! enhancement factor of IAGGS due to Efield
68 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRCHONI ! Homogeneous nucleation
69 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRVDEPS ! Deposition on r_s
70 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRIAGGS ! Aggregation on r_s
71 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRIAUTS ! Autoconversion of r_i for r_s production
72 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRVDEPG ! Deposition on r_g

80 !

83 !-------------------------------------------------------------------------------
84 !
85 !
86 !* 3.2 compute the homogeneous nucleation source: RCHONI
87 !
88 DO jl=1, ksize
89  IF(pt(jl)<cst%XTT-35.0 .AND. prct(jl)>iced%XRTMIN(2) .AND. ldcompute(jl)) THEN
90  IF(.NOT. ldsoft) THEN
91  prchoni(jl) = min(1000.,icep%XHON*prhodref(jl)*prct(jl) &
92  *exp( icep%XALPHA3*(pt(jl)-cst%XTT)-icep%XBETA3 ))
93  ENDIF
94  ELSE
95  prchoni(jl) = 0.
96  ENDIF
97 ENDDO
98 !
99 !* 3.4 compute the deposition, aggregation and autoconversion sources
100 !
101 !
102 !* 3.4.2 compute the riming-conversion of r_c for r_i production: RCAUTI
103 !
104 ! ZZW(:) = 0.0
105 ! ZTIMAUTIC = SQRT( ICEP%XTIMAUTI*ICEP%XTIMAUTC )
106 ! WHERE ( (PRCT(:)>0.0) .AND. (PRIT(:)>0.0) .AND. (PRCS(:)>0.0) )
107 ! ZZW(:) = MIN( PRCS(:),ZTIMAUTIC * MAX( SQRT( PRIT(:)*PRCT(:) ),0.0 ) )
108 ! PRIS(:) = PRIS(:) + ZZW(:)
109 ! PRCS(:) = PRCS(:) - ZZW(:)
110 ! PTHS(:) = PTHS(:) + ZZW(:)*(PLSFACT(:)-PLVFACT(:)) ! f(L_f*(RCAUTI))
111 ! END WHERE
112 !
113 !* 3.4.3 compute the deposition on r_s: RVDEPS
114 !
115 DO jl=1, ksize
116  IF(prvt(jl)>iced%XRTMIN(1) .AND. prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
117  IF(.NOT. ldsoft) THEN
118 #ifdef REPRO48
119  prvdeps(jl) = ( pssi(jl)/(prhodref(jl)*pai(jl)) ) * &
120  ( icep%X0DEPS*plbdas(jl)**icep%XEX0DEPS + icep%X1DEPS*pcj(jl)*plbdas(jl)**icep%XEX1DEPS )
121 #else
122  prvdeps(jl) = ( prst(jl)*(pssi(jl)/pai(jl)) ) * &
123  ( icep%X0DEPS*plbdas(jl)**(iced%XBS+icep%XEX0DEPS) + icep%X1DEPS*pcj(jl) * &
124  (1+0.5*(iced%XFVELOS/plbdas(jl))**iced%XALPHAS)**(-iced%XNUS+icep%XEX1DEPS/iced%XALPHAS) &
125  *(plbdas(jl))**(iced%XBS+icep%XEX1DEPS) )
126 #endif
127  ENDIF
128  ELSE
129  prvdeps(jl) = 0.
130  ENDIF
131 ENDDO
132 !
133 !* 3.4.4 compute the aggregation on r_s: RIAGGS
134 !
135 DO jl=1, ksize
136  IF(prit(jl)>iced%XRTMIN(4) .AND. prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
137  IF(.NOT. ldsoft) THEN
138 #ifdef REPRO48
139  priaggs(jl) = icep%XFIAGGS * exp( icep%XCOLEXIS*(pt(jl)-cst%XTT) ) &
140  * prit(jl) &
141  * plbdas(jl)**icep%XEXIAGGS &
142  * prhodref(jl)**(-iced%XCEXVT)
143 #else
144  priaggs(jl) = icep%XFIAGGS * exp( icep%XCOLEXIS*(pt(jl)-cst%XTT) ) &
145  * prit(jl) &
146  * prst(jl) * (1+(iced%XFVELOS/plbdas(jl))**iced%XALPHAS)**&
147  (-iced%XNUS+icep%XEXIAGGS/iced%XALPHAS) &
148  * prhodref(jl)**(-iced%XCEXVT+1.) &
149  * ((plbdas(jl))**(iced%XBS+icep%XEXIAGGS))
150 #endif
151  IF (oelec) priaggs(jl) = priaggs(jl) * platham_iaggs(jl)
152  ENDIF
153  ELSE
154  priaggs(jl) = 0.
155  ENDIF
156 ENDDO
157 !
158 !* 3.4.5 compute the autoconversion of r_i for r_s production: RIAUTS
159 !
160 DO jl=1, ksize
161  IF(phli_hri(jl)>iced%XRTMIN(4) .AND. ldcompute(jl)) THEN
162  IF(.NOT. ldsoft) THEN
163  !ZCRIAUTI(:)=MIN(ICEP%XCRIAUTI,10**(0.06*(PT(:)-CST%XTT)-3.5))
164  zcriauti(jl)=min(icep%XCRIAUTI,10**(icep%XACRIAUTI*(pt(jl)-cst%XTT)+icep%XBCRIAUTI))
165  priauts(jl) = icep%XTIMAUTI * exp( icep%XTEXAUTI*(pt(jl)-cst%XTT) ) &
166  * max(phli_hri(jl)-zcriauti(jl)*phli_hcf(jl), 0.)
167  ENDIF
168  ELSE
169  priauts(jl) = 0.
170  ENDIF
171 ENDDO
172 !
173 !* 3.4.6 compute the deposition on r_g: RVDEPG
174 !
175 !
176 DO jl=1, ksize
177  IF(prvt(jl)>iced%XRTMIN(1) .AND. prgt(jl)>iced%XRTMIN(6) .AND. ldcompute(jl)) THEN
178  IF(.NOT. ldsoft) THEN
179  prvdepg(jl) = ( pssi(jl)/(prhodref(jl)*pai(jl)) ) * &
180  ( icep%X0DEPG*plbdag(jl)**icep%XEX0DEPG + icep%X1DEPG*pcj(jl)*plbdag(jl)**icep%XEX1DEPG )
181  ENDIF
182  ELSE
183  prvdepg(jl) = 0.
184  ENDIF
185 ENDDO
186 !
187 IF (lhook) CALL dr_hook('ICE4_SLOW', 1, zhook_handle)
188 !
189 END SUBROUTINE ice4_slow
