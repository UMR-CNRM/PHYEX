17!!**** Adjust up- and downdraft mass fluxes to be consistent with the
18!! mass transport at the LFS given by the precipitation efficiency
19!! relation.

21!!
22!! PURPOSE
23!! -------
24!! The purpose of this routine is to adjust up- and downdraft mass
25!! fluxes below the LFS to be consistent with the precipitation
26!! efficiency relation
27!!
28!!
29!!

43!! JCVEXB, JCVEXT ! extra levels on the vertical boundaries

47!! ! supply rate for downdraft

69USE modd_convpar, ONLY: xusrdpth

73!* 0.1 Declarations of dummy arguments :
74!
75!
76INTEGER, INTENT(IN) :: KLON ! horizontal dimension
77INTEGER, INTENT(IN) :: KLEV ! vertical dimension
78REAL, DIMENSION(KLON,KLEV), INTENT(IN) :: PPRES ! pressure (Pa)
79REAL, DIMENSION(KLON,KLEV), INTENT(IN) :: PURW ! updraft total water (kg/kg)
80REAL, DIMENSION(KLON), INTENT(IN) :: PUTPR ! updraft total precipit. (kg/s
81REAL, DIMENSION(KLON), INTENT(IN) :: PPREF ! precipitation efficiency
82REAL, DIMENSION(KLON), INTENT(IN) :: PMIXF ! critical mixed fraction at LCL
83INTEGER, DIMENSION(KLON), INTENT(IN) :: KLCL ! contains vert. index of LCL
84INTEGER, DIMENSION(KLON), INTENT(IN) :: KCTL ! contains vert. index of CTL
85INTEGER, DIMENSION(KLON), INTENT(IN) :: KETL ! contains vert. index of equilibrium
86 ! (zero buoyancy) level
87INTEGER, DIMENSION(KLON), INTENT(INOUT) :: KLFS ! contains vert. index of LFS
88INTEGER, DIMENSION(KLON), INTENT(INOUT) :: KDBL ! contains vert. index of DBL
89!
90REAL, DIMENSION(KLON), INTENT(INOUT) :: PDTEVR ! total downdraft evaporation
91 ! rate at LFS
92REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PDTEVRF! downdraft evaporation rate
93REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PUMF ! updraft mass flux (kg/s)
94REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PUER ! updraft entrainment (kg/s)
95REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PUDR ! updraft detrainment (kg/s)
96REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PUPR ! updraft precipit. (kg/s)
97REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PDMF ! downdraft mass flux (kg/s)
98REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PDER ! downdraft entrainment (kg/s)
99REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PDDR ! downdraft detrainment (kg/s)
100REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PDTHL ! downdraft enthalpy (J/kg)
101REAL, DIMENSION(KLON,KLEV), INTENT(INOUT) :: PDRW ! downdraft total water (kg/kg)
102!
103REAL, DIMENSION(KLON), INTENT(OUT) :: PTPR ! total precipitation (kg/s)
104 ! = downdraft precipitation
105!
106!* 0.2 Declarations of local variables :
107!
108INTEGER :: IIE, IKB, IKE ! horizontal + vertical loop bounds
109INTEGER :: JK, JKT1, JKT2, JKT3 ! vertical loop index
110INTEGER :: JI ! horizontal loop index
111!
112INTEGER, DIMENSION(KLON) :: IPRL
113REAL, DIMENSION(KLON) :: ZWORK1, ZWORK2, ZWORK3, &
114 ZWORK4, ZWORK5, ZWORK6 ! work arrays
115REAL(KIND=jphook) :: zhook_handle
116!
117!
118!-------------------------------------------------------------------------------
119!
120! 0.3 Set loop bounds
121! ---------------
122!
123IF (lhook) CALL dr_hook('CONVECT_PRECIP_ADJUST',0,zhook_handle)
124ikb = 1 + jcvexb
125ike = klev - jcvext
126iie = klon
127jkt1 = maxval( klfs(:) )
128jkt2 = maxval( kctl(:) )
129jkt3 = minval( klcl(:) )
130!
131!
132! 1. Set some output variables for columns where no downdraft
133! exists. Exit if there is no downdraft at all.
134! --------------------------------------------------------
135!
136iprl(:) = ikb
137ptpr(:) = 0.
138!
139WHERE ( pdtevr(:) == 0. )
140 ptpr(:) = putpr(:) ! no downdraft evaporation => no downdraft, all
141 ! precipitation occurs in updraft
142END WHERE
143IF ( count( pdtevr(:) > 0. ) == 0 ) THEN ! exit routine if no downdraft exists
144 IF (lhook) CALL dr_hook('CONVECT_PRECIP_ADJUST',1,zhook_handle)
145 RETURN
146ENDIF
147!
148!* 2. The total mass transported from the updraft to the down-
149! draft at the LFS must be consistent with the three water
150! budget terms :
151! ---------------------------------------------------------
152!
153!* 2.1 Downdraft evaporation rate at the DBL. The evaporation
154! rate in downdraft must be consistent with precipitation
155! efficiency relation.
156! --------------------------------------------------------
157!
158!
159DO ji = 1, iie
160 jk = klfs(ji)
161 zwork1(ji) = pdtevr(ji) / min( -1.e-1, pdmf(ji,jk) )
162 zwork6(ji) = pdmf(ji,jk)
163END DO
164!
165!* 2.2 Some preliminar computations for downdraft = total
166! precipitation rate. The precipitation is evaluated in
167! a layer thickness DP=XUSRDPTH=165 hPa above the LCL.
168! The difference between updraft precipitation and downdraft
169! precipitation (updraft supply rate) is used to drive the
170! downdraft through evaporational cooling.
171! --------------------------------------------------------
172!
173DO ji = 1, iie
174 jk = klcl(ji)
175 zwork5(ji) = ppres(ji,jk)
176END DO
177!
178ptpr(:) = 0.
179DO jk = jkt3, jkt2
180 WHERE ( jk >= klcl(:) .AND. ppres(:,jk) >= zwork5(:) - xusrdpth )
181 ptpr(:) = ptpr(:) + pupr(:,jk)
182 iprl(:) = jk
183 END WHERE
184END DO
185iprl(:) = min( ketl(:), iprl(:) )
186!
187DO ji = 1, iie
188 jk = iprl(ji)
189 ptpr(ji) = pumf(ji,jk+1) * purw(ji,jk+1) + ptpr(ji)
190END DO
191!
192ptpr(:) = ppref(:) * min( putpr(:), ptpr(:) )
193zwork4(:) = putpr(:) - ptpr(:)
194!
195!
196!* 2.3 Total amount of precipitation that falls out of the up-
197! draft between the LCL and the LFS.
198! Condensate transfer from up to downdraft at LFS
199! ---------------------------------------------------------
200!
201zwork5(:) = 0.
202DO jk = jkt3, jkt1
203 WHERE ( jk >= klcl(:) .AND. jk <= klfs(:) )
204 zwork5(:) = zwork5(:) + pupr(:,jk)
205 END WHERE
206END DO
207!
208DO ji = 1, iie
209 jk = klfs(ji)
210 zwork2(ji) = ( 1. - ppref(ji) ) * zwork5(ji) * &
211 ( 1. - pmixf(ji) ) / max( 1.e-1, pumf(ji,jk) )
212END DO
213!
214!
215!* 2.4 Increase the first guess downdraft mass flux to satisfy
216! precipitation efficiency relation.
217! If downdraft does not evaporate any water at the DBL for
218! the specified relative humidity, or if the corrected mass
219! flux at the LFS is positive no downdraft is allowed
220! ---------------------------------------------------------
221!
222!
223zwork1(:) = -zwork4(:) / ( -zwork1(:) + zwork2(:) + 1.e-8 )
224zwork2(:) = zwork1(:) / min( -1.e-1, zwork6(:) ) ! ratio of budget consistent to actual DMF
225!
226zwork3(:) = 1.
227zwork6(:) = 1.
228WHERE ( zwork1(:) > 0. .OR. pdtevr(:) < 1. )
229 kdbl(:) = ikb
230 klfs(:) = ikb
231 pdtevr(:) = 0.
232 zwork2(:) = 0.
233 zwork3(:) = 0.
234 zwork6(:) = 0.
235END WHERE
236!
237DO jk = ikb, jkt1
238 pdmf(:,jk) = pdmf(:,jk) * zwork2(:)
239 pder(:,jk) = pder(:,jk) * zwork2(:)
240 pddr(:,jk) = pddr(:,jk) * zwork2(:)
241 pdtevrf(:,jk) = pdtevrf(:,jk)* zwork2(:)
242 pdrw(:,jk) = pdrw(:,jk) * zwork3(:)
243 pdthl(:,jk) = pdthl(:,jk) * zwork3(:)
244END DO
245zwork4(:) = zwork2(:)
246!
247!
248!* 3. Increase updraft mass flux, mass detrainment rate, and water
249! substance detrainment rates to be consistent with the transfer
250! of the estimated mass from the up- to the downdraft at the LFS
251! --------------------------------------------------------------
252!
253DO ji = 1, iie
254 jk = klfs(ji)
255 zwork2(ji) = ( 1. - zwork6(ji) ) + zwork6(ji) * &
256 ( pumf(ji,jk) - ( 1. - pmixf(ji) ) * zwork1(ji) ) / &
257 max( 1.e-1, pumf(ji,jk) )
258END DO
259!
260!
261jkt1 = maxval( klfs(:) ) ! value of KLFS might have been reset to IKB above
262DO jk = ikb, jkt1
263 DO ji = 1, iie
264 IF ( jk <= klfs(ji) ) THEN
265 pumf(ji,jk) = pumf(ji,jk) * zwork2(ji)
266 puer(ji,jk) = puer(ji,jk) * zwork2(ji)
267 pudr(ji,jk) = pudr(ji,jk) * zwork2(ji)
268 pupr(ji,jk) = pupr(ji,jk) * zwork2(ji)
269 END IF
270 END DO
271END DO
272!
273!
274!* 4. Increase total = downdraft precipitation and evaporation rate
275! -------------------------------------------------------------
276!
277WHERE ( pdtevr(:) > 0. )
278 ptpr(:) = ptpr(:) + ppref(:) * zwork5(:) * ( zwork2(:) - 1. )
279 pdtevr(:) = putpr(:) - ptpr(:)
280 pdtevrf(:,ikb+1) = pdtevr(:)
281ELSEWHERE
282 ptpr(:) = putpr(:)
283END WHERE
284!
285!
286IF (lhook) CALL dr_hook('CONVECT_PRECIP_ADJUST',1,zhook_handle)
287END SUBROUTINE convect_precip_adjust
288END MODULE mode_convect_precip_adjust
