88LOGICAL, INTENT(IN) :: OOCEAN ! switch for Ocean model version
89LOGICAL, INTENT(IN) :: OFLAT ! Logical for zero ororography
90LOGICAL, INTENT(IN) :: OCOMPUTE_SRC ! flag to define dimensions of SIGS and SRCT variables
91LOGICAL, INTENT(IN) :: OBLOWSNOW ! switch to activate pronostic blowing snow
92LOGICAL, INTENT(IN) :: ONOMIXLG ! to use turbulence for lagrangian variables (modd_conf)
93LOGICAL, INTENT(IN) :: O2D ! Logical for 2D model version (modd_conf)
94REAL, INTENT(IN) :: PRSNOW ! Ratio for diffusion coeff. scalar (blowing snow)
95REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PDXX, PDYY, PDZZ, PDZX, PDZY
96 ! Metric coefficients
97REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PLM ! mixing length
98REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PLEPS ! dissipative length
99REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PTKEM ! tke
100REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PTHVREF ! reference Thv
101REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PTHLM ! potential temperature at t-Delta t
102REAL, DIMENSION(D%NIT,D%NJT,D%NKT,KRR), INTENT(IN) :: PRM ! Mixing ratios at t-Delta t
103REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PLOCPEXNM ! Lv(T)/Cp/Exnref at time t-1
104REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PATHETA ! coefficients between
105REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PAMOIST ! s and Thetal and Rnp
106REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PSRCM ! normalized
107 ! 2nd-order flux s'r'c/2Sigma_s2 at t-1 multiplied by Lambda_3
108REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PWM ! w at t-1
109REAL, DIMENSION(D%NIT,D%NJT,D%NKT,KSV), INTENT(IN) :: PSVM ! scalar var. at t-1
110!
111!
112!
113!* 0.2 declaration of local variables
114!
115REAL, DIMENSION(D%NIT,D%NJT,D%NKT) &
116 :: ZFLX, ZA
117!
118INTEGER :: JSV ! loop counter
119INTEGER :: IKU, IIT, IJT, IKT, JI, JJ, JK
120!
121REAL :: ZTIME1, ZTIME2
122!
123REAL :: ZCSVD = 1.2 ! constant for scalar variance dissipation
124REAL :: ZCTSVD = 2.4 ! constant for temperature - scalar covariance dissipation
125REAL :: ZCQSVD = 2.4 ! constant for humidity - scalar covariance dissipation
126!
127REAL :: ZCSV !constant for the scalar flux
128! ---------------------------------------------------------------------------
129!
130iku=SIZE(ptkem,3)
131iit=d%NIT
132ijt=d%NJT
133ikt=d%NKT
134CALL second_mnh(ztime1)
135!
136IF(oblowsnow) THEN
137! See Vionnet (PhD, 2012) for a complete discussion around the value of the Schmidt number for blowing snow variables
138 zcsv= turbn%XCHF/prsnow
139ELSE
140 zcsv= turbn%XCHF
141ENDIF
142!
143DO jsv=1,ksv
144!
145 IF (onomixlg .AND. jsv >= ksv_lgbeg .AND. jsv<= ksv_lgend) cycle
146 !
147 ! variance Sv2
148 !
149 IF (tles%LLES_CALL) THEN
150 IF (.NOT. o2d) THEN
151 zflx(:,:,:) = zcsv / zcsvd * plm(:,:,:) * pleps(:,:,:) * &
152 ( gx_m_m(oflat,psvm(:,:,:,jsv),pdxx,pdzz,pdzx)**2 &
153 + gy_m_m(oflat,psvm(:,:,:,jsv),pdyy,pdzz,pdzy)**2 )
154 ELSE
155 zflx(:,:,:) = zcsv / zcsvd * plm(:,:,:) * pleps(:,:,:) * &
156 gx_m_m(oflat,psvm(:,:,:,jsv),pdxx,pdzz,pdzx)**2
157 END IF
158 CALL les_mean_subgrid( -2.*zcsvd*sqrt(ptkem)*zflx/pleps, &
159 tles%X_LES_SUBGRID_DISS_Sv2(:,:,:,jsv), .true. )
160 CALL les_mean_subgrid( mzf(pwm)*zflx, tles%X_LES_RES_W_SBG_Sv2(:,:,:,jsv), .true. )
161 END IF
162 !
163 ! covariance SvThv
164 !
165 IF (tles%LLES_CALL) THEN
166 CALL etheta(d,cst,krr,krri,pthlm,prm,plocpexnm,patheta,psrcm,oocean,ocompute_src,za)
167 IF (.NOT. o2d) THEN
168 zflx(:,:,:)= plm(:,:,:) * pleps(:,:,:) &
169 * ( gx_m_m(oflat,pthlm,pdxx,pdzz,pdzx) * gx_m_m(oflat,psvm(:,:,:,jsv),pdxx,pdzz,pdzx) &
170 + gy_m_m(oflat,pthlm,pdyy,pdzz,pdzy) * gy_m_m(oflat,psvm(:,:,:,jsv),pdyy,pdzz,pdzy) &
171 ) * (turbn%XCSHF+zcsv) / (2.*zctsvd)
172 ELSE
173 zflx(:,:,:)= plm(:,:,:) * pleps(:,:,:) &
174 * gx_m_m(oflat,pthlm,pdxx,pdzz,pdzx) * gx_m_m(oflat,psvm(:,:,:,jsv),pdxx,pdzz,pdzx) &
175 * (turbn%XCSHF+zcsv) / (2.*zctsvd)
176 END IF
177 CALL les_mean_subgrid( za*zflx, tles%X_LES_SUBGRID_SvThv(:,:,:,jsv) , .true.)
178 CALL les_mean_subgrid( -cst%XG/pthvref/3.*za*zflx, tles%X_LES_SUBGRID_SvPz(:,:,:,jsv), .true. )
179 !
180 IF (krr>=1) THEN
181 CALL emoist(d,cst,krr,krri,pthlm,prm,plocpexnm,pamoist,psrcm,oocean,za)
182 IF (.NOT. o2d) THEN
183 zflx(:,:,:)= plm(:,:,:) * pleps(:,:,:) &
184 * ( gx_m_m(oflat,prm(:,:,:,1),pdxx,pdzz,pdzx) * gx_m_m(oflat,psvm(:,:,:,jsv),pdxx,pdzz,pdzx) &
185 + gy_m_m(oflat,prm(:,:,:,1),pdyy,pdzz,pdzy) * gy_m_m(oflat,psvm(:,:,:,jsv),pdyy,pdzz,pdzy) &
186 ) * (turbn%XCHF+zcsv) / (2.*zcqsvd)
187 ELSE
188 zflx(:,:,:)= plm(:,:,:) * pleps(:,:,:) &
189 * gx_m_m(oflat,prm(:,:,:,1),pdxx,pdzz,pdzx) * gx_m_m(oflat,psvm(:,:,:,jsv),pdxx,pdzz,pdzx) &
190 * (turbn%XCHF+zcsv) / (2.*zcqsvd)
191 END IF
192 CALL les_mean_subgrid( za*zflx, tles%X_LES_SUBGRID_SvThv(:,:,:,jsv) , .true.)
193 CALL les_mean_subgrid( -cst%XG/pthvref/3.*za*zflx, tles%X_LES_SUBGRID_SvPz(:,:,:,jsv), .true. )
194 END IF
195 END IF
196!
197END DO ! end loop JSV
198!
199CALL second_mnh(ztime2)
200tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1

202END SUBROUTINE turb_hor_sv_corr
