38 TYPE dimphyex_t
39  !
40  !On x direction
41  INTEGER :: nit ! Array total dimension

44  !
45  !On y direction
46  INTEGER :: njt ! Array total dimension
47  INTEGER :: njb ! First inner mass point index
48  INTEGER :: nje ! Last inner mass point index
49  !
50  !On z direction
51  !Ordering can be different depending on the host model
52  INTEGER :: nkl ! Order of the vertical levels
53  ! 1: as for Méso-NH, levels are numbered from ground to space
54  ! -1: as for AROME, levels are numbered from space to ground
55  INTEGER :: nkt ! Array total dimension
56  INTEGER :: nkles ! Number of vertical levels for LES diagnostics
57  INTEGER :: nka ! Near ground array index (is an unphysical level if JPVEXT!=0)
58  INTEGER :: nku ! Uppest atmosphere array index (is an unphysical level if JPVEXT!=0)
59  INTEGER :: nkb ! Near ground physical array index (e.g. equal to 1+JPVEXT if NKL==1)
60  INTEGER :: nke ! Uppest physical atmosphere array index (e.g. equal to 1+JPVEXT if NKL==-1)
61  INTEGER :: nktb ! Smaller index of the physical domain (equals to MIN(NKB, NKE)=1+JPVEXT)
62  INTEGER :: nkte ! Greater index of the physical domain (equals to MAX(NKB, NKE)=NKT-JPVEXT)
63  !Explanations about the different values. To loop on:
64  !* all (including non physical) levels from ground to top of atm: DO JK=NKA, NKU, KKL
65  !* all (including non physical) levels from top of atm to ground: DO JK=NKU, NKA, -KKL
66  !* physical levels only from ground to top of atm: DO JK=NKB, NKE, KKL
67  !* physical levels only from top of atm to ground: DO JK=NKE, NKB, -KKL
68  !* all (including non physical) following the array ordering: DO JK=1, NKT
69  !* physical levels only following the array ordering: DO JK=NKTB, NKTE
70  INTEGER :: nibc ! Computational indices used in DO LOOP
71  INTEGER :: njbc ! = NIB/NJC/NIE/NJE in all schemes
72  INTEGER :: niec ! except in turbulence where external HALO points must be
73  INTEGER :: njec ! included so NIBC=NJBC=1 and NIEC/NJEC=NIT/NJT
74  INTEGER :: nijt ! NIT*NJT for horizontal packing
75  INTEGER :: nijb ! First horizontal inner mass point index
76  INTEGER :: nije ! Last horizontal inner mass point index
77  !
78  INTEGER :: nlesmask ! Number of LES masks
79  INTEGER :: nles_times ! Number of LES time data storage
