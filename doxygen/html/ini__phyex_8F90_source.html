7!
8!IMPORTANT NOTE ON HOW TO DEAL WITH KEYS USED IN THE PHYSICS *AND* IN THE HOST MODEL
9!
10!When this situation occurs:
11!For example, if a physics option (eg logical switch) has an impact on the model setup, this
12!option must be accessible in the host model *and* in the physics.
13!
14!The prefered solution (when it is possible):
15!The option is implemented in a module and in a namelist of PHYEX. And, after the initialisation
16!is done, the option is available in the PHYEX_OUT structure (in addition to the PHYEX module) for use
17!by the host model.
18!
19!When the physics option is needed in the host model before the PHYEX initialisation, or if the physics
20!option must be computed from other options known by the host model, there are several solutions:
21!
22! If this situation is specific to only one host model:
23! The option is declared (module) and initialised (namelist) twice: once in the host model, once in PHYEX;
24! and a consistency check is added after the physics initialisation.
25! This solution is preferred to overwriting the variable stored in the PHYEX module, because between
26! initialization and overwriting, the variable may have been used to initialize a parameterization.
27!
28! If this issue is common to all the host models:
29! The option is declared and initialised in the host model. Then, the value is given, through INI_PHYEX, to
30! the parametrisation INIT subroutine (eg TURBN_INIT). In the INIT subroutine the value is assigned
31! to a module variable specific to PHYEX (in order to be accessible from inside the parametrisations).
32! In this case the variable *must not* be added in a PHYEX namelist.
33! An alternative to this solution is not to copy the variable inside a PHYEX module but add it directly
34! to the parametrisation call arguments.
35!

111!IMPORTANT NOTE on PHYEX_OUT arguments.
112!Logically this argument should be declared with INTENT(OUT) but in this case ifort (at least) breaks the
113!execution when the same structure is given for the PHYEX_IN and the PHYEX_OUT argument.
114!When INITENT(INOUT) is used, execution is OK on ifort.
115
116LOGICAL :: LLINIT, LLCHANGEMODEL, LLCHECK
117INTEGER :: IPRINT
118!
119!** ARGUMENTS
120!
121llinit=.true.
122IF(PRESENT(ldinit)) llinit=ldinit
123llcheck=.true.
124IF(PRESENT(ldcheck)) llcheck=ldcheck
125llchangemodel=.true.
126IF(PRESENT(ldchangemodel)) llchangemodel=ldchangemodel
127iprint=0
128IF(PRESENT(kprint)) iprint=kprint
129!
130!** CST
131!
132IF(llinit) THEN
133 IF(iprint==2) WRITE(unit=kluout,fmt='('' MODD_CST '')')
134 IF(PRESENT(phyex_in)) cst=phyex_in%CST
135 CALL ini_cst()
136 IF(iprint==2) CALL print_cst(kluout)
137 IF(PRESENT(phyex_out)) phyex_out%CST=cst
138ENDIF
139!
140!** MICROPHYSICS SCHEME
141!
142IF(cmicro=='ICE3' .OR. cmicro=='ICE4' .OR. cmicro=='LIMA') THEN
143 !The LIMA scheme makes use of the condensation routine of the ICE3/ICE4 scheme
144 !It is why the ICE3/ICE4 initialisation is needed in for the LIMA scheme
145 IF(iprint==2) WRITE(unit=kluout,fmt='('' MODD_PARAM_ICEN, MODD_RAIN_ICE_DESCRN, MODD_RAIN_ICE_PARAMN, MODD_CLOUDPARN '')')
146 IF(llchangemodel) THEN
147 CALL cloudpar_goto_model(kfrom, kto)
148 CALL param_ice_goto_model(kfrom, kto)
149 CALL rain_ice_descr_goto_model(kfrom, kto)
150 CALL rain_ice_param_goto_model(kfrom, kto)
151 IF(cmicro=='LIMA') THEN
152 CALL param_lima_associate()
153 ENDIF
154 ENDIF
155 IF(PRESENT(phyex_in)) THEN
156 param_icen=phyex_in%PARAM_ICEN
157 rain_ice_descrn=phyex_in%RAIN_ICE_DESCRN
158 rain_ice_paramn=phyex_in%RAIN_ICE_PARAMN
159 cloudparn=phyex_in%CLOUDPARN
160 IF(cmicro=='LIMA') THEN
161 param_lima=phyex_in%PARAM_LIMA
162 param_lima_warm=phyex_in%PARAM_LIMA_WARM
163 param_lima_cold=phyex_in%PARAM_LIMA_COLD
164 param_lima_mixed=phyex_in%PARAM_LIMA_MIXED
165 ENDIF
166 ENDIF
167
168 CALL param_icen_init(hprogram, tpfile, ldneednam, kluout, &
169 &lddefaultval, ldreadnam, ldcheck, kprint)
170 IF(cmicro=='LIMA') THEN
171 CALL param_lima_init(hprogram, tpfile, ldneednam, kluout, &
172 &lddefaultval, ldreadnam, ldcheck, kprint)
173 ENDIF
174
175 IF(llinit) THEN
176 CALL ini_rain_ice(hprogram, kluout, ptstep, pdzmin, cloudparn%NSPLITR, cmicro)
177 CALL ini_tiwmx
178 rain_ice_paramn%TIWMX=>tiwmx
179
180 IF(cmicro=='LIMA') THEN
181 CALL init_aerosol_properties
182 CALL ini_lima(ptstep, pdzmin, cloudparn%NSPLITR, cloudparn%NSPLITG)
183 ENDIF
184 ENDIF
185
186 IF(PRESENT(phyex_out)) THEN
187 phyex_out%PARAM_ICEN=param_icen
188 phyex_out%RAIN_ICE_DESCRN=rain_ice_descrn
189 phyex_out%RAIN_ICE_PARAMN=rain_ice_paramn
190 phyex_out%CLOUDPARN=cloudparn
191 IF(cmicro=='LIMA') THEN
192 phyex_out%PARAM_LIMA=param_lima
193 phyex_out%PARAM_LIMA_WARM=param_lima_warm
194 phyex_out%PARAM_LIMA_COLD=param_lima_cold
195 phyex_out%PARAM_LIMA_MIXED=param_lima_mixed
196 ENDIF
197 ENDIF
198ENDIF
199!
200!** NSV init (must be after LIMA init)
201!
202IF(llinit) THEN
203 IF(llchangemodel) CALL nsv_associate()
204 IF(PRESENT(phyex_in)) tnsv=phyex_in%TNSV
205 tnsv%NSV=0
206 CALL lima_update_nsv(param_lima, tnsv, odinit=.true., kmi=kto, ksv=tnsv%NSV, &
207 &hdcloud=cmicro, odupdate=.true.)
208 tnsv%NSV=tnsv%NSV_LIMA
209 IF(PRESENT(phyex_out)) phyex_out%TNSV=tnsv
210ENDIF
211!
212!** CLOUD SCHEME
213!
214! Must be before microphysics (consistency checks in PARAM_LIMA_INIT)
215IF(.true.) THEN !Placeholder for configuration without cloud scheme or a different one
216 IF(iprint==2) WRITE(unit=kluout,fmt='('' MODD_NEB_n '')')
217 IF(llchangemodel) CALL neb_goto_model(kfrom, kto)
218 IF(PRESENT(phyex_in)) nebn=phyex_in%NEBN
219
220 CALL nebn_init(hprogram, tpfile, ldneednam, kluout, &
221 &lddefaultval, ldreadnam, ldcheck, kprint)
222 IF(llinit) THEN
223 !Nothing to do, everything is read from namelist
224 ENDIF
225
226 IF(PRESENT(phyex_out)) phyex_out%NEBN=nebn
227ENDIF
228!
229!** SHALLOW CONVECTION SCHEME
230!
231IF(csconv=='EDKF') THEN
232 IF(iprint==2) WRITE(unit=kluout,fmt='('' MODD_PARAM_MFSHALL_n '')')
233 IF(llchangemodel) CALL param_mfshall_goto_model(kfrom, kto)
234 IF(PRESENT(phyex_in)) param_mfshalln=phyex_in%PARAM_MFSHALLN
235
236 CALL param_mfshalln_init(hprogram, tpfile, ldneednam, kluout, &
237 &lddefaultval, ldreadnam, ldcheck, kprint)
238 IF(llinit) THEN
239 CALL ini_mfshall()
240 ENDIF
241
242 IF(PRESENT(phyex_out)) phyex_out%PARAM_MFSHALLN=param_mfshalln
243ENDIF
244!
245!** TURBULENCE SCHEME
246!
247IF(cturb=='TKEL') THEN
248 IF(iprint==2) WRITE(unit=kluout,fmt='('' MODD_TURB_n MODD_CTURB '')')
249 IF(llchangemodel) THEN
250 CALL turb_goto_model(kfrom, kto)
251 CALL cturb_associate()
252 ENDIF
253 IF(PRESENT(phyex_in)) THEN
254 turbn=phyex_in%TURBN
255 csturb=phyex_in%CSTURB
256 ENDIF
257
258 CALL turbn_init(hprogram, tpfile, ldneednam, kluout, &
259 &lddefaultval, ldreadnam, ldcheck, kprint)
260 IF(llinit) THEN
261 CALL ini_turb(hprogram)
262 ENDIF
263
264 IF(PRESENT(phyex_out)) THEN
265 phyex_out%TURBN=turbn
266 phyex_out%CSTURB=csturb
267 ENDIF
268ENDIF
269!
270!** GLOBAL CONSISTENCY TESTS
271!
272IF(llcheck) THEN
273 IF((cmicro=='ICE3' .OR. cmicro=='ICE4' .OR. cmicro=='LIMA') .AND. cturb=='TKEL') THEN
274 IF ((csubg_aucv_rc == 'ADJU' .OR. csubg_aucv_ri == 'ADJU') .AND. ccondens /= 'GAUS') THEN
275 CALL print_msg(nverb_fatal, 'GEN', 'INI_PHYEX', &
276 &"CSUBG_AUCV_RC and/or CSUBG_AUCV_RI cannot be 'ADJU' if CCONDENS is not 'GAUS'")
277 ENDIF
278 IF ((csubg_aucv_rc == 'ADJU' .OR. csubg_aucv_ri == 'ADJU') .AND. cmicro=='LIMA') THEN
279 CALL print_msg(nverb_fatal, 'GEN', 'INI_PHYEX', &
280 &"CSUBG_AUCV_RC and/or CSUBG_AUCV_RI cannot be 'ADJU' if CMICRO is 'LIMA'")
281 ENDIF
282 IF (.NOT. lharat .AND. lstatnw) THEN
283 CALL print_msg(nverb_fatal, 'GEN', 'INI_PHYEX', &
284 &'LSTATNW only tested in combination with HARATU and EDMFm!')
285 ENDIF
286 ENDIF
287 !
288 IF(cmicro=='LIMA') THEN
289 IF (lsubg_cond .AND. (.NOT. lptsplit)) THEN
290 CALL print_msg(nverb_fatal, 'GEN', 'INI_PHYEX', &
291 &"YOU MUST USE LPTSPLIT=T WITH CMICRO=LIMA AND LSUBG_COND")
292 END IF
293 IF (lsubg_cond .AND. (.NOT. ladj)) THEN
294 CALL print_msg(nverb_fatal, 'GEN', 'INI_PHYEX', &
295 &"YOU MUST USE LADJ=t WITH CMICRO=LIMA AND LSUBG_COND")
296 END IF
297 IF (csconv=='EDKF') THEN
298 IF (param_mfshalln%LRELAX_ALPHA_MF) THEN !PARAM_MFSHALLN%LRELAX_ALPHA_MF undefined if CSCONV/=EDKF
299 CALL print_msg(nverb_fatal, 'GEN', 'INI_PHYEX', &
300 &"YOU CANNOT USE LRELAX_ALPHA_MF when CMICRO is 'LIMA'")
301 ENDIF
302 ENDIF
303 IF(.NOT. nebn%LCONDBORN) THEN
304 CALL print_msg(nverb_fatal, 'GEN', 'INI_PHYEX', &
305 &"LCONDBORN (MODD_NEB_n) must be .TRUE. with LIMA")
306 ENDIF
307 ENDIF
308ENDIF
