28 !!**** *HYPSER* - hypergeometric function
29 !!
30 !!
31 !! PURPOSE
32 !! -------

35 !!
36 !!
37 !! A*B (A+1)A*(B+1)B X^2
38 !! HYPSER(A,B,C,X)= 1 + ----- * X + ------------- * --- + ... +
39 !! C (C+1)C 2
40 !!
41 !! (A+n)...A*(B+n)...B X^n
42 !! --------------------- * ----- + ... ...
43 !! (C+n)...C n!
44 !!
45 !!** METHOD
46 !! ------
47 !!
48 !! EXTERNAL

50 !! HYPSER

84 !

89 INTEGER :: JN,JFLAG
90 REAL :: ZXH,ZZA,ZZB,ZZC,ZFAC,ZTEMP
91 REAL :: ZPREC

93 !------------------------------------------------------------------------------
94 !
95 zprec = 1.0e-04
96 zxh = px
97 zfac = 1.0
98 ztemp = zfac
99 zza = pa
100 zzb = pb
101 zzc = pc
102 jflag = 0
103 serie: DO jn = 1,5000
104  zfac = zfac * zza * zzb / zzc
105  zfac = zfac * zxh / real(jn)
106  phyp = ztemp + zfac
107  IF (abs(phyp-ztemp).LE.zprec) THEN
108  jflag = 1
109  EXIT serie
110  END IF
111  ztemp = phyp
112  zza = zza + 1.
113  zzb = zzb + 1.
114  zzc = zzc + 1.
115 END DO serie
116 IF (jflag == 0) call print_msg(nverb_fatal,'GEN','HYPSER','convergence failure')
117 !
