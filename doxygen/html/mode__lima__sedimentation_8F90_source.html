17 !! PURPOSE
18 !! -------
19 !! The purpose of this routine is to compute the sedimentation of any hydrometeor,
20 !! also accounting for the transport of heat
21 !!
22 !! METHOD
23 !! ------
24 !! The sedimentation rates are computed with a time spliting technique:
25 !! an upstream scheme, written as a difference of non-advective fluxes.
26 !! This source term is added to the next coming time step (split-implicit
27 !! process).

139 !

141 !
142 ! Time splitting
143 !
144 ztsplitg= ptstep / real(nsplitsed(kid))
145 !
146 zwdt=0.
147 pinpr(:,:) = 0.
148 zwsedr(:,:,:) = 0.
149 zwsedc(:,:,:) = 0.
150 !
151 prs(:,:,:) = prs(:,:,:) * ptstep
152 IF (kmoments==2) pcs(:,:,:) = pcs(:,:,:) * ptstep
153 IF (oelec) pqs(:,:,:) = pqs(:,:,:) * ptstep
154 DO jk = d%NKTB , d%NKTE
155  zw(:,:,jk)=ztsplitg/pdzz(:,:,jk)
156 END DO
157 !
158 IF (hphase=='L') zc=cst%XCL
159 IF (hphase=='I') zc=cst%XCI
160 !
161 ! When pristine ice is 1-moment, nb concentration is parameterized following
162 ! McFarquhar and Heymsfield (1997) for columns as in ICE3
163 IF (kid==4 .AND. imoments==1) THEN
164  imoments=2
165  WHERE(prs(:,:,:)>0) pcs(:,:,:)=1/(4*cst%XPI*900.) * prs(:,:,:) * &
166  max(0.05e6,-0.15319e6-0.021454e6*alog(prhodref(:,:,:)*prs(:,:,:)))**3
167 END IF
168 !
169 ! ################################
170 ! Compute the sedimentation fluxes
171 ! ################################
172 !
173 DO jn = 1 , nsplitsed(kid)
174  ! Computation only where enough ice, snow, graupel or hail
175  gsedim(:,:,:) = .false.
176  gsedim(d%NIB:d%NIE,d%NJB:d%NJE,d%NKTB:d%NKTE) = prs(d%NIB:d%NIE,d%NJB:d%NJE,d%NKTB:d%NKTE)>xrtmin(kid)
177  IF (imoments==2) gsedim(:,:,:) = gsedim(:,:,:) .AND. pcs(:,:,:)>xctmin(kid)
178  isedim = countjv( gsedim(:,:,:),i1(:),i2(:),i3(:))
179 !
180  IF( isedim >= 1 ) THEN
181 !
182  ALLOCATE(zrhodref(isedim))
183  ALLOCATE(zpabst(isedim))
184  ALLOCATE(zt(isedim))
185  ALLOCATE(zrs(isedim))
186  ALLOCATE(zcs(isedim))
187  ALLOCATE(zlbda(isedim)) ; zlbda(:) = 1.e10
188  ALLOCATE(zcc(isedim)) ; zcc(:) = 1.0
189  ALLOCATE(zzw(isedim)) ; zzw(:) = 0.0
190  ALLOCATE(zzx(isedim)) ; zzx(:) = 0.0
191  ALLOCATE(zzy(isedim)) ; zzy(:) = 0.0
192  !
193  IF (oelec) THEN
194  zwsedq(:,:,:) = 0.
195  ALLOCATE(zes(isedim)) ; zes(:) = 0.0
196  ALLOCATE(zqs(isedim)) ; zqs(:) = 0.0
197  ALLOCATE(zzq(isedim)) ; zzq(:) = 0.0
198  END IF
199 !
200  DO jl = 1,isedim
201  zrhodref(jl) = prhodref(i1(jl),i2(jl),i3(jl))
202  zpabst(jl) = ppabst(i1(jl),i2(jl),i3(jl))
203  zt(jl) = pt(i1(jl),i2(jl),i3(jl))
204  zrs(jl) = prs(i1(jl),i2(jl),i3(jl))
205  IF (imoments==2) zcs(jl) = pcs(i1(jl),i2(jl),i3(jl))
206  IF (oelec) zqs(jl) = pqs(i1(jl),i2(jl),i3(jl))
207  END DO
208 !
209 ! Compute lambda
210  IF (kid == 5 .AND. nmom_s.EQ.1 .AND. lsnow_t) THEN
211  zlbda(:) = 1.e10
212  WHERE(zt(:)>263.15 .AND. zrs(:)>xrtmin(5))
213  zlbda(:) = max(min(xlbdas_max, 10**(14.554-0.0423*zt(:))),xlbdas_min)
214  END WHERE
215  WHERE(zt(:)<=263.15 .AND. zrs(:)>xrtmin(5))
216  zlbda(:) = max(min(xlbdas_max, 10**(6.226-0.0106*zt(:))),xlbdas_min)
217  END WHERE
218  zlbda(:) = zlbda(:)*xtrans_mp_gammas
219  zzw(:) = xfsedr(kid) * zrhodref(:)**(1.-xcexvt)*zrs(:)* &
220  (1 + (xfvelos/zlbda(:))**xalphas)**(-xnus-(xd(kid)+xbs)/xalphas) * zlbda(:)**(-xd(kid))
221  ELSE
222  IF (imoments==1) zlbda(:) = xlb(kid) * ( zrhodref(:) * zrs(:) )**xlbex(kid)
223  IF (imoments==2) zlbda(:) = ( xlb(kid)*zcs(:) / zrs(:) )**xlbex(kid)
224  zzy(:) = zrhodref(:)**(-xcexvt) * zlbda(:)**(-xd(kid))
225  IF (lsnow_t .AND. kid==5) &
226  zzy(:) = zzy(:) * (1 + (xfvelos/zlbda(:))**xalphas)**(-xnus-(xd(kid)+xbs)/xalphas)
227  zzw(:) = xfsedr(kid) * zrs(:) * zzy(:) * zrhodref(:)
228  END IF ! Wurtz
229 !
230  IF (kmoments==2) zzx(:) = xfsedc(kid) * zcs(:) * zzy(:) * zrhodref(:)
231 
232  IF (kid==2) THEN
233  ! mean cloud droplet diameter
234  zcc(:) = 0.5*gamma_x0d(xnuc+1./xalphac)/(gamma_x0d(xnuc)*zlbda(:))
235  ! correction factor for cloud droplet terminal fall speed
236  zcc(:) = 1.+1.26*6.6e-8*(101325./zpabst(:))*(zt(:)/293.15)/zcc(:)
237  zzw(:) = zcc(:) * zzw(:)
238  zzx(:) = zcc(:) * zzx(:)
239  END IF
240 !
241 ! If the electrical scheme is activated, the electric field can impact the sedimentation
242  zbeardcoeff(:,:,:) = 1.0
243  IF (oelec .AND. lsedim_beard) THEN
244  zlbda3(:,:,:) = unpack( zlbda(:),mask=gsedim(:,:,:),field=0.0 )
245  CALL elec_beard_effect(d, cst, 'LIMA', kid, gsedim, pt, prhodref, pthvrefzikb, &
246  prs, pqs, pefieldw, zlbda3, zbeardcoeff, &
247  limap=param_lima, limapc=param_lima_cold, limapw=param_lima_warm, limapm=param_lima_mixed)
248  END IF
249 !
250  zwsedr(:,:,1:d%NKT) = unpack( zzw(:),mask=gsedim(:,:,:),field=0.0 )
251  zwsedr(:,:,d%NKTB:d%NKTE) = min( zwsedr(:,:,d%NKTB:d%NKTE) * zbeardcoeff(:,:,d%NKTB:d%NKTE), &
252  prs(:,:,d%NKTB:d%NKTE) * prhodref(:,:,d%NKTB:d%NKTE) / &
253  zw(:,:,d%NKTB:d%NKTE) )
254  IF (kmoments==2) THEN
255  zwsedc(:,:,1:d%NKT) = unpack( zzx(:),mask=gsedim(:,:,:),field=0.0 )
256  zwsedc(:,:,d%NKTB:d%NKTE) = min( zwsedc(:,:,d%NKTB:d%NKTE) * zbeardcoeff(:,:,d%NKTB:d%NKTE), &
257  pcs(:,:,d%NKTB:d%NKTE) * prhodref(:,:,d%NKTB:d%NKTE) / &
258  zw(:,:,d%NKTB:d%NKTE) )
259  END IF
260 !
261 ! Sedimentation of electric charges
262  IF (oelec) THEN
263  ! compute e of the q-D relationship
264  IF (imoments == 2) THEN ! 2-moment species
265  CALL elec_compute_ex (kid, imoments, isedim, 'LIMA', ptstep, zrhodref, xrtmin(kid), &
266  zrs, zqs, zes, plbdx=zlbda, pcx=zcs)
267  ELSE ! 1-moment species
268  CALL elec_compute_ex (kid, imoments, isedim, 'LIMA', ptstep, zrhodref, xrtmin(kid), &
269  zrs, zqs, zes, plbdx=zlbda)
270  END IF
271  !
272  ! number concentration for 1-moment species
273  ! for precipitating hydrometeors, N=C\lambda^x, except for snow if lsnow_t=t
274  IF (imoments == 1) THEN
275  IF (kid == 5) THEN
276  zcx = xccs
277  zxx = xcxs
278  ELSE IF (kid == 6) THEN
279  zcx = xccg
280  zxx = xcxg
281  ELSE IF (kid == 7) THEN
282  zcx = xcch
283  zxx = xcxh
284  END IF
285  zcs(:) = zcx * zlbda(:)**zxx
286  END IF
287  !
288  zzq(:) = zrhodref(:)**(1.-xcexvt) * zes(:) * zcs(:) * xfqsed(kid) * zlbda(:)**(-xdq(kid))
289  !
290  ! correction for cloud droplet terminal fall speed
291  IF (kid == 2) zzq(:) = zzq(:) * zcc(:)
292  !
293  zwsedq(:,:,1:d%NKT) = unpack( zzq(:),mask=gsedim(:,:,:),field=0.0 )
294  zwsedq(:,:,1:d%NKT) = zwsedq(:,:,1:d%NKT) * zbeardcoeff(:,:,1:d%NKT)
295  zwsedq(:,:,d%NKTB:d%NKTE) = sign(min(abs(zwsedq(:,:,d%NKTB:d%NKTE)), &
296  abs(pqs(:,:,d%NKTB:d%NKTE)*prhodref(:,:,d%NKTB:d%NKTE)/zw(:,:,d%NKTB:d%NKTE))), &
297  zwsedq(:,:,d%NKTB:d%NKTE))
298  END IF
299 
300  DO jk = d%NKTB , d%NKTE
301  prs(:,:,jk) = prs(:,:,jk) + zw(:,:,jk)* &
302  (zwsedr(:,:,jk+d%NKL)-zwsedr(:,:,jk))/prhodref(:,:,jk)
303  pfpr(:,:,jk) = zwsedr(:,:,jk)
304  IF (kmoments==2) pcs(:,:,jk) = pcs(:,:,jk) + zw(:,:,jk)* &
305  (zwsedc(:,:,jk+d%NKL)-zwsedc(:,:,jk))/prhodref(:,:,jk)
306  ! Heat transport
307  !PRT_SUM(:,:,JK-D%NKL) = PRT_SUM(:,:,JK-D%NKL) + ZW(:,:,JK-D%NKL)*ZWSEDR(:,:,JK)/PRHODREF(:,:,JK-D%NKL)
308  !PRT_SUM(:,:,JK) = PRT_SUM(:,:,JK) - ZW(:,:,JK)*ZWSEDR(:,:,JK)/PRHODREF(:,:,JK)
309  !PCPT(:,:,JK-D%NKL) = PCPT(:,:,JK-D%NKL) + ZC * (ZW(:,:,JK-D%NKL)*ZWSEDR(:,:,JK)/PRHODREF(:,:,JK-D%NKL))
310  !PCPT(:,:,JK) = PCPT(:,:,JK) - ZC * (ZW(:,:,JK)*ZWSEDR(:,:,JK)/PRHODREF(:,:,JK))
311  !ZWDT(:,:,JK) =(PRHODREF(:,:,JK+D%NKL)*(1.+PRT_SUM(:,:,JK))*PCPT(:,:,JK)*PT(:,:,JK) + &
312  ! ZW(:,:,JK)*ZWSEDR(:,:,JK+1)*ZC*PT(:,:,JK+D%NKL)) / &
313  ! (PRHODREF(:,:,JK+D%NKL)*(1.+PRT_SUM(:,:,JK))*PCPT(:,:,JK) + ZW(:,:,JK)*ZWSEDR(:,:,JK+D%NKL)*ZC)
314  !ZWDT(:,:,JK) = ZWDT(:,:,JK) - PT(:,:,JK)
315  IF (oelec) pqs(:,:,jk) = pqs(:,:,jk) + zw(:,:,jk) * &
316  (zwsedq(:,:,jk+d%NKL) - zwsedq(:,:,jk)) / prhodref(:,:,jk)
317 
318  END DO
319  !
320  DEALLOCATE(zrhodref)
321  DEALLOCATE(zpabst)
322  DEALLOCATE(zt)
323  DEALLOCATE(zrs)
324  DEALLOCATE(zcs)
325  DEALLOCATE(zcc)
326  DEALLOCATE(zlbda)
327  DEALLOCATE(zzw)
328  DEALLOCATE(zzx)
329  DEALLOCATE(zzy)
330  IF (ALLOCATED(zqs)) DEALLOCATE(zqs)
331  IF (ALLOCATED(zzq)) DEALLOCATE(zzq)
332  IF (ALLOCATED(zes)) DEALLOCATE(zes)
333  !
334  pinpr(:,:) = pinpr(:,:) + zwsedr(:,:,d%NKB)/cst%XRHOLW/nsplitsed(kid) ! in m/s
335  !PT(:,:,:) = PT(:,:,:) + ZWDT(:,:,:)
336 
337  END IF
338 END DO
339 !
340 prs(:,:,:) = prs(:,:,:) / ptstep
341 IF (kmoments==2) pcs(:,:,:) = pcs(:,:,:) / ptstep
342 IF (oelec) pqs(:,:,:) = pqs(:,:,:) / ptstep
343 !
344 END SUBROUTINE lima_sedimentation
