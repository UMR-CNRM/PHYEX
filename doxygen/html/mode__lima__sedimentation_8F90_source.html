17!! PURPOSE
18!! -------
19!! The purpose of this routine is to compute the sedimentation of any hydrometeor,
20!! also accounting for the transport of heat
21!!
22!! METHOD
23!! ------
24!! The sedimentation rates are computed with a time spliting technique:
25!! an upstream scheme, written as a difference of non-advective fluxes.
26!! This source term is added to the next coming time step (split-implicit
27!! process).
28!!
29!! AUTHOR

31!! J.-M. Cohard * Laboratoire d'Aerologie*

34!! B. Vié * CNRM *
35!!
36!! MODIFICATIONS
37!! -------------
38!! Original 15/03/2018
39!!
40!! B.Vie 02/2019 Desactivate (comment) the heat transport by droplets

43! B. Vie 03/2020: disable temperature change of droplets by air temperature

102!
103LOGICAL, DIMENSION(D%NIJT, D%NKT) :: GSEDIM ! Test where to compute the SED processes
104REAL, DIMENSION(D%NIJT, D%NKT) :: ZW, & ! Work array
105 ZWDT ! Temperature change
106REAL, DIMENSION(D%NIJT,0:D%NKT+1) &
107 :: ZWSEDR, & ! Sedimentation of MMR
108 ZWSEDC ! Sedimentation of number conc.

111 :: ZRS, & ! m.r. source
112 ZCS, & ! conc. source
113 ZRHODREF, & ! RHO Dry REFerence
114 ZPABST, & ! Pressure
115 ZT, & ! Temperature
116 ZZW, & ! Work array
117 ZZX, & ! Work array
118 ZZY, & ! Work array
119 ZLBDA, & ! Slope parameter
120 ZCC ! Cunningham corrective term for droplets fall speed
121!
122INTEGER , DIMENSION(D%NIJT*D%NKT) :: I1,I2,I3 ! Indexes for PACK replacement
123!
124REAL :: ZTSPLITG ! Small time step for rain sedimentation
125REAL :: ZC ! Cpl or Cpi
126INTEGER :: IMOMENTS
127REAL(KIND=jphook) :: zhook_handle
128!
129! Variables for cloud electricity
130REAL :: ZCX, ZXX ! C and x parameters for N-lambda relationship
131REAL, DIMENSION(:), ALLOCATABLE :: ZQS, & ! Electric charge density source
132 ZZQ, & ! Work array
133 ZES ! e in q-D relationship
134REAL, DIMENSION(MERGE(D%NIJT, 0, OELEC), & &MERGE(D%NKT, 0, OELEC)) :: ZWSEDQ, & ! Sedimentation of electric charge density
135 & ZLBDA3
136TYPE(PARAM_LIMA_MIXED_T),INTENT(IN)::LIMAM
137TYPE(param_lima_cold_t),INTENT(IN)::LIMAC
138TYPE(param_lima_warm_t),INTENT(IN)::LIMAW
139TYPE(param_lima_t),INTENT(IN)::LIMAP
140REAL, DIMENSION(D%NIJT, D%NKT):: ZBEARDCOEFF ! effect of electrical forces on terminal fall speed
141!
142!-------------------------------------------------------------------------------
143!
144IF (lhook) CALL dr_hook('LIMA_SEDIMENTATION', 0, zhook_handle)
145imoments=kmoments
146!
147! Time splitting
148!
149ztsplitg= ptstep / real(limap%NSPLITSED(kid))
150!
151zwdt(:,:)=0.
152pinpr(:) = 0.
153pfpr(:,:) = 0.
154zwsedr(:,:) = 0.
155zwsedc(:,:) = 0.
156!
157prs(:,:) = prs(:,:) * ptstep
158IF (kmoments==2) pcs(:,:) = pcs(:,:) * ptstep
159IF (oelec) pqs(:,:) = pqs(:,:) * ptstep
160DO ik = d%NKTB , d%NKTE
161 zw(:,ik)=ztsplitg/pdzz(:,ik)
162END DO
163!
164IF (hphase=='L') zc=cst%XCL
165IF (hphase=='I') zc=cst%XCI
166!
167! When pristine ice is 1-moment, nb concentration is parameterized following
168! McFarquhar and Heymsfield (1997) for columns as in ICE3
169IF (kid==4 .AND. imoments==1) THEN
170 imoments=2
171 WHERE(prs(:,:)>0) pcs(:,:)=1/(4*cst%XPI*900.) * prs(:,:) * &
172 max(0.05e6,-0.15319e6-0.021454e6*log(prhodref(:,:)*prs(:,:)))**3
173END IF
174!
175! ################################
176! Compute the sedimentation fluxes
177! ################################
178!
179DO in = 1 , limap%NSPLITSED(kid)
180 ! Computation only where enough ice, snow, graupel or hail
181 gsedim(:,:) = .false.
182 gsedim(d%NIJB:d%NIJE,d%NKTB:d%NKTE) = prs(d%NIJB:d%NIJE,d%NKTB:d%NKTE)>limap%XRTMIN(kid)
183 IF (imoments==2) gsedim(:,:) = gsedim(:,:) .AND. pcs(:,:)>limap%XCTMIN(kid)
184 isedim = countjv( gsedim(:,:),i1(:),i3(:))
185!
186 IF( isedim >= 1 ) THEN
187!
188 ALLOCATE(zrhodref(isedim))
189 ALLOCATE(zpabst(isedim))
190 ALLOCATE(zt(isedim))
191 ALLOCATE(zrs(isedim))
192 ALLOCATE(zcs(isedim))
193 ALLOCATE(zlbda(isedim)) ; zlbda(:) = 1.e10
194 ALLOCATE(zcc(isedim)) ; zcc(:) = 1.0
195 ALLOCATE(zzw(isedim)) ; zzw(:) = 0.0
196 ALLOCATE(zzx(isedim)) ; zzx(:) = 0.0
197 ALLOCATE(zzy(isedim)) ; zzy(:) = 0.0
198 !
199 IF (oelec) THEN
200 zwsedq(:,:) = 0.
201 ALLOCATE(zes(isedim)) ; zes(:) = 0.0
202 ALLOCATE(zqs(isedim)) ; zqs(:) = 0.0
203 ALLOCATE(zzq(isedim)) ; zzq(:) = 0.0
204 END IF
205!
206 DO il = 1,isedim
207 zrhodref(il) = prhodref(i1(il),i3(il))
208 zpabst(il) = ppabst(i1(il),i3(il))
209 zt(il) = pt(i1(il),i3(il))
210 zrs(il) = prs(i1(il),i3(il))
211 IF (imoments==2) zcs(il) = pcs(i1(il),i3(il))
212 IF (oelec) zqs(il) = pqs(i1(il),i3(il))
213 IF (limap%LCRYSTAL_SHAPE .AND. imoments == 2 .AND. kid == 4) &
214 zlbda(il) = plbdai_shape(i1(il),i3(il))
215 END DO
216!
217! Compute lambda
218 IF (kid == 5 .AND. limap%NMOM_S.EQ.1 .AND. limap%LSNOW_T) THEN
219 zlbda(:) = 1.e10
220 WHERE(zt(:)>263.15 .AND. zrs(:)>limap%XRTMIN(5))
221 zlbda(:) = max(min(limac%XLBDAS_MAX, 10**(14.554-0.0423*zt(:))),limac%XLBDAS_MIN)
222 END WHERE
223 WHERE(zt(:)<=263.15 .AND. zrs(:)>limap%XRTMIN(5))
224 zlbda(:) = max(min(limac%XLBDAS_MAX, 10**(6.226-0.0106*zt(:))),limac%XLBDAS_MIN)
225 END WHERE
226 zlbda(:) = zlbda(:)*limac%XTRANS_MP_GAMMAS
227 zzw(:) = limap%XFSEDR(kid) * zrhodref(:)**(1.-limap%XCEXVT)*zrs(:)* &
228 (1 + (limac%XFVELOS/zlbda(:))**limap%XALPHAS)**(-limap%XNUS-(limap%XD(kid)+limac%XBS)/limap%XALPHAS) &
229 * zlbda(:)**(-limap%XD(kid))
230 ELSE IF (kid == 4 .AND. imoments==2 .AND. limap%LCRYSTAL_SHAPE) THEN
231 zzy(:) = zrhodref(:)**(-limap%XCEXVT) * zlbda(:)**(-limac%XDI_SHAPE(kishape))
232 zzw(:) = limac%XFSEDRI_SHAPE(kishape) * zrs(:) * zzy(:) * zrhodref(:)
233 zzx(:) = limac%XFSEDCI_SHAPE(kishape) * zcs(:) * zzy(:) * zrhodref(:)
234 ELSE
235 IF (imoments==1) zlbda(:) = limap%XLB(kid) * ( zrhodref(:) * zrs(:) )**limap%XLBEX(kid)
236 IF (imoments==2) zlbda(:) = ( limap%XLB(kid)*zcs(:) / zrs(:) )**limap%XLBEX(kid)
237 zzy(:) = zrhodref(:)**(-limap%XCEXVT) * zlbda(:)**(-limap%XD(kid))
238 IF (limap%LSNOW_T .AND. kid==5) &
239 zzy(:) = zzy(:) * (1 + (limac%XFVELOS/zlbda(:))**limap%XALPHAS)**(-limap%XNUS-(limap%XD(kid)+limac%XBS)/limap%XALPHAS)
240 zzw(:) = limap%XFSEDR(kid) * zrs(:) * zzy(:) * zrhodref(:)
241 END IF ! Wurtz
242!
243 IF (kmoments==2 .AND. .NOT.(limap%LCRYSTAL_SHAPE .AND. kid==4)) &
244 zzx(:) = limap%XFSEDC(kid) * zcs(:) * zzy(:) * zrhodref(:)
245
246 IF (kid==2) THEN
247 ! mean cloud droplet diameter
248 zcc(:) = limaw%XGCC/zlbda(:)
249 ! correction factor for cloud droplet terminal fall speed
250 zcc(:) = 1.+1.26*6.6e-8*(101325./zpabst(:))*(zt(:)/293.15)/zcc(:)
251 zzw(:) = zcc(:) * zzw(:)
252 zzx(:) = zcc(:) * zzx(:)
253 END IF
254!
255! If the electrical scheme is activated, the electric field can impact the sedimentation
256 zbeardcoeff(:,:) = 1.0
257 IF (oelec .AND. elecd%LSEDIM_BEARD) THEN
258 zlbda3(:,:) = unpack( zlbda(:),mask=gsedim(:,:),field=0.0 )
259 CALL elec_beard_effect(d, cst, 'LIMA', kid, gsedim, pt, prhodref, pthvrefzikb, &
260 prs, pqs, pefieldw, zlbda3, zbeardcoeff, &
261 limap=limap, limapc=limac, limapw=limaw, limapm=limam)
262 END IF
263!
264 zwsedr(:,1:d%NKT) = unpack( zzw(:),mask=gsedim(:,:),field=0.0 )
265 zwsedr(:,d%NKTB:d%NKTE) = min( zwsedr(:,d%NKTB:d%NKTE) * zbeardcoeff(:,d%NKTB:d%NKTE), &
266 prs(:,d%NKTB:d%NKTE) * prhodref(:,d%NKTB:d%NKTE) / &
267 zw(:,d%NKTB:d%NKTE) )
268 IF (kmoments==2) THEN
269 zwsedc(:,1:d%NKT) = unpack( zzx(:),mask=gsedim(:,:),field=0.0 )
270 zwsedc(:,d%NKTB:d%NKTE) = min( zwsedc(:,d%NKTB:d%NKTE) * zbeardcoeff(:,d%NKTB:d%NKTE), &
271 pcs(:,d%NKTB:d%NKTE) * prhodref(:,d%NKTB:d%NKTE) / &
272 zw(:,d%NKTB:d%NKTE) )
273 END IF
274!
275! Sedimentation of electric charges
276 IF (oelec) THEN
277 ! compute e of the q-D relationship
278 IF (imoments == 2) THEN ! 2-moment species
279 CALL elec_compute_ex (kid, imoments, isedim, 'LIMA', ptstep, zrhodref, limap%XRTMIN(kid), &
280 zrs, zqs, zes, plbdx=zlbda, pcx=zcs)
281 ELSE ! 1-moment species
282 CALL elec_compute_ex (kid, imoments, isedim, 'LIMA', ptstep, zrhodref, limap%XRTMIN(kid), &
283 zrs, zqs, zes, plbdx=zlbda)
284 END IF
285 !
286 ! number concentration for 1-moment species
287 ! for precipitating hydrometeors, N=C\lambda^x, except for snow if lsnow_t=t
288 IF (imoments == 1) THEN
289 IF (kid == 5) THEN
290 zcx = limac%XCCS
291 zxx = limac%XCXS
292 ELSE IF (kid == 6) THEN
293 zcx = limam%XCCG
294 zxx = limam%XCXG
295 ELSE IF (kid == 7) THEN
296 zcx = limam%XCCH
297 zxx = limam%XCXH
298 END IF
299 zcs(:) = zcx * zlbda(:)**zxx
300 END IF
301 !
302 zzq(:) = zrhodref(:)**(1.-limap%XCEXVT) * zes(:) * zcs(:) * elecp%XFQSED(kid) * zlbda(:)**(-elecp%XDQ(kid))
303 !
304 ! correction for cloud droplet terminal fall speed
305 IF (kid == 2) zzq(:) = zzq(:) * zcc(:)
306 !
307 zwsedq(:,1:d%NKT) = unpack( zzq(:),mask=gsedim(:,:),field=0.0 )
308 zwsedq(:,1:d%NKT) = zwsedq(:,1:d%NKT) * zbeardcoeff(:,1:d%NKT)
309 zwsedq(:,d%NKTB:d%NKTE) = sign(min(abs(zwsedq(:,d%NKTB:d%NKTE)), &
310 abs(pqs(:,d%NKTB:d%NKTE)*prhodref(:,d%NKTB:d%NKTE)/zw(:,d%NKTB:d%NKTE))), &
311 zwsedq(:,d%NKTB:d%NKTE))
312 END IF
313
314 DO ik = d%NKTB , d%NKTE
315 prs(:,ik) = prs(:,ik) + zw(:,ik)* &
316 (zwsedr(:,ik+d%NKL)-zwsedr(:,ik))/prhodref(:,ik)
317 pfpr(:,ik) = zwsedr(:,ik)
318 IF (kmoments==2) pcs(:,ik) = pcs(:,ik) + zw(:,ik)* &
319 (zwsedc(:,ik+d%NKL)-zwsedc(:,ik))/prhodref(:,ik)
320 ! Heat transport
321 !PRT_SUM(:,IK-D%NKL) = PRT_SUM(:,IK-D%NKL) + ZW(:,IK-D%NKL)*ZWSEDR(:,IK)/PRHODREF(:,IK-D%NKL)
322 !PRT_SUM(:,IK) = PRT_SUM(:,IK) - ZW(:,IK)*ZWSEDR(:,IK)/PRHODREF(:,IK)
323 !PCPT(:,IK-D%NKL) = PCPT(:,IK-D%NKL) + ZC * (ZW(:,IK-D%NKL)*ZWSEDR(:,IK)/PRHODREF(:,IK-D%NKL))
324 !PCPT(:,IK) = PCPT(:,IK) - ZC * (ZW(:,IK)*ZWSEDR(:,IK)/PRHODREF(:,IK))
325 !ZWDT(:,IK) =(PRHODREF(:,IK+D%NKL)*(1.+PRT_SUM(:,IK))*PCPT(:,IK)*PT(:,IK) + &
326 ! ZW(:,IK)*ZWSEDR(:,IK+1)*ZC*PT(:,IK+D%NKL)) / &
327 ! (PRHODREF(:,IK+D%NKL)*(1.+PRT_SUM(:,IK))*PCPT(:,IK) + ZW(:,IK)*ZWSEDR(:,IK+D%NKL)*ZC)
328 !ZWDT(:,IK) = ZWDT(:,IK) - PT(:,IK)
329 IF (oelec) pqs(:,ik) = pqs(:,ik) + zw(:,ik) * &
330 (zwsedq(:,ik+d%NKL) - zwsedq(:,ik)) / prhodref(:,ik)
331
332 END DO
333 !
334 DEALLOCATE(zrhodref)
335 DEALLOCATE(zpabst)
336 DEALLOCATE(zt)
337 DEALLOCATE(zrs)
338 DEALLOCATE(zcs)
339 DEALLOCATE(zcc)
340 DEALLOCATE(zlbda)
341 DEALLOCATE(zzw)
342 DEALLOCATE(zzx)
343 DEALLOCATE(zzy)
344 IF (ALLOCATED(zqs)) DEALLOCATE(zqs)
345 IF (ALLOCATED(zzq)) DEALLOCATE(zzq)
346 IF (ALLOCATED(zes)) DEALLOCATE(zes)
347 !
348 pinpr(:) = pinpr(:) + zwsedr(:,d%NKB)/cst%XRHOLW/limap%NSPLITSED(kid) ! in m/s
349 !PT(:,:) = PT(:,:) + ZWDT(:,:)
350
351 END IF
352END DO
353!
354prs(:,:) = prs(:,:) / ptstep
355IF (kmoments==2) pcs(:,:) = pcs(:,:) / ptstep
356IF (oelec) pqs(:,:) = pqs(:,:) / ptstep
357!
358IF (lhook) CALL dr_hook('LIMA_SEDIMENTATION', 1, zhook_handle)
359END SUBROUTINE lima_sedimentation
