19 !
20 ! Purpose : update the parameter e_x in the relation q_x = e_x d**f_x
21 ! e_x = q_x/(N_x * M(f_x))
22 !
23 ! AUTHOR
24 ! ------
25 ! C. Barthe * LAERO *

27 ! MODIFICATIONS

31 !------------------------------------------------------------------
32 !
33 !* 0. DECLARATIONS
34 ! ------------
35 !
36 USE modd_elec_param, ONLY : xecmax, xermax, xeimax, xesmax, xegmax, xehmax, &
37  xfqupdc, xfqupdr, xfqupdi, xexfqupdi, xfqupds, xfqupdg, xfqupdh
38 USE modd_elec_descr, ONLY : xcxr, xfc, xfr, xfi, xfs, xfg, xfh
39 USE modd_rain_ice_descr_n, ONLY : xcxs_i=>xcxs, xcxg_i=>xcxg, xcxh_i=>xcxh
40 USE modd_param_lima_cold, ONLY : xcxs_l=>xcxs
41 USE modd_param_lima_mixed, ONLY : xcxg_l=>xcxg, xcxh_l=>xcxh, xalphah, xnuh
42 USE modd_param_lima, ONLY : xalphac, xalphar, xalphai, xalphas, xalphag, &
43  xnuc, xnur, xnui, xnus, xnug
44 !
45 USE modi_momg
46 !
47 IMPLICIT NONE
48 !
49 !* 0.1 Declaration of dummy arguments
50 !
51 INTEGER, INTENT(IN) :: KID ! nb of the hydrometeor category
52 INTEGER, INTENT(IN) :: KMOMENT ! nb of moments of the microphysics scheme
53 INTEGER, INTENT(IN) :: KSIZE
54 REAL, INTENT(IN) :: PDUM ! =1. if mixing ratio
55  ! =timestep if source
56 REAL, INTENT(IN) :: PRTMIN
57 REAL, DIMENSION(KSIZE), INTENT(IN) :: PRHO ! reference density
58 REAL, DIMENSION(KSIZE), INTENT(IN) :: PQX ! Electric charge
59 REAL, DIMENSION(KSIZE), INTENT(IN) :: PRX ! Mixing ratio
60 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PEX ! e coef of the q-D relation
61 REAL, DIMENSION(KSIZE), OPTIONAL, INTENT(IN) :: PLBDX ! Slope parameter of the distribution
62 REAL, DIMENSION(KSIZE), OPTIONAL, INTENT(IN) :: PCX ! Nb concentration
63 CHARACTER (LEN=4), INTENT(IN) :: HCLOUD ! Kind of microphysical scheme
64 !
65 !* 0.2 Declaration of local variables
66 !
67 REAL :: ZRTMIN, ZFX, ZCX, ZEXMAX, ZFQUPDX, ZALPHAX, ZNUX
68 !
69 !---------------------------------------------------------------------------------
70 !
71 !* 1. PRELIMINARIES
72 ! -------------
73 !
74 zrtmin = prtmin / pdum
75 pex(:) = 0.
76 !
77 IF (kid == 2) THEN ! parameters for cloud droplets
78  zfx = xfc
79  zexmax = xecmax
80  IF (hcloud(1:3) == 'ICE') zfqupdx = xfqupdc
81  IF (hcloud == 'LIMA') THEN
82  zalphax = xalphac
83  znux = xnuc
84  END IF
85 ELSE IF (kid == 3) THEN ! parameters for raindrops
86  zfx = xfr
87  zcx = xcxr
88  zexmax = xermax
89  IF (hcloud(1:3) == 'ICE') zfqupdx = xfqupdr
90  IF (hcloud == 'LIMA' .AND. kmoment == 2) THEN
91  zalphax = xalphar
92  znux = xnur
93  END IF
94 ELSE IF (kid == 4) THEN ! parameters for ice crystals
95  zfx = xfi
96  zexmax = xeimax
97  IF (hcloud(1:3) == 'ICE') zfqupdx = xfqupdi
98  IF (hcloud == 'LIMA' .AND. kmoment == 2) THEN
99  zalphax = xalphai
100  znux = xnui
101  END IF
102 ELSE IF (kid == 5) THEN ! parameters for snow/aggregates
103  zfx = xfs
104  zexmax = xesmax
105  zfqupdx = xfqupds
106  IF (hcloud == 'LIMA' .AND. kmoment == 2) THEN
107  zalphax = xalphas
108  znux = xnus
109  ELSE IF (hcloud == 'LIMA' .AND. kmoment == 1) THEN
110  zcx = xcxs_l
111  ELSE IF (hcloud(1:3) == 'ICE') THEN
112  zcx = xcxs_i
113  END IF
114 ELSE IF (kid == 6) THEN ! parameters for graupel
115  zfx = xfg
116  zexmax = xegmax
117  zfqupdx = xfqupdg
118  IF (hcloud == 'LIMA' .AND. kmoment == 2) THEN
119  zalphax = xalphag
120  znux = xnug
121  ELSE IF (hcloud == 'LIMA' .AND. kmoment == 1) THEN
122  zcx = xcxg_l
123  ELSE IF (hcloud(1:3) == 'ICE') THEN
124  zcx = xcxg_i
125  END IF
126 ELSE IF (kid == 7) THEN ! parameters for hail
127  zfx = xfh
128  zexmax = xehmax
129  zfqupdx = xfqupdh
130  IF (hcloud == 'LIMA' .AND. kmoment == 2) THEN
131  zalphax = xalphah
132  znux = xnuh
133  ELSE IF (hcloud == 'LIMA' .AND. kmoment == 1) THEN
134  zcx = xcxh_l
135  ELSE IF (hcloud(1:3) == 'ICE') THEN
136  zcx = xcxh_i
137  END IF
138 END IF
139 !
140 IF (hcloud == 'LIMA') THEN
141  IF (kid == 2) THEN
142  zalphax = xalphac
143  znux = xnuc
144  ELSE IF (kid == 3) THEN
145  zalphax = xalphar
146  znux = xnur
147  ELSE IF (kid == 4) THEN
148  zalphax = xalphai
149  znux = xnui
150  ELSE IF (kid == 5) THEN
151  zalphax = xalphas
152  znux = xnus
153  ELSE IF (kid == 6) THEN
154  zalphax = xalphag
155  znux = xnug
156  ELSE IF (kid == 7) THEN
157  zalphax = xalphah
158  znux = xnuh
159  END IF
160 END IF
161 !
162 !
163 !* 2. UPDATE E_x FOR 2-MOMENT SPECIES
164 ! -------------------------------
165 !
166 IF (kmoment == 2) THEN
167  WHERE (prx(:) > zrtmin .AND. pcx(:) > 0.0)
168  pex(:) = pdum * prho(:) * pqx(:) * plbdx(:)**zfx / (pcx(:) * momg(zalphax,znux,zfx))
169  ENDWHERE
170 !
171 !
172 !* 3. UPDATE E_x FOR 1-MOMENT SPECIES
173 ! -------------------------------
174 !
175 ELSE IF (kmoment == 1) THEN
176 !
177 !* 3.1 Special case of cloud droplets
178 !
179  IF (kid == 2) THEN
180  WHERE (prx(:) > zrtmin)
181  pex(:) = pdum * prho(:) * pqx(:) / zfqupdx
182  pex(:) = sign( min(abs(pex(:)), zexmax), pex(:))
183  ENDWHERE
184 !
185 !* 3.2 Special case of ice crystals
186 !
187  ELSE IF (kid == 4) THEN
188  WHERE (prx(:) > zrtmin .AND. pcx(:) > 0.0)
189  pex(:) = pdum * prho(:) * pqx(:) / &
190  ((pcx**(1 - xexfqupdi)) * zfqupdx * (prho(:) * &
191  pdum * prx(:))**xexfqupdi)
192  pex(:) = sign( min(abs(pex(:)), zexmax), pex(:))
193  ENDWHERE
194 !
195 !* 3.3 Computation for all other hydrometeors
196 !
197  ELSE
198  WHERE (prx(:) > zrtmin .AND. plbdx(:) > 0.)
199  pex(:) = pdum * prho(:) * pqx(:) / (zfqupdx * plbdx(:)**(zcx - zfx))
200  pex(:) = sign( min(abs(pex(:)), zexmax), pex(:))
201  ENDWHERE
202  END IF
203 END IF
204 !
205 END SUBROUTINE elec_compute_ex
206 END MODULE mode_elec_compute_ex
