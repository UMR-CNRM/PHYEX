8SUBROUTINE rmc01(D,CST,CSTURB,TURBN,PZZ,PDXX,PDYY,PDZZ,PDIRCOSZW,PSBL_DEPTH,PLMO,PLK,PLEPS)
9 USE yomhook , ONLY : lhook, dr_hook, jphook
10! ##############################################################
11!
12!!**** *RMC01* -
13!!
14!! PURPOSE
15!! -------
16!! This routine modifies the mixing and dissipative length near the SBL.
17!! (Redelsperger, Mahe and Carlotti, 2001)
18!!
19!!** METHOD
20!! ------
21!!
22!! EXTERNAL
23!! --------
24!!
25!! IMPLICIT ARGUMENTS
26!! ------------------
27!!
28!!
29!! REFERENCE
30!! ---------
31!!
32!! Book 2
33!!
34!! AUTHOR
35!! ------
36!!
37!! V. Masson - Meteo-France -
38!!
39!! MODIFICATIONS
40!! -------------
41!! Original 14/02/02
42!-------------------------------------------------------------------------------
43!
44!* 0. DECLARATIONS
45! ------------
46!
47USE modd_parameters, ONLY: xundef
48USE modd_cst, ONLY : cst_t
49USE modd_dimphyex, ONLY: dimphyex_t
50USE modd_cturb, ONLY: csturb_t
51USE modd_turb_n, ONLY: turb_t
52!
53USE mode_update_iiju_phy, ONLY: update_iiju_phy
54USE mode_sbl_phy, ONLY: businger_phim, businger_phie
55!
56USE mode_shuman_phy, ONLY: mzf_phy, myf_phy, mxf_phy
57!
58IMPLICIT NONE
59!
60!* 0.1 Declaration of arguments
61! ------------------------
62!
63TYPE(dimphyex_t), INTENT(IN) :: D
64TYPE(cst_t), INTENT(IN) :: CST
65TYPE(csturb_t), INTENT(IN) :: CSTURB
66TYPE(turb_t), INTENT(IN) :: TURBN
67REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PZZ ! altitude of flux points
68REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDXX ! width of grid mesh (X dir)
69REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDYY ! width of grid mesh (Y dir)
70REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDZZ ! width of vert. layers
71REAL, DIMENSION(D%NIJT), INTENT(IN) :: PDIRCOSZW ! Director Cosinus
72REAL, DIMENSION(D%NIJT), INTENT(IN) :: PSBL_DEPTH! SBL depth
73REAL, DIMENSION(D%NIJT), INTENT(IN) :: PLMO ! Monin Obuhkov length
74REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PLK ! Mixing length
75REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PLEPS ! Dissipative length
76!
77!* 0.2 Declaration of local variables
78! ------------------------------
79!
80INTEGER :: IKB,IKE ! first,last physical level
81INTEGER :: IKTB,IKTE,IKT,IKA,IKU,IKL ! start, end of k loops in physical domain
82INTEGER :: JK,JIJ ! loop counter
83INTEGER :: IIJB,IIJE
84!
85REAL, DIMENSION(D%NIJT,D%NKT) :: ZZZ ! height of mass
86 ! points above ground
87REAL, DIMENSION(D%NIJT,D%NKT) :: ZZ_O_LMO ! height / LMO
88REAL, DIMENSION(D%NIJT,D%NKT) :: ZGAM ! factor controling
89 ! transition betw.
90 ! SBL and free BL
91REAL, DIMENSION(D%NIJT,D%NKT) :: ZPHIM! MO function
92 ! for stress
93REAL, DIMENSION(D%NIJT,D%NKT) :: ZPHIE! MO function
94 ! for TKE
95REAL, DIMENSION(D%NIJT,D%NKT) :: ZZC ! alt. where turb. is isotr.
96 ! size
97REAL, DIMENSION(D%NIJT,D%NKT) :: ZL ! SBL length
98REAL, DIMENSION(D%NIJT,D%NKT) :: ZWORK1, ZWORK2
99REAL, DIMENSION(D%NIJT,D%NKT) :: ZDH ! hor. grid mesh
100!-------------------------------------------------------------------------------
101!
102!* 1. Initializations
103! ---------------
104!
105! horizontal boundaries
106REAL(KIND=jphook) :: zhook_handle
107IF (lhook) CALL dr_hook('RMC01',0,zhook_handle)
108iktb=d%NKTB
109ikte=d%NKTE
110ikb=d%NKB
111ike=d%NKE
112iijb=d%NIJB
113iije=d%NIJE
114ikt=d%NKT
115ika=d%NKA
116iku=d%NKU
117ikl=d%NKL
118!
119! altitude of mass points
120CALL mzf_phy(d,pzz,zzz)
121! replace by height of mass points
122DO jk=1,ikt
123 !$mnh_expand_array(JIJ=IIJB:IIJE)
124 zzz(iijb:iije,jk) = zzz(iijb:iije,jk) - pzz(iijb:iije,ikb)
125 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
126END DO
127! fill upper level with physical value
128!$mnh_expand_array(JIJ=IIJB:IIJE)
129zzz(iijb:iije,iku) = 2.*zzz(iijb:iije,iku-ikl) - zzz(iijb:iije,iku-2*ikl)
130!$mnh_end_expand_array(JIJ=IIJB:IIJE)
131!
132!-------------------------------------------------------------------------------
133!
134!* 2. MO quantities
135! -------------
136!
137! z/LMO
138DO jk=1,ikt
139 !$mnh_expand_where(JIJ=IIJB:IIJE)
140 WHERE (plmo(iijb:iije)==xundef)
141 zz_o_lmo(iijb:iije,jk)=0.
142 ELSEWHERE
143 zz_o_lmo(iijb:iije,jk)=zzz(iijb:iije,jk)*pdircoszw(iijb:iije)/plmo(iijb:iije)
144 END WHERE
145 !$mnh_end_expand_where(JIJ=IIJB:IIJE)
146END DO
147!$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
148zz_o_lmo(iijb:iije,1:ikt) = max(zz_o_lmo(iijb:iije,1:ikt),-10.)
149zz_o_lmo(iijb:iije,1:ikt) = min(zz_o_lmo(iijb:iije,1:ikt), 10.)
150!$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
151!
152!
153! MO function for stress
154CALL businger_phim(d,zz_o_lmo,zphim)
155!
156! MO function for TKE
157CALL businger_phie(d,csturb,zz_o_lmo,zphie)
158!
159!-------------------------------------------------------------------------------
160SELECT CASE (turbn%CTURBLEN)
161!-------------------------------------------------------------------------------
162!
163!* 3. altitude where turbulence is isotropic inside a layer of given width (3D case)
164! --------------------------------------------------------------------
165!
166!
167!* LES subgrid mixing (unresolved eddies all below mesh size)
168! For stable cases, the vertical size of eddies is supposed to be given by the
169! same law as in the neutral case (i.e. with Phim = 1).
170!
171 CASE ('DELT','DEAR')
172 CALL mxf_phy(d,pdxx,zwork1)
173 CALL myf_phy(d,pdyy,zwork2)
174 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
175 zdh(iijb:iije,1:ikt) = sqrt(zwork1(iijb:iije,1:ikt)*zwork2(iijb:iije,1:ikt))
176 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
177 !
178 CALL update_iiju_phy(d,zzc)
179 !
180 DO jk=1,ikt
181 !$mnh_expand_array(JIJ=IIJB:IIJE)
182 zzc(iijb:iije,jk) = 2.*min(zphim(iijb:iije,jk),1.)/cst%XKARMAN &
183 * max( pdzz(iijb:iije,jk)*pdircoszw(iijb:iije) , &
184 zdh(iijb:iije,jk)/pdircoszw(iijb:iije)/3. )
185 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
186 END DO
187 !
188 !* 4. factor controling the transition between SBL and free isotropic turb. (3D case)
189 ! --------------------------------------------------------------------
190 !
191 zgam(iijb:iije,ika) = 0.
192 DO jk=iktb,ikte
193 !$mnh_expand_array(JIJ=IIJB:IIJE)
194 zgam(iijb:iije,jk) = 1. - exp( -3.*(zzz(iijb:iije,jk)-zzz(iijb:iije,ikb))/(zzc(iijb:iije,jk)) )
195 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
196 !$mnh_expand_where(JIJ=IIJB:IIJE)
197 WHERE (zgam(iijb:iije,jk-ikl)>zgam(iijb:iije,jk) .OR. zgam(iijb:iije,jk-ikl)>0.99 )
198 zgam(iijb:iije,jk) = 1.
199 END WHERE
200 !$mnh_end_expand_where(JIJ=IIJB:IIJE)
201 END DO
202 !$mnh_expand_array(JIJ=IIJB:IIJE)
203 zgam(iijb:iije,iku) = 1. - exp( -3.*(zzz(iijb:iije,iku)-zzz(iijb:iije,ikb))&
204 /(zzc(iijb:iije,iku)) )
205 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
206 !$mnh_expand_where(JIJ=IIJB:IIJE)
207 WHERE (zgam(iijb:iije,iku-ikl)>zgam(iijb:iije,iku) .OR. zgam(iijb:iije,iku-ikl)>0.99 )
208 zgam(iijb:iije,iku) = 1.
209 END WHERE
210 !$mnh_end_expand_where(JIJ=IIJB:IIJE)
211 !
212 !
213 !-------------------------------------------------------------------------------
214 !
215 !* 5. factor controling the transition between SBL and free isotropic turb.(1D case)
216 ! --------------------------------------------------------------------
217 !
218 CASE DEFAULT
219 !* SBL depth is used
220 zgam(iijb:iije,1:ikt) = 1.
221 zgam(iijb:iije,ika) = 0.
222 DO jk=iktb,ikte
223 !$mnh_expand_where(JIJ=IIJB:IIJE)
224 WHERE(psbl_depth(iijb:iije)>0.)
225 zgam(iijb:iije,jk) = tanh( (zzz(iijb:iije,jk)-zzz(iijb:iije,ikb))/psbl_depth(iijb:iije) )
226 END WHERE
227 !$mnh_end_expand_where(JIJ=IIJB:IIJE)
228 !$mnh_expand_where(JIJ=IIJB:IIJE)
229 WHERE (zgam(iijb:iije,jk-ikl)>0.99 )
230 zgam(iijb:iije,jk) = 1.
231 END WHERE
232 !$mnh_end_expand_where(JIJ=IIJB:IIJE)
233 END DO
234 !$mnh_expand_where(JIJ=IIJB:IIJE)
235 WHERE(psbl_depth(iijb:iije)>0.)
236 zgam(iijb:iije,iku) = tanh( (zzz(iijb:iije,iku)-zzz(iijb:iije,ikb))/psbl_depth(iijb:iije) )
237 END WHERE
238 !$mnh_end_expand_where(JIJ=IIJB:IIJE)
239 !$mnh_expand_where(JIJ=IIJB:IIJE)
240 WHERE (zgam(iijb:iije,iku-ikl)>0.99 )
241 zgam(iijb:iije,jk) = 1.
242 END WHERE
243 !$mnh_end_expand_where(JIJ=IIJB:IIJE)
244 !
245 !-------------------------------------------------------------------------------
246END SELECT
247!-------------------------------------------------------------------------------
248!
249!* 6. Modification of the mixing length
250! ---------------------------------
251!
252DO jk=1,ikt
253 !$mnh_expand_array(JIJ=IIJB:IIJE)
254 zl(iijb:iije,jk) = cst%XKARMAN/sqrt(csturb%XALPSBL)/csturb%XCMFS &
255 * zzz(iijb:iije,jk)*pdircoszw(iijb:iije)/(zphim(iijb:iije,jk)**2*sqrt(zphie(iijb:iije,jk)))
256 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
257END DO
258!
259!$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
260plk(iijb:iije,1:ikt)=(1.-zgam(iijb:iije,1:ikt))*zl(iijb:iije,1:ikt) &
261 +zgam(iijb:iije,1:ikt)*plk(iijb:iije,1:ikt)
262!$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
263!
264plk(iijb:iije,ika) = plk(iijb:iije,ikb)
265plk(iijb:iije,iku) = plk(iijb:iije,ike)
266!-------------------------------------------------------------------------------
267!
268!* 7. Modification of the dissipative length
269! --------------------------------------
270!
271!$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
272zl(iijb:iije,1:ikt) = zl(iijb:iije,1:ikt) * (csturb%XALPSBL**(3./2.)*cst%XKARMAN*turbn%XCED) &
273 / (cst%XKARMAN/sqrt(csturb%XALPSBL)/csturb%XCMFS)
274!$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
275!
276!$mnh_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
277WHERE (zz_o_lmo(iijb:iije,1:ikt)<0.)
278 zl(iijb:iije,1:ikt) = zl(iijb:iije,1:ikt)/(1.-1.9*zz_o_lmo(iijb:iije,1:ikt))
279ELSEWHERE
280 zl(iijb:iije,1:ikt) = zl(iijb:iije,1:ikt)/(1.-0.3*sqrt(zz_o_lmo(iijb:iije,1:ikt)))
281END WHERE
282!$mnh_end_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
283!
284!$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
285pleps(iijb:iije,1:ikt)=(1.-zgam(iijb:iije,1:ikt))*zl(iijb:iije,1:ikt) &
286 +zgam(iijb:iije,1:ikt)*pleps(iijb:iije,1:ikt)
287!$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
288!
289pleps(iijb:iije,ika) = pleps(iijb:iije,ikb)
290pleps(iijb:iije,iku) = pleps(iijb:iije,ike)
291!-------------------------------------------------------------------------------
292!
293IF (lhook) CALL dr_hook('RMC01',1,zhook_handle)
294END SUBROUTINE rmc01
295END MODULE mode_rmc01
