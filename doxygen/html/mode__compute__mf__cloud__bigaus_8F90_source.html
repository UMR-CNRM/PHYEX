19!!
20!!**** *COMPUTE_MF_CLOUD_BIGAUS* -
21!! compute diagnostic subgrid cumulus cloud caracteristics with a statistical scheme
22!! based on a bi-gaussian PDF. In this routine, we only compute the shallow convection
23!! part of this bi-gaussian
24!!
25!! PURPOSE
26!! -------
27!!**** With this option, a formulation for the computation of the variance of the departure
28!! to saturation is proposed. This variance is used to compute the cloud fraction and
29!! the mean convective cloud content from the bi-gaussian PDF
30!!
31!
32!!** METHOD
33!! ------
34!! Updraft variables are used to diagnose the variance
35!!
36!! EXTERNAL
37!! --------
38!!
39!! IMPLICIT ARGUMENTS
40!! ------------------
41!!
42!! REFERENCE
43!! ---------
44!!
45!!
46!! AUTHOR
47!! ------
48!! S. Riette moving of code previously in compute_mf_cloud code
49!!
50!! MODIFICATIONS
51!! -------------
52!! Original 25 Aug 2011
53!! S. Riette Jan 2012: support for both order of vertical levels
54!! S. Riette Jun 2019: remove unused PRC_UP and PRI_UP, use SIGN in ERFC computation
55!! A. Marcel Jun 2025: new bi-gaussian cloud scheme implementation
56!! A. Marcel Jan 2025: bi-Gaussian PDF and associated subgrid precipitation
57!! A. Marcel Jan 2025: relaxation of the small fraction assumption
58!! Jan 2025: protection against too small values
59!! --------------------------------------------------------------------------

61!* 0. DECLARATIONS
62! ------------
63USE modd_dimphyex, ONLY: dimphyex_t
64USE modd_cst, ONLY: cst_t
65USE modd_param_mfshall_n, ONLY: param_mfshall_t
66USE modd_rain_ice_param_n, ONLY: rain_ice_param_t
67!
68USE modi_shuman_mf, ONLY: mzf_mf, gz_m_w_mf
69!
70USE yomhook , ONLY : lhook, dr_hook, jphook

72IMPLICIT NONE
73!
74!* 0.1 Declaration of Arguments
75!
76TYPE(dimphyex_t), INTENT(IN) :: D
77TYPE(cst_t), INTENT(IN) :: CST
78TYPE(param_mfshall_t), INTENT(IN) :: PARAMMF
79TYPE(rain_ice_param_t), INTENT(IN) :: ICEP
80INTEGER, INTENT(IN) :: KRR ! number of moist var.
81REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRT_UP, PRV_UP, PRC_UP, PRI_UP, PTH_UP, PFRAC_UP ! updraft characteritics
82REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTHM, PRTM ! env. var. at t-dt
83REAL, DIMENSION(D%NIJT,D%NKT, KRR), INTENT(IN) :: PRM ! hydrometeors
84REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODREF, PEXNM, PPABSM ! env density and pressure
85REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PRC_MF, PRI_MF ! cloud content
86REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PCF_MF ! and cloud fraction for MF scheme
87REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PSIGMF ! contribution to the env s variance
88REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PHLC_HCF, PHLC_HRC, PHLI_HCF, PHLI_HRI ! low/high cloud diagnostics
89REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PWEIGHT_MF_CLOUD ! weight coefficient for the mass-flux cloud
90!
91!* 0.1 Declaration of local variables
92!
93!
94INTEGER :: JK, JIJ ! loop index
95INTEGER :: IIJE, IIJB, IKT ! bounds
96REAL, DIMENSION(D%NIJT,D%NKT) :: ZRT_UP_M,&
97 & ZRV_UP_M, ZRC_UP_M, ZRI_UP_M,&
98 & ZTH_UP_M,&
99 & ZFRAC_UP_M
100REAL :: ZQ1, ZCPH, ZTEMP, ZTEMP_UP, ZLV, ZLS, ZLVS, ZQSL, ZAH, ZA, &
101 ZSBAR_MEAN, ZSBAR_ENV, ZSIGS_UP, ZPV, ZQSI, &
102 ZAUTC, ZGAUTC, ZGAUC, ZAUTI, ZGAUTI, ZGAUI, ZCRIAUTI, ZAUT, &
103 ZCOND, ZGAM, ZFRAC_ICE_UP_M, ZFRAC_ICE, ZSBAR_UP
104REAL(KIND=jphook) :: zhook_handle
105!
106IF (lhook) CALL dr_hook('COMPUTE_MF_CLOUD_BIGAUS',0,zhook_handle)
107!
108iije=d%NIJE
109iijb=d%NIJB
110ikt=d%NKT
111!
112!Computation is done on mass points
113!Interpolation on mass points
114CALL mzf_mf(d, prc_up(:,:), zrc_up_m(:,:))
115CALL mzf_mf(d, pri_up(:,:), zri_up_m(:,:))
116CALL mzf_mf(d, prt_up(:,:), zrt_up_m(:,:))
117CALL mzf_mf(d, prv_up(:,:), zrv_up_m(:,:))
118CALL mzf_mf(d, pfrac_up(:,:), zfrac_up_m(:,:))
119CALL mzf_mf(d, pth_up(:,:), zth_up_m(:,:))
120pweight_mf_cloud(:,:)=0.
121
122DO jk=1, ikt
123 DO jij=iijb, iije
124 IF (zfrac_up_m(jij, jk)>0.) THEN
125 ! Ice franction in environment and in updraft
126 zfrac_ice_up_m=0.
127 IF (zrc_up_m(jij, jk)+zri_up_m(jij, jk) > 1.e-20) THEN
128 zfrac_ice_up_m = zri_up_m(jij, jk) / (zrc_up_m(jij, jk)+zri_up_m(jij, jk))
129 ENDIF
130 zfrac_ice = 0.
131 IF (krr > 3) THEN
132 IF (prm(jij, jk, 2)+prm(jij, jk, 4) > 1.e-20) THEN
133 zfrac_ice = prm(jij, jk, 4) / (prm(jij, jk, 2)+prm(jij, jk, 4))
134 ENDIF
135 ENDIF
136
137 !----------------------------------------------------------------------------
138 !
139 !* 1. Computation of the variance
140 ! ------------------------------------------------
141 !
142 !
143 !#### Computation of S in updraft
144 zcph = cst%XCPD + cst%XCPV*zrv_up_m(jij, jk) + cst%XCL*zrc_up_m(jij, jk) + cst%XCI*zri_up_m(jij, jk)
145 ztemp_up = zth_up_m(jij, jk)*pexnm(jij, jk)
146 zlv = cst%XLVTT + (cst%XCPV-cst%XCL) * (ztemp_up-cst%XTT)
147 zls = cst%XLSTT + (cst%XCPV-cst%XCI) * (ztemp_up-cst%XTT)
148
149 zpv = min(exp( cst%XALPW - cst%XBETAW / ztemp_up - cst%XGAMW * log(ztemp_up)), .99*ppabsm(jij, jk))
150 zqsl = cst%XRD / cst%XRV * zpv / ( ppabsm(jij, jk) - zpv )
151 zpv = min(exp( cst%XALPI - cst%XBETAI / ztemp_up - cst%XGAMI * log(ztemp_up)), .99*ppabsm(jij, jk))
152 zqsi = cst%XRD / cst%XRV * zpv / ( ppabsm(jij, jk) - zpv )
153
154 zqsl = (1. - zfrac_ice_up_m) * zqsl + zfrac_ice_up_m * zqsi
155 zlvs = (1. - zfrac_ice_up_m) * zlv + zfrac_ice_up_m * zls
156
157 zah = zlvs * zqsl / ( cst%XRV * ztemp_up**2 ) * (1. + (cst%XRV * zqsl / cst%XRD))
158 za = 1. / ( 1. + zlvs/zcph * zah )
159
160 zsbar_up = za * (zrt_up_m(jij, jk) - zqsl + zah * zlvs * (zrc_up_m(jij, jk)+zri_up_m(jij, jk)) / zcph)
161
162 !#### Computation of S on mean grid
163 IF (krr > 3) THEN
164 zcph = cst%XCPD + cst%XCPV*prm(jij, jk, 1) + cst%XCL*prm(jij, jk, 2) + cst%XCI*prm(jij, jk, 4)
165 ELSE
166 zcph = cst%XCPD + cst%XCPV*prm(jij, jk, 1) + cst%XCL*prm(jij, jk, 2)
167 ENDIF
168 ztemp = pthm(jij, jk)*pexnm(jij, jk)
169 zlv = cst%XLVTT + (cst%XCPV-cst%XCL) * (ztemp-cst%XTT)
170 zls = cst%XLSTT + (cst%XCPV-cst%XCI) * (ztemp-cst%XTT)
171
172 zpv = min(exp( cst%XALPW - cst%XBETAW / ztemp - cst%XGAMW * log(ztemp)), .99*ppabsm(jij, jk))
173 zqsl = cst%XRD / cst%XRV * zpv / ( ppabsm(jij, jk) - zpv )
174 zpv = min(exp( cst%XALPI - cst%XBETAI / ztemp - cst%XGAMI * log(ztemp)), .99*ppabsm(jij, jk))
175 zqsi = cst%XRD / cst%XRV * zpv / ( ppabsm(jij, jk) - zpv )
176
177 zqsl = (1. - zfrac_ice) * zqsl + zfrac_ice * zqsi
178 zlvs = (1. - zfrac_ice) * zlv + zfrac_ice * zls
179
180 zah = zlvs * zqsl / ( cst%XRV * ztemp**2 ) * (1. + (cst%XRV * zqsl / cst%XRD))
181 za = 1. / ( 1. + zlvs/zcph * zah )
182
183 IF (krr > 3) THEN
184 zsbar_mean = za * (prtm(jij, jk) - zqsl + zah * zlvs * (prm(jij, jk, 2)+prm(jij, jk, 4)) / zcph)
185 ELSE
186 zsbar_mean = za * (prtm(jij, jk) - zqsl + zah * zlvs * prm(jij, jk, 2) / zcph)
187 ENDIF
188
189 !#### Computation of S on environment
190 zsbar_env = (zsbar_mean-zfrac_up_m(jij, jk)*zsbar_up)/(1.-zfrac_up_m(jij, jk))
191
192 !#### Compute parametrized variances
193 ! Contribution to the environment internal variance
194 psigmf(jij, jk)=parammf%XSIGMA_ENV*(1. -zfrac_up_m(jij, jk))*(zsbar_env-zsbar_mean)**2
195 psigmf(jij, jk)=sqrt(abs(psigmf(jij, jk)))
196
197 ! Updraft internal variance
198 zsigs_up =parammf%XSIGMA_MF*zfrac_up_m(jij, jk)*(zsbar_up-zsbar_mean)**2
199 zsigs_up = sqrt(max(abs(zsigs_up), 1.e-40))
200
201 !* 2. PDF integration
202 ! ------------------------------------------------
203 !
204 pweight_mf_cloud(jij, jk) = zfrac_up_m(jij, jk)*parammf%XALPHA_MF ! XALPHA_MF = 1 --> EDMF :: XALPHA_MF > 1 --> PARAM
205 pweight_mf_cloud(jij, jk) = max(0., min(pweight_mf_cloud(jij, jk), 1.))
206
207 !The mean of the distribution is ZSBAR_UP
208 !Computation of ZA and ZGAM (=efrc(ZA)) coefficient
209
210 !Gaussian computation for S variable
211 zq1 = zsbar_up/zsigs_up
212 za = -zq1/sqrt(2.)
213
214 zgam = 1 + erf(-za)
215 zcond = (exp(-za**2)-za*sqrt(cst%XPI)*zgam)*zsigs_up/sqrt(2.*cst%XPI) * pweight_mf_cloud(jij, jk)
216
217 !Cloud fraction
218 pcf_mf(jij, jk) = max(0., min(1., 0.5*zgam * pweight_mf_cloud(jij, jk)))
219 !computation of condensate, then PRC and PRI
220 zcond = max(zcond, 0.)
221 IF(zcond < 1.e-12 .OR. pcf_mf(jij, jk) == 0.) THEN
222 pcf_mf(jij, jk)=0.
223 zcond=0.
224 ENDIF
225 prc_mf(jij, jk) = (1.-zfrac_ice_up_m) * zcond
226 pri_mf(jij, jk) = zfrac_ice_up_m * zcond
227
228 ! For low/high cloud diagnostics
229 IF(1-zfrac_ice_up_m > 1.e-20)THEN
230 zautc = (zsbar_up - icep%XCRIAUTC/(prhodref(jij, jk)*(1-zfrac_ice_up_m)))/zsigs_up
231 zgautc = -zautc/sqrt(2.)
232
233 zgauc = 1 + erf(-zgautc)
234 phlc_hrc(jij, jk) = (1-zfrac_ice_up_m)*(exp(-zgautc**2)-zgautc*sqrt(cst%XPI)*zgauc)*zsigs_up/sqrt(2.*cst%XPI) &
235 * pweight_mf_cloud(jij, jk)
236
237 phlc_hcf(jij, jk) = max( 0., min(1., 0.5*zgauc)) * pweight_mf_cloud(jij, jk)
238 phlc_hrc(jij, jk) = phlc_hrc(jij, jk) + icep%XCRIAUTC/prhodref(jij, jk) * phlc_hcf(jij, jk)
239 phlc_hrc(jij, jk) = max(phlc_hrc(jij, jk), 0.)
240 IF(phlc_hrc(jij,jk) < 1.e-12 .OR. phlc_hcf(jij,jk) < 1.e-6) THEN
241 phlc_hrc(jij,jk)=0.
242 phlc_hcf(jij,jk)=0.
243 ENDIF
244 ELSE
245 phlc_hcf(jij, jk)=0.
246 phlc_hrc(jij, jk)=0.
247 ENDIF
248
249 IF(zfrac_ice_up_m > 1.e-20)THEN
250 zcriauti = min(icep%XCRIAUTI, 10**(icep%XACRIAUTI*(ztemp_up-cst%XTT)+icep%XBCRIAUTI))
251 zauti = (zsbar_up - zcriauti/zfrac_ice_up_m)/zsigs_up
252 zgauti = -zauti/sqrt(2.)
253
254 zgaui = 1 + erf(-zgauti)
255 phli_hri(jij, jk) = zfrac_ice_up_m*(exp(-zgauti**2)-zgauti*sqrt(cst%XPI)*zgaui)*zsigs_up/sqrt(2.*cst%XPI) &
256 * pweight_mf_cloud(jij, jk)
257
258 phli_hcf(jij, jk) = max( 0., min(1., 0.5*zgaui)) * pweight_mf_cloud(jij, jk)
259 phli_hri(jij, jk) = phli_hri(jij, jk) + zcriauti*phli_hcf(jij, jk)
260 phli_hri(jij, jk) = max(phli_hri(jij, jk), 0.)
261 IF(phli_hri(jij,jk) < 1.e-12 .OR. phli_hcf(jij,jk) < 1.e-6) THEN
262 phli_hri(jij,jk)=0.
263 phli_hcf(jij,jk)=0.
264 ENDIF
265 ELSE
266 phli_hcf(jij, jk)=0.
267 phli_hri(jij, jk)=0.
268 ENDIF
269 ELSE
270 pcf_mf(jij, jk) = 0.
271 prc_mf(jij, jk) = 0.
272 pri_mf(jij, jk) = 0.
273 phlc_hcf(jij, jk) = 0.
274 phlc_hrc(jij, jk) = 0.
275 phli_hcf(jij, jk) = 0.
276 phli_hri(jij, jk) = 0.
277 ENDIF
278 ENDDO
279ENDDO
280
281IF(.NOT. parammf%LRELAX_ALPHA_MF) THEN
282 pweight_mf_cloud(:,:)=0.
283ENDIF
284!
285IF (lhook) CALL dr_hook('COMPUTE_MF_CLOUD_BIGAUS',1,zhook_handle)
286
287END SUBROUTINE compute_mf_cloud_bigaus
