15! ##########################################################################
16!
17!! PURPOSE
18!! -------
19!! The purpose of this routine is to compute the microphysical sources
20!! for slow cold processes :
21!! - conversion of snow to ice
22!! - deposition of vapor on snow
23!! - conversion of ice to snow (Harrington 1995)
24!!
25!!
26!! AUTHOR
27!! ------
28!! J.-M. Cohard * Laboratoire d'Aerologie*
29!! J.-P. Pinty * Laboratoire d'Aerologie*
30!! S. Berthet * Laboratoire d'Aerologie*
31!! B. Vié * CNRM *
32!!
33!! MODIFICATIONS
34!! -------------
35!! Original 15/03/2018
36!!
37! J. Wurtz 03/2022: new snow characteristics
38! B. Vie 03/2022: Add option for 1-moment pristine ice
39! M. Taufour 07/2022: add snow concentration
40!-------------------------------------------------------------------------------
41!
42!* 0. DECLARATIONS
43! ------------
44!
45USE modd_param_lima, ONLY : xrtmin, xctmin, xalphas, xnus, nmom_i
46USE modd_param_lima_cold, ONLY : xdscnvi_lim, xlbdascnvi_max, &
47 xc0depsi, xc1depsi, xr0depsi, xr1depsi, &

73!
74!* 0.2 Declarations of local variables :
75!
76LOGICAL, DIMENSION(SIZE(PRHODREF)) :: GMICRO ! Computations only where necessary
77REAL, DIMENSION(SIZE(PRHODREF)) :: ZZW, ZZW2, ZZX ! Work array
78!
79!-------------------------------------------------------------------------------
80!
81p_ri_cnvi(:) = 0.
82p_ci_cnvi(:) = 0.
83p_th_deps(:) = 0.
84p_rs_deps(:) = 0.
85!
86! Looking for regions where computations are necessary
87gmicro(:) = ldcompute(:) .AND. prst(:)>xrtmin(5)
88!
89IF (nmom_i.EQ.1) THEN
90 WHERE( gmicro )
91!
92! Deposition of water vapor on r_s: RVDEPS
93!
94 zzw(:) = 0.0
95 WHERE ( prst(:)>xrtmin(5) )
96 zzw(:) = pcst(:) * pssi(:) / pai(:) * &
97 ( x0deps*plbds(:)**xex0deps + &
98 x1deps*plbds(:)**xex1deps *pcj(:) * &
99 (1+0.5*(xfvelos/plbds(:))**xalphas)**(-xnus+xex1deps/xalphas) )
100 zzw(:) = zzw(:)*(0.5+sign(0.5,zzw(:))) - abs(zzw(:))*(0.5-sign(0.5,zzw(:)))
101 END WHERE
102 p_rs_deps(:) = zzw(:)
103 END WHERE
104ELSE
105 WHERE( gmicro )
106!
107!* 2.1 Conversion of snow to r_i: RSCNVI
108! ----------------------------------------
109!
110!
111 zzw2(:) = 0.0
112 zzw(:) = 0.0
113 WHERE ( plbds(:)<xlbdascnvi_max .AND. prst(:)>xrtmin(5) .AND. pcst(:)>xctmin(5) &
114 .AND. pssi(:)<0.0 )
115 zzw(:) = (plbds(:)*xdscnvi_lim)**(xalphas)
116 zzx(:) = ( -pssi(:)/pai(:) ) * pcst(:) * (zzw(:)**xnus) * exp(-zzw(:))
117!
118 zzw(:) = ( xr0depsi+xr1depsi*pcj(:) )*zzx(:)
119!
120 zzw2(:)= ( xc0depsi+xc1depsi*pcj(:) )*zzx(:)
121 END WHERE
122!
123 p_ri_cnvi(:) = zzw(:)
124 p_ci_cnvi(:) = zzw2(:)
125!
126!
127!* 2.2 Deposition of water vapor on r_s: RVDEPS
128! -----------------------------------------------
129!
130!
131 zzw(:) = 0.0
132 WHERE ( prst(:)>xrtmin(5) .AND. pcst(:)>xctmin(5) )
133 zzw(:) = ( pcst(:)*pssi(:)/pai(:) ) * &
134 ( x0deps*plbds(:)**xex0deps + &
135 ( x1deps*pcj(:)*plbds(:)**xex1deps * &
136 (1+0.5*(xfvelos/plbds(:))**xalphas)**(-xnus+xex1deps/xalphas)) )
137 zzw(:) = zzw(:)*(0.5+sign(0.5,zzw(:))) - abs(zzw(:))*(0.5-sign(0.5,zzw(:)))
138 END WHERE
139!
140 p_rs_deps(:) = zzw(:)
141!
142 END WHERE
143END IF
144!
145END SUBROUTINE lima_snow_deposition
