80 ! Compute log in advance for efficiency

82 ! ZFACTOR = f_c

88  zembryo(:) = zsubsat(:)*delta(1.,0.,zzt(:),xtx1(jspecie),xtx2(jspecie)) &
89  * xfrac_ref(jspecie)*zzy(:)/xarea1(jspecie)

95 ! ZZX(:) = ZZX(:) + ZEMBRYO(:) * CST%XPI * (XMDIAM_IFN(JSPECIE))**2 / 2.0 &

105  END IF
106  ENDDO

111 ! Beware : here, weights are normalized : XWEIGHT = wi/sqrt(pi)
112 !
113  ginteg(:) = zzt(:)<=(cst%XTT-35.) .AND. zsi(:)>1.0 .AND. zembryo(:)>1.0e-8
114 !
115  DO jl = 1, ndiam
116  DO jl2 = 1, SIZE(ginteg)
117  IF (ginteg(jl2)) THEN
118  zzx(jl2) = zzx(jl2) - xweight(jl)*exp(-zembryo(jl2)*cst%XPI*(xmdiam_ifn(jspecie))**2 &
119  * exp(2.0*sqrt(2.0)*log(xsigma_ifn(jspecie)) * xabsciss(jl)) )
120  END IF
121  ENDDO
122  ENDDO
123 !
124 ! DO JL2 = 1, SIZE(GINTEG)
125 ! IF (GINTEG(JL2)) THEN
126 ! ZZX(JL2) = ZZX(JL2) + 0.5* CST%XPI*ZEMBRYO(JL2)*(XMDIAM_IFN(JSPECIE))**2 &
127 ! * (1.0-( 1.0-GAMMA_INC(0.5,(SQRT(2.0)*LOG(XSIGMA_IFN(JSPECIE))-XB)**2)) &

129 ! END IF
130 ! ENDDO
131  DO jl2 = 1, SIZE(ginteg)
132  IF (ginteg(jl2)) THEN
133  zzx(jl2) = 1 + zzx(jl2) &
134  - ( 0.5* cst%XPI*zembryo(jl2)*(xmdiam_ifn(jspecie))**2 * exp( 2.0*(log(xsigma_ifn(jspecie)))**2) &
135  * ( 1.0-sign(1.,sqrt(2.0)*log(xsigma_ifn(jspecie))-xb)*gamma_inc(0.5,(sqrt(2.0)*log(xsigma_ifn(jspecie))-xb)**2)) )
136  END IF
137  ENDDO
138 !
139  z_frac_act(:,jspecie)=zzx(:)

147 ENDDO
