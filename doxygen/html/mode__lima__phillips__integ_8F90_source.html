8!! PURPOSE

11!! species (DM1, DM2, BC, O) that may be activated, following Phillips (2008)
12!!

15!! ---------
16!!
17!! Phillips et al., 2008: An empirical parameterization of heterogeneous
18!! ice nucleation for multiple chemical species of aerosols, J. Atmos. Sci.

21!! ------
22!! J.-M. Cohard * Laboratoire d'Aerologie*
23!! J.-P. Pinty * Laboratoire d'Aerologie*

28!! -------------
29!! Original ??/??/13

32!
33!* 0. DECLARATIONS
34! ------------
35!
36USE modd_cst, ONLY: cst_t

39USE yomhook, ONLY:lhook, dr_hook, jphook

41IMPLICIT NONE

43!* 0.1 Declarations of dummy arguments :
44!

46TYPE(cst_t), INTENT(IN) :: CST
47INTEGER, INTENT(IN) :: ISIZE

56!
57INTEGER :: ISPECIE, IL, IL2
58REAL :: ZB
59!
60REAL, DIMENSION(:), ALLOCATABLE :: ZZX, & ! Work array
61 ZFACTOR, &
62 ZSUBSAT, &
63 ZEMBRYO
64!
65REAL(KIND=jphook) :: zhook_handle
66LOGICAL, DIMENSION(:), ALLOCATABLE :: GINTEG ! Mask to integrate over the
67 ! AP size spectrum

69!
70!-------------------------------------------------------------------------------
71!

75DO ispecie = 1, limap%NSPECIE ! = 4 = {DM1, DM2, BC, O} respectively

77 ALLOCATE(zzx(SIZE(pzt)) ) ; zzx(:) = 0.0
78 ALLOCATE(zfactor(SIZE(pzt)) )
79 ALLOCATE(zsubsat(SIZE(pzt)) )
80 ALLOCATE(zembryo(SIZE(pzt)) )
81 ALLOCATE(ginteg(SIZE(pzt)) )
82
83! Compute log in advance for efficiency
84 zb = log(0.1e-6/limap%XMDIAM_IFN(ispecie))/(sqrt(2.0)*log(limap%XSIGMA_IFN(ispecie)))
85! ZFACTOR = f_c
86 zfactor(:) = delta(1.,limap%XH(ispecie),pzt(:),limap%XT0(ispecie),limap%XT0(ispecie)+limap%XDT0(ispecie)) &
87 * delta_vec(0.,1.,psi(:),psi0(:,ispecie),psi0(:,ispecie)+limap%XDSI0(ispecie)) / limap%XGAMMA
88! ZSUBSAT = H_X
89 zsubsat(:) = min(zfactor(:)+(1.0-zfactor(:))*delta(0.,1.,psw(:),limap%XSW0,1.) , 1.0)
90! ZEMBRYO = µ_X/(pi*(D_X)**2) = A
91 zembryo(:) = zsubsat(:)*delta(1.,0.,pzt(:),limap%XTX1(ispecie),limap%XTX2(ispecie)) &
92 * limap%XFRAC_REF(ispecie)*pzy(:)/limap%XAREA1(ispecie)
93!
94! For T warmer than -35°C, the integration is approximated with µ_X << 1
95! Error function : GAMMA_INC(1/2, x**2) = ERF(x) !!! for x>=0 !!!
96!
97! WHERE (PZT(:)>(CST%XTT-35.) .AND. ZEMBRYO(:)>1.0E-8)
98! ZZX(:) = ZZX(:) + ZEMBRYO(:) * CST%XPI * (LIMAP%XMDIAM_IFN(ISPECIE))**2 / 2.0 &
99! * EXP(2*(LOG(LIMAP%XSIGMA_IFN(ISPECIE)))**2) &
100! * (1.0+GAMMA_INC(0.5,(SQRT(2.0)*LOG(LIMAP%XSIGMA_IFN(ISPECIE))-ZB)**2))
101! END WHERE

106 * exp(2*(log(limap%XSIGMA_IFN(ispecie)))**2) &
107 * (1.0+sign(1.,sqrt(2.0)*log(limap%XSIGMA_IFN(ispecie))-zb)*limap%XGINC_IFN(ispecie))
108 END IF
109 ENDDO
110
111!
112! For other T, integration between 0 and infinity is made with a Gauss-Hermite
113! quadrature method and integration between 0 and 0.1 uses e(x) ~ 1+x+O(x**2)
114! Beware : here, weights are normalized : LIMAP%XWEIGHT = wi/sqrt(pi)
115!
116 ginteg(:) = pzt(:)<=(cst%XTT-35.) .AND. psi(:)>1.0 .AND. zembryo(:)>1.0e-8
117!
118 DO il = 1, limap%NDIAM
119 DO il2 = 1, SIZE(ginteg)
120 IF (ginteg(il2)) THEN
121 zzx(il2) = zzx(il2) - limap%XWEIGHT(il)*exp(-zembryo(il2)*cst%XPI*(limap%XMDIAM_IFN(ispecie))**2 &
122 * exp(2.0*sqrt(2.0)*log(limap%XSIGMA_IFN(ispecie)) * limap%XABSCISS(il)) )
123 END IF
124 ENDDO
125 ENDDO
126!
127! DO IL2 = 1, SIZE(GINTEG)
128! IF (GINTEG(IL2)) THEN
129! ZZX(IL2) = ZZX(IL2) + 0.5* CST%XPI*ZEMBRYO(IL2)*(LIMAP%XMDIAM_IFN(ISPECIE))**2 &
130! * (1.0-( 1.0-GAMMA_INC(0.5,(SQRT(2.0)*LOG(LIMAP%XSIGMA_IFN(ISPECIE))-ZB)**2)) &
131! * EXP( 2.0*(LOG(LIMAP%XSIGMA_IFN(ISPECIE)))**2) )
132! END IF
133! ENDDO
134 DO il2 = 1, SIZE(ginteg)
135 IF (ginteg(il2)) THEN
136 zzx(il2) = 1 + zzx(il2) &
137 - ( 0.5* cst%XPI*zembryo(il2)*(limap%XMDIAM_IFN(ispecie))**2 * exp( 2.0*(log(limap%XSIGMA_IFN(ispecie)))**2) &
138 * ( 1.0-sign(1.,sqrt(2.0)*log(limap%XSIGMA_IFN(ispecie))-zb)*limap%XGINC_IFN(ispecie)) )
139 END IF
140 ENDDO
141!
142 p_frac_act(:,ispecie)=zzx(:)
143!
144 DEALLOCATE(zzx)
145 DEALLOCATE(zfactor)
146 DEALLOCATE(zsubsat)
147 DEALLOCATE(zembryo)
148 DEALLOCATE(ginteg)
149!
150ENDDO
151!
152IF (lhook) CALL dr_hook('LIMA_PHILLIPS_INTEG', 1, zhook_handle)
