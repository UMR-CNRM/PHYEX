5 SUBROUTINE lima_phillips_integ (CST, ZZT, ZSI, ZSI0, ZSW, ZZY, Z_FRAC_ACT)
6! ######################################################################
7!!
8!! PURPOSE
9!! -------
10!! The purpose of this routine is to compute the fraction of each aerosol
11!! species (DM1, DM2, BC, O) that may be activated, following Phillips (2008)
12!!
13!!
14!! REFERENCE
15!! ---------
16!!
17!! Phillips et al., 2008: An empirical parameterization of heterogeneous
18!! ice nucleation for multiple chemical species of aerosols, J. Atmos. Sci.
19!!
20!! AUTHOR
21!! ------
22!! J.-M. Cohard * Laboratoire d'Aerologie*
23!! J.-P. Pinty * Laboratoire d'Aerologie*
24!! S. Berthet * Laboratoire d'Aerologie*
25!! B. Vié * Laboratoire d'Aerologie*
26!!
27!! MODIFICATIONS
28!! -------------
29!! Original ??/??/13
30!!
31!-------------------------------------------------------------------------------
32!
33!* 0. DECLARATIONS
34! ------------
35!
36USE modd_cst, ONLY: cst_t
37USE modd_param_lima, ONLY : xmdiam_ifn, xsigma_ifn, nspecie, xfrac_ref, &
38 xh, xarea1, xgamma, xabsciss, xweight, ndiam, &
39 xt0, xdt0, xdsi0, xsw0, xtx1, xtx2
40USE mode_lima_functions, ONLY : delta, delta_vec
41USE modi_gamma_inc
42!
43IMPLICIT NONE
44!
45!* 0.1 Declarations of dummy arguments :
46!
47TYPE(cst_t), INTENT(IN) :: CST
48REAL, DIMENSION(:), INTENT(IN) :: ZZT
49REAL, DIMENSION(:), INTENT(IN) :: ZSI
50REAL, DIMENSION(:,:), INTENT(IN) :: ZSI0
51REAL, DIMENSION(:), INTENT(IN) :: ZSW
52REAL, DIMENSION(:), INTENT(IN) :: ZZY
53REAL, DIMENSION(:,:), INTENT(INOUT) :: Z_FRAC_ACT
54!
55!* 0.2 Declarations of local variables :
56!
57INTEGER :: JSPECIE, JL, JL2
58REAL :: XB
59!
60REAL, DIMENSION(:), ALLOCATABLE :: ZZX, & ! Work array
61 ZFACTOR, &
62 ZSUBSAT, &
63 ZEMBRYO
64!
65LOGICAL, DIMENSION(:), ALLOCATABLE :: GINTEG ! Mask to integrate over the
66 ! AP size spectrum
67!
68!
69!-------------------------------------------------------------------------------
70!
71!
72DO jspecie = 1, nspecie ! = 4 = {DM1, DM2, BC, O} respectively
73!
74 ALLOCATE(zzx(SIZE(zzt)) ) ; zzx(:) = 0.0
75 ALLOCATE(zfactor(SIZE(zzt)) )
76 ALLOCATE(zsubsat(SIZE(zzt)) )
77 ALLOCATE(zembryo(SIZE(zzt)) )
78 ALLOCATE(ginteg(SIZE(zzt)) )
79
80! Compute log in advance for efficiency
81 xb = log(0.1e-6/xmdiam_ifn(jspecie))/(sqrt(2.0)*log(xsigma_ifn(jspecie)))
82! ZFACTOR = f_c
83 zfactor(:) = delta(1.,xh(jspecie),zzt(:),xt0(jspecie),xt0(jspecie)+xdt0(jspecie)) &
84 * delta_vec(0.,1.,zsi(:),zsi0(:,jspecie),zsi0(:,jspecie)+xdsi0(jspecie)) / xgamma
85! ZSUBSAT = H_X
86 zsubsat(:) = min(zfactor(:)+(1.0-zfactor(:))*delta(0.,1.,zsw(:),xsw0,1.) , 1.0)
87! ZEMBRYO = µ_X/(pi*(D_X)**2) = A
88 zembryo(:) = zsubsat(:)*delta(1.,0.,zzt(:),xtx1(jspecie),xtx2(jspecie)) &
89 * xfrac_ref(jspecie)*zzy(:)/xarea1(jspecie)
90!
91! For T warmer than -35°C, the integration is approximated with µ_X << 1
92! Error function : GAMMA_INC(1/2, x**2) = ERF(x) !!! for x>=0 !!!
93!
94! WHERE (ZZT(:)>(CST%XTT-35.) .AND. ZEMBRYO(:)>1.0E-8)
95! ZZX(:) = ZZX(:) + ZEMBRYO(:) * CST%XPI * (XMDIAM_IFN(JSPECIE))**2 / 2.0 &
96! * EXP(2*(LOG(XSIGMA_IFN(JSPECIE)))**2) &
97! * (1.0+GAMMA_INC(0.5,(SQRT(2.0)*LOG(XSIGMA_IFN(JSPECIE))-XB)**2))
98! END WHERE
99
100 DO jl = 1, SIZE(zzt)
101 IF (zzt(jl)>(cst%XTT-35.) .AND. zembryo(jl)>1.0e-8) THEN
102 zzx(jl) = zzx(jl) + zembryo(jl) * cst%XPI * (xmdiam_ifn(jspecie))**2 / 2.0 &
103 * exp(2*(log(xsigma_ifn(jspecie)))**2) &
104 * (1.0+sign(1.,sqrt(2.0)*log(xsigma_ifn(jspecie))-xb)*gamma_inc(0.5,(sqrt(2.0)*log(xsigma_ifn(jspecie))-xb)**2))
105 END IF
106 ENDDO
107
108!
109! For other T, integration between 0 and infinity is made with a Gauss-Hermite
110! quadrature method and integration between 0 and 0.1 uses e(x) ~ 1+x+O(x**2)
111! Beware : here, weights are normalized : XWEIGHT = wi/sqrt(pi)
112!
113 ginteg(:) = zzt(:)<=(cst%XTT-35.) .AND. zsi(:)>1.0 .AND. zembryo(:)>1.0e-8
114!
115 DO jl = 1, ndiam
116 DO jl2 = 1, SIZE(ginteg)
117 IF (ginteg(jl2)) THEN
118 zzx(jl2) = zzx(jl2) - xweight(jl)*exp(-zembryo(jl2)*cst%XPI*(xmdiam_ifn(jspecie))**2 &
119 * exp(2.0*sqrt(2.0)*log(xsigma_ifn(jspecie)) * xabsciss(jl)) )
120 END IF
121 ENDDO
122 ENDDO
123!
124! DO JL2 = 1, SIZE(GINTEG)
125! IF (GINTEG(JL2)) THEN
126! ZZX(JL2) = ZZX(JL2) + 0.5* CST%XPI*ZEMBRYO(JL2)*(XMDIAM_IFN(JSPECIE))**2 &
127! * (1.0-( 1.0-GAMMA_INC(0.5,(SQRT(2.0)*LOG(XSIGMA_IFN(JSPECIE))-XB)**2)) &
128! * EXP( 2.0*(LOG(XSIGMA_IFN(JSPECIE)))**2) )
129! END IF
130! ENDDO
131 DO jl2 = 1, SIZE(ginteg)
132 IF (ginteg(jl2)) THEN
133 zzx(jl2) = 1 + zzx(jl2) &
134 - ( 0.5* cst%XPI*zembryo(jl2)*(xmdiam_ifn(jspecie))**2 * exp( 2.0*(log(xsigma_ifn(jspecie)))**2) &
135 * ( 1.0-sign(1.,sqrt(2.0)*log(xsigma_ifn(jspecie))-xb)*gamma_inc(0.5,(sqrt(2.0)*log(xsigma_ifn(jspecie))-xb)**2)) )
136 END IF
137 ENDDO
138!
139 z_frac_act(:,jspecie)=zzx(:)
140!
141 DEALLOCATE(zzx)
142 DEALLOCATE(zfactor)
143 DEALLOCATE(zsubsat)
144 DEALLOCATE(zembryo)
145 DEALLOCATE(ginteg)
146!
147ENDDO
148!
149END SUBROUTINE lima_phillips_integ
