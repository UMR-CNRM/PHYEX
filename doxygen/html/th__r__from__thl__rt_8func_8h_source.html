62 TYPE(dimphyex_t), INTENT(IN) :: D
63 TYPE(cst_t), INTENT(IN) :: CST
64 TYPE(neb_t), INTENT(IN) :: NEBN
65 CHARACTER(LEN=1), INTENT(IN) :: HFRAC_ICE
66 LOGICAL, INTENT(IN) :: OOCEAN ! switch for Ocean model version
67 REAL, DIMENSION(D%NIJT), INTENT(INOUT) :: PFRAC_ICE
68 REAL, DIMENSION(D%NIJT), INTENT(IN) :: PP ! Pressure
69 REAL, DIMENSION(D%NIJT), INTENT(IN) :: PTHL ! thetal to transform into th
70 REAL, DIMENSION(D%NIJT),INTENT(IN) :: PRT ! Total mixing ratios to transform into rv,rc and ri
71 REAL, DIMENSION(D%NIJT),OPTIONAL,INTENT(IN) :: PRR, PRS, PRG, PRH
72 REAL, DIMENSION(D%NIJT), INTENT(OUT):: PTH ! th
73 REAL, DIMENSION(D%NIJT), INTENT(OUT):: PRV ! vapor mixing ratio
74 REAL, DIMENSION(D%NIJT), INTENT(INOUT):: PRL ! vapor mixing ratio
75 REAL, DIMENSION(D%NIJT), INTENT(INOUT):: PRI ! vapor mixing ratio
76 REAL, DIMENSION(D%NIJT), INTENT(OUT) :: PRSATW ! estimated mixing ration at saturation over water
77 REAL, DIMENSION(D%NIJT), INTENT(OUT) :: PRSATI ! estimated mixing ration at saturation over ice
78 REAL, DIMENSION(D%NIJT, 16), INTENT(OUT) :: PBUF ! buffer to replace automatic arrays
79 !
80 !-------------------------------------------------------------------------------
81 !
82 ! 0.2 declaration of local variables
83 INTEGER :: II ! Loop control
84 INTEGER :: JITER ! number of iterations
85 INTEGER :: JIJ, IIJB, IIJE
86 INTEGER, PARAMETER :: IEXN=1, irvsat=2, icph=3, irltemp=4, icph2=5, it=6, ilvocpexn=7, ilsocpexn=8, &
87  & idrsatodt=9, idrsatodtw=10, idrsatodti=11, ifoesw=12, ifoesi=13, &
88  & ilogt=14, i99pp=15, i1prt=16
89 REAL :: ZVAR1, ZVAR2, ZTPOW2, ZDELT
90 
91 !----------------------------------------------------------------------------
92 !
93 !* 1 Initialisation
94 ! --------------
95 !
96 !
97 !
98 iije=d%NIJE
99 iijb=d%NIJB
100 
101 !Number of iterations
102 jiter=2
103 !
104 !Computation of PBUF(IIJB:IIJE, ICPH2) depending on dummy arguments received
105 pbuf(iijb:iije, icph2)=0
106 IF(PRESENT(prr)) pbuf(iijb:iije, icph2)=pbuf(iijb:iije, icph2) + cst%XCL*prr(iijb:iije)
107 IF(PRESENT(prs)) pbuf(iijb:iije, icph2)=pbuf(iijb:iije, icph2) + cst%XCI*prs(iijb:iije)
108 IF(PRESENT(prg)) pbuf(iijb:iije, icph2)=pbuf(iijb:iije, icph2) + cst%XCI*prg(iijb:iije)
109 IF(PRESENT(prh)) pbuf(iijb:iije, icph2)=pbuf(iijb:iije, icph2) + cst%XCI*prh(iijb:iije)
110 !
111 !Computation of an approximate state thanks to PRL and PRI guess
112 pbuf(iijb:iije, iexn)=(pp(iijb:iije)/cst%XP00) ** cst%RDSCPD
113 
114 DO jij=iijb,iije
115  pbuf(jij, i99pp)=0.99*pp(jij)
116  prv(jij)=prt(jij)-prl(jij)-pri(jij)
117  pbuf(jij, icph)=cst%XCPD+ cst%XCPV * prv(jij)+ cst%XCL * prl(jij) + cst%XCI * pri(jij) + pbuf(jij, icph2)
118  zvar2=pbuf(jij, icph)*pbuf(jij, iexn)
119  zdelt=(pthl(jij)*pbuf(jij, iexn))-cst%XTT
120  pbuf(jij, ilvocpexn) = (cst%XLVTT + (cst%XCPV-cst%XCL) * zdelt) /zvar2
121  pbuf(jij, ilsocpexn) = (cst%XLSTT + (cst%XCPV-cst%XCI) * zdelt) /zvar2
122  pth(jij)=pthl(jij)+pbuf(jij, ilvocpexn)*prl(jij)+pbuf(jij, ilsocpexn)*pri(jij)
123  pbuf(jij, i1prt)=1+prt(jij)
124 ENDDO
125 !
126 !
127 ! 2 Iteration
128 ! ---------
129 
130 DO ii=1,jiter
131  IF (oocean) THEN
132  pbuf(iijb:iije, it)=pth(iijb:iije)
133  ELSE
134  pbuf(iijb:iije, it)=pth(iijb:iije)*pbuf(iijb:iije, iexn)
135  END IF
136  !Computation of liquid/ice fractions
137  pfrac_ice(iijb:iije) = 0.
138  DO jij=iijb, iije
139  IF(prl(jij)+pri(jij) > 1.e-20) THEN
140  pfrac_ice(jij) = pri(jij) / (prl(jij)+pri(jij))
141  ENDIF
142  ENDDO
143  CALL compute_frac_ice(cst, hfrac_ice,nebn,pfrac_ice(iijb:iije),pbuf(iijb:iije, it))
144 
145  !Computation of Rvsat and dRsat/dT
146  !In this version QSAT, QSATI, DQSAT and DQASATI functions are not used
147  !due to performance issue
148 
149  ! Log does not vectorize on all compilers:
150  pbuf(iijb:iije, ilogt)=log(pbuf(iijb:iije, it))
151 
152  DO jij=iijb, iije
153  pbuf(jij, ifoesw) = min(exp( cst%XALPW - cst%XBETAW/pbuf(jij, it) - cst%XGAMW*pbuf(jij, ilogt) ), pbuf(jij, i99pp))
154  pbuf(jij, ifoesi) = min(exp( cst%XALPI - cst%XBETAI/pbuf(jij, it) - cst%XGAMI*pbuf(jij, ilogt) ), pbuf(jij, i99pp))
155  prsatw(jij) = cst%XRD/cst%XRV*pbuf(jij, ifoesw)/pp(jij) / (1.+(cst%XRD/cst%XRV-1.)*pbuf(jij, ifoesw)/pp(jij))
156  prsati(jij) = cst%XRD/cst%XRV*pbuf(jij, ifoesi)/pp(jij) / (1.+(cst%XRD/cst%XRV-1.)*pbuf(jij, ifoesi)/pp(jij))
157  ztpow2=pbuf(jij, it)**2
158  pbuf(jij, idrsatodtw) = prsatw(jij) / (1.+(cst%XRD/cst%XRV-1.)*pbuf(jij, ifoesw)/pp(jij) ) &
159  * (cst%XBETAW/ztpow2 - cst%XGAMW/pbuf(jij, it))*pbuf(jij, i1prt)
160  pbuf(jij, idrsatodti) = prsati(jij) / (1.+(cst%XRD/cst%XRV-1.)*pbuf(jij, ifoesi)/pp(jij) ) &
161  * (cst%XBETAI/ztpow2 - cst%XGAMI/pbuf(jij, it))*pbuf(jij, i1prt)
162  !PRSATW(JIJ) = QSAT(PBUF(JIJ, IT),PP(JIJ)) !qsatw
163  !PRSATI(JIJ) = QSATI(PBUF(JIJ, IT),PP(JIJ)) !qsati
164  !PBUF(JIJ, IDRSATODTW) = DQSAT(PBUF(JIJ, IT),PP(JIJ),PRSATW(JIJ))*PBUF(JIJ, I1PRT)
165  !PBUF(JIJ, IDRSATODTI) = DQSATI(PBUF(JIJ, IT),PP(JIJ),PRSATI(JIJ))*PBUF(JIJ, I1PRT)
166  prsatw(jij) = prsatw(jij)*pbuf(jij, i1prt)
167  prsati(jij) = prsati(jij)*pbuf(jij, i1prt)
168  pbuf(jij, irvsat) = prsatw(jij)*(1-pfrac_ice(jij)) + prsati(jij)*pfrac_ice(jij)
169  pbuf(jij, idrsatodt) = (pbuf(jij, idrsatodtw)*(1-pfrac_ice(jij))+ &
170  & pbuf(jij, idrsatodti)*pfrac_ice(jij))
171 
172  !Computation of new PRL, PRI and PRV
173  !Correction term applied to (PRV(JIJ)-PBUF(JIJ, IRVSAT)) is computed assuming that
174  !PBUF(JIJ, ILVOCPEXN), PBUF(JIJ, ILSOCPEXN) and PBUF(JIJ, ICPH) don't vary too much with T. It takes into account
175  !the variation (estimated linear) of Qsat with T
176  pbuf(jij, irltemp)=(prv(jij)-pbuf(jij, irvsat))/ &
177  &(1 + pbuf(jij, idrsatodt)*pbuf(jij, iexn)* &
178  & (pbuf(jij, ilvocpexn)*(1-pfrac_ice(jij))+pbuf(jij, ilsocpexn)*pfrac_ice(jij)))
179  pbuf(jij, irltemp)=min(max(-prl(jij)-pri(jij), pbuf(jij, irltemp)),prv(jij))
180  prv(jij)=prv(jij)-pbuf(jij, irltemp)
181  prl(jij)=prl(jij)+pri(jij)+pbuf(jij, irltemp)
182  pri(jij)=pfrac_ice(jij) * (prl(jij))
183  prl(jij)=(1-pfrac_ice(jij)) * (prt(jij) - prv(jij))
184 
185  !Computation of Cph (as defined in Meso-NH doc, equation 2.2, to be used with mixing ratios)
186  pbuf(jij, icph)=cst%XCPD+ cst%XCPV * prv(jij)+ cst%XCL * prl(jij) + cst%XCI * pri(jij) + pbuf(jij, icph2)
187 
188  !Computation of L/Cph/EXN, then new PTH
189  zvar2=pbuf(jij, icph)*pbuf(jij, iexn)
190  pbuf(jij, ilvocpexn) = (cst%XLVTT + (cst%XCPV-cst%XCL) * (pbuf(jij, it)-cst%XTT)) /zvar2
191  pbuf(jij, ilsocpexn) = (cst%XLSTT + (cst%XCPV-cst%XCI) * (pbuf(jij, it)-cst%XTT)) /zvar2
192  pth(jij)=pthl(jij)+pbuf(jij, ilvocpexn)*prl(jij)+pbuf(jij, ilsocpexn)*pri(jij)
193 
194  !Computation of estimated mixing ration at saturation
195  !To compute the adjustement a first order development was used
196  zvar1=pth(jij)*pbuf(jij, iexn)-pbuf(jij, it)
197  prsatw(jij)=prsatw(jij) + pbuf(jij, idrsatodtw)*zvar1
198  prsati(jij)=prsati(jij) + pbuf(jij, idrsatodti)*zvar1
199  ENDDO
200 ENDDO
201 
202 END SUBROUTINE th_r_from_thl_rt
