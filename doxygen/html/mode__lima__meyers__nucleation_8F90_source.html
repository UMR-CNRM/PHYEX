131!-------------------------------------------------------------------------------
132!
133!
134!* 1. PRELIMINARY COMPUTATIONS
135! ------------------------
136!
137IF (lhook) CALL dr_hook('LIMA_MEYERS_NUCLEATION', 0, zhook_handle)
138p_th_hind(:,:) = 0.
139p_ri_hind(:,:) = 0.
140p_ci_hind(:,:) = 0.
141p_th_hinc(:,:) = 0.
142p_rc_hinc(:,:) = 0.
143p_cc_hinc(:,:) = 0.
144!
145! Temperature
146!
147zt(:,:) = ptht(:,:) * ( ppabst(:,:)/cst%XP00 ) ** (cst%XRD/cst%XCPD)
148!
149! Saturation over ice
150!
151zw(:,:) = exp( cst%XALPI - cst%XBETAI/zt(:,:) - cst%XGAMI*log(zt(:,:) ) )
152zw(:,:) = prvt(:,:)*( ppabst(:,:)-zw(:,:) ) / ( (cst%XMV/cst%XMD) * zw(:,:) )
153!
154!
155!-------------------------------------------------------------------------------
156!
157! optimization by looking for locations where
158! the temperature is negative only !!!
159!
160gnegt(:,:) = .false.
161gnegt(d%NIJB:d%NIJE,d%NKB:d%NKE) = zt(d%NIJB:d%NIJE,d%NKB:d%NKE)<cst%XTT .AND. &
162 zw(d%NIJB:d%NIJE,d%NKB:d%NKE)>0.8
163inegt = countjv( gnegt(:,:),i1(:),i3(:))
164IF( inegt >= 1 ) THEN
165 ALLOCATE(zrvt(inegt))
166 ALLOCATE(zrct(inegt))
167 ALLOCATE(zrrt(inegt))
168 ALLOCATE(zrit(inegt))
169 ALLOCATE(zrst(inegt))
170 ALLOCATE(zrgt(inegt))
171!
172 ALLOCATE(ztht(inegt))
173!
174 ALLOCATE(zcct(inegt))
175 ALLOCATE(zint(inegt,1))
176 ALLOCATE(zcit(inegt))
177!
178 ALLOCATE(zrhodref(inegt))
179 ALLOCATE(zzt(inegt))
180 ALLOCATE(zpres(inegt))
181 ALLOCATE(zexnref(inegt))
182 DO il=1,inegt
183 zrvt(il) = prvt(i1(il),i3(il))
184 zrct(il) = prct(i1(il),i3(il))
185 zrrt(il) = prrt(i1(il),i3(il))
186 zrit(il) = prit(i1(il),i3(il))
187 zrst(il) = prst(i1(il),i3(il))
188 zrgt(il) = prgt(i1(il),i3(il))
189!
190 zcct(il) = pcct(i1(il),i3(il))
191!
192 ztht(il) = ptht(i1(il),i3(il))
193!
194 zcct(il) = pcct(i1(il),i3(il))
195 zcit(il) = pcit(i1(il),i3(il))
196!
197 zrhodref(il) = prhodref(i1(il),i3(il))
198 zzt(il) = zt(i1(il),i3(il))
199 zpres(il) = ppabst(i1(il),i3(il))
200 zexnref(il) = pexnref(i1(il),i3(il))
201 ENDDO
202 ALLOCATE(zzw(inegt))
203 ALLOCATE(zzx(inegt))
204 ALLOCATE(zzy(inegt))
205 ALLOCATE(zlsfact(inegt))
206 ALLOCATE(zlvfact(inegt))
207 ALLOCATE(zssi(inegt))
208 ALLOCATE(ztcelsius(inegt))
209!
210 zzw(:) = zexnref(:)*( cst%XCPD+cst%XCPV*zrvt(:)+cst%XCL*(zrct(:)+zrrt(:)) &
211 +cst%XCI*(zrit(:)+zrst(:)+zrgt(:)) )
212 ztcelsius(:) = max( zzt(:)-cst%XTT,-50.0 )
213 zlsfact(:) = (cst%XLSTT+(cst%XCPV-cst%XCI)*ztcelsius(:))/zzw(:) ! L_s/(Pi_ref*C_ph)
214 zlvfact(:) = (cst%XLVTT+(cst%XCPV-cst%XCL)*ztcelsius(:))/zzw(:) ! L_v/(Pi_ref*C_ph)
215!
216 zzw(:) = exp( cst%XALPI - cst%XBETAI/zzt(:) - cst%XGAMI*log(zzt(:)) ) ! es_i
217 zssi(:) = zrvt(:)*(zpres(:)-zzw(:))/((cst%XMV/cst%XMD)*zzw(:)) - 1.0
218 ! Supersaturation over ice
219!
220!---------------------------------------------------------------------------
221!
222!* compute the heterogeneous nucleation by deposition: RVHNDI
223!
224 DO il=1,inegt
225 zint(il,1) = pint(i1(il),i3(il),1)
226 END DO
227 zzw(:) = 0.0
228 zzx(:) = 0.0
229 zzy(:) = 0.0
230!
231 WHERE( zzt(:)<cst%XTT-5.0 .AND. zssi(:)>0.0 )
232 zzy(:) = limac%XNUC_DEP*exp( limac%XEXSI_DEP*100.*min(1.,zssi(:))+limac%XEX_DEP)/zrhodref(:)
233 zzx(:) = max( zzy(:)-zint(:,1) , 0.0 ) ! number of ice crystals formed at this time step #/kg
234 zzw(:) = min( limac%XMNU0*zzx(:) , zrvt(:) ) ! mass of ice formed at this time step (kg/kg)
235 END WHERE
236 !
237 p_ci_hind(:,:) = unpack( zzx(:), mask=gnegt(:,:), field=0. )
238 p_ri_hind(:,:) = unpack( zzw(:), mask=gnegt(:,:), field=0. )
239 p_th_hind(:,:) = unpack( zzw(:)*(zlsfact(:)-zlvfact(:)), mask=gnegt(:,:), field=0. )
240 ptht(:,:) = ptht(:,:) + p_th_hind(:,:)
241 prvt(:,:) = prvt(:,:) - p_ri_hind(:,:)
242 prit(:,:) = prit(:,:) + p_ri_hind(:,:)
243 pcit(:,:) = pcit(:,:) + p_ci_hind(:,:)
244 pint(:,:,1) = pint(:,:,1) + p_ci_hind(:,:)
245!
246!---------------------------------------------------------------------------
247!
248!* compute the heterogeneous nucleation by contact: RVHNCI
249!
250!
251 DO il=1,inegt
252 zint(il,1) = pint(i1(il),i3(il),1)
253 END DO
254 zzw(:) = 0.0
255 zzx(:) = 0.0
256 zzy(:) = 0.0
257!
258 WHERE( zzt(:)<cst%XTT-2.0 .AND. zcct(:)>limap%XCTMIN(2) .AND. zrct(:)>limap%XRTMIN(2) )
259 zzy(:) = min( limac%XNUC_CON * exp( limac%XEXTT_CON*ztcelsius(:)+limac%XEX_CON ) &
260 /zrhodref(:) , zcct(:) )
261 zzx(:) = max( zzy(:)-zint(:,1),0.0 )
262 zzw(:) = min( (zrct(:)/zcct(:))*zzx(:),zrct(:) )
263 END WHERE
264!
265 p_rc_hinc(:,:) = - unpack( zzw(:), mask=gnegt(:,:), field=0. )
266 p_cc_hinc(:,:) = - unpack( zzx(:), mask=gnegt(:,:), field=0. )
267 p_th_hinc(:,:) = unpack( zzw(:)*(zlsfact(:)-zlvfact(:)), mask=gnegt(:,:), field=0. )
268 ptht(:,:) = ptht(:,:) + p_th_hinc(:,:)
269 prct(:,:) = prct(:,:) + p_rc_hinc(:,:)
270 prit(:,:) = prit(:,:) - p_rc_hinc(:,:)
271 pcct(:,:) = pcct(:,:) + p_cc_hinc(:,:)
272 pcit(:,:) = pcit(:,:) - p_cc_hinc(:,:)
273 pint(:,:,1) = pint(:,:,1) - p_cc_hinc(:,:)
274!
275 DEALLOCATE(zrvt)
276 DEALLOCATE(zrct)
277 DEALLOCATE(zrrt)
278 DEALLOCATE(zrit)
279 DEALLOCATE(zrst)
280 DEALLOCATE(zrgt)
281!
282 DEALLOCATE(ztht)
283!
284 DEALLOCATE(zcct)
285 DEALLOCATE(zint)
286 DEALLOCATE(zcit)
287!
288 DEALLOCATE(zrhodref)
289 DEALLOCATE(zzt)
290 DEALLOCATE(ztcelsius)
291 DEALLOCATE(zpres)
292 DEALLOCATE(zexnref)
293 DEALLOCATE(zssi)
294 DEALLOCATE(zzw)
295 DEALLOCATE(zzx)
296 DEALLOCATE(zzy)
297 DEALLOCATE(zlsfact)
298 DEALLOCATE(zlvfact)
299!
300END IF
301!
302!-------------------------------------------------------------------------------
303!
304IF (lhook) CALL dr_hook('LIMA_MEYERS_NUCLEATION', 1, zhook_handle)
