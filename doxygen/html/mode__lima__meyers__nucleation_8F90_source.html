126 !-------------------------------------------------------------------------------
127 !
128 !
129 !* 1. PRELIMINARY COMPUTATIONS
130 ! ------------------------
131 !
132 p_th_hind(:,:,:) = 0.
133 p_ri_hind(:,:,:) = 0.
134 p_ci_hind(:,:,:) = 0.
135 p_th_hinc(:,:,:) = 0.
136 p_rc_hinc(:,:,:) = 0.
137 p_cc_hinc(:,:,:) = 0.
138 !
139 ! Physical domain
140 !
141 iib=1+jphext
142 iie=SIZE(ptht,1) - jphext
143 ijb=1+jphext
144 ije=SIZE(ptht,2) - jphext
145 ikb=1+jpvext
146 ike=SIZE(ptht,3) - jpvext
147 !
148 ! Temperature
149 !
150 zt(:,:,:) = ptht(:,:,:) * ( ppabst(:,:,:)/cst%XP00 ) ** (cst%XRD/cst%XCPD)
151 !
152 ! Saturation over ice
153 !
154 zw(:,:,:) = exp( cst%XALPI - cst%XBETAI/zt(:,:,:) - cst%XGAMI*alog(zt(:,:,:) ) )
155 zw(:,:,:) = prvt(:,:,:)*( ppabst(:,:,:)-zw(:,:,:) ) / ( (cst%XMV/cst%XMD) * zw(:,:,:) )
156 !
157 !
158 !-------------------------------------------------------------------------------
159 !
160 ! optimization by looking for locations where
161 ! the temperature is negative only !!!
162 !
163 gnegt(:,:,:) = .false.
164 gnegt(iib:iie,ijb:ije,ikb:ike) = zt(iib:iie,ijb:ije,ikb:ike)<cst%XTT .AND. &
165  zw(iib:iie,ijb:ije,ikb:ike)>0.8
166 inegt = countjv( gnegt(:,:,:),i1(:),i2(:),i3(:))
167 IF( inegt >= 1 ) THEN
168  ALLOCATE(zrvt(inegt))
169  ALLOCATE(zrct(inegt))
170  ALLOCATE(zrrt(inegt))
171  ALLOCATE(zrit(inegt))
172  ALLOCATE(zrst(inegt))
173  ALLOCATE(zrgt(inegt))
174 !
175  ALLOCATE(ztht(inegt))
176 !
177  ALLOCATE(zcct(inegt))
178  ALLOCATE(zint(inegt,1))
179  ALLOCATE(zcit(inegt))
180 !
181  ALLOCATE(zrhodref(inegt))
182  ALLOCATE(zzt(inegt))
183  ALLOCATE(zpres(inegt))
184  ALLOCATE(zexnref(inegt))
185  DO jl=1,inegt
186  zrvt(jl) = prvt(i1(jl),i2(jl),i3(jl))
187  zrct(jl) = prct(i1(jl),i2(jl),i3(jl))
188  zrrt(jl) = prrt(i1(jl),i2(jl),i3(jl))
189  zrit(jl) = prit(i1(jl),i2(jl),i3(jl))
190  zrst(jl) = prst(i1(jl),i2(jl),i3(jl))
191  zrgt(jl) = prgt(i1(jl),i2(jl),i3(jl))
192 !
193  zcct(jl) = pcct(i1(jl),i2(jl),i3(jl))
194 !
195  ztht(jl) = ptht(i1(jl),i2(jl),i3(jl))
196 !
197  zcct(jl) = pcct(i1(jl),i2(jl),i3(jl))
198  zcit(jl) = pcit(i1(jl),i2(jl),i3(jl))
199 !
200  zrhodref(jl) = prhodref(i1(jl),i2(jl),i3(jl))
201  zzt(jl) = zt(i1(jl),i2(jl),i3(jl))
202  zpres(jl) = ppabst(i1(jl),i2(jl),i3(jl))
203  zexnref(jl) = pexnref(i1(jl),i2(jl),i3(jl))
204  ENDDO
205  ALLOCATE(zzw(inegt))
206  ALLOCATE(zzx(inegt))
207  ALLOCATE(zzy(inegt))
208  ALLOCATE(zlsfact(inegt))
209  ALLOCATE(zlvfact(inegt))
210  ALLOCATE(zssi(inegt))
211  ALLOCATE(ztcelsius(inegt))
212 !
213  zzw(:) = zexnref(:)*( cst%XCPD+cst%XCPV*zrvt(:)+cst%XCL*(zrct(:)+zrrt(:)) &
214  +cst%XCI*(zrit(:)+zrst(:)+zrgt(:)) )
215  ztcelsius(:) = max( zzt(:)-cst%XTT,-50.0 )
216  zlsfact(:) = (cst%XLSTT+(cst%XCPV-cst%XCI)*ztcelsius(:))/zzw(:) ! L_s/(Pi_ref*C_ph)
217  zlvfact(:) = (cst%XLVTT+(cst%XCPV-cst%XCL)*ztcelsius(:))/zzw(:) ! L_v/(Pi_ref*C_ph)
218 !
219  zzw(:) = exp( cst%XALPI - cst%XBETAI/zzt(:) - cst%XGAMI*alog(zzt(:)) ) ! es_i
220  zssi(:) = zrvt(:)*(zpres(:)-zzw(:))/((cst%XMV/cst%XMD)*zzw(:)) - 1.0
221  ! Supersaturation over ice
222 !
223 !---------------------------------------------------------------------------
224 !
225 !* compute the heterogeneous nucleation by deposition: RVHNDI
226 !
227  DO jl=1,inegt
228  zint(jl,1) = pint(i1(jl),i2(jl),i3(jl),1)
229  END DO
230  zzw(:) = 0.0
231  zzx(:) = 0.0
232  zzy(:) = 0.0
233 !
234  WHERE( zzt(:)<cst%XTT-5.0 .AND. zssi(:)>0.0 )
235  zzy(:) = xnuc_dep*exp( xexsi_dep*100.*min(1.,zssi(:))+xex_dep)/zrhodref(:)
236  zzx(:) = max( zzy(:)-zint(:,1) , 0.0 ) ! number of ice crystals formed at this time step #/kg
237  zzw(:) = min( xmnu0*zzx(:) , zrvt(:) ) ! mass of ice formed at this time step (kg/kg)
238  END WHERE
239  !
240  p_ci_hind(:,:,:) = unpack( zzx(:), mask=gnegt(:,:,:), field=0. )
241  p_ri_hind(:,:,:) = unpack( zzw(:), mask=gnegt(:,:,:), field=0. )
242  p_th_hind(:,:,:) = unpack( zzw(:)*(zlsfact(:)-zlvfact(:)), mask=gnegt(:,:,:), field=0. )
243  ptht(:,:,:) = ptht(:,:,:) + p_th_hind(:,:,:)
244  prvt(:,:,:) = prvt(:,:,:) - p_ri_hind(:,:,:)
245  prit(:,:,:) = prit(:,:,:) + p_ri_hind(:,:,:)
246  pcit(:,:,:) = pcit(:,:,:) + p_ci_hind(:,:,:)
247  pint(:,:,:,1) = pint(:,:,:,1) + p_ci_hind(:,:,:)
248 !
249 !---------------------------------------------------------------------------
250 !
251 !* compute the heterogeneous nucleation by contact: RVHNCI
252 !
253 !
254  DO jl=1,inegt
255  zint(jl,1) = pint(i1(jl),i2(jl),i3(jl),1)
256  END DO
257  zzw(:) = 0.0
258  zzx(:) = 0.0
259  zzy(:) = 0.0
260 !
261  WHERE( zzt(:)<cst%XTT-2.0 .AND. zcct(:)>xctmin(2) .AND. zrct(:)>xrtmin(2) )
262  zzy(:) = min( xnuc_con * exp( xextt_con*ztcelsius(:)+xex_con ) &
263  /zrhodref(:) , zcct(:) )
264  zzx(:) = max( zzy(:)-zint(:,1),0.0 )
265  zzw(:) = min( (zrct(:)/zcct(:))*zzx(:),zrct(:) )
266  END WHERE
267 !
268  p_rc_hinc(:,:,:) = - unpack( zzw(:), mask=gnegt(:,:,:), field=0. )
269  p_cc_hinc(:,:,:) = - unpack( zzx(:), mask=gnegt(:,:,:), field=0. )
270  p_th_hinc(:,:,:) = unpack( zzw(:)*(zlsfact(:)-zlvfact(:)), mask=gnegt(:,:,:), field=0. )
271  ptht(:,:,:) = ptht(:,:,:) + p_th_hinc(:,:,:)
272  prct(:,:,:) = prct(:,:,:) + p_rc_hinc(:,:,:)
273  prit(:,:,:) = prit(:,:,:) - p_rc_hinc(:,:,:)
274  pcct(:,:,:) = pcct(:,:,:) + p_cc_hinc(:,:,:)
275  pcit(:,:,:) = pcit(:,:,:) - p_cc_hinc(:,:,:)
276  pint(:,:,:,1) = pint(:,:,:,1) - p_cc_hinc(:,:,:)
277 !

279  DEALLOCATE(zrct)

300  DEALLOCATE(zlsfact)
301  DEALLOCATE(zlvfact)
302 !
303 END IF
