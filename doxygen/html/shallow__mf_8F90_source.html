124REAL, INTENT(IN) :: PTSTEP ! Dynamical timestep
125REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PZZ ! Height of flux point
126REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDZZ ! Metric coefficients
127REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODJ ! dry density * Grid size
128REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODREF ! dry density of the reference state
129REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PPABSM ! Pressure at time t-1
130REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PEXNM ! Exner function at t-dt
131
132REAL, DIMENSION(D%NIJT), INTENT(IN) :: PSFTH,PSFRV ! normal surface fluxes of theta and Rv
133REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTHM ! Theta at t-dt
134REAL, DIMENSION(D%NIJT,D%NKT,KRR), INTENT(IN) :: PRM ! water var. at t-dt
135REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PUM,PVM ! wind components at t-dt
136REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTKEM ! tke at t-dt
137REAL, DIMENSION(D%NIJT,D%NKT,KSV), INTENT(IN) :: PSVM ! scalar variable a t-dt
138
139REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PDUDT_MF ! tendency of U by massflux scheme
140REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PDVDT_MF ! tendency of V by massflux scheme
141REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PDTKEDT_MF ! tendency of TKE by massflux scheme
142REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PDTHLDT_MF ! tendency of thl by massflux scheme
143REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PDRTDT_MF ! tendency of rt by massflux scheme
144REAL, DIMENSION(D%NIJT,D%NKT,KSV), INTENT(OUT):: PDSVDT_MF ! tendency of Sv by massflux scheme
145
146REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PSIGMF,PRC_MF,PRI_MF,PCF_MF ! cloud info for the cloud scheme
147REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PHLC_HRC, PHLC_HCF, PHLI_HRI, PHLI_HCF ! low/high cloud diagnostics
148REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PWEIGHT_MF_CLOUD ! weight coefficient for the mass-flux cloud
149REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZTHVMF ! Thermal production for TKE scheme
150REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZTHMF
151REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZRMF
152REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZUMF
153REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZVMF
154REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZTKEMF
155REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PTHL_UP ! Thl updraft characteristics
156REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRT_UP ! Rt updraft characteristics
157REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRV_UP ! Vapor updraft characteristics
158REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PU_UP ! U wind updraft characteristics
159REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PV_UP ! V wind updraft characteristics
160REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PTKE_UP ! TKE updraft characteristics
161REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRC_UP ! cloud content updraft characteristics
162REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRI_UP ! ice content updraft characteristics
163REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PTHV_UP ! Thv updraft characteristics
164REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PW_UP ! vertical speed updraft characteristics
165REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PFRAC_UP ! updraft fraction
166REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PEMF ! updraft mass flux
167REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PDETR ! updraft detrainment
168REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PENTR ! updraft entrainment
169INTEGER,DIMENSION(D%NIJT), INTENT(OUT) :: KKLCL,KKETL,KKCTL ! level of LCL,ETL and CTL
170REAL, INTENT(IN) :: PDX, PDY
171REAL, DIMENSION(D%NIJT,D%NKT,KSV), INTENT(IN), OPTIONAL :: PRSVS ! sources of sv (for Budgets with lagrangian tracer)
172REAL,DIMENSION(JPSVMAX), INTENT(IN), OPTIONAL :: PSVMIN ! minimum value for SV variables (for Budgets)
173TYPE(tbudgetconf_t), INTENT(IN), OPTIONAL :: BUCONF ! budget structure
174TYPE(tbudgetdata), DIMENSION(KBUDGETS), INTENT(INOUT), OPTIONAL :: TBUDGETS
175INTEGER, INTENT(IN) :: KBUDGETS ! option. because not used in arpifs
176
177!
178! 0.2 Declaration of local variables
179!
180REAL, DIMENSION(D%NIJT,D%NKT) :: &
181 ZTHLM, & !
182 ZRTM, & !
183 ZTHVM, & !
184 ZWORK,ZWORK2, &
185 ZEMF_O_RHODREF, & ! entrainment/detrainment
186 ZBUO_INTEG ! integrated buoyancy
187REAL, DIMENSION(D%NIJT,D%NKT) :: ZFRAC_ICE
188REAL, DIMENSION(D%NIJT,D%NKT) :: ZWK
189
190REAL, DIMENSION(D%NIJT,D%NKT,KSV) :: &
191 ZSV_UP,& ! updraft scalar var.
192 ZFLXZSVMF ! Flux
193REAL, DIMENSION(D%NIJT) :: ZDEPTH ! Deepness of cloud
194REAL, DIMENSION(D%NIJT,D%NKT) :: ZFRAC_ICE_UP ! liquid/solid fraction in updraft
195REAL, DIMENSION(D%NIJT,D%NKT) :: ZRSAT_UP ! Rsat in updraft
196REAL, DIMENSION(D%NIJT,D%NKT) :: ZTH_UP
197
198LOGICAL :: GENTR_DETR ! flag to recompute entrainment, detrainment and mass flux
199INTEGER, DIMENSION(D%NIJT,D%NKT) :: IERR
200INTEGER :: JIJ, JK, JSV
201INTEGER :: IIJB,IIJE ! physical horizontal domain indices
202INTEGER :: IKT
203!
204REAL(KIND=jphook) :: zhook_handle
205!------------------------------------------------------------------------
206
207!!! 1. Initialisation
208IF (lhook) CALL dr_hook('SHALLOW_MF',0,zhook_handle)
209!
210iije=d%NIJE
211iijb=d%NIJB
212ikt=d%NKT
213!
214! updraft governing variables
215IF (parammf%CMF_UPDRAFT == 'EDKF' .OR. parammf%CMF_UPDRAFT == 'RHCJ') THEN
216 pentr(:,:) = 1.e20
217 pdetr(:,:) = 1.e20
218 pemf(:,:) = 1.e20
219 zbuo_integ(:,:) = 1.e20
220ENDIF
221
222! Thermodynamics functions
223zfrac_ice(:,:) = 0.
224IF (krr.GE.4) THEN
225 !$mnh_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
226 WHERE(prm(iijb:iije,1:ikt,2)+prm(iijb:iije,1:ikt,4) > 1.e-20)
227 zfrac_ice(iijb:iije,1:ikt) = prm(iijb:iije,1:ikt,4) / (prm(iijb:iije,1:ikt,2)+prm(iijb:iije,1:ikt,4))
228 ENDWHERE
229 !$mnh_end_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
230ENDIF
231!$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
232zwk(iijb:iije,1:ikt)=pthm(iijb:iije,1:ikt)*pexnm(iijb:iije,1:ikt)
233!$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
234CALL compute_frac_ice(cst, nebn%CFRAC_ICE_SHALLOW_MF,nebn,zfrac_ice(:,:),zwk(:,:), ierr(:,:))
235
236! Conservative variables at t-dt
237CALL thl_rt_from_th_r_mf(d, cst, krr,krrl,krri, &
238 pthm, prm, pexnm, &
239 zthlm, zrtm )
240
241! Virtual potential temperature at t-dt
242!$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
243zthvm(iijb:iije,1:ikt) = pthm(iijb:iije,1:ikt)*&
244 & ((1.+cst%XRV / cst%XRD *prm(iijb:iije,1:ikt,1))/(1.+zrtm(iijb:iije,1:ikt)))
245!$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
246!
247!!! 2. Compute updraft
248!!! ---------------
249!
250IF (parammf%CMF_UPDRAFT == 'EDKF') THEN
251 gentr_detr = .true.
252 CALL compute_updraft(d, cst, nebn, parammf, turbn, csturb, &
253 ksv, gentr_detr, &
254 onomixlg,ksv_lgbeg,ksv_lgend, &
255 pzz,pdzz, &
256 psfth,psfrv,pexnm, ppabsm,prhodref, &
257 pum,pvm,ptkem, &
258 pthm,prm(:,:,1),zthlm,zrtm,psvm, &
259 zth_up,pthl_up,prt_up,prv_up,prc_up,pri_up, &
260 pthv_up, pw_up, pu_up, pv_up, zsv_up, &
261 pfrac_up,zfrac_ice_up,zrsat_up,ptke_up,pemf,pdetr,&
262 pentr,zbuo_integ,kklcl,kketl,kkctl,zdepth,&
263 pdx,pdy)
264ELSEIF (parammf%CMF_UPDRAFT == 'RHCJ') THEN
265 gentr_detr = .true.
266 CALL compute_updraft_rhcj10(d, cst, nebn, parammf, turbn, csturb,&
267 ksv, gentr_detr, &
268 onomixlg,ksv_lgbeg,ksv_lgend, &
269 pzz,pdzz, &
270 psfth,psfrv,ppabsm,prhodref, &
271 pum,pvm,ptkem, &
272 pthm,prm(:,:,1),zthlm,zrtm,psvm, &
273 zth_up,pthl_up,prt_up,prv_up,prc_up,pri_up,&
274 pthv_up, pw_up, pu_up, pv_up, zsv_up, &
275 pfrac_up,zfrac_ice_up,zrsat_up,pemf,pdetr,&
276 pentr,zbuo_integ,kklcl,kketl,kkctl,zdepth )
277ELSEIF (parammf%CMF_UPDRAFT == 'RAHA') THEN
278 CALL compute_updraft_raha(d, cst, nebn, parammf, &
279 ksv, gentr_detr, &
280 onomixlg,ksv_lgbeg,ksv_lgend, &
281 pzz,pdzz, &
282 psfth,psfrv, &
283 ppabsm,prhodref,pum,pvm,ptkem, &
284 pexnm,pthm,prm(:,:,1),zthlm,zrtm, &
285 psvm,zth_up,pthl_up,prt_up, &
286 prv_up,prc_up,pri_up, pthv_up, &
287 pw_up, pu_up, pv_up, zsv_up, &
288 pfrac_up,zfrac_ice_up,zrsat_up, &
289 pemf,pdetr,pentr, &
290 zbuo_integ,kklcl,kketl,kkctl, &
291 zdepth )
292ELSEIF (parammf%CMF_UPDRAFT == 'DUAL') THEN
293 !Updraft characteristics are already computed and received by interface
294ELSE
295 CALL print_msg( nverb_fatal, 'GEN', 'SHALLOW_MF', 'no updraft model for EDKF: CMF_UPDRAFT='//parammf%CMF_UPDRAFT)
296ENDIF
297
298!!! 5. Compute diagnostic convective cloud fraction and content
299!!! --------------------------------------------------------
300!
301CALL compute_mf_cloud(d,cst,turbn,parammf,icep,nebn%LSTATNW, &
302 krr, krrl, krri, &
303 zfrac_ice, &
304 prv_up,prc_up,pri_up,pemf, &
305 pthl_up,prt_up,pfrac_up,zth_up, &
306 pexnm,zthlm,zrtm, &
307 pthm, prm, &
308 pdzz,kklcl, &
309 ppabsm,prhodref, &
310 prc_mf,pri_mf,pcf_mf,psigmf, &
311 phlc_hrc, phlc_hcf, phli_hri, phli_hcf,&
312 pweight_mf_cloud)
313
314!!! 3. Compute fluxes of conservative variables and their divergence = tendency
315!!! ------------------------------------------------------------------------
316!
317!$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
318zemf_o_rhodref(iijb:iije,1:ikt)=pemf(iijb:iije,1:ikt)/prhodref(iijb:iije,1:ikt)
319!$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
320
321IF ( parammf%XIMPL_MF > 1.e-10 ) THEN
322 CALL mf_turb(d, ksv, parammf, &
323 onomixlg,ksv_lgbeg,ksv_lgend, &
324 ptstep, pdzz, prhodj, &
325 zthlm,zthvm,zrtm,pum,pvm,psvm,ptkem, &
326 pdthldt_mf,pdrtdt_mf,pdudt_mf,pdvdt_mf,pdsvdt_mf,pdtkedt_mf, &
327 zemf_o_rhodref,pthl_up,pthv_up,prt_up,pu_up,pv_up,zsv_up,ptke_up,&
328 pflxzthmf,pflxzthvmf,pflxzrmf,pflxzumf,pflxzvmf, &
329 zflxzsvmf,pflxztkemf,pfrac_up )
330ELSE
331 CALL mf_turb_expl(d, parammf, &
332 prhodj,zthlm,zthvm,zrtm,pum,pvm,ptkem, &
333 pdthldt_mf,pdrtdt_mf,pdudt_mf,pdvdt_mf,pdtkedt_mf, &
334 zemf_o_rhodref,pthl_up,pthv_up,prt_up,pu_up,pv_up,ptke_up, &
335 pflxzthmf,pflxzthvmf,pflxzrmf,pflxzumf,pflxzvmf,pflxztkemf, &
336 pfrac_up)
337ENDIF
338
339! security in the case PARAMMF%CMF_UPDRAFT = 'DUAL'
340! to be modified if 'DUAL' is evolving (momentum mixing for example)
341IF( parammf%CMF_UPDRAFT == 'DUAL') THEN
342 ! Now thetav_up from vdfhghtnn is used!
343 pflxzthvmf(:,:)=0.
344 ! Yes/No UV mixing!
345! PDUDT_MF=0.
346! PDVDT_MF=0.
347ENDIF
348!
349IF(PRESENT(buconf)) THEN
350 IF( buconf%LBUDGET_U ) THEN
351 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
352 zwork(iijb:iije,1:ikt)=prhodj(iijb:iije,1:ikt)*pdudt_mf(iijb:iije,1:ikt)
353 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
354 CALL mxm_phy(d, zwork, zwork2)
355 CALL budget_store_add_phy(d, tbudgets(nbudget_u ), 'MAFL', zwork2)
356 END IF
357!
358 IF( buconf%LBUDGET_V ) THEN
359 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
360 zwork(iijb:iije,1:ikt)=prhodj(iijb:iije,1:ikt)*pdvdt_mf(iijb:iije,1:ikt)
361 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
362 CALL mym_phy(d, zwork, zwork2)
363 CALL budget_store_add_phy(d, tbudgets(nbudget_v ), 'MAFL', zwork2)
364 END IF
365!
366 IF( buconf%LBUDGET_TH ) THEN
367 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
368 zwork(iijb:iije,1:ikt)=prhodj(iijb:iije,1:ikt)*pdthldt_mf(iijb:iije,1:ikt)
369 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
370 CALL budget_store_add_phy(d, tbudgets(nbudget_th), 'MAFL', zwork)
371 END IF
372!
373 IF( buconf%LBUDGET_RV ) THEN
374 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
375 zwork(iijb:iije,1:ikt)=prhodj(iijb:iije,1:ikt)*pdrtdt_mf(iijb:iije,1:ikt)
376 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
377 CALL budget_store_add_phy(d, tbudgets(nbudget_rv), 'MAFL', zwork)
378 END IF
379!
380 IF( buconf%LBUDGET_SV ) THEN
381 DO jsv=1,ksv
382 IF (onomixlg .AND. jsv >= ksv_lgbeg .AND. jsv<= ksv_lgend) THEN
383 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
384 zwork(iijb:iije,1:ikt)=max(prsvs(iijb:iije,1:ikt,jsv) + prhodj(iijb:iije,1:ikt)* &
385 pdsvdt_mf(iijb:iije,1:ikt,jsv),psvmin(jsv))
386 zwork(iijb:iije,1:ikt)=prsvs(iijb:iije,1:ikt,jsv) - zwork(iijb:iije,1:ikt)
387 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
388 ELSE
