104 TYPE(cst_t), INTENT(IN) :: CST ! modd_cst general constant structure

113 LOGICAL, INTENT(IN) :: ONOMIXLG ! False if mixing of lagrangian tracer
114 INTEGER, INTENT(IN) :: KSV_LGBEG ! first index of lag. tracer
115 INTEGER, INTENT(IN) :: KSV_LGEND ! last index of lag. tracer
116 REAL, INTENT(IN) :: PTSTEP ! Dynamical timestep
117 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PZZ ! Height of flux point
118 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDZZ ! Metric coefficients
119 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODJ ! dry density * Grid size
120 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODREF ! dry density of the reference state
121 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PPABSM ! Pressure at time t-1
122 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PEXNM ! Exner function at t-dt
123 
124 REAL, DIMENSION(D%NIJT), INTENT(IN) :: PSFTH,PSFRV ! normal surface fluxes of theta and Rv
125 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTHM ! Theta at t-dt
126 REAL, DIMENSION(D%NIJT,D%NKT,KRR), INTENT(IN) :: PRM ! water var. at t-dt
127 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PUM,PVM ! wind components at t-dt
128 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTKEM ! tke at t-dt
129 REAL, DIMENSION(D%NIJT,D%NKT,KSV), INTENT(IN) :: PSVM ! scalar variable a t-dt
130 
131 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PDUDT_MF ! tendency of U by massflux scheme
132 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PDVDT_MF ! tendency of V by massflux scheme
133 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PDTHLDT_MF ! tendency of thl by massflux scheme
134 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PDRTDT_MF ! tendency of rt by massflux scheme
135 REAL, DIMENSION(D%NIJT,D%NKT,KSV), INTENT(OUT):: PDSVDT_MF ! tendency of Sv by massflux scheme
136 
137 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PSIGMF,PRC_MF,PRI_MF,PCF_MF ! cloud info for the cloud scheme
138 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZTHVMF ! Thermal production for TKE scheme
139 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZTHMF
140 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZRMF
141 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZUMF
142 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PFLXZVMF
143 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PTHL_UP ! Thl updraft characteristics
144 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRT_UP ! Rt updraft characteristics
145 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRV_UP ! Vapor updraft characteristics
146 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PU_UP ! U wind updraft characteristics
147 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PV_UP ! V wind updraft characteristics
148 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRC_UP ! cloud content updraft characteristics
149 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRI_UP ! ice content updraft characteristics
150 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PTHV_UP ! Thv updraft characteristics
151 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PW_UP ! vertical speed updraft characteristics
152 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PFRAC_UP ! updraft fraction
153 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PEMF ! updraft mass flux
154 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PDETR ! updraft detrainment
155 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PENTR ! updraft entrainment
156 INTEGER,DIMENSION(D%NIJT), INTENT(OUT) :: KKLCL,KKETL,KKCTL ! level of LCL,ETL and CTL
157 REAL, INTENT(IN) :: PDX, PDY
158 REAL, DIMENSION(D%NIJT,D%NKT,KSV), INTENT(IN), OPTIONAL :: PRSVS ! sources of sv (for Budgets with lagrangian tracer)
159 REAL,DIMENSION(JPSVMAX), INTENT(IN), OPTIONAL :: PSVMIN ! minimum value for SV variables (for Budgets)
160 TYPE(tbudgetconf_t), INTENT(IN), OPTIONAL :: BUCONF ! budget structure
161 TYPE(tbudgetdata), DIMENSION(KBUDGETS), INTENT(INOUT), OPTIONAL :: TBUDGETS
162 INTEGER, INTENT(IN) :: KBUDGETS ! option. because not used in arpifs
163 
164 !
165 ! 0.2 Declaration of local variables
166 !
167 REAL, DIMENSION(D%NIJT,D%NKT) :: &
168  ZTHLM, & !
169  ZRTM, & !
170  ZTHVM, & !
171  ZWORK,ZWORK2, &
172  ZEMF_O_RHODREF, & ! entrainment/detrainment
173  ZBUO_INTEG ! integrated buoyancy
174 REAL, DIMENSION(D%NIJT,D%NKT) :: ZFRAC_ICE
175 REAL, DIMENSION(D%NIJT,D%NKT) :: ZWK
176 
177 REAL, DIMENSION(D%NIJT,D%NKT,KSV) :: &
178  ZSV_UP,& ! updraft scalar var.
179  ZFLXZSVMF ! Flux
180 REAL, DIMENSION(D%NIJT) :: ZDEPTH ! Deepness of cloud
181 REAL, DIMENSION(D%NIJT,D%NKT) :: ZFRAC_ICE_UP ! liquid/solid fraction in updraft
182 REAL, DIMENSION(D%NIJT,D%NKT) :: ZRSAT_UP ! Rsat in updraft
183 
184 LOGICAL :: GENTR_DETR ! flag to recompute entrainment, detrainment and mass flux
185 INTEGER, DIMENSION(D%NIJT,D%NKT) :: IERR
186 INTEGER :: JIJ, JK, JSV
187 INTEGER :: IIJB,IIJE ! physical horizontal domain indices
188 INTEGER :: IKT
189 !
190 REAL(KIND=jphook) :: zhook_handle
191 !------------------------------------------------------------------------
192 
193 !!! 1. Initialisation
194 IF (lhook) CALL dr_hook('SHALLOW_MF',0,zhook_handle)
195 !
196 iije=d%NIJE
197 iijb=d%NIJB
198 ikt=d%NKT
199 !
200 ! updraft governing variables
201 IF (parammf%CMF_UPDRAFT == 'EDKF' .OR. parammf%CMF_UPDRAFT == 'RHCJ') THEN
202  pentr(:,:) = 1.e20
203  pdetr(:,:) = 1.e20
204  pemf(:,:) = 1.e20
205  zbuo_integ(:,:) = 1.e20
206 ENDIF
207 
208 ! Thermodynamics functions
209 zfrac_ice(:,:) = 0.
210 IF (krr.GE.4) THEN
211  !$mnh_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
212  WHERE(prm(iijb:iije,1:ikt,2)+prm(iijb:iije,1:ikt,4) > 1.e-20)
213  zfrac_ice(iijb:iije,1:ikt) = prm(iijb:iije,1:ikt,4) / (prm(iijb:iije,1:ikt,2)+prm(iijb:iije,1:ikt,4))
214  ENDWHERE
215  !$mnh_end_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
216 ENDIF
217 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
218 zwk(iijb:iije,1:ikt)=pthm(iijb:iije,1:ikt)*pexnm(iijb:iije,1:ikt)
219 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
220 CALL compute_frac_ice(cst, nebn%CFRAC_ICE_SHALLOW_MF,nebn,zfrac_ice(:,:),zwk(:,:), ierr(:,:))
221 
222 ! Conservative variables at t-dt
223 CALL thl_rt_from_th_r_mf(d, cst, krr,krrl,krri, &
224  pthm, prm, pexnm, &
225  zthlm, zrtm )
226 
227 ! Virtual potential temperature at t-dt
228 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
229 zthvm(iijb:iije,1:ikt) = pthm(iijb:iije,1:ikt)*&
230  & ((1.+cst%XRV / cst%XRD *prm(iijb:iije,1:ikt,1))/(1.+zrtm(iijb:iije,1:ikt)))
231 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
232 !
233 !!! 2. Compute updraft
234 !!! ---------------
235 !
236 IF (parammf%CMF_UPDRAFT == 'EDKF') THEN
237  gentr_detr = .true.
238  CALL compute_updraft(d, cst, nebn, parammf, turbn, csturb, &
239  ksv, gentr_detr, &
240  onomixlg,ksv_lgbeg,ksv_lgend, &
241  pzz,pdzz, &
242  psfth,psfrv,ppabsm,prhodref, &
243  pum,pvm,ptkem, &
244  pthm,prm(:,:,1),zthlm,zrtm,psvm, &
245  pthl_up,prt_up,prv_up,prc_up,pri_up, &
246  pthv_up, pw_up, pu_up, pv_up, zsv_up, &
247  pfrac_up,zfrac_ice_up,zrsat_up,pemf,pdetr,&
248  pentr,zbuo_integ,kklcl,kketl,kkctl,zdepth,&
249  pdx,pdy)
250 ELSEIF (parammf%CMF_UPDRAFT == 'RHCJ') THEN
251  gentr_detr = .true.
252  CALL compute_updraft_rhcj10(d, cst, nebn, parammf, turbn, csturb,&
253  ksv, gentr_detr, &
254  onomixlg,ksv_lgbeg,ksv_lgend, &
255  pzz,pdzz, &
256  psfth,psfrv,ppabsm,prhodref, &
257  pum,pvm,ptkem, &
258  pthm,prm(:,:,1),zthlm,zrtm,psvm, &
259  pthl_up,prt_up,prv_up,prc_up,pri_up, &
260  pthv_up, pw_up, pu_up, pv_up, zsv_up, &
261  pfrac_up,zfrac_ice_up,zrsat_up,pemf,pdetr,&
262  pentr,zbuo_integ,kklcl,kketl,kkctl,zdepth )
263 ELSEIF (parammf%CMF_UPDRAFT == 'RAHA') THEN
264  CALL compute_updraft_raha(d, cst, nebn, parammf, &
265  ksv, gentr_detr, &
266  onomixlg,ksv_lgbeg,ksv_lgend, &
267  pzz,pdzz, &
268  psfth,psfrv, &
269  ppabsm,prhodref,pum,pvm,ptkem, &
270  pexnm,pthm,prm(:,:,1),zthlm,zrtm, &
271  psvm,pthl_up,prt_up, &
272  prv_up,prc_up,pri_up, pthv_up, &
273  pw_up, pu_up, pv_up, zsv_up, &
274  pfrac_up,zfrac_ice_up,zrsat_up, &
275  pemf,pdetr,pentr, &
276  zbuo_integ,kklcl,kketl,kkctl, &
277  zdepth )
278 ELSEIF (parammf%CMF_UPDRAFT == 'DUAL') THEN
279  !Updraft characteristics are already computed and received by interface
280 ELSE
281  CALL print_msg( nverb_fatal, 'GEN', 'SHALLOW_MF', 'no updraft model for EDKF: CMF_UPDRAFT='//parammf%CMF_UPDRAFT)
282 ENDIF
283 
284 !!! 5. Compute diagnostic convective cloud fraction and content
285 !!! --------------------------------------------------------
286 !
287 CALL compute_mf_cloud(d,cst,turbn,parammf,nebn%LSTATNW, &
288  krr, krrl, krri, &
289  zfrac_ice, &
290  prc_up,pri_up,pemf, &
291  pthl_up,prt_up,pfrac_up, &
292  pthv_up,zfrac_ice_up, &
293  zrsat_up,pexnm,zthlm,zrtm, &
294  pthm, zthvm, prm, &
295  pdzz,pzz,kklcl, &
296  ppabsm,prhodref, &
297  prc_mf,pri_mf,pcf_mf,psigmf,zdepth)
298 
299 
300 !!! 3. Compute fluxes of conservative variables and their divergence = tendency
301 !!! ------------------------------------------------------------------------
302 !
303 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
304 zemf_o_rhodref(iijb:iije,1:ikt)=pemf(iijb:iije,1:ikt)/prhodref(iijb:iije,1:ikt)
305 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
306 
307 IF ( parammf%XIMPL_MF > 1.e-10 ) THEN
308  CALL mf_turb(d, ksv, parammf%LMIXUV, &
309  onomixlg,ksv_lgbeg,ksv_lgend, &
310  parammf%XIMPL_MF, ptstep, &
311  pdzz, &
312  prhodj, &
313  zthlm,zthvm,zrtm,pum,pvm,psvm, &
314  pdthldt_mf,pdrtdt_mf,pdudt_mf,pdvdt_mf,pdsvdt_mf, &
315  zemf_o_rhodref,pthl_up,pthv_up,prt_up,pu_up,pv_up,zsv_up,&
316  pflxzthmf,pflxzthvmf,pflxzrmf,pflxzumf,pflxzvmf, &
317  zflxzsvmf )
318 ELSE
319  CALL mf_turb_expl(d, parammf, &
320  prhodj,zthlm,zthvm,zrtm,pum,pvm, &
321  pdthldt_mf,pdrtdt_mf,pdudt_mf,pdvdt_mf, &
322  zemf_o_rhodref,pthl_up,pthv_up,prt_up,pu_up,pv_up, &
323  pflxzthmf,pflxzthvmf,pflxzrmf,pflxzumf,pflxzvmf)
324 ENDIF
325 
326 ! security in the case PARAMMF%CMF_UPDRAFT = 'DUAL'
327 ! to be modified if 'DUAL' is evolving (momentum mixing for example)
328 IF( parammf%CMF_UPDRAFT == 'DUAL') THEN
329  ! Now thetav_up from vdfhghtnn is used!
330  pflxzthvmf(:,:)=0.
331  ! Yes/No UV mixing!
332 ! PDUDT_MF=0.
333 ! PDVDT_MF=0.
334 ENDIF
335 !
336 IF(PRESENT(buconf)) THEN
337  IF( buconf%LBUDGET_U ) THEN
338  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
339  zwork(iijb:iije,1:ikt)=prhodj(iijb:iije,1:ikt)*pdudt_mf(iijb:iije,1:ikt)
340  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
341  CALL mxm_phy(d, zwork, zwork2)
342  CALL budget_store_add_phy(d, tbudgets(nbudget_u ), 'MAFL', zwork2)
343  END IF
344 !
345  IF( buconf%LBUDGET_V ) THEN
346  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
347  zwork(iijb:iije,1:ikt)=prhodj(iijb:iije,1:ikt)*pdvdt_mf(iijb:iije,1:ikt)
348  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
349  CALL mym_phy(d, zwork, zwork2)
350  CALL budget_store_add_phy(d, tbudgets(nbudget_v ), 'MAFL', zwork2)
351  END IF
352 !
353  IF( buconf%LBUDGET_TH ) THEN
354  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
355  zwork(iijb:iije,1:ikt)=prhodj(iijb:iije,1:ikt)*pdthldt_mf(iijb:iije,1:ikt)
356  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
357  CALL budget_store_add_phy(d, tbudgets(nbudget_th), 'MAFL', zwork)
358  END IF
359 !
360  IF( buconf%LBUDGET_RV ) THEN
361  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
362  zwork(iijb:iije,1:ikt)=prhodj(iijb:iije,1:ikt)*pdrtdt_mf(iijb:iije,1:ikt)
363  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
364  CALL budget_store_add_phy(d, tbudgets(nbudget_rv), 'MAFL', zwork)
365  END IF
366 !
367  IF( buconf%LBUDGET_SV ) THEN
368  DO jsv=1,ksv
369  IF (onomixlg .AND. jsv >= ksv_lgbeg .AND. jsv<= ksv_lgend) THEN
370  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
371  zwork(iijb:iije,1:ikt)=max(prsvs(iijb:iije,1:ikt,jsv) + prhodj(iijb:iije,1:ikt)* &
372  pdsvdt_mf(iijb:iije,1:ikt,jsv),psvmin(jsv))
373  zwork(iijb:iije,1:ikt)=prsvs(iijb:iije,1:ikt,jsv) - zwork(iijb:iije,1:ikt)
374  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
375  ELSE
376  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
377  zwork(iijb:iije,1:ikt)=prhodj(iijb:iije,1:ikt)*pdsvdt_mf(iijb:iije,1:ikt,jsv)
378  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
379  END IF
380  CALL budget_store_add_phy(d, tbudgets(nbudget_sv1 - 1 + jsv), 'MAFL', zwork )
381  END DO
382  END IF
383 END IF
384 !
385 IF (lhook) CALL dr_hook('SHALLOW_MF',1,zhook_handle)
386 !
387 CONTAINS
388 include "compute_frac_ice.func.h"
389 !
