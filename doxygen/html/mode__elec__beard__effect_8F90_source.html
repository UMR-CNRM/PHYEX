18!! PURPOSE
19!! -------
20!! The purpose of this routine is to compute the effect of the electric field
21!! on the terminal velocity of hydrometeors.
22!!
23!! METHOD
24!! ------
25!! From Beard, K. V., 1980: The Effects of Altitude and Electrical Force on
26!! the Terminal Velocity of Hydrometeors. J. Atmos. Sci., 37, 1363–1374,
27!! https://doi.org/10.1175/1520-0469(1980)037<1363:TEOAAE>2.0.CO;2.

29!! AUTHOR
30!! ------
31!! J.-P. Pinty * LAERO *
32!! C. Barthe * LAERO *

34!! MODIFICATIONS
35!! -------------
36!! Original 21/08/2013 first coded in rain_ice_elec
37!! C. Barthe 01/06/2023 : externalize the code to use it with ICE3 and LIMA
38!! C. Barthe 08/06/2023 : correction by 10-5 of the dynamic viscosity of air
39!! (unecessary for eta0/eta but necessary for Re0)

63TYPE(dimphyex_t), INTENT(IN) :: D
64TYPE(cst_t), INTENT(IN) :: CST
65INTEGER, INTENT(IN) :: KID ! Hydrometeor ID
66LOGICAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: OSEDIM ! if T, compute the sedim. proc.
67REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODREF ! Reference density
68REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PT ! Temperature
69REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRX ! m.r. source
70REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PQX ! Elec. charge density source
71REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PEFIELDW ! Vertical component of the electric field
72REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PLBDA ! Slope param. of the distribution
73REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PBEARDCOEF ! Beard coefficient
74CHARACTER (LEN=4), INTENT(IN) :: HCLOUD ! Kind of microphysical scheme
75!
76REAL, INTENT(IN) :: PTHVREFZIKB ! Reference thv at IKB for electricity
77TYPE(rain_ice_descr_t), OPTIONAL, INTENT(IN) :: ICED
78TYPE(param_lima_t), OPTIONAL, INTENT(IN) :: LIMAP
79TYPE(param_lima_cold_t), OPTIONAL, INTENT(IN) :: LIMAPC
80TYPE(param_lima_warm_t), OPTIONAL, INTENT(IN) :: LIMAPW
81TYPE(param_lima_mixed_t), OPTIONAL, INTENT(IN) :: LIMAPM
82!* 0.2 Declarations of local variables
83!
84INTEGER :: JIJ, JK ! loop indexes
85INTEGER :: IIJB, IIJE, IKTB, IKTE
86REAL :: ZCEXVT, ZBX, ZCX, ZDX, ZALPHAX, ZNUX
87REAL :: ZRE0
88REAL :: ZETA0
89REAL :: ZVX
90REAL :: ZK
91REAL :: ZCOR00, ZRHO00
92REAL :: ZT ! Temperature (C)
93REAL :: ZCOR ! To remove the Foote-duToit correction
94REAL :: ZF0, ZF1 ! Coef. in Beard's equation
95real, dimension(D%NIJT,D%NKT) :: zreynolds
96!
97!-------------------------------------------------------------------------------
98!
99
100!* 1. COMPUTE USEFULL PARAMETERS
101! --------------------------
102!
103iktb = d%NKTB
104ikte = d%NKTE
105iijb = d%NIJB
106iije = d%NIJE
107!
108!* 1.1 Select the right parameters
109! --> depend on the microphysics scheme and the hydrometeor species
110!
111IF (hcloud(1:3) == 'ICE') THEN
112 IF(.NOT. PRESENT(iced)) THEN
113 CALL print_msg( nverb_fatal, 'GEN', 'ELEC_BEARD_EFFECT', 'ICED mandatory with ICE3/ICE4')
114 ENDIF
115 zcexvt = iced%XCEXVT
116 !
117 IF (kid == 2) THEN
118 zbx = iced%XBC
119 zcx = iced%XCC
120 zdx = iced%XDC
121 zalphax = iced%XALPHAC
122 znux = iced%XNUC
123 ELSE IF (kid == 3) THEN
124 zbx = iced%XBR
125 zcx = iced%XCR
126 zdx = iced%XDR
127 zalphax = iced%XALPHAR
128 znux = iced%XNUR
129 ELSE IF (kid == 4) THEN
130 ! values for columns are used to be consistent with the McF&H formula
131 zbx = 1.7
132 zcx = 2.1e5
133 zdx = 1.585
134 zalphax = iced%XALPHAI
135 znux = iced%XNUI
136 ELSE IF (kid == 5) THEN
137 zbx = iced%XBS
138 zcx = iced%XCS
139 zdx = iced%XDS
140 zalphax = iced%XALPHAS
141 znux = iced%XNUS
142 ELSE IF (kid == 6) THEN
143 zbx = iced%XBG
144 zcx = iced%XCG
145 zdx = iced%XDG
146 zalphax = iced%XALPHAG
147 znux = iced%XNUG
148 ELSE IF (kid == 7) THEN
149 zbx = iced%XBH
150 zcx = iced%XCH
151 zdx = iced%XDH
152 zalphax = iced%XALPHAH
153 znux = iced%XNUH
154 END IF
155ELSE IF (hcloud == 'LIMA') THEN
156 IF(.NOT. PRESENT(limap)) THEN
157 CALL print_msg( nverb_fatal, 'GEN', 'ELEC_BEARD_EFFECT', 'LIMAP mandatory with ICE3/ICE4')
158 ENDIF
159 IF(.NOT. PRESENT(limapc)) THEN
160 CALL print_msg( nverb_fatal, 'GEN', 'ELEC_BEARD_EFFECT', 'LIMAPC mandatory with ICE3/ICE4')
161 ENDIF
162 IF(.NOT. PRESENT(limapw)) THEN
163 CALL print_msg( nverb_fatal, 'GEN', 'ELEC_BEARD_EFFECT', 'LIMAPW mandatory with ICE3/ICE4')
164 ENDIF
165 IF(.NOT. PRESENT(limapm)) THEN
166 CALL print_msg( nverb_fatal, 'GEN', 'ELEC_BEARD_EFFECT', 'LIMAPM mandatory with ICE3/ICE4')
167 ENDIF
168
169 zcexvt = limap%XCEXVT
170 !
171 IF (kid == 2) THEN
172 zbx = limapw%XBC
173 zcx = limapw%XCC
174 zdx = limapw%XDC
175 zalphax = limap%XALPHAC
176 znux = limap%XNUC
177 ELSE IF (kid == 3) THEN
178 zbx = limapw%XBR
179 zcx = limapw%XCR
180 zdx = limapw%XDR
181 zalphax = limap%XALPHAR
182 znux = limap%XNUR
183 ELSE IF (kid == 4) THEN
184 zbx = 1.7
185 zcx = 2.1e5
186 zdx = 1.585
187 zalphax = limap%XALPHAI
188 znux = limap%XNUI
189 ELSE IF (kid == 5) THEN
190 zbx = limapc%XBS
191 zcx = limapc%XCS
192 zdx = limapc%XDS
193 zalphax = limap%XALPHAS
194 znux = limap%XNUS
195 ELSE IF (kid == 6) THEN
196 zbx = limapm%XBG
197 zcx = limapm%XCG
198 zdx = limapm%XDG
199 zalphax = limap%XALPHAG
200 znux = limap%XNUG
201 ELSE IF (kid == 7) THEN
202 zbx = limapm%XBH
203 zcx = limapm%XCH
204 zdx = limapm%XDH
205 zalphax = limapm%XALPHAH
206 znux = limapm%XNUH
207 END IF
208 !
209END IF
210!
211!* 1.2 Parameters from Table 1 in Beard (1980)
212!
213! Reference value of the dynamic viscosity of air
214zeta0 = (1.718e-5 + 0.0049e-5 * (pthvrefzikb - cst%XTT))
215!
216zrho00 = cst%XP00 / (cst%XRD * pthvrefzikb)
217zcor00 = zrho00**zcexvt
218!
219! (rho_0 / eta_0) * (v * lambda^d)
220zvx = (zrho00 / zeta0) * zcx * momg(zalphax,znux,zbx+zdx) / momg(zalphax,znux,zbx)
221!
222!-------------------------------------------------------------------------------
223!
224!* 2. COMPUTE THE VELOCITY ADJUSTMENT FACTOR
225! --------------------------------------
226!
227zreynolds(:,:) = 0.
228pbeardcoef(:,:) = 1.0
229!
230DO jk = iktb, ikte
231 DO jij = iijb, iije
232!++cb++ 09/06/23 on n'applique l'effet Beard que pour les points ou le rapport de melange est
233! suffisamment eleve pour eviter que qE >> mg => coef de Beard tres eleve !
234! Ce pb intervient avec ICE3 pour lequel xrtmin est tres bas par rapport a LIMA.
235 IF (osedim(jij,jk) .AND. prx(jij,jk) .GT. xrtmin_elec(kid) .AND. plbda(jij,jk) .GT. 0.) THEN
236!--cb--
237 ! Temperature K --> C
238 zt = pt(jij,jk) - cst%XTT
239 !
240 ! Pre-factor of f_0
241 IF (zt >= 0.0) THEN
242 zf0 = zeta0 / (1.718e-5 + 0.0049e-5 * zt)
243 ELSE
244 zf0 = zeta0 / (1.718e-5 + 0.0049e-5 * zt - 1.2e-10 * zt * zt)
245 END IF
246 !
247 ! Pre-factor of f_infty
248 zf1 = sqrt(zrho00/prhodref(jij,jk))
249 !
250 ! compute (1 - K) = 1 - qE/mg
251 zk = 1. - pqx(jij,jk) * pefieldw(jij,jk) / (prx(jij,jk) * cst%XG)
252 !
253 ! Hyp : K_0 ~ 0
254 ! Hyp : si qE > mg, K > 1
255 IF (zk <= 0.0) THEN
256 pbeardcoef(jij,jk) = 0. ! levitation
257 ELSE
258 ! Reynolds number
259 zre0 = zvx / plbda(jij,jk)**(1.+zdx)
260 zreynolds(jij,jk) = zre0
261 IF (zre0 <= 0.2) THEN
262 pbeardcoef(jij,jk) = zf0 * zk
263 ELSE IF (zre0 >= 1000.) THEN
264 pbeardcoef(jij,jk) = zf1 * sqrt(zk)
265 ELSE
266 pbeardcoef(jij,jk) = zf0 * zk + &
267 (zf1 * sqrt(zk) - zf0 * zk) * &
268 (1.61 + log(zre0)) / 8.52
269 END IF
270 ! remove the Foote-duToit correction
271 zcor = (prhodref(jij,jk) / zrho00)**zcexvt
272 pbeardcoef(jij,jk) = pbeardcoef(jij,jk) * zcor
273 END IF
274 ELSE
275 pbeardcoef(jij,jk) = 1.0 ! No "Beard" effect
276 END IF
277 END DO
278END DO
279!
280!-------------------------------------------------------------------------------
281!
282END SUBROUTINE elec_beard_effect
