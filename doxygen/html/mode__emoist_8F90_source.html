8 SUBROUTINE emoist(D,CST,KRR,KRRI,PTHLM,PRM,PLOCPEXNM,PAMOIST,PSRCM,OOCEAN,PEMOIST)
9 USE yomhook , ONLY : lhook, dr_hook, jphook
10 ! ############################################################################
11 !
12 ! PURPOSE
13 !! -------
14 ! EMOIST computes the coefficient Emoist in the flottability turbulent
15 ! flux. This coefficient relates the vertical flux of the virtual potential
16 ! temperature ( <Thv' W'> ) to the vertical flux of the conservative mixing
17 ! ratio ( <Rnp' W'> ).
18 !
19 !!** METHOD
20 !! ------
21 !! The virtual potential temperature perturbation is linearized in function
22 !! of Thl' and Rnp'. The result is
23 !! Thv'= ( ZA + ZC * Atheta * 2 * SRC ) Thl'
24 !! +( ZB + ZC * Amoist * 2 * SRC ) Rnp'
25 !! From this relation, we can compute the verical turbulent fluxes.
26 !!
27 !! EXTERNAL
28 !! --------
29 !! NONE
30 !!
31 !! IMPLICIT ARGUMENTS
32 !! ------------------
33 !! Module MODD_CST : contains physical constants.
34 !! XRV, XRD : R for water vapor and dry air
35 !!
36 !! REFERENCE
37 !! ---------
38 !!
39 !! NONE
40 !!
41 !!
42 !! AUTHOR
43 !! ------
44 !! Jean-Marie Carriere * Meteo-France *
45 !!
46 !! MODIFICATIONS
47 !! -------------
48 !! Original 20/03/95
49 !!
50 !! J. Stein Feb 28, 1996 optimization + Doctorization
51 !! J. Stein Spet 15, 1996 Amoist previously computed
52 !! J.-P. Pinty May 20, 2003 Improve EMOIST expression
53 !! 03/2021 (JL Redelsperger) Ocean model case
54 !!
55 !! ----------------------------------------------------------------------
56 !
57 !* 0. DECLARATIONS
58 ! ------------
59 USE modd_cst, ONLY : cst_t
60 USE modd_dimphyex, ONLY: dimphyex_t
61 
62 !
63 IMPLICIT NONE
64 !
65 !* 0.1 declarations of arguments and result
66 !
67 !
68 TYPE(dimphyex_t), INTENT(IN) :: D
69 TYPE(cst_t), INTENT(IN) :: CST
70 INTEGER, INTENT(IN) :: KRR ! number of moist var.
71 INTEGER, INTENT(IN) :: KRRI ! number of ice var.
72 LOGICAL, INTENT(IN) :: OOCEAN ! switch for Ocean model version
73 !
74 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTHLM ! Conservative pot. temperature
75 REAL, DIMENSION(D%NIJT,D%NKT,KRR), INTENT(IN) :: PRM ! Mixing ratios, where
76 ! PRM(:,:,:,1) = conservative mixing ratio
77 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PLOCPEXNM ! Lv(T)/Cp/Exner at time t-1
78 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PAMOIST ! Amoist
79 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PSRCM ! Normalized 2dn_order
80  ! moment s'r'c/2Sigma_s2
81 !
82 REAL,DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PEMOIST ! result
83 !
84 !* 0.2 declarations of local variables
85 !
86 REAL,DIMENSION(D%NIJT,D%NKT) :: &
87  ZA, ZRW
88 ! ZA = coeft A, ZRW = total mixing ratio rw
89 REAL :: ZDELTA ! = Rv/Rd - 1
90 INTEGER :: JRR ! moist loop counter
91 INTEGER :: JIJ,JK ! loop counter
92 INTEGER :: IIJB,IIJE,IKT
93 !
94 !---------------------------------------------------------------------------
95 !
96 !
97 !* 1. COMPUTE EMOIST
98 ! --------------
99 !
100 REAL(KIND=jphook) :: zhook_handle
101 IF (lhook) CALL dr_hook('EMOIST',0,zhook_handle)
102 !
103 iijb=d%NIJB
104 iije=d%NIJE
105 ikt=d%NKT
106 !
107 IF (oocean) THEN
108  IF ( krr == 0 ) THEN ! Unsalted
109  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
110  pemoist(iijb:iije,:) = 0.
111  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
112  ELSE
113  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
114  pemoist(iijb:iije,:) = 1. ! Salted case
115  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
116  END IF
117  !
118 ELSE
119  !
120  IF ( krr == 0 ) THEN ! dry case
121  pemoist(iijb:iije,:) = 0.
122  ELSE IF ( krr == 1 ) THEN ! only vapor
123  zdelta = (cst%XRV/cst%XRD) - 1.
124  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
125  pemoist(iijb:iije,:) = zdelta*pthlm(iijb:iije,:)
126  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
127  ELSE ! liquid water & ice present
128  zdelta = (cst%XRV/cst%XRD) - 1.
129  zrw(iijb:iije,:) = prm(iijb:iije,:,1)
130  !
131  IF ( krri>0) THEN ! rc and ri case
132  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
133  zrw(iijb:iije,:) = zrw(iijb:iije,:) + prm(iijb:iije,:,3)
134  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
135  DO jrr=5,krr
136  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
137  zrw(iijb:iije,:) = zrw(iijb:iije,:) + prm(iijb:iije,:,jrr)
138  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
139  ENDDO
140  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
141  za(iijb:iije,:) = 1. + ( & ! Compute A
142  (1.+zdelta) * (prm(iijb:iije,:,1) - prm(iijb:iije,:,2) - prm(iijb:iije,:,4)) &
143  -zrw(iijb:iije,:) &
144  ) / (1. + zrw(iijb:iije,:))
145  !
146  ! Emoist = ZB + ZC * Amoist
147  ! ZB is computed from line 1 to line 2
148  ! ZC is computed from line 3 to line 5
149  ! Amoist* 2 * SRC is computed at line 6
150  !
151  pemoist(iijb:iije,:) = zdelta * (pthlm(iijb:iije,:) + plocpexnm(iijb:iije,:)*( &
152  prm(iijb:iije,:,2)+prm(iijb:iije,:,4)))&
153  / (1. + zrw(iijb:iije,:)) &
154  +( plocpexnm(iijb:iije,:) * za(iijb:iije,:) &
155  -(1.+zdelta) * (pthlm(iijb:iije,:) + plocpexnm(iijb:iije,:)*( &
156  prm(iijb:iije,:,2)+prm(iijb:iije,:,4)))&
157  / (1. + zrw(iijb:iije,:)) &
158  ) * pamoist(iijb:iije,:) * 2. * psrcm(iijb:iije,:)
159  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
160  ELSE
161  DO jrr=3,krr
162  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
163  zrw(iijb:iije,:) = zrw(iijb:iije,:) + prm(iijb:iije,:,jrr)
164  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
165  ENDDO
166  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
167  za(iijb:iije,:) = 1. + ( & ! Compute ZA
168  (1.+zdelta) * (prm(iijb:iije,:,1) - prm(iijb:iije,:,2)) &
169  -zrw(iijb:iije,:) &
170  ) / (1. + zrw(iijb:iije,:))
171  !
172  ! Emoist = ZB + ZC * Amoist
173  ! ZB is computed from line 1 to line 2
174  ! ZC is computed from line 3 to line 5
175  ! Amoist* 2 * SRC is computed at line 6
176  !
177  pemoist(iijb:iije,:) = zdelta * (pthlm(iijb:iije,:) + plocpexnm(iijb:iije,:)* &
178  prm(iijb:iije,:,2)) / (1. + zrw(iijb:iije,:)) &
179  +( plocpexnm(iijb:iije,:) * za(iijb:iije,:) &
180  -(1.+zdelta) * (pthlm(iijb:iije,:) + plocpexnm(iijb:iije,:)* &
181  prm(iijb:iije,:,2)) / (1. + zrw(iijb:iije,:)) &
182  ) * pamoist(iijb:iije,:) * 2. * psrcm(iijb:iije,:)
183  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
184  END IF
185  END IF
186  !
187 END IF
188 !---------------------------------------------------------------------------
189 !
190 IF (lhook) CALL dr_hook('EMOIST',1,zhook_handle)
191 END SUBROUTINE emoist
192 END MODULE mode_emoist
