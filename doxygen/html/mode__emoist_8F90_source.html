8SUBROUTINE emoist(D,CST,KRR,KRRI,PTHLM,PRM,PLOCPEXNM,PAMOIST,PSRCM,OOCEAN,PEMOIST)
9USE yomhook , ONLY : lhook, dr_hook, jphook
10! ############################################################################
11!
12! PURPOSE
13!! -------
14! EMOIST computes the coefficient Emoist in the flottability turbulent
15! flux. This coefficient relates the vertical flux of the virtual potential
16! temperature ( <Thv' W'> ) to the vertical flux of the conservative mixing
17! ratio ( <Rnp' W'> ).
18!
19!!** METHOD
20!! ------
21!! The virtual potential temperature perturbation is linearized in function
22!! of Thl' and Rnp'. The result is
23!! Thv'= ( ZA + ZC * Atheta * 2 * SRC ) Thl'
24!! +( ZB + ZC * Amoist * 2 * SRC ) Rnp'
25!! From this relation, we can compute the verical turbulent fluxes.
26!!
27!! EXTERNAL
28!! --------
29!! NONE
30!!
31!! IMPLICIT ARGUMENTS
32!! ------------------
33!! Module MODD_CST : contains physical constants.
34!! XRV, XRD : R for water vapor and dry air
35!!
36!! REFERENCE
37!! ---------
38!!
39!! NONE
40!!
41!!
42!! AUTHOR
43!! ------
44!! Jean-Marie Carriere * Meteo-France *
45!!
46!! MODIFICATIONS
47!! -------------
48!! Original 20/03/95
49!!
50!! J. Stein Feb 28, 1996 optimization + Doctorization
51!! J. Stein Spet 15, 1996 Amoist previously computed
52!! J.-P. Pinty May 20, 2003 Improve EMOIST expression
53!! 03/2021 (JL Redelsperger) Ocean model case
54!!
55!! ----------------------------------------------------------------------
56!
57!* 0. DECLARATIONS
58! ------------
59USE modd_cst, ONLY : cst_t
60USE modd_dimphyex, ONLY: dimphyex_t
61
62!
63IMPLICIT NONE
64!
65!* 0.1 declarations of arguments and result
66!
67!
68TYPE(dimphyex_t), INTENT(IN) :: D
69TYPE(cst_t), INTENT(IN) :: CST
70INTEGER, INTENT(IN) :: KRR ! number of moist var.
71INTEGER, INTENT(IN) :: KRRI ! number of ice var.
72LOGICAL, INTENT(IN) :: OOCEAN ! switch for Ocean model version
73!
74REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTHLM ! Conservative pot. temperature
75REAL, DIMENSION(D%NIJT,D%NKT,KRR), INTENT(IN) :: PRM ! Mixing ratios, where
76! PRM(:,:,:,1) = conservative mixing ratio
77REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PLOCPEXNM ! Lv(T)/Cp/Exner at time t-1
78REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PAMOIST ! Amoist
79REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PSRCM ! Normalized 2dn_order
80 ! moment s'r'c/2Sigma_s2
81!
82REAL,DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PEMOIST ! result
83!
84!* 0.2 declarations of local variables
85!
86REAL,DIMENSION(D%NIJT,D%NKT) :: &
87 ZA, ZRW
88! ZA = coeft A, ZRW = total mixing ratio rw
89REAL :: ZDELTA ! = Rv/Rd - 1
90INTEGER :: JRR ! moist loop counter
91INTEGER :: JIJ,JK ! loop counter
92INTEGER :: IIJB,IIJE,IKT
93!
94!---------------------------------------------------------------------------
95!
96!
97!* 1. COMPUTE EMOIST
98! --------------
99!
100REAL(KIND=jphook) :: zhook_handle
101IF (lhook) CALL dr_hook('EMOIST',0,zhook_handle)
102!
103iijb=d%NIJB
104iije=d%NIJE
105ikt=d%NKT
106!
107IF (oocean) THEN
108 IF ( krr == 0 ) THEN ! Unsalted
109!$acc kernels
110 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
111 pemoist(iijb:iije,:) = 0.
112 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
113!$acc end kernels
114 ELSE
115!$acc kernels
116 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
117 pemoist(iijb:iije,:) = 1. ! Salted case
118 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
119!$acc end kernels
120 END IF
121 !
122ELSE
123 !
124 IF ( krr == 0 ) THEN ! dry case
125!$acc kernels
126 pemoist(iijb:iije,:) = 0.
127!$acc end kernels
128 ELSE IF ( krr == 1 ) THEN ! only vapor
129!$acc kernels
130 zdelta = (cst%XRV/cst%XRD) - 1.
131 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
132 pemoist(iijb:iije,:) = zdelta*pthlm(iijb:iije,:)
133 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
134!$acc end kernels
135 ELSE ! liquid water & ice present
136!$acc kernels present_cr(ZRW)
137 zdelta = (cst%XRV/cst%XRD) - 1.
138 zrw(iijb:iije,:) = prm(iijb:iije,:,1)
139!$acc end kernels
140 !
141 IF ( krri>0) THEN ! rc and ri case
142!$acc kernels
143 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
144 zrw(iijb:iije,:) = zrw(iijb:iije,:) + prm(iijb:iije,:,3)
145 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
146!$acc loop seq
147 DO jrr=5,krr
148 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
149 zrw(iijb:iije,:) = zrw(iijb:iije,:) + prm(iijb:iije,:,jrr)
150 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
151 ENDDO
152 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
153 za(iijb:iije,:) = 1. + ( & ! Compute A
154 (1.+zdelta) * (prm(iijb:iije,:,1) - prm(iijb:iije,:,2) - prm(iijb:iije,:,4)) &
155 -zrw(iijb:iije,:) &
156 ) / (1. + zrw(iijb:iije,:))
157 !
158 ! Emoist = ZB + ZC * Amoist
159 ! ZB is computed from line 1 to line 2
160 ! ZC is computed from line 3 to line 5
161 ! Amoist* 2 * SRC is computed at line 6
162 !
163 pemoist(iijb:iije,:) = zdelta * (pthlm(iijb:iije,:) + plocpexnm(iijb:iije,:)*( &
164 prm(iijb:iije,:,2)+prm(iijb:iije,:,4)))&
165 / (1. + zrw(iijb:iije,:)) &
166 +( plocpexnm(iijb:iije,:) * za(iijb:iije,:) &
167 -(1.+zdelta) * (pthlm(iijb:iije,:) + plocpexnm(iijb:iije,:)*( &
168 prm(iijb:iije,:,2)+prm(iijb:iije,:,4)))&
169 / (1. + zrw(iijb:iije,:)) &
170 ) * pamoist(iijb:iije,:) * 2. * psrcm(iijb:iije,:)
171 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
172!$acc end kernels
173 ELSE
174!$acc kernels
175!$acc loop seq
176 DO jrr=3,krr
177 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
178 zrw(iijb:iije,:) = zrw(iijb:iije,:) + prm(iijb:iije,:,jrr)
179 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
180 ENDDO
181 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
182 za(iijb:iije,:) = 1. + ( & ! Compute ZA
183 (1.+zdelta) * (prm(iijb:iije,:,1) - prm(iijb:iije,:,2)) &
184 -zrw(iijb:iije,:) &
185 ) / (1. + zrw(iijb:iije,:))
186 !
187 ! Emoist = ZB + ZC * Amoist
188 ! ZB is computed from line 1 to line 2
189 ! ZC is computed from line 3 to line 5
190 ! Amoist* 2 * SRC is computed at line 6
191 !
192 pemoist(iijb:iije,:) = zdelta * (pthlm(iijb:iije,:) + plocpexnm(iijb:iije,:)* &
193 prm(iijb:iije,:,2)) / (1. + zrw(iijb:iije,:)) &
194 +( plocpexnm(iijb:iije,:) * za(iijb:iije,:) &
195 -(1.+zdelta) * (pthlm(iijb:iije,:) + plocpexnm(iijb:iije,:)* &
196 prm(iijb:iije,:,2)) / (1. + zrw(iijb:iije,:)) &
197 ) * pamoist(iijb:iije,:) * 2. * psrcm(iijb:iije,:)
198 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
199!$acc end kernels
200 END IF
201 END IF
202 !
203END IF
204!---------------------------------------------------------------------------
205!
206IF (lhook) CALL dr_hook('EMOIST',1,zhook_handle)
207END SUBROUTINE emoist
208END MODULE mode_emoist
