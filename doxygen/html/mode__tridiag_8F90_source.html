11 ! #################################################
12 !
13 !
14 !!**** *TRIDIAG* - routine to solve a time implicit scheme
15 !!
16 !!
17 !! PURPOSE
18 !! -------
19 ! The purpose of this routine is to give a field PVARP at t+1, by
20 ! solving an implicit tridiagonal system obtained by the
21 ! discretization of the vertical turbulent diffusion. It should be noted
22 ! that the degree of implicitness can be varied (PIMPL parameter) and the
23 ! sources of evolution other than the turbulent diffusion can be taken
24 ! into account through the PSOURCE field. PVARP is localized at a mass
25 ! point.
26 !
27 !!** METHOD
28 !! ------
29 !! First, the Right Hand Side of the implicit equation is computed.
30 !! It is build as follows:
31 !! ZY = PVARM + PTSTEP*PSOURCE + DIFF_EXPLI
32 !! where PVARM is the variable at t-dt, PSOURCE the supplementary sources of
33 !! PVAR ( and not PVAR * PRHODJ !!) and DIFF_EXPLI is the explicit part
34 !! of the vertical turbulent diffusion. This operator is spatially
35 !! discretized as the implicit one, thus:
36 !! DIFF_EXPLI(k) = - PEXPL / PRHODJ(k) *
37 !! ( PA(k+1) * (PVARM(k+1) - PVARM(k) )
38 !! -PA(k) * (PVARM(k) - PVARM(k-1)) )
39 !! For the first level, only the upper part is considered, the lower one
40 !! is replaced by the turbulent surface flux (taken into account in the
41 !! PSOURCE(ikb) term).
42 !! DIFF_EXPLI(ikb) = - PEXPL / PRHODJ(ikb) *
43 !! ( PA(ikb+1) * (PVARM(ikb+1) - PVARM(ikb)) )
44 !! For the last level, only the lower part is considered, the upper one
45 !! is replaced by the turbulent flux which is taken equal to 0
46 !! (taken into account in the PSOURCE(ike) term).
47 !!
48 !! DIFF_EXPLI(ike) = + PEXPL / PRHODJ(ike) *
49 !! ( PA(ike) * (PVARM(ike) - PVARM(ike-1)) )
50 !!
51 !! Then, the classical tridiagonal algorithm is used to invert the
52 !! implicit operator. Its matrix is given by:
53 !!
54 !! ( b(ikb) c(ikb) 0 0 0 0 0 0 )
55 !! ( 0 a(ikb+1) b(ikb+1) c(ikb+1) 0 ... 0 0 0 )
56 !! ( 0 0 a(ikb+2) b(ikb+2) c(ikb+2). 0 0 0 )
57 !! .......................................................................
58 !! ( 0 ... 0 a(k) b(k) c(k) 0 ... 0 0 )
59 !! .......................................................................
60 !! ( 0 0 0 0 0 ...a(ike-1) b(ike-1) c(ike-1))
61 !! ( 0 0 0 0 0 ... 0 a(ike) b(ike) )
62 !!
63 !! ikb and ike represent the first and the last inner mass levels of the
64 !! model. The coefficients are:
65 !!
66 !! a(k) = PIMPL * PA(k)/PRHODJ(k)
67 !! b(k) = 1 - PIMPL * PA(k)/PRHODJ(k) - PIMPL * PA(k+1)/PRHODJ(k)
68 !! c(k) = PIMPL * PA(k+1)/PRHODJ(k)
69 !!
70 !! for all k /= ikb or ike
71 !!
72 !! b(ikb) = 1 - PIMPL * PA(ikb+1)/PRHODJ(ikb)
73 !! c(ikb) = PIMPL * PA(ikb+1)/PRHODJ(ikb)
74 !! (discretization of the upper part of the implicit operator)
75 !! b(ike) = 1 - PIMPL * PA(ike)/PRHODJ(ike)
76 !! a(ike) = PIMPL * PA(ike)/PRHODJ(ike)
77 !! (discretization of the lower part of the implicit operator)
78 !! Finally, the marginal points are prescribed.
79 !!
80 !! All these computations are purely vertical and vectorizations are
81 !! easely achieved by processing all the verticals in parallel.
82 !!
83 !! EXTERNAL
84 !! --------
85 !!
86 !! NONE
87 !!
88 !! IMPLICIT ARGUMENTS
89 !! ------------------
90 !! MODD_PARAMETERS
91 !! JPVEXT_TURB: number of vertical external points
92 !!
93 !! REFERENCE
94 !! ---------
95 !! Book 1 of Meso-NH documentation (chapter Turbulence)
96 !! Press et al: Numerical recipes (1986) Cambridge Univ. Press
97 !!
98 !! AUTHOR
99 !! ------
100 !! Joan Cuxart * INM and Meteo-France *
101 !!
102 !! MODIFICATIONS
103 !! -------------
104 !! Original August 29, 1994
105 !! Modification : January 29, 1995 Algorithm written with two
106 !! local variables less
107 !! (Cuxart, Stein) August 21, 1995 Bug correction for PRHODJ
108 !! (Stein) November 16, 1995 new version
109 !! (Stein) February 28, 1995 no inversion in the explicit case
110 !! (Seity) February 2012 add possibility to run with reversed
111 !! vertical levels
112 !! ---------------------------------------------------------------------
113 !
114 !* 0. DECLARATIONS
115 !
116 USE modd_dimphyex, ONLY: dimphyex_t
117 !
118 IMPLICIT NONE
119 !
120 !
121 !* 0.1 declarations of arguments
122 !
123 TYPE(dimphyex_t), INTENT(IN) :: D
124 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PVARM ! variable at t-1
125 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PA ! upper diag. elements
126 REAL, INTENT(IN) :: PTSTEP ! Double time step
127 REAL, INTENT(IN) :: PEXPL,PIMPL ! weights of the temporal scheme
128 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODJ ! (dry rho)*J
129 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PSOURCE ! source term of PVAR
130 !
131 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PVARP ! variable at t+1
132 !
133 !* 0.2 declarations of local variables
134 !
135 REAL, DIMENSION(D%NIJT,D%NKT) :: ZY ,ZGAM
136  ! RHS of the equation, 3D work array
137 REAL, DIMENSION(D%NIJT) :: ZBET
138  ! 2D work array
139 INTEGER :: JIJ,JK ! loop counter
140 INTEGER :: IKB,IKE ! inner vertical limits
141 INTEGER :: IKT,IKA,IKU,IKL! array size in k direction
142 INTEGER :: IKTB,IKTE ! start, end of k loops in physical domain
143 INTEGER :: IIJB, IIJE ! start, end of ij loops in physical domain
144 
145 !
146 ! ---------------------------------------------------------------------------
147 !
148 !* 1. COMPUTE THE RIGHT HAND SIDE
149 ! ---------------------------
150 !
151 REAL(KIND=jphook) :: zhook_handle
152 IF (lhook) CALL dr_hook('TRIDIAG',0,zhook_handle)
153 !
154 ikt=d%NKT
155 iktb=d%NKTB
156 ikte=d%NKTE
157 ikb=d%NKB
158 ike=d%NKE
159 ika=d%NKA
160 iku=d%NKU
161 ikl=d%NKL
162 iijb=d%NIJB
163 iije=d%NIJE
164 !
165 !$mnh_expand_array(JIJ=IIJB:IIJE)
166 zy(iijb:iije,ikb) = pvarm(iijb:iije,ikb) + ptstep*psource(iijb:iije,ikb) - &
167  pexpl / prhodj(iijb:iije,ikb) * pa(iijb:iije,ikb+ikl) * &
168  (pvarm(iijb:iije,ikb+ikl) - pvarm(iijb:iije,ikb))
169 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
170 !
171 DO jk=iktb+1,ikte-1
172  !$mnh_expand_array(JIJ=IIJB:IIJE)
173  zy(iijb:iije,jk)= pvarm(iijb:iije,jk) + ptstep*psource(iijb:iije,jk) - &
174  pexpl / prhodj(iijb:iije,jk) * &
175  ( pvarm(iijb:iije,jk-ikl)*pa(iijb:iije,jk) &
176  -pvarm(iijb:iije,jk)*(pa(iijb:iije,jk)+pa(iijb:iije,jk+ikl)) &
177  +pvarm(iijb:iije,jk+ikl)*pa(iijb:iije,jk+ikl) &
178  )
179  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
180 END DO
181 !
182 !$mnh_expand_array(JIJ=IIJB:IIJE)
183 zy(iijb:iije,ike)= pvarm(iijb:iije,ike) + ptstep*psource(iijb:iije,ike) + &
184  pexpl / prhodj(iijb:iije,ike) * pa(iijb:iije,ike) * (pvarm(iijb:iije,ike)-pvarm(iijb:iije,ike-ikl))
185 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
186 !
187 !
188 !* 2. INVERSION OF THE TRIDIAGONAL SYSTEM
189 ! -----------------------------------
190 !
191 IF ( pimpl > 1.e-10 ) THEN
192 !
193  !
194  ! going up
195  !
196  !$mnh_expand_array(JIJ=IIJB:IIJE)
197  zbet(iijb:iije) = 1. - pimpl * pa(iijb:iije,ikb+ikl) / prhodj(iijb:iije,ikb) ! bet = b(ikb)
198  pvarp(iijb:iije,ikb) = zy(iijb:iije,ikb) / zbet(iijb:iije)
199  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
200  !
201  DO jk = ikb+ikl,ike-ikl,ikl
202  !$mnh_expand_array(JIJ=IIJB:IIJE)
203  zgam(iijb:iije,jk) = pimpl * pa(iijb:iije,jk) / prhodj(iijb:iije,jk-ikl) / zbet(iijb:iije)
204  ! gam(k) = c(k-1) / bet
205  zbet(iijb:iije) = 1. - pimpl * ( pa(iijb:iije,jk) * (1. + zgam(iijb:iije,jk)) &
206  + pa(iijb:iije,jk+ikl) &
207  ) / prhodj(iijb:iije,jk)
208  ! bet = b(k) - a(k)* gam(k)
209  pvarp(iijb:iije,jk)= ( zy(iijb:iije,jk) - pimpl * pa(iijb:iije,jk) / prhodj(iijb:iije,jk) &
210  * pvarp(iijb:iije,jk-ikl) &
211  ) / zbet(iijb:iije)
212  ! res(k) = (y(k) -a(k)*res(k-1))/ bet
213  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
214  END DO
215  !$mnh_expand_array(JIJ=IIJB:IIJE)
216  ! special treatment for the last level
217  zgam(iijb:iije,ike) = pimpl * pa(iijb:iije,ike) / prhodj(iijb:iije,ike-ikl) / zbet(iijb:iije)
218  ! gam(k) = c(k-1) / bet
219  zbet(iijb:iije) = 1. - pimpl * ( pa(iijb:iije,ike) * (1. + zgam(iijb:iije,ike)) &
220  ) / prhodj(iijb:iije,ike)
221  ! bet = b(k) - a(k)* gam(k)
222  pvarp(iijb:iije,ike)= ( zy(iijb:iije,ike) - pimpl * pa(iijb:iije,ike) / prhodj(iijb:iije,ike) &
223  * pvarp(iijb:iije,ike-ikl) &
224  ) / zbet(iijb:iije)
225  ! res(k) = (y(k) -a(k)*res(k-1))/ bet
226  !
227  ! going down
228  !
229  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
230  DO jk = ike-ikl,ikb,-1*ikl
231  !$mnh_expand_array(JIJ=IIJB:IIJE)
232  pvarp(iijb:iije,jk) = pvarp(iijb:iije,jk) - zgam(iijb:iije,jk+ikl) * pvarp(iijb:iije,jk+ikl)
233  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
234  END DO
235 !
236 ELSE
237 !
238  DO jk=iktb,ikte
239  !$mnh_expand_array(JIJ=IIJB:IIJE)
240  pvarp(iijb:iije,jk) = zy(iijb:iije,jk)
241  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
242  END DO
243 !
244 END IF
245 !
246 !
247 !* 3. FILL THE UPPER AND LOWER EXTERNAL VALUES
248 ! ----------------------------------------
249 !
250 !$mnh_expand_array(JIJ=IIJB:IIJE)
251 pvarp(iijb:iije,ika)=pvarp(iijb:iije,ikb)
252 pvarp(iijb:iije,iku)=pvarp(iijb:iije,ike)
253 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
254 !
255 !-------------------------------------------------------------------------------
256 !
257 IF (lhook) CALL dr_hook('TRIDIAG',1,zhook_handle)
258 END SUBROUTINE tridiag
259 END MODULE mode_tridiag
