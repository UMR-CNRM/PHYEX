11 PRHODREF, PT, &
12 PRCT, PCCT, PRST, PCST, PLBDC, PLBDS, PLVFACT, PLSFACT, &
13!++cb++
14! P_TH_RIM, P_RC_RIM, P_CC_RIM, P_RS_RIM, P_CS_RIM, P_RG_RIM, &
15 P_TH_RIM, P_CC_RIM, P_CS_RIM, &
16 P_RC_RIMSS, P_RC_RIMSG, P_RS_RIMCG, &
17 P_RI_HMS, P_CI_HMS, P_RS_HMS )
18! ###########################################################################################
19!
20!! PURPOSE
21!! -------
22!! Compute the cloud droplets riming of the aggregates rate, and the associated
23!! Hallett-Mossop ice production rate
24!!
25!! AUTHOR
26!! ------
27!! J.-M. Cohard * Laboratoire d'Aerologie*
28!! J.-P. Pinty * Laboratoire d'Aerologie*
29!! S. Berthet * Laboratoire d'Aerologie*
30!! B. Vié * CNRM *
31!!
32!! MODIFICATIONS
33!! -------------
34!! Original 15/03/2018
35! P. Wautelet 26/04/2019: replace non-standard FLOAT function by REAL function
36! J. Wurtz 03/2022: new snow characteristics
37! C. Barthe 06/2022: modify the microphysics terms to save to simplify the merging with the electrification scheme
38! (same terms as in ICE3)
39!
40!-------------------------------------------------------------------------------
41!
42!* 0. DECLARATIONS
43! ------------
44!
45USE modd_cst, ONLY : xtt
46USE modd_param_lima, ONLY : xrtmin, xctmin, xcexvt, xnus, xalphas, lmurakami
47USE modd_param_lima_mixed, ONLY : ngaminc, xrimintp1, xrimintp2, xgaminc_rim1, xgaminc_rim2, xgaminc_rim4, &
48 xcrimss, xexcrimss, xsrimcg, xexsrimcg, xsrimcg2, xsrimcg3, xexsrimcg2, &
49 xhmlintp1, xhmlintp2, xgaminc_hmc, xhm_facts, xhmtmin, xhmtmax
50USE modd_param_lima_cold, ONLY : xmnu0, xfvelos

52IMPLICIT NONE
53!
54!* 0.1 Declarations of dummy arguments :
55!
56REAL, INTENT(IN) :: PTSTEP
57LOGICAL, DIMENSION(:),INTENT(IN) :: LDCOMPUTE
58!
59REAL, DIMENSION(:), INTENT(IN) :: PRHODREF !
60REAL, DIMENSION(:), INTENT(IN) :: PT !
61!
62REAL, DIMENSION(:), INTENT(IN) :: PRCT ! Cloud water mr at t
63REAL, DIMENSION(:), INTENT(IN) :: PCCT ! Cloud water C. at t
64REAL, DIMENSION(:), INTENT(IN) :: PRST ! Snow mr at t
65REAL, DIMENSION(:), INTENT(IN) :: PCST ! Snow C. at t
66REAL, DIMENSION(:), INTENT(IN) :: PLBDC !
67REAL, DIMENSION(:), INTENT(IN) :: PLBDS !
68REAL, DIMENSION(:), INTENT(IN) :: PLVFACT !
69REAL, DIMENSION(:), INTENT(IN) :: PLSFACT !
70!
71!++cb++
72!REAL, DIMENSION(:), INTENT(OUT) :: P_RC_RIM
73REAL, DIMENSION(:), INTENT(OUT) :: P_CC_RIM
74!REAL, DIMENSION(:), INTENT(OUT) :: P_RS_RIM
75REAL, DIMENSION(:), INTENT(OUT) :: P_CS_RIM
76!REAL, DIMENSION(:), INTENT(OUT) :: P_RG_RIM
77REAL, DIMENSION(:), INTENT(OUT) :: P_RC_RIMSS
78REAL, DIMENSION(:), INTENT(OUT) :: P_RC_RIMSG
79REAL, DIMENSION(:), INTENT(OUT) :: P_RS_RIMCG
80!--cb--
81!
82REAL, DIMENSION(:), INTENT(OUT) :: P_TH_RIM
83REAL, DIMENSION(:), INTENT(OUT) :: P_RI_HMS
84REAL, DIMENSION(:), INTENT(OUT) :: P_CI_HMS
85REAL, DIMENSION(:), INTENT(OUT) :: P_RS_HMS
86!
87!* 0.2 Declarations of local variables :
88!
89REAL, DIMENSION(SIZE(PRCT)) :: ZZW1, ZZW2, ZZW3, ZZW4, ZZW5
90REAL, DIMENSION(SIZE(PRCT)) :: Z_RC_RIM, Z_RS_RIM, Z_RG_RIM !++cb--
91!
92INTEGER, DIMENSION(SIZE(PRCT)) :: IVEC2 ! Vector of indices
93REAL, DIMENSION(SIZE(PRCT)) :: ZVEC1,ZVEC2,ZVEC1W ! Work vectors
94INTEGER :: JI
95!
96!-------------------------------------------------------------------------------
97!
98!
99DO ji = 1, SIZE(prct)
100!
101!* Cloud droplet riming of the aggregates
102! --------------------------------------
103!
104 IF ( prct(ji)>xrtmin(2) .AND. prst(ji)>xrtmin(5) .AND. pt(ji)<xtt .AND. &
105 pcct(ji)>xctmin(2) .AND. pcst(ji)>xctmin(5) .AND. ldcompute(ji) ) THEN
106!
107 zvec1(ji) = plbds(ji)
108 zvec1w(ji)= ( xfvelos**xalphas + plbds(ji)**xalphas ) ** (1./xalphas) ! modified equivalent lambda
109!
110! 2. perform the linear interpolation of the normalized
111! "2+XDS"-moment of the incomplete gamma function using the modified equivalent lambda
112!
113 zvec2(ji) = max( 1.0001, min( real(ngaminc)-0.0001, &
114 xrimintp1 * log( zvec1w(ji) ) + xrimintp2 ) )
115 ivec2(ji) = int( zvec2(ji) )
116 zvec2(ji) = zvec2(ji) - real( ivec2(ji) )
117!
118 zzw1(ji) = xgaminc_rim1( ivec2(ji)+1 )* zvec2(ji) &
119 - xgaminc_rim1( ivec2(ji) )*(zvec2(ji) - 1.0)
120!
121! 3. perform the linear interpolation of the normalized
122! "XBS"-moment of the incomplete gamma function
123!
124 zvec2(ji) = max( 1.0001, min( real(ngaminc)-0.0001, &
125 xrimintp1 * log( zvec1(ji) ) + xrimintp2 ) )
126 ivec2(ji) = int( zvec2(ji) )
127 zvec2(ji) = zvec2(ji) - real( ivec2(ji) )
128!
129 zzw2(ji) = xgaminc_rim2( ivec2(ji)+1 )* zvec2(ji) &
130 - xgaminc_rim2( ivec2(ji) )*(zvec2(ji) - 1.0)
131!
132! 4. riming
133!
134!++cb++
135 ! Cloud droplets collected
136! P_RC_RIM(JI) = - XCRIMSS * PRCT(JI) * PCST(JI)*(1+(XFVELOS/PLBDS(JI))**XALPHAS)**(-XNUS+XEXCRIMSS/XALPHAS) &
137! * PRHODREF(JI)**(-XCEXVT+1) * PLBDS(JI)**XEXCRIMSS
138! P_CC_RIM(JI) = P_RC_RIM(JI) * PCCT(JI)/PRCT(JI) ! Lambda_c**3
139! total mass loss of cloud droplets, < 0
140 z_rc_rim(ji) = - xcrimss * prct(ji) * pcst(ji)*(1+(xfvelos/plbds(ji))**xalphas)**(-xnus+xexcrimss/xalphas) &
141 * prhodref(ji)**(-xcexvt+1) * plbds(ji)**xexcrimss
142 p_cc_rim(ji) = z_rc_rim(ji) * (pcct(ji) / prct(ji)) ! Lambda_c**3
143 !
144 ! Cloud droplets collected on small aggregates add to snow
145! P_RS_RIM(JI) = - P_RC_RIM(JI) * ZZW1(JI)
146 z_rs_rim(ji) = -z_rc_rim(ji) * zzw1(ji)
147 p_rc_rimss(ji) = z_rc_rim(ji) * zzw1(ji) ! < 0, loss of mass for rc
148 !
149 ! Cloud droplets collected on large aggregates add to graupel
150! P_RG_RIM(JI) = - P_RC_RIM(JI) - P_RS_RIM(JI)
151 z_rg_rim(ji) = -z_rc_rim(ji) - z_rs_rim(ji)
152 p_rc_rimsg(ji) = z_rc_rim(ji) - p_rc_rimss(ji) ! < 0, loss of mass for rc
153 !
154 IF (lmurakami) THEN
155 ! Graupel formation based on Murakami
156 zvec1(ji) = xgaminc_rim4( ivec2(ji)+1 )* zvec2(ji) &
157 - xgaminc_rim4( ivec2(ji) )*(zvec2(ji) - 1.0)
158 zzw5(ji) = zvec1(ji)
159 zzw3(ji) = xsrimcg * prhodref(ji) * pcst(ji) * plbds(ji)**xexsrimcg * (1.0 - zzw2(ji))!/(PTSTEP*PRHODREF(JI))
160 zzw3(ji) = z_rg_rim(ji)*zzw3(ji)/ &
161 max(1.e-10, & !-20
162 xsrimcg3*xsrimcg2*pcst(ji)*prhodref(ji)*plbds(ji)**(xexsrimcg2)*(1.-zzw5(ji))- &
163 xsrimcg3*zzw3(ji))
164 ELSE
165 ! Large aggregates collecting droplets add to graupel (instant process ???)
166 zzw3(ji) = prst(ji)*(1.0 - zzw2(ji))/ptstep
167 END IF
168 !
169 p_rs_rimcg(ji) = zzw3(ji)
170 p_cs_rim(ji) = -zzw3(ji) * pcst(ji)/prst(ji)
171! P_RS_RIM(JI) = P_RS_RIM(JI) - ZZW3(JI)
172! P_RG_RIM(JI) = P_RG_RIM(JI) + ZZW3(JI)
173 !
174! P_TH_RIM(JI) = - P_RC_RIM(JI)*(PLSFACT(JI)-PLVFACT(JI))
175 p_th_rim(ji) = - z_rc_rim(ji)*(plsfact(ji)-plvfact(ji))
176!--cb--
177 ELSE
178 p_th_rim(ji) = 0.
179 p_rc_rimss(ji) = 0.
180 p_rc_rimsg(ji) = 0.
181 p_rs_rimcg(ji) = 0.
182 z_rc_rim(ji) = 0.
183 p_cc_rim(ji) = 0.
184 z_rs_rim(ji) = 0.
185 p_cs_rim(ji) = 0.
186 z_rg_rim(ji) = 0.
187 END IF
188!
189!* Hallett-Mossop ice production (HMS)
190! -----------------------------------
191!
192 IF ( prst(ji)>xrtmin(5) .AND. prct(ji)>xrtmin(2) .AND. pt(ji)<xhmtmax .AND. pt(ji)>xhmtmin .AND. &
193 pcst(ji)>xctmin(5) .AND. pcct(ji)>xctmin(2) .AND. ldcompute(ji) ) THEN
194!
195 zvec1(ji) = plbdc(ji)
196 zvec2(ji) = max( 1.0001, min( real(ngaminc)-0.0001, &
197 xhmlintp1 * log( zvec1(ji) ) + xhmlintp2 ) )
198 ivec2(ji) = int( zvec2(ji) )
199 zvec2(ji) = zvec2(ji) - real( ivec2(ji) )
200 zvec1(ji) = xgaminc_hmc( ivec2(ji)+1 )* zvec2(ji) &
201 - xgaminc_hmc( ivec2(ji) )*(zvec2(ji) - 1.0)
202 zzw4(ji) = zvec1(ji) ! Large droplets
203!
204 IF ( zzw4(ji)<0.99 ) THEN
205 p_ci_hms(ji) = - z_rc_rim(ji) * (pcct(ji)/prct(ji)) * (1.0-zzw4(ji)) * xhm_facts * &
206 max( 0.0, min( (pt(ji)-xhmtmin)/3.0,(xhmtmax-pt(ji))/2.0 ) ) ! CCHMSI
207!
208 p_ri_hms(ji) = p_ci_hms(ji) * xmnu0 ! RCHMSI
209 p_rs_hms(ji) = - p_ri_hms(ji)
210 ELSE
211 p_ri_hms(ji) = 0.
212 p_ci_hms(ji) = 0.
213 p_rs_hms(ji) = 0.
214 END IF
215 ELSE
216 p_ri_hms(ji) = 0.
217 p_ci_hms(ji) = 0.
218 p_rs_hms(ji) = 0.
219 END IF
220END DO
221!
222!-------------------------------------------------------------------------------
223!
224END SUBROUTINE lima_droplets_riming_snow
225END MODULE mode_lima_droplets_riming_snow
