66REAL, DIMENSION(KSIZE), INTENT(IN) :: PRCT ! Cloud water mr at t

77REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_CC_RIM
78!REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_RS_RIM
79REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_CS_RIM
80!REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_RG_RIM

87REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_RI_HMS
88REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_CI_HMS
89REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_RS_HMS
90!
91!* 0.2 Declarations of local variables :
92!
93REAL, DIMENSION(SIZE(PRCT)) :: ZZW1, ZZW2, ZZW3, ZZW4, ZZW5
94REAL, DIMENSION(SIZE(PRCT)) :: Z_RC_RIM, Z_RS_RIM, Z_RG_RIM !++cb--
95!
96INTEGER, DIMENSION(SIZE(PRCT)) :: IVEC2 ! Vector of indices
97REAL, DIMENSION(SIZE(PRCT)) :: ZVEC1,ZVEC2,ZVEC1W ! Work vectors
98INTEGER :: II
99REAL(KIND=jphook) :: zhook_handle
100!
101!-------------------------------------------------------------------------------
102!
103!
104IF (lhook) CALL dr_hook('LIMA_DROPLETS_RIMING_SNOW', 0, zhook_handle)
105DO ii = 1, SIZE(prct)
106!
107!* Cloud droplet riming of the aggregates
108! --------------------------------------
109!
110 IF ( prct(ii)>limap%XRTMIN(2) .AND. prst(ii)>limap%XRTMIN(5) .AND. pt(ii)<cst%XTT .AND. &
111 pcct(ii)>limap%XCTMIN(2) .AND. pcst(ii)>limap%XCTMIN(5) .AND. odcompute(ii) ) THEN
112!
113 zvec1(ii) = plbds(ii)
114 zvec1w(ii)= ( limac%XFVELOS**limap%XALPHAS + plbds(ii)**limap%XALPHAS ) ** (1./limap%XALPHAS) ! modified equivalent lambda
115!
116! 2. perform the linear interpolation of the normalized
117! "2+LIMAC%XDS"-moment of the incomplete gamma function using the modified equivalent lambda
118!
119 zvec2(ii) = max( 1.0001, min( real(limam%NGAMINC)-0.0001, &
120 limam%XRIMINTP1 * log( zvec1w(ii) ) + limam%XRIMINTP2 ) )
121 ivec2(ii) = int( zvec2(ii) )
122 zvec2(ii) = zvec2(ii) - real( ivec2(ii) )
123!
124 zzw1(ii) = limam%XGAMINC_RIM1( ivec2(ii)+1 )* zvec2(ii) &
125 - limam%XGAMINC_RIM1( ivec2(ii) )*(zvec2(ii) - 1.0)
126!
127! 3. perform the linear interpolation of the normalized
128! "LIMAC%XBS"-moment of the incomplete gamma function
129!
130 zvec2(ii) = max( 1.0001, min( real(limam%NGAMINC)-0.0001, &
131 limam%XRIMINTP1 * log( zvec1(ii) ) + limam%XRIMINTP2 ) )
132 ivec2(ii) = int( zvec2(ii) )
133 zvec2(ii) = zvec2(ii) - real( ivec2(ii) )
134!
135 zzw2(ii) = limam%XGAMINC_RIM2( ivec2(ii)+1 )* zvec2(ii) &
136 - limam%XGAMINC_RIM2( ivec2(ii) )*(zvec2(ii) - 1.0)
137!
138! 4. riming
139!
140 ! Cloud droplets collected
141! total mass loss of cloud droplets, < 0
142 z_rc_rim(ii) = - limam%XCRIMSS * prct(ii) * pcst(ii)*(1+(limac%XFVELOS/plbds(ii))&
143 **limap%XALPHAS)**(-limap%XNUS+limam%XEXCRIMSS/limap%XALPHAS) &
144 * prhodref(ii)**(-limap%XCEXVT+1) * plbds(ii)**limam%XEXCRIMSS
145 p_cc_rim(ii) = z_rc_rim(ii) * (pcct(ii) / prct(ii)) ! Lambda_c**3
146 !
147 ! Cloud droplets collected on small aggregates add to snow
148! P_RS_RIM(II) = - P_RC_RIM(II) * ZZW1(II)
149 z_rs_rim(ii) = -z_rc_rim(ii) * zzw1(ii)
150 p_rc_rimss(ii) = z_rc_rim(ii) * zzw1(ii) ! < 0, loss of mass for rc
151 !
152 ! Cloud droplets collected on large aggregates add to graupel
153 z_rg_rim(ii) = -z_rc_rim(ii) - z_rs_rim(ii)
154 p_rc_rimsg(ii) = z_rc_rim(ii) - p_rc_rimss(ii) ! < 0, loss of mass for rc
155 !
156 IF (limap%LMURAKAMI) THEN
157 ! Graupel formation based on Murakami
158 zvec1(ii) = limam%XGAMINC_RIM4( ivec2(ii)+1 )* zvec2(ii) &
159 - limam%XGAMINC_RIM4( ivec2(ii) )*(zvec2(ii) - 1.0)
160 zzw5(ii) = zvec1(ii)
161 zzw3(ii) = limam%XSRIMCG * prhodref(ii) * pcst(ii) * plbds(ii)**limam%XEXSRIMCG * (1.0 - zzw2(ii))!/(PTSTEP*PRHODREF(II))
162 zzw3(ii) = z_rg_rim(ii)*zzw3(ii)/ &
163 max(1.e-10, & !-20
164 limam%XSRIMCG3*limam%XSRIMCG2*pcst(ii)*prhodref(ii)*plbds(ii)**(limam%XEXSRIMCG2)*(1.-zzw5(ii))- &
165 limam%XSRIMCG3*zzw3(ii))
166 ELSE
167 ! Large aggregates collecting droplets add to graupel (instant process ???)
168 zzw3(ii) = prst(ii)*(1.0 - zzw2(ii))/ptstep
169 END IF
170 !
171 p_rs_rimcg(ii) = zzw3(ii)
172 p_cs_rim(ii) = -zzw3(ii) * pcst(ii)/prst(ii)
173 !
174 p_th_rim(ii) = - z_rc_rim(ii)*(plsfact(ii)-plvfact(ii))
175 ELSE
176 p_th_rim(ii) = 0.
177 p_rc_rimss(ii) = 0.
178 p_rc_rimsg(ii) = 0.
179 p_rs_rimcg(ii) = 0.
180 z_rc_rim(ii) = 0.
181 p_cc_rim(ii) = 0.
182 z_rs_rim(ii) = 0.
183 p_cs_rim(ii) = 0.
184 z_rg_rim(ii) = 0.
185 END IF
186!
187!* Hallett-Mossop ice production (HMS)
188! -----------------------------------
189!
190 IF ( prst(ii)>limap%XRTMIN(5) .AND. prct(ii)>limap%XRTMIN(2) .AND. pt(ii)<limam%XHMTMAX .AND. pt(ii)>limam%XHMTMIN .AND. &
191 pcst(ii)>limap%XCTMIN(5) .AND. pcct(ii)>limap%XCTMIN(2) .AND. odcompute(ii) ) THEN
192!
193 zvec1(ii) = plbdc(ii)
194 zvec2(ii) = max( 1.0001, min( real(limam%NGAMINC)-0.0001, &
195 limam%XHMLINTP1 * log( zvec1(ii) ) + limam%XHMLINTP2 ) )
196 ivec2(ii) = int( zvec2(ii) )
197 zvec2(ii) = zvec2(ii) - real( ivec2(ii) )
198 zvec1(ii) = limam%XGAMINC_HMC( ivec2(ii)+1 )* zvec2(ii) &
199 - limam%XGAMINC_HMC( ivec2(ii) )*(zvec2(ii) - 1.0)
200 zzw4(ii) = zvec1(ii) ! Large droplets
201!
202 IF ( zzw4(ii)<0.99 ) THEN
203 p_ci_hms(ii) = - z_rc_rim(ii) * (pcct(ii)/prct(ii)) * (1.0-zzw4(ii)) * limam%XHM_FACTS * &
204 max( 0.0, min( (pt(ii)-limam%XHMTMIN)/3.0,(limam%XHMTMAX-pt(ii))/2.0 ) ) ! CCHMSI
205!
206 p_ri_hms(ii) = p_ci_hms(ii) * limac%XMNU0 ! RCHMSI
207 p_rs_hms(ii) = - p_ri_hms(ii)
208 ELSE
209 p_ri_hms(ii) = 0.
210 p_ci_hms(ii) = 0.
211 p_rs_hms(ii) = 0.
212 END IF
213 ELSE
214 p_ri_hms(ii) = 0.
215 p_ci_hms(ii) = 0.
216 p_rs_hms(ii) = 0.
217 END IF
218END DO
219!
220!-------------------------------------------------------------------------------
221!
222IF (lhook) CALL dr_hook('LIMA_DROPLETS_RIMING_SNOW', 1, zhook_handle)
223END SUBROUTINE lima_droplets_riming_snow
