154!
155! budget of electric charges
156IF (buconf%LBUDGET_SV .AND. oelec) THEN
157 IF (parami%LSEDIC) CALL budget_store_init_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+1), 'SEDI', pqcs(:, :) * prhodj(:, :))
158 CALL budget_store_init_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+2), 'SEDI', pqrs(:, :) * prhodj(:, :))
159 CALL budget_store_init_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+3), 'SEDI', pqis(:, :) * prhodj(:, :))
160 CALL budget_store_init_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+4), 'SEDI', pqss(:, :) * prhodj(:, :))
161 CALL budget_store_init_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+5), 'SEDI', pqgs(:, :) * prhodj(:, :))
162 IF (krr == 7) CALL budget_store_init_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+6), 'SEDI', pqhs(:, :) * prhodj(:, :))
163END IF
164!
165IF(parami%CSEDIM=='STAT') THEN
166 IF(oelec) THEN
167 CALL print_msg(nverb_fatal, 'GEN', 'RAIN_ICE', "OELEC=.T. not allowed with CSEDIM='STAT'")
168 ENDIF
169!$acc kernels
170!$acc loop independent collapse(2)
171 DO jk = iktb,ikte
172 DO jij = iijb,iije
173 zrct(jij,jk)=prcs(jij,jk)*ptstep
174 zrrt(jij,jk)=prrs(jij,jk)*ptstep
175 zrit(jij,jk)=pris(jij,jk)*ptstep
176 zrst(jij,jk)=prss(jij,jk)*ptstep
177 zrgt(jij,jk)=prgs(jij,jk)*ptstep
178 IF (krr==7) zrht(jij,jk)=prhs(jij,jk)*ptstep
179 ENDDO
180 ENDDO
181!$acc end kernels
182 CALL ice4_sedimentation_stat(d, cst, icep, iced, parami, &
183 &ptstep, krr, pdzz, &
184 &prhodref, ppabst, ptht, pt, prhodj, &
185 &prcs, zrct, prrs, zrrt, pris, zrit,&
186 &prss, zrst, prgs, zrgt,&
187 &pinprc, pinprr, zinpri, pinprs, pinprg, &
188 &psea=psea, ptown=ptown, &
189 &pinprh=pinprh, prht=zrht, prhs=prhs, pfpr=pfpr)
190!$acc kernels
191 pinprs(iijb:iije) = pinprs(iijb:iije) + zinpri(iijb:iije)
192!$acc end kernels
193 !No negativity correction here as we apply sedimentation on PR.S*PTSTEP variables
194ELSEIF(parami%CSEDIM=='SPLI') THEN
195 CALL ice4_sedimentation_split(d, cst, icep, iced, parami, elecp, elecd, &
196 &oelec, osedim_beard, pthvrefzikb, ptstep, krr, pdzz, &
197 &prhodref, ppabst, ptht, pt, prhodj, &
198 &prcs, prct, prrs, prrt, pris, prit, prss, prst, prgs, prgt,&
199 &pinprc, pinprr, zinpri, pinprs, pinprg, &
200 &pqct, pqrt, pqit, pqst, pqgt, pqcs, pqrs, pqis, pqss, pqgs, &
201 &pefieldw, &
202 &psea=psea, ptown=ptown, &
203 &pinprh=pinprh, prht=prht, prhs=prhs, pfpr=pfpr, &
204 &pqht=pqht, pqhs=pqhs)
205!$acc kernels
206 pinprs(iijb:iije) = pinprs(iijb:iije) + zinpri(iijb:iije)
207!$acc end kernels
208 !We correct negativities with conservation
209 !SPLI algorith uses a time-splitting. Inside the loop a temporary m.r. is used.
210 ! It is initialized with the m.r. at T and is modified by two tendencies:
211 ! sedimentation tendency and an external tendency which represents all other
212 ! processes (mainly advection and microphysical processes). If both tendencies
213 ! are negative, sedimentation can remove a species at a given sub-timestep. From
214 ! this point sedimentation stops for the remaining sub-timesteps but the other tendency
215 ! will be still active and will lead to negative values.
216 ! We could prevent the algorithm to not consume too much a species, instead we apply
217 ! a correction here.
218!++cb-- il faudrait faire la correction correspondante sur les charges electriques pour eviter de se retrouver
219! avec des points ou il y a de la charge mais pas de masse !
220 CALL ice4_correct_negativities(d, iced, krr, prvs, prcs, prrs, &
221 &pris, prss, prgs, &
222 &pths, plvfact, plsfact, prhs)
223! CALL ICE4_CORRECT_NEGATIVITIES(D, ICED, KRR, PRVS, PRCS, PRRS, &
224! &PRIS, PRSS, PRGS, &
225! &PQPIS, PQCS, PQRS, PQIS, PQSS, PQGS, PQNIS, &
226! &PTHS, PLVFACT, PLSFACT, PRHS, PQHS)
227ELSEIF(parami%CSEDIM=='NONE') THEN
228ELSE
229 CALL print_msg(nverb_fatal, 'GEN', 'RAIN_ICE', 'no sedimentation scheme for PARAMI%CSEDIM='//parami%CSEDIM)
230END IF
231!
232!
233IF (buconf%LBUDGET_RC .AND. parami%LSEDIC) CALL budget_store_end_phy(d, tbudgets(nbudget_rc), 'SEDI', prcs(:, :) * prhodj(:, :))
234IF (buconf%LBUDGET_RR) CALL budget_store_end_phy(d, tbudgets(nbudget_rr), 'SEDI', prrs(:, :) * prhodj(:, :))
235IF (buconf%LBUDGET_RI) CALL budget_store_end_phy(d, tbudgets(nbudget_ri), 'SEDI', pris(:, :) * prhodj(:, :))
236IF (buconf%LBUDGET_RS) CALL budget_store_end_phy(d, tbudgets(nbudget_rs), 'SEDI', prss(:, :) * prhodj(:, :))
237IF (buconf%LBUDGET_RG) CALL budget_store_end_phy(d, tbudgets(nbudget_rg), 'SEDI', prgs(:, :) * prhodj(:, :))
238IF (buconf%LBUDGET_RH .AND. krr==7) CALL budget_store_end_phy(d, tbudgets(nbudget_rh), 'SEDI', prhs(:, :) * prhodj(:, :))
239!
240! Budget for electric charges
241IF (buconf%LBUDGET_SV .AND. oelec) THEN
242 IF (parami%LSEDIC) CALL budget_store_end_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+1), 'SEDI', pqcs(:, :) * prhodj(:, :))
243 CALL budget_store_end_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+2), 'SEDI', pqrs(:, :) * prhodj(:, :))
244 CALL budget_store_end_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+3), 'SEDI', pqis(:, :) * prhodj(:, :))
245 CALL budget_store_end_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+4), 'SEDI', pqss(:, :) * prhodj(:, :))
246 CALL budget_store_end_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+5), 'SEDI', pqgs(:, :) * prhodj(:, :))
247 IF (krr == 7) CALL budget_store_end_phy(d, tbudgets(nbudget_sv1-1+nsv_elecbeg+6), 'SEDI', pqhs(:, :) * prhodj(:, :))
248END IF
249!
250IF (lhook) CALL dr_hook('ICE4_SEDIMENTATION', 1, zhook_handle)
251!
252END SUBROUTINE ice4_sedimentation
