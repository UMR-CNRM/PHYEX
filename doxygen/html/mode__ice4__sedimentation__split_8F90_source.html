100 REAL, DIMENSION(MERGE(D%NIJT,0,OELEC),MERGE(D%NKT,0,OELEC)), INTENT(IN) :: PQIT ! Pristine ice electric charge at t

113 !
114 !
115 INTEGER :: JIJ, JK
116 INTEGER :: IKTB, IKTE, IKB, IKL, IIJE, IIJB
117 INTEGER :: IRR !Workaround of PGI bug with OpenACC (at least up to 18.10 version)
118 LOGICAL :: GSEDIC !Workaround of PGI bug with OpenACC (at least up to 18.10 version)
119 LOGICAL :: GPRESENT_PFPR, GPRESENT_PSEA
120 REAL :: ZINVTSTEP
121 REAL, DIMENSION(D%NIJT) :: ZCONC_TMP ! Weighted concentration
122 REAL, DIMENSION(D%NIJT,D%NKTB:D%NKTE) :: ZW ! work array
123 REAL, DIMENSION(D%NIJT, D%NKT) :: ZCONC3D, & ! droplet condensation
124  & ZRAY, & ! Cloud Mean radius
125  & ZLBC, & ! XLBC weighted by sea fraction
126  & ZFSEDC, &

139 !-------------------------------------------------------------------------------
140 !
141 IF (lhook) CALL dr_hook('ICE4_SEDIMENTATION_SPLIT', 0, zhook_handle)
142 
143 !-------------------------------------------------------------------------------
144 !
145 !
146 gsedic = parami%LSEDIC
147 irr = krr
148 !
149 iktb=d%NKTB
150 ikte=d%NKTE
151 iijb=d%NIJB
152 iije=d%NIJE
153 !
154 IF (PRESENT(pfpr)) THEN
155  gpresent_pfpr = .true.
156 ELSE
157  gpresent_pfpr = .false.
158 END IF
159 !
160 IF (PRESENT(psea)) THEN
161  gpresent_psea = .true.
162 ELSE
163  gpresent_psea = .false.
164 END IF
165 !
166 IF(krr==7) THEN
167  hcloud='ICE4'
168 ELSE
169  hcloud='ICE3'
170 ENDIF
171 !
172 ! O. Initialization of for sedimentation
173 !
174 zinvtstep=1./ptstep
175 IF (gpresent_pfpr) THEN
176  pfpr(:,:,:) = 0.
177 END IF
178 !
179 !* 1. Parameters for cloud sedimentation
180 !
181 IF (gsedic) THEN
182  zray(:,:) = 0.
183  zlbc(:,:) = iced%XLBC(1)
184  zfsedc(:,:) = icep%XFSEDC(1)
185  zconc3d(:,:)= iced%XCONC_LAND
186  zconc_tmp(:)= iced%XCONC_LAND
187  IF (gpresent_psea) THEN
188  DO jij = iijb, iije
189  zconc_tmp(jij)=psea(jij)*iced%XCONC_SEA+(1.-psea(jij))*iced%XCONC_LAND
190  ENDDO
191  DO jk=iktb, ikte
192  DO jij = iijb, iije
193  zlbc(jij,jk) = psea(jij)*iced%XLBC(2)+(1.-psea(jij))*iced%XLBC(1)
194  zfsedc(jij,jk) = (psea(jij)*icep%XFSEDC(2)+(1.-psea(jij))*icep%XFSEDC(1))
195  zfsedc(jij,jk) = max(min(icep%XFSEDC(1),icep%XFSEDC(2)),zfsedc(jij,jk))
196  zconc3d(jij,jk)= (1.-ptown(jij))*zconc_tmp(jij)+ptown(jij)*iced%XCONC_URBAN
197  zray(jij,jk) = 0.5*((1.-psea(jij))*gamma(iced%XNUC+1.0/iced%XALPHAC)/(gamma(iced%XNUC)) + &
198  & psea(jij)*gamma(iced%XNUC2+1.0/iced%XALPHAC2)/(gamma(iced%XNUC2)))
199  ENDDO
200  END DO
201  ELSE
202  zconc3d(:,:) = iced%XCONC_LAND
203  zray(:,:) = 0.5*(gamma(iced%XNUC+1.0/iced%XALPHAC)/(gamma(iced%XNUC)))
204  END IF
205  DO jk=iktb, ikte
206  DO jij = iijb, iije
207  zray(jij,jk) = max(1.,zray(jij,jk))
208  zlbc(jij,jk) = max(min(iced%XLBC(1),iced%XLBC(2)),zlbc(jij,jk))
209  ENDDO
210  ENDDO
211 ENDIF
212 !
213 !* 2. compute the fluxes
214 !
215 ! optimization by looking for locations where
216 ! the precipitating fields are larger than a minimal value only !!!
217 ! For optimization we consider each variable separately
218 !
219 DO jk=iktb, ikte
220  DO jij = iijb, iije
221  ! External tendecies
222  IF (gsedic) THEN
223  zprcs(jij,jk) = prcs(jij,jk)-prct(jij,jk)*zinvtstep
224  ENDIF
225  zprrs(jij,jk) = prrs(jij,jk)-prrt(jij,jk)*zinvtstep
226  zpris(jij,jk) = pris(jij,jk)-prit(jij,jk)*zinvtstep
227  zprss(jij,jk) = prss(jij,jk)-prst(jij,jk)*zinvtstep
228  zprgs(jij,jk) = prgs(jij,jk)-prgt(jij,jk)*zinvtstep
229  IF ( irr == 7 ) THEN
230  zprhs(jij,jk) = prhs(jij,jk)-prht(jij,jk)*zinvtstep
231  END IF
232  !
233  ! mr values inside the time-splitting loop
234  zrct(jij,jk) = prct(jij,jk)
235  zrrt(jij,jk) = prrt(jij,jk)
236  zrit(jij,jk) = prit(jij,jk)
237  zrst(jij,jk) = prst(jij,jk)
238  zrgt(jij,jk) = prgt(jij,jk)
239  IF (irr==7) THEN
240  zrht(jij,jk) = prht(jij,jk)
241  END IF
242  !
243  zw(jij,jk) =1./(prhodref(jij,jk)* pdzz(jij,jk))
244  !
245  ! Cloud electricity
246  IF (oelec) THEN
247  IF (gsedic) zpqcs(jij,jk) = pqcs(jij,jk) - pqct(jij,jk) * zinvtstep
248  zpqrs(jij,jk) = pqrs(jij,jk) - pqrt(jij,jk) * zinvtstep
249  zpqis(jij,jk) = pqis(jij,jk) - pqit(jij,jk) * zinvtstep
250  zpqss(jij,jk) = pqss(jij,jk) - pqst(jij,jk) * zinvtstep
251  zpqgs(jij,jk) = pqgs(jij,jk) - pqgt(jij,jk) * zinvtstep
252  IF (irr==7) zpqhs(jij,jk) = pqhs(jij,jk) - pqht(jij,jk) * zinvtstep
253  !
254  zqct(jij,jk) = pqct(jij,jk)
255  zqrt(jij,jk) = pqst(jij,jk)
256  zqit(jij,jk) = pqit(jij,jk)
257  zqst(jij,jk) = pqst(jij,jk)
258  zqgt(jij,jk) = pqgt(jij,jk)
259  IF (irr==7) zqht(jij,jk) = pqht(jij,jk)
260  ENDIF
261  ENDDO
262 ENDDO
263 !
264 !
265 !* 2.1 for cloud
266 !
267 IF (gsedic) THEN
268  CALL internal_sedim_spli(d, cst, icep, iced, parami, elecp, elecd, &
269  &krr, oelec, osedim_beard, &
270  &prhodref, zw, pdzz, ppabst, ptht, pt, ptstep, &
271  &2, &
272  &zrct, prcs, pinprc, zprcs, &
273  &zqct, pqcs, zpqcs, pefieldw, &
274  &pray=zray, plbc=zlbc, pfsedc=zfsedc, pconc3d=zconc3d, &
275  &pfpr=pfpr)
276 ENDIF
277 !
278 !* 2.2 for rain
279 !
280  CALL internal_sedim_spli(d, cst, icep, iced, parami, elecp, elecd, &
281  &krr, oelec, osedim_beard, &
282  &prhodref, zw, pdzz, ppabst, ptht, pt, ptstep, &
283  &3, &
284  &zrrt, prrs, pinprr, zprrs, &
285  &zqrt, pqrs, zpqrs, pefieldw, &
286  &pfpr=pfpr)
287 !
288 !* 2.3 for pristine ice
289 !
290  CALL internal_sedim_spli(d, cst, icep, iced, parami, elecp, elecd, &
291  &krr, oelec, osedim_beard, &
292  &prhodref, zw, pdzz, ppabst, ptht, pt, ptstep, &
293  &4, &
294  &zrit, pris, pinpri, zpris, &
295  &zqit, pqis, zpqis, pefieldw, &
296  &pfpr=pfpr)
297 !
298 !* 2.4 for aggregates/snow
299 !
300  CALL internal_sedim_spli(d, cst, icep, iced, parami, elecp, elecd, &
301  &krr, oelec, osedim_beard, &
302  &prhodref, zw, pdzz, ppabst, ptht, pt, ptstep, &
303  &5, &
304  &zrst, prss, pinprs, zprss, &
305  &zqst, pqss, zpqss, pefieldw, &
306  &pfpr=pfpr)
307 !
308 !* 2.5 for graupeln
309 !
310  CALL internal_sedim_spli(d, cst, icep, iced, parami, elecp, elecd, &
311  &krr, oelec, osedim_beard, &
312  &prhodref, zw, pdzz, ppabst, ptht, pt, ptstep, &
313  &6, &
314  &zrgt, prgs, pinprg, zprgs, &
315  &zqgt, pqgs, zpqgs, pefieldw, &
316  &pfpr=pfpr)
317 !
318 !* 2.6 for hail
319 !
320 IF (irr==7) THEN
321  CALL internal_sedim_spli(d, cst, icep, iced, parami, elecp, elecd, &
322  &krr, oelec, osedim_beard, &
323  &prhodref, zw, pdzz, ppabst, ptht, pt, ptstep, &
324  &7, &
325  &zrht, prhs, pinprh, zprhs, &
326  &zqht, pqhs, zpqhs, pefieldw, &
327  &pfpr=pfpr)
328 ENDIF
329 !
330 IF (lhook) CALL dr_hook('ICE4_SEDIMENTATION_SPLIT', 1, zhook_handle)
331 !
332 CONTAINS
