287 SUBROUTINE turbn_init(HPROGRAM, TFILENAM, LDNEEDNAM, KLUOUT, &
288  &LDDEFAULTVAL, LDREADNAM, LDCHECK, KPRINT)
289 !!*** *TURBN_INIT* - Code needed to initialize the MODD_TURB_n module
290 !!
291 !!* PURPOSE
292 !! -------
293 !! Sets the default values, reads the namelist, performs the checks and prints
294 !!
295 !!* METHOD
296 !! ------
297 !! 0. Declarations
298 !! 1. Declaration of arguments
299 !! 2. Declaration of local variables
300 !! 1. Default values
301 !! 2. Namelist
302 !! 3. Checks
303 !! 4. Prints
304 !!
305 !! AUTHOR
306 !! ------
307 !! S. Riette
308 !!
309 !! MODIFICATIONS
310 !! -------------
311 !! Original Feb 2023
312 !-------------------------------------------------------------------------------
313 !
314 !* 0. DECLARATIONS
315 ! ---------------
316 !
317 USE mode_posnam_phy, ONLY: posnam_phy
318 USE mode_check_nam_val, ONLY: check_nam_val_char
319 USE modd_parameters, ONLY: xundef
320 USE modd_io, ONLY: tfiledata
321 
322 !
323 IMPLICIT NONE
324 !
325 !* 0.1. Declaration of arguments
326 ! ------------------------
327 !
328 CHARACTER(LEN=6), INTENT(IN) :: HPROGRAM
329 TYPE(tfiledata), INTENT(IN) :: TFILENAM
330 LOGICAL, INTENT(IN) :: LDNEEDNAM
331 INTEGER, INTENT(IN) :: KLUOUT
332 LOGICAL, OPTIONAL, INTENT(IN) :: LDDEFAULTVAL
333 LOGICAL, OPTIONAL, INTENT(IN) :: LDREADNAM
334 LOGICAL, OPTIONAL, INTENT(IN) :: LDCHECK
335 INTEGER, OPTIONAL, INTENT(IN) :: KPRINT
337 !
338 !* 0.2 Declaration of local variables
339 ! ------------------------------
340 !
341 LOGICAL :: LLDEFAULTVAL, LLREADNAM, LLCHECK, LLFOUND
342 INTEGER :: IPRINT
343 
344 lldefaultval=.true.
345 llreadnam=.true.
346 llcheck=.true.
347 iprint=0
348 IF(PRESENT(lddefaultval)) lldefaultval=lddefaultval
349 IF(PRESENT(ldreadnam )) llreadnam =ldreadnam
350 IF(PRESENT(ldcheck )) llcheck =ldcheck
351 IF(PRESENT(kprint )) iprint =kprint
352 !
353 !* 1. DEFAULT VALUES
354 ! -----------------
355 !
356 IF(lldefaultval) THEN
357  !NOTES ON GENERAL DEFAULTS AND MODEL-SPECIFIC DEFAULTS :
358  !- General default values *MUST* remain unchanged.
359  !- To change the default value for a given application,
360  ! an "IF(HPROGRAM=='...')" condition must be used.
361 
362  ximpl = 1.
363  xtkemin = 0.01
364  xced = xundef
365  xctp = xundef
366  xcadap = 0.5
367  cturblen = 'BL89'
368  cturbdim = '1DIM'
369  lturb_flx =.false.
370  lturb_diag=.false.
371  lsig_conv =.false.
372  lrmc01 =.false.
373  ctom ='NONE'
374  lleonard =.false.
375  xcoefhgradthl = 1.0
376  xcoefhgradrm = 1.0
377  xalthgrad = 2000.0
378  xcldthold = -1.0
379  xlini=0.1 !old value: 10.
380  lharat=.false.
381  lrotate_wind=.false.
382  ltkeminturb=.true.
383  lprojqiturb=.true.
384  lsmooth_prandtl=.true.
385  xminsigs=0.
386  nturbsplit=1
387  lcloudmodiflm = .false.
388  cturblen_cloud = 'DELT'
389  xcoef_ampl_sat = 5.
390  xcei_min = 0.001e-06
391  xcei_max = 0.01e-06
392  lturb_precip =.false.
393  !
394  IF(hprogram=='AROME') THEN
395  xtkemin=1.e-6
396  xlini=0.
397  lprojqiturb=.false.
398  lsmooth_prandtl=.false.
399  ELSEIF(hprogram=='MESONH') THEN
400  lrotate_wind=.true.
401  ltkeminturb=.false.
402  xminsigs=1.e-12
403  ELSEIF(hprogram=='LMDZ') THEN
404  xtkemin=1.e-6
405  xlini=0.
406  ENDIF
407 ENDIF
408 !
409 !* 2. NAMELIST
410 ! -----------
411 !
412 IF(llreadnam) THEN
413  CALL posnam_phy(tfilenam, 'NAM_TURBN', ldneednam, llfound)
414  IF(llfound) READ(unit=tfilenam%NLU, nml=nam_turbn)
415 ENDIF
416 !
417 !* 3. CHECKS
418 ! ---------
419 !
420 IF(llcheck) THEN
421  CALL check_nam_val_char(kluout, 'CTURBDIM', cturbdim, '1DIM', '3DIM')
422  CALL check_nam_val_char(kluout, 'CTURBLEN', cturblen, 'DELT', 'BL89', 'RM17', 'DEAR', 'BLKR', 'HM21')
423  CALL check_nam_val_char(kluout, 'CTOM', ctom, 'NONE', 'TM06')
424  CALL check_nam_val_char(kluout, 'CTURBLEN_CLOUD', cturblen_cloud, 'DELT', 'BL89', 'RM17', 'DEAR', 'BLKR', 'HM21')
425 
426 ENDIF
427 !
428 !* 3. PRINTS
429 ! ---------
430 !
431 IF(iprint>=1) THEN
432  WRITE(unit=kluout,nml=nam_turbn)
433 ENDIF
434 !
435 END SUBROUTINE turbn_init
436 !
