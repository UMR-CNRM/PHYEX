213SUBROUTINE turb_goto_model(KFROM, KTO)
214!! This subroutine associate all the pointers to the right component of
215!! the right strucuture. A value can be accessed through the structure TURBN
216!! or through the strucuture TURB_MODEL(KTO) or directly through these pointers.

219!
220IF(.NOT. ASSOCIATED(turbn, turb_model(kto))) THEN
221!

223!
224! Save current state for allocated arrays
225!
226IF(kfrom>0 .AND. kfrom<=jpmodelmax) THEN
227 !TURB_MODEL(KFROM)%XBL_DEPTH=>XBL_DEPTH !Done in FIELDLIST_GOTO_MODEL
228 !TURB_MODEL(KFROM)%XSBL_DEPTH=>XSBL_DEPTH !Done in FIELDLIST_GOTO_MODEL

238 turb_model(kfrom)%XSSRFL_C=>xssrfl_c
239 turb_model(kfrom)%XCEI=>xcei
240ENDIF
241!

245xced=>turb_model(kto)%XCED
246xctp=>turb_model(kto)%XCTP

306SUBROUTINE turbn_init(HPROGRAM, TFILENAM, LDNEEDNAM, KLUOUT, &
307 &LDDEFAULTVAL, LDREADNAM, LDCHECK, KPRINT)
308!!*** *TURBN_INIT* - Code needed to initialize the MODD_TURB_n module
309!!
310!!* PURPOSE
311!! -------
312!! Sets the default values, reads the namelist, performs the checks and prints
313!!
314!!* METHOD
315!! ------
316!! 0. Declarations
317!! 1. Declaration of arguments
318!! 2. Declaration of local variables
319!! 1. Default values
320!! 2. Namelist
321!! 3. Checks
322!! 4. Prints
323!!
324!! AUTHOR
325!! ------
326!! S. Riette
327!!
328!! MODIFICATIONS
329!! -------------
330!! Original Feb 2023
331!-------------------------------------------------------------------------------
332!
333!* 0. DECLARATIONS
334! ---------------
335!
336USE mode_posnam_phy, ONLY: posnam_phy
337USE mode_check_nam_val, ONLY: check_nam_val_char
338USE modd_parameters, ONLY: xundef
339USE modd_io, ONLY: tfiledata
340
341!
342IMPLICIT NONE
343!
344!* 0.1. Declaration of arguments
345! ------------------------
346!
347CHARACTER(LEN=6), INTENT(IN) :: HPROGRAM
348TYPE(tfiledata), INTENT(IN) :: TFILENAM
349LOGICAL, INTENT(IN) :: LDNEEDNAM
350INTEGER, INTENT(IN) :: KLUOUT
351LOGICAL, OPTIONAL, INTENT(IN) :: LDDEFAULTVAL
352LOGICAL, OPTIONAL, INTENT(IN) :: LDREADNAM
353LOGICAL, OPTIONAL, INTENT(IN) :: LDCHECK
354INTEGER, OPTIONAL, INTENT(IN) :: KPRINT
356!
357!* 0.2 Declaration of local variables
358! ------------------------------
359!
360LOGICAL :: LLDEFAULTVAL, LLREADNAM, LLCHECK, LLFOUND
361INTEGER :: IPRINT
362
363lldefaultval=.true.
364llreadnam=.true.
365llcheck=.true.
366iprint=0
367IF(PRESENT(lddefaultval)) lldefaultval=lddefaultval
368IF(PRESENT(ldreadnam )) llreadnam =ldreadnam
369IF(PRESENT(ldcheck )) llcheck =ldcheck
370IF(PRESENT(kprint )) iprint =kprint
371!
372!* 1. DEFAULT VALUES
373! -----------------
374!
375IF(lldefaultval) THEN
376 !NOTES ON GENERAL DEFAULTS AND MODEL-SPECIFIC DEFAULTS :
377 !- General default values *MUST* remain unchanged.
378 !- To change the default value for a given application,
379 ! an "IF(HPROGRAM=='...')" condition must be used.
380
381 ximpl = 1.
382 xtkemin = 0.01
383 xced = xundef
384 xctp = xundef
385 xcadap = 0.5
386 cturblen = 'BL89'
387 cturbdim = '1DIM'
388 lturb_flx =.false.
389 lturb_diag=.false.
390 lsig_conv =.false.
391 lrmc01 =.false.
392 ctom ='NONE'
393 lleonard =.false.
394 xcoefhgradthl = 1.0
395 xcoefhgradrm = 1.0
396 xalthgrad = 2000.0
397 lgoger=.false.
398 xsmag=0.20
399 xcldthold = -1.0
400 xlini=0.1 !old value: 10.
401 lharat=.false.
402 lbl89top=.false.
403 lrotate_wind=.false.
404 ltkeminturb=.true.
405 lprojqiturb=.true.
406 lsmooth_prandtl=.true.
407 xminsigs=0.
408 nturbsplit=1
409 lcloudmodiflm = .false.
410 cturblen_cloud = 'DELT'
411 xcoef_ampl_sat = 5.
412 xcei_min = 0.001e-06
413 xcei_max = 0.01e-06
414 lturb_precip =.false.
415 ldynmf = .false.
416 lthermmf = .true.
417 !
418 IF(hprogram=='AROME') THEN
419 xtkemin=1.e-6
420 xlini=0.
421 lprojqiturb=.false.
422 lsmooth_prandtl=.false.
423 ELSEIF(hprogram=='MESONH') THEN
424 lrotate_wind=.true.
425 ltkeminturb=.false.
426 xminsigs=1.e-12
427 ELSEIF(hprogram=='LMDZ') THEN
428 xtkemin=1.e-6
429 xlini=0.
430 ENDIF
431ENDIF
432!
433!* 2. NAMELIST
434! -----------
435!
436IF(llreadnam) THEN
437 CALL posnam_phy(tfilenam, 'NAM_TURBN', ldneednam, llfound)
438 IF(llfound) READ(unit=tfilenam%NLU, nml=nam_turbn)
439ENDIF
440!
441!* 3. CHECKS
442! ---------
443!
444IF(llcheck) THEN
445 CALL check_nam_val_char(kluout, 'CTURBDIM', cturbdim, '1DIM', '3DIM')
446 CALL check_nam_val_char(kluout, 'CTURBLEN', cturblen, 'DELT', 'BL89', 'RM17', 'DEAR', 'BLKR', 'HM21')
447 CALL check_nam_val_char(kluout, 'CTOM', ctom, 'NONE', 'TM06')
448 CALL check_nam_val_char(kluout, 'CTURBLEN_CLOUD', cturblen_cloud, 'DELT', 'BL89', 'RM17', 'DEAR', 'BLKR', 'HM21')
449
450ENDIF
451!
452!* 3. PRINTS
453! ---------
454!
455IF(iprint>=1) THEN
456 WRITE(unit=kluout,nml=nam_turbn)
457ENDIF
458!
459END SUBROUTINE turbn_init
