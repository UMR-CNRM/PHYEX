72 ! --------------------------
73 !
74 !
75 p_rs_cmel(:)=0.
76 p_cs_cmel(:)=0.
77 !
78 zw(:) = 0.0
79 WHERE( prst(:)>xrtmin(5) .AND. pcst(:)>xctmin(5) .AND. pt(:)>xtt .AND. ldcompute(:) )
80  zw(:) = prvt(:)*ppres(:)/((xmv/xmd)+prvt(:)) ! Vapor pressure
81  zw(:) = pka(:)*(xtt-pt(:)) + &
82  ( pdv(:)*(xlvtt + ( xcpv - xcl ) * ( pt(:) - xtt )) &
83  *(xestt-zw(:))/(xrv*pt(:)) )
84 !
85 ! compute RSMLT
86 !
87  zw(:) = xfscvmg*max( 0.0,( -zw(:) * pcst(:) * &
88  ( x0deps*plbds(:)**xex0deps + &
89  x1deps*pcj(:)*plbds(:)**xex1deps * &
90  (1+0.5*(xfvelos/plbds(:))**xalphas)**(-xnus+xex1deps/xalphas)) ))
91 ! On ne tient pas compte de la collection de pluie et gouttelettes par la neige si T>0 !!!!
92 ! Note that no heat is exchanged because the graupeln produced are still icy!!!
93  p_rs_cmel(:) = - zw(:)
94  p_cs_cmel(:) = - zw(:) * pcst(:) / prst(:)
95 !
96 END WHERE
