142 SUBROUTINE lmo(D,CST,PUSTAR,PTHETA,PRV,PSFTH,PSFRV,PLMO)

151  REAL, DIMENSION(D%NIJT), INTENT(IN) :: PRV
152  REAL, DIMENSION(D%NIJT), INTENT(IN) :: PSFTH
153  REAL, DIMENSION(D%NIJT), INTENT(IN) :: PSFRV
154  REAL, DIMENSION(D%NIJT),INTENT(OUT) :: PLMO
155 !
156  REAL, DIMENSION(D%NIJT) :: ZTHETAV, ZQ0
157  REAL :: ZEPS
158  INTEGER :: IIJB,IIJE, JIJ,IKT
159 !
160  REAL(KIND=jphook) :: zhook_handle
161  IF (lhook) CALL dr_hook('MODE_SBL:LMO',0,zhook_handle)
162 !
163  iije=d%NIJE
164  iijb=d%NIJB
165  ikt=d%NKT
166  zeps=(cst%XRV-cst%XRD)/cst%XRD
167 !
168  !$mnh_expand_array(JIJ=IIJB:IIJE)
169  zthetav(iijb:iije) = ptheta(iijb:iije) * ( 1. +zeps * prv(iijb:iije))
170  zq0(iijb:iije) = psfth(iijb:iije) + zthetav(iijb:iije) * zeps * psfrv(iijb:iije)
171 !
172  plmo(iijb:iije) = xundef
173  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
174  !$mnh_expand_where(JIJ=IIJB:IIJE)
175  WHERE ( zq0(iijb:iije)/=0. )
176  plmo(iijb:iije) = - max(pustar(iijb:iije),1.e-6)**3 &
177  / ( cst%XKARMAN * cst%XG / zthetav(iijb:iije) *zq0(iijb:iije) )
178  END WHERE
179  !$mnh_end_expand_where(JIJ=IIJB:IIJE)
