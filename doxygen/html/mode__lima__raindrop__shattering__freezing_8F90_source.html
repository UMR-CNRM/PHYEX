70 !
71 !-------------------------------------------------------------------------------

74 !
75 grdsf(:) = lrdsf .AND. ldcompute .AND. (prit(:)>xrtmin(4)) .AND. (prrt(:)>xrtmin(3)) &
76  .AND. (pcit(:)>xctmin(4)) .AND. (pcrt(:)>xctmin(3)) &
77  .AND. (prgt(:)>xrtmin(6))
78 
79 irdsf = count( grdsf(:) )
80 !
81 IF (irdsf > 0) THEN
82 !
83  ALLOCATE(zvec1_r(irdsf))
84  ALLOCATE(zvec1_r1(irdsf))
85  ALLOCATE(zvec2_r(irdsf))
86  ALLOCATE(ivec2_r(irdsf))
87 !
88 !* 2.2.1 select the ZLBDAR
89 !
90  zvec1_r(:) = pack( plbdr(:),mask=grdsf(:) )
91 !
92 !* 2.2.2 find the next lower indice for the ZLBDAR in the
93 ! geometrical set of Lbda_r used to tabulate some moments of the
94 ! incomplete gamma function, for the lower boundary (0.1 mm)
95 !
96  zvec2_r(1:irdsf) = max( 1.00001, min( float(ngaminc)-0.00001,xrdsfintp_r &
97  * log( zvec1_r(1:irdsf) ) + xrdsfintp1_r ) )
98  ivec2_r(1:irdsf) = int( zvec2_r(1:irdsf) )
99  zvec2_r(1:irdsf) = zvec2_r(1:irdsf) - float( ivec2_r(1:irdsf) )
100 !
101 !* 2.2.3 perform the linear interpolation of the
102 ! normalized "2+XDR"-moment of the incomplete gamma function
103 !
104  zvec1_r1(1:irdsf) = xgaminc_rdsf_r(ivec2_r(1:irdsf)+1) * zvec2_r(1:irdsf) &
105  - xgaminc_rdsf_r(ivec2_r(1:irdsf)) * (zvec2_r(1:irdsf) - 1.0)
106 !
107 ! From 0.1 mm to infinity we need
108  zvec1_r1(1:irdsf) = xmomgr_rdsf * (1.0 - zvec1_r1(1:irdsf))
109 !
110  zintg_rain(:) = unpack( vector=zvec1_r1(:),mask=grdsf,field=0.0 )
111 !
112 !* 2.2.4 To compute final "RDSF" contributions
113 !
114  zni_rdsf(:) = (xfactor_rdsf_ni / (prhodref(:)**(xcexvt-1.0))) * ( &
115  pcit(:) * pcrt(:) * zintg_rain(:) * plbdr(:)**(-(xdr+6.0)) )
116 !
117  p_ci_rdsf(:) = max(zni_rdsf(:),0.)
118 !
119 ! The value of rg removed by RDSF is determined by the mean mass of pristine ice
120  zri_rdsf(:) = max( zni_rdsf(:)*xmnu0,xrtmin(5) )
121 !
122  p_ri_rdsf(:) = zri_rdsf(:)
123 !
124  DEALLOCATE(zvec1_r)
125  DEALLOCATE(zvec1_r1)
126  DEALLOCATE(zvec2_r)
127  DEALLOCATE(ivec2_r)
128  !
129 ENDIF
