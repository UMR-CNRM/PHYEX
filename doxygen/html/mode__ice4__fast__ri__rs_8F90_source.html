105prilars(:) = 0.
106zwcitred(:)= 0.1 ! ratio of ice crystal concentration wet to dry

108DO jl=1, ksize
109 IF( .NOT. ldsoft) THEN
110 zzw(jl) = 0.
111 IF(parami%LMODICEDEP .AND. ztimesc > 0. )THEN
112 ! Turn ice to snow if ice crystal distribution is such that
113 ! the ice crystal diameter for the (mass x N_i) maximum
114 ! is larger than a prescribed size,
115 ! (ZDICRIT).
116 zzwc(jl) = (icenumber3(prit(jl),pt(jl))+pcit(jl))* &
117 prhodref(jl)
118
119 IF (prit(jl)>0.0 .AND. zzwc(jl)>0.0) THEN
120 zzw(jl) = min(1.e8,iced%XLBI*( prhodref(jl)*prit(jl)/zzwc(jl) )**iced%XLBEXI) ! LAMBDA for ICE
121 zzw(jl) = min(0.5, 1. - 0.5**( zkvo /zzw(jl)))
122 zzw(jl) = min(0.9*prit(jl), zzw(jl)*prit(jl))
123 prilars(jl) = zzw(jl)
124 ENDIF
125 ELSE
126 ! Turn ice crystals lagrer than a precribed size into snow:
127 ! (For the moment sperical ice crystals are assumed)
128 IF ( (prit(jl)>0.0) .AND.(pssi(jl)>0.001) ) THEN
129 zzw(jl) = min(prit(jl),prit(jl)/(87.5*(zdicrit)**2*pai(jl)/ pssi(jl)))
130 prilars(jl) = zzw(jl)
131 ENDIF
132 ENDIF
133 ENDIF
134ENDDO

136prvdepi(:) = 0.
137zzw(:) = 0.
138zw2d(:) = 1./(pifr(:)*picldfr(:) + 1. -picldfr(:))
139zzwc(:) = 0.
140DO jl=1, ksize
141 IF( .NOT. ldsoft) THEN
142 IF(parami%LMODICEDEP)THEN
143 zzwc(jl) = (icenumber3(prit(jl),pt(jl))+pcit(jl))* &
144 prhodref(jl)
145 IF( (ppres(jl)*0.5 - pesi(jl))>0. .AND. zzwc(jl)>0. .AND. ldcompute(jl)) THEN

147 zwcitred23(jl)=zwcitred(jl)**(1.+ iced%XLBEXI)
148 zzw(jl)= icep%X0DEPI/(iced%XLBI*pai(jl)) *(zzwc(jl)/prhodref(jl))**(1.+iced%XLBEXI) * &
149 & (max(iced%XRTMIN(4),prit(jl))*zw2d(jl) )**(-iced%XLBEXI)
150 zzw(jl)= zzw(jl)* ( pssio(jl)* picldfr(jl)* zxw2d13(jl) + zwcitred23(jl)*pssiu(jl)* &
151 & (1.-picldfr(jl)) )
152 ENDIF
153 prvdepi(jl) = prvdepi(jl) + zzw(jl)
154 ELSE
155 IF ( (ppres(jl)*0.5 - pesi(jl))>0. .AND. pcit(jl)>0. .AND. ldcompute(jl) ) THEN
156 zxw2d13(jl)=pifr(jl)**0.333
157 zwcitred23(jl)=zwcitred(jl)**0.667
158 ztc = max(-18.,min(-1.,pt(jl)-cst%XTT))
159 zhu = min(0.15,max(0.,pssi(jl)))
160 zcrysha(jl)=1.1+ 3.*zhu*(1.+ sin(0.64*ztc -1.3))
161 ! icedensity*4/3 *pi /8. =366.5 ; icedensity=700 kg/m3
162 zqimax = 366.5 * zdicrit**3 * pcit(jl)*zwcitred(jl)/prhodref(jl)
163 zci2s(jl) = 0.
164 IF(prit(jl) > 1.0e-12 .AND. ztimesc > 0. )THEN
165 zci2s(jl) = prit(jl)*(1. - min(1., 0.5*zqimax /prit(jl)))* &
166 & (1.-picldfr(jl))*zw2d(jl)/(ztimesc*0.5)
167 ENDIF
168
169 zzwc(jl)=zcrysha(jl)*0.878/pai(jl)*(pcit(jl)/prhodref(jl))**0.667 &
170 &*(max(iced%XRTMIN(4),prit(jl))*zw2d(jl))**0.333
171 ! Ice supersaturated part of grid box:
172 IF( pssio(jl)>0. .AND. picldfr(jl) > 0.02 ) THEN
173 zzw(jl) = zzwc(jl)*zxw2d13(jl)*pssio(jl)*picldfr(jl)
174 prvdepi(jl) = prvdepi(jl) + zzw(jl)
175 ENDIF
176 ! Ice subsaturated part of grid box:
177 IF ( pssiu(jl)<0. .AND. picldfr(jl) <0.98 ) THEN
178 prilars(jl) = prilars(jl) + zci2s(jl)
179 zzw(jl) = zzwc(jl)*zwcitred23(jl)*pssiu(jl)*(1. - picldfr(jl))
180 prvdepi(jl) = prvdepi(jl) + zzw(jl)
181 ENDIF
182 ENDIF
183 ENDIF
184 ENDIF
185ENDDO

192 FUNCTION icenumber3 (Q_ICE, T3D) RESULT (RET)

201 !+---+-----------------------------------------------------------------+
202 !..Table of lookup values of radiative effective radius of ice crystals
203 !.. as a function of Temperature from -94C to 0C. Taken from WRF RRTMG
204 !.. radiation code where it is attributed to Jon Egill Kristjansson
205 !.. and coauthors.
206 !+---+-----------------------------------------------------------------+
207
208 REAL, DIMENSION(95), PARAMETER :: retab = (/ &
209 5.92779, 6.26422, 6.61973, 6.99539, 7.39234, &
210 7.81177, 8.25496, 8.72323, 9.21800, 9.74075, 10.2930, &
211 10.8765, 11.4929, 12.1440, 12.8317, 13.5581, 14.2319, &
212 15.0351, 15.8799, 16.7674, 17.6986, 18.6744, 19.6955, &
213 20.7623, 21.8757, 23.0364, 24.2452, 25.5034, 26.8125, &
214 27.7895, 28.6450, 29.4167, 30.1088, 30.7306, 31.2943, &
215 31.8151, 32.3077, 32.7870, 33.2657, 33.7540, 34.2601, &
216 34.7892, 35.3442, 35.9255, 36.5316, 37.1602, 37.8078, &
217 38.4720, 39.1508, 39.8442, 40.5552, 41.2912, 42.0635, &
218 42.8876, 43.7863, 44.7853, 45.9170, 47.2165, 48.7221, &
219 50.4710, 52.4980, 54.8315, 57.4898, 60.4785, 63.7898, &
220 65.5604, 71.2885, 75.4113, 79.7368, 84.2351, 88.8833, &
221 93.6658, 98.5739, 103.603, 108.752, 114.025, 119.424, &
222 124.954, 130.630, 136.457, 142.446, 148.608, 154.956, &
223 161.503, 168.262, 175.248, 182.473, 189.952, 197.699, &
224 205.728, 214.055, 222.694, 231.661, 240.971, 250.639/)
225
226 !+---+-----------------------------------------------------------------+
227 !..From the model 3D temperature field, subtract 179K for which
228 !.. index value of retab as a start. Value of corr is for
229 !.. interpolating between neighboring values in the table.
230 !+---+-----------------------------------------------------------------+
231
232 idx_rei = int(t3d-179.)
233 idx_rei = min(max(idx_rei,1),95)
234 corr = t3d - int(t3d)
235 reice = retab(idx_rei)*(1.-corr) + retab(min(95,idx_rei+1))*corr
236 deice = 2.*reice * 1.e-6
237
238 !+---+-----------------------------------------------------------------+
239 !..Now we have the final radiative effective size of ice (as function
240 !.. of temperature only). This size represents 3rd moment divided by
241 !.. second moment of the ice size distribution, so we can compute a
242 !.. number concentration from the mean size and mass mixing ratio.
243 !.. The mean (radiative effective) diameter is 3./Slope for an inverse
244 !.. exponential size distribution. So, starting with slope, work
245 !.. backwords to get number concentration.
246 !+---+-----------------------------------------------------------------+
247
248 lambda = 3.0 / deice
249 ret = q_ice * lambda*lambda*lambda / (pi*ice_density)
250
251 !+---+-----------------------------------------------------------------+
252 !..Example1:Common ice size coming from Thompson scheme is about 30 microns.
253 !.. An example ice mixing ratio could be 0.001 g/kg for a temp. of -50C.
254 !.. Remember to convert both into MKS units. This gives N_ice=357652 per kg.
255 !..Example2: Lower in atmosphere at T=-10C matching ~162 microns in retab,
256 !.. and assuming we have 0.1 g/kg mixing ratio, then N_ice=28122 per kg,
257 !.. which is 28 crystals per liter of air if the air density is 1.0.
258 !+---+-----------------------------------------------------------------+
259
260 END FUNCTION icenumber3
