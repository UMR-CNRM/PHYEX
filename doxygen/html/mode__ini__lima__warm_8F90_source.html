36!
37!* 0. DECLARATIONS
38! ------------
39!

55REAL, INTENT(IN) :: PTSTEP ! Effective Time step
56REAL, INTENT(IN) :: PDZMIN ! minimun vertical mesh size
57!
58!* 0.2 Declarations of local variables :
59!
60INTEGER :: IKB ! Coordinates of the first and last physical
61 ! points along z
62INTEGER :: I1 ! Internal loop indexes
63INTEGER :: IMOD ! Internal loop to index the CCN modes
64!
65REAL, DIMENSION(6) :: ZGAMC, ZGAMR ! parameters involving various moments of
66 ! the generalized gamma law
67!
68REAL :: ZTT ! Temperature in Celsius
69REAL :: ZLV ! Latent heat of vaporization
70REAL :: ZSS ! Supersaturation
71REAL :: ZG ! G function
72REAL :: ZRHO00 ! Surface reference air density
73REAL :: ZSURF_TEN ! Water drop surface tension
74REAL :: ZSMIN, ZSMAX ! Minimal and maximal supersaturation used to
75 ! discretize the HYP functions
76REAL(KIND=jphook) :: zhook_handle
77!
78!
79!INTEGER :: ILUOUT0 ! Logical unit number for output-listing
80!INTEGER :: IRESP ! Return code of FM-routines
81!LOGICAL :: GFLAG ! Logical flag for printing the constatnts on the output
82 ! listing

85!
86IF (lhook) CALL dr_hook('INI_LIMA_WARM', 0, zhook_handle)
87CALL param_lima_warm_associate()
88!
89!* 1. CHARACTERISTICS OF THE SPECIES
90! ------------------------------
91!
92!
93!* 1.1 Cloud droplet characteristics
94!
95xac = (xpi/6.0)*xrholw
96xbc = 3.0
97xcc = xrholw*xg/(18.0*1.816e-5) ! Stokes flow (Pruppacher eq. 10-138 for T=293K)
98!XCC = XRHOLW*XG/(18.0*1.7E-5) ! Stokes flow (Pruppacher eq. 10-138 for T=273K)
99xdc = 2.0
100!
101xf0c = 1.00
102xf2c = 0.108
103!
104xc1c = 1./2.
105!
106!* 1.2 Raindrops characteristics
107!
108xar = (xpi/6.0)*xrholw
109xbr = 3.0
110xcr = 842.
111xdr = 0.8
112!
113xf0r = 0.780
114!Correction BVIE Pruppacher 1997 eq. 13-61
115!XF1R = 0.265
116xf1r = 0.308
117!
118!
119!------------------------------------------------------------------------------
120!
121!
122!* 2. DIMENSIONAL DISTRIBUTIONS OF THE SPECIES
123! ----------------------------------------
124!
125!
126!* 2.1 Cloud droplet distribution
127!
128!XALPHAC = 3.0 ! Gamma law of the Cloud droplet (here volume-like distribution)
129!XNUC = 3.0 ! Gamma law with little dispersion
130!
131!* 2.2 Raindrop distribution
132!
133!XALPHAR = 3.0 ! Gamma law of the raindrops (here volume-like distribution)
134!XNUR = 3.0 ! Gamma law for the raindrops
135!XNUR = 0.1
136!
137!* 2.3 Precalculation of the gamma function momentum
138!
139!
140zgamc(1) = gamma_x0d(xnuc)
141zgamc(2) = momg(xalphac,xnuc,3.)
142zgamc(3) = momg(xalphac,xnuc,6.)
143zgamc(4) = zgamc(3)-zgamc(2)**2 ! useful for Sig_c
144zgamc(5) = momg(xalphac,xnuc,9.)
145zgamc(6) = momg(xalphac,xnuc,3.)**(2./3.)/momg(xalphac,xnuc,2.)
146!
147zgamr(1) = gamma_x0d(xnur)
148zgamr(2) = momg(xalphar,xnur,3.)
149zgamr(3) = momg(xalphar,xnur,6.)
150zgamr(4) = momg(xalphar,xnur,6.)
151zgamr(5) = momg(xalphar,xnur,9.)
152zgamr(6) = momg(xalphar,xnur,3.)**(2./3.)/momg(xalphar,xnur,2.)
153!
154!* 2.4 Csts for the shape parameter
155!
156xlbc = xar*zgamc(2)
157xlbexc = 1.0/xbc
158!
159xnr = 1.0/(xar*momg(xalphar,xnur,xbr))
160xccr = 8.e6
161xcxr = -1.
162IF (nmom_r.EQ.1) THEN
163 xlbexr = 1.0/(xcxr-xbr)
164 xlbr = ( xar*xccr*momg(xalphar,xnur,xbr) )**(-xlbexr)
165ELSE
166 xlbr = xar*zgamr(2)
167 xlbexr = 1.0/xbr
168END IF
169!
170!
171!------------------------------------------------------------------------------
172!
173!
174!* 3. CONSTANTS FOR THE SEDIMENTATION
175! -------------------------------
176!
177!
178!* 4.1 Exponent of the fall-speed air density correction
179!
180ikb = 1 + jpvext
181! Correction
182!ZRHO00 = XP00/(XRD*XTHVREFZ(IKB))
183zrho00 = 1.2041 ! at P=1013.25hPa and T=20°C
184!
185!* 4.2 Constants for sedimentation
186!
187xfsedrr = xcr*gamma_x0d(xnur+(xdr+3.)/xalphar)/gamma_x0d(xnur+3./xalphar)* &
188 (zrho00)**xcexvt
189xfsedcr = xcr*gamma_x0d(xnur+xdr/xalphar)/gamma_x0d(xnur)* &
190 (zrho00)**xcexvt
191xfsedrc = xcc*gamma_x0d(xnuc+(xdc+3.)/xalphac)/gamma_x0d(xnuc+3./xalphac)* &
192 (zrho00)**xcexvt
193xfsedcc = xcc*gamma_x0d(xnuc+xdc/xalphac)/gamma_x0d(xnuc)* &
194 (zrho00)**xcexvt
195
196!
197xlb(2) = xlbc
198xlbex(2) = xlbexc
199xd(2) = xdc
200xfsedr(2) = xfsedrc
201xfsedc(2) = xfsedcc
202!
203xlb(3) = xlbr
204xlbex(3) = xlbexr
205xd(3) = xdr
206xfsedr(3) = xfsedrr
207xfsedc(3) = xfsedcr
208!
209xgcc = 0.5*gamma_x0d(xnuc+1./xalphac)/gamma_x0d(xnuc)
210!------------------------------------------------------------------------------
211!
212!
213!* 4. CONSTANTS FOR THE NUCLEATION PROCESS
214! ------------------------------------
215!
216!
217xwmin = 0.01 ! Minimal positive vertical velocity required
218 ! for the activation process in Twomey and CPB scheme
219xtmin = -0.000278 ! Minimal cooling required 1K/h
220!
221xdiva = 226.e-7 ! Diffusivity of water vapor in the air
222xthco = 24.3e-3 ! Air thermal conductivity
223!
224! ( 8 Mw (Sigma)sw )3 Pi*Rho_l
225! XCSTDCRIT = ( -------------- ) * --------
226! ( 3 Ra Rhow ) 6
227!
228zsurf_ten = 76.1e-3 ! Surface tension of a water drop at T=0 C
229xcstdcrit = (xpi/6.)*xrholw*( (8.0*zsurf_ten )/( 3.0*xrv*xrholw ) )**3
230!
231!
232!
233! 4.1 Tabulation of the hypergeometric functions in 'no units'
234! --------------------------------------------------------
235!
236! In LIMA's nucleation parameterization,
237! supersaturation is not in % : Smax=0.01 for a 1% supersaturation.
238! This is accounted for in the modified Beta and C values.
239!
240! Here, we tabulate the
241! F(mu,k/2, k/2+1 ,-Beta S**2) -> XHYPF12
242! F(mu,k/2,(k+3)/2,-Beta S**2) -> XHYPF32 functions
243! using a logarithmic scale for S
244!
245nhyp = 500 ! Number of points for the tabulation
246CALL param_lima_warm_allocate('XHYPF12', nhyp, nmod_ccn)
247CALL param_lima_warm_allocate('XHYPF32', nhyp, nmod_ccn)
248!
249zsmin = 1.0e-5 ! Minimum supersaturation set at 0.001 %
250zsmax = 5.0e-2 ! Maximum supersaturation set at 5 %
251xhypintp1 = real(nhyp-1)/log(zsmax/zsmin)
252xhypintp2 = real(nhyp)-xhypintp1*log(zsmax)
253!
254DO imod = 1,nmod_ccn
255 DO i1 = 1,nhyp
256 zss =zsmax*(zsmin/zsmax)**(real(nhyp-i1)/real(nhyp-1))
257 xhypf12(i1,imod) = hypgeo(xmuhen_multi(imod),0.5*xkhen_multi(imod),&
258 0.5*xkhen_multi(imod)+1.0,xbetahen_multi(imod),zss)
259 xhypf32(i1,imod) = hypgeo(xmuhen_multi(imod),0.5*xkhen_multi(imod),&
260 0.5*xkhen_multi(imod)+1.5,xbetahen_multi(imod),zss)
261 END DO
262ENDDO
263!
264nahen = 81 ! Tabulation for each Kelvin degree in the range XTT-40 to XTT+40
265xahenintp1 = 1.0
266xahenintp2 = 0.5*real(nahen-1) - xtt
267!
268! Compute the tabulation of function of T :
269!
270! 1
271! XAHENG = -----------------------
272! XCSTHEN * G**(3/2)
273!
274! Compute constants for the calculation of Smax.
275! XCSTHEN = 1/(rho_l 2 pi)
276! PSI1
277! PSI3
278! T
279! Lv
280! G
281!
282CALL param_lima_warm_allocate('XAHENG', nahen)
283CALL param_lima_warm_allocate('XAHENG2', nahen)
284CALL param_lima_warm_allocate('XAHENG3', nahen)
285CALL param_lima_warm_allocate('XPSI1', nahen)
286CALL param_lima_warm_allocate('XPSI3', nahen)
287xcsthen = 1.0 / ( xrholw*2.0*xpi )
288DO i1 = 1,nahen
289 ztt = xtt + real(i1-(nahen-1)/2) ! T
290 zlv = xlvtt+(xcpv-xcl)*(ztt-xtt) ! Lv
291 xpsi1(i1) = (xg/(xrd*ztt))*(xmv*zlv/(xmd*xcpd*ztt)-1.) ! Psi1
292 xpsi3(i1) = -1*xmv*zlv/(xmd*xrd*(ztt**2)) ! Psi3
293 zg = 1./( xrholw*( (xrv*ztt)/ & ! G
294 (xdiva*exp(xalpw-(xbetaw/ztt)-(xgamw*log(ztt)))) &
295 + (zlv/ztt)**2/(xthco*xrv) ) )
296 xaheng(i1) = xcsthen/(zg)**(3./2.)
297 xaheng2(i1) = 1/(zg)**(1./2.) * gamma_x0d(xnuc+1./xalphac)/gamma_x0d(xnuc)
298 xaheng3(i1) = (zg) * gamma_x0d(xnuc+1./xalphac)/gamma_x0d(xnuc)
299END DO
300!-------------------------------------------------------------------------------
301!
302! Parameters used to initialise the droplet and drop concentration
303! from the respective mixing ratios (used in RESTART_RAIN_C2R2)
304!
305! Droplet case
306!
307!!ALLOCATE(XCONCC_INI(SIZE(PNFS,1),SIZE(PNFS,2),SIZE(PNFS,3),SIZE(PNFS,4))) !NMOD_CCN))
308!! XCONCC_INI(:,:,:,:) = 0.8 * PNFS(:,:,:,:) ! 80% of the maximum CCN conc. is assumed
309!
310! Raindrop case
311!
312xconcr_param_ini = (1.e7)**3/(xpi*xrholw) ! MP law with N_O=1.E7 m-1 is assumed
313!
314!
315!------------------------------------------------------------------------------
316!
317!
318!* 5. CONSTANTS FOR THE COALESCENCE PROCESSES
319! ---------------------------------------
320!
321!
322!* 6.1 Csts for the coalescence processes
323!
324xkera1 = 2.59e15 ! From Long a1=9.44E9 cm-3 so XKERA1= 9.44E9*1E6*(PI/6)**2
325xkera2 = 3.03e3 ! From Long a2=5.78E3 so XKERA2= 5.78E3* (PI/6)
326!
327! Cst for the cloud droplet selfcollection process
328!
329xselfc = xkera1*zgamc(3)
330!
331! Cst for the autoconversion process
332!
333xauto1 = 6.25e18*(zgamc(2))**(1./3.)*sqrt(zgamc(4))
334xauto2 = 0.5e6*(zgamc(4))**(1./6.)
335xlautr = 2.7e-2
336xlautr_threshold = 0.4
337xitautr= 0.27 ! (Notice that T2 of BR74 is uncorrect and that 0.27=1./3.7
338xitautr_threshold = 7.5
339xcautr = 3.5e9
340xr0 = 25.0e-6
341!
342! Cst for the accretion process
343!
344xaccr1 = zgamr(2)**(1./3.)
345xaccr2 = 5.0e-6
346xaccr3 = 12.6e-4
347xaccr4 = xauto2
348xaccr5 = 3.5
349xaccr6 = 1.2*xcautr
350xaccr_clarge1 = xkera2*zgamc(2)
351xaccr_clarge2 = xkera2*zgamr(2)
352xaccr_rlarge1 = xkera2*zgamc(3)*xrholw*(xpi/6.0)
353xaccr_rlarge2 = xkera2*zgamc(2)*zgamr(2)*xrholw*(xpi/6.0)
354xaccr_csmall1 = xkera1*zgamc(3)
355xaccr_csmall2 = xkera1*zgamr(3)
356xaccr_rsmall1 = xkera1*zgamc(5)*xrholw*(xpi/6.0)
357xaccr_rsmall2 = xkera1*zgamc(2)*zgamr(3)*xrholw*(xpi/6.0)
358!
359! ICE3 accretion of cloud droplets by rain drops
360!
361xfcaccr = (xpi/4.0)*xccr*xcr*(zrho00**xcexvt)*momg(xalphar,xnur,xdr+2.0)
362xexcaccr = -xdr-3.0
363!
364! Cst for the raindrop self-collection/breakup process
365!
366xscbu2 = xkera2*zgamr(2)
367xscbu3 = xkera1*zgamr(3)
368xscbu_eff1 = 0.6e-3
369xscbu_eff2 = 2.0e-3
370xscbuexp1 = -2500.0
371!
372!
373!------------------------------------------------------------------------------
374!
375!
376!* 6. CONSTANTS FOR THE "SONTANEOUS" BREAK-UP
377! ---------------------------------------
378!
379!
380xsponbud1 = 3.0e-3
381xsponbud2 = 4.0e-3
382xsponbud3 = 5.0e-3
383xsponcoef2 = ((xsponbud3/xsponbud2)**3 - 1.0)/(xsponbud3-xsponbud1)**2
384!
385!
386!------------------------------------------------------------------------------
387!
388!
389!* 7. CONSTANTS FOR EVAPORATION PROCESS
390! ---------------------------------------
391!
392!
393x0cndc = (4.0*xpi)*xc1c*xf0c*momg(xalphac,xnuc,1.)
394x2cndc = (4.0*xpi)*xc1c*xf2c*xcc*momg(xalphac,xnuc,xdc+2.0)
395!
396! Valeurs utiles pour le calcul de l'évaporation en fonction de N_r
397!
398!XEX0EVAR = -1.0
399!XEX1EVAR = -1.0 - (XDR+1.0)*0.5
400!XEX2EVAR = -0.5*XCEXVT
401!
402!X0EVAR = (2.0*XPI)*XF0R*GAMMA_X0D(XNUR+1./XALPHAR)/GAMMA_X0D(XNUR)
403!X1EVAR = (2.0*XPI)*XF1R*((ZRHO00)**(XCEXVT)*(XCR/0.15E-4))**0.5* &
404! GAMMA_X0D(XNUR+(XDR+3.0)/(2.0*XALPHAR))/GAMMA_X0D(XNUR)
405!
406!
407! Valeurs utiles pour le calcul de l'évaporation en fonction de r_r
408!
409xex0evar = 2.0
410xex1evar = 2.0 - (xdr+1.0)*0.5
411xex2evar = -0.5*xcexvt
412!
413x0evar = (12.0)*xf0r*gamma_x0d(xnur+1./xalphar)/gamma_x0d(xnur+3./xalphar)
414x1evar = (12.0)*xf1r*((zrho00)**(xcexvt)*(xcr/0.15e-4))**0.5* &
415 gamma_x0d(xnur+(xdr+3.0)/(2.0*xalphar))/gamma_x0d(xnur+3./xalphar)
416!
417xcevap = 0.86
418!
419!------------------------------------------------------------------------------
420!
421!
422!* 8. SET-UP RADIATIVE PARAMETERS
423! ---------------------------
424!
425!
426! R_eff_c = XFREFFC * (rho*r_c/N_c)**(1/3)
427!
428!
429xfreffc = 0.5 * zgamc(6) * (1.0/xac)**(1.0/3.0)
430xfreffr = 0.5 * zgamr(6) * (1.0/xar)**(1.0/3.0)
431!
432! Coefficients used to compute reff when both cloud and rain are present
433!
434xcrec = 1.0/ (zgamc(6) * xac**(2.0/3.0))
435xcrer = 1.0/ (zgamr(6) * xar**(2.0/3.0))
436!
437!
438!------------------------------------------------------------------------------
439!
440!
441!* 9. SOME PRINTS FOR CONTROL
442! -----------------------
443!
444!
445!!$GFLAG = .TRUE.
446!!$IF (GFLAG) THEN
447!!$ ILUOUT0 = TLUOUT0%NLU
448!!$ WRITE(UNIT=ILUOUT0,FMT='(" Summary of the cloud particule characteristics")')
449!!$ WRITE(UNIT=ILUOUT0,FMT='(" CLOUD")')
450!!$ WRITE(UNIT=ILUOUT0,FMT='(" masse: A=",E13.6," B=",E13.6)') &
451!!$ XAR,XBR
452!!$ WRITE(UNIT=ILUOUT0,FMT='(" vitesse: C=",E13.6," D=",E13.6)') &
453!!$ XCC,XDC
454!!$ WRITE(UNIT=ILUOUT0,FMT='(" distribution:AL=",E13.6,"NU=",E13.6)') &
455!!$ XALPHAC,XNUC
456!!$ WRITE(UNIT=ILUOUT0,FMT='(" RAIN")')
457!!$ WRITE(UNIT=ILUOUT0,FMT='(" masse: A=",E13.6," B=",E13.6)') &
458!!$ XAR,XBR
459!!$ WRITE(UNIT=ILUOUT0,FMT='(" vitesse: C=",E13.6," D=",E13.6)') &
460!!$ XCR,XDR
461!!$ WRITE(UNIT=ILUOUT0,FMT='(" distribution:AL=",E13.6,"NU=",E13.6)') &
462!!$ XALPHAR,XNUR
463!!$ WRITE(UNIT=ILUOUT0,FMT='(" Description of the nucleation spectrum")')
464!!$ WRITE(UNIT=ILUOUT0,FMT='(" C=",E13.6," k=",E13.6)') XCHEN, XKHEN
465!!$ WRITE(UNIT=ILUOUT0,FMT='(" Beta=",E13.6," MU=",E13.6)') XBETAHEN, XMUHEN
466!!$ WRITE(UNIT=ILUOUT0,FMT='(" CCN max=",E13.6)') XCONC_CCN
467!!$END IF
468!
469!------------------------------------------------------------------------------
470!
471IF (lhook) CALL dr_hook('INI_LIMA_WARM', 1, zhook_handle)
472END SUBROUTINE ini_lima_warm
