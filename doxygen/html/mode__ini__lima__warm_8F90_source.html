87 ! ------------------------------
88 !
89 !
90 !* 1.1 Cloud droplet characteristics
91 !
92 xac = (xpi/6.0)*xrholw
93 xbc = 3.0
94 xcc = xrholw*xg/(18.0*1.816e-5) ! Stokes flow (Pruppacher eq. 10-138 for T=293K)
95 !XCC = XRHOLW*XG/(18.0*1.7E-5) ! Stokes flow (Pruppacher eq. 10-138 for T=273K)

97 !

104 !
105 xar = (xpi/6.0)*xrholw
106 xbr = 3.0
107 xcr = 842.
108 xdr = 0.8
109 !
110 xf0r = 0.780
111 !Correction BVIE Pruppacher 1997 eq. 13-61
112 !XF1R = 0.265
113 xf1r = 0.308
114 !
115 !
116 !------------------------------------------------------------------------------
117 !
118 !
119 !* 2. DIMENSIONAL DISTRIBUTIONS OF THE SPECIES
120 ! ----------------------------------------
121 !
122 !
123 !* 2.1 Cloud droplet distribution
124 !
125 !XALPHAC = 3.0 ! Gamma law of the Cloud droplet (here volume-like distribution)
126 !XNUC = 3.0 ! Gamma law with little dispersion
127 !
128 !* 2.2 Raindrop distribution
129 !
130 !XALPHAR = 3.0 ! Gamma law of the raindrops (here volume-like distribution)
131 !XNUR = 3.0 ! Gamma law for the raindrops
132 !XNUR = 0.1
133 !
134 !* 2.3 Precalculation of the gamma function momentum
135 !
136 !
137 zgamc(1) = gamma_x0d(xnuc)
138 zgamc(2) = momg(xalphac,xnuc,3.)
139 zgamc(3) = momg(xalphac,xnuc,6.)
140 zgamc(4) = zgamc(3)-zgamc(2)**2 ! useful for Sig_c
141 zgamc(5) = momg(xalphac,xnuc,9.)
142 zgamc(6) = momg(xalphac,xnuc,3.)**(2./3.)/momg(xalphac,xnuc,2.)
143 !
144 zgamr(1) = gamma_x0d(xnur)
145 zgamr(2) = momg(xalphar,xnur,3.)
146 zgamr(3) = momg(xalphar,xnur,6.)
147 zgamr(4) = momg(xalphar,xnur,6.)
148 zgamr(5) = momg(xalphar,xnur,9.)
149 zgamr(6) = momg(xalphar,xnur,3.)**(2./3.)/momg(xalphar,xnur,2.)
150 !
151 !* 2.4 Csts for the shape parameter
152 !
153 xlbc = xar*zgamc(2)
154 xlbexc = 1.0/xbc
155 !
156 xnr = 1.0/(xar*momg(xalphar,xnur,xbr))
157 xccr = 8.e6
158 xcxr = -1.
159 IF (nmom_r.EQ.1) THEN
160  xlbexr = 1.0/(xcxr-xbr)
161  xlbr = ( xar*xccr*momg(xalphar,xnur,xbr) )**(-xlbexr)
162 ELSE
163  xlbr = xar*zgamr(2)
164  xlbexr = 1.0/xbr
165 END IF
166 !
167 !
168 !------------------------------------------------------------------------------
169 !
170 !
171 !* 3. CONSTANTS FOR THE SEDIMENTATION
172 ! -------------------------------
173 !
174 !
175 !* 4.1 Exponent of the fall-speed air density correction
176 !
177 ikb = 1 + jpvext
178 ! Correction
179 !ZRHO00 = XP00/(XRD*XTHVREFZ(IKB))
180 zrho00 = 1.2041 ! at P=1013.25hPa and T=20°C
181 !
182 !* 4.2 Constants for sedimentation
183 !
184 xfsedrr = xcr*gamma_x0d(xnur+(xdr+3.)/xalphar)/gamma_x0d(xnur+3./xalphar)* &
185  (zrho00)**xcexvt
186 xfsedcr = xcr*gamma_x0d(xnur+xdr/xalphar)/gamma_x0d(xnur)* &
187  (zrho00)**xcexvt
188 xfsedrc = xcc*gamma_x0d(xnuc+(xdc+3.)/xalphac)/gamma_x0d(xnuc+3./xalphac)* &
189  (zrho00)**xcexvt
190 xfsedcc = xcc*gamma_x0d(xnuc+xdc/xalphac)/gamma_x0d(xnuc)* &
191  (zrho00)**xcexvt
192 
193 !
194 xlb(2) = xlbc
195 xlbex(2) = xlbexc
196 xd(2) = xdc
197 xfsedr(2) = xfsedrc
198 xfsedc(2) = xfsedcc
199 !
200 xlb(3) = xlbr
201 xlbex(3) = xlbexr
202 xd(3) = xdr
203 xfsedr(3) = xfsedrr
204 xfsedc(3) = xfsedcr
205 !
206 !------------------------------------------------------------------------------
207 !
208 !
209 !* 4. CONSTANTS FOR THE NUCLEATION PROCESS
210 ! ------------------------------------
211 !
212 !
213 xwmin = 0.01 ! Minimal positive vertical velocity required
214  ! for the activation process in Twomey and CPB scheme
215 xtmin = -0.000278 ! Minimal cooling required 1K/h
216 !
217 xdiva = 226.e-7 ! Diffusivity of water vapor in the air
218 xthco = 24.3e-3 ! Air thermal conductivity
219 !
220 ! ( 8 Mw (Sigma)sw )3 Pi*Rho_l
221 ! XCSTDCRIT = ( -------------- ) * --------
222 ! ( 3 Ra Rhow ) 6
223 !
224 zsurf_ten = 76.1e-3 ! Surface tension of a water drop at T=0 C
225 xcstdcrit = (xpi/6.)*xrholw*( (8.0*zsurf_ten )/( 3.0*xrv*xrholw ) )**3
226 !
227 !
228 !
229 ! 4.1 Tabulation of the hypergeometric functions in 'no units'
230 ! --------------------------------------------------------
231 !
232 ! In LIMA's nucleation parameterization,
233 ! supersaturation is not in % : Smax=0.01 for a 1% supersaturation.
234 ! This is accounted for in the modified Beta and C values.
235 !
236 ! Here, we tabulate the
237 ! F(mu,k/2, k/2+1 ,-Beta S**2) -> XHYPF12
238 ! F(mu,k/2,(k+3)/2,-Beta S**2) -> XHYPF32 functions
239 ! using a logarithmic scale for S
240 !
241 nhyp = 500 ! Number of points for the tabulation
242 CALL param_lima_warm_allocate('XHYPF12', nhyp, nmod_ccn)
243 CALL param_lima_warm_allocate('XHYPF32', nhyp, nmod_ccn)
244 !
245 zsmin = 1.0e-5 ! Minimum supersaturation set at 0.001 %
246 zsmax = 5.0e-2 ! Maximum supersaturation set at 5 %
247 xhypintp1 = real(nhyp-1)/log(zsmax/zsmin)
248 xhypintp2 = real(nhyp)-xhypintp1*log(zsmax)
249 !
250 DO jmod = 1,nmod_ccn
251  DO j1 = 1,nhyp
252  zss =zsmax*(zsmin/zsmax)**(real(nhyp-j1)/real(nhyp-1))
253  xhypf12(j1,jmod) = hypgeo(xmuhen_multi(jmod),0.5*xkhen_multi(jmod),&
254  0.5*xkhen_multi(jmod)+1.0,xbetahen_multi(jmod),zss)
255  xhypf32(j1,jmod) = hypgeo(xmuhen_multi(jmod),0.5*xkhen_multi(jmod),&
256  0.5*xkhen_multi(jmod)+1.5,xbetahen_multi(jmod),zss)
257  END DO
258 ENDDO
259 !
260 nahen = 81 ! Tabulation for each Kelvin degree in the range XTT-40 to XTT+40
261 xahenintp1 = 1.0
262 xahenintp2 = 0.5*real(nahen-1) - xtt
263 !
264 ! Compute the tabulation of function of T :
265 !
266 ! 1
267 ! XAHENG = -----------------------
268 ! XCSTHEN * G**(3/2)
269 !
270 ! Compute constants for the calculation of Smax.
271 ! XCSTHEN = 1/(rho_l 2 pi)
272 ! PSI1
273 ! PSI3
274 ! T
275 ! Lv
276 ! G
277 !
278 CALL param_lima_warm_allocate('XAHENG', nahen)
279 CALL param_lima_warm_allocate('XAHENG2', nahen)
280 CALL param_lima_warm_allocate('XAHENG3', nahen)
281 CALL param_lima_warm_allocate('XPSI1', nahen)
282 CALL param_lima_warm_allocate('XPSI3', nahen)
283 xcsthen = 1.0 / ( xrholw*2.0*xpi )
284 DO j1 = 1,nahen
285  ztt = xtt + real(j1-(nahen-1)/2) ! T
286  zlv = xlvtt+(xcpv-xcl)*(ztt-xtt) ! Lv
287  xpsi1(j1) = (xg/(xrd*ztt))*(xmv*zlv/(xmd*xcpd*ztt)-1.) ! Psi1
288  xpsi3(j1) = -1*xmv*zlv/(xmd*xrd*(ztt**2)) ! Psi3
289  zg = 1./( xrholw*( (xrv*ztt)/ & ! G
290  (xdiva*exp(xalpw-(xbetaw/ztt)-(xgamw*alog(ztt)))) &
291  + (zlv/ztt)**2/(xthco*xrv) ) )
292  xaheng(j1) = xcsthen/(zg)**(3./2.)
293  xaheng2(j1) = 1/(zg)**(1./2.) * gamma_x0d(xnuc+1./xalphac)/gamma_x0d(xnuc)
294  xaheng3(j1) = (zg) * gamma_x0d(xnuc+1./xalphac)/gamma_x0d(xnuc)
295 END DO
296 !-------------------------------------------------------------------------------
297 !
298 ! Parameters used to initialise the droplet and drop concentration
299 ! from the respective mixing ratios (used in RESTART_RAIN_C2R2)
300 !
301 ! Droplet case
302 !
303 !!ALLOCATE(XCONCC_INI(SIZE(PNFS,1),SIZE(PNFS,2),SIZE(PNFS,3),SIZE(PNFS,4))) !NMOD_CCN))
304 !! XCONCC_INI(:,:,:,:) = 0.8 * PNFS(:,:,:,:) ! 80% of the maximum CCN conc. is assumed
305 !
306 ! Raindrop case
307 !
308 xconcr_param_ini = (1.e7)**3/(xpi*xrholw) ! MP law with N_O=1.E7 m-1 is assumed
309 !
310 !
311 !------------------------------------------------------------------------------
312 !
313 !
314 !* 5. CONSTANTS FOR THE COALESCENCE PROCESSES
315 ! ---------------------------------------
316 !
317 !
318 !* 6.1 Csts for the coalescence processes
319 !
320 xkera1 = 2.59e15 ! From Long a1=9.44E9 cm-3 so XKERA1= 9.44E9*1E6*(PI/6)**2
321 xkera2 = 3.03e3 ! From Long a2=5.78E3 so XKERA2= 5.78E3* (PI/6)
322 !
323 ! Cst for the cloud droplet selfcollection process
324 !
325 xselfc = xkera1*zgamc(3)
326 !
327 ! Cst for the autoconversion process
328 !
329 xauto1 = 6.25e18*(zgamc(2))**(1./3.)*sqrt(zgamc(4))
330 xauto2 = 0.5e6*(zgamc(4))**(1./6.)
331 xlautr = 2.7e-2
332 xlautr_threshold = 0.4
333 xitautr= 0.27 ! (Notice that T2 of BR74 is uncorrect and that 0.27=1./3.7
334 xitautr_threshold = 7.5
335 xcautr = 3.5e9
336 xr0 = 25.0e-6
337 !
338 ! Cst for the accretion process
339 !
340 xaccr1 = zgamr(2)**(1./3.)
341 xaccr2 = 5.0e-6
342 xaccr3 = 12.6e-4
343 xaccr4 = xauto2
344 xaccr5 = 3.5
345 xaccr6 = 1.2*xcautr
346 xaccr_clarge1 = xkera2*zgamc(2)
347 xaccr_clarge2 = xkera2*zgamr(2)
348 xaccr_rlarge1 = xkera2*zgamc(3)*xrholw*(xpi/6.0)
349 xaccr_rlarge2 = xkera2*zgamc(2)*zgamr(2)*xrholw*(xpi/6.0)
350 xaccr_csmall1 = xkera1*zgamc(3)
351 xaccr_csmall2 = xkera1*zgamr(3)
352 xaccr_rsmall1 = xkera1*zgamc(5)*xrholw*(xpi/6.0)
353 xaccr_rsmall2 = xkera1*zgamc(2)*zgamr(3)*xrholw*(xpi/6.0)
354 !
355 ! ICE3 accretion of cloud droplets by rain drops
356 !
357 xfcaccr = (xpi/4.0)*xccr*xcr*(zrho00**xcexvt)*momg(xalphar,xnur,xdr+2.0)
358 xexcaccr = -xdr-3.0
359 !
360 ! Cst for the raindrop self-collection/breakup process
361 !
362 xscbu2 = xkera2*zgamr(2)
363 xscbu3 = xkera1*zgamr(3)
364 xscbu_eff1 = 0.6e-3
365 xscbu_eff2 = 2.0e-3
366 xscbuexp1 = -2500.0
367 !
368 !
369 !------------------------------------------------------------------------------
370 !
371 !
372 !* 6. CONSTANTS FOR THE "SONTANEOUS" BREAK-UP
373 ! ---------------------------------------
374 !
375 !
376 xsponbud1 = 3.0e-3
377 xsponbud2 = 4.0e-3
378 xsponbud3 = 5.0e-3
379 xsponcoef2 = ((xsponbud3/xsponbud2)**3 - 1.0)/(xsponbud3-xsponbud1)**2
380 !
381 !
382 !------------------------------------------------------------------------------
383 !
384 !
385 !* 7. CONSTANTS FOR EVAPORATION PROCESS
386 ! ---------------------------------------
387 !
388 !
389 x0cndc = (4.0*xpi)*xc1c*xf0c*momg(xalphac,xnuc,1.)
390 x2cndc = (4.0*xpi)*xc1c*xf2c*xcc*momg(xalphac,xnuc,xdc+2.0)
391 !
392 ! Valeurs utiles pour le calcul de l'évaporation en fonction de N_r
393 !
394 !XEX0EVAR = -1.0
395 !XEX1EVAR = -1.0 - (XDR+1.0)*0.5
396 !XEX2EVAR = -0.5*XCEXVT
397 !
398 !X0EVAR = (2.0*XPI)*XF0R*GAMMA_X0D(XNUR+1./XALPHAR)/GAMMA_X0D(XNUR)
399 !X1EVAR = (2.0*XPI)*XF1R*((ZRHO00)**(XCEXVT)*(XCR/0.15E-4))**0.5* &
400 ! GAMMA_X0D(XNUR+(XDR+3.0)/(2.0*XALPHAR))/GAMMA_X0D(XNUR)
401 !
402 !
403 ! Valeurs utiles pour le calcul de l'évaporation en fonction de r_r
404 !
405 xex0evar = 2.0
406 xex1evar = 2.0 - (xdr+1.0)*0.5
407 xex2evar = -0.5*xcexvt
408 !
409 x0evar = (12.0)*xf0r*gamma_x0d(xnur+1./xalphar)/gamma_x0d(xnur+3./xalphar)
410 x1evar = (12.0)*xf1r*((zrho00)**(xcexvt)*(xcr/0.15e-4))**0.5* &
411  gamma_x0d(xnur+(xdr+3.0)/(2.0*xalphar))/gamma_x0d(xnur+3./xalphar)
412 !
413 xcevap = 0.86
414 !
415 !------------------------------------------------------------------------------
416 !
417 !
418 !* 8. SET-UP RADIATIVE PARAMETERS
419 ! ---------------------------
420 !
421 !
422 ! R_eff_c = XFREFFC * (rho*r_c/N_c)**(1/3)
423 !
424 !
425 xfreffc = 0.5 * zgamc(6) * (1.0/xac)**(1.0/3.0)
426 xfreffr = 0.5 * zgamr(6) * (1.0/xar)**(1.0/3.0)
427 !
428 ! Coefficients used to compute reff when both cloud and rain are present
429 !
430 xcrec = 1.0/ (zgamc(6) * xac**(2.0/3.0))
431 xcrer = 1.0/ (zgamr(6) * xar**(2.0/3.0))
432 !
433 !
434 !------------------------------------------------------------------------------
435 !
436 !
437 !* 9. SOME PRINTS FOR CONTROL
438 ! -----------------------
439 !
440 !
441 !!$GFLAG = .TRUE.
442 !!$IF (GFLAG) THEN
443 !!$ ILUOUT0 = TLUOUT0%NLU
444 !!$ WRITE(UNIT=ILUOUT0,FMT='(" Summary of the cloud particule characteristics")')
445 !!$ WRITE(UNIT=ILUOUT0,FMT='(" CLOUD")')
446 !!$ WRITE(UNIT=ILUOUT0,FMT='(" masse: A=",E13.6," B=",E13.6)') &
447 !!$ XAR,XBR
448 !!$ WRITE(UNIT=ILUOUT0,FMT='(" vitesse: C=",E13.6," D=",E13.6)') &
449 !!$ XCC,XDC
450 !!$ WRITE(UNIT=ILUOUT0,FMT='(" distribution:AL=",E13.6,"NU=",E13.6)') &
451 !!$ XALPHAC,XNUC
452 !!$ WRITE(UNIT=ILUOUT0,FMT='(" RAIN")')
453 !!$ WRITE(UNIT=ILUOUT0,FMT='(" masse: A=",E13.6," B=",E13.6)') &
454 !!$ XAR,XBR
455 !!$ WRITE(UNIT=ILUOUT0,FMT='(" vitesse: C=",E13.6," D=",E13.6)') &
456 !!$ XCR,XDR
457 !!$ WRITE(UNIT=ILUOUT0,FMT='(" distribution:AL=",E13.6,"NU=",E13.6)') &
458 !!$ XALPHAR,XNUR
459 !!$ WRITE(UNIT=ILUOUT0,FMT='(" Description of the nucleation spectrum")')
460 !!$ WRITE(UNIT=ILUOUT0,FMT='(" C=",E13.6," k=",E13.6)') XCHEN, XKHEN

463 !!$END IF
