18!!**** Determine convective columns as well as the cloudy values of theta,
19!! and qv at the lifting condensation level (LCL)
20!!
21!! PURPOSE
22!! -------
23!! The purpose of this routine is to determine convective columns
24!!
25!!
26!!
27!!** METHOD
28!! ------
29!! Computations are done at every model level starting from bottom.
30!! The use of masks allows to optimise the inner loops (horizontal loops).
31!! What we look for is the undermost unstable level at each grid point.
32!!
33!!
34!!
35!! EXTERNAL
36!! --------
37!! Routine CONVECT_SATMIXRATIO
38!!
39!!
40!! IMPLICIT ARGUMENTS
41!! ------------------
42!! Module MODD_CST
43!! XG ! gravity constant
44!! XP00 ! Reference pressure
45!! XRD, XRV ! Gaz constants for dry air and water vapor
46!! XCPD ! Cpd (dry air)
47!! XTT ! triple point temperature
48!! XBETAW, XGAMW ! constants for vapor saturation pressure
49!!
50!! Module MODD_CONVPAR
51!! XA25 ! reference grid area
52!! XZLCL ! maximum height difference between
53!! ! the surface and the DPL
54!! XZPBL ! minimum mixed layer depth to sustain convection
55!! XCDEPTH ! minimum necessary cloud depth
56!! XCDEPTH_D ! maximum allowed cloud depth
57!! XDTPERT ! add small Temp peturbation
58!! XNHGAM ! coefficient for buoyancy term in w eq.
59!! ! accounting for nh-pressure
60!! XAW, XBW, XATPERT, XBTPERT
61!!
62!! Module MODD_CONVPAREXT
63!! JCVEXB, JCVEXT ! extra levels on the vertical boundaries
64!!
65!! REFERENCE
66!! ---------
67!!
68!! Book2 of documentation ( routine TRIGGER_FUNCT)
69!! Fritsch and Chappell (1980), J. Atm. Sci., Vol. 37, 1722-1761.

165!-------------------------------------------------------------------------------
166!
167!* 0.3 Compute array bounds
168! --------------------
169!
170
171IF (lhook) CALL dr_hook('CONVECT_TRIGGER_SHAL',0,zhook_handle)
172ikb = 1 + cvpext%JCVEXB
173ike = d%NKT - cvpext%JCVEXT
174!
175!
176!* 1. Initialize local variables
177! --------------------------
178!
179zeps = cst%XRD / cst%XRV
180zepsa = cst%XRV / cst%XRD
181zcpord = cst%XCPD / cst%XRD
182zrdocp = cst%XRD / cst%XCPD
183otrig(:) = .false.
184idpl(:) = kdpl(:)
185ipbl(:) = kpbl(:)
186ilcl(:) = klcl(:)
187pwlcl(:) = 0.
188zwlcl(:) = 0.
189pthlcl(:) = 1.
190pthvelcl(:)= 1.
191ptlcl(:) = 1.
192prvlcl(:) = 0.
193pwlcl(:) = 0.
194pzlcl(:) = pz(:,ikb)
195zzdpl(:) = pz(:,ikb)
196gtrig2(:) = .true.
197!
198!
199!
200! 0. Auxiliary arrays
201! ----------------
202!
203!!!Auxiliary arrays, computed once here and used for each JKK in 3.
204DO jk=ikb+1,ike-1
205 DO ji=d%NIJB,d%NIJE
206 zzzx1(ji,jk)=ppres(ji,jk)-ppres(ji,jk+1)
207 zzppres(ji,jk)=ppres(ji,jk)*zzzx1(ji,jk)
208 zzpth(ji,jk)=pth(ji,jk)*zzzx1(ji,jk)
209 zzprv(ji,jk)=max(0.,prv(ji,jk))*zzzx1(ji,jk)
210 ENDDO
211ENDDO
212!
213! 1. Determine highest necessary loop test layer
214! -------------------------------------------
215!
216jt = ike - 2
217!
218!* 2. Enter loop for convection test
219! ------------------------------
220!
221DO jkk = ikb + 1, ike - 2
222!
223 DO ji=d%NIJB, d%NIJE
224 gwork1(ji) = zzdpl(ji) - pz(ji,ikb) < cvp_shal%XZLCL
225 ENDDO
226 ! we exit the trigger test when the center of the mixed layer is more
227 ! than 1500 m above soil level.
228 DO ji=d%NIJB, d%NIJE
229 IF ( gwork1(ji) ) THEN
230 zdpthmix(ji) = 0.
231 zpresmix(ji) = 0.
232 zthlcl(ji) = 0.
233 zrvlcl(ji) = 0.
234 zzdpl(ji) = pz(ji,jkk)
235 idpl(ji) = jkk
236 END IF
237 ENDDO
238!
239!
240!* 3. Construct a mixed layer of at least 50 hPa (XZPBL)
241! ------------------------------------------
242!
243 DO jk = jkk, ike - 1
244 jkm = jk + 1
245 DO ji = d%NIJB, d%NIJE
246 IF ( gwork1(ji) .AND. zdpthmix(ji) < cvp_shal%XZPBL ) THEN
247 ipbl(ji) = jk
248 zdpthmix(ji) = zdpthmix(ji) + zzzx1(ji,jk) !!!uses the auxiliary arrays of 0.
249 zpresmix(ji) = zpresmix(ji) + zzppres(ji,jk)
250 zthlcl(ji) = zthlcl(ji) + zzpth(ji,jk)
251 zrvlcl(ji) = zrvlcl(ji) + zzprv(ji,jk)
252 END IF
253 END DO
254 END DO
255!
256!
257 DO ji=d%NIJB, d%NIJE
258 IF ( gwork1(ji) ) THEN
259!
260 zpresmix(ji) = zpresmix(ji) / zdpthmix(ji)
261 zthlcl(ji) = zthlcl(ji) / zdpthmix(ji) + &
262 & (cvp_shal%XATPERT * min(3.,ptkecls(ji))/cst%XCPD +cvp_shal%XBTPERT) * cvp_shal%XDTPERT ! add small Temp Perturb.
263 zrvlcl(ji) = zrvlcl(ji) / zdpthmix(ji)
264 zthvlcl(ji) = zthlcl(ji) * ( 1. + zepsa * zrvlcl(ji) ) &
265 / ( 1. + zrvlcl(ji) )
266!
267!* 4.1 Use an empirical direct solution ( Bolton formula )
268! to determine temperature and pressure at LCL.
269! Nota: the adiabatic saturation temperature is not
270! equal to the dewpoint temperature
271! ----------------------------------------------------
272!
273!
274 ztmix(ji) = zthlcl(ji) * ( zpresmix(ji) / cst%XP00 ) ** zrdocp
275 zevmix(ji) = zrvlcl(ji) * zpresmix(ji) / ( zrvlcl(ji) + zeps )
276 zevmix(ji) = max( 1.e-8, zevmix(ji) )
277 zx1 = log( zevmix(ji) / 613.3 )
278 ! dewpoint temperature
279 zx1 = ( 4780.8 - 32.19 * zx1 ) / ( 17.502 - zx1 )
280 ! adiabatic saturation temperature
281 ztlcl(ji) = zx1 - ( .212 + 1.571e-3 * ( zx1 - cst%XTT ) &
282 - 4.36e-4 * ( ztmix(ji) - cst%XTT ) ) * ( ztmix(ji) - zx1 )
283 ztlcl(ji) = min( ztlcl(ji), ztmix(ji) )
284 zplcl(ji) = cst%XP00 * ( ztlcl(ji) / zthlcl(ji) ) ** zcpord
285!
286 END IF
287 ENDDO
288!
289 DO ji=d%NIJB, d%NIJE
290 !arrays ZEWA, ZLVA, ZLSA, ZCPHA,
291 ! ZEWB, ZLVB, ZLSB, ZCPHB are only used if GWORK1(JI)=.TRUE.
292 !in the rest of the code.
293 IF ( gwork1(ji) ) THEN
294 CALL convect_satmixratio( zplcl(ji), ztlcl(ji), zeps, zewa(ji), zlva(ji), zlsa(ji), zcpha(ji) )
295 CALL convect_satmixratio( zpresmix(ji), ztmix(ji), zeps, zewb(ji), zlvb(ji), zlsb(ji), zcphb(ji) )
296 ENDIF
297 ENDDO
298!
299!* 4.2 Correct ZTLCL in order to be completely consistent
300! with MNH saturation formula
301! ---------------------------------------------
302!
303 DO ji=d%NIJB, d%NIJE
304 IF( gwork1(ji) ) THEN
305 zlsa(ji) = zewa(ji) / ztlcl(ji) * ( cst%XBETAW / ztlcl(ji) - cst%XGAMW ) ! dr_sat/dT
306 zlsa(ji) = ( zewa(ji) - zrvlcl(ji) ) / &
307 ( 1. + zlva(ji) / zcpha(ji) * zlsa(ji) )
308 ztlcl(ji) = ztlcl(ji) - zlva(ji) / zcpha(ji) * zlsa(ji)
309!
310 END IF
311 ENDDO
312!
313!
314!* 4.3 If ZRVLCL = PRVMIX is oversaturated set humidity
315! and temperature to saturation values.
316! ---------------------------------------------
317!
318 DO ji=d%NIJB, d%NIJE
319 IF( gwork1(ji) .AND. zrvlcl(ji) > zewb(ji) ) THEN
320 zlsb(ji) = zewb(ji) / ztmix(ji) * ( cst%XBETAW / ztmix(ji) - cst%XGAMW ) ! dr_sat/dT
321 zlsb(ji) = ( zewb(ji) - zrvlcl(ji) ) / &
322 ( 1. + zlvb(ji) / zcphb(ji) * zlsb(ji) )
323 ztlcl(ji) = ztmix(ji) - zlvb(ji) / zcphb(ji) * zlsb(ji)
324 zrvlcl(ji) = zrvlcl(ji) - zlsb(ji)
325 zplcl(ji) = zpresmix(ji)
326 zthlcl(ji) = ztlcl(ji) * ( cst%XP00 / zplcl(ji) ) ** zrdocp
327 zthvlcl(ji)= zthlcl(ji) * ( 1. + zepsa * zrvlcl(ji) ) &
328 / ( 1. + zrvlcl(ji) )
329 END IF
330 ENDDO
331!
332!
333!* 5.1 Determine vertical loop index at the LCL and DPL
334! --------------------------------------------------
335!
336 !!!Every 4 vertical level, the first loop checks if computations need
337 !!!to be done in the second loop for a given JI.
338 !!!PPRES comes from array PREHYDF, which is decreasing with
339 !!!increasing vertical level.
340 jlstep=4
341 DO jlstart=jkk,ike-1,jlstep
342 jlsize=min(jlstep,jt-jlstart+1)
343 IF (jlstart .EQ. jkk) THEN
344 ijimin=d%NIJB
345 ijimax=d%NIJE
346 llcompute=.true.
347 ELSE
348 IF (llcompute) THEN
349 llcompute=.false.
350 ijimin2=ijimin
351 ijimax2=ijimax
352 DO ji=d%NIJB,d%NIJE
353 IF ((ji .GE. ijimin2) .AND. (ji .LE. ijimax2) .AND. gwork1(ji) .AND. (zplcl(ji) <= ppres(ji,jlstart))) THEN
354 IF (.NOT. llcompute) ijimin=ji
355 ijimax=ji
356 llcompute=.true.
357 ENDIF
358 ENDDO
359 ENDIF
360 ENDIF
361
362 IF (llcompute) THEN
363 DO jk = jlstart, jlstart+jlsize-1
364 DO ji = d%NIJB,d%NIJE
365 IF ( (ji .GE. ijimin) .AND. (ji .LE. ijimax) .AND. zplcl(ji) <= ppres(ji,jk) .AND. gwork1(ji) ) ilcl(ji) = jk + 1
366 END DO
367 END DO
368 ENDIF
369 ENDDO
370
371!
372!
373!* 5.2 Estimate height and environm. theta_v at LCL
374! --------------------------------------------------
375!
376 jlclmin=ilcl(d%NIJB)
377 DO ji = d%NIJB, d%NIJE
378 IF ( gwork1(ji) ) THEN
379 jk = ilcl(ji)
380 jkm = jk - 1
381 zdp(ji) = log( zplcl(ji) / ppres(ji,jkm) ) / &
382 log( ppres(ji,jk) / ppres(ji,jkm) )
383 zthvelcl(ji) = pthv(ji,jkm) + ( pthv(ji,jk) - pthv(ji,jkm) ) * zdp(ji)
384 ! we compute the precise value of the LCL
385 ! The precise height is between the levels ILCL and ILCL-1.
386 zzlcl(ji) = pz(ji,jkm) + ( pz(ji,jk) - pz(ji,jkm) ) * zdp(ji)
387 END IF
388 END DO
389!
390!* 6. Check to see if cloud is bouyant
391! --------------------------------
392!
393!* 6.1 Compute grid scale vertical velocity perturbation term ZWORK1
394! -------------------------------------------------------------
395!
396! ! normalize w grid scale to a 25 km refer. grid
397! DO JI = 1, D%NIJT
398! JK = ILCL(JI)
399! JKM = JK - 1
400! ZWORK1(JI) = ( PW(JI,JKM) + ( PW(JI,JK) - PW(JI,JKM) ) * ZDP(JI) ) &
401! * SQRT( PDXDY(JI) / XA25 )
402! - 0.02 * ZZLCL(JI) / XZLCL ! avoid spurious convection
403! END DO
404! ! compute sign of normalized grid scale w
405! ZWORK2(:) = SIGN( 1., ZWORK1(:) )
406! ZWORK1(:) = XWTRIG * ZWORK2(:) * ABS( ZWORK1(:) ) ** 0.333 &
407! * ( XP00 / ZPLCL(:) ) ** ZRDOCP
408!
409!* 6.2 Compute parcel vertical velocity at LCL
410! ---------------------------------------
411!
412! DO JI = 1, D%NIJT
413! JKDL = IDPL(JI)
414! ZWORK3(JI) = XG * ZWORK1(JI) * ( ZZLCL(JI) - PZ(JI,JKDL) ) &
415! / ( PTHV(JI,JKDL) + ZTHVELCL(JI) )
416! END DO
417! WHERE( GWORK1(:) )
418! ZWLCL(:) = 1. + .5 * ZWORK2(:) * SQRT( ABS( ZWORK3(:) ) )
419! GTRIG(:) = ZTHVLCL(:) - ZTHVELCL(:) + ZWORK1(:) > 0. .AND. &
420! ZWLCL(:) > 0.
421! END WHERE
422 DO ji = d%NIJB, d%NIJE
423 jlclmin=min(jlclmin,ilcl(ji))
424 zwlcl(ji) = cvp_shal%XAW * max(0.,pw(ji,ikb)) + cvp_shal%XBW
425 ! the factor 1.05 takes entrainment into account
426 zwlclsqrent(ji) = 1.05*zwlcl(ji)*zwlcl(ji)
427!
428!
429!* 6.3 Look for parcel that produces sufficient cloud depth.
430! The cloud top is estimated as the level where the CAPE
431! is smaller than a given value (based on vertical velocity eq.)
432! --------------------------------------------------------------
433!
434 ztheul(ji) = ztlcl(ji) * ( zthlcl(ji) / ztlcl(ji) ) &
435 ** ( 1. - 0.28 * zrvlcl(ji) ) &
436 * exp( ( 3374.6525 / ztlcl(ji) - 2.5403 ) * &
437 zrvlcl(ji) * ( 1. + 0.81 * zrvlcl(ji) ) )
438 END DO
439
440 zcape(d%NIJB:d%NIJE) = max(jlclmin-1-ikb,0)*cst%XG
441 zcap(d%NIJB:d%NIJE) = 0.
442 ztop(d%NIJB:d%NIJE) = 0.
443 zwork3(d%NIJB:d%NIJE)= 0.
444 jkm=max(ikb,jlclmin-1)
445
446!REK : should try if's instead of sign & max
447
448 !!!Every 4 vertical level, the first loop checks if computations need to be done in the second loop.
449 !!!If ZWORK3(JI)==-1 and ZCAPE(JI) >10., no more computation are necessary for JI :
450 !!!if ZWORK3(JI)==-1, it remains equal to -1, and ZTOP(JI) does not change.
451 !!!if ZCAPE(JI)>10., it remains >10., because MAX(1.,ZX1) is a positive
452 !!!number ; the last loop of CONVECT_TRIGGER_SHAL uses ZCAPE(JI) as a
453 !!!criterion.
454 jlstep=4
455 DO jlstart=jkm,jt,jlstep
456 jlsize=min(jlstep,jt-jlstart+1)
457
458 IF (jlstart .EQ. jkm) THEN
459 llcompute=.true.
460 ELSE
461 IF (llcompute) THEN
462 llcompute=.false.
463 DO ji=d%NIJB,d%NIJE
464 IF (gtrig2(ji) .AND. ((zwork3(ji) .GT. -0.5) .OR. (zcape(ji) .LE. 10.))) llcompute=.true.
465 ENDDO
466 ENDIF
467 ENDIF
468 IF (llcompute) THEN
469
470 DO jl = jlstart,jlstart+jlsize-1
471 jk = jl + 1
472
473 DO ji = d%NIJB, d%NIJE
474 !!!it was not possible to use a mask IF (JI>=IJIMIN .AND. JI
475 !!!<=IJIMAX) here, as the loop did not vectorize.
476 zx1 = ( 2. * ztheul(ji) / ( pthes(ji,jk) + pthes(ji,jl) ) - 1. ) *&
477 & (pz(ji,jk) - pz(ji,jl))
478 IF ((jl < ilcl(ji))) zx1 = 0.
479 zcape(ji) = zcape(ji) + cst%XG * max( 1., zx1 )
480 zcap(ji) = zcap(ji) + zx1
481 ! the factor 1.05 in ZWLCLSQRENT takes entrainment into account
482 zx2 = sign( 1., ( cvp_shal%XNHGAM * cst%XG * zcap(ji) + zwlclsqrent(ji) ) )
483 zwork3(ji) = max( -1., (zwork3(ji) + min(0.,zx2)) )
484 ! Nota, the factors ZX2 and ZWORK3 are only used to avoid
485 ! if and goto statements, the difficulty is to extract only
486 ! the level where the criterium is first fullfilled
487 ztopp(ji)=ztop(ji)
488 ztop(ji) = pz(ji,jl) * .5 * ( 1. + zx2 ) * ( 1. + zwork3(ji) ) + &
489 & ztop(ji) * .5 * ( 1. - zx2 )
490 ztop(ji)=max(ztop(ji),ztopp(ji))
491 ENDDO
492
493 ENDDO
494
495 ENDIF !!LLCOMPUTE
496
497 END DO
498!
499!
500
501 DO ji=d%NIJB, d%NIJE
502 IF( (ztop(ji)-zzlcl(ji)) .GE. cvp_shal%XCDEPTH .AND. gtrig2(ji) .AND. zcape(ji) > 10. )THEN
503 gtrig2(ji) = .false.
