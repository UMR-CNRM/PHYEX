126  :: ZFLXX,ZFLXY
127  ! work arrays

129 !
130 REAL :: ZCSV !constant for the scalar flux
131

133  ! Index values for the Beginning and End
134  ! mass points of the domain

136 INTEGER :: ISV ! number of scalar var.
137 REAL, DIMENSION(SIZE(PDZZ,1),SIZE(PDZZ,2),1+JPVEXT:3+JPVEXT) :: ZCOEFF
138  ! coefficients for the uncentred gradient
139  ! computation near the ground
140 !
141 CHARACTER(LEN=NMNHNAMELGTMAX) :: YMNHNAME
142 INTEGER :: IKU
143 TYPE(tfieldmetadata) :: TZFIELD
144 REAL :: ZTIME1, ZTIME2
145 ! ---------------------------------------------------------------------------
146 !
147 !* 1. PRELIMINARY COMPUTATIONS
148 ! ------------------------
149 !
150 ikb = 1+jpvext
151 ike = SIZE(psvm,3)-jpvext
152 iku = SIZE(psvm,3)
153 !
154 isv = SIZE(psvm,4)
155 !
156 IF(oblowsnow) THEN
157 ! See Vionnet (PhD, 2012) for a complete discussion around the value of the Schmidt number for blowing snow variables
158  zcsv= turbn%XCHF/prsnow
159 ELSE
160  zcsv= turbn%XCHF
161 ENDIF
162 !
163 ! compute the coefficients for the uncentred gradient computation near the
164 ! ground
165 zcoeff(:,:,ikb+2)= - pdzz(:,:,ikb+1) / &
166  ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+2) )
167 zcoeff(:,:,ikb+1)= (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) / &
168  ( pdzz(:,:,ikb+1) * pdzz(:,:,ikb+2) )
169 zcoeff(:,:,ikb)= - (pdzz(:,:,ikb+2)+2.*pdzz(:,:,ikb+1)) / &
170  ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+1) )
171 !
172 !
173 !* 15. HORIZONTAL FLUXES OF PASSIVE SCALARS
174 ! ------------------------------------
175 !
176 !
177 DO jsv=1,isv
178 !
179  IF (onomixlg .AND. jsv >= ksv_lgbeg .AND. jsv<= ksv_lgend) cycle
180 !
181 ! 15.1 <U' SVth'>
182 ! ----------
183 !
184  ! Computes the flux in the X direction
185  zflxx(:,:,:) = -zcsv * mxm(pk) * gx_m_u(1,iku,1,psvm(:,:,:,jsv),pdxx,pdzz,pdzx)
186  zflxx(:,:,ike+1) = zflxx(:,:,ike)
187 !
188 ! Compute the flux at the first inner U-point with an uncentred vertical
189 ! gradient
190  zflxx(:,:,ikb:ikb) = -zcsv * mxm( pk(:,:,ikb:ikb) ) * &
191  ( dxm(psvm(:,:,ikb:ikb,jsv)) * pinv_pdxx(:,:,ikb:ikb) &
192  -mxm( zcoeff(:,:,ikb+2:ikb+2)*psvm(:,:,ikb+2:ikb+2,jsv) &
193  +zcoeff(:,:,ikb+1:ikb+1)*psvm(:,:,ikb+1:ikb+1,jsv) &
194  +zcoeff(:,:,ikb :ikb )*psvm(:,:,ikb :ikb ,jsv) &
195  ) * 0.5 * ( pdzx(:,:,ikb+1:ikb+1)+pdzx(:,:,ikb:ikb) ) &
196  * pinv_pdxx(:,:,ikb:ikb) &
197  )
198 ! extrapolates the flux under the ground so that the vertical average with
199 ! the IKB flux gives the ground value
200  zwork2d(:,:,1)=psfsvm(:,:,jsv) * pdircosxw(:,:)
201  zflxx(:,:,ikb-1:ikb-1) = 2. * mxm( zwork2d(:,:,1:1) ) - zflxx(:,:,ikb:ikb)
202  !
203  ! stores <U SVth>
204  IF ( tpfile%LOPENED .AND. turbn%LTURB_FLX ) THEN
205  WRITE(ymnhname,'("USV_FLX_",I3.3)') jsv
206  tzfield = tfieldmetadata( &
207  cmnhname = trim( ymnhname ), &
208  cstdname = '', &
209  clongname = trim( ymnhname ), &
210  cunits = 'SVUNIT m s-1', &
211  cdir = 'XY', &
212  ccomment = 'X_Y_Z_' // trim( ymnhname ), &
213  ngrid = 2, &
214  ntype = typereal, &
215  ndims = 3, &
216  ltimedep = .true. )
217  CALL io_field_write(tpfile,tzfield,zflxx)
218  END IF
219 !
220  IF (tles%LLES_CALL .AND. ksplt==1) THEN
221  CALL second_mnh(ztime1)
222  CALL les_mean_subgrid( mxf(zflxx), tles%X_LES_SUBGRID_USv(:,:,:,jsv) )
223  CALL les_mean_subgrid( mzf(mxf(gx_w_uw(pwm,pdxx,pdzz,pdzx)*mzm(zflxx))), &
224  tles%X_LES_RES_ddxa_W_SBG_UaSv(:,:,:,jsv) , .true. )
225  CALL les_mean_subgrid( gx_m_m(psvm(:,:,:,jsv),pdxx,pdzz,pdzx)*mxf(zflxx), &
226  tles%X_LES_RES_ddxa_Sv_SBG_UaSv(:,:,:,jsv), .true. )
227  CALL second_mnh(ztime2)
228  tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
229  END IF
230 !
231 ! 15.2 <V' SVth'>
232 ! ----------
233 !
234  IF (.NOT. o2d) THEN
235 !
236 ! Computes the flux in the Y direction
237  zflxy(:,:,:)=-zcsv * mym(pk) * gy_m_v(1,iku,1,psvm(:,:,:,jsv),pdyy,pdzz,pdzy)
238  zflxy(:,:,ike+1) = zflxy(:,:,ike)
239 !
240 ! Compute the flux at the first inner V-point with an uncentred vertical
241 ! gradient
242 !
243  zflxy(:,:,ikb:ikb) = -zcsv * mym( pk(:,:,ikb:ikb) ) * &
244  ( dym(psvm(:,:,ikb:ikb,jsv)) * pinv_pdyy(:,:,ikb:ikb) &
245  -mym( zcoeff(:,:,ikb+2:ikb+2)*psvm(:,:,ikb+2:ikb+2,jsv) &
246  +zcoeff(:,:,ikb+1:ikb+1)*psvm(:,:,ikb+1:ikb+1,jsv) &
247  +zcoeff(:,:,ikb :ikb )*psvm(:,:,ikb :ikb ,jsv) &
248  ) * 0.5 * ( pdzy(:,:,ikb+1:ikb+1)+pdzy(:,:,ikb:ikb) ) &
249  * pinv_pdyy(:,:,ikb:ikb) &
250  )
251 ! extrapolates the flux under the ground so that the vertical average with
252 ! the IKB flux gives the ground value
253  zwork2d(:,:,1)=psfsvm(:,:,jsv) * pdircosyw(:,:)
254  zflxy(:,:,ikb-1:ikb-1) = 2. * mym( zwork2d(:,:,1:1) ) - zflxy(:,:,ikb:ikb)
255  !
256  ! stores <V SVth>
257  IF ( tpfile%LOPENED .AND. turbn%LTURB_FLX ) THEN
258  WRITE(ymnhname,'("VSV_FLX_",I3.3)') jsv
259  tzfield = tfieldmetadata( &
260  cmnhname = trim( ymnhname ), &
261  cstdname = '', &
262  clongname = trim(tzfield%CMNHNAME), &
263  cunits = 'SVUNIT m s-1', &
264  cdir = 'XY', &
265  ccomment = 'X_Y_Z_'//trim(tzfield%CMNHNAME), &
266  ngrid = 3, &
267  ntype = typereal, &
268  ndims = 3, &
269  ltimedep = .true. )
270  CALL io_field_write(tpfile,tzfield,zflxy)
271  END IF
272 !
273  ELSE
274  zflxy=0.
275  END IF
276 !
277  IF (tles%LLES_CALL .AND. ksplt==1) THEN
278  CALL second_mnh(ztime1)
279  CALL les_mean_subgrid( myf(zflxy), tles%X_LES_SUBGRID_VSv(:,:,:,jsv) )
280  CALL les_mean_subgrid( mzf(myf(gy_w_vw(pwm,pdyy,pdzz,pdzy)*mzm(zflxy))), &
281  tles%X_LES_RES_ddxa_W_SBG_UaSv(:,:,:,jsv) , .true. )
282  CALL les_mean_subgrid( gy_m_m(psvm(:,:,:,jsv),pdyy,pdzz,pdzy)*myf(zflxy), &
283  tles%X_LES_RES_ddxa_Sv_SBG_UaSv(:,:,:,jsv) , .true. )
284  CALL second_mnh(ztime2)
285  tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
286  END IF
287 !
288 !
289 ! 15.3 Horizontal source terms
290 ! -----------------------
291 !
292  IF (.NOT. o2d) THEN
293  IF (.NOT. oflat) THEN
294  prsvs(:,:,:,jsv)= prsvs(:,:,:,jsv) &
295  -dxf( mxm(prhodj) * zflxx * pinv_pdxx ) &
296  -dyf( mym(prhodj) * zflxy * pinv_pdyy ) &
297  +dzf( pmzm_prhodj * pinv_pdzz * &
298  ( mxf( mzm(zflxx * pinv_pdxx) * pdzx ) + myf( mzm(zflxy * pinv_pdyy) * pdzy ) ) &
299  )
300  ELSE
301  prsvs(:,:,:,jsv)= prsvs(:,:,:,jsv) &
302  -dxf( mxm(prhodj) * zflxx * pinv_pdxx ) &
303  -dyf( mym(prhodj) * zflxy * pinv_pdyy )
304  END IF
305  ELSE
306  IF (.NOT. oflat) THEN
307  prsvs(:,:,:,jsv)= prsvs(:,:,:,jsv) &
308  -dxf( mxm(prhodj) * zflxx * pinv_pdxx ) &
309  +dzf( pmzm_prhodj * pinv_pdzz * &
310  ( mxf( mzm(zflxx * pinv_pdxx) * pdzx ) ) &
311  )
312  ELSE
313  prsvs(:,:,:,jsv)= prsvs(:,:,:,jsv) &
314  -dxf( mxm(prhodj) *zflxx * pinv_pdxx )
315  END IF
316  END IF
317 !
318 !
319 END DO ! end loop JSV
320 !
321 !
322 END SUBROUTINE turb_hor_sv_flux
