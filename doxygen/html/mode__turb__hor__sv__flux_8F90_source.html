99INTEGER, INTENT(IN) :: KSV,KSV_LGBEG,KSV_LGEND ! number of sv var.
100LOGICAL, INTENT(IN) :: OFLAT ! Logical for zero ororography
101LOGICAL, INTENT(IN) :: ONOMIXLG ! to use turbulence for lagrangian variables (modd_conf)
102LOGICAL, INTENT(IN) :: O2D ! Logical for 2D model version (modd_conf)
103LOGICAL, INTENT(IN) :: OBLOWSNOW ! switch to activate pronostic blowing snow
104!
105REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PK ! Turbulent diffusion doef.
106 ! PK = PLM * SQRT(PTKEM)
107REAL, INTENT(IN) :: PRSNOW ! Ratio for diffusion coeff. scalar (blowing snow)
108REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDXX ! 1./PDXX
109REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDYY ! 1./PDYY
110REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDZZ ! 1./PDZZ
111REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PMZM_PRHODJ ! MZM(PRHODJ)
112REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PDXX, PDYY, PDZZ, PDZX, PDZY
113 ! Metric coefficients
114REAL, DIMENSION(:,:), INTENT(IN) :: PDIRCOSXW, PDIRCOSYW
115! Director Cosinus along x and y directions at surface w-point
116REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PRHODJ ! density * grid volume
117REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PWM ! vertical wind
118!
119REAL, DIMENSION(D%NIT,D%NJT,KSV), INTENT(IN) :: PSFSVM ! surface fluxes
120!
121!
122! Variables at t-1
123REAL, DIMENSION(D%NIT,D%NJT,D%NKT,KSV), INTENT(IN) :: PSVM ! scalar var. at t-1
124!
125REAL, DIMENSION(D%NIT,D%NJT,D%NKT,KSV), INTENT(INOUT) :: PRSVS ! var. at t+1 -split-
126!
127!
128!
129!* 0.2 declaration of local variables
130!
131REAL, DIMENSION(D%NIT,D%NJT,D%NKT) &
132 :: ZFLXX,ZFLXY
133 ! work arrays
134REAL, DIMENSION(D%NIT,D%NJT) :: ZWORK2D
135!
136REAL :: ZCSV !constant for the scalar flux
137
138INTEGER :: IKB,IKE
139 ! Index values for the Beginning and End
140 ! mass points of the domain
141INTEGER :: JSV ! loop counter
142INTEGER :: ISV ! number of scalar var.
143REAL, DIMENSION(SIZE(PDZZ,1),SIZE(PDZZ,2),1+JPVEXT:3+JPVEXT) :: ZCOEFF
144 ! coefficients for the uncentred gradient
145 ! computation near the ground
146!
147CHARACTER(LEN=NMNHNAMELGTMAX) :: YMNHNAME
148INTEGER :: IKU, IIT, IJT, IKT, JI, JJ, JK
149TYPE(tfieldmetadata) :: TZFIELD
150REAL :: ZTIME1, ZTIME2
151! ---------------------------------------------------------------------------
152!
153!* 1. PRELIMINARY COMPUTATIONS
154! ------------------------
155!
156ikb = 1+jpvext
157ike = SIZE(psvm,3)-jpvext
158iku = SIZE(psvm,3)
159iit=d%NIT
160ijt=d%NJT
161ikt=d%NKT
162!
163isv = SIZE(psvm,4)
164!
165IF(oblowsnow) THEN
166! See Vionnet (PhD, 2012) for a complete discussion around the value of the Schmidt number for blowing snow variables
167 zcsv= turbn%XCHF/prsnow
168ELSE
169 zcsv= turbn%XCHF
170ENDIF
171!
172! compute the coefficients for the uncentred gradient computation near the
173! ground
174!$acc kernels
175!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
176zcoeff(:,:,ikb+2)= - pdzz(:,:,ikb+1) / &
177 ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+2) )
178zcoeff(:,:,ikb+1)= (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) / &
179 ( pdzz(:,:,ikb+1) * pdzz(:,:,ikb+2) )
180zcoeff(:,:,ikb)= - (pdzz(:,:,ikb+2)+2.*pdzz(:,:,ikb+1)) / &
181 ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+1) )
182!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
183!$acc end kernels
184!
185!
186!* 15. HORIZONTAL FLUXES OF PASSIVE SCALARS
187! ------------------------------------
188!
189!
190DO jsv=1,isv
191!
192 IF (onomixlg .AND. jsv >= ksv_lgbeg .AND. jsv<= ksv_lgend) cycle
193!
194! 15.1 <U' SVth'>
195! ----------
196!
197 ! Computes the flux in the X direction
198 zflxx(:,:,:) = -zcsv * mxm(pk) * gx_m_u(oflat,psvm(:,:,:,jsv),pdxx,pdzz,pdzx)
199
200!$acc kernels
201!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
202 zflxx(:,:,ike+1) = zflxx(:,:,ike)
203!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
204!$acc end kernels
205!
206! Compute the flux at the first inner U-point with an uncentred vertical
207! gradient
208 zflxx(:,:,ikb) = -zcsv * mxm( pk(:,:,ikb) ) * &
209 ( dxm(psvm(:,:,ikb,jsv)) * pinv_pdxx(:,:,ikb) &
210 -mxm( zcoeff(:,:,2)*psvm(:,:,2,jsv) &
211 +zcoeff(:,:,ikb+1)*psvm(:,:,ikb+1,jsv) &
212 +zcoeff(:,:,ikb )*psvm(:,:,ikb ,jsv) &
213 ) * 0.5 * ( pdzx(:,:,ikb+1)+pdzx(:,:,ikb) ) &
214 * pinv_pdxx(:,:,ikb) &
215 )
216! extrapolates the flux under the ground so that the vertical average with
217! the IKB flux gives the ground value
218!$acc kernels
219!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
220 zwork2d(:,:)=psfsvm(:,:,jsv) * pdircosxw(:,:)
221!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
222!$acc end kernels
223 zflxx(:,:,ikb-1) = 2. * mxm( zwork2d(:,:) ) - zflxx(:,:,ikb)
224 !
225 ! stores <U SVth>
226 IF ( tpfile%LOPENED .AND. turbn%LTURB_FLX ) THEN
227 WRITE(ymnhname,'("USV_FLX_",I3.3)') jsv
228 tzfield = tfieldmetadata( &
229 cmnhname = trim( ymnhname ), &
230 cstdname = '', &
231 clongname = trim( ymnhname ), &
232 cunits = 'SVUNIT m s-1', &
233 cdir = 'XY', &
234 ccomment = 'X_Y_Z_' // trim( ymnhname ), &
235 ngrid = 2, &
236 ntype = typereal, &
237 ndims = 3, &
238 ltimedep = .true. )
239!$acc update self(ZFLXX)
240 CALL io_field_write(tpfile,tzfield,zflxx)
241 END IF
242!
243 IF (tles%LLES_CALL .AND. ksplt==1) THEN
244 CALL second_mnh(ztime1)
245 CALL les_mean_subgrid( mxf(zflxx), tles%X_LES_SUBGRID_USv(:,:,:,jsv) )
246 CALL les_mean_subgrid( mzf(mxf(gx_w_uw(oflat,pwm,pdxx,pdzz,pdzx)*mzm(zflxx))), &
247 tles%X_LES_RES_ddxa_W_SBG_UaSv(:,:,:,jsv) , .true. )
248 CALL les_mean_subgrid( gx_m_m(oflat,psvm(:,:,:,jsv),pdxx,pdzz,pdzx)*mxf(zflxx), &
249 tles%X_LES_RES_ddxa_Sv_SBG_UaSv(:,:,:,jsv), .true. )
250 CALL second_mnh(ztime2)
251 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
252 END IF
253!
254! 15.2 <V' SVth'>
255! ----------
256!
257 IF (.NOT. o2d) THEN
258!
259! Computes the flux in the Y direction
260 zflxy(:,:,:)=-zcsv * mym(pk) * gy_m_v(oflat,psvm(:,:,:,jsv),pdyy,pdzz,pdzy)
261!$acc kernels
262!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
263 zflxy(:,:,ike+1) = zflxy(:,:,ike)
264!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
265!$acc end kernels
266!
267! Compute the flux at the first inner V-point with an uncentred vertical
268! gradient
269!
270 zflxy(:,:,ikb) = -zcsv * mym( pk(:,:,ikb) ) * &
271 ( dym(psvm(:,:,ikb,jsv)) * pinv_pdyy(:,:,ikb) &
272 -mym( zcoeff(:,:,2)*psvm(:,:,2,jsv) &
273 +zcoeff(:,:,ikb+1)*psvm(:,:,ikb+1,jsv) &
274 +zcoeff(:,:,ikb )*psvm(:,:,ikb ,jsv) &
275 ) * 0.5 * ( pdzy(:,:,ikb+1)+pdzy(:,:,ikb) ) &
276 * pinv_pdyy(:,:,ikb) &
277 )
278! extrapolates the flux under the ground so that the vertical average with
279! the IKB flux gives the ground value
280!$acc kernels
281!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
282 zwork2d(:,:)=psfsvm(:,:,jsv) * pdircosyw(:,:)
283!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
284!$acc end kernels
285 zflxy(:,:,ikb-1) = 2. * mym( zwork2d(:,:) ) - zflxy(:,:,ikb)
286 !
287 ! stores <V SVth>
288 IF ( tpfile%LOPENED .AND. turbn%LTURB_FLX ) THEN
289 WRITE(ymnhname,'("VSV_FLX_",I3.3)') jsv
290 tzfield = tfieldmetadata( &
291 cmnhname = trim( ymnhname ), &
292 cstdname = '', &
293 clongname = trim(tzfield%CMNHNAME), &
294 cunits = 'SVUNIT m s-1', &
295 cdir = 'XY', &
296 ccomment = 'X_Y_Z_'//trim(tzfield%CMNHNAME), &
297 ngrid = 3, &
298 ntype = typereal, &
299 ndims = 3, &
300 ltimedep = .true. )
301!$acc update self(ZFLXY)
302 CALL io_field_write(tpfile,tzfield,zflxy)
303 END IF
304!
305 ELSE
306 !$acc kernels
307 !$mnh_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
308 zflxy(:,:,:)=0.
309 !$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
310 !$acc end kernels
311 END IF
312!
313 IF (tles%LLES_CALL .AND. ksplt==1) THEN
314 CALL second_mnh(ztime1)
315 CALL les_mean_subgrid( myf(zflxy), tles%X_LES_SUBGRID_VSv(:,:,:,jsv) )
316 CALL les_mean_subgrid( mzf(myf(gy_w_vw(oflat,pwm,pdyy,pdzz,pdzy)*mzm(zflxy))), &
317 tles%X_LES_RES_ddxa_W_SBG_UaSv(:,:,:,jsv) , .true. )
318 CALL les_mean_subgrid( gy_m_m(oflat,psvm(:,:,:,jsv),pdyy,pdzz,pdzy)*myf(zflxy), &
319 tles%X_LES_RES_ddxa_Sv_SBG_UaSv(:,:,:,jsv) , .true. )
320 CALL second_mnh(ztime2)
321 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
322 END IF
323!
324!
325! 15.3 Horizontal source terms
326! -----------------------
327!
328 IF (.NOT. o2d) THEN
329 IF (.NOT. oflat) THEN
330 prsvs(:,:,:,jsv)= prsvs(:,:,:,jsv) &
331 -dxf( mxm(prhodj) * zflxx * pinv_pdxx ) &
332 -dyf( mym(prhodj) * zflxy * pinv_pdyy ) &
333 +dzf( pmzm_prhodj * pinv_pdzz * &
334 ( mxf( mzm(zflxx * pinv_pdxx) * pdzx ) + myf( mzm(zflxy * pinv_pdyy) * pdzy ) ) &
335 )
336 ELSE
337 prsvs(:,:,:,jsv)= prsvs(:,:,:,jsv) &
338 -dxf( mxm(prhodj) * zflxx * pinv_pdxx ) &
339 -dyf( mym(prhodj) * zflxy * pinv_pdyy )
340 END IF
341 ELSE
342 IF (.NOT. oflat) THEN
343 prsvs(:,:,:,jsv)= prsvs(:,:,:,jsv) &
344 -dxf( mxm(prhodj) * zflxx * pinv_pdxx ) &
345 +dzf( pmzm_prhodj * pinv_pdzz * &
346 ( mxf( mzm(zflxx * pinv_pdxx) * pdzx ) ) &
347 )
348 ELSE
349 prsvs(:,:,:,jsv)= prsvs(:,:,:,jsv) &
350 -dxf( mxm(prhodj) *zflxx * pinv_pdxx )
351 END IF
352 END IF
353!
354!
355END DO ! end loop JSV
356!
357!
358END SUBROUTINE turb_hor_sv_flux
359END MODULE mode_turb_hor_sv_flux
