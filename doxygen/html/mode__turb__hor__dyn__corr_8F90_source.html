192! --------------------------------------------------------------------------
193!
194!* 1. PRELIMINARY COMPUTATIONS
195! ------------------------
196NULLIFY(tzfields_ll)
197!
198ikb = 1+jpvext
199ike = SIZE(pum,3)-jpvext
200iku = SIZE(pum,3)
201!
202!
203zdirsinzw(:,:) = sqrt( 1. - pdircoszw(:,:)**2 )
204!
205gx_u_m_pum = gx_u_m(pum,pdxx,pdzz,pdzx)
206IF (.NOT. o2d) gy_v_m_pvm = gy_v_m(pvm,pdyy,pdzz,pdzy)
207gz_w_m_pwm = gz_w_m(pwm,pdzz)
208!
209zmzf_dzz = mzf(pdzz)
210!
211CALL add3dfield_ll( tzfields_ll, zflx, 'TURB_HOR_DYN_CORR::ZFLX' )
212
213
214! compute the coefficients for the uncentred gradient computation near the
215! ground
216!
217!* 9. < U'U'>
218! -------
219!
220! Computes the U variance
221IF (.NOT. o2d) THEN
222 zflx(:,:,:)= (2./3.) * ptkem &
223 - xcmfs * pk *( (4./3.) * gx_u_m_pum &
224 -(2./3.) * ( gy_v_m_pvm &
225 +gz_w_m_pwm ) )
226 !! & to be tested later
227 !! + XCMFB * PLM / SQRT(PTKEM) * (-2./3.) * PTP
228ELSE
229 zflx(:,:,:)= (2./3.) * ptkem &
230 - xcmfs * pk *( (4./3.) * gx_u_m_pum &
231 -(2./3.) * ( gz_w_m_pwm ) )
232 !! & to be tested later
233 !! + XCMFB * PLM / SQRT(PTKEM) * (-2./3.) * PTP
234END IF
235!
236zflx(:,:,ike+1) = zflx(:,:,ike)
237!
238!* prescription of du/dz and dv/dz with uncentered gradient at the surface
239! prescription of dw/dz at Dz/2 above ground using the continuity equation
240! using a Boussinesq hypothesis to remove the z dependance of rhod_ref
241! (div u = 0)
242!
243zdzz(:,:,:) = mxm(pdzz(:,:,ikb:ikb+2))
244zcoeff(:,:,ikb+2)= - zdzz(:,:,2) / &
245 ( (zdzz(:,:,3)+zdzz(:,:,2)) * zdzz(:,:,3) )
246zcoeff(:,:,ikb+1)= (zdzz(:,:,3)+zdzz(:,:,2)) / &
247 ( zdzz(:,:,2) * zdzz(:,:,3) )
248zcoeff(:,:,ikb)= - (zdzz(:,:,3)+2.*zdzz(:,:,2)) / &
249 ( (zdzz(:,:,3)+zdzz(:,:,2)) * zdzz(:,:,2) )
250!
251zdu_dz_dzs_dx(:,:,:)=mxf((zcoeff(:,:,ikb+2:ikb+2)*pum(:,:,ikb+2:ikb+2) &
252 +zcoeff(:,:,ikb+1:ikb+1)*pum(:,:,ikb+1:ikb+1) &
253 +zcoeff(:,:,ikb :ikb )*pum(:,:,ikb :ikb ) &
254 )* 0.5 * ( pdzx(:,:,ikb+1:ikb+1)+pdzx(:,:,ikb:ikb)) &
255 )/ mxf(pdxx(:,:,ikb:ikb))
256!
257zdzz(:,:,:) = mym(pdzz(:,:,ikb:ikb+2))
258zcoeff(:,:,ikb+2)= - zdzz(:,:,2) / &
259 ( (zdzz(:,:,3)+zdzz(:,:,2)) * zdzz(:,:,3) )
260zcoeff(:,:,ikb+1)= (zdzz(:,:,3)+zdzz(:,:,2)) / &
261 ( zdzz(:,:,2) * zdzz(:,:,3) )
262zcoeff(:,:,ikb)= - (zdzz(:,:,3)+2.*zdzz(:,:,2)) / &
263 ( (zdzz(:,:,3)+zdzz(:,:,2)) * zdzz(:,:,2) )
264!
265
266zdv_dz_dzs_dy(:,:,:)=myf((zcoeff(:,:,ikb+2:ikb+2)*pvm(:,:,ikb+2:ikb+2) &
267 +zcoeff(:,:,ikb+1:ikb+1)*pvm(:,:,ikb+1:ikb+1) &
268 +zcoeff(:,:,ikb :ikb )*pvm(:,:,ikb :ikb ) &
269 )* 0.5 * ( pdzy(:,:,ikb+1:ikb+1)+pdzy(:,:,ikb:ikb)) &
270 )/ myf(pdyy(:,:,ikb:ikb))
271!
272!
273zdu_dx(:,:,:)= dxf(pum(:,:,ikb:ikb)) / mxf(pdxx(:,:,ikb:ikb)) &
274 - zdu_dz_dzs_dx(:,:,:)
275
276zdv_dy(:,:,:)= dyf(pvm(:,:,ikb:ikb)) / myf(pdyy(:,:,ikb:ikb)) &
277 - zdv_dz_dzs_dy(:,:,:)
278!
279zdw_dz(:,:,:)=-zdu_dx(:,:,:)-zdv_dy(:,:,:)
280!
281!* computation
282!
283zflx(:,:,ikb) = (2./3.) * ptkem(:,:,ikb) &
284 - xcmfs * pk(:,:,ikb) * 2. * zdu_dx(:,:,1)
285
286
287!! & to be tested later
288!! + XCMFB * PLM(:,:,IKB:IKB) /SQRT(PTKEM(:,:,IKB:IKB)) * &
289!! (-2./3.) * PTP(:,:,IKB:IKB)
290!
291! extrapolates this flux under the ground with the surface flux
292zflx(:,:,ikb-1) = &
293 ptau11m(:,:) * pcosslope(:,:)**2 * pdircoszw(:,:)**2 &
294 -2. * ptau12m(:,:) * pcosslope(:,:)* psinslope(:,:) * pdircoszw(:,:) &
295 + ptau22m(:,:) * psinslope(:,:)**2 &
296 + ptau33m(:,:) * pcosslope(:,:)**2 * zdirsinzw(:,:)**2 &
297 +2. * pcdueff(:,:) * ( &
298 pvslopem(:,:) * pcosslope(:,:) * psinslope(:,:) * zdirsinzw(:,:) &
299 - puslopem(:,:) * pcosslope(:,:)**2 * zdirsinzw(:,:) * pdircoszw(:,:) )
300!
301zflx(:,:,ikb-1) = 2. * zflx(:,:,ikb-1) - zflx(:,:,ikb)
302!
303CALL update_halo_ll(tzfields_ll, iinfo_ll)
304IF ( tpfile%LOPENED .AND. turbn%LTURB_FLX ) THEN
305 ! stores <U U>
306 tzfield = tfieldmetadata( &
307 cmnhname = 'U_VAR', &
308 cstdname = '', &
309 clongname = 'U_VAR', &
310 cunits = 'm2 s-2', &
311 cdir = 'XY', &
312 ccomment = 'X_Y_Z_U_VAR', &
313 ngrid = 1, &
314 ntype = typereal, &
315 ndims = 3, &
316 ltimedep = .true. )
317 CALL io_field_write(tpfile,tzfield,zflx)
318END IF
319!
320! Complete the U tendency
321IF (.NOT. oflat) THEN
322CALL mppdb_check3dm("before turb_corr:PRUS,PRHODJ,ZFLX,PDXX,PDZX,PINV_PDZZ",precision,&
323 & prus,prhodj,zflx,pdxx,pdzx,pinv_pdzz )
324
325 prus(:,:,:)=prus &
326 -dxm(prhodj * zflx / mxf(pdxx) ) &
327 +dzf( pdzx / mzm(pdxx) * mxm( mzm(prhodj*zflx) * pinv_pdzz ) )
328CALL mppdb_check3dm("after turb_corr:PRUS,PRHODJ,ZFLX,PDXX,PDZX,PINV_PDZZ",precision,&
329 & prus,prhodj,zflx,pdxx,pdzx,pinv_pdzz )
330ELSE
331 prus(:,:,:)=prus -dxm(prhodj * zflx / mxf(pdxx) )
332END IF
333!
334IF (ksplt==1) THEN
335 ! Contribution to the dynamic production of TKE:
336 zwork(:,:,:) = - zflx(:,:,:) * gx_u_m_pum
337 !
338 ! evaluate the dynamic production at w(IKB+1) in PDP(IKB)
339 !
340 zwork(:,:,ikb) = 0.5* ( -zflx(:,:,ikb)*zdu_dx(:,:,1) + zwork(:,:,ikb+1) )
341 !
342 pdp(:,:,:) = pdp(:,:,:) + zwork(:,:,:)
343END IF
344!
345! Storage in the LES configuration
346!
347IF (tles%LLES_CALL .AND. ksplt==1) THEN
348 CALL second_mnh(ztime1)
349 CALL les_mean_subgrid( zflx, tles%X_LES_SUBGRID_U2 )
350 CALL les_mean_subgrid( -zwork, tles%X_LES_RES_ddxa_U_SBG_UaU , .true.)
351 CALL second_mnh(ztime2)
352 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
353END IF
354
355!
356!* 10. < V'V'>
357! -------
358!
359! Computes the V variance
360IF (.NOT. o2d) THEN
361 zflx(:,:,:)= (2./3.) * ptkem &
362 - xcmfs * pk *( (4./3.) * gy_v_m_pvm &
363 -(2./3.) * ( gx_u_m_pum &
364 +gz_w_m_pwm ) )
365 !! & to be tested
366 !! + XCMFB * PLM / SQRT(PTKEM) * (-2./3.) * PTP
367 !
368ELSE
369 zflx(:,:,:)= (2./3.) * ptkem &
370 - xcmfs * pk *(-(2./3.) * ( gx_u_m_pum &
371 +gz_w_m_pwm ) )
372 !! & to be tested
373 !! + XCMFB * PLM / SQRT(PTKEM) * (-2./3.) * PTP
374 !
375END IF
376!
377zflx(:,:,ike+1) = zflx(:,:,ike)
378!
379zflx(:,:,ikb) = (2./3.) * ptkem(:,:,ikb) &
380 - xcmfs * pk(:,:,ikb) * 2. * zdv_dy(:,:,1)
381
382!! & to be tested
383!! + XCMFB * PLM(:,:,IKB:IKB) /SQRT(PTKEM(:,:,IKB:IKB)) * &
384!! (-2./3.) * PTP(:,:,IKB:IKB)
385!
386! extrapolates this flux under the ground with the surface flux
387zflx(:,:,ikb-1) = &
388 ptau11m(:,:) * psinslope(:,:)**2 * pdircoszw(:,:)**2 &
389 +2. * ptau12m(:,:) * pcosslope(:,:)* psinslope(:,:) * pdircoszw(:,:) &
390 + ptau22m(:,:) * pcosslope(:,:)**2 &
391 + ptau33m(:,:) * psinslope(:,:)**2 * zdirsinzw(:,:)**2 &
392 -2. * pcdueff(:,:)* ( &
393 puslopem(:,:) * psinslope(:,:)**2 * zdirsinzw(:,:) * pdircoszw(:,:) &
394 + pvslopem(:,:) * pcosslope(:,:) * psinslope(:,:) * zdirsinzw(:,:) )
395!
396zflx(:,:,ikb-1) = 2. * zflx(:,:,ikb-1) - zflx(:,:,ikb)
397!
398CALL update_halo_ll(tzfields_ll, iinfo_ll)
399!
400IF ( tpfile%LOPENED .AND. turbn%LTURB_FLX ) THEN
401 ! stores <V V>
402 tzfield = tfieldmetadata( &
403 cmnhname = 'V_VAR', &
404 cstdname = '', &
405 clongname = 'V_VAR', &
406 cunits = 'm2 s-2', &
407 cdir = 'XY', &
408 ccomment = 'X_Y_Z_V_VAR', &
409 ngrid = 1, &
410 ntype = typereal, &
411 ndims = 3, &
412 ltimedep = .true. )
413 CALL io_field_write(tpfile,tzfield,zflx)
414END IF
415!
416! Complete the V tendency
417IF (.NOT. o2d) THEN
418 IF (.NOT. oflat) THEN
419 prvs(:,:,:)=prvs &
420 -dym(prhodj * zflx / myf(pdyy) ) &
421 +dzf( pdzy / mzm(pdyy) * &
422 mym( mzm(prhodj*zflx) * pinv_pdzz ) )
423 ELSE
424 prvs(:,:,:)=prvs -dym(prhodj * zflx / myf(pdyy) )
425 END IF
426!
427! Contribution to the dynamic production of TKE:
428 IF (ksplt==1) zwork(:,:,:) = - zflx(:,:,:) * gy_v_m_pvm
429ELSE
430 zwork(:,:,:) = 0.
431END IF
432!
433IF (ksplt==1) THEN
434 !
435 ! evaluate the dynamic production at w(IKB+1) in PDP(IKB)
436 !
437 zwork(:,:,ikb) = 0.5* ( -zflx(:,:,ikb)*zdv_dy(:,:,1) + zwork(:,:,ikb+1) )
438 !
439 pdp(:,:,:) = pdp(:,:,:) + zwork(:,:,:)
440END IF
441!
442! Storage in the LES configuration
443!
444IF (tles%LLES_CALL .AND. ksplt==1) THEN
445 CALL second_mnh(ztime1)
446 CALL les_mean_subgrid( zflx, tles%X_LES_SUBGRID_V2 )
447 CALL les_mean_subgrid( -zwork, tles%X_LES_RES_ddxa_V_SBG_UaV , .true.)
448 CALL second_mnh(ztime2)
449 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
450END IF
451!
452!* 11. < W'W'>
453! -------
454!
455! Computes the W variance
456IF (.NOT. o2d) THEN
457 zflx(:,:,:)= (2./3.) * ptkem &
458 - xcmfs * pk *( (4./3.) * gz_w_m_pwm &
459 -(2./3.) * ( gx_u_m_pum &
460 +gy_v_m_pvm ) )
461 !! & to be tested
462 !! -2.* XCMFB * PLM / SQRT(PTKEM) * (-2./3.) * PTP
463ELSE
464 zflx(:,:,:)= (2./3.) * ptkem &
465 - xcmfs * pk *( (4./3.) * gz_w_m_pwm &
466 -(2./3.) * ( gx_u_m_pum ) )
467 !! & to be tested
468 !! -2.* XCMFB * PLM / SQRT(PTKEM) * (-2./3.) * PTP
469END IF
470!
471zflx(:,:,ike+1)= zflx(:,:,ike)
472!
473zflx(:,:,ikb) = (2./3.) * ptkem(:,:,ikb) &
474 - xcmfs * pk(:,:,ikb) * 2. * zdw_dz(:,:,1)
475
476! & to be tested
477! - 2.* XCMFB * PLM(:,:,IKB:IKB) /SQRT(PTKEM(:,:,IKB:IKB)) * &
478! (-2./3.) * PTP(:,:,IKB:IKB)
479!
480! extrapolates this flux under the ground with the surface flux
481zflx(:,:,ikb-1) = &
482 ptau11m(:,:) * zdirsinzw(:,:)**2 &
483 + ptau33m(:,:) * pdircoszw(:,:)**2 &

518 - xcmfs * pk(:,:,ikb+1:) * (4./3.) * (gz_w_m_zwp(:,:,ikb+1:) - gz_w_m_pwm(:,:,ikb+1:))
519!

523 zwork(:,:,:) = - zflx(:,:,:) * gz_w_m_zwp

540 CALL les_mean_subgrid(zflx*mzf(gz_m_w(1,iku,1,pthlm,pdzz)),tles%X_LES_RES_ddz_Thl_SBG_W2)
541 IF (krr>=1) THEN
542 CALL les_mean_subgrid( gz_m_m(prm(:,:,:,1),pdzz)*zflx, &
543 tles%X_LES_RES_ddxa_Rt_SBG_UaW , .true.)

545 tles%X_LES_RES_ddz_Rt_SBG_W2)
546 END IF
547 DO jsv=1,ksv
548 CALL les_mean_subgrid( gz_m_m(psvm(:,:,:,jsv),pdzz)*zflx, &

551 tles%X_LES_RES_ddz_Sv_SBG_W2(:,:,:,jsv))
552 END DO
553 CALL second_mnh(ztime2)
554 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
555END IF

557CALL cleanlist_ll(tzfields_ll)
