273!$acc end kernels
274!
275!* prescription of du/dz and dv/dz with uncentered gradient at the surface
276! prescription of dw/dz at Dz/2 above ground using the continuity equation
277! using a Boussinesq hypothesis to remove the z dependance of rhod_ref
278! (div u = 0)
279!
280! ZDZZ(:,:,:) = MXM(PDZZ(:,:,IKB:IKB+2)) ! case not handled yet by pyft --expandAllArraysConcurrent and shumanFUNCtoCALL

282zdzz(:,:,ikb+1) = mxm(pdzz(:,:,ikb+1))
283zdzz(:,:,ikb+2) = mxm(pdzz(:,:,ikb+2))
284!$acc kernels present_cr(zdzz,zcoeff) ! async(3)
285!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
286 zcoeff(:,:,ikb+2)= - zdzz(:,:,2) / &
287 ( (zdzz(:,:,3)+zdzz(:,:,2)) * zdzz(:,:,3) )
288 zcoeff(:,:,ikb+1)= (zdzz(:,:,3)+zdzz(:,:,2)) / &
289 ( zdzz(:,:,2) * zdzz(:,:,3) )
290 zcoeff(:,:,ikb)= - (zdzz(:,:,3)+2.*zdzz(:,:,2)) / &
291 ( (zdzz(:,:,3)+zdzz(:,:,2)) * zdzz(:,:,2) )
292!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
293!$acc end kernels
294!
295zdu_dz_dzs_dx(:,:)=mxf((zcoeff(:,:,ikb+2)*pum(:,:,ikb+2) &
296 +zcoeff(:,:,ikb+1)*pum(:,:,ikb+1) &
297 +zcoeff(:,:,ikb)*pum(:,:,ikb) &
298 )* 0.5 * ( pdzx(:,:,ikb+1)+pdzx(:,:,ikb)) &
299 )/ mxf(pdxx(:,:,ikb))
300!
301zdzz(:,:,ikb) = mym(pdzz(:,:,ikb))
302zdzz(:,:,ikb+1) = mym(pdzz(:,:,ikb+1))
303zdzz(:,:,ikb+2) = mym(pdzz(:,:,ikb+2))
304!
305!$acc kernels present_cr(zdzz,zcoeff) ! async(4)
306!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
307 zcoeff(:,:,ikb+2)= - zdzz(:,:,2) / &
308 ( (zdzz(:,:,3)+zdzz(:,:,2)) * zdzz(:,:,3) )
309 zcoeff(:,:,ikb+1)= (zdzz(:,:,3)+zdzz(:,:,2)) / &
310 ( zdzz(:,:,2) * zdzz(:,:,3) )
311 zcoeff(:,:,ikb)= - (zdzz(:,:,3)+2.*zdzz(:,:,2)) / &
312 ( (zdzz(:,:,3)+zdzz(:,:,2)) * zdzz(:,:,2) )
313!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
314!$acc end kernels
315
316zdv_dz_dzs_dy(:,:)=myf((zcoeff(:,:,ikb+2)*pvm(:,:,ikb+2) &
317 +zcoeff(:,:,ikb+1)*pvm(:,:,ikb+1) &
318 +zcoeff(:,:,ikb)*pvm(:,:,ikb) &
319 )* 0.5 * ( pdzy(:,:,ikb+1)+pdzy(:,:,ikb)) &
320 )/ myf(pdyy(:,:,ikb))
321!
322!
323zdu_dx(:,:)= dxf(pum(:,:,ikb)) / mxf(pdxx(:,:,ikb)) &
324 - zdu_dz_dzs_dx(:,:)
325
326zdv_dy(:,:)= dyf(pvm(:,:,ikb)) / myf(pdyy(:,:,ikb)) &
327 - zdv_dz_dzs_dy(:,:)

329!$acc kernels present_cr(zdv_dy,zdw_dz) ! async(4)
330!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
331zdw_dz(:,:)=-zdu_dx(:,:)-zdv_dy(:,:)
332!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
333!$acc end kernels
334!
335!* computation
336!
337!!! wait for the computation of ZFLX
338!!$acc ! wait(2) async(4)
339!!! wait for the computation of ZDW_DZ
340!!$acc ! wait(4)
341!
342! ! !!! we can launch the update of ZFLX on the part that has already been computed
343! ! !$acc update self(ZFLX(:,:,IKB+1:)) async(10)
344!attention !!!!! je ne comprends pas pourquoi mais ce update plante √† l'execution...

439IF (ksplt==1) THEN

583 -dym(prhodj * zflx / myf(pdyy) ) &

606IF (ksplt==1) THEN

611 !$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
612 zwork(:,:,ikb) = 0.5* ( -zflx(:,:,ikb)*zdv_dy(:,:) + zwork(:,:,ikb+1) )
613 !$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)

618 pdp(:,:,:) = pdp(:,:,:) + zwork(:,:,:)

627 CALL les_mean_subgrid( zflx, tles%X_LES_SUBGRID_V2 )

630 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1

635!
636! Computes the W variance
637IF (.NOT. o2d) THEN
638 !$acc kernels present_cr(gy_v_m_pvm,zflx) ! async(2)
639 !$mnh_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)

653 -(2./3.) * ( gx_u_m_pum(:,:,:) ) )
654 !$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)

660!$acc kernels ! async(2)
661!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
662zflx(:,:,ike+1)= zflx(:,:,ike)
663!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
664!$acc end kernels
665!!! wait for the computation of ZWORK, PDP and ZFLX
666!!$acc ! wait(2)
667!
668!
669!$acc kernels present_cr(zdw_dz,zflx) ! async(2)
670!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
671 zflx(:,:,ikb) = (2./3.) * ptkem(:,:,ikb) &
672 - xcmfs * pk(:,:,ikb) * 2. * zdw_dz(:,:)
673!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
674
675!$acc end kernels
676!
677
678! & to be tested
679! - 2.* XCMFB * PLM(:,:,IKB:IKB) /SQRT(PTKEM(:,:,IKB:IKB)) * &
680! (-2./3.) * PTP(:,:,IKB:IKB)
681! extrapolates this flux under the ground with the surface flux
682!$acc kernels present_cr(ZFLX) ! async(3)

685 ptau11m(:,:) * zdirsinzw(:,:)**2 &

773 !$acc kernels ! async(2)

775 pdp(:,:,:) = pdp(:,:,:) + zwork(:,:,:)

777 !$acc end kernels
778END IF
779!
780! Storage in the LES configuration
781!
782!
783IF (tles%LLES_CALL .AND. ksplt==1) THEN
784 CALL second_mnh(ztime1)
785 CALL les_mean_subgrid( zflx, tles%X_LES_SUBGRID_W2 )
786 CALL les_mean_subgrid( -zwork, tles%X_LES_RES_ddxa_W_SBG_UaW , .true.)
787 CALL les_mean_subgrid( gz_m_m(pthlm,pdzz)*zflx, tles%X_LES_RES_ddxa_Thl_SBG_UaW , .true.)
788 CALL les_mean_subgrid(zflx*mzf(gz_m_w(pthlm,pdzz)),tles%X_LES_RES_ddz_Thl_SBG_W2)
789 IF (krr>=1) THEN
790 CALL les_mean_subgrid( gz_m_m(prm(:,:,:,1),pdzz)*zflx, &
791 tles%X_LES_RES_ddxa_Rt_SBG_UaW , .true.)
792 CALL les_mean_subgrid(zflx*mzf(gz_m_w(prm(:,:,:,1),pdzz)), &
793 tles%X_LES_RES_ddz_Rt_SBG_W2)
794 END IF
795 DO jsv=1,ksv
796 CALL les_mean_subgrid( gz_m_m(psvm(:,:,:,jsv),pdzz)*zflx, &
797 tles%X_LES_RES_ddxa_Sv_SBG_UaW(:,:,:,jsv) , .true.)
798 CALL les_mean_subgrid(zflx*mzf(gz_m_w(psvm(:,:,:,jsv),pdzz)), &
799 tles%X_LES_RES_ddz_Sv_SBG_W2(:,:,:,jsv))

801 CALL second_mnh(ztime2)
802 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
803END IF

805CALL cleanlist_ll(tzfields_ll)
