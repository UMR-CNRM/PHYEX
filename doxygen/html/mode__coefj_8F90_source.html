16  FUNCTION coefj(PTHL,PEXNREF,PFRAC_ICE) RESULT(PCOEFJ)
17 ! #######################################################
18 !
19 ! PURPOSE
20 !! -------
21 ! COEFJ computes the coefficient J of the documentation.
22 !
23 !!** METHOD
24 !! ------ rvs(Tl) Lv(Tl)
25 !! The value of this coefficient is J = --------------, for rc only
26 !! Rv Tl THETAl
27 !!
28 !! rvsw(Tl) Lv(Tl) rvsi(Tl) Ls(Tl)
29 !! or --------------- (1-Pfrac_ri) + --------------- Pfrac_ri, for rc+ri.
30 !! Rv Tl THETAl Rv Tl THETAl
31 !!
32 !! EXTERNAL
33 !! --------
34 !! None.
35 !!
36 !! IMPLICIT ARGUMENTS
37 !! ------------------
38 !! Module MODD_CST : contains physical constants.
39 !!
40 !! REFERENCE
41 !! ---------
42 !! Book 1 of documentation of Meso-NH
43 !! Book 2 of documentation of Meso-NH
44 !!
45 !!
46 !! AUTHOR
47 !! ------
48 !! Jean-Marie Carriere * Meteo-France *
49 !!
50 !! MODIFICATIONS
51 !! -------------
52 !! Original 20/03/95
53 !! J.-P. Pinty 20/02/03 add non-precipitating ice
54 !!
55 !! ----------------------------------------------------------------------
56 !
57 !* 0. DECLARATIONS
58 ! ------------
59 USE modd_cst
60 !
61 IMPLICIT NONE
62 !
63 !* 0.1 declarations of arguments and result
64 !
65 REAL, DIMENSION(:,:,:), INTENT(IN) :: pthl ! Temperature variable
66 REAL, DIMENSION(:,:,:), INTENT(IN) :: pexnref ! Exner function of the
67 ! reference state
68 REAL, DIMENSION(:,:,:), INTENT(IN), OPTIONAL :: pfrac_ice
69  ! Fraction of ri in the
70  ! non-precipating
71  ! "rc+ri" condensate
72 !
73 REAL,DIMENSION(SIZE(PTHL,1),SIZE(PTHL,2),SIZE(PTHL,3)):: pcoefj ! result
74 !
75 !* 0.2 declarations of local variables
76 !
77 REAL,DIMENSION(SIZE(PTHL,1),SIZE(PTHL,2),SIZE(PTHL,3)) :: &
78  ztl, zl, zes, zrvs, zp
79 ! ZTL = Tl, ZL = Lv(Tl) or Ls(Tl), ZES = esw(Tl) or esi(Tl)
80 ! ZRVS = rvsw(Tl) or rvsi(Tl), ZP = p
81 !
82 REAL :: zeps ! = Mv/Md
83 !---------------------------------------------------------------------------
84 !
85 !* 1. COMPUTATION OF Tl
86 ! -----------------
87 !
88 ztl(:,:,:) = pthl(:,:,:) * pexnref(:,:,:)
89 !
90 !* 2. COMPUTATION OF Lv(Tl)
91 ! ---------------------
92 !
93 zl(:,:,:) = xlvtt + ( xcpv - xcl ) * ( ztl(:,:,:) -xtt )
94 !
95 !* 3. COMPUTATION OF rvs(Tl)
96 ! ----------------------
97 !
98 zeps = xmv/xmd
99 zp(:,:,:) = (pexnref(:,:,:)**(xcpd/xrd))*xp00
100 zes(:,:,:) = exp( xalpw - xbetaw/ztl(:,:,:) - xgamw*alog(ztl(:,:,:) ) )
101 zrvs(:,:,:) = zes(:,:,:) * zeps / ( zp(:,:,:) - zes(:,:,:) )
102 !
103 ! 4. RESULT FOR rc only
104 ! ------------------
105 !
106 pcoefj(:,:,:) = zrvs(:,:,:)*zl(:,:,:)/ &
107  ( xrv*ztl(:,:,:)*pthl(:,:,:) )
108 !
109 ! Add case when rc+ri
110 !
111 IF(PRESENT(pfrac_ice)) THEN
112 !
113 !* 5. COMPUTATION OF Ls(Tl)
114 ! ---------------------
115 !
116  zl(:,:,:) = xlstt + ( xcpv - xci ) * ( ztl(:,:,:) -xtt )
117 !
118 !* 6. COMPUTATION OF rvs(Tl)
119 ! ----------------------
120 !
121  zes(:,:,:) = exp( xalpi - xbetai/ztl(:,:,:) - xgami*alog(ztl(:,:,:) ) )
122  zrvs(:,:,:) = zes(:,:,:) * zeps / ( zp(:,:,:) - zes(:,:,:) )
123 !
124 ! 7. RESULT FOR rc and ri
125 ! --------------------
126 !
127  pcoefj(:,:,:) = (1.0 - pfrac_ice(:,:,:))*pcoefj(:,:,:) &
128  + pfrac_ice(:,:,:) *zrvs(:,:,:)*zl(:,:,:)/ &
129  ( xrv*ztl(:,:,:)*pthl(:,:,:) )
130 END IF
131 !
132 !---------------------------------------------------------------------------
133 !
134 END FUNCTION coefj
135 END MODULE mode_coefj
