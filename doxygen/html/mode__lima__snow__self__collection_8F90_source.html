39USE modd_param_lima_cold, ONLY : nsclbdas, xscintp1s, xscintp2s, xker_n_sscs, xfnsscs, xcolexss, &
40 xlbnsscs1, xlbnsscs2

42IMPLICIT NONE
43!
44!* 0.1 Declarations of dummy arguments :
45!
46LOGICAL, DIMENSION(:),INTENT(IN) :: LDCOMPUTE

48REAL, DIMENSION(:), INTENT(IN) :: PRHODREF ! Reference Exner function

52REAL, DIMENSION(:), INTENT(IN) :: PCST ! Snow C. at t
53REAL, DIMENSION(:), INTENT(IN) :: PLBDS !
54!
55REAL, DIMENSION(:), INTENT(OUT) :: P_CS_SSC
56!
57!* 0.2 Declarations of local variables :
58!
59REAL, DIMENSION(SIZE(PCST)) :: &
60 ZW1, & ! work arrays
61 ZW2
62LOGICAL, DIMENSION(SIZE(PCST)) :: GSSC
63INTEGER :: IGSSC, JL
64INTEGER, DIMENSION(:), ALLOCATABLE :: IVEC1 ! Vectors of indices
65REAL, DIMENSION(:), ALLOCATABLE :: ZVEC1, ZVEC3 ! Work vectors
66!
67!-------------------------------------------------------------------------------
68!
69!
70!* 1. Snow self-collection and break-up
71! ---------------------------------
72!
73!
74p_cs_ssc(:)=0.
75!
76zw1(:) =0.
77zw2(:) =0.
78!
79gssc(:) = pcst(:)>xctmin(5) .AND. prst(:)>xrtmin(5)
80igssc = count(gssc(:))
81!
82IF( igssc>0 ) THEN
83!
84! 1.3N.0 allocations
85!
86 ALLOCATE(zvec1(igssc))
87 ALLOCATE(ivec1(igssc))
88!
89! 1.3N.1 select the (ZLBDAS,ZLBDAS) couplet
90!
91 zvec1(:) = pack( plbds(:),mask=gssc(:) )
92!
93! 1.3N.2 find the next lower indice for the ZLBDAS and for the ZLBDAS
94! in the geometrical set of (Lbda_s,Lbda_s) couplet use to
95! tabulate the SACCS-kernel
96!
97 zvec1(1:igssc) = max( 1.0001, min( float(nsclbdas)-0.0001, &
98 xscintp1s * log( zvec1(1:igssc) ) + xscintp2s ) )
99 ivec1(1:igssc) = int( zvec1(1:igssc) )
100 zvec1(1:igssc) = zvec1(1:igssc) - float( ivec1(1:igssc) )
101!
102! 1.3N.3 perform the bilinear interpolation of the normalized
103! SSCS-kernel
104!
105 ALLOCATE(zvec3(igssc))
106 DO jl = 1,igssc
107 zvec3(jl) = ( xker_n_sscs(ivec1(jl)+1,ivec1(jl)+1)* zvec1(jl) &
108 - xker_n_sscs(ivec1(jl)+1,ivec1(jl) )*(zvec1(jl) - 1.0) ) &
109 * zvec1(jl) &
110 - ( xker_n_sscs(ivec1(jl) ,ivec1(jl)+1)* zvec1(jl) &
111 - xker_n_sscs(ivec1(jl) ,ivec1(jl) )*(zvec1(jl) - 1.0) ) &
112 * (zvec1(jl) - 1.0)
113 END DO
114 zw1(:) = unpack( vector=zvec3(:),mask=gssc(:),field=0.0 ) !! NSACCS
115 DEALLOCATE(zvec3)
116!
117 WHERE( gssc(:) )
118 p_cs_ssc(:) = - xfnsscs * zw1(:) * exp( xcolexss*(pt(:)-xtt) ) * pcst(:)**2 &
119 * prhodref(:)**(-xcexvt-1.) * (xlbnsscs1+xlbnsscs2) / plbds(:)**2
120 END WHERE
121 DEALLOCATE(ivec1)
122 DEALLOCATE(zvec1)
123END IF
124!
125!-------------------------------------------------------------------------------
126!
127END SUBROUTINE lima_snow_self_collection
