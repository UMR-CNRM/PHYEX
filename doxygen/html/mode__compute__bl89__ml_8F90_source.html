138  (0.5*(zhlvpt(jij,kk)+ pvpt(jij,kk)) - zvpt_dep(jij)) + &

140  pdzz2d(jij,kk)*0.5
141  ! Test if it rests some energy to consume
142  ztest =0.5+sign(0.5,zinte(jij)-zpote(jij))
143  ! Length travelled by parcel if it rests energy to consume
144  zlwork1(jij)=pdzz2d(jij,kk)*0.5
145  ! Lenght travelled by parcel to nullify energy
146  zlwork2(jij)= ( - pg_o_thvref(jij) * &
147  ( zhlvpt(jij,kk) - zvpt_dep(jij) ) &
148  - csturb%XRM17*pshear(jij,kk)*sqrt(abs(ptkem_dep(jij))) &
149  + sqrt(abs( &

153  * zdeltvpt(jij,kk) / pdzz2d(jij,kk) )) ) / &
154  ( pg_o_thvref(jij) * zdeltvpt(jij,kk) / pdzz2d(jij,kk) )

158  ! Rest of energy to consume
159  zinte(jij) = zinte(jij) - zpote(jij)
160  ENDDO
161  ELSE
162  !$mnh_expand_array(JIJ=IIJB:IIJE)
163  zvpt_dep(iijb:iije)=pvpt(iijb:iije,kk) ! departure point is on mass level
164  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
165  ENDIF

168  IF(ztestm > 0.) THEN

172  zpote(jij) = ztest0*(pg_o_thvref(jij) * &
173  (zhlvpt(jij,jkk) - zvpt_dep(jij)) &

180  ( pvpt(jij,jkk-ikl) - zvpt_dep(jij) ) &

184  pg_o_thvref(jij) * (pvpt(jij,jkk-ikl) - zvpt_dep(jij)) )**2 &

191  zinte(jij) = zinte(jij) - zpote(jij)
192  END DO
193  ENDIF
194  END DO
195 ENDIF
196 !!
197 !* 2. CALCULATION OF THE DOWNWARD MIXING LENGTH
198 ! ---------------------------------------
199 !
200 
201 IF (oupordn.EQV..false.) THEN
202  IF(oflux) CALL print_msg(nverb_fatal,'GEN','COMPUTE_BL89_ML','OFLUX option not coded for downward mixing length')
203  !$mnh_expand_array(JIJ=IIJB:IIJE)

205  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
206  plwork=0.
207  ztestm=1.
208  DO jkk=kk,ikb,-ikl
209  IF(ztestm > 0.) THEN
210  ztestm=0
211  DO jij=iijb,iije

213  zpote(jij) = ztest0*(-pg_o_thvref(jij) * &

216  ztest =0.5+sign(0.5,zinte(jij)-zpote(jij))
217  ztestm=ztestm+ztest0
