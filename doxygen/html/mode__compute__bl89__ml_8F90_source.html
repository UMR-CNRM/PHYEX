43USE modd_dimphyex, ONLY: dimphyex_t
44USE modd_cst, ONLY: cst_t
45USE modd_cturb, ONLY: csturb_t
46!
47USE mode_msg
48!
49USE modi_shuman_mf, ONLY: dzm_mf, mzm_mf
50!
51IMPLICIT NONE
52!
53! 0.1 arguments
54!
55TYPE(dimphyex_t), INTENT(IN) :: D
56TYPE(cst_t), INTENT(IN) :: CST
57TYPE(csturb_t), INTENT(IN) :: CSTURB
58REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDZZ2D ! height difference between two mass levels
59REAL, DIMENSION(D%NIJT), INTENT(IN) :: PTKEM_DEP ! TKE to consume
60REAL, DIMENSION(D%NIJT), INTENT(IN) :: PG_O_THVREF ! g/ThetaVRef at the departure point
61REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PVPT ! ThetaV on mass levels
62INTEGER, INTENT(IN) :: KK ! index of departure level
63LOGICAL, INTENT(IN) :: OUPORDN ! switch to compute upward (true) or
64 ! downward (false) mixing length
65LOGICAL, INTENT(IN) :: OFLUX ! Computation must be done from flux level
66REAL, DIMENSION(D%NIJT), INTENT(OUT) :: PLWORK ! Resulting mixing length
67REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PSHEAR ! vertical wind shear for RM17 mixing length
68
69! 0.2 Local variable
70!
71REAL, DIMENSION(D%NIJT) :: ZLWORK1,ZLWORK2 ! Temporary mixing length
72REAL, DIMENSION(D%NIJT) :: ZINTE,ZPOTE ! TKE and potential energy
73 ! between 2 levels
74REAL, DIMENSION(D%NIJT) :: ZVPT_DEP ! Thetav on departure point
75!
76REAL, DIMENSION(D%NIJT,D%NKT) :: ZDELTVPT,ZHLVPT
77 !Virtual Potential Temp at Half level and DeltaThv between
78 !2 mass levels
79
80INTEGER :: JKK !loop counters
81INTEGER :: JIJ, JK
82INTEGER :: IIJB,IIJE ! physical horizontal domain indices
83INTEGER :: IKT,IKB,IKA,IKE,IKL
84REAL :: ZTEST,ZTEST0,ZTESTM !test for vectorization
85!-------------------------------------------------------------------------------------
86!
87!* 1. INITIALISATION
88! --------------
89REAL(KIND=jphook) :: zhook_handle
90IF (lhook) CALL dr_hook('COMPUTE_BL89_ML',0,zhook_handle)
91!
92iije=d%NIJE
93iijb=d%NIJB
94ikt=d%NKT
95ikb=d%NKB
96ika=d%NKA
97ike=d%NKE
98ikl=d%NKL
99!
100CALL dzm_mf(d, pvpt(:,:), zdeltvpt(:,:))
101zdeltvpt(iijb:iije,ika)=0.
102!$mnh_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
103WHERE (abs(zdeltvpt(iijb:iije,1:ikt))<csturb%XLINF)
104 zdeltvpt(iijb:iije,1:ikt)=csturb%XLINF
105END WHERE
106!$mnh_end_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
107!
108CALL mzm_mf(d, pvpt(:,:), zhlvpt(:,:))
109!
110!We consider that gradient between mass levels KKB and KKB+KKL is the same as
111!the gradient between flux level KKB and mass level KKB
112!$mnh_expand_array(JIJ=IIJB:IIJE)
113zdeltvpt(iijb:iije,ikb)=pdzz2d(iijb:iije,ikb)*zdeltvpt(iijb:iije,ikb+ikl)/pdzz2d(iijb:iije,ikb+ikl)
114zhlvpt(iijb:iije,ikb)=pvpt(iijb:iije,ikb)-zdeltvpt(iijb:iije,ikb)*0.5
115!$mnh_end_expand_array(JIJ=IIJB:IIJE)
116!
117!
118!
119!* 2. CALCULATION OF THE UPWARD MIXING LENGTH
120! ---------------------------------------
121!
122
123IF (oupordn.EQV..true.) THEN
124 !$mnh_expand_array(JIJ=IIJB:IIJE)
125 zinte(iijb:iije)=ptkem_dep(iijb:iije)
126 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
127 plwork=0.
128 ztestm=1.
129 IF(oflux)THEN
130 !$mnh_expand_array(JIJ=IIJB:IIJE)
131 zvpt_dep(iijb:iije)=zhlvpt(iijb:iije,kk) ! departure point is on flux level
132 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
133 !We must compute what happens between flux level KK and mass level KK
134 DO jij=iijb,iije
135 ztest0=0.5+sign(0.5,zinte(jij)) ! test if there's energy to consume
136 ! Energy consumed if parcel cross the entire layer
137 zpote(jij) = ztest0*(pg_o_thvref(jij) * &
138 (0.5*(zhlvpt(jij,kk)+ pvpt(jij,kk)) - zvpt_dep(jij)) + &
139 csturb%XRM17*pshear(jij,kk)*sqrt(abs(ptkem_dep(jij)))) * &
140 pdzz2d(jij,kk)*0.5
141 ! Test if it rests some energy to consume
142 ztest =0.5+sign(0.5,zinte(jij)-zpote(jij))
143 ! Length travelled by parcel if it rests energy to consume
144 zlwork1(jij)=pdzz2d(jij,kk)*0.5
145 ! Lenght travelled by parcel to nullify energy
146 zlwork2(jij)= ( - pg_o_thvref(jij) * &
147 ( zhlvpt(jij,kk) - zvpt_dep(jij) ) &
148 - csturb%XRM17*pshear(jij,kk)*sqrt(abs(ptkem_dep(jij))) &
149 + sqrt(abs( &
150 (csturb%XRM17*pshear(jij,kk)*sqrt(abs(ptkem_dep(jij))) + &
151 pg_o_thvref(jij) * (zhlvpt(jij,kk) - zvpt_dep(jij)) )**2 &
152 + 2. * zinte(jij) * pg_o_thvref(jij) &
153 * zdeltvpt(jij,kk) / pdzz2d(jij,kk) )) ) / &
154 ( pg_o_thvref(jij) * zdeltvpt(jij,kk) / pdzz2d(jij,kk) )
155 ! Effective length travelled by parcel
156 plwork(jij)=plwork(jij)+ztest0*(ztest*zlwork1(jij)+ &
157 (1-ztest)*zlwork2(jij))
158 ! Rest of energy to consume
159 zinte(jij) = zinte(jij) - zpote(jij)
160 ENDDO
161 ELSE
162 !$mnh_expand_array(JIJ=IIJB:IIJE)
163 zvpt_dep(iijb:iije)=pvpt(iijb:iije,kk) ! departure point is on mass level
164 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
165 ENDIF
166
167 DO jkk=kk+ikl,ike,ikl
168 IF(ztestm > 0.) THEN
169 ztestm=0
170 DO jij=iijb,iije
171 ztest0=0.5+sign(0.5,zinte(jij))
172 zpote(jij) = ztest0*(pg_o_thvref(jij) * &
173 (zhlvpt(jij,jkk) - zvpt_dep(jij)) &
174 + csturb%XRM17*pshear(jij,jkk)*sqrt(abs(ptkem_dep(jij))))* pdzz2d(jij,jkk)
175 ztest =0.5+sign(0.5,zinte(jij)-zpote(jij))
176 ztestm=ztestm+ztest0
177 zlwork1(jij)=pdzz2d(jij,jkk)
178 !ZLWORK2 jump of the last reached level
179 zlwork2(jij)= ( - pg_o_thvref(jij) * &
180 ( pvpt(jij,jkk-ikl) - zvpt_dep(jij) ) &
181 - csturb%XRM17*pshear(jij,jkk)*sqrt(abs(ptkem_dep(jij))) &
182 + sqrt(abs( &
183 (csturb%XRM17*pshear(jij,jkk)*sqrt(abs(ptkem_dep(jij))) + &
184 pg_o_thvref(jij) * (pvpt(jij,jkk-ikl) - zvpt_dep(jij)) )**2 &
185 + 2. * zinte(jij) * pg_o_thvref(jij) &
186 * zdeltvpt(jij,jkk) / pdzz2d(jij,jkk) )) ) / &
187 ( pg_o_thvref(jij) * zdeltvpt(jij,jkk) / pdzz2d(jij,jkk) )
188 !
189 plwork(jij)=plwork(jij)+ztest0*(ztest*zlwork1(jij)+ &
190 (1-ztest)*zlwork2(jij))
191 zinte(jij) = zinte(jij) - zpote(jij)
192 END DO
193 ENDIF
194 END DO
195ENDIF
196!!
197!* 2. CALCULATION OF THE DOWNWARD MIXING LENGTH
198! ---------------------------------------
199!
200
201IF (oupordn.EQV..false.) THEN
202 !$mnh_expand_array(JIJ=IIJB:IIJE)
203 zinte(iijb:iije)=ptkem_dep(iijb:iije)
204 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
205 plwork=0.
206 ztestm=1.
207 IF(oflux) THEN
208 CALL print_msg(nverb_fatal,'GEN','COMPUTE_BL89_ML','OFLUX option not coded for downward mixing length')
209 ELSE
210 !$mnh_expand_array(JIJ=IIJB:IIJE)
211 zvpt_dep(iijb:iije)=pvpt(iijb:iije,kk) ! departure point is on mass level
212 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
213 ENDIF
214
215 DO jkk=kk,ikb,-ikl
216 IF(ztestm > 0.) THEN
217 ztestm=0
218 DO jij=iijb,iije
219 ztest0=0.5+sign(0.5,zinte(jij))
220 zpote(jij) = ztest0*(-pg_o_thvref(jij) * &
221 (zhlvpt(jij,jkk) - zvpt_dep(jij)) &
222 + csturb%XRM17*pshear(jij,jkk)*sqrt(abs(ptkem_dep(jij))))* pdzz2d(jij,jkk)
223 ztest =0.5+sign(0.5,zinte(jij)-zpote(jij))
224 ztestm=ztestm+ztest0
225 zlwork1(jij)=pdzz2d(jij,jkk)
226 zlwork2(jij)= ( + pg_o_thvref(jij) * &
227 ( pvpt(jij,jkk) - zvpt_dep(jij) ) &
228 -csturb%XRM17*pshear(jij,jkk)*sqrt(abs(ptkem_dep(jij))) &
229 + sqrt(abs( &
230 (csturb%XRM17*pshear(jij,jkk)*sqrt(abs(ptkem_dep(jij))) - &
231 pg_o_thvref(jij) * (pvpt(jij,jkk) - zvpt_dep(jij)) )**2 &
232 + 2. * zinte(jij) * pg_o_thvref(jij) &
233 * zdeltvpt(jij,jkk) / pdzz2d(jij,jkk) )) ) / &
234 ( pg_o_thvref(jij) * zdeltvpt(jij,jkk) / pdzz2d(jij,jkk) )
235 !
236 plwork(jij)=plwork(jij)+ztest0*(ztest*zlwork1(jij)+ &
237 (1-ztest)*zlwork2(jij))
238 zinte(jij) = zinte(jij) - zpote(jij)
239 END DO
240 ENDIF
241 END DO
242ENDIF
243
244IF (lhook) CALL dr_hook('COMPUTE_BL89_ML',1,zhook_handle)
245END SUBROUTINE compute_bl89_ml
246END MODULE mode_compute_bl89_ml
