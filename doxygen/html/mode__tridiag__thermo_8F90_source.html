10 ! #################################################
11 !
12 !
13 !!**** *TRIDIAG_THERMO* - routine to solve a time implicit scheme
14 !!
15 !!
16 !! PURPOSE
17 !! -------
18 ! The purpose of this routine is to give a field PVARP at t+1, by
19 ! solving an implicit TRIDIAGonal system obtained by the
20 ! discretization of the vertical turbulent diffusion. It should be noted
21 ! that the degree of implicitness can be varied (PIMPL parameter) and that
22 ! the function of F(dT/dz) must have been linearized.
23 ! PVARP is localized at a mass point.
24 !
25 !!** METHOD
26 !! ------
27 !!
28 !! [T(+) - T(-)]/2Dt = -d{ F + dF/d(dT/dz) * [impl*dT/dz(+) + expl* dT/dz(-)] }/dz
29 !!
30 !! It is discretized as follows:
31 !!
32 !! PRHODJ(k)*PVARP(k)/PTSTEP
33 !! =
34 !! PRHODJ(k)*PVARM(k)/PTSTEP
35 !! - (PRHODJ(k+1)+PRHODJ(k) )/2. * PF(k+1)/PDZZ(k+1)
36 !! + (PRHODJ(k) +PRHODJ(k-1))/2. * PF(k) /PDZZ(k)
37 !! + (PRHODJ(k+1)+PRHODJ(k) )/2. * ZEXPL* PDFDDTDZ(k+1) * PVARM(k+1)/PDZZ(k+1)**2
38 !! - (PRHODJ(k+1)+PRHODJ(k) )/2. * PIMPL* PDFDDTDZ(k+1) * PVARP(k+1)/PDZZ(k+1)**2
39 !! - (PRHODJ(k+1)+PRHODJ(k) )/2. * ZEXPL* PDFDDTDZ(k+1) * PVARM(k) /PDZZ(k+1)**2
40 !! + (PRHODJ(k+1)+PRHODJ(k) )/2. * PIMPL* PDFDDTDZ(k+1) * PVARP(k) /PDZZ(k+1)**2
41 !! - (PRHODJ(k) +PRHODJ(k-1))/2. * ZEXPL* PDFDDTDZ(k) * PVARM(k) /PDZZ(k)**2
42 !! + (PRHODJ(k) +PRHODJ(k-1))/2. * PIMPL* PDFDDTDZ(k) * PVARP(k) /PDZZ(k)**2
43 !! + (PRHODJ(k) +PRHODJ(k-1))/2. * ZEXPL* PDFDDTDZ(k) * PVARM(k-1)/PDZZ(k)**2
44 !! - (PRHODJ(k) +PRHODJ(k-1))/2. * PIMPL* PDFDDTDZ(k) * PVARP(k-1)/PDZZ(k)**2
45 !!
46 !!
47 !! The system to solve is:
48 !!
49 !! A*PVARP(k-1) + B*PVARP(k) + C*PVARP(k+1) = Y(k)
50 !!
51 !!
52 !! The RHS of the linear system in PVARP writes:
53 !!
54 !! y(k) = PRHODJ(k)*PVARM(k)/PTSTEP
55 !! - (PRHODJ(k+1)+PRHODJ(k) )/2. * PF(k+1)/PDZZ(k+1)
56 !! + (PRHODJ(k) +PRHODJ(k-1))/2. * PF(k) /PDZZ(k)
57 !! + (PRHODJ(k+1)+PRHODJ(k) )/2. * ZEXPL* PDFDDTDZ(k+1) * PVARM(k+1)/PDZZ(k+1)**2
58 !! - (PRHODJ(k+1)+PRHODJ(k) )/2. * ZEXPL* PDFDDTDZ(k+1) * PVARM(k) /PDZZ(k+1)**2
59 !! - (PRHODJ(k) +PRHODJ(k-1))/2. * ZEXPL* PDFDDTDZ(k) * PVARM(k) /PDZZ(k)**2
60 !! + (PRHODJ(k) +PRHODJ(k-1))/2. * ZEXPL* PDFDDTDZ(k) * PVARM(k-1)/PDZZ(k)**2
61 !!
62 !!
63 !! Then, the classical TRIDIAGonal algorithm is used to invert the
64 !! implicit operator. Its matrix is given by:
65 !!
66 !! ( b(ikb) c(ikb) 0 0 0 0 0 0 )
67 !! ( 0 a(ikb+1) b(ikb+1) c(ikb+1) 0 ... 0 0 0 )
68 !! ( 0 0 a(ikb+2) b(ikb+2) c(ikb+2). 0 0 0 )
69 !! .......................................................................
70 !! ( 0 ... 0 a(k) b(k) c(k) 0 ... 0 0 )
71 !! .......................................................................
72 !! ( 0 0 0 0 0 ...a(ike-1) b(ike-1) c(ike-1))
73 !! ( 0 0 0 0 0 ... 0 a(ike) b(ike) )
74 !!
75 !! ikb and ike represent the first and the last inner mass levels of the
76 !! model. The coefficients are:
77 !!
78 !! a(k) = + (PRHODJ(k) +PRHODJ(k-1))/2. * PIMPL* PDFDDTDZ(k) /PDZZ(k)**2
79 !! b(k) = PRHODJ(k) / PTSTEP
80 !! - (PRHODJ(k+1)+PRHODJ(k) )/2. * PIMPL* PDFDDTDZ(k+1)/PDZZ(k+1)**2
81 !! - (PRHODJ(k) +PRHODJ(k-1))/2. * PIMPL* PDFDDTDZ(k) /PDZZ(k)**2
82 !! c(k) = + (PRHODJ(k+1)+PRHODJ(k) )/2. * PIMPL* PDFDDTDZ(k+1)/PDZZ(k+1)**2
83 !!
84 !! for all k /= ikb or ike
85 !!
86 !!
87 !! b(ikb) = PRHODJ(ikb) / PTSTEP
88 !! -(PRHODJ(ikb+1)+PRHODJ(ikb))/2.*PIMPL*PDFDDTDZ(ikb+1)/PDZZ(ikb+1)**2
89 !! c(ikb) = +(PRHODJ(ikb+1)+PRHODJ(ikb))/2.*PIMPL*PDFDDTDZ(ikb+1)/PDZZ(ikb+1)**2
90 !!
91 !! b(ike) = PRHODJ(ike) / PTSTEP
92 !! -(PRHODJ(ike)+PRHODJ(ike-1))/2.*PIMPL*PDFDDTDZ(ike)/PDZZ(ike)**2
93 !! a(ike) = +(PRHODJ(ike)+PRHODJ(ike-1))/2.*PIMPL*PDFDDTDZ(ike)/PDZZ(ike)**2
94 !!
95 !!
96 !! EXTERNAL
97 !! --------
98 !!
99 !! NONE
100 !!
101 !! IMPLICIT ARGUMENTS
102 !! ------------------
103 !!
104 !! REFERENCE
105 !! ---------
106 !! Press et al: Numerical recipes (1986) Cambridge Univ. Press
107 !!
108 !! AUTHOR
109 !! ------
110 !! V. Masson * Meteo-France *

112 !! MODIFICATIONS

117 !* 0. DECLARATIONS

119 USE yomhook , ONLY : lhook, dr_hook, jphook
120 USE modd_dimphyex, ONLY : dimphyex_t

127 !* 0.1 declarations of arguments

129 TYPE(dimphyex_t), INTENT(IN) :: D
130 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PVARM ! variable at t-1 at mass point
131 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PF ! flux in dT/dt=-dF/dz at flux point
132 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDFDDTDZ! dF/d(dT/dz) at flux point
133 REAL, INTENT(IN) :: PTSTEP ! Double time step
134 REAL, INTENT(IN) :: PIMPL ! implicit weight
135 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDZZ ! Dz at flux point
136 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODJ ! (dry rho)*J at mass point
137 !
138 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT):: PVARP ! variable at t+1 at mass point
139 !
140 !
141 !* 0.2 declarations of local variables
142 !
143 REAL, DIMENSION(D%NIJT,D%NKT) :: ZRHODJ_DFDDTDZ_O_DZ2
144 REAL, DIMENSION(D%NIJT,D%NKT) :: ZMZM_RHODJ
145 REAL, DIMENSION(D%NIJT,D%NKT) :: ZA, ZB, ZC
146 REAL, DIMENSION(D%NIJT,D%NKT) :: ZY ,ZGAM
147  ! RHS of the equation, 3D work array
148 REAL, DIMENSION(D%NIJT) :: ZBET
149  ! 2D work array
150 INTEGER :: JIJ,JK ! loop counter
151 INTEGER :: IKB,IKE ! inner limits
152 INTEGER :: IKT,IKA,IKU ! array size in k direction
153 INTEGER :: IKTB,IKTE ! start, end of k loops in physical domain
154 INTEGER :: IIJB,IIJE ! start, end of ij loops in physical domain
155 INTEGER :: IKL
156 !
157 ! ---------------------------------------------------------------------------
158 !
159 !* 1. Preliminaries
160 ! -------------
161 !
162 REAL(KIND=jphook) :: zhook_handle
163 IF (lhook) CALL dr_hook('TRIDIAG_THERMO',0,zhook_handle)
164 ikt=d%NKT
165 iktb=d%NKTB
166 ikte=d%NKTE
167 ikb=d%NKB
168 ike=d%NKE
169 ika=d%NKA
170 iku=d%NKU
171 ikl=d%NKL
172 iijb=d%NIJB
173 iije=d%NIJE
174 !
175 CALL mzm_phy(d,prhodj,zmzm_rhodj)
176 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
177 zrhodj_dfddtdz_o_dz2(iijb:iije,1:ikt) = zmzm_rhodj(iijb:iije,1:ikt)*pdfddtdz(iijb:iije,1:ikt) &
178  /pdzz(iijb:iije,1:ikt)**2
179 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
180 !
181 za=0.
182 zb=0.
183 zc=0.
184 zy=0.
185 !
186 !
187 !* 2. COMPUTE THE RIGHT HAND SIDE
188 ! ---------------------------
189 !
190 !$mnh_expand_array(JIJ=IIJB:IIJE)
191 zy(iijb:iije,ikb) = prhodj(iijb:iije,ikb)*pvarm(iijb:iije,ikb)/ptstep &
192  - zmzm_rhodj(iijb:iije,ikb+ikl) * pf(iijb:iije,ikb+ikl)/pdzz(iijb:iije,ikb+ikl) &
193  + zmzm_rhodj(iijb:iije,ikb ) * pf(iijb:iije,ikb )/pdzz(iijb:iije,ikb ) &
194  + zrhodj_dfddtdz_o_dz2(iijb:iije,ikb+ikl) * pimpl * pvarm(iijb:iije,ikb+ikl) &
195  - zrhodj_dfddtdz_o_dz2(iijb:iije,ikb+ikl) * pimpl * pvarm(iijb:iije,ikb )
196 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
197 !
198 DO jk=iktb+1,ikte-1
199  !$mnh_expand_array(JIJ=IIJB:IIJE)
200  zy(iijb:iije,jk) = prhodj(iijb:iije,jk)*pvarm(iijb:iije,jk)/ptstep &
201  - zmzm_rhodj(iijb:iije,jk+ikl) * pf(iijb:iije,jk+ikl)/pdzz(iijb:iije,jk+ikl) &
202  + zmzm_rhodj(iijb:iije,jk ) * pf(iijb:iije,jk )/pdzz(iijb:iije,jk ) &
203  + zrhodj_dfddtdz_o_dz2(iijb:iije,jk+ikl) * pimpl * pvarm(iijb:iije,jk+ikl) &
204  - zrhodj_dfddtdz_o_dz2(iijb:iije,jk+ikl) * pimpl * pvarm(iijb:iije,jk ) &
205  - zrhodj_dfddtdz_o_dz2(iijb:iije,jk ) * pimpl * pvarm(iijb:iije,jk ) &
206  + zrhodj_dfddtdz_o_dz2(iijb:iije,jk ) * pimpl * pvarm(iijb:iije,jk-ikl)
207  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
208 END DO
209 !
210 !$mnh_expand_array(JIJ=IIJB:IIJE)
211 zy(iijb:iije,ike) = prhodj(iijb:iije,ike)*pvarm(iijb:iije,ike)/ptstep &
212  - zmzm_rhodj(iijb:iije,ike+ikl) * pf(iijb:iije,ike+ikl)/pdzz(iijb:iije,ike+ikl) &
213  + zmzm_rhodj(iijb:iije,ike ) * pf(iijb:iije,ike )/pdzz(iijb:iije,ike ) &
214  - zrhodj_dfddtdz_o_dz2(iijb:iije,ike ) * pimpl * pvarm(iijb:iije,ike ) &
215  + zrhodj_dfddtdz_o_dz2(iijb:iije,ike ) * pimpl * pvarm(iijb:iije,ike-ikl)
216 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
217 !
218 !
219 !* 3. INVERSION OF THE TRIDIAGONAL SYSTEM
220 ! -----------------------------------
221 !
222 IF ( pimpl > 1.e-10 ) THEN
223 !
224 !* 3.1 arrays A, B, C
225 ! --------------
226 !
227  !$mnh_expand_array(JIJ=IIJB:IIJE)
228  zb(iijb:iije,ikb) = prhodj(iijb:iije,ikb)/ptstep &
229  - zrhodj_dfddtdz_o_dz2(iijb:iije,ikb+ikl) * pimpl
230  zc(iijb:iije,ikb) = zrhodj_dfddtdz_o_dz2(iijb:iije,ikb+ikl) * pimpl
231  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
232 !
233  DO jk=iktb+1,ikte-1
234  !$mnh_expand_array(JIJ=IIJB:IIJE)
235  za(iijb:iije,jk) = zrhodj_dfddtdz_o_dz2(iijb:iije,jk) * pimpl
236  zb(iijb:iije,jk) = prhodj(iijb:iije,jk)/ptstep &
237  - zrhodj_dfddtdz_o_dz2(iijb:iije,jk+ikl) * pimpl &
238  - zrhodj_dfddtdz_o_dz2(iijb:iije,jk) * pimpl
239  zc(iijb:iije,jk) = zrhodj_dfddtdz_o_dz2(iijb:iije,jk+ikl) * pimpl
240  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
241  END DO
242 !
243  !$mnh_expand_array(JIJ=IIJB:IIJE)
244  za(iijb:iije,ike) = zrhodj_dfddtdz_o_dz2(iijb:iije,ike ) * pimpl
245  zb(iijb:iije,ike) = prhodj(iijb:iije,ike)/ptstep &
246  - zrhodj_dfddtdz_o_dz2(iijb:iije,ike ) * pimpl
247 !
248 !* 3.2 going up
249 ! --------
250 !
251  zbet(iijb:iije) = zb(iijb:iije,ikb) ! bet = b(ikb)
252  pvarp(iijb:iije,ikb) = zy(iijb:iije,ikb) / zbet(iijb:iije)
253  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
254 
255  !
256  DO jk = ikb+ikl,ike-ikl,ikl
257  !$mnh_expand_array(JIJ=IIJB:IIJE)
258  zgam(iijb:iije,jk) = zc(iijb:iije,jk-ikl) / zbet(iijb:iije)
259  ! gam(k) = c(k-1) / bet
260  zbet(iijb:iije) = zb(iijb:iije,jk) - za(iijb:iije,jk) * zgam(iijb:iije,jk)
261  ! bet = b(k) - a(k)* gam(k)
262  pvarp(iijb:iije,jk)= ( zy(iijb:iije,jk) - za(iijb:iije,jk) * pvarp(iijb:iije,jk-ikl) ) &
263  / zbet(iijb:iije)
264  ! res(k) = (y(k) -a(k)*res(k-1))/ bet
265  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
266  END DO
267  ! special treatment for the last level
268  !$mnh_expand_array(JIJ=IIJB:IIJE)
269  zgam(iijb:iije,ike) = zc(iijb:iije,ike-ikl) / zbet(iijb:iije)
270  ! gam(k) = c(k-1) / bet
271  zbet(iijb:iije) = zb(iijb:iije,ike) - za(iijb:iije,ike) * zgam(iijb:iije,ike)
272  ! bet = b(k) - a(k)* gam(k)
273  pvarp(iijb:iije,ike)= ( zy(iijb:iije,ike) - za(iijb:iije,ike) * pvarp(iijb:iije,ike-ikl) ) &
274  / zbet(iijb:iije)
275  ! res(k) = (y(k) -a(k)*res(k-1))/ bet
276  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
277 !
278 !* 3.3 going down
279 ! ----------
280 !
281  DO jk = ike-ikl,ikb,-1*ikl
282  !$mnh_expand_array(JIJ=IIJB:IIJE)
283  pvarp(iijb:iije,jk) = pvarp(iijb:iije,jk) - zgam(iijb:iije,jk+ikl) * pvarp(iijb:iije,jk+ikl)
284  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
285  END DO
286 !
287 ELSE
288 !
289  DO jk=iktb,ikte
290  !$mnh_expand_array(JIJ=IIJB:IIJE)
291  pvarp(iijb:iije,jk) = zy(iijb:iije,jk) * ptstep / prhodj(iijb:iije,jk)
292  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
293  END DO
294 !
295 END IF
296 !
297 !
298 !* 4. FILL THE UPPER AND LOWER EXTERNAL VALUES
299 ! ----------------------------------------
300 !
301 !$mnh_expand_array(JIJ=IIJB:IIJE)
302 pvarp(iijb:iije,ika)=pvarp(iijb:iije,ikb)
303 pvarp(iijb:iije,iku)=pvarp(iijb:iije,ike)
304 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
305 !
306 !-------------------------------------------------------------------------------
307 !
308 IF (lhook) CALL dr_hook('TRIDIAG_THERMO',1,zhook_handle)
309 END SUBROUTINE tridiag_thermo
