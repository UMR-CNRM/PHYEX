14SUBROUTINE lima_update_nsv(LDINIT, KMI, KSV, CDCLOUD, LDUPDATE)
15!!*** *LIMA_UPDATE_NSV* - update modd_nsv values realtive to LIMA
16!!
17!!* PURPOSE
18!! -------
19!! The modd_nsv values relative to the LIMA scheme are initialised (if LDINIT is .TRUE)
20!! according to the micromisics scheme used.
21!! If LDUPDATE is .TRUE., the scalar values of modd_nsv module receive the values
22!! assigned for the KMI model.
23!!
24!!* METHOD
25!! ------
26!!
27!!* EXTERNAL
28!! --------
29!!
30!!* IMPLICIT ARGUMENTS
31!! ------------------
32!! MODD_NSV, MODD_PARAM_LIMA
33!!
34!!* REFERENCE
35!! ---------
36!!
37!! AUTHOR
38!! ------
39!! S. Riette
40!!
41!! MODIFICATIONS
42!! -------------
43!! Original April 2023
44!-------------------------------------------------------------------------------
45!
46!* 0. DECLARATIONS
47! ---------------
48!
49USE modd_nsv, ONLY: nsv_lima_beg_a, nsv_lima_end_a, nsv_lima_a, &
50 & nsv_lima_nc_a, nsv_lima_nr_a, nsv_lima_ni_a, nsv_lima_ns_a, nsv_lima_ng_a, nsv_lima_nh_a, &
51 & nsv_lima_ccn_free_a, nsv_lima_ccn_acti_a, nsv_lima_scavmass_a, &
52 & nsv_lima_ifn_free_a, nsv_lima_ifn_nucl_a, nsv_lima_imm_nucl_a, &
53 & nsv_lima_hom_haze_a, nsv_lima_spro_a, &
54 & nsv_lima_beg, nsv_lima_end, nsv_lima, &
55 & nsv_lima_nc, nsv_lima_nr, nsv_lima_ni, nsv_lima_ns, nsv_lima_ng, nsv_lima_nh, &
56 & nsv_lima_ccn_free, nsv_lima_ccn_acti, nsv_lima_scavmass, &
57 & nsv_lima_ifn_free, nsv_lima_ifn_nucl, nsv_lima_imm_nucl, &
58 & nsv_lima_hom_haze, nsv_lima_spro
59USE modd_param_lima, ONLY: nmod_ccn, lscav, laero_mass, &
60 nmod_ifn, nmod_imm, lhhoni, &
61 lspro, &
62 nmom_c, nmom_r, nmom_i, nmom_s, nmom_g, nmom_h
63!
64!* 0.1. Declaration of arguments
65! ------------------------
66!
67IMPLICIT NONE
68LOGICAL, INTENT(IN) :: LDINIT
69INTEGER, INTENT(IN) :: KMI
70INTEGER, INTENT(INOUT) :: KSV
72CHARACTER(LEN=4), INTENT(IN) :: CDCLOUD
73LOGICAL, INTENT(IN) :: LDUPDATE
74!
75!* 1. INITIALISATION
76! -----------------
77!
78IF(ldinit) THEN
79 IF(cdcloud=='LIMA') THEN
80 ksv = ksv+1
81 nsv_lima_beg_a(kmi) = ksv
82 ! Nc
83 IF (nmom_c.GE.2) THEN
84 nsv_lima_nc_a(kmi) = ksv
85 ksv = ksv+1
86 END IF
87 ! Nr
88 IF (nmom_r.GE.2) THEN
89 nsv_lima_nr_a(kmi) = ksv
90 ksv = ksv+1
91 END IF
92 ! CCN
93 IF (nmod_ccn .GT. 0) THEN
94 nsv_lima_ccn_free_a(kmi) = ksv
95 ksv = ksv + nmod_ccn
96 nsv_lima_ccn_acti_a(kmi) = ksv
97 ksv = ksv + nmod_ccn
98 END IF
99 ! Scavenging
100 IF (lscav .AND. laero_mass) THEN
101 nsv_lima_scavmass_a(kmi) = ksv
102 ksv = ksv+1
103 END IF
104 ! Ni
105 IF (nmom_i.GE.2) THEN
106 nsv_lima_ni_a(kmi) = ksv
107 ksv = ksv+1
108 END IF
109 ! Ns
110 IF (nmom_s.GE.2) THEN
111 nsv_lima_ns_a(kmi) = ksv
112 ksv = ksv+1
113 END IF
114 ! Ng
115 IF (nmom_g.GE.2) THEN
116 nsv_lima_ng_a(kmi) = ksv
117 ksv = ksv+1
118 END IF
119 ! Nh
120 IF (nmom_h.GE.2) THEN
121 nsv_lima_nh_a(kmi) = ksv
122 ksv = ksv+1
123 END IF
124 ! IFN
125 IF (nmod_ifn .GT. 0) THEN
126 nsv_lima_ifn_free_a(kmi) = ksv
127 ksv = ksv + nmod_ifn
128 nsv_lima_ifn_nucl_a(kmi) = ksv
129 ksv = ksv + nmod_ifn
130 END IF
131 ! IMM
132 IF (nmod_imm .GT. 0) THEN
133 nsv_lima_imm_nucl_a(kmi) = ksv
134 ksv = ksv + max(1,nmod_imm)
135 END IF
136 ! Homogeneous freezing of CCN
137 IF (lhhoni) THEN
138 nsv_lima_hom_haze_a(kmi) = ksv
139 ksv = ksv + 1
140 END IF
141 ! Supersaturation
142 IF (lspro) THEN
143 nsv_lima_spro_a(kmi) = ksv
144 ksv = ksv + 1
145 END IF
146 !
147 ! End and total variables
148 !
149 ksv = ksv - 1
150 nsv_lima_end_a(kmi) = ksv
151 nsv_lima_a(kmi) = nsv_lima_end_a(kmi) - nsv_lima_beg_a(kmi) + 1
152 ELSE
153 nsv_lima_a(kmi) = 0
154 !
155 ! force First index to be superior to last index
156 ! in order to create a null section
157 !
158 nsv_lima_beg_a(kmi) = 1
159 nsv_lima_end_a(kmi) = 0
160 ENDIF
161ENDIF
162!
163!* 2. UPDATE
164! ---------
165!
166IF(ldupdate) THEN
167 nsv_lima = nsv_lima_a(kmi)
168 nsv_lima_beg = nsv_lima_beg_a(kmi)
169 nsv_lima_end = nsv_lima_end_a(kmi)
170 nsv_lima_nc = nsv_lima_nc_a(kmi)
171 nsv_lima_nr = nsv_lima_nr_a(kmi)
172 nsv_lima_ccn_free = nsv_lima_ccn_free_a(kmi)
173 nsv_lima_ccn_acti = nsv_lima_ccn_acti_a(kmi)
174 nsv_lima_scavmass = nsv_lima_scavmass_a(kmi)
175 nsv_lima_ni = nsv_lima_ni_a(kmi)
176 nsv_lima_ns = nsv_lima_ns_a(kmi)
177 nsv_lima_ng = nsv_lima_ng_a(kmi)
178 nsv_lima_nh = nsv_lima_nh_a(kmi)
179 nsv_lima_ifn_free = nsv_lima_ifn_free_a(kmi)
180 nsv_lima_ifn_nucl = nsv_lima_ifn_nucl_a(kmi)
181 nsv_lima_imm_nucl = nsv_lima_imm_nucl_a(kmi)
182 nsv_lima_hom_haze = nsv_lima_hom_haze_a(kmi)
183 nsv_lima_spro = nsv_lima_spro_a(kmi)
184ENDIF
185!
186END SUBROUTINE lima_update_nsv
187!
188END MODULE mode_lima_update_nsv
