213 ! 1.a Collection of rc and ri in the dry mode
214 ! --------------------------------------------
215 !
216 WHERE( prgt(:)>xrtmin(6) .AND. ldcompute(:) )
217  zzw(:) = pcgt(:) * plbdg(:)**(-xdg-2.0) * prhodref(:)**(1-xcexvt)
218  zzw1(:) = xfcdryg * prct(:) * zzw(:) ! RCDRYG - rc collected by graupel in dry mode
219  zzw2(:) = xfidryg * exp( xcolexig*(pt(:)-xtt) ) * prit(:) * zzw(:) ! RIDRYG - ri collected by graupel in dry mode
220 END WHERE
221 !
222 !* 1.b Collection of rs in the dry mode
223 ! ------------------------------------
224 !
225 gdry(:) = prst(:)>xrtmin(5) .AND. pcst(:)>xctmin(5) .AND. prgt(:)>xrtmin(6) .AND. pcgt(:)>xctmin(6) .AND. ldcompute(:)
226 !
227 WHERE( gdry )
228 !
229 !* Select the (ZLBDAG,ZLBDAS) couplet
230 !
231  zvec1(:) = plbdg(:)
232  zvec2(:) = plbds(:)
233 !
234 !* find the next lower indice for the ZLBDAG and for the ZLBDAS
235 ! in the geometrical set of (Lbda_g,Lbda_s) couplet use to
236 ! tabulate the SDRYG-kernel
237 !
238  zvec1(:) = max( 1.0001, min( real(ndrylbdag)-0.0001, &
239  xdryintp1g * log( zvec1(:) ) + xdryintp2g ) )
240  ivec1(:) = int( zvec1(:) )
241  zvec1(:) = zvec1(:) - real( ivec1(:) )
242 !
243  zvec2(:) = max( 1.0001, min( real(ndrylbdas)-0.0001, &
244  xdryintp1s * log( zvec2(:) ) + xdryintp2s ) )
245  ivec2(:) = int( zvec2(:) )
246  zvec2(:) = zvec2(:) - real( ivec2(:) )
247 !
248 !* perform the bilinear interpolation of the normalized
249 ! SDRYG-kernel
250  !
251  z1(:) = get_xker_sdryg(ivec1(:)+1,ivec2(:)+1)
252  z2(:) = get_xker_sdryg(ivec1(:)+1,ivec2(:) )
253  z3(:) = get_xker_sdryg(ivec1(:) ,ivec2(:)+1)
254  z4(:) = get_xker_sdryg(ivec1(:) ,ivec2(:) )
255  zvec3(:) = ( z1(:)* zvec2(:) &
256  - z2(:)*(zvec2(:) - 1.0) ) &
257  * zvec1(:) &
258  - ( z3(:)* zvec2(:) &
259  - z4(:)*(zvec2(:) - 1.0) ) &
260  * (zvec1(:) - 1.0)
261  zzw(:) = zvec3(:)
262  zzw3(:) = xfsdryg * zzw(:) * exp( xcolexsg*(pt(:)-xtt) ) & ! RSDRYG - rs collected by graupel in dry mode
263  * prst(:) * pcgt(:) &
264  * prhodref(:)**(1-xcexvt) &
265  *( xlbsdryg1/( plbdg(:)**2 ) + &
266  xlbsdryg2/( plbdg(:) * plbds(:) ) + &
267  xlbsdryg3/( plbds(:)**2) )
268 !
269  z1(:) = get_xker_n_sdryg(ivec1(:)+1,ivec2(:)+1)
270  z2(:) = get_xker_n_sdryg(ivec1(:)+1,ivec2(:) )
271  z3(:) = get_xker_n_sdryg(ivec1(:) ,ivec2(:)+1)
272  z4(:) = get_xker_n_sdryg(ivec1(:) ,ivec2(:) )
273  zvec3(:) = ( z1(:)* zvec2(:) &
274  - z2(:)*(zvec2(:) - 1.0) ) &
275  * zvec1(:) &
276  - ( z3(:)* zvec2(:) &
277  - z4(:)*(zvec2(:) - 1.0) ) &
278  * (zvec1(:) - 1.0)
279  zzw(:) = zvec3(:)
280  zzw3n(:) = xfnsdryg * zzw(:) * exp( xcolexsg*(pt(:)-xtt) ) & ! NSDRYG - Ns collected by graupel in dry mode
281  * pcst(:) * pcgt(:) &
282  * prhodref(:)**(1-xcexvt) &
283  *( xlbnsdryg1/( plbdg(:)**2 ) + &
284  xlbnsdryg2/( plbdg(:) * plbds(:) ) + &
285  xlbnsdryg3/( plbds(:)**2) )
286 END WHERE
287 !
288 !* 1.c Collection of rr in the dry mode
289 ! -------------------------------------
290 !
291 gdry(:) = prrt(:)>xrtmin(3) .AND. pcrt(:)>xctmin(3) .AND. prgt(:)>xrtmin(6) .AND. pcgt(:)>xctmin(6) .AND. ldcompute(:)
292 !
293 WHERE( gdry )
294 !
295 !* Select the (ZLBDAG,ZLBDAR) couplet
296 !
297  zvec1(:) = plbdg(:)
298  zvec2(:) = plbdr(:)
299 !
300 !* Find the next lower indice for the ZLBDAG and for the ZLBDAR
301 ! in the geometrical set of (Lbda_g,Lbda_r) couplet use to
302 ! tabulate the RDRYG-kernel
303 !
304  zvec1(:) = max( 1.0001, min( real(ndrylbdag)-0.0001, &
305  xdryintp1g * log( zvec1(:) ) + xdryintp2g ) )
306  ivec1(:) = int( zvec1(:) )
307  zvec1(:) = zvec1(:) - real( ivec1(:) )
308 !
309  zvec2(:) = max( 1.0001, min( real(ndrylbdar)-0.0001, &
310  xdryintp1r * log( zvec2(:) ) + xdryintp2r ) )
311  ivec2(:) = int( zvec2(:) )
312  zvec2(:) = zvec2(:) - real( ivec2(:) )
313 !
314 !* Perform the bilinear interpolation of the normalized
315 ! RDRYG-kernel
316 !
317  z1(:) = get_xker_rdryg(ivec1(:)+1,ivec2(:)+1)
318  z2(:) = get_xker_rdryg(ivec1(:)+1,ivec2(:) )
319  z3(:) = get_xker_rdryg(ivec1(:) ,ivec2(:)+1)
320  z4(:) = get_xker_rdryg(ivec1(:) ,ivec2(:) )
321  zvec3(:) = ( z1(:)* zvec2(:) &
322  - z2(:)*(zvec2(:) - 1.0) ) &
323  * zvec1(:) &
324  - ( z3(:)* zvec2(:) &
325  - z4(:)*(zvec2(:) - 1.0) ) &
326  * (zvec1(:) - 1.0)
327  zzw(:) = zvec3(:)
328  zzw4(:) = xfrdryg * zzw(:) & ! RRDRYG
329  * prrt(:) * pcgt(:) &
330  * prhodref(:)**(1-xcexvt) &
331  *( xlbrdryg1/( plbdg(:)**2 ) + &
332  xlbrdryg2/( plbdg(:) * plbdr(:) ) + &
333  xlbrdryg3/( plbdr(:)**2) )
334 !
335  z1(:) = get_xker_n_rdryg(ivec1(:)+1,ivec2(:)+1)
336  z2(:) = get_xker_n_rdryg(ivec1(:)+1,ivec2(:) )
337  z3(:) = get_xker_n_rdryg(ivec1(:) ,ivec2(:)+1)
338  z4(:) = get_xker_n_rdryg(ivec1(:) ,ivec2(:) )
339  zvec3(:) = ( z1(:)* zvec2(:) &
340  - z2(:)*(zvec2(:) - 1.0) ) &
341  * zvec1(:) &
342  - ( z3(:)* zvec2(:) &
343  - z4(:)*(zvec2(:) - 1.0) ) &
344  * (zvec1(:) - 1.0)
345  zzw(:) = zvec3(:)
346  zzw4n(:) = xfnrdryg * zzw(:) & ! NRDRYG
347  * pcrt(:) * pcgt(:) &
348  * prhodref(:)**(1-xcexvt) &
349  *( xlbnrdryg1/( plbdg(:)**2 ) + &
350  xlbnrdryg2/( plbdg(:) * plbdr(:) ) + &
351  xlbnrdryg3/( plbdr(:)**2) )
352 
353 END WHERE
354 !
355 ! 1.d Total collection in the dry mode
356 ! ------------------------------------
357 !
358 zrdryg(:) = zzw1(:) + zzw2(:) + zzw3(:) + zzw4(:)
359 !
360 ! 1.e Collection in the wet mode
361 ! ------------------------------
362 !
363 zzw(:) = 0.0
364 WHERE( prgt(:)>xrtmin(6) .AND. pcgt(:)>xctmin(6) .AND. ldcompute(:) )
365  zzw5(:) = zzw2(:) / (xcolig*exp(xcolexig*(pt(:)-xtt)) ) ! RIWETG
366  zzw6(:) = zzw3(:) / (xcolsg*exp(xcolexsg*(pt(:)-xtt)) ) ! RSWETG
367  zzw6n(:)= zzw3n(:)/ (xcolsg*exp(xcolexsg*(pt(:)-xtt)) ) ! NSWETG
368 !
369  zzw(:) = prvt(:)*ppres(:)/((xmv/xmd)+prvt(:)) ! Vapor pressure
370  zzw(:) = pka(:)*(xtt-pt(:)) + &
371  ( pdv(:)*(xlvtt + ( xcpv - xcl ) * ( pt(:) - xtt )) &
372  *(xestt-zzw(:))/(xrv*pt(:)) )
373 !
374 ! Total mass gained by graupel in wet mode
375  zrwetg(:) = max( 0.0, &
376  ( zzw(:) * pcgt(:) * ( x0depg* plbdg(:)**xex0depg + &
377  x1depg*pcj(:)*plbdg(:)**xex1depg ) + &
378  ( zzw5(:)+zzw6(:) ) * ( xlmtt + (xci-xcl)*(xtt-pt(:)) ) ) &
379  / (xlmtt-xcl*(xtt-pt(:))) )
380  !We must agregate, at least, the cold species
381  zrwetg(:)=max(zrwetg(:), zzw5(:)+zzw6(:))
382 END WHERE
383 !
384 ! 1.f Wet mode and partial conversion to hail
385 ! -------------------------------------------
386 !
387 zzw(:) = 0.0
388 nhail = 0.
389 IF (nmom_h.GE.1) nhail = 1.
390 WHERE( ldcompute(:) .AND. prgt(:)>xrtmin(6) .AND. pcgt(:)>xctmin(6) .AND. pt(:)<xtt .AND. &
391  (zrdryg(:)-zzw2(:)-zzw3(:))>=(zrwetg(:)-zzw5(:)-zzw6(:)) .AND. zrwetg(:)-zzw5(:)-zzw6(:)>0.0 )
392 !
393 ! Mass of rain and cloud droplets frozen by graupel in wet mode : RCWETG + RRWETG = RWETG - RIWETG - RSWETG
394  zzw7(:) = zrwetg(:) - zzw5(:) - zzw6(:)
395 !
396 ! assume a linear percent of conversion of graupel into hail
397 ! ZZW = percentage of graupel transformed
398 !
399  zzw(:) = zrdryg(:)*nhail/(zrwetg(:)+zrdryg(:))
400 !
401  p_rc_wetg(:) = - zzw1(:)
402  p_cc_wetg(:) = p_rc_wetg(:) * pcct(:)/max(prct(:),xrtmin(2))
403  p_rr_wetg(:) = - zzw7(:) + zzw1(:)
404  p_cr_wetg(:) = p_rr_wetg(:) * pcrt(:)/max(prrt(:),xrtmin(3))
405  p_ri_wetg(:) = - zzw5(:)
406  p_ci_wetg(:) = p_ri_wetg(:) * pcit(:)/max(prit(:),xrtmin(4))
407  p_rs_wetg(:) = - zzw6(:)
408  p_cs_wetg(:) = - zzw6n(:)
409  p_rg_wetg(:) = - prgt(:)/ptstep * zzw(:) + zrwetg(:) * (1.-zzw(:))
410  p_cg_wetg(:) = - pcgt(:)/ptstep * zzw(:)
411  p_rh_wetg(:) = prgt(:)/ptstep * zzw(:) + zrwetg(:) * zzw(:)
412  !
413  p_th_wetg(:) = zzw7(:) * (plsfact(:)-plvfact(:))
414 END WHERE
415 !
416 ! 1.g Dry mode
417 ! ------------

428  p_rs_dryg(:) = - zzw3(:)
429  p_cs_dryg(:) = - zzw3n(:)
430  p_rg_dryg(:) = zrdryg(:)

433 END WHERE
434 !
435 !
436 !* 2. Hallett-Mossop process (HMG)
437 ! --------------------------------
438 !
439 ! BVIE test ZRDRYG<ZZW ?????????????????????????

465  END WHERE
466 END WHERE

470 ! -------------------
471 !
472 zzx(:) = 0.0

475  zzx(:) = pka(:)*(xtt-pt(:)) + &
476  ( pdv(:)*(xlvtt + ( xcpv - xcl ) * ( pt(:) - xtt )) &
477  *(xestt-zzx(:))/(xrv*pt(:)) )
478 !
479 ! compute RGMLTR
480 !
481  zzx(:) = max( 0.0,( -zzx(:) * pcgt(:) * &
482  ( x0depg* plbdg(:)**xex0depg + &
483  x1depg*pcj(:)*plbdg(:)**xex1depg ) - &
484  ( zzw1(:)+zzw4(:) ) * ( xcl*(xtt-pt(:))) ) &
485  / xlmtt )
486  p_rr_gmlt(:) = zzx(:)
487  p_cr_gmlt(:) = zzx(:) * 5.0e6 ! obtained after averaging, Dshed=1mm and 500 microns
488  p_cg_gmlt(:) = - zzx(:) * pcgt(:)/prgt(:)
489  !
490  p_th_gmlt(:) = - p_rr_gmlt(:) * (plsfact(:)-plvfact(:))
491 END WHERE
492 !
493 !
494 !
495 !
496 pa_rc(:) = pa_rc(:) + p_rc_wetg(:) + p_rc_dryg(:)
497 pa_cc(:) = pa_cc(:) + p_cc_wetg(:) + p_cc_dryg(:)
