63 zwlbdr3(:) = 1.e30

65 WHERE ( prrt(:)>xrtmin(3) .AND. pcrt(:)>xctmin(3) .AND. ldcompute(:) )
66  zwlbdr3(:) = xlbr * pcrt(:) / prrt(:)
67  zwlbdr(:) = zwlbdr3(:)**xlbexr
68 END WHERE
69 WHERE (zwlbdr(:)<(xaccr1/xsponbud1) .AND. ldcompute(:))
70  p_cr_brku(:) = pcrt(:)*( max((1.+xsponcoef2*(xaccr1/zwlbdr(:)-xsponbud1)**2),&
71  (xaccr1/zwlbdr(:)/xsponbud3)**3) -1. )
72 END WHERE
73 !
74 pb_cr(:) = pb_cr(:) + p_cr_brku(:)
