55 & irv, & ! Water vapor
56 & irc, & ! Cloud water
57 & irr, & ! Rain water

74TYPE(cst_t), INTENT(IN) :: CST
75TYPE(param_ice_t), INTENT(IN) :: PARAMI
76TYPE(rain_ice_param_t), INTENT(IN) :: ICEP
77TYPE(rain_ice_descr_t), INTENT(IN) :: ICED
78TYPE(tbudgetconf_t), INTENT(IN) :: BUCONF
79LOGICAL, INTENT(IN) :: LDSIGMA_RC
80LOGICAL, INTENT(IN) :: LDAUCV_ADJU
81LOGICAL, INTENT(IN) :: LDEXT_TEND
82INTEGER, INTENT(IN) :: KPROMA ! cache-blocking factor for microphysic loop
83INTEGER, INTENT(IN) :: KMICRO ! Case r_x>0 locations
84LOGICAL, DIMENSION(KPROMA), INTENT(IN) :: LDMICRO
85REAL, INTENT(IN) :: PTSTEP ! Double Time step (single if cold start)
86INTEGER, INTENT(IN) :: KRR ! Number of moist variable
87LOGICAL, INTENT(IN) :: OSAVE_MICRO ! if true, save the microphysical tendencies
88LOGICAL, INTENT(IN) :: OELEC ! if true, cloud electricity is activated
89!
90REAL, DIMENSION(KPROMA), INTENT(IN) :: PEXN ! Exner function
91REAL, DIMENSION(KPROMA), INTENT(IN) :: PRHODREF! Reference density
92INTEGER, DIMENSION(KPROMA), INTENT(IN) :: K1,K2 ! Used to replace the COUNT and PACK intrinsics on variables
93REAL, DIMENSION(KPROMA), INTENT(IN) :: PPRES
94REAL, DIMENSION(KPROMA), INTENT(IN) :: PCF ! Cloud fraction
95REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PSIGMA_RC
96REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PCIT
97REAL, DIMENSION(KPROMA,0:7), INTENT(INOUT) :: PVART !Packed variables
98REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PHLC_HRC
99REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PHLC_HCF
100REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PHLI_HRI
101REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PHLI_HCF
102REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT) :: PRAINFR
103REAL, DIMENSION(KPROMA,0:7), INTENT(INOUT) :: PEXTPK !To take into acount external tendencies inside the splitting
104REAL, DIMENSION(KPROMA, IBUNUM-IBUNUM_EXTRA),INTENT(OUT) :: PBU_SUM

107 ! enhancement of IAGGS
108!
109!
110!* 0.2 Declarations of local variables :
111!
112REAL(KIND=jphook) :: zhook_handle
113!
114LOGICAL :: LSOFT ! Must we really compute tendencies or only adjust them to new T variables
115INTEGER :: INB_ITER_MAX ! Maximum number of iterations (with real tendencies computation)
116REAL :: ZTSTEP ! length of sub-timestep in case of time splitting
117REAL :: ZINV_TSTEP ! Inverse ov PTSTEP
118REAL :: ZTIME_THRESHOLD ! Time to reach threshold
119!
120INTEGER :: JL, JV, JJV
121REAL, DIMENSION(KPROMA) :: &
122 & ZTIME, & ! Current integration time (starts with 0 and ends with PTSTEP)
123 & ZMAXTIME, & ! Time on which we can apply the current tendencies
124 & ZTIME_LASTCALL, & ! Integration time when last tendecies call has been done
125 & ZSSI, &
126 & ZZT, & ! Temperature
127 & ZLSFACT, & ! L_s/(Pi*C_ph)
128 & ZLVFACT, & ! L_v/(Pi*C_ph)
129 & ZHLC_LCF, & ! HLCLOUDS : fraction of Low Cloud Fraction in grid
130 ! note that PCF = PHLC_HCF + ZHLC_LCF
131 & zhlc_lrc, & ! HLCLOUDS : LWC that is Low LWC in grid
132 ! note that ZRC = PHLC_HRC + ZHLC_LRC
133 & zhli_lcf, &
134 & zhli_lri
135LOGICAL, DIMENSION(KPROMA) :: LLCOMPUTE ! .TRUE. or points where we must compute tendencies,
136REAL, DIMENSION(SIZE(ICED%XRTMIN)) :: ZRSMIN
137!
138!Output packed tendencies (for budgets only)
139REAL, DIMENSION(KPROMA, IBUNUM) :: ZBU_INST
140!
141!For mixing-ratio-splitting
142LOGICAL :: LLCPZ0RT
143REAL :: ZTIME_THRESHOLD1D(KPROMA) ! Time to reach threshold
144REAL, DIMENSION(KPROMA, KRR) :: Z0RT ! Mixing-ratios at the beginig of the current loop
145!
146REAL, DIMENSION(KPROMA,0:7) :: ZA, ZB
147!
148REAL, DIMENSION(KPROMA, 8) :: ZRS_TEND, ZRG_TEND
149REAL, DIMENSION(KPROMA,10) :: ZRH_TEND
150
151INTEGER, DIMENSION(KPROMA) :: IITER ! Number of iterations done (with real tendencies computation)
152!
153REAL, DIMENSION(KPROMA) :: ZSUM2, ZMAXB
154REAL :: ZDEVIDE, ZX
155!
156LOGICAL :: LL_ANY_ITER
157!
158!-------------------------------------------------------------------------------
159IF (lhook) CALL dr_hook('ICE4_STEPPING', 0, zhook_handle)
160!
161!* 1. GENERALITIES
162! ------------
163!
164zinv_tstep=1./ptstep
165zrsmin = iced%XRTMIN
166!
167IF(buconf%LBU_ENABLE .OR. osave_micro) THEN
168 DO jv=1, ibunum-ibunum_extra
169 pbu_sum(:, jv)=0.
170 ENDDO
171ENDIF
172
173!Maximum number of iterations
174!We only count real iterations (those for which we *compute* tendencies)
175inb_iter_max=parami%NMAXITER_MICRO
176IF(parami%XTSTEP_TS/=0.)THEN
177 inb_iter_max=max(1, int(ptstep/parami%XTSTEP_TS)) !At least the number of iterations needed for the time-splitting
178 ztstep=ptstep/inb_iter_max
179 inb_iter_max=max(parami%NMAXITER_MICRO, inb_iter_max) !For the case XMRSTEP/=0. at the same time
180ENDIF
181
182IF (ldext_tend) THEN
183!$acc kernels
184!$acc loop independent collapse(2)
185 DO jv=0, krr
186 DO jl=1, kmicro
187 pextpk(jl, jv)=pextpk(jl, jv)-pvart(jl, jv)*zinv_tstep
188 ENDDO
189 ENDDO
190!$acc end kernels
191ENDIF
192IF (ldsigma_rc) THEN
193!$acc kernels
194!$acc loop independent
195 DO jl=1, kmicro
196 psigma_rc(jl)=psigma_rc(jl)*2.
197 ENDDO
198!$acc end kernels
199ENDIF
200IF (ldaucv_adju) THEN
201!$acc kernels
202!$acc loop independent
203 DO jl=1, kmicro
204 zhlc_lrc(jl) = pvart(jl, irc) - phlc_hrc(jl)
205 zhli_lri(jl) = pvart(jl, iri) - phli_hri(jl)
206 IF(pvart(jl, irc)>0.) THEN
207 zhlc_lcf(jl) = pcf(jl)- phlc_hcf(jl)
208 ELSE
209 zhlc_lcf(jl)=0.
210 ENDIF
211 IF(pvart(jl, iri)>0.) THEN
212 zhli_lcf(jl) = pcf(jl)- phli_hcf(jl)
213 ELSE
214 zhli_lcf(jl)=0.
215 ENDIF
216 ENDDO
217!$acc end kernels
218ENDIF
219
220!-------------------------------------------------------------------------------
221!
222!*** 4.4 temporal loop
223!
224!
225!$acc kernels
226iiter(1:kmicro)=0
227!$acc loop independent
228DO jl=1, kmicro
229 IF(ldmicro(jl)) THEN
230 ztime(jl)=0. ! Current integration time (all points may have a different integration time)
231 ELSE
232 ztime(jl)=ptstep ! Nothing to do on this point, it has already reached the end of the timestep
233 ENDIF
234ENDDO
235!$acc end kernels
236
237DO WHILE(any(ztime(1:kmicro)<ptstep)) ! Loop to *really* compute tendencies
238
239 IF(parami%XTSTEP_TS/=0.) THEN
240 ! In this case we need to remember the time when tendencies were computed
241 ! because when time has evolved more than a limit, we must re-compute tendencies
242 ztime_lastcall(1:kmicro)=ztime(1:kmicro)
243 ENDIF
244!$acc kernels
245!$acc loop independent
246 DO jl=1, kmicro
247 IF (ztime(jl) < ptstep) THEN
248 llcompute(jl)=.true. ! Computation (.TRUE.) only for points for which integration time has not reached the timestep
249 iiter(jl)=iiter(jl)+1
250 ELSE
251 llcompute(jl)=.false.
252 ENDIF
253 ENDDO
254!$acc end kernels
255 ll_any_iter=any(iiter(1:kmicro) < inb_iter_max)
256 llcpz0rt=.true.
257 lsoft=.false. ! We *really* compute the tendencies
258
259 DO WHILE(any(llcompute(1:kmicro))) ! Loop to adjust tendencies when we cross the 0¬∞C or when a species disappears
260!$acc kernels
261 zsum2(1:kmicro)=pvart(1:kmicro, iri)
262!$acc loop independent collapse(2)
263 DO jv=iri+1,krr
264 DO jl=1, kmicro
265 zsum2(jl)=zsum2(jl)+pvart(jl, jv)
266 ENDDO
267 ENDDO
268!$acc end kernels
269!$acc kernels
270!$acc loop independent
271 DO jl=1, kmicro
272 zdevide=(cst%XCPD + cst%XCPV*pvart(jl, irv) + cst%XCL*(pvart(jl, irc)+pvart(jl, irr)) + cst%XCI*zsum2(jl)) * pexn(jl)
273 zzt(jl) = pvart(jl, ith) * pexn(jl)
274 zlsfact(jl)=(cst%XLSTT+(cst%XCPV-cst%XCI)*(zzt(jl)-cst%XTT)) / zdevide
275 zlvfact(jl)=(cst%XLVTT+(cst%XCPV-cst%XCL)*(zzt(jl)-cst%XTT)) / zdevide
276 ENDDO
277!$acc end kernels
278 !-------------------------------------------------------------------------------
279 !
280 !*** 4.5 Effective tendencies computation
281 !
282 !
283 ! Tendencies are *really* computed when LSOFT==.FALSE. and only adjusted otherwise
284 CALL ice4_tendencies(cst, parami, icep, iced, buconf, &
285 &kproma, kmicro, &
286 &krr, lsoft, llcompute, &
287 &osave_micro, oelec, &
288 &pexn, prhodref, zlvfact, zlsfact, k1, k2, &
289 &ppres, pcf, psigma_rc, &
290 &pcit, &
291 &zzt, pvart, &
292 &platham_iaggs, &
293 &zbu_inst, &
294 &zrs_tend, zrg_tend, zrh_tend, zssi, &
295 &za, zb, &
296 &phlc_hcf, zhlc_lcf, phlc_hrc, zhlc_lrc, &
297 &phli_hcf, zhli_lcf, phli_hri, zhli_lri, prainfr)
298
299 ! External tendencies
300 IF(ldext_tend) THEN
301!$acc kernels
302!$acc loop independent
303 DO jv=0, krr
304 DO jl=1, kmicro
305 za(jl, jv) = za(jl, jv) + pextpk(jl, jv)
306 ENDDO
307 ENDDO
308!$acc end kernels
309 ENDIF
310 !-------------------------------------------------------------------------------
311 !
312 !*** 4.6 Time integration
313 !
314 !
315 ! If we can, we shall use these tendencies until the end of the timestep
316!$acc kernels
317!$acc loop independent
318 DO jl=1, kmicro
319 IF(llcompute(jl)) THEN
320 zmaxtime(jl)=(ptstep-ztime(jl)) ! Remaining time until the end of the timestep
321 ELSE
322 zmaxtime(jl)=0.
323 ENDIF
324 ENDDO
325!$acc end kernels
326 !We need to adjust tendencies when temperature reaches 0
327 IF(parami%LFEEDBACKT) THEN
328!$acc kernels
329!$acc loop independent
330 DO jl=1, kmicro
331 !Is ZB(:, ITH) enough to change temperature sign?
332 zx=cst%XTT/pexn(jl)
333 IF ((pvart(jl, ith) - zx) * (pvart(jl, ith) + zb(jl, ith) - zx) < 0.) THEN
334 zmaxtime(jl)=0.
335 ENDIF
336 !Can ZA(:, ITH) make temperature change of sign?
337 IF (abs(za(jl,ith)) > 1.e-20 ) THEN
338 ztime_threshold=(zx - zb(jl, ith) - pvart(jl, ith))/za(jl, ith)
339 IF (ztime_threshold > 0.) THEN
340 zmaxtime(jl)=min(zmaxtime(jl), ztime_threshold)
341 ENDIF
342 ENDIF
343 ENDDO
344!$acc end kernels
345 ENDIF
346
347 !We need to adjust tendencies when a species disappears
348 !When a species is missing, only the external tendencies can be negative (and we must keep track of it)
349!$acc kernels
350 DO jv=1, krr
351!$acc loop independent
352 DO jl=1, kmicro
353 IF (za(jl, jv) < -1.e-20 .AND. pvart(jl, jv) > zrsmin(jv)) THEN
354 zmaxtime(jl)=min(zmaxtime(jl), -(zb(jl, jv)+pvart(jl, jv))/za(jl, jv))
355 zmaxtime(jl)=max(zmaxtime(jl), cst%XMNH_TINY) !to prevent rounding errors
356 ENDIF
357 ENDDO
358 ENDDO
359!$acc end kernels
360 !We stop when the end of the timestep is reached
361!$acc kernels
362!$acc loop independent
363 DO jl=1, kmicro
364 IF (ztime(jl)+zmaxtime(jl) >= ptstep) THEN
365 llcompute(jl)=.false.
366 ENDIF
367 ENDDO
368!$acc end kernels
369 !We must recompute tendencies when the end of the sub-timestep is reached
370 IF (parami%XTSTEP_TS/=0.) THEN
371!$acc kernels
372!$acc loop independent
373 DO jl=1, kmicro
374 IF ((iiter(jl) < inb_iter_max) .AND. (ztime(jl)+zmaxtime(jl) > ztime_lastcall(jl)+ztstep)) THEN
375 zmaxtime(jl)=ztime_lastcall(jl)-ztime(jl)+ztstep
376 llcompute(jl)=.false.
377 ENDIF
378 ENDDO
379!$acc end kernels
380 ENDIF
381
382 !We must recompute tendencies when the maximum allowed change is reached
383 !When a species is missing, only the external tendencies can be active and we do not want to recompute
384 !the microphysical tendencies when external tendencies are negative (results won't change because species was already missing)
385 IF (parami%XMRSTEP/=0.) THEN
386 IF (ll_any_iter) THEN
387 ! In this case we need to remember the initial mixing ratios used to compute the tendencies
388 ! because when mixing ratio has evolved more than a threshold, we must re-compute tendencies
389 ! Thus, at first iteration (ie when LLCPZ0RT=.TRUE.) we copy PVART into Z0RT
390 DO jv=1,krr
391!$acc kernels
392 IF (llcpz0rt) THEN
393 z0rt(1:kmicro, jv)=pvart(1:kmicro, jv)
394 ENDIF
395!$acc loop independent
396 DO jl=1, kmicro
397 IF (iiter(jl)<inb_iter_max .AND. abs(za(jl,jv))>1.e-20) THEN
398 ztime_threshold1d(jl)=(sign(1., za(jl, jv))*parami%XMRSTEP+ &
399 &z0rt(jl, jv)-pvart(jl, jv)-zb(jl, jv))/za(jl, jv)
400 ELSE
401 ztime_threshold1d(jl)=-1.
402 ENDIF
403 ENDDO
404!$acc loop independent
405 DO jl=1, kmicro
406 IF (ztime_threshold1d(jl)>=0 .AND. ztime_threshold1d(jl)<zmaxtime(jl) .AND. &
407 &(pvart(jl, jv)>zrsmin(jv) .OR. za(jl, jv)>0.)) THEN
408 zmaxtime(jl)=min(zmaxtime(jl), ztime_threshold1d(jl))
409 llcompute(jl)=.false.
410 ENDIF
411 ENDDO
412 IF (jv == 1) THEN
413!$acc loop independent
414 DO jl=1, kmicro
415 zmaxb(jl)=abs(zb(jl, jv))
416 ENDDO
417 ELSE
418!$acc loop independent
419 DO jl=1, kmicro
420 zmaxb(jl)=max(zmaxb(jl), abs(zb(jl, jv)))
421 ENDDO
422 ENDIF
423!$acc end kernels
424 ENDDO
425 llcpz0rt=.false.
426!$acc kernels
427!$acc loop independent
428 DO jl=1, kmicro
429 IF (iiter(jl)<inb_iter_max .AND. zmaxb(jl)>parami%XMRSTEP) THEN
430 zmaxtime(jl)=0.
431 llcompute(jl)=.false.
432 ENDIF
433 ENDDO
434!$acc end kernels
435 ENDIF ! LL_ANY_ITER
436 ENDIF ! XMRSTEP/=0.
437 !-------------------------------------------------------------------------------
438 !
439 !*** 4.7 New values of variables for next iteration
440 !
441 !
442!$acc kernels
443!$acc loop independent collapse(2)
444 DO jv=0, krr
445 DO jl=1, kmicro
446 IF(ldmicro(jl)) THEN
447 pvart(jl, jv)=pvart(jl, jv)+za(jl, jv)*zmaxtime(jl)+zb(jl, jv)
448 ENDIF
449 ENDDO
450 ENDDO
451!$acc end kernels
452!$acc kernels
453!$acc loop independent
454 DO jl=1, kmicro
455 IF (pvart(jl,iri)<=0. .AND. ldmicro(jl)) pcit(jl) = 0.
456 ztime(jl)=ztime(jl)+zmaxtime(jl)
457 ENDDO
458!$acc end kernels
459 !-------------------------------------------------------------------------------
460 !
461 !*** 4.8 Mixing ratio change due to each process
462 !
463 IF(buconf%LBU_ENABLE .OR. osave_micro) THEN
464 !Mixing ratio change due to a tendency
465!$acc kernels
466!$acc loop independent collapse(2)
467 DO jv=1, ibunum-ibunum_mr-ibunum_extra
468 DO jl=1, kmicro
469 pbu_sum(jl, jv) = pbu_sum(jl, jv) + zbu_inst(jl, jv)*zmaxtime(jl)
470 ENDDO
471 ENDDO
472!$acc end kernels
473
474 !Mixing ratio change due to a mixing ratio change
475!$acc kernels
476!$acc loop independent collapse(2)
477 DO jv=ibunum-ibunum_mr-ibunum_extra+1, ibunum-ibunum_extra
478 DO jl=1, kmicro
479 pbu_sum(jl, jv) = pbu_sum(jl, jv) + zbu_inst(jl, jv)
480 ENDDO
481 ENDDO
482!$acc end kernels
483
484 !Extra contribution as a mixing ratio change
485 DO jv=ibunum-ibunum_extra+1, ibunum
486 jjv=ibuextraind(jv)
487!$acc kernels
488!$acc loop independent
489 DO jl=1, kmicro
490 pbu_sum(jl, jjv) = pbu_sum(jl, jjv) + zbu_inst(jl, jv)
491 ENDDO
492!$acc end kernels
493 ENDDO
494 ENDIF
495 !-------------------------------------------------------------------------------
496 !
497 !*** 4.9 Next loop
498 !
499 lsoft=.true. ! We try to adjust tendencies (inner while loop)
500 ENDDO !Iterations on tendency computations (WHILE ANY(LLCOMPUTE))
501ENDDO !Temporal loop
502
503IF(ldext_tend) THEN
504 !Z..T variables contain the external tendency, we substract it
505!$acc kernels
506!$acc loop independent collapse(2)
507 DO jv=0, krr
508 DO jl=1, kmicro
509 IF(ldmicro(jl)) THEN
510 pvart(jl, jv) = pvart(jl, jv) - pextpk(jl, jv) * ptstep
511 ENDIF
512 ENDDO
513 ENDDO
514!$acc end kernels
515ENDIF
516!$acc kernels
517DO jl=1, kmicro
518 prrevav(jl)=zbu_inst(jl, irrevav)
519ENDDO
520!$acc end kernels
521!
522IF (lhook) CALL dr_hook('ICE4_STEPPING', 1, zhook_handle)
523END SUBROUTINE ice4_stepping
524END MODULE mode_ice4_stepping
