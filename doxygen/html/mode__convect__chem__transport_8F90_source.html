17!!**** Compute modified chemical tracer values due to convective event
18!!
19!!
20!! PURPOSE
21!! -------
22!! The purpose of this routine is to determine the final adjusted
23!! environmental values of the chemical tracers
24!! The final convective tendencies can then be evaluated in the main
25!! routine DEEP_CONVECT by (PCH1C-PCH1)/PTIMEC
26!!
27!!
28!!** METHOD
29!! ------
30!! Identical to the computation of the conservative variables in the
31!! main deep convection code
32!!
33!! EXTERNAL
34!! --------
35!!
36!! IMPLICIT ARGUMENTS
37!! ------------------
38!! Module MODD_CST

42!! JCVEXB, JCVEXT ! extra levels on the vertical boundaries

72INTEGER, INTENT(IN) :: KCH ! number of passive tracers

74REAL,DIMENSION(D%NIJT,D%NKT,KCH),INTENT(IN) :: PCH1 ! grid scale tracer concentr.
75REAL,DIMENSION(D%NIJT,D%NKT,KCH),INTENT(OUT):: PCH1C! conv adjusted tracer concntr.
76!
77INTEGER, DIMENSION(D%NIJT), INTENT(IN) :: KDPL ! index for departure level
78INTEGER, DIMENSION(D%NIJT), INTENT(IN) :: KPBL ! index for top of source layer
79INTEGER, DIMENSION(D%NIJT), INTENT(IN) :: KLCL ! index lifting condens. level
80INTEGER, DIMENSION(D%NIJT), INTENT(IN) :: KCTL ! index for cloud top level
81INTEGER, DIMENSION(D%NIJT), INTENT(IN) :: KLFS ! index for level of free sink
82INTEGER, DIMENSION(D%NIJT), INTENT(IN) :: KDBL ! index for downdraft base level
83!
84REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PUMF ! updraft mass flux (kg/s)
85REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PUER ! updraft entrainment (kg/s)
86REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PUDR ! updraft detrainment (kg/s)
87REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDMF ! downdraft mass flux (kg/s)
88REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDER ! downdraft entrainment (kg/s)
89REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDDR ! downdraft detrainment (kg/s)
90!
91REAL, DIMENSION(D%NIJT), INTENT(IN) :: PTIMEC! convection time step
92REAL, DIMENSION(D%NIJT), INTENT(IN) :: PDXDY ! grid area (m^2)
93REAL, DIMENSION(D%NIJT), INTENT(IN) :: PMIXF ! mixed fraction at LFS
94REAL, DIMENSION(D%NIJT,D%NKT),INTENT(IN) :: PLMASS! mass of model layer (kg)
95REAL, DIMENSION(D%NIJT,D%NKT),INTENT(IN) :: PWSUB ! envir. compensating subsidence(Pa/s)
96INTEGER, INTENT(IN) :: KFTSTEPS ! maximum fractional time steps
97!
98!
99!* 0.2 Declarations of local variables :
100!
101INTEGER :: INCH1 ! number of chemical tracers
102INTEGER :: IKB, IKE ! horizontal + vertical loop bounds
103INTEGER :: IKS ! vertical dimension
104INTEGER :: JI ! horizontal loop index
105INTEGER :: JK, JKP ! vertical loop index
106INTEGER :: JN ! chemical tracer loop index
107INTEGER :: JCH
108INTEGER :: JSTEP ! fractional time loop index
109INTEGER :: JKLD, JKLP, JKMIN, JKMAX, JKMAX2 ! loop index for levels
110!
111REAL, DIMENSION(D%NIJT,D%NKT) :: ZOMG ! compensat. subsidence (Pa/s)
112REAL, DIMENSION(D%NIJT,D%NKT,KCH) :: ZUCH1, ZDCH1 ! updraft/downdraft values
113REAL, DIMENSION(D%NIJT) :: ZTIMEC ! fractional convective time step
114REAL, DIMENSION(D%NIJT,D%NKT) :: ZTIMC! 2D work array for ZTIMEC
115REAL, DIMENSION(D%NIJT,D%NKT,KCH) :: ZCH1MFIN, ZCH1MFOUT
116 ! work arrays for environm. compensat. mass
117REAL, DIMENSION(D%NIJT,KCH) :: ZWORK1, ZWORK2, ZWORK3
118!
119!-------------------------------------------------------------------------------
120!
121!* 0.3 Compute loop bounds
122! -------------------
123!
124REAL(KIND=jphook) :: zhook_handle
125IF (lhook) CALL dr_hook('CONVECT_CHEM_TRANSPORT',0,zhook_handle)
126inch1 = kch
127ikb = 1 + cvpext%JCVEXB
128iks = d%NKT
129ike = d%NKT - cvpext%JCVEXT
130jkmax = 0
131jkmin = 999999999
132DO ji=d%NIJB, d%NIJE
133 jkmin = min(jkmin, kdpl(ji))
134 jkmax = max(jkmax, kctl(ji))
135ENDDO
136
137!
138!
139!* 2. Updraft computations
140! --------------------
141!
142zuch1(:,:,:) = 0.
143!
144!* 2.1 Initialization at LCL
145! ----------------------------------
146!
147DO ji = d%NIJB, d%NIJE
148 DO jn = 1, inch1
149 jkld = kdpl(ji)
150 jklp = kpbl(ji)
151 zwork1(ji,jn) = .5 * ( pch1(ji,jkld,jn) + pch1(ji,jklp,jn) )
152 ENDDO
153END DO
154!
155!* 2.2 Final updraft loop
156! ------------------
157!
158DO jk = jkmin, jkmax
159 jkp = jk + 1
160 !
161 DO jn = 1, inch1
162 DO ji = d%NIJB, d%NIJE
163 IF ( kdpl(ji) <= jk .AND. min(klcl(ji), kctl(ji)) > jk ) &
164 zuch1(ji,jk,jn) = zwork1(ji,jn)
165 !
166 IF ( min(klcl(ji), kctl(ji)) - 1 <= jk .AND. kctl(ji) > jk ) THEN
167 zuch1(ji,jkp,jn) = zuch1(ji,jk,jn)
168 !if you have reactive i.e. non-passive tracers
169 ! update their values here and add the corresponding
170 ! sink term in the following equation
171 zuch1(ji,jkp,jn) = ( pumf(ji,jk) * zuch1(ji,jk,jn) + &
172 puer(ji,jkp) * pch1(ji,jk,jn) ) / &
173 ( pumf(ji,jkp) + pudr(ji,jkp) + 1.e-7 )
174 END IF
175 END DO
176 END DO
177 !
178END DO
179!
180!* 3. Downdraft computations
181! ----------------------
182!
183zdch1(:,:,:) = 0.
184!
185!* 3.1 Initialization at the LFS
186! -------------------------
187!
188DO ji=d%NIJB, d%NIJE
189 DO jk=1, inch1
190 zwork1(ji,jk) = pmixf(ji)
191 ENDDO
192ENDDO
193DO ji = d%NIJB, d%NIJE
194 DO jn = 1, inch1
195 jk = klfs(ji)
196 zdch1(ji,jk,jn) = zwork1(ji,jn) * pch1(ji,jk,jn) + &
197 ( 1. - zwork1(ji,jn) ) * zuch1(ji,jk,jn)
198 ENDDO
199END DO
200!
201!* 3.2 Final downdraft loop
202! --------------------
203!
204jkmax2 = 0
205DO ji=d%NIJB, d%NIJE
206 jkmax2 = max(jkmax2, klfs(ji))
207ENDDO
208DO jk = jkmax2, ikb + 1, -1
209jkp = jk - 1
210 DO jn = 1, inch1
211 DO ji = d%NIJB, d%NIJE
212 IF ( jk <= klfs(ji) .AND. jkp >= kdbl(ji) ) THEN
213 zdch1(ji,jkp,jn) = ( zdch1(ji,jk,jn) * pdmf(ji,jk) - &
214 pch1(ji,jk,jn) * pder(ji,jkp) ) / &
215 ( pdmf(ji,jkp) - pddr(ji,jkp) - 1.e-7 )
216 END IF
217 END DO
218 END DO
219END DO
220!
221!
222!* 4. Final closure (environmental) computations
223! ------------------------------------------
224!
225DO jch = 1, kch
226 pch1c(:,ikb:ike,jch) = pch1(:,ikb:ike,jch) ! initialize adjusted envir. values
227ENDDO
228!
229DO jk = ikb, ike
230 DO ji=d%NIJB,d%NIJE
231 zomg(ji,jk) = pwsub(ji,jk) * pdxdy(ji) / cst%XG ! environmental subsidence
232 ENDDO
233END DO
234!
235DO ji=d%NIJB,d%NIJE
236 ztimec(ji) = ptimec(ji) / real( kftsteps ) ! adjust fractional time step
237ENDDO ! to be an integer multiple of PTIMEC
238
239DO ji=d%NIJB,d%NIJE
240 IF(ptimec(ji) < 1.) ztimec(ji) = 0
241ENDDO
242DO ji=d%NIJB,d%NIJE
243 DO jk=1,iks
244 ztimc(ji,jk) = ztimec(ji)
245 ENDDO
246ENDDO
247!
248DO jch = 1, kch
249 zch1mfin(:,1:d%NKT,jch) = 0.
250 zch1mfout(:,1:d%NKT,jch) = 0.
251ENDDO
252!
253DO jstep = 1, kftsteps ! Enter the fractional time step loop
254 DO jk = ikb + 1, jkmax
255 jkp = max( ikb + 1, jk - 1 )
256 DO ji=d%NIJB,d%NIJE
257 DO jn=1,inch1
258 zwork3(ji,jn) = zomg(ji,jk)
259 ENDDO
260 ENDDO
261 zwork1(:,:) = sign( 1., zwork3(:,:) )
262 zwork2(:,:) = 0.5 * ( 1. + zwork1(:,:) )
263 zwork1(:,:) = 0.5 * ( 1. - zwork1(:,:) )
264 zch1mfin(:,jk,:) = - zwork3(:,:) * pch1c(:,jkp,:) * zwork1(:,:)
265 zch1mfout(:,jk,:) = zwork3(:,:) * pch1c(:,jk,:) * zwork2(:,:)
266 zch1mfin(:,jkp,:) = zch1mfin(:,jkp,:) + zch1mfout(:,jk,:) * zwork2(:,:)
267 zch1mfout(:,jkp,:)= zch1mfout(:,jkp,:) + zch1mfin(:,jk,:) * zwork1(:,:)
268 END DO
269 !
270 DO jn = 1, inch1
271 DO jk = ikb + 1, jkmax
272 DO ji=d%NIJB,d%NIJE
273 pch1c(ji,jk,jn) = pch1c(ji,jk,jn) + ztimc(ji,jk) / plmass(ji,jk) * ( &
274 zch1mfin(ji,jk,jn) + pudr(ji,jk) * zuch1(ji,jk,jn) + &
275 pddr(ji,jk) * zdch1(ji,jk,jn) - zch1mfout(ji,jk,jn) - &
276 ( puer(ji,jk) + pder(ji,jk) ) * pch1(ji,jk,jn) )
277 IF(jn < nsv%NSV_LGBEG .OR. jn>nsv%NSV_LGEND-1) THEN
278 pch1c(ji,jk,jn) = max( 0., pch1c(ji,jk,jn) )
279 ELSE
280 ! no tendency for horizontal Lagrangian variables
281 pch1c(ji,jk,jn) = pch1(ji,jk,jn)
282 END IF
283 ENDDO
284 END DO
285 END DO
286 !
287END DO ! Exit the fractional time step loop
288!
289!
290IF (lhook) CALL dr_hook('CONVECT_CHEM_TRANSPORT',1,zhook_handle)
291END SUBROUTINE convect_chem_transport
