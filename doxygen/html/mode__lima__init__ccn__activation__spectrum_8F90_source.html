102 CASE('Na2SO4') ! Sodium sulfate
103  xkappa = 0.80
104 CASE('NH43HSO42') ! Letovicite (rare ammonium sulfate mineral)
105  xkappa = 0.65
106 CASE('SOA') ! Secondary organic aerosol (alpha-pinene, beta-pinene)

116 ! Initialize supersaturation values (in %)
117 !
118 DO ii=1, SIZE(xs)
119  xs(ii)=exp( log(10.**(-3.)) + real(ii) / real(SIZE(xs)) * (log(10.**2.)-log(10.**(-3.))) )
120 END DO
121 
122 DO ij=1, SIZE(xt)
123 !
124 !* 2. Compute Nccn(s) for several supersaturation values

128 !
129  xddry = xd
130  xmin = xd
131  xmax = xd*10.
132  xprec = xd/100.

137 !
138 ! Get the XDCRIT values for XS using the approx.
139 ! ln(100*(Sw))~Dcrit^(-3/2) where Sw is in % (Sw=1 for a 1% supersaturation)
140 !
141  xw = xddry * xscrit**0.66 ! "a" factor in Ddry_crit = a*S**-0.66
142  xdcrit(:) = xw * xs(:)**(-0.66) ! Ddry_crit for each value of S
143 !
144 ! Compute Nccn(S) as the incomplete integral of n(D) from 0 to Ddry_crit(S)
145 !
146  DO ii=1, SIZE(xs)
147  xnccn(ii) = 1- ( 0.5 + sign(0.5,xdcrit(ii)-xd) * gamma_inc(0.5,(log(xdcrit(ii)/xd)/sqrt(2.)/log(xsigma))**2) )
148  END DO

161  xmu = params(2)
162  xbeta = params(3)
163 !
164 END DO ! loop on temperatures
