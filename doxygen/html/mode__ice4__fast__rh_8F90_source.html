54 LOGICAL, DIMENSION(KPROMA), INTENT(IN) :: LDCOMPUTE

56 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRHODREF ! Reference density

59 REAL, DIMENSION(KPROMA), INTENT(IN) :: PPRES ! absolute pressure at t
60 REAL, DIMENSION(KPROMA), INTENT(IN) :: PDV ! Diffusivity of water vapor in the air
61 REAL, DIMENSION(KPROMA), INTENT(IN) :: PKA ! Thermal conductivity of the air
62 REAL, DIMENSION(KPROMA), INTENT(IN) :: PCJ ! Function to compute the ventilation coefficient
63 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAS ! Slope parameter of the aggregate distribution

66 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAH ! Slope parameter of the hail distribution
67 REAL, DIMENSION(KPROMA), INTENT(IN) :: PT ! Temperature
68 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRVT ! Water vapor m.r. at t
69 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRCT ! Cloud water m.r. at t
70 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRRT ! Rain m.r. at t

72 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRST ! Snow/aggregate m.r. at t
73 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRGT ! Graupel m.r. at t
74 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRHT ! Hail m.r. at t
75 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRCWETH ! Dry growth of hailstone
76 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRIWETH ! Dry growth of hailstone
77 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSWETH ! Dry growth of hailstone
78 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRGWETH ! Dry growth of hailstone
79 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRWETH ! Dry growth of hailstone
80 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRCDRYH ! Wet growth of hailstone
81 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRIDRYH ! Wet growth of hailstone
82 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSDRYH ! Wet growth of hailstone
83 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRDRYH ! Wet growth of hailstone
84 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRGDRYH ! Wet growth of hailstone
85 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRDRYHG ! Conversion of hailstone into graupel
86 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRHMLTR ! Melting of the hailstones
87 REAL, DIMENSION(KPROMA, 10), INTENT(INOUT) :: PRH_TEND ! Individual tendencies
88 !
89 !* 0.2 declaration of local variables
90 !
91 INTEGER, PARAMETER :: IRCWETH=1, irrweth=2, iridryh=3, iriweth=4, irsdryh=5, irsweth=6, irgdryh=7, irgweth=8, &
92  & ifreez1=9, ifreez2=10
93 LOGICAL, DIMENSION(KPROMA) :: GWET
94 INTEGER :: IGWET
95 REAL, DIMENSION(KPROMA) :: ZBUF1, ZBUF2, ZBUF3
96 INTEGER, DIMENSION(KPROMA) :: IBUF1, IBUF2, IBUF3
97 REAL, DIMENSION(KPROMA) :: ZZW, &
98  ZRDRYH_INIT, ZRWETH_INIT, &
99  ZRDRYHG
100 INTEGER :: JL
101 REAL(KIND=jphook) :: zhook_handle
102 LOGICAL, DIMENSION(KPROMA) :: LLWETH, LLDRYH
103 !
104 !-------------------------------------------------------------------------------
105 IF (lhook) CALL dr_hook('ICE4_FAST_RH',0,zhook_handle)
106 !
107 !
108 !* 7.2 compute the Wet and Dry growth of hail
109 !
110 DO jl=1, ksize
111  IF(prht(jl)>iced%XRTMIN(7) .AND. prct(jl)>iced%XRTMIN(2) .AND. ldcompute(jl)) THEN
112  IF(.NOT. ldsoft) THEN
113  prh_tend(jl, ircweth)=plbdah(jl)**(iced%XCXH-iced%XDH-2.0) * prhodref(jl)**(-iced%XCEXVT)
114  prh_tend(jl, ircweth)=icep%XFWETH * prct(jl) * prh_tend(jl, ircweth)
115  ENDIF
116  ELSE
117  prh_tend(jl, ircweth)=0.
118  ENDIF
119 
120  IF(prht(jl)>iced%XRTMIN(7) .AND. prit(jl)>iced%XRTMIN(4) .AND. ldcompute(jl)) THEN
121  IF(.NOT. ldsoft) THEN
122  prh_tend(jl, iriweth)=plbdah(jl)**(iced%XCXH-iced%XDH-2.0) * prhodref(jl)**(-iced%XCEXVT)
123  prh_tend(jl, iriweth)=icep%XFWETH * prit(jl) * prh_tend(jl, iriweth) ! RIWETH
124  prh_tend(jl, iridryh)=prh_tend(jl, iriweth)*(icep%XCOLIH*exp(icep%XCOLEXIH*(pt(jl)-cst%XTT))) ! RIDRYH
125  ENDIF
126  ELSE
127  prh_tend(jl, iriweth)=0.
128  prh_tend(jl, iridryh)=0.
129  ENDIF
130 ENDDO
131 
132 !
133 !* 7.2.1 accretion of aggregates on the hailstones
134 !
135 DO jl = 1, ksize
136  IF (prht(jl)>iced%XRTMIN(7) .AND. prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
137  gwet(jl) = .true.
138  ELSE
139  gwet(jl) = .false.
140  prh_tend(jl, irsweth)=0.
141  prh_tend(jl, irsdryh)=0.
142  ENDIF
143 ENDDO
144 IF(.NOT. ldsoft) THEN
145  CALL interp_micro_2d(kproma, ksize, plbdah(:), plbdas(:), icep%NWETLBDAH, icep%NWETLBDAS, &
146  &icep%XWETINTP1H, icep%XWETINTP2H, icep%XWETINTP1S, icep%XWETINTP2S, &
147  &parami%LPACK_INTERP, gwet(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
148  &igwet, &
149  &icep%XKER_SWETH(:,:), zzw(:))
150  IF(igwet>0)THEN
151  !$mnh_expand_where(JL=1:KSIZE)
152  WHERE(gwet(1:ksize))
153 #ifdef REPRO48
154  prh_tend(1:ksize, irsweth)=icep%XFSWETH*zzw(1:ksize) & ! RSWETH
155  *( plbdas(1:ksize)**(iced%XCXS-iced%XBS) )*( plbdah(1:ksize)**iced%XCXH ) &
156  *( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
157  *( icep%XLBSWETH1/( plbdah(1:ksize)**2 ) + &
158  icep%XLBSWETH2/( plbdah(1:ksize) * plbdas(1:ksize) ) + &
159  icep%XLBSWETH3/( plbdas(1:ksize)**2) )
160 #else
161  prh_tend(1:ksize, irsweth)=icep%XFSWETH*zzw(1:ksize) & ! RSWETH
162  *( prst(1:ksize))*( plbdah(1:ksize)**iced%XCXH ) &
163  *( prhodref(1:ksize)**(-iced%XCEXVT) ) &
164  *( icep%XLBSWETH1/( plbdah(1:ksize)**2 ) + &
165  icep%XLBSWETH2/( plbdah(1:ksize) * plbdas(1:ksize) ) + &
166  icep%XLBSWETH3/( plbdas(1:ksize)**2) )
167 #endif
168  prh_tend(1:ksize, irsdryh)=prh_tend(1:ksize, irsweth)*(icep%XCOLSH*exp(icep%XCOLEXSH*(pt(1:ksize)-cst%XTT)))
169  END WHERE
170  !$mnh_end_expand_where(JL=1:KSIZE)
171  ENDIF
172 ENDIF
173 !
174 !* 7.2.6 accretion of graupeln on the hailstones
175 !
176 DO jl = 1, ksize
177  IF (prht(jl)>iced%XRTMIN(7) .AND. prgt(jl)>iced%XRTMIN(6) .AND. ldcompute(jl)) THEN
178  gwet(jl) = .true.
179  ELSE
180  gwet(jl) = .false.
181  prh_tend(jl, irgweth)=0.
182  prh_tend(jl, irgdryh)=0.
183  END IF
184 ENDDO
185 IF(.NOT. ldsoft) THEN
186  CALL interp_micro_2d(kproma, ksize, plbdah(:), plbdag(:), icep%NWETLBDAH, icep%NWETLBDAG, &
187  &icep%XWETINTP1H, icep%XWETINTP2H, icep%XWETINTP1G, icep%XWETINTP2G, &
188  &parami%LPACK_INTERP, gwet(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
189  &igwet, &
190  &icep%XKER_GWETH(:,:), zzw(:))
191  IF(igwet>0)THEN
192  !$mnh_expand_where(JL=1:KSIZE)
193  WHERE(gwet(1:ksize))
194  prh_tend(1:ksize, irgweth)=icep%XFGWETH*zzw(1:ksize) & ! RGWETH
195  *( plbdag(1:ksize)**(iced%XCXG-iced%XBG) )*( plbdah(1:ksize)**iced%XCXH ) &
196  *( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
197  *( icep%XLBGWETH1/( plbdah(1:ksize)**2 ) + &
198  icep%XLBGWETH2/( plbdah(1:ksize) * plbdag(1:ksize) ) + &
199  icep%XLBGWETH3/( plbdag(1:ksize)**2) )
200  prh_tend(1:ksize, irgdryh)=prh_tend(1:ksize, irgweth)
201  END WHERE
202  !When graupel grows in wet mode, graupel is wet (!) and collection efficiency must remain the same
203  WHERE(gwet(1:ksize) .AND. .NOT. ldwetg(1:ksize))
204  prh_tend(1:ksize, irgdryh)=prh_tend(1:ksize, irgdryh)*(icep%XCOLGH*exp(icep%XCOLEXGH*(pt(1:ksize)-cst%XTT)))
205  END WHERE
206  !$mnh_end_expand_where(JL=1:KSIZE)
207  END IF
208 ENDIF
209 !
210 !* 7.2.11 accretion of raindrops on the hailstones
211 !
212 DO jl = 1, ksize
213  IF (prht(jl)>iced%XRTMIN(7) .AND. prrt(jl)>iced%XRTMIN(3) .AND. ldcompute(jl)) THEN
214  gwet(jl) = .true.
215  ELSE
216  gwet(jl) = .false.
217  prh_tend(jl, irrweth)=0.
218  ENDIF
219 ENDDO
220 IF(.NOT. ldsoft) THEN
221  CALL interp_micro_2d(kproma, ksize, plbdah(:), plbdar(:), icep%NWETLBDAH, icep%NWETLBDAR, &
222  &icep%XWETINTP1H, icep%XWETINTP2H, icep%XWETINTP1R, icep%XWETINTP2R, &
223  &parami%LPACK_INTERP, gwet(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
224  &igwet, &
225  &icep%XKER_RWETH(:,:), zzw(:))
226  IF(igwet>0)THEN
227  !$mnh_expand_where(JL=1:KSIZE)
228  WHERE(gwet(1:ksize))
229  prh_tend(1:ksize, irrweth) = icep%XFRWETH*zzw(1:ksize) & ! RRWETH
230  *( plbdar(1:ksize)**(-4) )*( plbdah(1:ksize)**iced%XCXH ) &
231  *( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
232  *( icep%XLBRWETH1/( plbdah(1:ksize)**2 ) + &
233  icep%XLBRWETH2/( plbdah(1:ksize) * plbdar(1:ksize) ) + &
234  icep%XLBRWETH3/( plbdar(1:ksize)**2) )
235  END WHERE
236  !$mnh_end_expand_where(JL=1:KSIZE)
237  ENDIF
238 ENDIF
239 !
240 DO jl=1, ksize
241  zrdryh_init(jl)=prh_tend(jl, ircweth)+prh_tend(jl, iridryh)+ &
242  &prh_tend(jl, irsdryh)+prh_tend(jl, irrweth)+prh_tend(jl, irgdryh)
243 ENDDO
244 !
245 !* 7.3 compute the Wet growth of hail
246 ! and
247 !* 7.4 Select Wet or Dry case
248 !
249 DO jl=1, ksize
250  IF(prht(jl)>iced%XRTMIN(7) .AND. ldcompute(jl)) THEN
251  IF(.NOT. ldsoft) THEN
252  prh_tend(jl, ifreez1)=prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
253  IF(parami%LEVLIMIT) THEN
254  prh_tend(jl, ifreez1)=min(prh_tend(jl, ifreez1), exp(cst%XALPI-cst%XBETAI/pt(jl)-cst%XGAMI*alog(pt(jl)))) ! min(ev, es_i(T))
255  ENDIF
256  prh_tend(jl, ifreez1)=pka(jl)*(cst%XTT-pt(jl)) + &
257  (pdv(jl)*(cst%XLVTT+(cst%XCPV-cst%XCL)*(pt(jl)-cst%XTT)) &
258  *(cst%XESTT-prh_tend(jl, ifreez1))/(cst%XRV*pt(jl)))
259  prh_tend(jl, ifreez1)=prh_tend(jl, ifreez1)* (icep%X0DEPH* plbdah(jl)**icep%XEX0DEPH + &
260  icep%X1DEPH*pcj(jl)*plbdah(jl)**icep%XEX1DEPH)/ &
261  (prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
262  prh_tend(jl, ifreez2)=(prhodref(jl)*(cst%XLMTT+(cst%XCI-cst%XCL)*(cst%XTT-pt(jl)))) / &
263  (prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
264  ENDIF
265 
266  !We must agregate, at least, the cold species
267  zrweth_init(jl)=max(prh_tend(jl, iriweth)+prh_tend(jl, irsweth)+prh_tend(jl, irgweth), &
268  &max(0., prh_tend(jl, ifreez1) + &
269  &prh_tend(jl, ifreez2)*(prh_tend(jl, iriweth)+prh_tend(jl, irsweth)+prh_tend(jl, irgweth))))
270 
271  !Wet case
272  llweth(jl)=max(0., zrweth_init(jl)-prh_tend(jl, iriweth)-prh_tend(jl, irsweth)-prh_tend(jl, irgweth))<= &
273  max(0., zrdryh_init(jl)-prh_tend(jl, iridryh)-prh_tend(jl, irsdryh)-prh_tend(jl, irgdryh))
274  IF(parami%LNULLWETH) THEN
275  llweth(jl) = llweth(jl) .AND. zrdryh_init(jl)>0.
276  ELSE
277  llweth(jl) = llweth(jl) .AND. zrweth_init(jl)>0.
278  ENDIF
279  IF(.NOT. parami%LWETHPOST) THEN
280  llweth(jl) = llweth(jl) .AND. pt(jl)<cst%XTT
281  ENDIF
282 
283  !Dry case
284  lldryh(jl)=pt(jl)<cst%XTT .AND. zrdryh_init(jl)>1.e-20 .AND. &
285  &max(0., zrweth_init(jl)-prh_tend(jl, iriweth)-prh_tend(jl, irsweth))>&
286  &max(0., zrdryh_init(jl)-prh_tend(jl, iridryh)-prh_tend(jl, irsdryh))
287 
288  ELSE
289  prh_tend(jl, ifreez1)=0.
290  prh_tend(jl, ifreez2)=0.
291  zrweth_init(jl)=0.
292  llweth(jl)=.false.
293  lldryh(jl)=.false.
294  ENDIF
295 ENDDO
296 
297 IF(parami%LCONVHG)THEN
298  !$mnh_expand_where(JL=1:KSIZE)
299  WHERE(lldryh(1:ksize))
300  zrdryhg(1:ksize)=zrdryh_init(1:ksize)*zrweth_init(1:ksize)/(zrdryh_init(1:ksize)+zrweth_init(1:ksize))
301  ELSEWHERE
302  zrdryhg(1:ksize)=0.
303  END WHERE
304  !$mnh_end_expand_where(JL=1:KSIZE)
305 ELSE
306  zrdryhg(:)=0.
307 ENDIF
308 
309 DO jl=1, ksize
310  IF(llweth(jl)) THEN
311  prcweth(jl) = prh_tend(jl, ircweth)
312  priweth(jl) = prh_tend(jl, iriweth)
313  prsweth(jl) = prh_tend(jl, irsweth)
314  prgweth(jl) = prh_tend(jl, irgweth)
315  !Collected minus aggregated
316  prrweth(jl) = (zrweth_init(jl) - prh_tend(jl, iriweth) - &
317  prh_tend(jl, irsweth) - prh_tend(jl, irgweth) - &
318  prh_tend(jl, ircweth))
319  ELSE
320  prcweth(jl) = 0.
321  priweth(jl) = 0.
322  prsweth(jl) = 0.
323  prgweth(jl) = 0.
324  prrweth(jl) = 0.
325  ENDIF
326 
327  IF(lldryh(jl)) THEN
328  prcdryh(jl) = prh_tend(jl, ircweth)
329  pridryh(jl) = prh_tend(jl, iridryh)
330  prsdryh(jl) = prh_tend(jl, irsdryh)
331  prrdryh(jl) = prh_tend(jl, irrweth)
332  prgdryh(jl) = prh_tend(jl, irgdryh)
333  prdryhg(jl) = zrdryhg(jl)
334  ELSE
335  prcdryh(jl) = 0.
336  pridryh(jl) = 0.
337  prsdryh(jl) = 0.
338  prrdryh(jl) = 0.
339  prgdryh(jl) = 0.
340  prdryhg(jl) = 0.
341  ENDIF
342 ENDDO
343 !
344 !* 7.5 Melting of the hailstones
345 !
346 DO jl=1, ksize
347  IF(prht(jl)>iced%XRTMIN(7) .AND. pt(jl)>cst%XTT .AND. ldcompute(jl)) THEN
348  IF(.NOT. ldsoft) THEN
349  prhmltr(jl) = prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
350  IF(parami%LEVLIMIT) THEN
351  prhmltr(jl)=min(prhmltr(jl), exp(cst%XALPW-cst%XBETAW/pt(jl)-cst%XGAMW*alog(pt(jl)))) ! min(ev, es_w(T))
352  ENDIF
353  prhmltr(jl) = pka(jl)*(cst%XTT-pt(jl)) + &
354  pdv(jl)*(cst%XLVTT + (cst%XCPV - cst%XCL) * (pt(jl) - cst%XTT)) &
355  *(cst%XESTT-prhmltr(jl))/(cst%XRV*pt(jl))
356  prhmltr(jl) = max(0., (-prhmltr(jl) * (icep%X0DEPH* plbdah(jl)**icep%XEX0DEPH + &
357  icep%X1DEPH*pcj(jl)*plbdah(jl)**icep%XEX1DEPH) - &
358  (prh_tend(jl, ircweth)+prh_tend(jl, irrweth)) * &
359  (prhodref(jl)*cst%XCL*(cst%XTT-pt(jl)))) / &
360  (prhodref(jl)*cst%XLMTT))
361  ENDIF
362  ELSE
363  prhmltr(jl)=0.
364  ENDIF
365 ENDDO
366 !
367 IF (lhook) CALL dr_hook('ICE4_FAST_RH', 1, zhook_handle)
368 !
369 CONTAINS
370 !
371 include "interp_micro.func.h"
372 !
373 END SUBROUTINE ice4_fast_rh
374 END MODULE mode_ice4_fast_rh
