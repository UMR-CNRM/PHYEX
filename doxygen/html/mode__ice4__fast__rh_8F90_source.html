8SUBROUTINE ice4_fast_rh(CST, PARAMI, ICEP, ICED, KPROMA, KSIZE, LDSOFT, LDCOMPUTE, LDWETG, &
9 &PRHODREF, PLVFACT, PLSFACT, PPRES, &
10 &PDV, PKA, PCJ, &
11 &PLBDAS, PLBDAG, PLBDAR, PLBDAH, &
12 &PT, PRVT, PRCT, PRRT, PRIT, PRST, PRGT, PRHT, &
13 &PRCWETH, PRIWETH, PRSWETH, PRGWETH, PRRWETH, &
14 &PRCDRYH, PRIDRYH, PRSDRYH, PRRDRYH, PRGDRYH, PRDRYHG, PRHMLTR, &
15 &PRH_TEND)
16!!
17!!** PURPOSE
18!! -------
19!! Computes the fast rh process
20!!
21!! AUTHOR
22!! ------
23!! S. Riette from the splitting of rain_ice source code (nov. 2014)
24!!
25!! MODIFICATIONS
26!! -------------
27!!
28! P. Wautelet 26/04/2019: replace non-standard FLOAT function by REAL function
29! P. Wautelet 29/05/2019: remove PACK/UNPACK intrinsics (to get more performance and better OpenACC support)
30!! R. El Khatib 24-Aug-2021 Optimizations
31! J. Wurtz 03/2022: New snow characteristics with LSNOW_T
32!
33!
34!* 0. DECLARATIONS
35! ------------
36!
37USE modd_cst, ONLY: cst_t
38USE modd_param_ice_n, ONLY: param_ice_t
39USE modd_rain_ice_descr_n, ONLY: rain_ice_descr_t
40USE modd_rain_ice_param_n, ONLY: rain_ice_param_t
41!
42USE yomhook , ONLY : lhook, dr_hook, jphook
43!
44IMPLICIT NONE
45!
46!* 0.1 Declarations of dummy arguments :
47!
48TYPE(cst_t), INTENT(IN) :: CST
49TYPE(param_ice_t), INTENT(IN) :: PARAMI
50TYPE(rain_ice_param_t), INTENT(IN) :: ICEP
51TYPE(rain_ice_descr_t), INTENT(IN) :: ICED
52INTEGER, INTENT(IN) :: KPROMA, KSIZE
53LOGICAL, INTENT(IN) :: LDSOFT
54LOGICAL, DIMENSION(KPROMA), INTENT(IN) :: LDCOMPUTE
55LOGICAL, DIMENSION(KPROMA), INTENT(IN) :: LDWETG ! .TRUE. where graupel grows in wet mode
56REAL, DIMENSION(KPROMA), INTENT(IN) :: PRHODREF ! Reference density
57REAL, DIMENSION(KPROMA), INTENT(IN) :: PLVFACT
58REAL, DIMENSION(KPROMA), INTENT(IN) :: PLSFACT
59REAL, DIMENSION(KPROMA), INTENT(IN) :: PPRES ! absolute pressure at t
60REAL, DIMENSION(KPROMA), INTENT(IN) :: PDV ! Diffusivity of water vapor in the air
61REAL, DIMENSION(KPROMA), INTENT(IN) :: PKA ! Thermal conductivity of the air
62REAL, DIMENSION(KPROMA), INTENT(IN) :: PCJ ! Function to compute the ventilation coefficient
63REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAS ! Slope parameter of the aggregate distribution
64REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAG ! Slope parameter of the graupel distribution
65REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAR ! Slope parameter of the rain distribution
66REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAH ! Slope parameter of the hail distribution
67REAL, DIMENSION(KPROMA), INTENT(IN) :: PT ! Temperature
68REAL, DIMENSION(KPROMA), INTENT(IN) :: PRVT ! Water vapor m.r. at t
69REAL, DIMENSION(KPROMA), INTENT(IN) :: PRCT ! Cloud water m.r. at t
70REAL, DIMENSION(KPROMA), INTENT(IN) :: PRRT ! Rain m.r. at t
71REAL, DIMENSION(KPROMA), INTENT(IN) :: PRIT ! Pristine ice m.r. at t
72REAL, DIMENSION(KPROMA), INTENT(IN) :: PRST ! Snow/aggregate m.r. at t
73REAL, DIMENSION(KPROMA), INTENT(IN) :: PRGT ! Graupel m.r. at t
74REAL, DIMENSION(KPROMA), INTENT(IN) :: PRHT ! Hail m.r. at t
75REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRCWETH ! Dry growth of hailstone
76REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRIWETH ! Dry growth of hailstone
77REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSWETH ! Dry growth of hailstone
78REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRGWETH ! Dry growth of hailstone
79REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRWETH ! Dry growth of hailstone
80REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRCDRYH ! Wet growth of hailstone
81REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRIDRYH ! Wet growth of hailstone
82REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSDRYH ! Wet growth of hailstone
83REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRDRYH ! Wet growth of hailstone
84REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRGDRYH ! Wet growth of hailstone
85REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRDRYHG ! Conversion of hailstone into graupel
86REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRHMLTR ! Melting of the hailstones
87REAL, DIMENSION(KPROMA, 10), INTENT(INOUT) :: PRH_TEND ! Individual tendencies
88!
89!* 0.2 declaration of local variables
90!
91INTEGER, PARAMETER :: IRCWETH=1, irrweth=2, iridryh=3, iriweth=4, irsdryh=5, irsweth=6, irgdryh=7, irgweth=8, &
92 & ifreez1=9, ifreez2=10
93LOGICAL, DIMENSION(KPROMA) :: GWET
94INTEGER :: IGWET
95REAL, DIMENSION(KPROMA) :: ZBUF1, ZBUF2, ZBUF3
96INTEGER, DIMENSION(KPROMA) :: IBUF1, IBUF2, IBUF3
97REAL, DIMENSION(KPROMA) :: ZZW, &
98 ZRDRYH_INIT, ZRWETH_INIT, &
99 ZRDRYHG
100INTEGER :: JL
101REAL(KIND=jphook) :: zhook_handle
102LOGICAL, DIMENSION(KPROMA) :: LLWETH, LLDRYH
103!
104!-------------------------------------------------------------------------------
105IF (lhook) CALL dr_hook('ICE4_FAST_RH',0,zhook_handle)
106!
107!
108!* 7.2 compute the Wet and Dry growth of hail
109!
110!$acc kernels
111!$acc loop independent
112DO jl=1, ksize
113 IF(prht(jl)>iced%XRTMIN(7) .AND. prct(jl)>iced%XRTMIN(2) .AND. ldcompute(jl)) THEN
114 IF(.NOT. ldsoft) THEN
115 prh_tend(jl, ircweth)=plbdah(jl)**(iced%XCXH-iced%XDH-2.0) * prhodref(jl)**(-iced%XCEXVT)
116 prh_tend(jl, ircweth)=icep%XFWETH * prct(jl) * prh_tend(jl, ircweth)
117 ENDIF
118 ELSE
119 prh_tend(jl, ircweth)=0.
120 ENDIF
121
122 IF(prht(jl)>iced%XRTMIN(7) .AND. prit(jl)>iced%XRTMIN(4) .AND. ldcompute(jl)) THEN
123 IF(.NOT. ldsoft) THEN
124 prh_tend(jl, iriweth)=plbdah(jl)**(iced%XCXH-iced%XDH-2.0) * prhodref(jl)**(-iced%XCEXVT)
125 prh_tend(jl, iriweth)=icep%XFWETH * prit(jl) * prh_tend(jl, iriweth) ! RIWETH
126 prh_tend(jl, iridryh)=prh_tend(jl, iriweth)*(icep%XCOLIH*exp(icep%XCOLEXIH*(pt(jl)-cst%XTT))) ! RIDRYH
127 ENDIF
128 ELSE
129 prh_tend(jl, iriweth)=0.
130 prh_tend(jl, iridryh)=0.
131 ENDIF
132ENDDO
133!$acc end kernels
134
135!
136!* 7.2.1 accretion of aggregates on the hailstones
137!
138!$acc kernels
139!$acc loop independent
140DO jl = 1, ksize
141 IF (prht(jl)>iced%XRTMIN(7) .AND. prst(jl)>iced%XRTMIN(5) .AND. ldcompute(jl)) THEN
142 gwet(jl) = .true.
143 ELSE
144 gwet(jl) = .false.
145 prh_tend(jl, irsweth)=0.
146 prh_tend(jl, irsdryh)=0.
147 ENDIF
148ENDDO
149!$acc end kernels
150IF(.NOT. ldsoft) THEN
151 CALL interp_micro_2d(kproma, ksize, plbdah(:), plbdas(:), icep%NWETLBDAH, icep%NWETLBDAS, &
152 &icep%XWETINTP1H, icep%XWETINTP2H, icep%XWETINTP1S, icep%XWETINTP2S, &
153 &parami%LPACK_INTERP, gwet(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
154 &igwet, &
155 &icep%XKER_SWETH(:,:), zzw(:))
156 IF(igwet>0)THEN
157!$acc kernels
158 !$mnh_expand_where(JL=1:KSIZE)
159 IF(.NOT. icep%LNEWCOEFF) THEN
160 WHERE(gwet(1:ksize))
161 prh_tend(1:ksize, irsweth)=icep%XFSWETH*zzw(1:ksize) & ! RSWETH
162 *( plbdas(1:ksize)**(iced%XCXS-iced%XBS) )*( plbdah(1:ksize)**iced%XCXH ) &
163 *( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
164 *( icep%XLBSWETH1/( plbdah(1:ksize)**2 ) + &
165 icep%XLBSWETH2/( plbdah(1:ksize) * plbdas(1:ksize) ) + &
166 icep%XLBSWETH3/( plbdas(1:ksize)**2) )
167 END WHERE
168 ELSE
169 WHERE(gwet(1:ksize))
170 prh_tend(1:ksize, irsweth)=icep%XFSWETH*zzw(1:ksize) & ! RSWETH
171 *( prst(1:ksize))*( plbdah(1:ksize)**iced%XCXH ) &
172 *( prhodref(1:ksize)**(-iced%XCEXVT) ) &
173 *( icep%XLBSWETH1/( plbdah(1:ksize)**2 ) + &
174 icep%XLBSWETH2/( plbdah(1:ksize) * plbdas(1:ksize) ) + &
175 icep%XLBSWETH3/( plbdas(1:ksize)**2) )
176 END WHERE
177 ENDIF
178 WHERE(gwet(1:ksize))
179 prh_tend(1:ksize, irsdryh)=prh_tend(1:ksize, irsweth)*(icep%XCOLSH*exp(icep%XCOLEXSH*(pt(1:ksize)-cst%XTT)))
180 END WHERE
181 !$mnh_end_expand_where(JL=1:KSIZE)
182!$acc end kernels
183 ENDIF
184ENDIF
185!
186!* 7.2.6 accretion of graupeln on the hailstones
187!
188!$acc kernels
189!$acc loop independent
190DO jl = 1, ksize
191 IF (prht(jl)>iced%XRTMIN(7) .AND. prgt(jl)>iced%XRTMIN(6) .AND. ldcompute(jl)) THEN
192 gwet(jl) = .true.
193 ELSE
194 gwet(jl) = .false.
195 prh_tend(jl, irgweth)=0.
196 prh_tend(jl, irgdryh)=0.
197 END IF
198ENDDO
199!$acc end kernels
200IF(.NOT. ldsoft) THEN
201 CALL interp_micro_2d(kproma, ksize, plbdah(:), plbdag(:), icep%NWETLBDAH, icep%NWETLBDAG, &
202 &icep%XWETINTP1H, icep%XWETINTP2H, icep%XWETINTP1G, icep%XWETINTP2G, &
203 &parami%LPACK_INTERP, gwet(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
204 &igwet, &
205 &icep%XKER_GWETH(:,:), zzw(:))
206 IF(igwet>0)THEN
207!$acc kernels
208 !$mnh_expand_where(JL=1:KSIZE)
209 WHERE(gwet(1:ksize))
210 prh_tend(1:ksize, irgweth)=icep%XFGWETH*zzw(1:ksize) & ! RGWETH
211 *( plbdag(1:ksize)**(iced%XCXG-iced%XBG) )*( plbdah(1:ksize)**iced%XCXH ) &
212 *( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
213 *( icep%XLBGWETH1/( plbdah(1:ksize)**2 ) + &
214 icep%XLBGWETH2/( plbdah(1:ksize) * plbdag(1:ksize) ) + &
215 icep%XLBGWETH3/( plbdag(1:ksize)**2) )
216 prh_tend(1:ksize, irgdryh)=prh_tend(1:ksize, irgweth)
217 END WHERE
218 !When graupel grows in wet mode, graupel is wet (!) and collection efficiency must remain the same
219 WHERE(gwet(1:ksize) .AND. .NOT. ldwetg(1:ksize))
220 prh_tend(1:ksize, irgdryh)=prh_tend(1:ksize, irgdryh)*(icep%XCOLGH*exp(icep%XCOLEXGH*(pt(1:ksize)-cst%XTT)))
221 END WHERE
222 !$mnh_end_expand_where(JL=1:KSIZE)
223!$acc end kernels
224 END IF
225ENDIF
226!
227!* 7.2.11 accretion of raindrops on the hailstones
228!
229!$acc kernels
230!$acc loop independent
231DO jl = 1, ksize
232 IF (prht(jl)>iced%XRTMIN(7) .AND. prrt(jl)>iced%XRTMIN(3) .AND. ldcompute(jl)) THEN
233 gwet(jl) = .true.
234 ELSE
235 gwet(jl) = .false.
236 prh_tend(jl, irrweth)=0.
237 ENDIF
238ENDDO
239!$acc end kernels
240IF(.NOT. ldsoft) THEN
241 CALL interp_micro_2d(kproma, ksize, plbdah(:), plbdar(:), icep%NWETLBDAH, icep%NWETLBDAR, &
242 &icep%XWETINTP1H, icep%XWETINTP2H, icep%XWETINTP1R, icep%XWETINTP2R, &
243 &parami%LPACK_INTERP, gwet(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
244 &igwet, &
245 &icep%XKER_RWETH(:,:), zzw(:))
246 IF(igwet>0)THEN
247!$acc kernels
248 !$mnh_expand_where(JL=1:KSIZE)
249 WHERE(gwet(1:ksize))
250 prh_tend(1:ksize, irrweth) = icep%XFRWETH*zzw(1:ksize) & ! RRWETH
251 *( plbdar(1:ksize)**(-4) )*( plbdah(1:ksize)**iced%XCXH ) &
252 *( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
253 *( icep%XLBRWETH1/( plbdah(1:ksize)**2 ) + &
254 icep%XLBRWETH2/( plbdah(1:ksize) * plbdar(1:ksize) ) + &
255 icep%XLBRWETH3/( plbdar(1:ksize)**2) )
256 END WHERE
257 !$mnh_end_expand_where(JL=1:KSIZE)
258!$acc end kernels
259 ENDIF
260ENDIF
261!
262!$acc kernels
263!$acc loop independent
264DO jl=1, ksize
265 zrdryh_init(jl)=prh_tend(jl, ircweth)+prh_tend(jl, iridryh)+ &
266 &prh_tend(jl, irsdryh)+prh_tend(jl, irrweth)+prh_tend(jl, irgdryh)
267ENDDO
268!$acc end kernels
269!
270!* 7.3 compute the Wet growth of hail
271! and
272!* 7.4 Select Wet or Dry case
273!
274!$acc kernels
275!$acc loop independent
276DO jl=1, ksize
277 IF(prht(jl)>iced%XRTMIN(7) .AND. ldcompute(jl)) THEN
278 IF(.NOT. ldsoft) THEN
279 prh_tend(jl, ifreez1)=prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
280 IF(parami%LEVLIMIT) THEN
281 prh_tend(jl, ifreez1)=min(prh_tend(jl, ifreez1), exp(cst%XALPI-cst%XBETAI/pt(jl)-cst%XGAMI*log(pt(jl)))) ! min(ev, es_i(T))
282 ENDIF
283 prh_tend(jl, ifreez1)=pka(jl)*(cst%XTT-pt(jl)) + &
284 (pdv(jl)*(cst%XLVTT+(cst%XCPV-cst%XCL)*(pt(jl)-cst%XTT)) &
285 *(cst%XESTT-prh_tend(jl, ifreez1))/(cst%XRV*pt(jl)))
286 prh_tend(jl, ifreez1)=prh_tend(jl, ifreez1)* (icep%X0DEPH* plbdah(jl)**icep%XEX0DEPH + &
287 icep%X1DEPH*pcj(jl)*plbdah(jl)**icep%XEX1DEPH)/ &
288 (prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
289 prh_tend(jl, ifreez2)=(prhodref(jl)*(cst%XLMTT+(cst%XCI-cst%XCL)*(cst%XTT-pt(jl)))) / &
290 (prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))))
291 ENDIF
292
293 !We must agregate, at least, the cold species
294 zrweth_init(jl)=max(prh_tend(jl, iriweth)+prh_tend(jl, irsweth)+prh_tend(jl, irgweth), &
295 &max(0., prh_tend(jl, ifreez1) + &
296 &prh_tend(jl, ifreez2)*(prh_tend(jl, iriweth)+prh_tend(jl, irsweth)+prh_tend(jl, irgweth))))
297
298 !Wet case
299 llweth(jl)=max(0., zrweth_init(jl)-prh_tend(jl, iriweth)-prh_tend(jl, irsweth)-prh_tend(jl, irgweth))<= &
300 max(0., zrdryh_init(jl)-prh_tend(jl, iridryh)-prh_tend(jl, irsdryh)-prh_tend(jl, irgdryh))
301 IF(parami%LNULLWETH) THEN
302 llweth(jl) = llweth(jl) .AND. zrdryh_init(jl)>0.
303 ELSE
304 llweth(jl) = llweth(jl) .AND. zrweth_init(jl)>0.
305 ENDIF
306 IF(.NOT. parami%LWETHPOST) THEN
307 llweth(jl) = llweth(jl) .AND. pt(jl)<cst%XTT
308 ENDIF
309
310 !Dry case
311 lldryh(jl)=pt(jl)<cst%XTT .AND. zrdryh_init(jl)>1.e-20 .AND. &
312 &max(0., zrweth_init(jl)-prh_tend(jl, iriweth)-prh_tend(jl, irsweth))>&
313 &max(0., zrdryh_init(jl)-prh_tend(jl, iridryh)-prh_tend(jl, irsdryh))
314
315 ELSE
316 prh_tend(jl, ifreez1)=0.
317 prh_tend(jl, ifreez2)=0.
318 zrweth_init(jl)=0.
319 llweth(jl)=.false.
320 lldryh(jl)=.false.
321 ENDIF
322ENDDO
323!$acc end kernels
324
325IF(parami%LCONVHG)THEN
326!$acc kernels
327 !$mnh_expand_where(JL=1:KSIZE)
328 WHERE(lldryh(1:ksize))
329 zrdryhg(1:ksize)=zrdryh_init(1:ksize)*zrweth_init(1:ksize)/(zrdryh_init(1:ksize)+zrweth_init(1:ksize))
330 ELSEWHERE
331 zrdryhg(1:ksize)=0.
332 END WHERE
333 !$mnh_end_expand_where(JL=1:KSIZE)
334!$acc end kernels
335ELSE
336!$acc kernels
337 zrdryhg(:)=0.
338!$acc end kernels
339ENDIF
340
341!$acc kernels
342!$acc loop independent
343DO jl=1, ksize
344 IF(llweth(jl)) THEN
345 prcweth(jl) = prh_tend(jl, ircweth)
346 priweth(jl) = prh_tend(jl, iriweth)
347 prsweth(jl) = prh_tend(jl, irsweth)
348 prgweth(jl) = prh_tend(jl, irgweth)
349 !Collected minus aggregated
350 prrweth(jl) = (zrweth_init(jl) - prh_tend(jl, iriweth) - &
351 prh_tend(jl, irsweth) - prh_tend(jl, irgweth) - &
352 prh_tend(jl, ircweth))
353 ELSE
354 prcweth(jl) = 0.
355 priweth(jl) = 0.
356 prsweth(jl) = 0.
357 prgweth(jl) = 0.
358 prrweth(jl) = 0.
359 ENDIF
360
361 IF(lldryh(jl)) THEN
362 prcdryh(jl) = prh_tend(jl, ircweth)
363 pridryh(jl) = prh_tend(jl, iridryh)
364 prsdryh(jl) = prh_tend(jl, irsdryh)
365 prrdryh(jl) = prh_tend(jl, irrweth)
366 prgdryh(jl) = prh_tend(jl, irgdryh)
367 prdryhg(jl) = zrdryhg(jl)
368 ELSE
369 prcdryh(jl) = 0.
370 pridryh(jl) = 0.
371 prsdryh(jl) = 0.
372 prrdryh(jl) = 0.
373 prgdryh(jl) = 0.
374 prdryhg(jl) = 0.
375 ENDIF
376ENDDO
377!$acc end kernels
378!
379!* 7.5 Melting of the hailstones
380!
381!$acc kernels
382!$acc loop independent
383DO jl=1, ksize
384 IF(prht(jl)>iced%XRTMIN(7) .AND. pt(jl)>cst%XTT .AND. ldcompute(jl)) THEN
385 IF(.NOT. ldsoft) THEN
386 prhmltr(jl) = prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
387 IF(parami%LEVLIMIT) THEN
388 prhmltr(jl)=min(prhmltr(jl), exp(cst%XALPW-cst%XBETAW/pt(jl)-cst%XGAMW*log(pt(jl)))) ! min(ev, es_w(T))
389 ENDIF
390 prhmltr(jl) = pka(jl)*(cst%XTT-pt(jl)) + &
391 pdv(jl)*(cst%XLVTT + (cst%XCPV - cst%XCL) * (pt(jl) - cst%XTT)) &
392 *(cst%XESTT-prhmltr(jl))/(cst%XRV*pt(jl))
393 prhmltr(jl) = max(0., (-prhmltr(jl) * (icep%X0DEPH* plbdah(jl)**icep%XEX0DEPH + &
394 icep%X1DEPH*pcj(jl)*plbdah(jl)**icep%XEX1DEPH) - &
395 (prh_tend(jl, ircweth)+prh_tend(jl, irrweth)) * &
396 (prhodref(jl)*cst%XCL*(cst%XTT-pt(jl)))) / &
397 (prhodref(jl)*cst%XLMTT))
398 ENDIF
399 ELSE
400 prhmltr(jl)=0.
401 ENDIF
402ENDDO
403!$acc end kernels
404!
405IF (lhook) CALL dr_hook('ICE4_FAST_RH', 1, zhook_handle)
406!
407CONTAINS
408!
409include "interp_micro.func.h"
410!
411END SUBROUTINE ice4_fast_rh
412END MODULE mode_ice4_fast_rh
