19!! PURPOSE
20!! -------
21!!** Routine to diagnose cloud fraction, liquid and ice condensate mixing ratios
22!! and s'rl'/sigs^2
23!!
24!!
25!!** METHOD
26!! ------
27!! Based on the large-scale fields of temperature, water vapor, and possibly
28!! liquid and solid condensate, the conserved quantities r_t and h_l are constructed
29!! and then fractional cloudiness, liquid and solid condensate is diagnosed.
30!!
31!! The total variance is parameterized as the sum of stratiform/turbulent variance
32!! and a convective variance.
33!! The turbulent variance is parameterized as a function of first-order moments, and
34!! the convective variance is modelled as a function of the convective mass flux
35!! (units kg/s m^2) as provided by the mass flux convection scheme.
36!!
37!! Nota: if the host model does not use prognostic values for liquid and solid condensate
38!! or does not provide a convective mass flux, put all these values to zero.
39!!
40!!
41!! EXTERNAL
42!! --------
43!! INI_CST
44!!
45!! IMPLICIT ARGUMENTS
46!! ------------------
47!! Module MODD_CST : contains physical constants
48!!
49!! REFERENCE
50!! ---------

307 cst%XCL*prr(jij,jk) + &
308 cst%XCI*(prs(jij,jk) + prg(jij,jk) )

357 END DO

359END IF

394 END DO

396END IF

398!$acc kernels
399!$acc loop independent collapse(2)
400DO jk=iktb,ikte
401 DO jij = iijb, iije
402 IF (.NOT. ocnd2) THEN
403 ! latent heats
404 ! saturated water vapor mixing ratio over liquid water and ice
405 zpv(jij,jk) = min(exp( cst%XALPW - cst%XBETAW / pt(jij,jk) - cst%XGAMW * log( pt(jij,jk) ) ), .99*ppabs(jij,jk))
406 zpiv(jij,jk) = min(exp( cst%XALPI - cst%XBETAI / pt(jij,jk) - cst%XGAMI * log( pt(jij,jk) ) ), .99*ppabs(jij,jk))
407 ENDIF
408 !Ice fraction
409 zfrac(jij,jk) = 0.
410 IF (ouseri .AND. .NOT.ocnd2) THEN
411 IF (prc_in(jij,jk)+pri_in(jij,jk) > 1.e-20) THEN
412 zfrac(jij,jk) = pri_in(jij,jk) / (prc_in(jij,jk)+pri_in(jij,jk))
413 ENDIF
414 CALL compute_frac_ice(cst, hfrac_ice, nebn, zfrac(jij,jk), pt(jij,jk), ierr) !error code IERR cannot be checked here to not break vectorization
415 ENDIF
416 zqsl(jij,jk) = cst%XRD / cst%XRV * zpv(jij,jk) / ( ppabs(jij,jk) - zpv(jij,jk) )
417 zqsi(jij,jk) = cst%XRD / cst%XRV * zpiv(jij,jk) / ( ppabs(jij,jk) - zpiv(jij,jk) )
418
419 ! interpolate between liquid and solid as function of temperature
420 zqsl(jij,jk) = (1. - zfrac(jij,jk)) * zqsl(jij,jk) + zfrac(jij,jk) * zqsi(jij,jk)
421 zlvs(jij,jk) = (1. - zfrac(jij,jk)) * zlv(jij,jk) + &
422 & zfrac(jij,jk) * zls(jij,jk)
423
424 ! coefficients a and b
425 zah(jij,jk) = zlvs(jij,jk) * zqsl(jij,jk) / ( cst%XRV * pt(jij,jk)**2 ) * (cst%XRV * zqsl(jij,jk) / cst%XRD + 1.)
426 za(jij,jk) = 1. / ( 1. + zlvs(jij,jk)/zcpd(jij,jk) * zah(jij,jk) )
427 zb(jij,jk) = zah(jij,jk) * za(jij,jk)
428 zsbar(jij,jk) = za(jij,jk) * ( zrt(jij,jk) - zqsl(jij,jk) + &
429 & zah(jij,jk) * zlvs(jij,jk) * (prc_in(jij,jk)+pri_in(jij,jk)*zprifact(jij,jk)) / zcpd(jij,jk))
430 ! switch to take either present computed value of SIGMAS
431 ! or that of Meso-NH turbulence scheme
432 IF ( osigmas ) THEN
433 IF (zsigqsat(jij,jk)/=0.) THEN
434 zdzfact(jij,jk) = 1.
435 IF(nebn%LHGT_QS .AND. jk+1 <= ikte)THEN

437 ELSEIF(nebn%LHGT_QS)THEN
438 zdzfact(jij,jk)= max(icep%XFRMIN(23),min(icep%XFRMIN(24),((pzz(jij,jk-1) - pzz(jij,jk)))*0.8/zdzref(jij,jk)))
