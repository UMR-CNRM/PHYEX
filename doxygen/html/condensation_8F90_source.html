18 !!
19 !! PURPOSE
20 !! -------
21 !!** Routine to diagnose cloud fraction, liquid and ice condensate mixing ratios
22 !! and s'rl'/sigs^2
23 !!
24 !!
25 !!** METHOD
26 !! ------
27 !! Based on the large-scale fields of temperature, water vapor, and possibly
28 !! liquid and solid condensate, the conserved quantities r_t and h_l are constructed
29 !! and then fractional cloudiness, liquid and solid condensate is diagnosed.
30 !!
31 !! The total variance is parameterized as the sum of stratiform/turbulent variance
32 !! and a convective variance.
33 !! The turbulent variance is parameterized as a function of first-order moments, and
34 !! the convective variance is modelled as a function of the convective mass flux
35 !! (units kg/s m^2) as provided by the mass flux convection scheme.
36 !!
37 !! Nota: if the host model does not use prognostic values for liquid and solid condensate
38 !! or does not provide a convective mass flux, put all these values to zero.
39 !!
40 !!
41 !! EXTERNAL
42 !! --------
43 !! INI_CST
44 !!
45 !! IMPLICIT ARGUMENTS
46 !! ------------------
47 !! Module MODD_CST : contains physical constants
48 !!
49 !! REFERENCE
50 !! ---------

162 REAL, DIMENSION(D%NIJT,D%NKT) :: ZL ! length scale

183 ! end OCND2

199 REAL,PARAMETER :: ZCSIG_CONV = 0.30e-2 ! scaling factor for ZSIG_CONV as function of mass flux
200 !
201 
202 REAL, DIMENSION(-22:11),PARAMETER :: ZSRC_1D =(/ &
203  0. , 0. , 2.0094444e-04, 0.316670e-03, &
204  4.9965648e-04, 0.785956e-03 , 1.2341294e-03, 0.193327e-02, &
205  3.0190963e-03, 0.470144e-02 , 7.2950651e-03, 0.112759e-01, &
206  1.7350994e-02, 0.265640e-01 , 4.0427860e-02, 0.610997e-01, &
207  9.1578111e-02, 0.135888e+00 , 0.1991484 , 0.230756e+00, &
208  0.2850565 , 0.375050e+00 , 0.5000000 , 0.691489e+00, &
209  0.8413813 , 0.933222e+00 , 0.9772662 , 0.993797e+00, &

215 !
216 IF (lhook) CALL dr_hook('CONDENSATION',0,zhook_handle)
217 !
218 iktb=d%NKTB
219 ikte=d%NKTE
220 ikb=d%NKB
221 ike=d%NKE
222 ikl=d%NKL
223 iijb=d%NIJB
224 iije=d%NIJE
225 !
226 pcldfr(:,:) = 0. ! Initialize values
227 psigrc(:,:) = 0. ! Initialize values
228 prv_out(:,:)= 0. ! Initialize values
229 prc_out(:,:)= 0. ! Initialize values
230 pri_out(:,:)= 0. ! Initialize values
231 zprifact = 1. ! Initialize value
232 zardum2 = 0. ! Initialize values
233 zcldini = -1. ! Dummy Initialized cloud input to icecloud routine
234 pifr = 10. ! ratio of cloud ice water mixing ratio wet to dry
235  ! part of a gridbox
236 zdzref = icep%XFRMIN(25) ! Thickness for unchanged vqsigsat (only used for LHGT_QS)
237 !
238 IF(ocnd2)zprifact = 0.
239 !
240 !
241 !-------------------------------------------------------------------------------
242 ! store total water mixing ratio
243 DO jk=iktb,ikte
244  DO jij=iijb,iije
245  zrt(jij,jk) = prv_in(jij,jk) + prc_in(jij,jk) + pri_in(jij,jk)*zprifact
246  END DO
247 END DO
248 !-------------------------------------------------------------------------------
249 ! Preliminary calculations
250 ! latent heat of vaporisation/sublimation
251 IF(PRESENT(plv) .AND. PRESENT(pls)) THEN
252  zlv(:,:)=plv(:,:)
253  zls(:,:)=pls(:,:)
254 ELSE
255  DO jk=iktb,ikte
256  DO jij=iijb,iije
257  ! latent heat of vaporisation/sublimation
258  zlv(jij,jk) = cst%XLVTT + ( cst%XCPV - cst%XCL ) * ( pt(jij,jk) - cst%XTT )
259  zls(jij,jk) = cst%XLSTT + ( cst%XCPV - cst%XCI ) * ( pt(jij,jk) - cst%XTT )
260  ENDDO
261  ENDDO
262 ENDIF
263 IF(PRESENT(pcph)) THEN
264  zcpd(:,:)=pcph(:,:)
265 ELSE
266  DO jk=iktb,ikte
267  DO jij=iijb,iije
268  zcpd(jij,jk) = cst%XCPD + cst%XCPV*prv_in(jij,jk) + cst%XCL*prc_in(jij,jk) + cst%XCI*pri_in(jij,jk) + &
269  cst%XCL*prr(jij,jk) + &
270  cst%XCI*(prs(jij,jk) + prg(jij,jk) )
271  ENDDO
272  ENDDO
273 ENDIF
274 ! Preliminary calculations needed for computing the "turbulent part" of Sigma_s
275 IF ( .NOT. osigmas ) THEN
276  DO jk=iktb,ikte
277  DO jij=iijb,iije
278  ! store temperature at saturation
279  ztlk(jij,jk) = pt(jij,jk) - zlv(jij,jk)*prc_in(jij,jk)/zcpd(jij,jk) &
280  - zls(jij,jk)*pri_in(jij,jk)/zcpd(jij,jk)*zprifact
281  END DO
282  END DO
283  ! Determine tropopause/inversion height from minimum temperature
284  itpl(:) = ikb+ikl
285  ztmin(:) = 400.
286  DO jk = iktb+1,ikte-1
287  DO jij=iijb,iije
288  IF ( pt(jij,jk) < ztmin(jij) ) THEN
289  ztmin(jij) = pt(jij,jk)
290  itpl(jij) = jk
291  ENDIF
292  END DO
293  END DO
294  ! Set the mixing length scale
295  zl(:,ikb) = 20.
296  DO jk = ikb+ikl,ike,ikl
297  DO jij=iijb,iije
298  ! free troposphere
299  zl(jij,jk) = zl0
300  zzz = pzz(jij,jk) - pzz(jij,ikb)
301  jkp = itpl(jij)
302  ! approximate length for boundary-layer
303  IF ( zl0 > zzz ) zl(jij,jk) = zzz
304  ! gradual decrease of length-scale near and above tropopause
305  IF ( zzz > 0.9*(pzz(jij,jkp)-pzz(jij,ikb)) ) &
306  zl(jij,jk) = .6 * zl(jij,jk-ikl)
307  END DO
308  END DO
309 END IF
310 !-------------------------------------------------------------------------------
311 !
312 DO jk=iktb,ikte
313  jkp=max(min(jk+ikl,ikte),iktb)
314  jkm=max(min(jk-ikl,ikte),iktb)
315  IF (ocnd2) THEN
316  DO jij = iijb, iije
317  zdz(jij) = pzz(jij,jkp) - pzz(jij,jkp-ikl)
318  ENDDO
319  CALL icecloud(d,cst,icep,ppabs(:,jk),pzz(:,jk),zdz(:), &
320  & pt(:,jk),prv_in(:,jk),1.,-1., &
321  & zcldini(:),pifr(iijb,jk),picldfr(:,jk), &
322  & pssio(:,jk),pssiu(:,jk),zardum2(:),zardum(:))
323  ! latent heats
324  ! saturated water vapor mixing ratio over liquid water and ice
325  DO jij=iijb,iije
326  esatw_t(jij)=esatw(icep%TIWMX, pt(jij,jk))
327  zpv(jij) = min(esatw_t(jij), .99*ppabs(jij,jk))
328  zpiv(jij) = min(esati(icep%TIWMX, pt(jij,jk)), .99*ppabs(jij,jk))
329  END DO
330  ELSE
331  ! latent heats
332  ! saturated water vapor mixing ratio over liquid water and ice
333  DO jij=iijb,iije
334  zpv(jij) = min(exp( cst%XALPW - cst%XBETAW / pt(jij,jk) - cst%XGAMW * log( pt(jij,jk) ) ), .99*ppabs(jij,jk))
335  zpiv(jij) = min(exp( cst%XALPI - cst%XBETAI / pt(jij,jk) - cst%XGAMI * log( pt(jij,jk) ) ), .99*ppabs(jij,jk))
336  END DO
337  ENDIF
338  !Ice fraction
339  zfrac(:) = 0.
340  IF (ouseri .AND. .NOT.ocnd2) THEN
341  DO jij=iijb,iije
342  IF (prc_in(jij,jk)+pri_in(jij,jk) > 1.e-20) THEN
343  zfrac(jij) = pri_in(jij,jk) / (prc_in(jij,jk)+pri_in(jij,jk))
344  ENDIF
345  END DO
346  DO jij=iijb,iije
347  CALL compute_frac_ice(cst, hfrac_ice, nebn, zfrac(jij), pt(jij,jk), ierr) !error code IERR cannot be checked here to not break vectorization
348  ENDDO
349  ENDIF
350  DO jij=iijb,iije
351  zqsl(jij) = cst%XRD / cst%XRV * zpv(jij) / ( ppabs(jij,jk) - zpv(jij) )
352  zqsi(jij) = cst%XRD / cst%XRV * zpiv(jij) / ( ppabs(jij,jk) - zpiv(jij) )
353 
354  ! interpolate between liquid and solid as function of temperature
355  zqsl(jij) = (1. - zfrac(jij)) * zqsl(jij) + zfrac(jij) * zqsi(jij)
356  zlvs = (1. - zfrac(jij)) * zlv(jij,jk) + &
357  & zfrac(jij) * zls(jij,jk)
358 
359  ! coefficients a and b
360  zah = zlvs * zqsl(jij) / ( cst%XRV * pt(jij,jk)**2 ) * (cst%XRV * zqsl(jij) / cst%XRD + 1.)
361  za(jij) = 1. / ( 1. + zlvs/zcpd(jij,jk) * zah )
362  zb(jij) = zah * za(jij)
363  zsbar(jij) = za(jij) * ( zrt(jij,jk) - zqsl(jij) + &
364  & zah * zlvs * (prc_in(jij,jk)+pri_in(jij,jk)*zprifact) / zcpd(jij,jk))
365  END DO
366  ! switch to take either present computed value of SIGMAS
367  ! or that of Meso-NH turbulence scheme
368  IF ( osigmas ) THEN
369  DO jij=iijb,iije
370  IF (psigqsat(jij)/=0.) THEN
371  zdzfact = 1.
372  IF(nebn%LHGT_QS .AND. jk+1 <= ikte)THEN
373  zdzfact= max(icep%XFRMIN(23),min(icep%XFRMIN(24),(pzz(jij,jk) - pzz(jij,jk+1))/zdzref))
374  ELSEIF(nebn%LHGT_QS)THEN
375  zdzfact= max(icep%XFRMIN(23),min(icep%XFRMIN(24),((pzz(jij,jk-1) - pzz(jij,jk)))*0.8/zdzref))
376  ENDIF
377  IF (nebn%LSTATNW) THEN
378  zsigma(jij) = sqrt((psigs(jij,jk))**2 + (psigqsat(jij)*zdzfact*zqsl(jij)*za(jij))**2)
379  ELSE
380  zsigma(jij) = sqrt((2*psigs(jij,jk))**2 + (psigqsat(jij)*zqsl(jij)*za(jij))**2)
381  ENDIF
382  ELSE
383  IF (nebn%LSTATNW) THEN
384  zsigma(jij) = psigs(jij,jk)
385  ELSE
386  zsigma(jij) = 2*psigs(jij,jk)
387  ENDIF
388  END IF
389  END DO
390  ELSE
391  DO jij=iijb,iije
392  ! parameterize Sigma_s with first_order closure
393  dzz = pzz(jij,jkp) - pzz(jij,jkm)
394  zdrw = zrt(jij,jkp) - zrt(jij,jkm)
395  zdtl = ztlk(jij,jkp) - ztlk(jij,jkm) + cst%XG/zcpd(jij,jk) * dzz
396  zll = zl(jij,jk)
397  ! standard deviation due to convection
398  zsig_conv =0.
399  IF(lmfconv) zsig_conv = zcsig_conv * pmfconv(jij,jk) / za(jij)
400  ! zsigma should be of order 4.e-4 in lowest 5 km of atmosphere
401  zsigma(jij) = sqrt( max( 1.e-25, zcsigma * zcsigma * zll*zll/(dzz*dzz)*(&
402  za(jij)*za(jij)*zdrw*zdrw - 2.*za(jij)*zb(jij)*zdrw*zdtl + zb(jij)*zb(jij)*zdtl*zdtl) + &
403  zsig_conv * zsig_conv ) )
404  END DO
405  END IF
406  DO jij=iijb,iije
407  zsigma(jij)= max( 1.e-10, zsigma(jij) )
408 
409  ! normalized saturation deficit
410  zq1(jij) = zsbar(jij)/zsigma(jij)
411  END DO
412  IF(hcondens == 'GAUS') THEN
413  DO jij=iijb,iije
414  ! Gaussian Probability Density Function around ZQ1
415  ! Computation of ZG and ZGAM(=erf(ZG))
