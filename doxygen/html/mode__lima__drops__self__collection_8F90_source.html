38 USE modd_param_lima_warm, ONLY : xaccr1, xscbuexp1, xscbu_eff1, xscbu_eff2, &
39  xscbu2, xscbu3
40 !
41 IMPLICIT NONE
42 !
43 !* 0.1 Declarations of dummy arguments :
44 !
45 LOGICAL, DIMENSION(:),INTENT(IN) :: LDCOMPUTE
46 !

48 !
49 REAL, DIMENSION(:), INTENT(IN) :: PCRT ! Rain drops C. at t
50 REAL, DIMENSION(:), INTENT(IN) :: PLBDR !
51 REAL, DIMENSION(:), INTENT(IN) :: PLBDR3 !
52 !
53 REAL, DIMENSION(:), INTENT(OUT) :: P_CR_SCBU

59  ZW2, &
60  ZW3, &

62  ZSCBU
63 !
64 !-------------------------------------------------------------------------------
65 !
66 !
67 !* 1. Rain drops self-collection and break-up
68 ! ---------------------------------------
69 !
70 !
71 p_cr_scbu(:)=0.
72 !
73 zw4(:) =0.
74 !
75 WHERE( pcrt(:)>xctmin(3) .AND. ldcompute(:) )
76  zw4(:) = xaccr1 / plbdr(:) ! Mean diameter
77 END WHERE
78 zscbu(:)=1.
79 WHERE (zw4(:)>=xscbu_eff1 .AND. pcrt(:)>xctmin(3) .AND. ldcompute(:)) &
80  zscbu(:) = exp(xscbuexp1*(zw4(:)-xscbu_eff1)) ! coalescence efficiency
81 WHERE (zw4(:)>=xscbu_eff2 .AND. ldcompute(:)) zscbu(:) = 0.0 ! Break-up
82 !
83 zw1(:) = 0.0
84 zw2(:) = 0.0
85 zw3(:) = 0.0
86 !
87 WHERE ( pcrt(:)>xctmin(3) .AND. zw4(:)>1.e-4 .AND. ldcompute(:)) ! analytical integration
88  zw1(:) = xscbu2 * pcrt(:)**2 / plbdr3(:) ! D>100 10-6 m
89  zw3(:) = zw1(:)*zscbu(:)
90 END WHERE
91 !
92 WHERE ( pcrt(:)>xctmin(3) .AND. zw4(:)<=1.e-4 .AND. ldcompute(:))
93  zw2(:) = xscbu3 *(pcrt(:) / plbdr3(:))**2 ! D<100 10-6 m
94  zw3(:) = zw2(:)
95 END WHERE
96 !
97 p_cr_scbu(:) = - zw3(:) * prhodref(:)
98 !
99 !
100 !-------------------------------------------------------------------------------
101 !
102 END SUBROUTINE lima_drops_self_collection
