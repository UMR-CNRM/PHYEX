19!!**** *MIMA_ADJUST* - compute the fast microphysical sources
20!!
21!! PURPOSE
22!! -------
23!! The purpose of this routine is to compute the fast microphysical sources
24!! through an explict scheme and a saturation ajustement procedure.
25!!
26!!
27!!** METHOD
28!! ------
29!! Reisin et al., 1996 for the explicit scheme when ice is present
30!! Langlois, Tellus, 1973 for the implict adjustment for the cloud water
31!! (refer also to book 1 of the documentation).
32!!
33!! Computations are done separately for three cases :
34!! - ri>0 and rc=0
35!! - rc>0 and ri=0
36!! - ri>0 and rc>0
37!!
38!!
39!! EXTERNAL
40!! --------
41!! None
42!!
43!!
44!! IMPLICIT ARGUMENTS
45!! ------------------
46!! Module MODD_CST
47!! XP00 ! Reference pressure
48!! XMD,XMV ! Molar mass of dry air and molar mass of vapor
49!! XRD,XRV ! Gaz constant for dry air, gaz constant for vapor
50!! XCPD,XCPV ! Cpd (dry air), Cpv (vapor)
51!! XCL ! Cl (liquid)
52!! XTT ! Triple point temperature
53!! XLVTT ! Vaporization heat constant
54!! XALPW,XBETAW,XGAMW ! Constants for saturation vapor
55!! ! pressure function
56!! Module MODD_CONF
57!! CCONF
58!! Module MODD_BUDGET:
59!! NBUMOD
60!! CBUTYPE
61!! LBU_RTH
62!! LBU_RRV
63!! LBU_RRC
64!! Module MODD_LES : NCTR_LES,LTURB_LES,NMODNBR_LES
65!! XNA declaration (cloud fraction as global var)
66!!
67!! REFERENCE
68!! ---------
69!!
70!! Book 1 and Book2 of documentation ( routine FAST_TERMS )
71!! Langlois, Tellus, 1973
72!!
73!! AUTHOR
74!! ------
75!! E. Richard * Laboratoire d'Aerologie*
76!! J.-M. Cohard * Laboratoire d'Aerologie*
77!! J.-P. Pinty * Laboratoire d'Aerologie*
78!! S. Berthet * Laboratoire d'Aerologie*
79!! B. Vi√© * Laboratoire d'Aerologie*
80!!
81!! MODIFICATIONS
82!! -------------

250!
251!* 1. PRELIMINARIES
252! -------------
253!
254IF (lhook) CALL dr_hook('LIMA_ADJUST_SPLIT', 0, zhook_handle)
255!
256isv_lima_nc = tnsv%NSV_LIMA_NC - tnsv%NSV_LIMA_BEG + 1
257isv_lima_ccn_free = tnsv%NSV_LIMA_CCN_FREE - tnsv%NSV_LIMA_BEG + 1
258isv_lima_ccn_acti = tnsv%NSV_LIMA_CCN_ACTI - tnsv%NSV_LIMA_BEG + 1
259isv_lima_scavmass = tnsv%NSV_LIMA_SCAVMASS - tnsv%NSV_LIMA_BEG + 1
260!
261isize = SIZE(limap%XRTMIN)
262ALLOCATE(zrtmin(isize))
263zrtmin(:) = limap%XRTMIN(:) / ptstep
264isize = SIZE(limap%XCTMIN)
265ALLOCATE(zctmin(isize))
266zctmin(:) = limap%XCTMIN(:) / ptstep
267!
268! Prepare 3D water mixing ratios
269!
270ztht = pths*ptstep
271!
272zrvt(:,:) = prs(:,:,1)*ptstep
273zrvs(:,:) = prs(:,:,1)
274!
275zrct(:,:) = 0.
276zrcs(:,:) = 0.
277zrrt(:,:) = 0.
278zrrs(:,:) = 0.
279zrit(:,:) = 0.
280zris(:,:) = 0.
281zrss(:,:) = 0.
282zrgs(:,:) = 0.
283zrhs(:,:) = 0.
284!
285IF ( krr .GE. 2 ) zrct(:,:) = prs(:,:,2)*ptstep
286IF ( krr .GE. 2 ) zrcs(:,:) = prs(:,:,2)
287IF ( krr .GE. 3 ) zrrt(:,:) = prt(:,:,3)
288IF ( krr .GE. 3 ) zrrs(:,:) = prs(:,:,3)
289IF ( krr .GE. 4 ) zrit(:,:) = prt(:,:,4)
290IF ( krr .GE. 4 ) zris(:,:) = prs(:,:,4)
291IF ( krr .GE. 5 ) zrss(:,:) = prs(:,:,5)
292IF ( krr .GE. 6 ) zrgs(:,:) = prs(:,:,6)
293IF ( krr .GE. 7 ) zrhs(:,:) = prs(:,:,7)
294!
295! Prepare 3D number concentrations
296zcct(:,:) = 0.
297zccs(:,:) = 0.
298!
299IF ( limap%NMOM_C.GE.2 ) zcct(:,:) = psvs(:,:,isv_lima_nc)*ptstep
300IF ( limap%NMOM_C.GE.2 ) zccs(:,:) = psvs(:,:,isv_lima_nc)
301!
302IF ( limap%LSCAV .AND. limap%LAERO_MASS ) zmas(:,:) = psvs(:,:,isv_lima_scavmass)
303!
304IF ( limap%NMOM_C.GE.1 .AND. limap%NMOD_CCN.GE.1 ) THEN
305 ALLOCATE( znfs(d%NIJT,d%NKT,limap%NMOD_CCN) )
306 ALLOCATE( znas(d%NIJT,d%NKT,limap%NMOD_CCN) )
307 ALLOCATE( znft(d%NIJT,d%NKT,limap%NMOD_CCN) )
308 ALLOCATE( znat(d%NIJT,d%NKT,limap%NMOD_CCN) )
309 znfs(:,:,:) = psvs(:,:,isv_lima_ccn_free:isv_lima_ccn_free+limap%NMOD_CCN-1)
310 znas(:,:,:) = psvs(:,:,isv_lima_ccn_acti:isv_lima_ccn_acti+limap%NMOD_CCN-1)
311 znft(:,:,:) = psvs(:,:,isv_lima_ccn_free:isv_lima_ccn_free+limap%NMOD_CCN-1)*ptstep
312 znat(:,:,:) = psvs(:,:,isv_lima_ccn_acti:isv_lima_ccn_acti+limap%NMOD_CCN-1)*ptstep
313END IF
314!
315! Initialize budgets
316!
317IF ( buconf%LBU_ENABLE ) then
318 IF ( buconf%LBUDGET_TH ) CALL budget_store_init_phy(d, tbudgets(nbudget_th), 'CEDS', pths(:,:) * prhodj(:,:) )
319 IF ( buconf%LBUDGET_RV ) CALL budget_store_init_phy(d, tbudgets(nbudget_rv), 'CEDS', zrvs(:,:) * prhodj(:,:) )
320 IF ( buconf%LBUDGET_RC ) CALL budget_store_init_phy(d, tbudgets(nbudget_rc), 'CEDS', zrcs(:,:) * prhodj(:,:) )
321 !Remark: ZRIS is not modified but source term kept for better coherence with lima_adjust and lima_notadjust
322 IF ( buconf%LBUDGET_RI ) CALL budget_store_init_phy(d, tbudgets(nbudget_ri), 'CEDS', zris(:,:) * prhodj(:,:) )
323 IF ( buconf%LBUDGET_SV ) then
324 IF ( limap%NMOM_C.GE.2) &
325 CALL budget_store_init_phy(d, tbudgets(nbudget_sv1 - 1 + tnsv%NSV_LIMA_NC), 'CEDS', zccs(:,:) * prhodj(:,:) )
326 IF ( limap%LSCAV .AND. limap%LAERO_MASS ) &
327 CALL budget_store_init_phy(d, tbudgets(nbudget_sv1 - 1 + tnsv%NSV_LIMA_SCAVMASS), 'CEDS', zmas(:,:) * prhodj(:,:) )
328 IF ( limap%NMOM_C.GE.1 ) then
329 DO il = 1, limap%NMOD_CCN
330 idx = nbudget_sv1 - 1 + tnsv%NSV_LIMA_CCN_FREE - 1 + il
331 CALL budget_store_init_phy(d, tbudgets(idx), 'CEDS', znfs(:,:, il) * prhodj(:,:) )
332 idx = nbudget_sv1 - 1 + tnsv%NSV_LIMA_CCN_ACTI - 1 + il
333 CALL budget_store_init_phy(d, tbudgets(idx), 'CEDS', znas(:,:, il) * prhodj(:,:) )
334 END DO
335 END IF
336 END IF
337END IF
338!
339!-------------------------------------------------------------------------------
340!
341!
342!* 2. CONDENSATION
343! ------------
344!
345!
346WHERE ( zrvs(:,:)+zrcs(:,:)+zris(:,:) < 0.)
347 zrvs(:,:) = - zrcs(:,:) - zris(:,:)
348END WHERE
349!
350zexns(:,:) = ( ppabst(:,:) / cst%XP00 ) ** (cst%XRD/cst%XCPD)
351!
352zt(:,:) = ( pths(:,:) * ptstep ) * zexns(:,:)
353zt2(:,:) = zt(:,:)
354zcph(:,:) = cst%XCPD + cst%XCPV *ptstep* zrvs(:,:) &
355 + cst%XCL *ptstep* ( zrcs(:,:) + zrrs(:,:) ) &
356 + cst%XCI *ptstep* ( zris(:,:) + zrss(:,:) + zrgs(:,:) + zrhs(:,:) )
357zlv(:,:) = cst%XLVTT + ( cst%XCPV - cst%XCL ) * ( zt(:,:) -cst%XTT )
358zls(:,:) = cst%XLSTT + ( cst%XCPV - cst%XCI ) * ( zt(:,:) -cst%XTT )
359!
360IF (limap%LADJ) THEN
361 zrv_in=zrvs*ptstep
362 zrc_in=zrcs*ptstep
363 IF (limap%NMOM_I.EQ.1) THEN
364 zri_in=zris*ptstep
365 guseri=.true.
366 ELSE
367 zri_in=0.
368 guseri=.false.
369 END IF
370 IF (osubg_cond) THEN
371 z_sigs=psigs
372 g_sigmas=osigmas
373 ELSE
374 z_sigs=0.
375 g_sigmas=.true.
376 END IF
377 !
378 CALL condensation(d, cst, rain_ice_paramn, nebn, turbn, &
379 'S', hcondens, hlambda3, &
380 ppabst, pzz, prhodref, zt, zrv_in, zrv, zrc_in, zrc, zri_in, zri, &
381 zrrs*ptstep,zrss*ptstep, zrgs*ptstep, &
382 z_sigs, .false., pmfconv, pcldfr, z_srcs, guseri, g_sigmas, .false., &
383 zdum, zdum, zdum, zdum, zdum, &
384 psigqsat, plv=zlv, pls=zls, pcph=zcph )
385 !
386 IF (limap%NMOM_C.GE.1) THEN
387 zw1(:,:) = (zrc(:,:) - zrcs(:,:)*ptstep) / ptstep
388 WHERE( zw1(:,:) < 0.0 )
389 zw1(:,:) = max( zw1(:,:), -zrcs(:,:) )
390 ELSEWHERE
391 zw1(:,:) = min( zw1(:,:), zrvs(:,:) )
392 END WHERE
393 zrvs(:,:) = zrvs(:,:) - zw1(:,:)
394 zrcs(:,:) = zrcs(:,:) + zw1(:,:)
395 pths(:,:) = pths(:,:) + &
396 zw1(:,:) * zlv(:,:) / (zcph(:,:) * pexnref(:,:))
397 END IF
398 !
399 IF (limap%NMOM_I.EQ.1) THEN
400 picefr(:,:)=pcldfr(:,:)
401 zw2(:,:) = (zri(:,:) - zris(:,:)*ptstep) / ptstep ! idem ZW1 but for Ri
402 !
403 WHERE( zw2(:,:) < 0.0 )
404 zw2(:,:) = max( zw2(:,:), -zris(:,:) )
405 ELSEWHERE
406 zw2(:,:) = min( zw2(:,:), zrvs(:,:) )
407 END WHERE
408 zrvs(:,:) = zrvs(:,:) - zw2(:,:)
409 zris(:,:) = zris(:,:) + zw2(:,:)
410 pths(:,:) = pths(:,:) + &
411 zw2(:,:) * zls(:,:) / (zcph(:,:) * pexnref(:,:))
412 END IF
413 !
414ELSE
415 DO ii=1,SIZE(zrcs,1)
416 DO ik=1,SIZE(zrcs,2)
417 IF (zrcs(ii,ik).GE.limap%XRTMIN(2) .AND. zccs(ii,ik).GE.limap%XCTMIN(2)) THEN
418 zvec1(ii,ik) = max( 1.0001, min( float(limaw%NAHEN)-0.0001, limaw%XAHENINTP1 * zt(ii,ik) + limaw%XAHENINTP2 ) )
419 ivec1(ii,ik) = int( zvec1(ii,ik) )
420 zvec1(ii,ik) = zvec1(ii,ik) - float( ivec1(ii,ik) )
421 zw(ii,ik) = exp( cst%XALPW - cst%XBETAW/zt(ii,ik) - cst%XGAMW*log(zt(ii,ik) ) ) ! es_w
422 zw(ii,ik) = cst%XMV / cst%XMD * zw(ii,ik) / ( ppabst(ii,ik)-zw(ii,ik) )
423 zs(ii,ik) = zrvs(ii,ik)*ptstep / zw(ii,ik) - 1.
424 zw(ii,ik) = zccs(ii,ik)*ptstep/(limaw%XLBC*zccs(ii,ik)/zrcs(ii,ik))**limaw%XLBEXC
425 zw2(ii,ik) = limaw%XAHENG3(ivec1(ii,ik)+1)*zvec1(ii,ik)-limaw%XAHENG3(ivec1(ii,ik))*(zvec1(ii,ik)-1.)
426 zcnd(ii,ik) = 2.*3.14*1000.*zw2(ii,ik)*zs(ii,ik)*zw(ii,ik)
427 IF(zcnd(ii,ik).LE.0.) THEN
428 zcnd(ii,ik) = max( zcnd(ii,ik), -zrcs(ii,ik) )
429 ELSE
430 zcnd(ii,ik) = min( zcnd(ii,ik), zrvs(ii,ik) )
431 END IF
432 zrvs(ii,ik) = zrvs(ii,ik) - zcnd(ii,ik)
433 zrcs(ii,ik) = zrcs(ii,ik) + zcnd(ii,ik)
434 pths(ii,ik) = pths(ii,ik) + zcnd(ii,ik) * zlv(ii,ik) / (zcph(ii,ik) * pexnref(ii,ik))
435 END IF
436 END DO
437 END DO
438END IF
439!
440IF (osubg_cond) THEN
441 psrcs=z_srcs
442ENDIF
443IF (osubg_cond .AND. limap%NMOM_C.GE.2 .AND. limap%LACTI) THEN
444 zw_mf=0.
445 zrv2=zrvt
446 zrc2=zrct
447 CALL lima_ccn_activation (limap, limaw, tnsv, d, cst, nebn,&
448 kcarb, ksoa, ksp, odust, osalt, oorilam, &
449 prhodref, pexnref, ppabst, zt2, pdthrad, pw_nu+zw_mf, &
450 paero,psolorg, pmi, hactccn, &
451 ztht, zrv2, zrc2, zcct, zrrt, znft, znat, &
452 pcldfr )
453END IF
454!
455!-------------------------------------------------------------------------------
456!
457!* 3. CLOUD FROM MASS-FLUX SCHEME
458! ---------------------------
459!
460IF ( .NOT. osubg_cond ) THEN
461 WHERE (zrcs(:,:) + zris(:,:) > 1.e-12 / ptstep)
462 pcldfr(:,:) = 1.
463 ELSEWHERE
464 pcldfr(:,:) = 0.
465 ENDWHERE
466 IF ( SIZE(psrcs,2) /= 0 ) THEN
467 WHERE (zrcs(:,:) + zris(:,:) > 1.e-12 / ptstep)
468 psrcs(:,:) = 1.
469 ELSEWHERE
470 psrcs(:,:) = 0.
471 ENDWHERE
472 END IF
473ELSE
474 ! We limit PRC_MF+PRI_MF to ZRVS*PTSTEP to avoid negative humidity
475 zw1(:,:)=prc_mf(:,:)/ptstep
476 zw2(:,:)=0.
477 IF (limap%NMOM_I.EQ.1) zw2(:,:)=pri_mf(:,:)/ptstep
478 WHERE(zw1(:,:)+zw2(:,:)>zrvs(:,:))
479 zw1(:,:)=zw1(:,:)*zrvs(:,:)/(zw1(:,:)+zw2(:,:))
480 zw2(:,:)=zrvs(:,:)-zw1(:,:)
481 ENDWHERE
482 ! Compute CF and update rc, ri from MF scheme
483 pcldfr(:,:) = min(1.,pcldfr(:,:)+pcf_mf(:,:))
484 zrvs(:,:) = zrvs(:,:) - zw1(:,:) -zw2(:,:)
485 zrcs(:,:) = zrcs(:,:) + zw1(:,:)
486 IF (limap%NMOM_I.EQ.1) zris(:,:) = zris(:,:) + zw2(:,:)
487 IF (limap%NMOM_C.GE.1) zccs(:,:) = zcct(:,:) / ptstep
488 IF (limap%NMOD_CCN.GE.1) znfs(:,:,:) = znft(:,:,:) / ptstep
489 IF (limap%NMOD_CCN.GE.1) znas(:,:,:) = znat(:,:,:) / ptstep
490 pths(:,:) = pths(:,:) + &
491 (zw1(:,:) * zlv(:,:) + zw2(:,:) * zls(:,:)) / zcph(:,:) &
492 / pexnref(:,:)
493END IF
494!
495!-------------------------------------------------------------------------------
496!
497!* 4. REMOVE SMALL NUMBERS OF DROPLETS
498! --------------------------------
499!
500IF (limap%NMOM_C .GE. 2) THEN
501 zmask(:,:) = 0.0
502 zw(:,:) = 0.
503 WHERE (zrcs(:,:) <= zrtmin(2) .OR. zccs(:,:) <=0.)
504 zrvs(:,:) = zrvs(:,:) + zrcs(:,:)
505 pths(:,:) = pths(:,:) - zrcs(:,:)*zlv(:,:)/(zcph(:,:)*zexns(:,:))
506 zrcs(:,:) = 0.0
507 zw(:,:) = max(zccs(:,:),0.)
508 zccs(:,:) = 0.0
509 pcldfr(:,:) = 0.
510 END WHERE
511 !
512 zw1(:,:) = 0.
513 IF (limap%NMOD_CCN.GE.1) zw1(:,:) = sum(znas,dim=3)
514 zw(:,:) = min( zw(:,:), zw1(:,:) )
515 zw2(:,:) = 0.
516 WHERE ( zw(:,:) > 0. )
517 zmask(:,:) = 1.0
518 zw2(:,:) = zw(:,:) / zw1(:,:)
519 ENDWHERE
520 !
521 IF (limap%NMOD_CCN.GE.1) THEN
522 DO imod = 1, limap%NMOD_CCN
523 znfs(:,:,imod) = znfs(:,:,imod) + &
524 zmask(:,:) * znas(:,:,imod) * zw2(:,:)
525 znas(:,:,imod) = znas(:,:,imod) - &
526 zmask(:,:) * znas(:,:,imod) * zw2(:,:)
527 znas(:,:,imod) = max( 0.0 , znas(:,:,imod) )
528 ENDDO
529 END IF
530 IF (limap%LSCAV .AND. limap%LAERO_MASS) zmas(:,:) = zmas(:,:) * (1-zmask(:,:))
531END IF
532!
533!-------------------------------------------------------------------------------
534!
535!* 5. SAVE CHANGES & CLEANING
536! -----------------------
537!
538! 3D water mixing ratios
539prs(:,:,1) = zrvs(:,:)
540IF ( krr .GE. 2 ) prs(:,:,2) = zrcs(:,:)
541IF ( krr .GE. 3 ) prs(:,:,3) = zrrs(:,:)
542IF ( krr .GE. 4 ) prs(:,:,4) = zris(:,:)
543IF ( krr .GE. 5 ) prs(:,:,5) = zrss(:,:)
544IF ( krr .GE. 6 ) prs(:,:,6) = zrgs(:,:)
545IF ( krr .GE. 7 ) prs(:,:,7) = zrhs(:,:)
546!
547! Prepare 3D number concentrations
548!
549IF ( limap%NMOM_C.GE.2 ) psvs(:,:,isv_lima_nc) = zccs(:,:)
550IF ( limap%LSCAV .AND. limap%LAERO_MASS ) psvs(:,:,isv_lima_scavmass) = zmas(:,:)
551IF ( limap%NMOM_C.GE.1 .AND. limap%NMOD_CCN.GE.1 ) THEN
552 psvs(:,:,isv_lima_ccn_free:isv_lima_ccn_free+limap%NMOD_CCN-1) = znfs(:,:,:)
553 psvs(:,:,isv_lima_ccn_acti:isv_lima_ccn_acti+limap%NMOD_CCN-1) = znas(:,:,:)
554END IF
555!
556! Initialize budgets
557!
558IF ( buconf%LBU_ENABLE ) THEN
559 IF ( buconf%LBUDGET_TH ) CALL budget_store_end_phy(d, tbudgets(nbudget_th), 'CEDS', pths(:,:) * prhodj(:,:) )
560 IF ( buconf%LBUDGET_RV ) CALL budget_store_end_phy(d, tbudgets(nbudget_rv), 'CEDS', zrvs(:,:) * prhodj(:,:) )
561 IF ( buconf%LBUDGET_RC ) CALL budget_store_end_phy(d, tbudgets(nbudget_rc), 'CEDS', zrcs(:,:) * prhodj(:,:) )
562 IF ( buconf%LBUDGET_RI ) CALL budget_store_end_phy(d, tbudgets(nbudget_ri), 'CEDS', zris(:,:) * prhodj(:,:) )
563 IF ( buconf%LBUDGET_SV ) THEN
564 IF ( limap%NMOM_C.GE.2) &
565 CALL budget_store_end_phy(d, tbudgets(nbudget_sv1 - 1 + tnsv%NSV_LIMA_NC), 'CEDS', zccs(:,:) * prhodj(:,:) )
566 IF ( limap%LSCAV .AND. limap%LAERO_MASS ) &
567 CALL budget_store_end_phy(d, tbudgets(nbudget_sv1 - 1 + tnsv%NSV_LIMA_SCAVMASS), 'CEDS', zmas(:,:) * prhodj(:,:) )
568 IF ( limap%NMOM_C.GE.1 ) THEN
569 DO il = 1, limap%NMOD_CCN
570 idx = nbudget_sv1 - 1 + tnsv%NSV_LIMA_CCN_FREE - 1 + il
571 CALL budget_store_end_phy(d, tbudgets(idx), 'CEDS', znfs(:,:, il) * prhodj(:,:) )
572 idx = nbudget_sv1 - 1 + tnsv%NSV_LIMA_CCN_ACTI - 1 + il
573 CALL budget_store_end_phy(d, tbudgets(idx), 'CEDS', znas(:,:, il) * prhodj(:,:) )
574 END DO
575 END IF
576 END IF
577END IF
578!
579DEALLOCATE(zrtmin)
580DEALLOCATE(zctmin)
581IF (ALLOCATED(znfs)) DEALLOCATE(znfs)
582IF (ALLOCATED(znas)) DEALLOCATE(znas)
583IF (ALLOCATED(znft)) DEALLOCATE(znft)
584IF (ALLOCATED(znat)) DEALLOCATE(znat)
585!
586!------------------------------------------------------------------------------
587!
588IF (lhook) CALL dr_hook('LIMA_ADJUST_SPLIT', 1, zhook_handle)
589END SUBROUTINE lima_adjust_split
