16 !!
17 !! PURPOSE
18 !! -------
19 !! The purpose of this routine is to compute the activation of CCN
20 !! according to Cohard and Pinty, QJRMS, 2000
21 !!
22 !!
23 !!** METHOD
24 !! ------
25 !! The activation of CCN is checked for quasi-saturated air parcels
26 !! to update the cloud droplet number concentration.
27 !!
28 !! Computation steps :
29 !! 1- Check where computations are necessary
30 !! 2- and 3- Compute the maximum of supersaturation using the iterative
31 !! Ridder algorithm
32 !! 4- Compute the nucleation source
33 !! 5- Deallocate local variables
34 !!
35 !! Contains :
36 !! 6- Functions : Ridder algorithm
37 !!
38 !!
39 !! REFERENCE
40 !! ---------
41 !!
42 !! Cohard, J.-M. and J.-P. Pinty, 2000: A comprehensive two-moment warm
43 !! microphysical bulk scheme.
44 !! Part I: Description and tests
45 !! Part II: 2D experiments with a non-hydrostatic model
46 !! Accepted for publication in Quart. J. Roy. Meteor. Soc.
47 !!
48 !! AUTHOR
49 !! ------

51 !! J.-P. Pinty * Laboratoire d'Aerologie*
52 !! S. Berthet * Laboratoire d'Aerologie*
53 !! B. Vié * Laboratoire d'Aerologie*

135  ZSW, & ! real supersaturation
136  ZSMAX, & ! Maximum supersaturation
137  ZVEC1
138 !

142  :: ZTDT, ZRVSAT, ZW, ZW2, ZCLDFR

144  :: ZCONC_TOT ! total CCN C. available
145 !

147  ! interpolations

149 !

151 REAL :: ZS1, ZS2, ZXACC

157 !-------------------------------------------------------------------------------
158 !
159 !ILUOUT = TLUOUT%NLU
160 !
161 !* 1. PREPARE COMPUTATIONS - PACK
162 ! ---------------------------
163 !
164 iib=1+jphext
165 iie=SIZE(prhodref,1) - jphext
166 ijb=1+jphext
167 ije=SIZE(prhodref,2) - jphext
168 ikb=1+jpvext
169 ike=SIZE(prhodref,3) - jpvext
170 !
171 ! Saturation vapor mixing ratio and radiative tendency
172 !
173 zeps= cst%XMV / cst%XMD
174 zrvsat(:,:,:) = zeps / (ppabst(:,:,:)*exp(-cst%XALPW+cst%XBETAW/pt(:,:,:)+cst%XGAMW*alog(pt(:,:,:))) - 1.0)
175 ztdt(:,:,:) = 0.
176 IF (lactit .AND. SIZE(pdthrad).GT.0) ztdt(:,:,:) = pdthrad(:,:,:) * pexnref(:,:,:)
177 !
178 ! find locations where CCN are available
179 !
180 zconc_tot(:,:,:) = 0.0
181 DO jmod = 1, nmod_ccn
182  zconc_tot(:,:,:) = zconc_tot(:,:,:) + pnft(:,:,:,jmod) ! sum over the free CCN
183 ENDDO
184 !
185 ! optimization by looking for locations where
186 ! the updraft velocity is positive!!!
187 !
188 gnuct(:,:,:) = .false.
189 !
190 IF (ladj) THEN
191  gnuct(iib:iie,ijb:ije,ikb:ike) = pw_nu(iib:iie,ijb:ije,ikb:ike)>xwmin &
192  .OR. prvt(iib:iie,ijb:ije,ikb:ike)>zrvsat(iib:iie,ijb:ije,ikb:ike)
193  IF (lactit) gnuct(iib:iie,ijb:ije,ikb:ike) = gnuct(iib:iie,ijb:ije,ikb:ike) &
194  .OR. ztdt(iib:iie,ijb:ije,ikb:ike)<xtmin
195 !
196  gnuct(iib:iie,ijb:ije,ikb:ike) = gnuct(iib:iie,ijb:ije,ikb:ike) &
197  .AND. pt(iib:iie,ijb:ije,ikb:ike)>(cst%XTT-22.) &
198  .AND. zconc_tot(iib:iie,ijb:ije,ikb:ike)>xctmin(2)
199 !
200  IF (lsubg_cond) gnuct(iib:iie,ijb:ije,ikb:ike) = gnuct(iib:iie,ijb:ije,ikb:ike) &
201  .AND. pcldfr(iib:iie,ijb:ije,ikb:ike)>0.01
202  IF (.NOT. lsubg_cond) gnuct(iib:iie,ijb:ije,ikb:ike) = gnuct(iib:iie,ijb:ije,ikb:ike) &
203  .AND. prvt(iib:iie,ijb:ije,ikb:ike).GE.zrvsat(iib:iie,ijb:ije,ikb:ike)
204 ELSE
205  gnuct(iib:iie,ijb:ije,ikb:ike) = prvt(iib:iie,ijb:ije,ikb:ike).GE.zrvsat(iib:iie,ijb:ije,ikb:ike) &
206  .AND. pt(iib:iie,ijb:ije,ikb:ike)>(cst%XTT-22.) &
207  .AND. zconc_tot(iib:iie,ijb:ije,ikb:ike)>xctmin(2)
208 END IF
209 !
210 IF (.NOT. lsubg_cond) THEN
211  zcldfr(:,:,:) = 1.
212 ELSE
213  zcldfr(:,:,:) = pcldfr(:,:,:)
214 END IF
215 !
216 inuct = countjv( gnuct(:,:,:),i1(:),i2(:),i3(:))
217 !
218 IF( inuct >= 1 ) THEN
219 !
220  ALLOCATE(znft(inuct,nmod_ccn))
221  ALLOCATE(znat(inuct,nmod_ccn))
222  ALLOCATE(ztmp(inuct,nmod_ccn))
223  ALLOCATE(zrct(inuct))
224  ALLOCATE(zcct(inuct))
225  ALLOCATE(zzt(inuct))
226  ALLOCATE(zztdt(inuct))
227  ALLOCATE(zsw(inuct))
228  ALLOCATE(zzw1(inuct))
229  ALLOCATE(zzw2(inuct))
230  ALLOCATE(zzw3(inuct))
231  ALLOCATE(zzw4(inuct))
232  ALLOCATE(zzw5(inuct))
233  ALLOCATE(zzw6(inuct))
234  ALLOCATE(zchen_multi(inuct,nmod_ccn))
235  ALLOCATE(zvec1(inuct))
236  ALLOCATE(ivec1(inuct))
237  ALLOCATE(zrhodref(inuct))
238  ALLOCATE(zexnref(inuct))
239  DO jl=1,inuct
240  zrct(jl) = prct(i1(jl),i2(jl),i3(jl))/zcldfr(i1(jl),i2(jl),i3(jl))
241  zcct(jl) = pcct(i1(jl),i2(jl),i3(jl))/zcldfr(i1(jl),i2(jl),i3(jl))
242  zzt(jl) = pt(i1(jl),i2(jl),i3(jl))
243  zzw1(jl) = zrvsat(i1(jl),i2(jl),i3(jl))
244  zzw2(jl) = pw_nu(i1(jl),i2(jl),i3(jl))
245  zztdt(jl) = ztdt(i1(jl),i2(jl),i3(jl))
246  zsw(jl) = prvt(i1(jl),i2(jl),i3(jl))/zrvsat(i1(jl),i2(jl),i3(jl)) - 1.
247  zrhodref(jl) = prhodref(i1(jl),i2(jl),i3(jl))
248  zexnref(jl) = pexnref(i1(jl),i2(jl),i3(jl))
249  DO jmod = 1,nmod_ccn
250  znft(jl,jmod) = pnft(i1(jl),i2(jl),i3(jl),jmod)
251  znat(jl,jmod) = pnat(i1(jl),i2(jl),i3(jl),jmod)
252  zchen_multi(jl,jmod) = (znft(jl,jmod)+znat(jl,jmod))*zrhodref(jl) &
253  / xlimit_factor(jmod)
254  ENDDO
255  ENDDO
256 !
257  ALLOCATE(zsmax(inuct))
258  IF (ladj) THEN
259  zzw1(:) = 1.0/zeps + 1.0/zzw1(:) &
260  + (((cst%XLVTT+(cst%XCPV-cst%XCL)*(zzt(:)-cst%XTT))/zzt(:))**2)/(cst%XCPD*cst%XRV) ! Psi2
261 !
262 !
263 !-------------------------------------------------------------------------------
264 !
265 !
266 !* 2. compute the constant term (ZZW3) relative to smax
267 ! ----------------------------------------------------
268 !
269 ! Remark : in LIMA's nucleation parameterization, Smax=0.01 for a supersaturation of 1% !
270 !
271 !
272  zvec1(:) = max( 1.0001, min( real(nahen)-0.0001, xahenintp1 * zzt(:) + xahenintp2 ) )
273  ivec1(:) = int( zvec1(:) )
274  zvec1(:) = zvec1(:) - real( ivec1(:) )
275 !
276 !
277  IF (lactit) THEN ! including a cooling rate
278 !
279 ! Compute the tabulation of function of ZZW3 :
280 !
281 ! (Psi1*w+Psi3*DT/Dt)**1.5
282 ! ZZW3 = XAHENG*(Psi1*w + Psi3*DT/Dt)**1.5 = ------------------------
283 ! 2*pi*rho_l*G**(3/2)
284 !
285 !
286  zzw4(:)=xpsi1( ivec1(:)+1)*zzw2(:)+xpsi3(ivec1(:)+1)*zztdt(:)
287  zzw5(:)=xpsi1( ivec1(:) )*zzw2(:)+xpsi3(ivec1(:) )*zztdt(:)
288  WHERE (zzw4(:) < 0. .OR. zzw5(:) < 0.)
289  zzw4(:) = 0.
290  zzw5(:) = 0.
291  END WHERE
292  zzw3(:) = xaheng( ivec1(:)+1)*(zzw4(:)**1.5)* zvec1(:) &
293  - xaheng( ivec1(:) )*(zzw5(:)**1.5)*(zvec1(:) - 1.0)
294  ! Cste*((Psi1*w+Psi3*dT/dt)/(G))**1.5
295  zzw6(:) = xaheng2( ivec1(:)+1)*(zzw4(:)**0.5)* zvec1(:) &
296  - xaheng2( ivec1(:) )*(zzw5(:)**0.5)*(zvec1(:) - 1.0)
297 !
298 !
299  ELSE ! LACTIT , for clouds
300 !
301 !
302 ! Compute the tabulation of function of ZZW3 :
303 !
304 ! (Psi1 * w)**1.5
305 ! ZZW3 = XAHENG * (Psi1 * w)**1.5 = -------------------------
306 ! 2 pi rho_l * G**(3/2)
307 !
308 !
309  zzw2(:)=max(zzw2(:),0.)
310  zzw3(:)=xaheng(ivec1(:)+1)*((xpsi1(ivec1(:)+1)*zzw2(:))**1.5)* zvec1(:) &
311  -xaheng(ivec1(:) )*((xpsi1(ivec1(:) )*zzw2(:))**1.5)*(zvec1(:)-1.0)
312 !
313  zzw6(:)=xaheng2(ivec1(:)+1)*((xpsi1(ivec1(:)+1)*zzw2(:))**0.5)* zvec1(:) &
314  -xaheng2(ivec1(:) )*((xpsi1(ivec1(:) )*zzw2(:))**0.5)*(zvec1(:)-1.0)
315 !
316  END IF ! LACTIT
317 !
318 !
319 ! (Psi1*w+Psi3*DT/Dt)**1.5 rho_air
320 ! ZZW3 = ------------------------ * -------
321 ! 2*pi*rho_l*G**(3/2) Psi2
322 !
323  zzw5(:) = 1.
324  zzw3(:) = (zzw3(:)/zzw1(:))*zrhodref(:) ! R.H.S. of Eq 9 of CPB 98 but
325  ! for multiple aerosol modes
326  WHERE (zrct(:) > xrtmin(2) .AND. zcct(:) > xctmin(2))
327  zzw6(:) = zzw6(:) * zrhodref(:) * zcct(:) / (xlbc*zcct(:)/zrct(:))**xlbexc
328  ELSEWHERE
329  zzw6(:)=0.
330  END WHERE
331 
332  WHERE (zzw3(:) == 0. .AND. .NOT.(zsw>0.))
333  zzw5(:) = -1.
334  END WHERE
335 !
336 !-------------------------------------------------------------------------------
337 !
338 !
339 !* 3. Compute the maximum of supersaturation
340 ! -----------------------------------------
341 !
342 !
343 ! estimate S_max for the CPB98 parameterization with SEVERAL aerosols mode
344 ! Reminder : Smax=0.01 for a 1% supersaturation
345 !
346 ! Interval bounds to tabulate sursaturation Smax
347 ! Check with values used for tabulation in ini_lima_warm.f90
348  zs1 = 1.0e-5 ! corresponds to 0.001% supersaturation
349  zs2 = 5.0e-2 ! corresponds to 5.0% supersaturation
350  zxacc = 1.0e-10 ! Accuracy needed for the search in [NO UNITS]
351 !
352  zsmax(:) = zriddr(zs1,zs2,zxacc,zzw3(:),zzw6(:),inuct) ! ZSMAX(:) is in [NO UNITS]
353  zsmax(:) = min(max(zsmax(:), zsw(:)),zs2)
354  !
355  ELSE
356  zsmax(:) = zsw(:)
357  zzw5(:) = 1.
358  END IF
359 !
360 !-------------------------------------------------------------------------------
361 !
362 !
363 !* 4. Compute the nucleus source
364 ! -----------------------------
365 !
366 !
367 ! Again : Smax=0.01 for a 1% supersaturation
368 ! Modified values for Beta and C (see in init_aerosol_properties) account for that
369 !
370  WHERE (zzw5(:) > 0. .AND. zsmax(:) > 0.)
371  zvec1(:) = max( 1.0001, min( real(nhyp)-0.0001, xhypintp1*log(zsmax(:))+xhypintp2 ) )
372  ivec1(:) = int( zvec1(:) )
373  zvec1(:) = zvec1(:) - real( ivec1(:) )
374  END WHERE
375  zzw6(:) = 0. ! initialize the change of cloud droplet concentration
376 !
377  ztmp(:,:)=0.0
378 !
379 ! Compute the concentration of activable aerosols for each mode
380 ! based on the max of supersaturation ( -> ZTMP )
381 !
382  DO jmod = 1, nmod_ccn ! iteration on mode number
383  zzw1(:) = 0.
384  zzw2(:) = 0.
385  zzw3(:) = 0.
386  !
387  WHERE( zzw5(:) > 0. .AND. zsmax(:)>0.0 )
388  zzw2(:) = xhypf12( ivec1(:)+1,jmod )* zvec1(:) & ! hypergeo function
389  - xhypf12( ivec1(:) ,jmod )*(zvec1(:) - 1.0) ! XHYPF12 is tabulated
390  !
391  ztmp(:,jmod) = zchen_multi(:,jmod)/zrhodref(:)*zsmax(:)**xkhen_multi(jmod)*zzw2(:)
392  ENDWHERE
393  ENDDO
394 !
395 ! Compute the concentration of aerosols activated at this time step
396 ! as the difference between ZTMP and the aerosols already activated at t-dt (ZZW1)
397 !
398  DO jmod = 1, nmod_ccn ! iteration on mode number
399  zzw1(:) = 0.
400  zzw2(:) = 0.
401  zzw3(:) = 0.
402  !
403  WHERE( sum(ztmp(:,:),dim=2) .GT. 0.01e6/zrhodref(:) )
404  zzw1(:) = min( znft(:,jmod),max( ztmp(:,jmod)- znat(:,jmod) , 0.0 ) )
405  ENDWHERE
406  !
407  !* update the concentration of activated CCN = Na
408  !
409  pnat(:,:,:,jmod) = pnat(:,:,:,jmod) + zcldfr(:,:,:) * unpack( zzw1(:), mask=gnuct(:,:,:), field=0.0 )
410  !
411  !* update the concentration of free CCN = Nf
412  !
413  pnft(:,:,:,jmod) = pnft(:,:,:,jmod) - zcldfr(:,:,:) * unpack( zzw1(:), mask=gnuct(:,:,:), field=0.0 )
414  !
415  !* prepare to update the cloud water concentration
416  !
417  zzw6(:) = zzw6(:) + zzw1(:)
418  ENDDO
419 !
420 ! Output tendencies
421 !
422  zzw1(:)=0.
423  WHERE (zzw5(:)>0.0 .AND. zsmax(:)>0.0) ! ZZW1 is computed with ZSMAX [NO UNIT]
424  zzw1(:) = min(xcstdcrit*zzw6(:)/(((zzt(:)*zsmax(:))**3)*zrhodref(:)),1.e-5)
425  END WHERE
426 !
427  IF(PRESENT(ptot_rv_henu)) ptot_rv_henu(:,:,:) = 0.
428  IF (.NOT.lsubg_cond) THEN
429  zw(:,:,:) = min( unpack( zzw1(:),mask=gnuct(:,:,:),field=0.0 ),prvt(:,:,:) )
430  IF(PRESENT(ptot_rv_henu)) ptot_rv_henu(:,:,:) = zw(:,:,:)
431  ptht(:,:,:) = ptht(:,:,:) + zw(:,:,:) * (cst%XLVTT+(cst%XCPV-cst%XCL)*(pt(:,:,:)-cst%XTT))/ &
432  (pexnref(:,:,:)*(cst%XCPD+cst%XCPV*prvt(:,:,:)+cst%XCL*(prct(:,:,:)+prrt(:,:,:))))
433  prvt(:,:,:) = prvt(:,:,:) - zw(:,:,:)
434  prct(:,:,:) = prct(:,:,:) + zw(:,:,:)
435  pcct(:,:,:) = pcct(:,:,:) + unpack( zzw6(:),mask=gnuct(:,:,:),field=0. )
436  ELSE
437  zw(:,:,:) = min( zcldfr(:,:,:) * unpack( zzw1(:),mask=gnuct(:,:,:),field=0.0 ),prvt(:,:,:) )
438  pcct(:,:,:) = pcct(:,:,:) + zcldfr(:,:,:) * unpack( zzw6(:),mask=gnuct(:,:,:),field=0. )
439  END IF
440 !
441 !++cb-- A quoi servent ces 2 dernieres lignes ? variables locales, non sauvees, et ne servent pas
442 ! a calculer quoi que ce soit (fin de la routine)
443  zw(:,:,:) = unpack( 100.0*zsmax(:),mask=gnuct(:,:,:),field=0.0 )
444  zw2(:,:,:) = zcldfr(:,:,:) * unpack( zzw6(:),mask=gnuct(:,:,:),field=0.0 )
445 !
446 !
447 !-------------------------------------------------------------------------------
448 !
449 !
450 !* 5. Cleaning
451 ! -----------

627  xl =pzriddr(jl)
