30 USE modd_rain_ice_descr_n, ONLY: rain_ice_descr_t
31 USE modd_rain_ice_param_n, ONLY: rain_ice_param_t
32 !
33 IMPLICIT NONE
34 !
35 !* 0.1 Declarations of dummy arguments :
36 !
37 TYPE(cst_t), INTENT(IN) :: CST

49 REAL, DIMENSION(KSIZE), INTENT(IN) :: PRVT ! Water vapor m.r. at t
50 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PCIT ! Pristine ice n.c. at t
51 REAL, DIMENSION(KSIZE), INTENT(OUT) :: PRVHENI_MR ! Mixing ratio change due to the heterogeneous nucleation
52 REAL, DIMENSION(KSIZE, 4),INTENT(OUT) :: PBUF ! Buffer
53 LOGICAL, DIMENSION(KSIZE),INTENT(OUT) :: LDBUF ! Buffer
54 !
55 !* 0.2 declaration of local variables
56 !
57 !This routine will be included in a CONTAIN part of other routine
58 !For GPU source-to-source transformation, this routine cannot declare
59 !local arrays. We must use buffer (declared in the calling routine)
60 INTEGER :: JL
61 INTEGER, PARAMETER :: IW1=1, iw2=2, iusw=3, issi=4
62 !-------------------------------------------------------------------------------
63 !
64 !
65 !$mnh_expand_where(JL=1:KSIZE)
66 WHERE(odcompute(:))
67  ldbuf(:)=pt(:)<cst%XTT .AND. prvt(:)>iced%XRTMIN(1)
68 ELSEWHERE
69  ldbuf(:)=.false.
70 ENDWHERE
71 !$mnh_end_expand_where(JL=1:KSIZE)
72 
73 pbuf(:, iusw)=0.
74 pbuf(:, iw2)=0.
75 !$mnh_expand_where(JL=1:KSIZE)
76 WHERE(ldbuf(:))
77  pbuf(:, iw2)=alog(pt(:))
78  pbuf(:, iusw)=exp(cst%XALPW - cst%XBETAW/pt(:) - cst%XGAMW*pbuf(:, iw2)) ! es_w
79  pbuf(:, iw2)=exp(cst%XALPI - cst%XBETAI/pt(:) - cst%XGAMI*pbuf(:, iw2)) ! es_i
80 END WHERE
81 !$mnh_end_expand_where(JL=1:KSIZE)
82 
83 pbuf(:, issi)=0.
84 !$mnh_expand_where(JL=1:KSIZE)
85 WHERE(ldbuf(:))
86  pbuf(:, iw2)=min(ppabst(:)/2., pbuf(:, iw2)) ! safety limitation
87  pbuf(:, issi)=prvt(:)*(ppabst(:)-pbuf(:, iw2)) / (cst%XEPSILO*pbuf(:, iw2)) - 1.0
88  ! Supersaturation over ice
89  pbuf(:, iusw)=min(ppabst(:)/2., pbuf(:, iusw)) ! safety limitation
90  pbuf(:, iusw)=(pbuf(:, iusw)/pbuf(:, iw2))*((ppabst(:)-pbuf(:, iw2))/(ppabst(:)-pbuf(:, iusw))) - 1.0
91  ! Supersaturation of saturated water vapor over ice
92  pbuf(:, issi)=min(pbuf(:, issi), pbuf(:, iusw)) ! limitation of SSi according to SSw=0
93 END WHERE
94 !$mnh_end_expand_where(JL=1:KSIZE)
95 
96 pbuf(:, iw2)=0.
97 DO jl=1,ksize
98  IF(ldbuf(jl)) THEN
99  IF(pt(jl)<cst%XTT-5.0 .AND. pbuf(jl, issi)>0.0) THEN
100  pbuf(jl, iw2)=icep%XNU20*exp(icep%XALPHA2*pbuf(jl, issi)-icep%XBETA2)
101  ELSEIF(pt(jl)<=cst%XTT-2.0 .AND. pt(jl)>=cst%XTT-5.0 .AND. pbuf(jl, issi)>0.0) THEN
102  pbuf(jl, iw2)=max(icep%XNU20*exp(-icep%XBETA2 ), &
103  icep%XNU10*exp(-icep%XBETA1*(pt(jl)-cst%XTT))*(pbuf(jl, issi)/pbuf(jl, iusw))**icep%XALPHA1)
104  ENDIF
105  ENDIF
106 ENDDO
107 !$mnh_expand_where(JL=1:KSIZE)
108 WHERE(ldbuf(:))
109  pbuf(:, iw2)=pbuf(:, iw2)-pcit(:)
110  pbuf(:, iw2)=min(pbuf(:, iw2), 50.e3) ! limitation provisoire a 50 l^-1
111 END WHERE
112 !$mnh_end_expand_where(JL=1:KSIZE)
113 
114 prvheni_mr(:)=0.
115 !$mnh_expand_where(JL=1:KSIZE)
116 WHERE(ldbuf(:))
117  prvheni_mr(:)=max(pbuf(:, iw2), 0.0)*icep%XMNU0/prhodref(:)
118  prvheni_mr(:)=min(prvt(:), prvheni_mr(:))
119 END WHERE
120 !$mnh_end_expand_where(JL=1:KSIZE)
121 !Limitation due to 0 crossing of temperature
122 IF(parami%LFEEDBACKT) THEN
123  pbuf(:, iw1)=0.
124  !$mnh_expand_where(JL=1:KSIZE)
125  WHERE(ldbuf(:))
126  pbuf(:, iw1)=min(prvheni_mr(:), &
127  max(0., (cst%XTT/pexn(:)-ptht(:))/plsfact(:))) / &
128  max(prvheni_mr(:), 1.e-20)
129  END WHERE
130  prvheni_mr(:)=prvheni_mr(:)*pbuf(:, iw1)
131  pbuf(:, iw2)=pbuf(:, iw2)*pbuf(:, iw1)
132  !$mnh_end_expand_where(JL=1:KSIZE)
133 ENDIF
134 !$mnh_expand_where(JL=1:KSIZE)
135 WHERE(ldbuf(:))
136  pcit(:)=max(pbuf(:, iw2)+pcit(:), pcit(:))
137 END WHERE
138 !$mnh_end_expand_where(JL=1:KSIZE)
139 !
140 END SUBROUTINE ice4_nucleation
