21 !!**** *TKE_EPS_SOURCES* - routine to compute the sources of the turbulent
22 !! evolutive variables: TKE and its dissipation when it is taken into
23 !! account. The contribution to the heating of tke dissipation is computed.
24 !!
25 !! PURPOSE
26 !! -------
27 ! The purpose of this routine is to compute the sources necessary for
28 ! the evolution of the turbulent kinetic energy and its dissipation
29 ! if necessary.
30 !
31 !!** METHOD
32 !! ------
33 !! The vertical turbulent flux is computed in an off-centered
34 !! implicit scheme (a Crank-Nicholson type with coefficients different
35 !! than 0.5), which allows to vary the degree of implicitness of the
36 !! formulation.
37 !! In high resolution, the horizontal transport terms are also
38 !! calculated, but explicitly.
39 !! The evolution of the dissipation as a variable is made if
40 !! the parameter TURBN%CTURBLEN is set equal to KEPS. The same reasoning
41 !! made for TKE applies.
42 !!
43 !! EXTERNAL
44 !! --------
45 !! GX_U_M,GY_V_M,GZ_W_M
46 !! GX_M_U,GY_M_V : Cartesian vertical gradient operators
47 !!
48 !! MXF,MXM.MYF,MYM,MZF,MZM: Shuman functions (mean operators)
49 !! DZF : Shuman functions (difference operators)
50 !!
51 !! SUBROUTINE TRIDIAG : to solve an implicit temporal scheme
52 !!
53 !!
54 !! IMPLICIT ARGUMENTS
55 !! ------------------
56 !! Module MODD_CST : contains physical constants
57 !!
58 !! XG : gravity constant
59 !!
60 !! Module MODD_CTURB: contains the set of constants for
61 !! the turbulence scheme
62 !!
63 !! CSTURB%XCET,TURBN%XCED : transport and dissipation cts. for the TKE
64 !! XCDP,XCDD,XCDT: constants from the parameterization of
65 !! the K-epsilon equation
66 !! TURBN%XTKEMIN,XEPSMIN : minimum values for the TKE and its
67 !! dissipation
68 !!
69 !! Module MODD_PARAMETERS:
70 !!
71 !! JPVEXT_TURB
72 !! Module MODD_BUDGET:
73 !! NBUMOD : model in which budget is calculated
74 !! CBUTYPE : type of desired budget
75 !! 'CART' for cartesian box configuration
76 !! 'MASK' for budget zone defined by a mask
77 !! 'NONE' ' for no budget
78 !! LBU_RTKE : logical for budget of RTKE (turbulent kinetic energy)
79 !! .TRUE. = budget of RTKE
80 !! .FALSE. = no budget of RTKE
81 !!
82 !!
83 !! REFERENCE
84 !! ---------
85 !! Book 2 of documentation (routine TKE_EPS_SOURCES)
86 !! Book 1 of documentation (Chapter: Turbulence)

156 !* 0.1 declarations of arguments
157 !
158 !
159 TYPE(dimphyex_t), INTENT(IN) :: D
160 TYPE(cst_t), INTENT(IN) :: CST
161 TYPE(csturb_t), INTENT(IN) :: CSTURB
162 TYPE(tbudgetconf_t), INTENT(IN) :: BUCONF
163 TYPE(turb_t), INTENT(IN) :: TURBN
164 TYPE(tles_t), INTENT(INOUT):: TLES
165 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTKEM ! TKE at t-deltat
166 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PLM ! mixing length
167 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PLEPS ! dissipative length
168 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT):: PDP ! Dyn. prod. of TKE
169 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTRH
170 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODJ ! density * grid volume
171 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PDXX,PDYY,PDZZ,PDZX,PDZY ! metric coefficients
172 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PZZ ! physical height w-pt
173 REAL, INTENT(IN) :: PTSTEP ! Time step
174 REAL, INTENT(IN) :: PEXPL ! Coef. temporal. disc.
175 TYPE(tfiledata), INTENT(IN) :: TPFILE ! Output file
176 LOGICAL, INTENT(IN) :: ODIAG_IN_RUN ! switch to activate online diagnostics (mesonh)
177 LOGICAL, INTENT(IN) :: OOCEAN ! switch for Ocean model version
178 REAL, DIMENSION(D%NIJT), INTENT(IN) :: PSFUM,PSFVM ! momentum sfc flux
179 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTP ! Ther. prod. of TKE
180 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT):: PRTKES ! RHOD * Jacobian * TKE at t+deltat
181 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT):: PRTHLS ! Source of Theta_l
182 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PCOEF_DISS ! 1/(Cph*Exner)
183 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PTDIFF ! Diffusion TKE term
184 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PTDISS ! Dissipation TKE term
185 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRTKEMS ! Advection source
186 TYPE(tbudgetdata), DIMENSION(KBUDGETS), INTENT(INOUT) :: TBUDGETS
187 INTEGER, INTENT(IN) :: KBUDGETS
188 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT), OPTIONAL :: PEDR ! EDR
189 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT), OPTIONAL :: PTR ! Transport prod. of TKE
190 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT), OPTIONAL :: PDISS ! Dissipation of TKE
191 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(INOUT), OPTIONAL :: PCURRENT_TKE_DISS ! if ODIAG_IN_RUN in mesonh
192 !
193 !
194 !
195 !* 0.2 declaration of local variables
196 !
197 REAL, DIMENSION(D%NIJT,D%NKT) :: &
198  ZA, & ! under diagonal elements of the tri-diagonal matrix involved
199  ! in the temporal implicit scheme
200  zres, & ! treated variable at t+ deltat when the turbu-
201  ! lence is the only source of evolution added to the ones
202  ! considered in ZSOURCE. This variable is also used to
203  ! temporarily store some diagnostics stored in FM file
204  zflx, & ! horizontal or vertical flux of the treated variable
205  zsource, & ! source of evolution for the treated variable
206  zkeff, & ! effectif diffusion coeff = LT * SQRT( TKE )
207  ztr, & ! Transport term
208  zmwork1,zmwork2,& ! working var. for MZM/MZF operators (array syntax)
209  zdwork1,zdwork2,& ! working var. for DZM/DZF operators (array syntax)
210  zw ! working array
211 
212 LOGICAL,DIMENSION(D%NIJT,D%NKT) :: GTKENEG
213  ! 3D mask .T. if TKE < CSTURB%XTKEMIN
214 INTEGER :: IIJB,IIJE,IKB,IKE,IKT,IKA,IKL ! Index value for the mass points of the domain
215 !
216 TYPE(list_ll), POINTER :: TZFIELDDISS_ll ! list of fields to exchange
217 INTEGER :: IINFO_ll ! return code of parallel routine
218 TYPE(tfieldmetadata) :: TZFIELD
219 REAL(KIND=jphook) :: zhook_handle
220 INTEGER :: JIJ,JK
221 !
222 !----------------------------------------------------------------------------
223 NULLIFY(tzfielddiss_ll)
224 !
225 !* 1. PRELIMINARY COMPUTATIONS
226 ! ------------------------
227 !
228 IF (lhook) CALL dr_hook('TKE_EPS_SOURCES',0,zhook_handle)
229 !
230 ikb=d%NKB
231 ike=d%NKE
232 iijb=d%NIJB
233 iije=d%NIJE
234 ikt=d%NKT
235 ika=d%NKA
236 ikl=d%NKL
237 !
238 ! compute the effective diffusion coefficient at the mass point
239 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
240 zkeff(iijb:iije,1:ikt) = plm(iijb:iije,1:ikt) * sqrt(ptkem(iijb:iije,1:ikt))
241 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
242 !
243 !----------------------------------------------------------------------------
244 !
245 !* 2. TKE EQUATION
246 ! ------------
247 !
248 !* 2.1 Horizontal turbulent explicit transport
249 !
250 !
251 ! Complete the sources of TKE with the horizontal turbulent explicit transport
252 !
253 IF (turbn%CTURBDIM=='3DIM') THEN
254  ztr(iijb:iije,1:ikt)=ptrh(iijb:iije,1:ikt)
255 ELSE
256  ztr(iijb:iije,1:ikt)=0.
257 END IF
258 !
259 !
260 !* 2.2 Explicit TKE sources except horizontal turbulent transport
261 !
262 ! extrapolate the dynamic production with a 1/Z law from its value at the
263 IF (oocean) THEN
264  ! W(IKE) value stored in PDP(IKE) to the mass localization of tke(IKE)
265  !$mnh_expand_array(JIJ=IIJB:IIJE)
266  pdp(iijb:iije,ike) = pdp(iijb:iije,ike) * (1. + pdzz(iijb:iije,ike)/pdzz(iijb:iije,ike+1))
267  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
268 ELSE
269  ! W(IKB+1) value stored in PDP(IKB) to the mass localization tke(IKB)
270  !$mnh_expand_array(JIJ=IIJB:IIJE)
271  pdp(iijb:iije,ikb) = pdp(iijb:iije,ikb) * (1. + pdzz(iijb:iije,ikb+ikl)/pdzz(iijb:iije,ikb))
272  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
273 END IF
274 !
275 ! Compute the source terms for TKE: ( ADVECtion + NUMerical DIFFusion + ..)
276 ! + (Dynamical Production) + (Thermal Production) - (dissipation)
277 !
278 CALL mzm_phy(d,zkeff,zmwork1)
279 CALL mzm_phy(d,prhodj,zmwork2)
280 !
281 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
282 zflx(iijb:iije,1:ikt) = turbn%XCED * sqrt(ptkem(iijb:iije,1:ikt)) / pleps(iijb:iije,1:ikt)
283 zsource(iijb:iije,1:ikt) = ( prtkes(iijb:iije,1:ikt) + prtkems(iijb:iije,1:ikt) ) &
284  / prhodj(iijb:iije,1:ikt) - ptkem(iijb:iije,1:ikt) / ptstep &
285  + pdp(iijb:iije,1:ikt) + ptp(iijb:iije,1:ikt) + ztr(iijb:iije,1:ikt) &
286  - pexpl * zflx(iijb:iije,1:ikt) * ptkem(iijb:iije,1:ikt)
287 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
288 !
289 !* 2.2 implicit vertical TKE transport
290 !
291 !
292 ! To add here in ZSOURCE surface flux of TKE
293 !(assumed to be 0 for ATM,
294 IF (oocean) THEN
295  !for ocean:wave breaking simple/very rough param wE = 100 Ustar**3 where ustar is the Tau_atmi/rhocea
296  !$mnh_expand_array(JIJ=IIJB:IIJE)
297  zsource(iijb:iije,ike)=zsource(iijb:iije,ike)-1.e2*((psfum(iijb:iije)**2 + psfvm(iijb:iije)**2)**1.5) /pdzz(iijb:iije,ike)
298  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
299 END IF
300 ! Compute the vector giving the elements just under the diagonal for the
301 ! matrix inverted in TRIDIAG
302 !
303 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
304 za(iijb:iije,1:ikt) = - ptstep * csturb%XCET * zmwork1(iijb:iije,1:ikt) &
305  * zmwork2(iijb:iije,1:ikt) / pdzz(iijb:iije,1:ikt)**2
306 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
307 !
308 ! Compute TKE at time t+deltat: ( stored in ZRES )
309 !
310 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
311 zw(:,:)=ptstep*zflx(:,:)
312 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
313 CALL tridiag_tke(d,ptkem,za,ptstep,pexpl,turbn%XIMPL,prhodj,zsource,zw,zres)
314 CALL get_halo_phy(d,zres)
315 !
316 !* diagnose the dissipation
317 !
318 IF (odiag_in_run) THEN
319  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
320  pcurrent_tke_diss(iijb:iije,1:ikt) = zflx(iijb:iije,1:ikt) * ptkem(iijb:iije,1:ikt) &
321  *(pexpl*ptkem(iijb:iije,1:ikt) + turbn%XIMPL*zres(iijb:iije,1:ikt))
322  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
323 !
324  CALL add3dfield_ll_phy(d, tzfielddiss_ll, pcurrent_tke_diss, 'TKE_EPS_SOURCES::PCURRENT_TKE_DISS' )
325  CALL update_halo_ll(tzfielddiss_ll,iinfo_ll)
326  CALL cleanlist_ll(tzfielddiss_ll)
327 ENDIF
328 !
329 ! TKE must be greater than its minimum value
330 ! CL : Now done at the end of the time step in ADVECTION_METSV for MesoNH
331 IF(turbn%LTKEMINTURB) THEN
332  !$mnh_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
333  gtkeneg(iijb:iije,1:ikt) = zres(iijb:iije,1:ikt) <= turbn%XTKEMIN
334  WHERE ( gtkeneg(iijb:iije,1:ikt) )
335  zres(iijb:iije,1:ikt) = turbn%XTKEMIN
336  END WHERE
337  !$mnh_end_expand_where(JIJ=IIJB:IIJE,JK=1:IKT)
338 END IF
339 !
340 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
341 ptdiss(iijb:iije,1:ikt) = - zflx(iijb:iije,1:ikt)*(pexpl*ptkem(iijb:iije,1:ikt) &
342  + turbn%XIMPL*zres(iijb:iije,1:ikt))
343 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
344 !
345 IF ( tles%LLES_CALL .OR. &
346  (turbn%LTURB_DIAG .AND. tpfile%LOPENED) ) THEN
347 !
348 ! Compute the cartesian vertical flux of TKE in ZFLX
349 !
350  CALL mzm_phy(d,zkeff,zmwork1)
351  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
352  zdwork1(iijb:iije,1:ikt) = turbn%XIMPL * zres(iijb:iije,1:ikt) + pexpl * ptkem(iijb:iije,1:ikt)
353  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
354  CALL dzm_phy(d,zdwork1,zdwork2)
355  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
356  zflx(iijb:iije,1:ikt) = - csturb%XCET * zmwork1(iijb:iije,1:ikt) &
357  * zdwork2(iijb:iije,1:ikt) / pdzz(iijb:iije,1:ikt)
358  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
359 !
360  zflx(iijb:iije,ikb) = 0.
361  zflx(iijb:iije,ika) = 0.
362 !
363 ! Compute the whole turbulent TRansport of TKE:
364 !
365  CALL mzm_phy(d,prhodj,zmwork1)
366  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
367  zmwork2(iijb:iije,1:ikt) = zmwork1(iijb:iije,1:ikt) * zflx(iijb:iije,1:ikt) &
368  / pdzz(iijb:iije,1:ikt)
369  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
370  CALL dzf_phy(d,zmwork2,zdwork1)
371  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
372  ztr(iijb:iije,1:ikt)= ztr(iijb:iije,1:ikt) - zdwork1(iijb:iije,1:ikt) &
373  /prhodj(iijb:iije,1:ikt)
374  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
375 !
376 ! Storage in the LES configuration
377 !
378  IF (tles%LLES_CALL) THEN
379  CALL mzf_phy(d,zflx,zmwork1)
380  CALL les_mean_subgrid_phy(d,tles,zmwork1, tles%X_LES_SUBGRID_WTke )
381  CALL les_mean_subgrid_phy(d,tles, -ztr, tles%X_LES_SUBGRID_ddz_WTke )
382  END IF
383 !
384 END IF
385 !
386 !* 2.4 stores the explicit sources for budget purposes
387 !
388 IF (buconf%LBUDGET_TKE) THEN
389  ! Dynamical production
390  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
391  zmwork1(iijb:iije,1:ikt) = pdp(iijb:iije,1:ikt) * prhodj(iijb:iije,1:ikt)
392  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
393  CALL budget_store_add_phy(d, tbudgets(nbudget_tke), 'DP', zmwork1)
394  !
395  ! Thermal production
396  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
397  zmwork1(iijb:iije,1:ikt) = ptp(iijb:iije,1:ikt) * prhodj(iijb:iije,1:ikt)
398  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
399  CALL budget_store_add_phy(d, tbudgets(nbudget_tke), 'TP', zmwork1)
400  !
401  ! Dissipation
402  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
403  zmwork1(iijb:iije,1:ikt) = -turbn%XCED * sqrt(ptkem(iijb:iije,1:ikt))/pleps(iijb:iije,1:ikt) * &
404  (pexpl*ptkem(iijb:iije,1:ikt) + turbn%XIMPL*zres(iijb:iije,1:ikt))*prhodj(iijb:iije,1:ikt)
405  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
406  CALL budget_store_add_phy(d, tbudgets(nbudget_tke), 'DISS',zmwork1)
407 END IF
408 !
409 !* 2.5 computes the final RTKE and stores the whole turbulent transport
410 ! with the removal of the advection part for MesoNH
411 
412 !Should be in IF LBUDGET_TKE only. Was removed out for a correct comput. of PTDIFF in case of LBUDGET_TKE=F in AROME
413 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
414 prtkes(iijb:iije,1:ikt) = prtkes(iijb:iije,1:ikt) + prhodj(iijb:iije,1:ikt) * &
415  ( pdp(iijb:iije,1:ikt) + ptp(iijb:iije,1:ikt) &
