20 !
21 !!**** *TRIDIAG_W* - routine to solve a time implicit scheme
22 !!
23 !!
24 !! PURPOSE
25 !! -------
26 ! The purpose of this routine is to give a field PVARP at t+1, by
27 ! solving an implicit TRIDIAGonal system obtained by the
28 ! discretization of the vertical turbulent diffusion.
29 ! The function of F(dT/dz) must have been linearized.
30 ! PVARP is localized at a flux point.
31 !
32 !!** METHOD
33 !! ------
34 !!
35 !! [W(+) - W(-)]/Dt = -d{ F + dF/d(dW/dz) * [dW/dz(+)-dW/dz(-)] }/dz
36 !!
37 !! It is discretized as follows:
38 !!
39 !! (PRHODJ(k)+PRHODJ(k-1))/2.*PVARP(k)/PTSTEP
40 !! =
41 !! (PRHODJ(k)+PRHODJ(k-1))/2.*PVARM(k)/PTSTEP
42 !! - PRHODJ(k) * PF(k )/PMZF_PDZZ(k )
43 !! + PRHODJ(k-1) * PF(k-1)/PMZF_PDZZ(k-1)
44 !! - PRHODJ(k) * PDFDDWDZ(k) * PVARP(k+1)/PMZF_DZZ(k)**2
45 !! + PRHODJ(k) * PDFDDWDZ(k) * PVARP(k) /PMZF_DZZ(k)**2
46 !! + PRHODJ(k-1) * PDFDDWDZ(k-1) * PVARP(k) /PMZF_DZZ(k-1)**2
47 !! - PRHODJ(k-1) * PDFDDWDZ(k-1) * PVARP(k-1)/PMZF_DZZ(k-1)**2
48 !! + PRHODJ(k) * PDFDDWDZ(k) * PVARM(k+1)/PMZF_DZZ(k)**2
49 !! - PRHODJ(k) * PDFDDWDZ(k) * PVARM(k) /PMZF_DZZ(k)**2
50 !! - PRHODJ(k-1) * PDFDDWDZ(k-1) * PVARM(k) /PMZF_DZZ(k-1)**2
51 !! + PRHODJ(k-1) * PDFDDWDZ(k-1) * PVARM(k-1)/PMZF_DZZ(k-1)**2
52 !!
53 !!
54 !! The system to solve is:
55 !!
56 !! A*PVARP(k-1) + B*PVARP(k) + C*PVARP(k+1) = Y(k)
57 !!
58 !!
59 !! The RHS of the linear system in PVARP writes:
60 !!
61 !! y(k) = (PRHODJ(k)+PRHODJ(k-1))/2.*PVARM(k)/PTSTEP
62 !! - PRHODJ(k) * PF(k )/PMZF_PDZZ(k )
63 !! + PRHODJ(k-1) * PF(k-1)/PMZF_PDZZ(k-1)
64 !! + PRHODJ(k) * PDFDDWDZ(k) * PVARM(k+1)/PMZF_DZZ(k)**2
65 !! - PRHODJ(k) * PDFDDWDZ(k) * PVARM(k) /PMZF_DZZ(k)**2
66 !! - PRHODJ(k-1) * PDFDDWDZ(k-1) * PVARM(k) /PMZF_DZZ(k-1)**2
67 !! + PRHODJ(k-1) * PDFDDWDZ(k-1) * PVARM(k-1)/PMZF_DZZ(k-1)**2
68 !!
69 !!
70 !! Then, the classical TRIDIAGonal algorithm is used to invert the
71 !! implicit operator. Its matrix is given by:
72 !!
73 !! ( b(ikb) c(ikb) 0 0 0 0 0 0 )
74 !! ( 0 a(ikb+1) b(ikb+1) c(ikb+1) 0 ... 0 0 0 )
75 !! ( 0 0 a(ikb+2) b(ikb+2) c(ikb+2). 0 0 0 )
76 !! .......................................................................
77 !! ( 0 ... 0 a(k) b(k) c(k) 0 ... 0 0 )
78 !! .......................................................................
79 !! ( 0 0 0 0 0 ...a(ike-1) b(ike-1) c(ike-1))
80 !! ( 0 0 0 0 0 ... 0 a(ike) b(ike) )
81 !!
82 !! ikb and ike represent the first and the last inner mass levels of the
83 !! model. The coefficients are:
84 !!
85 !! a(k) = + PRHODJ(k-1) * PDFDDWDZ(k-1)/PMZF_DZZ(k-1)**2
86 !! b(k) = (PRHODJ(k)+PRHODJ(k-1))/2. / PTSTEP
87 !! - PRHODJ(k) * PDFDDWDZ(k) /PMZF_DZZ(k)**2
88 !! - PRHODJ(k-1) * PDFDDWDZ(k-1)/PMZF_DZZ(k-1)**2
89 !! c(k) = + PRHODJ(k) * PDFDDWDZ(k)/PMZF_DZZ(k)**2
90 !!
91 !! for all k /= ikb or ike
92 !!
93 !! with the boundary conditions:
94 !! PVARP(ikb-1) = PVARP(ikb)
95 !! PVARP(ike+1) = 0.
96 !!
97 !! This induces:
98 !!
99 !! b(ikb) = (PRHODJ(ikb)+PRHODJ(ikb-1))/2. / PTSTEP
100 !! - PRHODJ(ikb) * PDFDDWDZ(ikb) /PMZF_DZZ(ikb)**2
101 !! c(ikb) = + PRHODJ(ikb) * PDFDDWDZ(ikb)/PMZF_DZZ(ikb)**2
102 !!
103 !! b(ike) = (PRHODJ(ike)+PRHODJ(ike-1))/2. / PTSTEP
104 !! - PRHODJ(ike-1) * PDFDDWDZ(ike-1)/PMZF_DZZ(ike-1)**2
105 !! - PRHODJ(ike ) * PDFDDWDZ(ike )/PMZF_DZZ(ike )**2
106 !! a(ike) = + PRHODJ(ike-1) * PDFDDWDZ(ike-1)/PMZF_DZZ(ike-1)**2
107 !!
108 !!
109 !! EXTERNAL
110 !! --------
111 !!
112 !! NONE
113 !!
114 !! IMPLICIT ARGUMENTS
115 !! ------------------
116 !!
117 !! REFERENCE
118 !! ---------
119 !! Press et al: Numerical recipes (1986) Cambridge Univ. Press
120 !!
121 !! AUTHOR
122 !! ------
123 !! V. Masson * Meteo-France *
124 !!
125 !! MODIFICATIONS
126 !! -------------
127 !! Original 04/2011 (from tridiag_thermo.f90)
128 !! 03/2014 modification of upper boundary condition
129 !! ---------------------------------------------------------------------
130 !
131 !* 0. DECLARATIONS
132 !
133 USE modd_parameters, ONLY : jpvext
134 !
135 USE modi_shuman
136 !
137 IMPLICIT NONE
138 !
139 !
140 !* 0.1 declarations of arguments
141 !
142 REAL, DIMENSION(:,:,:), INTENT(IN) :: PVARM ! variable at t-1 at flux point
143 REAL, DIMENSION(:,:,:), INTENT(IN) :: PF ! flux in dT/dt=-dF/dz at mass point
144 REAL, DIMENSION(:,:,:), INTENT(IN) :: PDFDDWDZ! dF/d(dW/dz) at mass point
145 REAL, INTENT(IN) :: PTSTEP ! Double time step
146 REAL, DIMENSION(:,:,:), INTENT(IN) :: PMZF_DZZ! Dz at mass point
147 REAL, DIMENSION(:,:,:), INTENT(IN) :: PRHODJ ! (dry rho)*J at mass point
148 !
149 REAL, DIMENSION(:,:,:), INTENT(OUT):: PVARP ! variable at t+1 at flux point
150 !
151 !
152 !* 0.2 declarations of local variables
153 !
154 REAL, DIMENSION(SIZE(PVARM,1),SIZE(PVARM,2),SIZE(PVARM,3)) :: ZRHODJ_DFDDWDZ_O_DZ2
155 REAL, DIMENSION(SIZE(PVARM,1),SIZE(PVARM,2),SIZE(PVARM,3)) :: ZMZM_RHODJ
156 REAL, DIMENSION(SIZE(PVARM,1),SIZE(PVARM,2),SIZE(PVARM,3)) :: ZA, ZB, ZC
157 REAL, DIMENSION(SIZE(PVARM,1),SIZE(PVARM,2),SIZE(PVARM,3)) :: ZY ,ZGAM
158  ! RHS of the equation, 3D work array
159 REAL, DIMENSION(SIZE(PVARM,1),SIZE(PVARM,2)) :: ZBET
160  ! 2D work array
161 INTEGER :: JK ! loop counter
162 INTEGER :: IKB,IKE ! inner vertical limits
163 !
164 ! ---------------------------------------------------------------------------
165 !
166 !* 1. Preliminaries
167 ! -------------
168 !
169 ikb=1+jpvext
170 ike=SIZE(pvarm,3)-jpvext
171 !
172 zmzm_rhodj = mzm(prhodj)
173 zrhodj_dfddwdz_o_dz2 = prhodj*pdfddwdz/pmzf_dzz**2
174 !
175 za=0.
176 zb=0.
177 zc=0.
178 zy=0.
179 !
180 !
181 !* 2. COMPUTE THE RIGHT HAND SIDE
182 ! ---------------------------
183 !
184 !! y(k) = (PRHODJ(k)+PRHODJ(k-1))/2.*PVARM(k)/PTSTEP
185 !! - PRHODJ(k) * PF(k )/PMZF_PDZZ(k )
186 !! + PRHODJ(k-1) * PF(k-1)/PMZF_PDZZ(k-1)
187 !! + PRHODJ(k) * PDFDDWDZ(k) * PVARM(k+1)/PMZF_DZZ(k)**2
188 !! - PRHODJ(k) * PDFDDWDZ(k) * PVARM(k) /PMZF_DZZ(k)**2
189 !! - PRHODJ(k-1) * PDFDDWDZ(k-1) * PVARM(k) /PMZF_DZZ(k-1)**2
190 !! + PRHODJ(k-1) * PDFDDWDZ(k-1) * PVARM(k-1)/PMZF_DZZ(k-1)**2
191 !
192 zy(:,:,ikb) = zmzm_rhodj(:,:,ikb)*pvarm(:,:,ikb)/ptstep &
193  - prhodj(:,:,ikb ) * pf(:,:,ikb )/pmzf_dzz(:,:,ikb ) &
194  + prhodj(:,:,ikb-1) * pf(:,:,ikb-1)/pmzf_dzz(:,:,ikb-1) &
195  + zrhodj_dfddwdz_o_dz2(:,:,ikb) * pvarm(:,:,ikb+1)&
196  - zrhodj_dfddwdz_o_dz2(:,:,ikb) * pvarm(:,:,ikb )
197 !
198  zy(:,:,ikb+1:ike-1) = zmzm_rhodj(:,:,ikb+1:ike-1)*pvarm(:,:,ikb+1:ike-1)/ptstep &
199  - prhodj(:,:,ikb+1:ike-1 ) * pf(:,:,ikb+1:ike-1 )/pmzf_dzz(:,:,ikb+1:ike-1 ) &
200  + prhodj(:,:,ikb:ike-2) * pf(:,:,ikb:ike-2)/pmzf_dzz(:,:,ikb:ike-2) &
201  + zrhodj_dfddwdz_o_dz2(:,:,ikb+1:ike-1 ) * pvarm(:,:,ikb+2:ike) &
202  - zrhodj_dfddwdz_o_dz2(:,:,ikb+1:ike-1 ) * pvarm(:,:,ikb+1:ike-1 ) &
203  - zrhodj_dfddwdz_o_dz2(:,:,ikb:ike-2) * pvarm(:,:,ikb+1:ike-1 ) &
204  + zrhodj_dfddwdz_o_dz2(:,:,ikb:ike-2) * pvarm(:,:,ikb:ike-2)
205 !
206 zy(:,:,ike) = zmzm_rhodj(:,:,ike)*pvarm(:,:,ike)/ptstep &
207  - prhodj(:,:,ike ) * pf(:,:,ike )/pmzf_dzz(:,:,ike ) &
208  + prhodj(:,:,ike-1) * pf(:,:,ike-1)/pmzf_dzz(:,:,ike-1) &
209  - zrhodj_dfddwdz_o_dz2(:,:,ike ) * pvarm(:,:,ike ) &
210  - zrhodj_dfddwdz_o_dz2(:,:,ike-1) * pvarm(:,:,ike ) &
211  + zrhodj_dfddwdz_o_dz2(:,:,ike-1) * pvarm(:,:,ike-1)
212 !
213 !
214 !* 3. INVERSION OF THE TRIDIAGONAL SYSTEM
215 ! -----------------------------------
216 !
217 !
218 !* 3.1 arrays A, B, C
219 ! --------------
220 !
221 !! a(k) = + PRHODJ(k-1) * PDFDDWDZ(k-1)/PMZF_DZZ(k-1)**2
222 !! b(k) = (PRHODJ(k)+PRHODJ(k-1))/2. / PTSTEP
223 !! - PRHODJ(k) * PDFDDWDZ(k) /PMZF_DZZ(k)**2
224 !! - PRHODJ(k-1) * PDFDDWDZ(k-1)/PMZF_DZZ(k-1)**2
225 !! c(k) = + PRHODJ(k) * PDFDDWDZ(k)/PMZF_DZZ(k)**2
226 !
227  zb(:,:,ikb) = zmzm_rhodj(:,:,ikb)/ptstep &
228  - zrhodj_dfddwdz_o_dz2(:,:,ikb)
229  zc(:,:,ikb) = zrhodj_dfddwdz_o_dz2(:,:,ikb)
230 
231  za(:,:,ikb+1:ike-1) = zrhodj_dfddwdz_o_dz2(:,:,ikb:ike-2)
232  zb(:,:,ikb+1:ike-1) = zmzm_rhodj(:,:,ikb+1:ike-1)/ptstep &
233  - zrhodj_dfddwdz_o_dz2(:,:,ikb+1:ike-1 ) &
234  - zrhodj_dfddwdz_o_dz2(:,:,ikb:ike-2)
235  zc(:,:,ikb+1:ike-1) = zrhodj_dfddwdz_o_dz2(:,:,ikb+1:ike-1 )
236 
237  za(:,:,ike) = zrhodj_dfddwdz_o_dz2(:,:,ike-1)
238  zb(:,:,ike) = zmzm_rhodj(:,:,ike)/ptstep &
239  - zrhodj_dfddwdz_o_dz2(:,:,ike ) &
240  - zrhodj_dfddwdz_o_dz2(:,:,ike-1)
241 !
242 !* 3.2 going up
243 ! --------
244 !
245  zbet(:,:) = zb(:,:,ikb) ! bet = b(ikb)
246  pvarp(:,:,ikb) = zy(:,:,ikb) / zbet(:,:)
247 
248  !
249  DO jk = ikb+1,ike-1
250  zgam(:,:,jk) = zc(:,:,jk-1) / zbet(:,:)
251  ! gam(k) = c(k-1) / bet
252  zbet(:,:) = zb(:,:,jk) - za(:,:,jk) * zgam(:,:,jk)
253  ! bet = b(k) - a(k)* gam(k)
254  pvarp(:,:,jk)= ( zy(:,:,jk) - za(:,:,jk) * pvarp(:,:,jk-1) ) / zbet(:,:)
255  ! res(k) = (y(k) -a(k)*res(k-1))/ bet
256  END DO
257  ! special treatment for the last level
258  zgam(:,:,ike) = zc(:,:,ike-1) / zbet(:,:)
259  ! gam(k) = c(k-1) / bet
260  zbet(:,:) = zb(:,:,ike) - za(:,:,ike) * zgam(:,:,ike)
261  ! bet = b(k) - a(k)* gam(k)
262  pvarp(:,:,ike)= ( zy(:,:,ike) - za(:,:,ike) * pvarp(:,:,ike-1) ) / zbet(:,:)
263  ! res(k) = (y(k) -a(k)*res(k-1))/ bet
264 !
265 !* 3.3 going down
266 ! ----------
267 !
268  DO jk = ike-1,ikb,-1
269  pvarp(:,:,jk) = pvarp(:,:,jk) - zgam(:,:,jk+1) * pvarp(:,:,jk+1)
270  END DO
271 !
272 !
273 !* 4. FILL THE UPPER AND LOWER EXTERNAL VALUES
274 ! ----------------------------------------
275 !
276 pvarp(:,:,ikb-1)=pvarp(:,:,ikb)
277 pvarp(:,:,ike+1)=0.
278 !
279 !-------------------------------------------------------------------------------
280 !
281 END SUBROUTINE tridiag_w
