83 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRWETG ! Graupel wet growth
84 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSWETG ! Graupel wet growth
85 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRCDRYG ! Graupel dry growth
86 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRIDRYG ! Graupel dry growth
87 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRRDRYG ! Graupel dry growth
88 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRSDRYG ! Graupel dry growth

90 REAL, DIMENSION(KPROMA), INTENT(OUT) :: PRWETGH_MR ! Conversion of graupel into hail, mr change
91 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRGMLTR ! Melting of the graupel
92 REAL, DIMENSION(KPROMA, 8), INTENT(INOUT) :: PRG_TEND ! Individual tendencies
93 !
94 !* 0.2 declaration of local variables
95 !
96 INTEGER, PARAMETER :: IRCDRYG=1, iridryg=2, iriwetg=3, irsdryg=4, irswetg=5, irrdryg=6, &
97  & ifreez1=7, ifreez2=8
98 LOGICAL, DIMENSION(KPROMA) :: GDRY, LLDRYG
99 INTEGER :: IGDRY
100 REAL, DIMENSION(KPROMA) :: ZBUF1, ZBUF2, ZBUF3
101 INTEGER, DIMENSION(KPROMA) :: IBUF1, IBUF2, IBUF3
102 REAL, DIMENSION(KPROMA) :: ZZW, &
103  ZRDRYG_INIT, & !Initial dry growth rate of the graupeln
104  ZRWETG_INIT !Initial wet growth rate of the graupeln
105 REAL :: ZZW0D
106 INTEGER :: JL
107 
108 REAL(KIND=jphook) :: zhook_handle
109 !-------------------------------------------------------------------------------
110 !
111 IF (lhook) CALL dr_hook('ICE4_FAST_RG', 0, zhook_handle)
112 !
113 !-------------------------------------------------------------------------------
114 !
115 !* 6.1 rain contact freezing
116 !
117 DO jl=1, ksize
118  IF(prit(jl)>iced%XRTMIN(4) .AND. prrt(jl)>iced%XRTMIN(3) .AND. ldcompute(jl)) THEN
119  IF(.NOT. ldsoft) THEN
120  pricfrrg(jl) = icep%XICFRR*prit(jl) & ! RICFRRG
121  *plbdar(jl)**icep%XEXICFRR &
122  *prhodref(jl)**(-iced%XCEXVT)
123  prrcfrig(jl) = icep%XRCFRI*pcit(jl) & ! RRCFRIG
124  * plbdar(jl)**icep%XEXRCFRI &
125  * prhodref(jl)**(-iced%XCEXVT-1.)
126  IF(parami%LCRFLIMIT) THEN
127  !Comparison between heat to be released (to freeze rain) and heat sink (rain and ice temperature change)
128  !ZZW0D is the proportion of process that can take place
129  zzw0d=max(0., min(1., (pricfrrg(jl)*cst%XCI+prrcfrig(jl)*cst%XCL)*(cst%XTT-pt(jl)) / &
130  max(1.e-20, cst%XLVTT*prrcfrig(jl))))
131  prrcfrig(jl) = zzw0d * prrcfrig(jl) !Part of rain that can be freezed
132  pricfrr(jl) = (1.-zzw0d) * pricfrrg(jl) !Part of collected pristine ice converted to rain
133  pricfrrg(jl) = zzw0d * pricfrrg(jl) !Part of collected pristine ice that lead to graupel
134  ELSE
135  pricfrr(jl) = 0.
136  ENDIF
137  ENDIF
138  ELSE
139  pricfrrg(jl)=0.
140  prrcfrig(jl)=0.
141  pricfrr(jl)=0.
142  ENDIF
143 ENDDO
144 !
145 !
146 !* 6.3 compute the graupel growth
147 !
148 ! Wet and dry collection of rc and ri on graupel
149 DO jl=1, ksize
150  IF(prgt(jl)>iced%XRTMIN(6) .AND. prct(jl)>iced%XRTMIN(2) .AND. ldcompute(jl)) THEN
151  IF(.NOT. ldsoft) THEN
152  prg_tend(jl, ircdryg)=plbdag(jl)**(iced%XCXG-iced%XDG-2.) * prhodref(jl)**(-iced%XCEXVT)
153  prg_tend(jl, ircdryg)=icep%XFCDRYG * prct(jl) * prg_tend(jl, ircdryg)
154  ENDIF
155  ELSE
156  prg_tend(jl, ircdryg)=0.
157  ENDIF
158 
159  IF(prgt(jl)>iced%XRTMIN(6) .AND. prit(jl)>iced%XRTMIN(4) .AND. ldcompute(jl)) THEN
160  IF(.NOT. ldsoft) THEN
161  prg_tend(jl, iridryg)=plbdag(jl)**(iced%XCXG-iced%XDG-2.) * prhodref(jl)**(-iced%XCEXVT)
162  prg_tend(jl, iridryg)=icep%XFIDRYG*exp(icep%XCOLEXIG*(pt(jl)-cst%XTT))*prit(jl)*prg_tend(jl, iridryg)
163  prg_tend(jl, iriwetg)=prg_tend(jl, iridryg) / (icep%XCOLIG*exp(icep%XCOLEXIG*(pt(jl)-cst%XTT)))
164  ENDIF
165  ELSE
166  prg_tend(jl, iridryg)=0.
167  prg_tend(jl, iriwetg)=0.
168  ENDIF
169 ENDDO
170 
171 ! Wet and dry collection of rs on graupel (6.2.1)
172 DO jl = 1, ksize
173  IF (prst(jl)>iced%XRTMIN(5) .AND. prgt(jl)>iced%XRTMIN(6) .AND. ldcompute(jl)) THEN
174  gdry(jl) = .true.
175  ELSE
176  gdry(jl) = .false.
177  prg_tend(jl, irsdryg)=0.
178  prg_tend(jl, irswetg)=0.
179  ENDIF
180 ENDDO
181 
182 IF(.NOT. ldsoft) THEN
183  CALL interp_micro_2d(kproma, ksize, plbdag(:), plbdas(:), icep%NDRYLBDAG, icep%NDRYLBDAS, &
184  &icep%XDRYINTP1G, icep%XDRYINTP2G, icep%XDRYINTP1S, icep%XDRYINTP2S, &
185  &parami%LPACK_INTERP, gdry(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
186  &igdry, &
187  &icep%XKER_SDRYG(:,:), zzw(:))
188  IF(igdry>0)THEN
189  !$mnh_expand_where(JL=1:KSIZE)
190  WHERE(gdry(1:ksize))
191 #ifdef REPRO48
192  prg_tend(1:ksize, irswetg)=icep%XFSDRYG*zzw(1:ksize) & ! RSDRYG
193  / icep%XCOLSG &
194  *(plbdas(1:ksize)**(iced%XCXS-iced%XBS))*( plbdag(1:ksize)**iced%XCXG ) &
195  *(prhodref(1:ksize)**(-iced%XCEXVT-1.)) &
196  *( icep%XLBSDRYG1/( plbdag(1:ksize)**2 ) + &
197  icep%XLBSDRYG2/( plbdag(1:ksize) * plbdas(1:ksize) ) + &
198  icep%XLBSDRYG3/( plbdas(1:ksize)**2))
199 #else
200  prg_tend(1:ksize, irswetg)=icep%XFSDRYG*zzw(1:ksize) & ! RSDRYG
201  / icep%XCOLSG &
202  *(prst(1:ksize))*( plbdag(1:ksize)**iced%XCXG ) &
203  *(prhodref(1:ksize)**(-iced%XCEXVT)) &
204  *( icep%XLBSDRYG1/( plbdag(1:ksize)**2 ) + &
205  icep%XLBSDRYG2/( plbdag(1:ksize) * plbdas(1:ksize) ) + &
206  icep%XLBSDRYG3/( plbdas(1:ksize)**2))
207 #endif
208  prg_tend(1:ksize, irsdryg)=prg_tend(1:ksize, irswetg)*icep%XCOLSG*exp(icep%XCOLEXSG*(pt(1:ksize)-cst%XTT))
209  END WHERE
210  !$mnh_end_expand_where(JL=1:KSIZE)
211  ENDIF
212 ENDIF
213 !
214 !* 6.2.6 accretion of raindrops on the graupeln
215 !
216 DO jl = 1, ksize
217  IF (prrt(jl)>iced%XRTMIN(3) .AND. prgt(jl)>iced%XRTMIN(6) .AND. ldcompute(jl)) THEN
218  gdry(jl) = .true.
219  ELSE
220  gdry(jl) = .false.
221  prg_tend(jl, irrdryg)=0.
222  ENDIF
223 ENDDO
224 IF(.NOT. ldsoft) THEN
225  !
226  CALL interp_micro_2d(kproma, ksize, plbdag(:), plbdar(:), icep%NDRYLBDAG, icep%NDRYLBDAR, &
227  &icep%XDRYINTP1G, icep%XDRYINTP2G, icep%XDRYINTP1R, icep%XDRYINTP2R, &
228  &parami%LPACK_INTERP, gdry(:), ibuf1(:), ibuf2(:), ibuf3(:), zbuf1(:), zbuf2(:), zbuf3(:), &
229  &igdry, &
230  &icep%XKER_RDRYG(:,:), zzw(:))
231  IF(igdry>0) THEN
232  !$mnh_expand_where(JL=1:KSIZE)
233  WHERE(gdry(1:ksize))
234  prg_tend(1:ksize, irrdryg) = icep%XFRDRYG*zzw(1:ksize) & ! RRDRYG
235  *( plbdar(1:ksize)**(-4) )*( plbdag(1:ksize)**iced%XCXG ) &
236  *( prhodref(1:ksize)**(-iced%XCEXVT-1.) ) &
237  *( icep%XLBRDRYG1/( plbdag(1:ksize)**2 ) + &
238  icep%XLBRDRYG2/( plbdag(1:ksize) * plbdar(1:ksize) ) + &
239  icep%XLBRDRYG3/( plbdar(1:ksize)**2) )
240  END WHERE
241  !$mnh_end_expand_where(JL=1:KSIZE)
242  ENDIF
243 ENDIF
244 
245 DO jl=1, ksize
246  zrdryg_init(jl)=prg_tend(jl, ircdryg)+prg_tend(jl, iridryg)+ &
247  &prg_tend(jl, irsdryg)+prg_tend(jl, irrdryg)
248 ENDDO
249 
250 !Freezing rate and growth mode
251 DO jl=1, ksize
252  IF(prgt(jl)>iced%XRTMIN(6) .AND. ldcompute(jl)) THEN
253  !Freezing rate
254  IF(.NOT. ldsoft) THEN
255  prg_tend(jl, ifreez1)=prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
256  IF(parami%LEVLIMIT) THEN
257  prg_tend(jl, ifreez1)=min(prg_tend(jl, ifreez1), exp(cst%XALPI-cst%XBETAI/pt(jl)-cst%XGAMI*alog(pt(jl)))) ! min(ev, es_i(T))
258  ENDIF
259  prg_tend(jl, ifreez1)=pka(jl)*(cst%XTT-pt(jl)) + &
260  (pdv(jl)*(cst%XLVTT+(cst%XCPV-cst%XCL)*(pt(jl)-cst%XTT)) &
261  *(cst%XESTT-prg_tend(jl, ifreez1))/(cst%XRV*pt(jl)) )
262  prg_tend(jl, ifreez1)=prg_tend(jl, ifreez1)* ( icep%X0DEPG* plbdag(jl)**icep%XEX0DEPG + &
263  icep%X1DEPG*pcj(jl)*plbdag(jl)**icep%XEX1DEPG )/ &
264  ( prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))) )
265  prg_tend(jl, ifreez2)=(prhodref(jl)*(cst%XLMTT+(cst%XCI-cst%XCL)*(cst%XTT-pt(jl))) ) / &
266  ( prhodref(jl)*(cst%XLMTT-cst%XCL*(cst%XTT-pt(jl))) )
267  ENDIF
268  zrwetg_init(jl)=max(prg_tend(jl, iriwetg)+prg_tend(jl, irswetg), &
269  &max(0., prg_tend(jl, ifreez1) + &
270  & prg_tend(jl, ifreez2) * (prg_tend(jl, iriwetg)+prg_tend(jl, irswetg))))
271 
272  !Growth mode
273  ldwetg(jl)=max(0., zrwetg_init(jl)-prg_tend(jl, iriwetg)-prg_tend(jl, irswetg))<= &
274  &max(0., zrdryg_init(jl)-prg_tend(jl, iridryg)-prg_tend(jl, irsdryg))
275 
276  IF(parami%LNULLWETG) THEN
277  ldwetg(jl) = ldwetg(jl) .AND. zrdryg_init(jl)>0.
278  ELSE
279  ldwetg(jl) = ldwetg(jl) .AND. zrwetg_init(jl)>0.
280  ENDIF
281  IF(.NOT. parami%LWETGPOST) THEN
282  ldwetg(jl) = ldwetg(jl) .AND. pt(jl)<cst%XTT
283  ENDIF
284 
285  lldryg(jl)=pt(jl)<cst%XTT .AND. zrdryg_init(jl)>1.e-20 .AND. &
286  &max(0., zrwetg_init(jl)-prg_tend(jl, iriwetg)-prg_tend(jl, irswetg))>&
287  &max(0., zrdryg_init(jl)-prg_tend(jl, iridryg)-prg_tend(jl, irsdryg))
288  ELSE
289  prg_tend(jl, ifreez1)=0.
290  prg_tend(jl, ifreez2)=0.
291  zrwetg_init(jl)=0.
292  ldwetg(jl)=.false.
293  lldryg(jl)=.false.
294  ENDIF
295 ENDDO
296 
297 ! Part of ZRWETG to be converted into hail
298 ! Graupel can be produced by other processes instantaneously (inducing a mixing ratio change, PRGSI_MR) or
299 ! as a tendency (PRWETGH)
300 IF(krr==7) THEN
301  !$mnh_expand_where(JL=1:KSIZE)
302  WHERE(ldwetg(1:ksize))
303  !assume a linear percent of conversion of produced graupel into hail
304  prwetgh(1:ksize)=(max(0., prgsi(1:ksize)+pricfrrg(1:ksize)+prrcfrig(1:ksize))+zrwetg_init(1:ksize))*&
305  &zrdryg_init(1:ksize)/(zrwetg_init(1:ksize)+zrdryg_init(1:ksize))
306  prwetgh_mr(1:ksize)=max(0., prgsi_mr(1:ksize))*zrdryg_init(1:ksize)/(zrwetg_init(1:ksize)+zrdryg_init(1:ksize))
307  ELSEWHERE
308  prwetgh(1:ksize)=0.
309  prwetgh_mr(1:ksize)=0.
310  END WHERE
311  !$mnh_end_expand_where(JL=1:KSIZE)
312 ELSE
313  prwetgh(:)=0.
314  prwetgh_mr(:)=0.
315 ENDIF
316 
317 DO jl=1, ksize
318  !Aggregated minus collected
319  IF(ldwetg(jl)) THEN
320  prrwetg(jl)=-(prg_tend(jl, iriwetg)+prg_tend(jl, irswetg)+&
321  &prg_tend(jl, ircdryg)-zrwetg_init(jl))
322  prcwetg(jl)=prg_tend(jl, ircdryg)
323  priwetg(jl)=prg_tend(jl, iriwetg)
324  prswetg(jl)=prg_tend(jl, irswetg)
325  ELSE
326  prrwetg(jl)=0.
327  prcwetg(jl)=0.
328  priwetg(jl)=0.
329  prswetg(jl)=0.
330  ENDIF
331 
332  IF(lldryg(jl)) THEN
333  prcdryg(jl)=prg_tend(jl, ircdryg)
334  prrdryg(jl)=prg_tend(jl, irrdryg)
335  pridryg(jl)=prg_tend(jl, iridryg)
336  prsdryg(jl)=prg_tend(jl, irsdryg)
337  ELSE
338  prcdryg(jl)=0.
339  prrdryg(jl)=0.
340  pridryg(jl)=0.
341  prsdryg(jl)=0.
342  ENDIF
343 ENDDO
344 !
345 !* 6.5 Melting of the graupeln
346 !
347 DO jl=1, ksize
348  IF(prgt(jl)>iced%XRTMIN(6) .AND. pt(jl)>cst%XTT .AND. ldcompute(jl)) THEN
349  IF(.NOT. ldsoft) THEN
350  prgmltr(jl)=prvt(jl)*ppres(jl)/(cst%XEPSILO+prvt(jl)) ! Vapor pressure
351  IF(parami%LEVLIMIT) THEN
352  prgmltr(jl)=min(prgmltr(jl), exp(cst%XALPW-cst%XBETAW/pt(jl)-cst%XGAMW*alog(pt(jl)))) ! min(ev, es_w(T))
353  ENDIF
354  prgmltr(jl)=pka(jl)*(cst%XTT-pt(jl)) + &
355  pdv(jl)*(cst%XLVTT + ( cst%XCPV - cst%XCL ) * ( pt(jl) - cst%XTT )) &
356  *(cst%XESTT-prgmltr(jl))/(cst%XRV*pt(jl))
357  prgmltr(jl)=max(0., (-prgmltr(jl)* &
358  (icep%X0DEPG* plbdag(jl)**icep%XEX0DEPG + &
359  icep%X1DEPG*pcj(jl)*plbdag(jl)**icep%XEX1DEPG) - &
360  (prg_tend(jl, ircdryg)+prg_tend(jl, irrdryg)) * &
361  (prhodref(jl)*cst%XCL*(cst%XTT-pt(jl)))) / &
362  ( prhodref(jl)*cst%XLMTT ) )
363  ENDIF
364  ELSE
365  prgmltr(jl)=0.
366  ENDIF
367 ENDDO
368 !
369 IF (lhook) CALL dr_hook('ICE4_FAST_RG', 1, zhook_handle)
370 !
371 CONTAINS
372 !
373 include "interp_micro.func.h"
374 !
375 END SUBROUTINE ice4_fast_rg
376 !
377 END MODULE mode_ice4_fast_rg
