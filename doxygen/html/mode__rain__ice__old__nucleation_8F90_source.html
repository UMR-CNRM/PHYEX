95 REAL, DIMENSION(D%NIT,D%NKT) :: ZW ! work array
96!
97! compute the temperature and the pressure
98!
99 REAL(KIND=jphook) :: zhook_handle
100
101 IF (lhook) CALL dr_hook('RAIN_ICE_OLD:RAIN_ICE_OLD_NUCLEATION',0,zhook_handle)
102
103 IF( ksize >= 1 ) THEN
104!
105! optimization by looking for locations where
106! the temperature is negative only !!!
107!
108 jl = 0
109 DO jk = 1, d%NKT
110 DO ji = 1, d%NIT
111 IF (pt(ji, jk) < cst%XTT) THEN
112 jl = jl + 1
113 i1(jl) = ji
114 i2(jl) = jk
115 ENDIF
116 ENDDO
117 ENDDO
118
119
120 DO jl=1, ksize
121 zrvt(jl) = prvt(i1(jl),i2(jl))
122 zcit(jl) = pcit(i1(jl),i2(jl))
123 zzt(jl) = pt(i1(jl),i2(jl))
124 zpres(jl) = ppabst(i1(jl),i2(jl))
125 zzicenu(jl) = picenu(i1(jl))
126
127 IF (ocnd2) THEN
128 zzz(jl) = pzzz(i1(jl),i2(jl))
129 zesi(jl) = esati(icep%TIWMX, zzt(jl))
130 zesw(jl) = esatw(icep%TIWMX, zzt(jl))
131 zam3(jl) = am3(icep%TIWMX, max(icep%XFRMIN(27),zzt(jl))) ! Avoid too high IN for very low temp.
132 zredin(jl) = redin(icep%TIWMX, zzt(jl))
133 zsifrc(jl) = picldfr(i1(jl),i2(jl))
134 ENDIF
135 IF (oaeronrt.AND.oaeifn) THEN
136 zzicenu(jl)=max(0.01,pifnnc(i1(jl),i2(jl))*1.0e-6) ! convert m-3 to cm-3
137 ENDIF
138
139 ENDDO
140
141 IF(ocnd2)THEN ! try to do some optimazation :
142
143 DO jl = 1, ksize
144 zzw(jl) = min(zpres(jl)/2., zesi(jl)) ! safety limitation es_i
145 zzw(jl) = min(zpres(jl)/2., zzw(jl)) ! safety limitation
146 zssi(jl) = zrvt(jl)*(zpres(jl)-zzw(jl)) / (cst%XEPSILO * zzw(jl)) - 1.0
147 ENDDO
148 ! Supersaturation over ice
149 DO jl = 1, ksize
150 zusw(jl) = min(zpres(jl)/2.,zesw(jl)) ! safety limitation es_w
151 zusw(jl) = (zusw(jl)/zzw(jl))*((zpres(jl)-zzw(jl))/(zpres(jl)-zusw(jl))) - 1.0
152 ENDDO
153 ! Supersaturation of saturated water vapor over ice
154 ELSE
155
156 DO jl = 1, ksize
157 zzw(jl) = exp( cst%XALPI - cst%XBETAI/zzt(jl) - cst%XGAMI*log(zzt(jl))) ! es_i
158 zzw(jl) = min(zpres(jl)/2., zzw(jl)) ! safety limitation
159 zssi(jl) = zrvt(jl)*( zpres(jl)-zzw(jl) ) / ( cst%XEPSILO * zzw(jl) ) - 1.0
160 ENDDO
161 ! Supersaturation over ice
162 DO jl = 1, ksize
163 zusw(jl) = exp( cst%XALPW - cst%XBETAW/zzt(jl) - cst%XGAMW*log(zzt(jl))) ! es_w
164 zusw(jl) = min(zpres(jl)/2.,zusw(jl)) ! safety limitation
165 ! Supersaturation of saturated water vapor over ice
166 zusw(jl) = (zusw(jl)/zzw(jl))*((zpres(jl)-zzw(jl))/(zpres(jl)-zusw(jl))) - 1.0
167 ENDDO
168
169 ENDIF
170!
171!* 3.1 compute the heterogeneous nucleation source: RVHENI
172!
173!* 3.1.1 compute the cloud ice concentration
174!
175 zzw(:) = 0.0
176
177 zssi(:) = min( zssi(:), zusw(:) ) ! limitation of SSi according to SSw=0
178
179 IF(ocnd2)THEN
180
181 IF (lmodicedep) THEN
182 DO jl = 1, ksize
183 zzw(jl) = 5.*exp(0.304*(cst%XTT-zzt(jl)))
184 zzw(jl) = min(1.,max(zssi(jl)*10.,0.01))*zzw(jl)
185 ENDDO
186 ELSE
187
188 DO jl = 1, ksize
189 zzw(jl) = zredin(jl)* max(0.1,((20000.- min(20000.,zzz(jl)))/20000.)**4) &
190 & *zam3(jl)*(0.0001 + 0.9999*zsifrc(jl))
191 ENDDO
192
193 ENDIF
194
195 ELSE
196
197 DO jl = 1, ksize
198 IF ((zzt(jl)<cst%XTT-5.0) .AND. (zssi(jl)>0.0)) THEN
199 zzw(jl) = icep%XNU20 * exp( icep%XALPHA2*zssi(jl)-icep%XBETA2 )
200 END IF
201 ENDDO
202
203 DO jl = 1, ksize
204 IF ((zzt(jl)<=cst%XTT-2.0) .AND. (zzt(jl)>=cst%XTT-5.0) .AND. (zssi(jl)>0.0)) THEN
205 zzw(jl) = max(icep%XNU20 * exp(-icep%XBETA2),icep%XNU10 * exp(-icep%XBETA1*(zzt(jl)-cst%XTT)) * &
206 (zssi(jl)/zusw(jl))**icep%XALPHA1)
207 END IF
208 ENDDO
209 ENDIF
210
211 zzw(:) = zzw(:)*zzicenu(:) - zcit(:)
212
213 IF( maxval(zzw(:)) > 0.0 ) THEN
214!
215!* 3.1.2 update the r_i and r_v mixing ratios
216!
217 zzw(:) = min( zzw(:),50.e3 ) ! limitation provisoire a 50 l^-1
218
219 IF(.NOT.ocnd2)THEN
220 zw(:,:) = 0.
221 DO jl=1, ksize
222 zw(i1(jl), i2(jl)) = zzw(jl)
223 ENDDO
224 zw(:,:) = max( zw(:,:) ,0.0 ) *icep%XMNU0/(prhodref(:,:)*ptstep)
225 pris(:,:) = pris(:,:) + zw(:,:)
226 prvs(:,:) = prvs(:,:) - zw(:,:)
227
228 IF ( krr == 7 ) THEN
229 pths(:,:) = pths(:,:) + zw(:,:)*(cst%XLSTT+(cst%XCPV-cst%XCI)*(pt(:,:)-cst%XTT)) &
230 /( (cst%XCPD + cst%XCPV*prvt(:,:) + cst%XCL*(prct(:,:)+prrt(:,:)) &
231 + cst%XCI*(prit(:,:)+prst(:,:)+prgt(:,:)+prht(:,:)))*pexnref(:,:) )
232 ELSE IF( krr == 6 ) THEN
233 pths(:,:) = pths(:,:) + zw(:,:)*(cst%XLSTT+(cst%XCPV-cst%XCI)*(pt(:,:)-cst%XTT)) &
234 /( (cst%XCPD + cst%XCPV*prvt(:,:) + cst%XCL*(prct(:,:)+prrt(:,:)) &
235 + cst%XCI*(prit(:,:)+prst(:,:)+prgt(:,:)))*pexnref(:,:) )
236 END IF
237 ENDIF
238 ! f(L_s*(RVHENI))
239 zzw(:) = max(zzw(:)+zcit(:),zcit(:))
240
241 DO jl=1, ksize
242 pcit(i1(jl), i2(jl)) = zzw(jl)
243 ENDDO
244 END IF
245
246 END IF
247
248 IF (lhook) CALL dr_hook('RAIN_ICE_OLD:RAIN_ICE_OLD_NUCLEATION',1,zhook_handle)

250 END SUBROUTINE rain_ice_old_nucleation
