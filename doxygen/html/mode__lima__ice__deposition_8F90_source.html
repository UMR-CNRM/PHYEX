18!! -------

56TYPE(param_lima_t),INTENT(IN)::LIMAP

69REAL, DIMENSION(KSIZE), INTENT(IN) :: PRIT ! Cloud ice m.r. at t
70!
71REAL, DIMENSION(KSIZE), INTENT(IN) :: PCIT ! Ice crystal C. at t
72REAL, DIMENSION(KSIZE,LIMAP%NNB_CRYSTAL_SHAPE), INTENT(IN) :: PCIT_SHAPE ! Ice crystal C. at t for each shape

82!
83!* 0.2 Declarations of local variables :
84!
85LOGICAL, DIMENSION(SIZE(PRHODREF)) :: GMICRO ! Computations only where necessary
86REAL, DIMENSION(SIZE(PRHODREF)) :: ZZW, ZZW2, ZZX, ZCRIAUTI ! Work array
87REAL(KIND=jphook) :: zhook_handle
88!
89REAL, DIMENSION(:,:), ALLOCATABLE :: ZRIT_SHAPE
90REAL, DIMENSION(:,:), ALLOCATABLE :: ZLBDAI_SHAPE
91INTEGER :: ISH
92INTEGER :: ISIZE
93!
94!-------------------------------------------------------------------------------
95!
96IF (lhook) CALL dr_hook('LIMA_ICE_DEPOSITION', 0, zhook_handle)
97p_th_depi(:) = 0.
98p_ri_depi(:) = 0.
99p_ri_cnvs(:) = 0.
100p_ci_cnvs(:) = 0.
101!
102! Physical limitations
103!
104!
105! Looking for regions where computations are necessary
106!
107gmicro(:) = odcompute(:) .AND. prit(:)>limap%XRTMIN(4)
108IF (limap%NMOM_I.GE.2) gmicro(:) = gmicro(:) .AND. pcit(:)>limap%XCTMIN(4)
109!
110zzw(:) = 0.0
111zzw2(:) = 0.0
112IF (limap%NMOM_I.EQ.1) THEN
113!
114!* Conversion of pristine ice to r_s: RICNVS
115! -----------------------------------------
116!
117 WHERE( gmicro )
118 zcriauti(:)=min(0.2e-4,10**(0.06*(pt(:)-cst%XTT)-3.5))
119 zzw(:) = 1.e-3 * exp( 0.015*(pt(:)-cst%XTT) ) * max( prit(:)-zcriauti(:),0.0 )
120 p_ri_cnvs(:) = - zzw(:)
121 END WHERE
122ELSE
123 IF (.NOT. limap%LCRYSTAL_SHAPE) THEN
124!
125!* Deposition of water vapor on r_i: RVDEPI
126! ----------------------------------------
127!
128 WHERE( gmicro )
129 zzw(:) = ( pssi(:) / pai(:) ) * pcit(:) * &
130 ( limac%X0DEPI/plbdi(:)+limac%X2DEPI*pcj(:)*pcj(:)/plbdi(:)**(limac%XDI+2.0) )
131 p_ri_depi(:) = zzw(:)
132 END WHERE
133!
134!* Conversion of pristine ice to r_s: RICNVS
135! -----------------------------------------
136!
137 zzw(:) = 0.0
138 zzw2(:) = 0.0
139 WHERE ( gmicro .AND. plbdi(:)<limac%XLBDAICNVS_LIM .AND. pssi(:)>0.0 )
140 zzw(:) = (plbdi(:)*limac%XDICNVS_LIM)**(limap%XALPHAI)
141 zzx(:) = (pssi(:)/pai(:))*pcit(:) * (zzw(:)**limap%XNUI) *exp(-zzw(:))
142!
143 zzw(:) = (limac%XR0DEPIS + limac%XR1DEPIS*pcj(:))*zzx(:)
144!
145 zzw2(:) = zzw(:) * (limac%XC0DEPIS+limac%XC1DEPIS*pcj(:)) / (limac%XR0DEPIS+limac%XR1DEPIS*pcj(:))
146 p_ri_cnvs(:) = - zzw(:)
147 p_ci_cnvs(:) = - zzw2(:)
148 END WHERE
149 ELSE ! LCRYSTAL_SHAPE
150!
151!* Deposition of water vapor on r_i: RVDEPI
152! ----------------------------------------
153 ALLOCATE(zrit_shape(SIZE(prhodref),limap%NNB_CRYSTAL_SHAPE))
154 ALLOCATE(zlbdai_shape(SIZE(prhodref),limap%NNB_CRYSTAL_SHAPE))
155 ! compute lambdai and ri per shape
156 CALL lima_shape_compute_lbda (limap, limac, ksize, prit, pcit_shape, &
157 zrit_shape, zlbdai_shape)
158 !
159 ! compute the deposition rate per shape
160 DO ish = 1, limap%NNB_CRYSTAL_SHAPE
161 zzw(:) = 0.
162 WHERE ( gmicro(:) .AND. (zrit_shape(:,ish)>limap%XRTMIN(4)) .AND. (pcit_shape(:,ish)>limap%XCTMIN(4)) )
163 zzw(:) = (pssi(:) / pai(:)) * pcit_shape(:,ish) * &
164 (limac%X0DEPI_SHAPE(ish) / zlbdai_shape(:,ish) + &
165 limac%X2DEPI_SHAPE(ish) * pcj(:) * pcj(:) / &
166 zlbdai_shape(:,ish)**(limac%XDI_SHAPE(ish)+2.0))
167 END WHERE
168 p_ri_depi(:) = p_ri_depi(:) + zzw(:)
169 END DO
170 !
171 ! plates and columns can change shape
172 CALL lima_change_shape (cst, limap, ksize, pt, limap%XCTMIN, p_ri_depi, pcit_shape, p_shci_hach)
173 !
174 DEALLOCATE(zrit_shape)
175!
176!* Conversion of pristine ice to r_s: RICNVS
177! -----------------------------------------
178 DO ish = 1, limap%NNB_CRYSTAL_SHAPE
179 zzw(:) = 0.0
180 zzw2(:) = 0.0
181 WHERE ( gmicro .AND. (zlbdai_shape(:,ish) < limac%XLBDAICNVS_LIM) &
182 .AND. (pcit_shape(:,ish) > limap%XCTMIN(4)) &
183 .AND. (pssi(:) > 0.0) )
184 zzw(:) = (zlbdai_shape(:,ish) * limac%XDICNVS_LIM)**(limap%XALPHAI)
185 zzx(:) = (pssi(:) / pai(:)) * pcit_shape(:,ish) * (zzw(:)**limap%XNUI) * exp(-zzw(:))
186 zzw(:) = (limac%XR0DEPIS_SHAPE(ish) + limac%XR1DEPIS_SHAPE(ish) * pcj(:)) * zzx(:)
187 !
188 zzw2(:) = zzw(:) * (limac%XC0DEPIS_SHAPE(ish) + limac%XC1DEPIS_SHAPE(ish) * pcj(:)) / &
189 (limac%XR0DEPIS_SHAPE(ish) + limac%XR1DEPIS_SHAPE(ish) * pcj(:))
190 END WHERE
191 p_ri_cnvs(:) = p_ri_cnvs(:) - zzw(:)
192 p_shci_cnvs(:,ish) = - zzw2(:)
193 END DO
194 DEALLOCATE(zlbdai_shape)
195 ENDIF ! LCRYSTAL_SHAPE
196END IF
197!
198IF (limap%NMOM_S.EQ.0) THEN
199 p_ri_cnvs(:) = 0.
200 p_ci_cnvs(:) = 0.
201END IF
202!
203IF (lhook) CALL dr_hook('LIMA_ICE_DEPOSITION', 1, zhook_handle)
204END SUBROUTINE lima_ice_deposition
