17 !! PURPOSE
18 !! -------
19 !! The purpose of this routine is to compute the microphysical sources
20 !! for slow cold processes :
21 !! - conversion of snow to ice
22 !! - deposition of vapor on snow
23 !! - conversion of ice to snow (Harrington 1995)

88 ! Physical limitations
89 !
90 !
91 ! Looking for regions where computations are necessary
92 !
93 gmicro(:) = ldcompute(:) .AND. prit(:)>xrtmin(4)
94 !
95 !
96 IF (nmom_i.EQ.1) THEN
97  WHERE( gmicro )
98 !
99 !* Conversion of pristine ice to r_s: RICNVS
100 ! -----------------------------------------
101 !
102  zcriauti(:)=min(0.2e-4,10**(0.06*(pt(:)-xtt)-3.5))
103  zzw(:) = 0.0
104  WHERE ( (prit(:)>xrtmin(4)))
105  zzw(:) = 1.e-3 * exp( 0.015*(pt(:)-xtt) ) * max( prit(:)-zcriauti(:),0.0 )
106  END WHERE
107 !
108  p_ri_cnvs(:) = - zzw(:)
109  END WHERE
110 ELSE
111  WHERE( gmicro )
112 !
113 !* Deposition of water vapor on r_i: RVDEPI
114 ! ----------------------------------------
115 !
116  zzw(:) = 0.0
117  WHERE ( (prit(:)>xrtmin(4)) .AND. (pcit(:)>xctmin(4)) )
118  zzw(:) = ( pssi(:) / pai(:) ) * pcit(:) * &
119  ( x0depi/plbdi(:)+x2depi*pcj(:)*pcj(:)/plbdi(:)**(xdi+2.0) )
120  END WHERE
121  p_ri_depi(:) = zzw(:)
122 !
123 !* Conversion of pristine ice to r_s: RICNVS
124 ! -----------------------------------------
125 !
126  zzw(:) = 0.0
127  zzw2(:) = 0.0
128  WHERE ( (plbdi(:)<xlbdaicnvs_lim) .AND. (pcit(:)>xctmin(4)) &
129  .AND. (pssi(:)>0.0) )
130  zzw(:) = (plbdi(:)*xdicnvs_lim)**(xalphai)
131  zzx(:) = (pssi(:)/pai(:))*pcit(:) * (zzw(:)**xnui) *exp(-zzw(:))
132 !
133  zzw(:) = (xr0depis + xr1depis*pcj(:))*zzx(:)
134 !
135  zzw2(:) = zzw(:) * (xc0depis+xc1depis*pcj(:)) / (xr0depis+xr1depis*pcj(:))
136  END WHERE
137  p_ri_cnvs(:) = - zzw(:)
138  p_ci_cnvs(:) = - zzw2(:)
139  END WHERE
140 END IF
141 !
142 IF (nmom_s.EQ.0) THEN
143  p_ri_cnvs(:) = 0.
144  p_ci_cnvs(:) = 0.
145 END IF
