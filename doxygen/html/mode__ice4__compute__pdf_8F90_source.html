44 TYPE(rain_ice_descr_t), INTENT(IN) :: ICED

47 CHARACTER(LEN=80), INTENT(IN) :: HSUBG_AUCV_RI ! Kind of Subgrid autoconversion method for cloud ice
48 CHARACTER(LEN=80), INTENT(IN) :: HSUBG_PR_PDF ! pdf for subgrid precipitation

51 REAL, DIMENSION(KSIZE), INTENT(IN) :: PRCT ! Cloud water m.r. at t
52 REAL, DIMENSION(KSIZE), INTENT(IN) :: PRIT ! Ice Crystal m.r. at t
53 REAL, DIMENSION(KSIZE), INTENT(IN) :: PCF ! Cloud fraction
54 REAL, DIMENSION(KSIZE), INTENT(IN) :: PT ! Temperature
55 REAL, DIMENSION(KSIZE), INTENT(IN) :: PSIGMA_RC ! Standard deviation of rc at time t
56 !Note for INTENT STATUS: in 'ADJU' case the PHL?_??? variables must be able to "cross" the subroutine untouched
57 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PHLC_HCF ! HLCLOUDS : fraction of High Cloud Fraction in grid
58 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PHLC_LCF ! HLCLOUDS : fraction of Low Cloud Fraction in grid
59  ! note that PCF = PHLC_HCF + PHLC_LCF
60 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PHLC_HRC ! HLCLOUDS : LWC that is High LWC in grid
61 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PHLC_LRC ! HLCLOUDS : LWC that is Low LWC in grid
62  ! note that PRC = PHLC_HRC + PHLC_LRC
63 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PHLI_HCF
64 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PHLI_LCF
65 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PHLI_HRI
66 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PHLI_LRI
67 REAL, DIMENSION(KSIZE), INTENT(OUT) :: PRF ! Rain fraction
68 !
69 !* 0.2 declaration of local variables
70 !
71 REAL, DIMENSION(KSIZE) :: ZRCRAUTC, & !RC value to begin rain formation =XCRIAUTC/RHODREF
72  ZCRIAUTI, & !RI value to begin snow formation
73  ZHLC_RCMAX, & !HLCLOUDS : maximum value for RC in distribution
74  ZHLC_LRCLOCAL, & !HLCLOUDS : LWC that is Low LWC local in LCF
75  ZHLC_HRCLOCAL, & !HLCLOUDS : LWC that is High LWC local in HCF
76  ! note that ZRC/CF = ZHLC_HRCLOCAL+ ZHLC_LRCLOCAL
77  ! = PHLC_HRC/HCF+ PHLC_LRC/LCF
78  zsumrc, zsumri
79 REAL :: ZCOEFFRCM
80 REAL(KIND=jphook) :: zhook_handle

83 !

86 !Cloud water split between high and low content part is done according to autoconversion option
87 !$mnh_expand_where(JL=1:KSIZE)
88 WHERE (ldmicro(:))
89  zrcrautc(:)=icep%XCRIAUTC/prhodref(:) ! Autoconversion rc threshold
90 ELSEWHERE
91  zrcrautc(:)=0.
92 END WHERE
93 !$mnh_end_expand_where(JL=1:KSIZE)
94 IF(hsubg_aucv_rc=='NONE') THEN
95  !Cloud water is entirely in low or high part
96  !$mnh_expand_where(JL=1:KSIZE)
97  WHERE(prct(:)>zrcrautc(:) .AND. ldmicro(:))
98  phlc_hcf(:)=1.
99  phlc_lcf(:)=0.
100  phlc_hrc(:)=prct(:)
101  phlc_lrc(:)=0.
102  ELSEWHERE(prct(:)>iced%XRTMIN(2) .AND. ldmicro(:))
103  phlc_hcf(:)=0.
104  phlc_lcf(:)=1.
105  phlc_hrc(:)=0.
106  phlc_lrc(:)=prct(:)
107  ELSEWHERE
108  phlc_hcf(:)=0.
109  phlc_lcf(:)=0.
110  phlc_hrc(:)=0.
111  phlc_lrc(:)=0.
112  END WHERE
113  !$mnh_end_expand_where(JL=1:KSIZE)
114 
115 ELSEIF(hsubg_aucv_rc=='CLFR') THEN
116  !Cloud water is only in the cloudy part and entirely in low or high part
117  !$mnh_expand_where(JL=1:KSIZE)
118  WHERE(pcf(:)>0. .AND. prct(:)>zrcrautc(:)*pcf(:) .AND. ldmicro(:))
119  phlc_hcf(:)=pcf(:)
120  phlc_lcf(:)=0.
121  phlc_hrc(:)=prct(:)
122  phlc_lrc(:)=0.
123  ELSEWHERE(pcf(:)>0. .AND. prct(:)>iced%XRTMIN(2) .AND. ldmicro(:))
124  phlc_hcf(:)=0.
125  phlc_lcf(:)=pcf(:)
126  phlc_hrc(:)=0.0
127  phlc_lrc(:)=prct(:)
128  ELSEWHERE
129  phlc_hcf(:)=0.
130  phlc_lcf(:)=0.
131  phlc_hrc(:)=0.
132  phlc_lrc(:)=0.
133  END WHERE
134  !$mnh_end_expand_where(JL=1:KSIZE)
135 ELSEIF(hsubg_aucv_rc=='ADJU') THEN
136  !$mnh_expand_where(JL=1:KSIZE)
137  WHERE(ldmicro(:))
138  zsumrc(:)=phlc_lrc(:)+phlc_hrc(:)
139  ELSEWHERE
140  zsumrc(:)=0.
141  ENDWHERE
142  !$mnh_end_expand_where(JL=1:KSIZE)
143  !$mnh_expand_where(JL=1:KSIZE)
144  WHERE(zsumrc(:) .GT. 0. .AND. ldmicro(:))
145  phlc_lrc(:)=phlc_lrc(:)*prct(:)/zsumrc(:)
146  phlc_hrc(:)=phlc_hrc(:)*prct(:)/zsumrc(:)
147  ELSEWHERE
148  phlc_lrc(:)=0.
149  phlc_hrc(:)=0.
150  ENDWHERE
151  !$mnh_end_expand_where(JL=1:KSIZE)
152 ELSEIF(hsubg_aucv_rc=='PDF ') THEN
153  !Cloud water is split between high and low part according to a PDF
154  ! 'HLCRECTPDF' : rectangular PDF form
155  ! 'HLCTRIANGPDF' : triangular PDF form
156  ! 'HLCQUADRAPDF' : second order quadratic PDF form
157  ! 'HLCISOTRIPDF' : isocele triangular PDF
158  ! 'SIGM' : Redelsperger and Sommeria (1986)
159  IF(hsubg_pr_pdf=='SIGM') THEN
160  ! Redelsperger and Sommeria (1986) but organised according to Turner (2011, 2012)
161  !$mnh_expand_where(JL=1:KSIZE)
162  WHERE (prct(:)>zrcrautc(:)+psigma_rc(:) .AND. ldmicro(:))
163  phlc_hcf(:)=1.
164  phlc_lcf(:)=0.
165  phlc_hrc(:)=prct(:)
166  phlc_lrc(:)=0.
167  ELSEWHERE(prct(:)> (zrcrautc(:)-psigma_rc(:)) .AND. prct(:)<=(zrcrautc(:)+psigma_rc(:)) .AND. ldmicro(:))
168  phlc_hcf(:)=(prct(:)+psigma_rc(:)-zrcrautc(:))/ &
169  &(2.*psigma_rc(:))
170  phlc_lcf(:)=max(0., pcf(:)-phlc_hcf(:))
171  phlc_hrc(:)=(prct(:)+psigma_rc(:)-zrcrautc(:))* &
172  &(prct(:)+psigma_rc(:)+zrcrautc(:))/ &
173  &(4.*psigma_rc(:))
174  phlc_lrc(:)=max(0., prct(:)-phlc_hrc(:))
175  ELSEWHERE(prct(:)>iced%XRTMIN(2) .AND. pcf(:)>0. .AND. ldmicro(:))
176  phlc_hcf(:)=0.
177  phlc_lcf(:)=pcf(:)
178  phlc_hrc(:)=0.
179  phlc_lrc(:)=prct(:)
180  ELSEWHERE
181  phlc_hcf(:)=0.
182  phlc_lcf(:)=0.
183  phlc_hrc(:)=0.
184  phlc_lrc(:)=0.
185  END WHERE
186  !$mnh_end_expand_where(JL=1:KSIZE)
187  ELSEIF(hsubg_pr_pdf=='HLCRECTPDF' .OR. hsubg_pr_pdf=='HLCISOTRIPDF' .OR. &
188  &hsubg_pr_pdf=='HLCTRIANGPDF' .OR. hsubg_pr_pdf=='HLCQUADRAPDF') THEN
189  ! Turner (2011, 2012)
190  ! Calculate maximum value r_cM from PDF forms
191  IF(hsubg_pr_pdf=='HLCRECTPDF' .OR. hsubg_pr_pdf=='HLCISOTRIPDF') THEN
192  zcoeffrcm=2.
193  ELSE IF(hsubg_pr_pdf=='HLCTRIANGPDF') THEN
194  zcoeffrcm=3.
195  ELSE IF(hsubg_pr_pdf=='HLCQUADRAPDF') THEN
196  zcoeffrcm=4.
197  END IF
198  !$mnh_expand_where(JL=1:KSIZE)
199  WHERE(prct(:).GT.0. .AND. pcf(:).GT.0. .AND. ldmicro(:))
200  zhlc_rcmax(:)=zcoeffrcm*prct(:)/pcf(:)
201  ELSEWHERE
202  zhlc_rcmax(:)=0.
203  END WHERE
204  !$mnh_end_expand_where(JL=1:KSIZE)
205 
206  ! Split available water and cloud fraction in two parts
207  ! Calculate local mean values int he low and high parts for the 3 PDF forms:
208  IF(hsubg_pr_pdf=='HLCRECTPDF') THEN
209  !$mnh_expand_where(JL=1:KSIZE)
210  WHERE(prct(:).GT.0. .AND. pcf(:).GT.0. .AND. zhlc_rcmax(:).GT.zrcrautc(:) .AND. ldmicro(:))
211  zhlc_lrclocal(:)=0.5*zrcrautc(:)
212  zhlc_hrclocal(:)=( zhlc_rcmax(:) + zrcrautc(:))/2.0
213  ELSEWHERE
214  zhlc_lrclocal(:)=0.
215  zhlc_hrclocal(:)=0.
216  END WHERE
217  !$mnh_end_expand_where(JL=1:KSIZE)
218  ELSE IF(hsubg_pr_pdf=='HLCTRIANGPDF') THEN
219  !$mnh_expand_where(JL=1:KSIZE)
220  WHERE(prct(:).GT.0. .AND. pcf(:).GT.0. .AND. zhlc_rcmax(:).GT.zrcrautc(:) .AND. ldmicro(:))
221  zhlc_lrclocal(:)=( zrcrautc(:) *(3.0 * zhlc_rcmax(:) - 2.0 * zrcrautc(:) ) ) &
222  / (3.0 * (2.0 * zhlc_rcmax(:) - zrcrautc(:) ) )
223  zhlc_hrclocal(:)=(zhlc_rcmax(:) + 2.0*zrcrautc(:)) / 3.0
224  ELSEWHERE
225  zhlc_lrclocal(:)=0.
226  zhlc_hrclocal(:)=0.
227  END WHERE
228  !$mnh_end_expand_where(JL=1:KSIZE)
229  ELSE IF(hsubg_pr_pdf=='HLCQUADRAPDF') THEN
230  !$mnh_expand_where(JL=1:KSIZE)
231  WHERE(prct(:).GT.0. .AND. pcf(:).GT.0. .AND. zhlc_rcmax(:).GT.zrcrautc(:) .AND. ldmicro(:))
232  zhlc_lrclocal(:)=(3.0 *zrcrautc(:)**3 - 8.0 *zrcrautc(:)**2 * zhlc_rcmax(:) &
233  + 6.0*zrcrautc(:) *zhlc_rcmax(:)**2 ) &
234  / &
235  (4.0* zrcrautc(:)**2 -12.0*zrcrautc(:) *zhlc_rcmax(:) &
236  + 12.0 * zhlc_rcmax(:)**2 )
237  zhlc_hrclocal(:)=(zhlc_rcmax(:) + 3.0*zrcrautc(:))/4.0
238  ELSEWHERE
239  zhlc_lrclocal(:)=0.
240  zhlc_hrclocal(:)=0.
241  END WHERE
242  !$mnh_end_expand_where(JL=1:KSIZE)
243  ELSE IF(hsubg_pr_pdf=='HLCISOTRIPDF') THEN
244  !$mnh_expand_where(JL=1:KSIZE)
245  WHERE (prct(:).LE.zrcrautc(:)*pcf(:) .AND. &
246  &prct(:).GT.0. .AND. pcf(:).GT.0. .AND. &
247  &zhlc_rcmax(:).GT.zrcrautc(:) .AND. ldmicro(:))
248  zhlc_lrclocal(:)=( (zhlc_rcmax(:))**3 &
249  -(12.0 * (zhlc_rcmax(:))*(zrcrautc(:))**2) &
250  +(8.0 * zrcrautc(:)**3) ) &
251  /( (6.0 * (zhlc_rcmax(:))**2) &
252  -(24.0 * (zhlc_rcmax(:)) * zrcrautc(:)) &
253  +(12.0 * zrcrautc(:)**2) )
254  zhlc_hrclocal(:)=( zhlc_rcmax(:) + 2.0 * zrcrautc(:) )/3.0
255  ELSEWHERE(prct(:).GT.0. .AND. pcf(:).GT.0. .AND. zhlc_rcmax(:).GT.zrcrautc(:) .AND. ldmicro(:))
256  zhlc_lrclocal(:)=(2.0/3.0) * zrcrautc(:)
257  zhlc_hrclocal(:)=(3.0*zhlc_rcmax(:)**3 - 8.0*zrcrautc(:)**3) &
258  / (6.0 * zhlc_rcmax(:)**2 - 12.0*zrcrautc(:)**2)
259  ELSEWHERE
260  zhlc_lrclocal(:)=0.
261  zhlc_hrclocal(:)=0.
262  END WHERE
263  !$mnh_end_expand_where(JL=1:KSIZE)
264  END IF
265  ! Compare r_cM to r_cR to know if cloud water content is high enough to split in two parts or not
266  !$mnh_expand_where(JL=1:KSIZE)
267  WHERE (prct(:).GT.0. .AND. pcf(:).GT.0. .AND. zhlc_rcmax(:).GT.zrcrautc(:) .AND. ldmicro(:))
268  ! Calculate final values for LCF and HCF:
269  phlc_lcf(:)=pcf(:) &
270  *(zhlc_hrclocal(:)- &
271  (prct(:) / pcf(:))) &
272  / (zhlc_hrclocal(:)-zhlc_lrclocal(:))
273  phlc_hcf(:)=max(0., pcf(:)-phlc_lcf(:))
274  !
275  ! Calculate final values for LRC and HRC:
276  phlc_lrc(:)=zhlc_lrclocal(:)*phlc_lcf(:)
277  phlc_hrc(:)=max(0., prct(:)-phlc_lrc(:))
278  ELSEWHERE (prct(:).GT.0. .AND. pcf(:).GT.0. .AND. zhlc_rcmax(:).LE.zrcrautc(:) .AND. ldmicro(:))
279  ! Put all available cloud water and his fraction in the low part
280  phlc_lcf(:)=pcf(:)
281  phlc_hcf(:)=0.
282  phlc_lrc(:)=prct(:)
283  phlc_hrc(:)=0.
284  ELSEWHERE
285  phlc_lcf(:)=0.
286  phlc_hcf(:)=0.
287  phlc_lrc(:)=0.
288  phlc_hrc(:)=0.
289  END WHERE
290  !$mnh_end_expand_where(JL=1:KSIZE)
291  ELSE
292  CALL print_msg(nverb_fatal,'GEN','ICE4_COMPUTE_PDF','wrong HSUBG_PR_PDF case')
293  ENDIF
294 ELSE
295  CALL print_msg(nverb_fatal,'GEN','ICE4_COMPUTE_PDF','wrong HSUBG_AUCV case')
296 ENDIF
297 !
298 !Ice water split between high and low content part is done according to autoconversion option
299 !$mnh_expand_where(JL=1:KSIZE)
300 WHERE(ldmicro(:))
301  zcriauti(:)=min(icep%XCRIAUTI,10**(icep%XACRIAUTI*(pt(:)-cst%XTT)+icep%XBCRIAUTI)) ! Autoconversion ri threshold
302 ELSEWHERE
303  zcriauti(:)=0.
304 ENDWHERE
305 !$mnh_end_expand_where(JL=1:KSIZE)
306 IF(hsubg_aucv_ri=='NONE') THEN
307  !Cloud water is entirely in low or high part
308  !$mnh_expand_where(JL=1:KSIZE)
309  WHERE(prit(:)>zcriauti(:) .AND. ldmicro(:))
310  phli_hcf(:)=1.
311  phli_lcf(:)=0.
312  phli_hri(:)=prit(:)
313  phli_lri(:)=0.
314  ELSEWHERE(prit(:)>iced%XRTMIN(4) .AND. ldmicro(:))
315  phli_hcf(:)=0.
316  phli_lcf(:)=1.
317  phli_hri(:)=0.
318  phli_lri(:)=prit(:)
319  ELSEWHERE
320  phli_hcf(:)=0.
321  phli_lcf(:)=0.
322  phli_hri(:)=0.
323  phli_lri(:)=0.
324  END WHERE
325  !$mnh_end_expand_where(JL=1:KSIZE)
326 ELSEIF(hsubg_aucv_ri=='CLFR') THEN
327  !Cloud water is only in the cloudy part and entirely in low or high part
328  !$mnh_expand_where(JL=1:KSIZE)
329  WHERE(pcf(:)>0. .AND. prit(:)>zcriauti(:)*pcf(:) .AND. ldmicro(:))
330  phli_hcf(:)=pcf(:)
331  phli_lcf(:)=0.
332  phli_hri(:)=prit(:)
333  phli_lri(:)=0.
334  ELSEWHERE(pcf(:)>0. .AND. prit(:)>iced%XRTMIN(4) .AND. ldmicro(:))
335  phli_hcf(:)=0.
336  phli_lcf(:)=pcf(:)
337  phli_hri(:)=0.0
338  phli_lri(:)=prit(:)
339  ELSEWHERE
340  phli_hcf(:)=0.
341  phli_lcf(:)=0.
342  phli_hri(:)=0.
343  phli_lri(:)=0.
344  END WHERE
345  !$mnh_end_expand_where(JL=1:KSIZE)
346 ELSEIF(hsubg_aucv_ri=='ADJU') THEN
347  !$mnh_expand_where(JL=1:KSIZE)
348  WHERE(ldmicro(:))
349  zsumri(:)=phli_lri(:)+phli_hri(:)
350  ELSEWHERE
351  zsumri(:)=0.
352  ENDWHERE
353  WHERE(zsumri(:) .GT. 0. .AND. ldmicro(:))
354  phli_lri(:)=phli_lri(:)*prit(:)/zsumri(:)
355  phli_hri(:)=phli_hri(:)*prit(:)/zsumri(:)
356  ELSEWHERE
357  phli_lri(:)=0.
358  phli_hri(:)=0.
359  ENDWHERE
360  !$mnh_end_expand_where(JL=1:KSIZE)
361 ELSE
362  !wrong HSUBG_AUCV_RI case
363  CALL print_msg( nverb_fatal, 'GEN', 'ICE4_COMPUTE_PDF', 'wrong HSUBG_AUCV_RI case' )
364 ENDIF
365 !
366 !$mnh_expand_where(JL=1:KSIZE)
367 WHERE(ldmicro(:))
368  prf(:)=max(phlc_hcf(:),phli_hcf(:))
369 ELSEWHERE
370  prf(:)=0.
371 ENDWHERE
372 !$mnh_end_expand_where(JL=1:KSIZE)
373 !
374 IF (lhook) CALL dr_hook('ICE4_COMPUTE_PDF', 1, zhook_handle)
375 END SUBROUTINE ice4_compute_pdf
