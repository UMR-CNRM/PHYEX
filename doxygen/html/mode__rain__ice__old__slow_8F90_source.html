102!* 3.2 compute the homogeneous nucleation source: RCHONI
103!
104 REAL, DIMENSION(KSIZE) :: ZBFT ! Mean time for a pristine ice crystal to reach
105 ! size of an snow/graupel particle (ZDICRIT)

117 zinvtstep=1./ptstep
118 zzw(:) = 0.0
119
120 IF (buconf%LBUDGET_TH) CALL budget_store_init_phy(d, tbudgets(nbudget_th), 'HON', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
121 IF (buconf%LBUDGET_RC) CALL budget_store_init_phy(d, tbudgets(nbudget_rc), 'HON', unpack(prcs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
122 IF (buconf%LBUDGET_RI) CALL budget_store_init_phy(d, tbudgets(nbudget_ri), 'HON', unpack(pris(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
123
124 DO jl = 1, ksize
125 IF ((pzt(jl)<cst%XTT-35.0) .AND. (prct(jl)>iced%XRTMIN(2)) .AND. (prcs(jl)>0.)) THEN
126 zzw(jl) = min( prcs(jl),icep%XHON*prhodref(jl)*prct(jl) &
127 *exp(icep%XALPHA3*(pzt(jl)-cst%XTT)-icep%XBETA3))
128 pris(jl) = pris(jl) + zzw(jl)
129 prcs(jl) = prcs(jl) - zzw(jl)
130 pzths(jl) = pzths(jl) + zzw(jl)*(plsfact(jl)-plvfact(jl)) ! f(L_f*(RCHONI))
131 END IF
132 END DO
133
134 IF (buconf%LBUDGET_TH) CALL budget_store_end_phy(d, tbudgets(nbudget_th), 'HON', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
135 IF (buconf%LBUDGET_RC) CALL budget_store_end_phy(d, tbudgets(nbudget_rc), 'HON', unpack(prcs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
136 IF (buconf%LBUDGET_RI) CALL budget_store_end_phy(d, tbudgets(nbudget_ri), 'HON', unpack(pris(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
137
138!* 3.3 compute the spontaneous freezing source: RRHONG
139
140 IF (buconf%LBUDGET_TH) CALL budget_store_init_phy(d, tbudgets(nbudget_th), 'SFR', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
141 IF (buconf%LBUDGET_RR) CALL budget_store_init_phy(d, tbudgets(nbudget_rr), 'SFR', unpack(prrs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
142 IF (buconf%LBUDGET_RG) CALL budget_store_init_phy(d, tbudgets(nbudget_rg), 'SFR', unpack(prgs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
143
144 zzw(:) = 0.0
145 DO jl = 1, ksize
146 IF ((pzt(jl)<cst%XTT-35.0) .AND. (prrt(jl)>iced%XRTMIN(3)) .AND. (prrs(jl)>0.)) THEN
147 zzw(jl) = min( prrs(jl),prrt(jl)* zinvtstep )
148 prgs(jl) = prgs(jl) + zzw(jl)
149 prrs(jl) = prrs(jl) - zzw(jl)
150 pzths(jl) = pzths(jl) + zzw(jl)*(plsfact(jl)-plvfact(jl)) ! f(L_f*(RRHONG))
151 END IF
152 END DO
153
154 IF (buconf%LBUDGET_TH) CALL budget_store_end_phy(d, tbudgets(nbudget_th), 'SFR', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
155 IF (buconf%LBUDGET_RR) CALL budget_store_end_phy(d, tbudgets(nbudget_rr), 'SFR', unpack(prrs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
156 IF (buconf%LBUDGET_RG) CALL budget_store_end_phy(d, tbudgets(nbudget_rg), 'SFR', unpack(prgs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
157
158!* 3.4 compute the deposition, aggregation and autoconversion sources
159
160 pka(:) = 2.38e-2 + 0.0071e-2 * ( pzt(:) - cst%XTT ) ! k_a
161 pdv(:) = 0.211e-4 * (pzt(:)/cst%XTT)**1.94 * (cst%XP00/ppres(:)) ! D_v
162!
163!* 3.4.1 compute the thermodynamical function A_i(T,P)
164!* and the c^prime_j (in the ventilation factor)
165!
166 IF(ocnd2)THEN
167 pai(:) = paa2(:) + pbb3(:)*ppres(:)
168 ELSE
169 pai(:) = exp( cst%XALPI - cst%XBETAI/pzt(:) - cst%XGAMI*log(pzt(:) ) ) ! es_i
170 pai(:) = ( cst%XLSTT + (cst%XCPV-cst%XCI)*(pzt(:)-cst%XTT) )**2 / (pka(:)*cst%XRV*pzt(:)**2) &
171 + ( cst%XRV*pzt(:) ) / (pdv(:)*pai(:))
172 ENDIF
173 pcj(:) = icep%XSCFAC * prhodref(:)**0.3 / sqrt( 1.718e-5+0.0049e-5*(pzt(:)-cst%XTT) )
174!
175!* 3.4.3 compute the deposition on r_s: RVDEPS
176!
177 DO jl = 1, ksize
178 IF (prst(jl)>0.0) THEN
179 pldbas(jl) = min( iced%XLBDAS_MAX, &
180 iced%XLBS*( prhodref(jl)*max( prst(jl),iced%XRTMIN(5) ) )**iced%XLBEXS )
181 END IF
182 END DO
183 zzw(:) = 0.0
184
185 IF (buconf%LBUDGET_TH) CALL budget_store_init_phy(d, tbudgets(nbudget_th), 'DEPS', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
186 IF (buconf%LBUDGET_RV) CALL budget_store_init_phy(d, tbudgets(nbudget_rv), 'DEPS', unpack(pzrvs(:),mask=gmicro(:,:),field=prvs)*prhodj(:,:))
187 IF (buconf%LBUDGET_RS) CALL budget_store_init_phy(d, tbudgets(nbudget_rs), 'DEPS', unpack(prss(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
188
189 IF(ocnd2)THEN
190 DO jl = 1, ksize
191 IF ((prst(jl)>iced%XRTMIN(5)) .AND. (prss(jl)>0.0)) THEN
192 zzw(jl) = (pssi(jl)/(prhodref(jl)*pai(jl))) * &
193 (icep%X0DEPS*pldbas(jl)**icep%XEX0DEPS + icep%X1DEPS*pcj(jl)*pldbas(jl)**icep%XEX1DEPS)
194 zzw(jl) = min( pzrvs(jl),max(-prss(jl),zzw(jl))) ! Simpler
195 zzw(jl) = zzw(jl)*zredsn ! Possible tuning by using ZREDSN /= 1
196 prss(jl) = prss(jl) + zzw(jl)
197 pzrvs(jl) = pzrvs(jl) - zzw(jl)
198 pzths(jl) = pzths(jl) + zzw(jl)*plsfact(jl)
199 END IF
200 END DO
201 ELSE
202 DO jl = 1, ksize
203 IF ((prst(jl)>iced%XRTMIN(5)) .AND. (prss(jl)>0.0)) THEN
204 zzw(jl) = ( pssi(jl)/(prhodref(jl)*pai(jl)) ) * &
205 ( icep%X0DEPS*pldbas(jl)**icep%XEX0DEPS + icep%X1DEPS*pcj(jl)*pldbas(jl)**icep%XEX1DEPS )
206 zzw(jl) = min(pzrvs(jl),zzw(jl) )*(0.5+sign(0.5,zzw(jl))) &
207 - min(prss(jl),abs(zzw(jl)))*(0.5-sign(0.5,zzw(jl)))
208
209 IF (zzw(jl) < 0.0) THEN
210 zzw(jl) = zzw(jl) * icep%XRDEPSRED
211 END IF
212
213 prss(jl) = prss(jl) + zzw(jl)
214 pzrvs(jl) = pzrvs(jl) - zzw(jl)
215 pzths(jl) = pzths(jl) + zzw(jl)*plsfact(jl)
216 END IF
217 END DO
218 ENDIF
219
220 IF (buconf%LBUDGET_TH) CALL budget_store_end_phy(d, tbudgets(nbudget_th), 'DEPS', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
221 IF (buconf%LBUDGET_RV) CALL budget_store_end_phy(d, tbudgets(nbudget_rv), 'DEPS', unpack(pzrvs(:),mask=gmicro(:,:),field=prvs)*prhodj(:,:))
222 IF (buconf%LBUDGET_RS) CALL budget_store_end_phy(d, tbudgets(nbudget_rs), 'DEPS', unpack(prss(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
223
224!* 3.4.4 compute the aggregation on r_s: RIAGGS
225
226 IF (buconf%LBUDGET_RI) CALL budget_store_init_phy(d, tbudgets(nbudget_ri), 'AGGS', unpack(pris(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
227 IF (buconf%LBUDGET_RS) CALL budget_store_init_phy(d, tbudgets(nbudget_rs), 'AGGS', unpack(prss(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
228
229 zzw(:) = 0.0
230 DO jl = 1, ksize
231 IF ((prit(jl)>iced%XRTMIN(4)) .AND. (prst(jl)>iced%XRTMIN(5)) .AND. (pris(jl)>0.0)) THEN
232 zzw(jl) = min(pris(jl),icep%XFIAGGS * exp( icep%XCOLEXIS*(pzt(jl)-cst%XTT)) &
233 * prit(jl) &
234 * pldbas(jl)**icep%XEXIAGGS &
235 * prhodref(jl)**(-iced%XCEXVT))
236 prss(jl) = prss(jl) + zzw(jl)
237 pris(jl) = pris(jl) - zzw(jl)
238 END IF
239 END DO
240
241 IF (buconf%LBUDGET_RI) CALL budget_store_end_phy(d, tbudgets(nbudget_ri), 'AGGS', unpack(pris(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
242 IF (buconf%LBUDGET_RS) CALL budget_store_end_phy(d, tbudgets(nbudget_rs), 'AGGS', unpack(prss(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
243
244!* 3.4.5 compute the autoconversion of r_i for r_s production: RIAUTS
245
246 zcriauti(:)=min(icep%XCRIAUTI,10**(icep%XACRIAUTI*(pzt(:)-cst%XTT)+icep%XBCRIAUTI))
247 zzw(:) = 0.0
248 IF (buconf%LBUDGET_RI) CALL budget_store_init_phy(d, tbudgets(nbudget_ri), 'AUTS', unpack(pris(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
249 IF (buconf%LBUDGET_RS) CALL budget_store_init_phy(d, tbudgets(nbudget_rs), 'AUTS', unpack(prss(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
250 DO jl = 1, ksize
251 IF ((prit(jl)>iced%XRTMIN(4)) .AND. (pris(jl)>0.0)) THEN
252 zzw(jl) = min(pris(jl),icep%XTIMAUTI * exp(icep%XTEXAUTI*(pzt(jl)-cst%XTT)) &
253 * max(prit(jl)-zcriauti(jl),0.0 ))
254 prss(jl) = prss(jl) + zzw(jl)
255 pris(jl) = pris(jl) - zzw(jl)
256 END IF
257 END DO
258
259 IF (ocnd2 .AND. .NOT. lmodicedep) THEN ! 3.4.5 B:
260
261 ! Turn ice crystals lagrer than a precribed size into snow:
262 ! (For the moment sperical ice crystals are assumed)
263
264 DO jl = 1, ksize
265 IF ((pris(jl)>0.0_jprb) .AND. (pssi(jl)>0.001_jprb)) THEN
266 zbft(jl) = 0.5_jprb*87.5_jprb*(zdicrit)**2*pai(jl)/ pssi(jl)
267 zbft(jl) = ptstep/ max(ptstep,zbft(jl)*2._jprb)
268 prss(jl) = prss(jl) + zbft(jl)*pris(jl)
269 pris(jl) = pris(jl) - zbft(jl)*pris(jl)
270 END IF
271 END DO
272 ENDIF
273
274 IF (ocnd2 .AND. lmodicedep) THEN ! 3.4.5 B:
275
276 ! Turn ice to snow if ice crystal distrubution is such that
277 ! the ice crystal diameter for the (mass x N_i) maximum
278 ! is lagrer than a precribed size.
279 ! (ZDICRIT) The general gamma function is assumed
280
281 DO jl=1,ksize
282 zzw2(jl) = &
283 max(pcit(jl),icenumber2(pris(jl)*ptstep,pzt(jl))*prhodref(jl))
284 ENDDO
285
286 DO jl = 1, ksize
287 IF (pris(jl)>icep%XFRMIN(13) .AND.pcit(jl) > 0.) THEN
288 ! LAMBDA for ICE
289 zzw2(jl) = min(1.e8,iced%XLBI*(prhodref(jl)*pris(jl)* ptstep/zzw2(jl))**iced%XLBEXI)
290 zbft(jl) = 1. - 0.5**(zkvo /zzw2(jl))
291 zbft(jl) = min(0.9*pris(jl)*ptstep, zbft(jl)*pris(jl)*ptstep)
292 prss(jl) = prss(jl) + zbft(jl)
293 pris(jl) = pris(jl) - zbft(jl)
294 END IF
295 END DO
296 ENDIF
297
298 IF (buconf%LBUDGET_RI) CALL budget_store_end_phy(d, tbudgets(nbudget_ri), 'AUTS', unpack(pris(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
299 IF (buconf%LBUDGET_RS) CALL budget_store_end_phy(d, tbudgets(nbudget_rs), 'AUTS', unpack(prss(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
300
301!* 3.4.6 compute the deposition on r_g: RVDEPG
302
303 zzw2(:) = 0.0
304 IF (icep%XFRMIN(5)> 1.0e-12 .AND. icep%XFRMIN(6) > 0.01) THEN
305 zzw2(:) = max(0., min(1., (icep%XFRMIN(5) - prgs(:))/icep%XFRMIN(5)))* &
306 & max(0., min(1., pssi(:)/icep%XFRMIN(6)))
307 ENDIF
308
309
310 DO jl = 1, ksize
311 IF (prgt(jl)>0.0) THEN
312 plbdag(jl) = iced%XLBG*(prhodref(jl)*max(prgt(jl), iced%XRTMIN(6)))**iced%XLBEXG
313 END IF
314 END DO
315
316 IF (buconf%LBUDGET_TH) CALL budget_store_init_phy(d, tbudgets(nbudget_th), 'DEPG', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
317 IF (buconf%LBUDGET_RV) CALL budget_store_init_phy(d, tbudgets(nbudget_rv), 'DEPG', unpack(pzrvs(:),mask=gmicro(:,:),field=prvs)*prhodj(:,:))
318 IF (buconf%LBUDGET_RG) CALL budget_store_init_phy(d, tbudgets(nbudget_rg), 'DEPG', unpack(prgs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
319
320 zzw(:) = 0.0
321 DO jl = 1, ksize
322 IF ((prgt(jl)>iced%XRTMIN(6)) .AND. (prgs(jl)>0.0)) THEN
323 zzw(jl) = (pssi(jl)/(prhodref(jl)*pai(jl))) * &
324 (icep%X0DEPG*plbdag(jl)**icep%XEX0DEPG + &
325 icep%X1DEPG*pcj(jl)*plbdag(jl)**icep%XEX1DEPG)
326
327 zzw(jl) = min(pzrvs(jl),zzw(jl) )*(0.5+sign(0.5,zzw(jl))) &
328 - min(prgs(jl),abs(zzw(jl)) )*(0.5-sign(0.5,zzw(jl)))
329 zzw(jl) = zzw(jl)*zredgr
330
331 IF (zzw(jl) < 0.0 ) THEN
332 zzw(jl) = zzw(jl) * icep%XRDEPGRED
333 END IF
334
335 prss(jl) = (zzw(jl) + prgs(jl))* zzw2(jl) + prss(jl)
336 prgs(jl) = (zzw(jl) + prgs(jl))*(1. - zzw2(jl))
337 pzrvs(jl) = pzrvs(jl) - zzw(jl)
338 pzths(jl) = pzths(jl) + zzw(jl)*plsfact(jl)
339 END IF
340 END DO
341
342 DO jl = 1, ksize
343 IF (zzw(jl) < 0.0) THEN
344 zzw(jl) = zzw(jl) * icep%XRDEPGRED
345 END IF
346 END DO
347
348 IF (buconf%LBUDGET_TH) CALL budget_store_end_phy(d, tbudgets(nbudget_th), 'DEPG', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
349 IF (buconf%LBUDGET_RV) CALL budget_store_end_phy(d, tbudgets(nbudget_rv), 'DEPG', unpack(pzrvs(:),mask=gmicro(:,:),field=prvs)*prhodj(:,:))
350 IF (buconf%LBUDGET_RG) CALL budget_store_end_phy(d, tbudgets(nbudget_rg), 'DEPG', unpack(prgs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
351
352 IF (lhook) CALL dr_hook('RAIN_ICE_OLD:RAIN_ICE_SLOW',1,zhook_handle)
353
354 END SUBROUTINE rain_ice_old_slow
