58CHARACTER (LEN=5),SAVE :: cconf ! Configuration of models
59 ! 'START' for start configuration (variables
60 ! at time t and t-dt are the same in the
61 ! initial file)
62 ! 'RESTART' for restart configuration
63 ! (variables at time t and t-dt are different)
64 ! 'POST' for post-treatment configuration
65LOGICAL,SAVE :: lthinshell ! Logical for thinshell approximation
66 ! .TRUE. = thinshell approximation
67 ! .FALSE. = no thinshell approximation
68LOGICAL,SAVE :: lcartesian ! Logical for cartesian geometry :
69 ! .TRUE. = cartesian geometry
70 ! .FALSE. = conformal projection
71LOGICAL,SAVE :: l2d=.false. ! Logical for 2D model version
72 ! .TRUE. = 2D model version
73 ! .FALSE. = 3D model version
74LOGICAL,SAVE :: l1d ! Logical for 1D model version
75 ! .TRUE. = 1D model version
76 ! .FALSE. = 2D or 3D model version
77LOGICAL,SAVE :: lflat ! Logical for zero ororography
78 ! .TRUE. = no orography (zs=0.)
79 ! .FALSE. = orography
80INTEGER,SAVE :: nmodel ! Number of nested models
81INTEGER,SAVE :: nverb ! Level of informations on output-listing
82 ! 0 for minimum of prints
83 ! 5 for intermediate level of prints
84 ! 10 for maximum of prints
85CHARACTER (LEN=5),SAVE :: cexp ! Experiment name
86CHARACTER (LEN=5),SAVE :: cseg ! name of segment
87LOGICAL,SAVE :: lforcing ! Logical for forcing sources
88 ! .TRUE. = add forcing sources
89 ! .FALSE. = no forcing fields
90!
91CHARACTER (LEN=3),SAVE :: ceqnsys! EQuatioN SYStem resolved by the MESONH model
92 ! 'LHE' Lipps and HEmler anelastic system
93 ! 'DUR' approximated form of the DURran version
94 ! of the anelastic sytem
95 ! 'MAE' classical Modified Anelastic Equations
96 ! but with not any approximation in the
97 ! momentum equation
98 ! 'FCE' fully compressible equations ( not
99 ! yet developped )
100LOGICAL,SAVE :: lpack ! Logical to compress 1D or 2D FM files
101!
102!
103INTEGER,DIMENSION(3),SAVE :: nmnhversion ! Version of MesoNH
104INTEGER,SAVE :: nmasdev ! NMASDEV=XY corresponds to the masdevX_Y

106CHARACTER(LEN=10),SAVE :: cbibuser! CBIBUSER is the name of the user binary library
107!
108CHARACTER(LEN=6),SAVE :: cprogram ! CPROGRAM is the program currently running:
109! ! 'PGD ','ADVPGD','NESPGD','REAL ','IDEAL '
110! ! 'MESONH','SPAWN ','DIAG ','SPEC '
111!
112INTEGER,SAVE :: nhalo ! Size of the halo for parallel distribution
113!
114CHARACTER (LEN=10),SAVE :: csplit ! kind of domain splitting for parallel distribution
115 ! "BSPLITTING","XSPLITTING","YSPLITTING"
116LOGICAL,SAVE :: llg ! Logical to use lagrangian variables
117LOGICAL,SAVE :: linit_lg ! to reinitialize lagrangian variables
118CHARACTER (LEN=5),SAVE :: cinit_lg ! to reinitialize LG variables at every output
119LOGICAL,SAVE :: lnomixlg ! to use turbulence for lagrangian variables
120!
121LOGICAL,SAVE :: lneutral ! True if ref. theta field is uniform
122!
123LOGICAL,SAVE :: lcpl_arome ! true if coupling file are issued from AROME
124LOGICAL,SAVE :: lcoupling ! true if coupling file (and not intial file)

126!
127LOGICAL,SAVE :: lcheck ! To test reproducibility
128!
