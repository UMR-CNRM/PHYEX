15SUBROUTINE bl_depth_diag_3d(D,KDIM, PSURF,PZS,PFLUX,PZZ,PFTOP_O_FSURF,BL_DEPTH_DIAG3D)

17!
18!
19!!**** *SBL_DEPTH* - computes SBL depth
20!!
21!! PURPOSE
22!! -------
23!
24!!** METHOD
25!! ------
26!!
27!! SBL is defined as the layer where momentum flux is equal to XSBL_FRAC of its surface value
28!!
29!! EXTERNAL
30!! --------
31!!
32!! IMPLICIT ARGUMENTS
33!! ------------------
34!!
35!!
36!! REFERENCE
37!! ---------
38!!
39!! AUTHOR
40!! ------
41!! V. Masson * Meteo-France *
42!!
43!! MODIFICATIONS
44!! -------------
45!! Original nov. 2005
46!!
47!! --------------------------------------------------------------------------
48!
49!* 0. DECLARATIONS
50! ------------
51!
52!* 0.1 declarations of arguments
53!
54USE modd_dimphyex, ONLY: dimphyex_t
55!
56IMPLICIT NONE
57!
58TYPE(dimphyex_t), INTENT(IN) :: D
59INTEGER, INTENT(IN) :: KDIM ! Vertical dimensions (LES grid vs full grid)
60REAL, DIMENSION(D%NIJT), INTENT(IN) :: PSURF ! surface flux
61REAL, DIMENSION(D%NIJT), INTENT(IN) :: PZS ! orography
62REAL, DIMENSION(D%NIJT,KDIM), INTENT(IN) :: PFLUX ! flux
63REAL, DIMENSION(D%NIJT,KDIM), INTENT(IN) :: PZZ ! altitude of flux points
64REAL, INTENT(IN) :: PFTOP_O_FSURF! Flux at BL top / Surface flux
65REAL, DIMENSION(D%NIJT), INTENT(OUT) :: BL_DEPTH_DIAG3D
66!
67!
68! 0.2 declaration of local variables
69!
70INTEGER :: JIJ,JK ! loop counters
71INTEGER :: IKB,IKE,IIJB,IIJE,IKL
72REAL :: ZFLX ! flux at top of BL
73!
74!----------------------------------------------------------------------------
75!
76REAL(KIND=jphook) :: zhook_handle
77IF (lhook) CALL dr_hook('BL_DEPTH_DIAG_3D',0,zhook_handle)
78ikb=d%NKTB
79ike=d%NKTE
80ikl=d%NKL
81iije=d%NIJE
82iijb=d%NIJB
83!
84bl_depth_diag3d(:) = 0.
85!
86
87DO jij=iijb,iije
88 IF (psurf(jij)/=0.) THEN
89 DO jk=ikb,ike,ikl
90 IF (pzz(jij,jk-ikl)>pzs(jij)) THEN
91 zflx = psurf(jij) * pftop_o_fsurf
92 IF ( (pflux(jij,jk)-zflx)*(pflux(jij,jk-ikl)-zflx) <= 0. ) THEN
93 bl_depth_diag3d(jij) = (pzz(jij,jk-ikl) - pzs(jij)) &
94 + (pzz(jij,jk) - pzz(jij,jk-ikl)) &
95 * (zflx - pflux(jij,jk-ikl) ) &
96 / (pflux(jij,jk) - pflux(jij,jk-ikl) )
97 END IF
98 END IF
99 END DO
100 END IF
101END DO
102!
103!$mnh_expand_array(JIJ=IIJB:IIJE)
104bl_depth_diag3d(iijb:iije) = bl_depth_diag3d(iijb:iije) / (1. - pftop_o_fsurf)
105!$mnh_end_expand_array(JIJ=IIJB:IIJE)
106!
107IF (lhook) CALL dr_hook('BL_DEPTH_DIAG_3D',1,zhook_handle)
108END SUBROUTINE bl_depth_diag_3d

110SUBROUTINE bl_depth_diag_1d(D,KDIM, PSURF,PZS,PFLUX,PZZ,PFTOP_O_FSURF,BL_DEPTH_DIAG1D)

128!
129REAL(kind=jphook) :: zhook_handle
130IF (lhook) CALL dr_hook('BL_DEPTH_DIAG_1D',0,zhook_handle)
131!
132ikb=d%NKTB
133ikl=d%NKL
134!
135bl_depth_diag1d = 0.
136IF (psurf/=0.) THEN
137 DO jk=ikb,kdim-1,ikl
138 IF (pzz(jk-ikl)>pzs) THEN
139 zflx = psurf * pftop_o_fsurf
140 IF ( (pflux(jk)-zflx)*(pflux(jk-ikl)-zflx) <= 0. ) THEN
141 bl_depth_diag1d = (pzz(jk-ikl) - pzs) &
142 + (pzz(jk) - pzz(jk-ikl)) &
143 * (zflx - pflux(jk-ikl) ) &
144 / (pflux(jk) - pflux(jk-ikl) )
145 END IF
146 END IF
147 END DO
148END IF
149!
150!-------------------------------------------------------------------------------
151!
152IF (lhook) CALL dr_hook('BL_DEPTH_DIAG_1D',1,zhook_handle)
