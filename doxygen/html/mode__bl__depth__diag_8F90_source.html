20 !!
21 !! PURPOSE
22 !! -------
23 !
24 !!** METHOD
25 !! ------
26 !!
27 !! SBL is defined as the layer where momentum flux is equal to XSBL_FRAC of its surface value
28 !!
29 !! EXTERNAL
30 !! --------
31 !!
32 !! IMPLICIT ARGUMENTS
33 !! ------------------
34 !!
35 !!

43 !! MODIFICATIONS

49 !* 0. DECLARATIONS
50 ! ------------
51 !
52 !* 0.1 declarations of arguments
53 !
54 USE modd_dimphyex, ONLY: dimphyex_t
55 !
56 IMPLICIT NONE
57 !
58 TYPE(dimphyex_t), INTENT(IN) :: D
59 REAL, DIMENSION(D%NIJT), INTENT(IN) :: PSURF ! surface flux
60 REAL, DIMENSION(D%NIJT), INTENT(IN) :: PZS ! orography
61 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PFLUX ! flux
62 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PZZ ! altitude of flux points
63 REAL, INTENT(IN) :: PFTOP_O_FSURF! Flux at BL top / Surface flux
64 REAL, DIMENSION(D%NIJT), INTENT(OUT) :: BL_DEPTH_DIAG3D
65 !
66 !
67 ! 0.2 declaration of local variables
68 !
69 INTEGER :: JIJ,JK ! loop counters
70 INTEGER :: IKB,IKE,IIJB,IIJE,IKL
71 REAL :: ZFLX ! flux at top of BL
72 !
73 !----------------------------------------------------------------------------
74 !
75 REAL(KIND=jphook) :: zhook_handle
76 IF (lhook) CALL dr_hook('BL_DEPTH_DIAG_3D',0,zhook_handle)
77 ikb=d%NKTB
78 ike=d%NKTE
79 ikl=d%NKL
80 iije=d%NIJE
81 iijb=d%NIJB
82 !
83 bl_depth_diag3d(:) = 0.
84 !
85 
86 DO jij=iijb,iije
87  IF (psurf(jij)/=0.) THEN
88  DO jk=ikb,ike,ikl
89  IF (pzz(jij,jk-ikl)>pzs(jij)) THEN
90  zflx = psurf(jij) * pftop_o_fsurf
91  IF ( (pflux(jij,jk)-zflx)*(pflux(jij,jk-ikl)-zflx) <= 0. ) THEN
92  bl_depth_diag3d(jij) = (pzz(jij,jk-ikl) - pzs(jij)) &
93  + (pzz(jij,jk) - pzz(jij,jk-ikl)) &
94  * (zflx - pflux(jij,jk-ikl) ) &
95  / (pflux(jij,jk) - pflux(jij,jk-ikl) )
96  END IF
97  END IF
98  END DO
99  END IF
100 END DO
101 !
102 !$mnh_expand_array(JIJ=IIJB:IIJE)
103 bl_depth_diag3d(iijb:iije) = bl_depth_diag3d(iijb:iije) / (1. - pftop_o_fsurf)
104 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
105 !
106 IF (lhook) CALL dr_hook('BL_DEPTH_DIAG_3D',1,zhook_handle)
107 END SUBROUTINE bl_depth_diag_3d
