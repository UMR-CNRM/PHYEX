296 SUBROUTINE lmo_2d(PUSTAR,PTHETA,PRV,PSFTH,PSFRV,LMO2D)
297  USE modd_cst
298  USE modd_parameters, ONLY: xundef
299  REAL, DIMENSION(:,:), INTENT(IN) :: PUSTAR
300  REAL, DIMENSION(:,:), INTENT(IN) :: PTHETA
301  REAL, DIMENSION(:,:), INTENT(IN) :: PRV
302  REAL, DIMENSION(:,:), INTENT(IN) :: PSFTH
303  REAL, DIMENSION(:,:), INTENT(IN) :: PSFRV
304  REAL, DIMENSION(SIZE(PUSTAR,1),SIZE(PUSTAR,2)),INTENT(OUT) :: LMO2D
305 !
306  REAL, DIMENSION(SIZE(PUSTAR,1),SIZE(PUSTAR,2)) :: ZTHETAV
307  REAL, DIMENSION(SIZE(PUSTAR,1),SIZE(PUSTAR,2)) :: ZQ0
308  REAL :: ZEPS
309 !
310 !
311  REAL(KIND=jphook) :: zhook_handle
312  IF (lhook) CALL dr_hook('MODE_SBL:LMO_2D',0,zhook_handle)
313  zeps=(xrv-xrd)/xrd
314  zthetav(:,:) = ptheta(:,:) * ( 1. +zeps * prv(:,:))
315  zq0(:,:) = psfth(:,:) + zthetav(:,:) * zeps * psfrv(:,:)
316 !
317  lmo2d(:,:) = xundef
318  WHERE ( zq0(:,:) /=0. ) &
319  lmo2d(:,:) = - max(pustar(:,:),1.e-6)**3 &
320  / ( xkarman * xg / zthetav(:,:) *zq0(:,:) )
321 
322  IF (lhook) CALL dr_hook('MODE_SBL:LMO_2D',1,zhook_handle)
323 END SUBROUTINE lmo_2d

389 SUBROUTINE ustar_2d(PU,PV,PZ,PZ0,PLMO,USTAR2D)
390  USE modd_cst
391  USE modd_parameters, ONLY: xundef
392  REAL, DIMENSION(:,:), INTENT(IN) :: PU
393  REAL, DIMENSION(:,:), INTENT(IN) :: PV
394  REAL, DIMENSION(:,:), INTENT(IN) :: PZ
395  REAL, DIMENSION(:,:), INTENT(IN) :: PZ0
396  REAL, DIMENSION(:,:), INTENT(IN) :: PLMO
397  REAL, DIMENSION(SIZE(PU,1),SIZE(PU,2)),INTENT(OUT) :: USTAR2D
398 
399  REAL, DIMENSION(SIZE(PU,1),SIZE(PU,2)) :: ZZ_O_LMO
400  REAL, DIMENSION(SIZE(PU,1),SIZE(PU,2)) :: ZZ0_O_LMO
401  REAL, DIMENSION(SIZE(PU,1),SIZE(PU,2)) :: ZWORK1,ZWORK2
402 !
403 !* purely unstable case
404  REAL(KIND=jphook) :: zhook_handle
405  IF (lhook) CALL dr_hook('MODE_SBL:USTAR_2D',0,zhook_handle)
406  ustar2d(:,:) = 0.
407  zz_o_lmo(:,:) = xundef
408  zz0_o_lmo(:,:) = xundef
409 !
410 !* general case
411  CALL paulson_psim(zz_o_lmo,zwork1)
412  CALL paulson_psim(zz0_o_lmo,zwork2)
413  WHERE(abs(plmo) > 1.e-20 .AND. plmo/=xundef)
414  zz_o_lmo = pz(:,:) / plmo(:,:)
415  zz0_o_lmo = pz0(:,:) / plmo(:,:)
416  ustar2d(:,:) = sqrt( pu(:,:)**2+pv(:,:)**2 ) &
417  * xkarman / ( log(pz(:,:)/pz0(:,:)) &
418  - zwork1(:,:) + zwork2(:,:))
419  END WHERE
420 !
421 !* purely neutral case
422  WHERE(plmo==xundef)
423  zz_o_lmo = 0.
424  ustar2d(:,:) = sqrt( pu(:,:)**2+pv(:,:)**2 ) &
425  * xkarman / log(pz(:,:)/pz0(:,:))
426  END WHERE
427 !
428  IF (lhook) CALL dr_hook('MODE_SBL:USTAR_2D',1,zhook_handle)
429 END SUBROUTINE ustar_2d

433 SUBROUTINE ustar_1d(PU,PV,PZ,PZ0,PLMO,USTAR1D)
434  USE modd_cst
435  USE modd_parameters, ONLY: xundef
436  REAL, DIMENSION(:), INTENT(IN) :: PU
437  REAL, DIMENSION(:), INTENT(IN) :: PV
438  REAL, DIMENSION(:), INTENT(IN) :: PZ
439  REAL, DIMENSION(:), INTENT(IN) :: PZ0
440  REAL, DIMENSION(:), INTENT(IN) :: PLMO
441  REAL, DIMENSION(SIZE(PU)),INTENT(OUT) :: USTAR1D
442 
443  REAL, DIMENSION(SIZE(PU)) :: ZZ_O_LMO
444  REAL, DIMENSION(SIZE(PU)) :: ZZ0_O_LMO
445  REAL, DIMENSION(SIZE(PU)) :: ZWORK1,ZWORK2
446 !
447 !* purely unstable case
448  REAL(KIND=jphook) :: zhook_handle
449  IF (lhook) CALL dr_hook('MODE_SBL:USTAR_1D',0,zhook_handle)
450  ustar1d(:) = 0.
451  zz_o_lmo(:) = xundef
452  zz0_o_lmo(:) = xundef
453 !
454 !* general case
455  CALL paulson_psim(zz_o_lmo,zwork1)
456  CALL paulson_psim(zz0_o_lmo,zwork2)
457  WHERE(abs(plmo) > 1.e-20 .AND. plmo/=xundef)
458  zz_o_lmo = pz(:) / plmo(:)
459  zz0_o_lmo = pz0(:) / plmo(:)
460  ustar1d(:) = sqrt( pu(:)**2+pv(:)**2 ) &
461  * xkarman / ( log(pz(:)/pz0(:)) &
462  - zwork1(:) + zwork2(:))
463  END WHERE
464 !
465 !* purely neutral case
466  WHERE(plmo==xundef)
467  zz_o_lmo = 0.
468  ustar1d(:) = sqrt( pu(:)**2+pv(:)**2 ) &
469  * xkarman / log(pz(:)/pz0(:))
