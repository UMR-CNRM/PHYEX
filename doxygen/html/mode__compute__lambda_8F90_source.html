14  SUBROUTINE compute_lambda (KID, KMOMENT, KSIZE, HCLOUD, &
15  PRHO, PRTMIN, PRX, PCX, PLBDX)
16 ! #########################################################
17 !
18 ! Purpose : compute lambda, the slope parameter of the distribution
19 ! - for 1-moment species: lbda_x = [(rho r_x) / (a_x C_x G(b/alpha))]^(1/(x-b))
20 ! - for 2-moment species: lbda_x = [(rho r_x) / (a_x N_x G(b/alpha))]^(-1/b)
21 !
22 ! AUTHOR
23 ! ------
24 ! C. Barthe * LAERO *
25 !
26 ! MODIFICATIONS
27 ! -------------
28 ! Original June 2022
29 ! C. Barthe 12/07/23 adapt the code for LIMA2
30 !
31 !------------------------------------------------------------------
32 !
33 !* 0. DECLARATIONS
34 ! ------------
35 !
36 USE modd_rain_ice_descr_n, ONLY: xlbc_i=>xlbc, xlbr_i=>xlbr, xlbi_i=>xlbi, xlbs_i=>xlbs, xlbg_i=>xlbg, xlbh_i=>xlbh, &
37  xlbexc_i=>xlbexc, xlbexr_i=>xlbexr, xlbexi_i=>xlbexi, xlbexs_i=>xlbexs, &
38  xlbexg_i=>xlbexg, xlbexh_i=>xlbexh, &
39  xlbdas_max_i=>xlbdas_max, &
40  xccr_i=>xccr, xccs_i=>xccs, xccg_i=>xccg, xcch_i=>xcch, &
41  xcxs_i=>xcxs, xcxg_i=>xcxg, xcxh_i=>xcxh
42 USE modd_elec_descr, ONLY: xcxr_i=>xcxr
43 USE modd_param_lima_warm, ONLY: xlbc_l=>xlbc, xlbr_l=>xlbr, xlbexc_l=>xlbexc, xlbexr_l=>xlbexr, &
44  xcxr_l=>xcxr, xccr_l=>xccr
45 USE modd_param_lima_cold, ONLY: xlbi_l=>xlbi, xlbs_l=>xlbs, xlbexi_l=>xlbexi, xlbexs_l=>xlbexs, &
46  xlbdas_max_l=>xlbdas_max, &
47  xcxs_l=>xcxs, xccs_l=>xccs
48 USE modd_param_lima_mixed, ONLY: xlbg_l=>xlbg, xlbh_l=>xlbh, xlbexg_l=>xlbexg, xlbexh_l=>xlbexh, &
49  xcxg_l=>xcxg, xccg_l=>xccg, xcxh_l=>xcxh, xcch_l=>xcch
50 !
51 USE modi_momg
52 !
53 IMPLICIT NONE
54 !
55 !* 0.1 Declaration of dummy arguments
56 !
57 INTEGER, INTENT(IN) :: KID ! nb of the hydrometeor category
58 CHARACTER (LEN=4), INTENT(IN) :: HCLOUD ! Kind of microphysical scheme
59 INTEGER, INTENT(IN) :: KMOMENT ! nb of moments of the microphysics scheme
60 INTEGER, INTENT(IN) :: KSIZE
61 REAL, INTENT(IN) :: PRTMIN
62 REAL, DIMENSION(KSIZE), INTENT(IN) :: PRHO ! reference density
63 REAL, DIMENSION(KSIZE), INTENT(IN) :: PRX ! Mixing ratio
64 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PCX ! Nb concentration
65 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PLBDX ! Slope parameter of the distribution
66 !
67 !* 0.2 Declaration of local variables
68 !
69 REAL :: ZRTMIN, ZLBX, ZLBEX, ZLBDAX_MAX, ZCCX, ZCXX
70 !
71 !---------------------------------------------------------------------------------
72 !
73 !* 1. PRELIMINARIES
74 ! -------------
75 !
76 zrtmin = prtmin
77 !
78 IF (kid == 2) THEN
79  IF (hcloud == 'LIMA' .AND. kmoment == 2) THEN
80  zlbx = xlbc_l
81  zlbex = xlbexc_l
82 ! ELSE
83 ! print*, 'ERROR: the computation of lambda_c is not available if c is 1-moment species'
84  END IF
85 ELSE IF (kid == 3) THEN
86  IF (hcloud == 'LIMA') THEN
87  zlbx = xlbr_l
88  zlbex = xlbexr_l
89  IF (kmoment == 1) THEN
90  zccx = xccr_l
91  zcxx = xcxr_l
92  END IF
93  ELSE IF (hcloud(1:3) == 'ICE') THEN
94  zlbx = xlbr_i
95  zlbex = xlbexr_i
96  zccx = xccr_i
97  zcxx = xcxr_i
98  ELSE
99  print*, 'ERROR: something wrong with the computation of lambda_r'
100  END IF
101 ELSE IF (kid == 4) THEN
102  IF (hcloud == 'LIMA') THEN
103  zlbx = xlbi_l
104  zlbex = xlbexi_l
105  ELSE IF (hcloud(1:3) == 'ICE') THEN
106  zlbx = xlbi_i
107  zlbex = xlbexi_i
108  ELSE
109  print*, 'ERROR: something wrong with the computation of lambda_i'
110  END IF
111 ELSE IF (kid == 5) THEN
112  IF (hcloud == 'LIMA') THEN
113  zlbx = xlbs_l
114  zlbex = xlbexs_l
115  zlbdax_max = xlbdas_max_l
116  IF (kmoment == 1) THEN
117  zccx = xccs_l
118  zcxx = xcxs_l
119  END IF
120  ELSE IF (hcloud(1:3) == 'ICE') THEN
121  zlbx = xlbs_i
122  zlbex = xlbexs_i
123  zlbdax_max = xlbdas_max_i
124  zccx = xccs_i
125  zcxx = xcxs_i
126  ELSE
127  print*, 'ERROR: something wrong with the computation of lambda_s'
128  END IF
129 ELSE IF (kid == 6) THEN
130  IF (hcloud == 'LIMA') THEN ! .AND. KMOMENT == 1) THEN
131  zlbx = xlbg_l
132  zlbex = xlbexg_l
133  IF (kmoment == 1) THEN
134  zccx = xccg_l
135  zcxx = xcxg_l
136  END IF
137  ELSE IF (hcloud(1:3) == 'ICE') THEN
138  zlbx = xlbg_i
139  zlbex = xlbexg_i
140  zccx = xccg_i
141  zcxx = xcxg_i
142  ELSE
143  print*, 'ERROR: something wrong with the computation of lambda_g'
144  END IF
145 ELSE IF (kid == 7) THEN
146  IF (hcloud == 'LIMA') THEN ! .AND. KMOMENT == 1) THEN
147  zlbx = xlbh_l
148  zlbex = xlbexh_l
149  IF (kmoment == 1) THEN
150  zccx = xcch_l
151  zcxx = xcxh_l
152  END IF
153  ELSE IF (hcloud(1:3) == 'ICE') THEN
154  zlbx = xlbh_i
155  zlbex = xlbexh_i
156  zccx = xcch_i
157  zcxx = xcxh_i
158  ELSE
159  print*, 'ERROR: something wrong with the computation of lambda_h'
160  END IF
161 END IF
162 !
163 plbdx(:) = 0. !1.E10
164 !
165 !
166 !* 2. COMPUTE LBDA_x FOR 2-MOMENT SPECIES
167 ! -----------------------------------
168 !
169 IF (kmoment == 2) THEN
170  WHERE (prx(:) > zrtmin .AND. pcx(:) > 0.)
171  plbdx(:) = (zlbx * pcx(:) / prx(:))**zlbex
172  END WHERE
173  IF (kid == 5) plbdx(:) = min(zlbdax_max, plbdx(:))
174 !
175 !
176 !* 3. COMPUTE LBDA_x and N_x FOR 1-MOMENT SPECIES
177 ! -------------------------------------------
178 !
179 ELSE IF (kmoment == 1) THEN
180 !
181 !* 3.1 Special case of cloud droplets
182 !
183  IF (kid == 2) THEN
184 ! print*, 'computation of lambda_c in 1-moment configuration not treated'
185 !
186 !* 3.2 Special case of ice crystals
187 !
188  ELSE IF (kid == 4) THEN
189 ! formulation utilisee dans rain_ice_fast_ri
190  WHERE (prx(:) > zrtmin .AND. pcx(:) > 0.0)
191  plbdx(:) = zlbx * (prho(:) * prx(:) / pcx(:))**zlbex
192  ENDWHERE
193 !
194 !* 3.3 Special case of snow
195 !
196  ELSE IF (kid == 5) THEN
197 ! limitation of lbdas
198  WHERE (prx(:) > zrtmin)
199  plbdx(:) = min(200000., zlbx * (prho(:) * prx(:))**zlbex)
200  pcx(:) = zccx * plbdx(:)**zcxx / prho(:)
201  ENDWHERE
202 !
203 !* 3.4 Computation for all other hydrometeors
204 !
205  ELSE
206  WHERE (prx(:) > zrtmin)
207  plbdx(:) = zlbx * (prho(:) * prx(:))**zlbex
208  pcx(:) = zccx * plbdx(:)**zcxx / prho(:)
209  ENDWHERE
210  END IF
211 END IF
212 !
213 END SUBROUTINE compute_lambda
214 END MODULE mode_compute_lambda
