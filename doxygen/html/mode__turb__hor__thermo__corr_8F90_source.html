137!
138!* 0.2 declaration of local variables
139!
140REAL, DIMENSION(D%NIT,D%NJT,D%NKT) :: ZFLX,ZWORK,ZA ! work arrays
141!

143INTEGER :: JI,JJ,JK
144 ! Index values for the Beginning and End
145 ! mass points of the domain
146REAL, DIMENSION(D%NIT,D%NJT,1+JPVEXT:3+JPVEXT) :: ZCOEFF
147 ! coefficients for the uncentred gradient
148 ! computation near the ground
149REAL :: ZTIME1, ZTIME2
150TYPE(tfieldmetadata) :: TZFIELD
151!
152! ---------------------------------------------------------------------------
153!
154!* 1. PRELIMINARY COMPUTATIONS
155! ------------------------
156!
157ikb = 1+jpvext
158ike = SIZE(pthlm,3)-jpvext
159iit=d%NIT
160ijt=d%NJT
161ikt=d%NKT
162!
163!
164!
165! compute the coefficients for the uncentred gradient computation near the
166! ground
167!$acc kernels present_cr(ZCOEFF)
168zcoeff(:,:,ikb+2)= - pdzz(:,:,ikb+1) / &
169 ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+2) )
170zcoeff(:,:,ikb+1)= (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) / &
171 ( pdzz(:,:,ikb+1) * pdzz(:,:,ikb+2) )
172zcoeff(:,:,ikb)= - (pdzz(:,:,ikb+2)+2.*pdzz(:,:,ikb+1)) / &
173 ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+1) )
174!$acc end kernels
175!
176!
177!* 8. TURBULENT CORRELATIONS : <THl THl>, <THl Rnp>, <Rnp Rnp>, Sigma_s
178! -----------------------------------------------------------------
179!
180!
181!
182IF ( ( krrl > 0 .AND. nebn%LSUBG_COND) .OR. ( turbn%LTURB_FLX .AND. tpfile%LOPENED ) &
183 .OR. ( tles%LLES_CALL ) ) THEN
184!
185!* 8.1 <THl THl>
186!
187 ! Computes the horizontal variance <THl THl>
188 IF (.NOT. o2d) THEN
189 zflx(:,:,:) = turbn%XCTV * plm(:,:,:) * pleps(:,:,:) * &
190 ( gx_m_m(oflat,pthlm,pdxx,pdzz,pdzx)**2 + gy_m_m(oflat,pthlm,pdyy,pdzz,pdzy)**2 )
191 ELSE
192 zflx(:,:,:) = turbn%XCTV * plm(:,:,:) * pleps(:,:,:) * &
193 gx_m_m(oflat,pthlm,pdxx,pdzz,pdzx)**2
194 END IF
195!
196! Compute the flux at the first inner U-point with an uncentred vertical
197! gradient
198!
199 zflx(:,:,ikb) = turbn%XCTV * plm(:,:,ikb) &
200 * pleps(:,:,ikb) * ( &
201 ( mxf(dxm(pthlm(:,:,ikb)) * pinv_pdxx(:,:,ikb)) &
202 - ( zcoeff(:,:,ikb+2)*pthlm(:,:,ikb+2) &
203 +zcoeff(:,:,ikb+1)*pthlm(:,:,ikb+1) &
204 +zcoeff(:,:,ikb)*pthlm(:,:,ikb) &
205 ) * 0.5 * ( pdzx(:,:,ikb+1)+pdzx(:,:,ikb) ) &
206 / mxf(pdxx(:,:,ikb)) &
207 ) ** 2 + &
208 ( myf(dym(pthlm(:,:,ikb)) * pinv_pdyy(:,:,ikb)) &
209 - ( zcoeff(:,:,ikb+2)*pthlm(:,:,ikb+2) &
210 +zcoeff(:,:,ikb+1)*pthlm(:,:,ikb+1) &
211 +zcoeff(:,:,ikb)*pthlm(:,:,ikb) &
212 ) * 0.5 * ( pdzy(:,:,ikb+1)+pdzy(:,:,ikb) ) &
213 / myf(pdyy(:,:,ikb)) &
214 ) ** 2 )
215 !
216!$acc kernels present_cr(ZFLX)
217 !$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
218 zflx(:,:,ikb-1) = zflx(:,:,ikb)
219 !$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
220 !
221 IF ( krrl > 0 ) THEN
222!$mnh_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
223 zwork(:,:,:) = zflx(:,:,:) * patheta(:,:,:) * patheta(:,:,:)
224!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
225 END IF
226!$acc end kernels
227 !
228 ! stores <THl THl>
229 IF ( turbn%LTURB_FLX .AND. tpfile%LOPENED ) THEN
230 tzfield = tfieldmetadata( &
231 cmnhname = 'THL_HVAR', &
232 cstdname = '', &
233 clongname = 'THL_HVAR', &
234 cunits = 'K2', &
235 cdir = 'XY', &
236 ccomment = 'X_Y_Z_THL_HVAR', &
237 ngrid = 1, &
238 ntype = typereal, &
239 ndims = 3, &
240 ltimedep = .true. )
241!$acc update self(ZFLX)
242 CALL io_field_write(tpfile,tzfield,zflx)
243 END IF
244!
245! Storage in the LES configuration (addition to TURB_VER computation)
246!
247 IF (tles%LLES_CALL) THEN
248 CALL second_mnh(ztime1)
249 CALL les_mean_subgrid( zflx, tles%X_LES_SUBGRID_Thl2, .true. )
250 CALL les_mean_subgrid( mzf(pwm)*zflx, tles%X_LES_RES_W_SBG_Thl2, .true. )
251 CALL les_mean_subgrid( -2.*xctd*sqrt(ptkem)*zflx/pleps ,tles%X_LES_SUBGRID_DISS_Thl2, .true. )
252 CALL etheta(d,cst,krr,krri,pthlm,prm,plocpexnm,patheta,psrcm,oocean,ocompute_src,za)
253 CALL les_mean_subgrid( za*zflx, tles%X_LES_SUBGRID_ThlThv, .true. )
254 CALL les_mean_subgrid( -cst%XG/pthvref/3.*za*zflx, tles%X_LES_SUBGRID_ThlPz, .true. )
255 CALL second_mnh(ztime2)
256 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
257 END IF
258!
259 IF ( krr /= 0 ) THEN
260!
261!* 8.3 <THl Rnp>
262!
263 ! Computes the horizontal correlation <THl Rnp>
264 IF (.NOT. o2d) THEN
265 zflx(:,:,:)= &
266 plm(:,:,:) * pleps(:,:,:) * &
267 (gx_m_m(oflat,pthlm,pdxx,pdzz,pdzx) * gx_m_m(oflat,prm(:,:,:,1),pdxx,pdzz,pdzx) &
268 + gy_m_m(oflat,pthlm,pdyy,pdzz,pdzy) * gy_m_m(oflat,prm(:,:,:,1),pdyy,pdzz,pdzy) &
269 ) * (turbn%XCHT1+turbn%XCHT2)
270 ELSE
271 zflx(:,:,:)= &
272 plm(:,:,:) * pleps(:,:,:) * &
273 (gx_m_m(oflat,pthlm,pdxx,pdzz,pdzx) * gx_m_m(oflat,prm(:,:,:,1),pdxx,pdzz,pdzx) &
274 ) * (turbn%XCHT1+turbn%XCHT2)
275
276 END IF
277!
278! Compute the flux at the first inner U-point with an uncentred vertical
279! gradient
280 zflx(:,:,ikb) = (turbn%XCHT1+turbn%XCHT2) * plm(:,:,ikb) &
281 * pleps(:,:,ikb) * ( &
282 ( mxf(dxm(pthlm(:,:,ikb)) * pinv_pdxx(:,:,ikb)) &
283 - ( zcoeff(:,:,ikb+2)*pthlm(:,:,ikb+2) &
284 +zcoeff(:,:,ikb+1)*pthlm(:,:,ikb+1) &
285 +zcoeff(:,:,ikb)*pthlm(:,:,ikb) &
286 ) * 0.5 * ( pdzx(:,:,ikb+1)+pdzx(:,:,ikb) ) &
287 / mxf(pdxx(:,:,ikb)) &
288 ) * &
289 ( mxf(dxm(prm(:,:,ikb,1)) * pinv_pdxx(:,:,ikb)) &
290 - ( zcoeff(:,:,ikb+2)*prm(:,:,ikb+2,1) &
291 +zcoeff(:,:,ikb+1)*prm(:,:,ikb+1,1) &
292 +zcoeff(:,:,ikb)*prm(:,:,ikb,1) &
293 ) * 0.5 * ( pdzx(:,:,ikb+1)+pdzx(:,:,ikb) ) &
294 / mxf(pdxx(:,:,ikb)) &
295 ) + &
296 ( myf(dym(pthlm(:,:,ikb)) * pinv_pdyy(:,:,ikb)) &
297 - ( zcoeff(:,:,ikb+2)*pthlm(:,:,ikb+2) &
298 +zcoeff(:,:,ikb+1)*pthlm(:,:,ikb+1) &
299 +zcoeff(:,:,ikb)*pthlm(:,:,ikb) &
300 ) * 0.5 * ( pdzy(:,:,ikb+1)+pdzy(:,:,ikb) ) &
301 / myf(pdyy(:,:,ikb)) &
302 ) * &
303 ( myf(dym(prm(:,:,ikb,1)) * pinv_pdyy(:,:,ikb)) &
304 - ( zcoeff(:,:,ikb+2)*prm(:,:,ikb+2,1) &
305 +zcoeff(:,:,ikb+1)*prm(:,:,ikb+1,1) &
306 +zcoeff(:,:,ikb)*prm(:,:,ikb,1) &
307 ) * 0.5 * ( pdzy(:,:,ikb+1)+pdzy(:,:,ikb) ) &
308 / myf(pdyy(:,:,ikb)) &
309 ) )
310 !
311!$acc kernels present_cr(ZFLX)
312 !$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
313 zflx(:,:,ikb-1) = zflx(:,:,ikb)
314 !$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
315 !
316 IF ( krrl > 0 ) THEN
317!$mnh_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
318 zwork(:,:,:) = zwork(:,:,:) + &
319 2. * patheta(:,:,:) * pamoist(:,:,:) * zflx(:,:,:)
320!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
321 END IF
322!$acc end kernels
323 ! stores <THl Rnp>
324 IF ( turbn%LTURB_FLX .AND. tpfile%LOPENED ) THEN
325 tzfield = tfieldmetadata( &
326 cmnhname = 'THLR_HCOR', &
327 cstdname = '', &
328 clongname = 'THLR_HCOR', &
329 cunits = 'K kg kg-1', &
330 cdir = 'XY', &
331 ccomment = 'X_Y_Z_THLR_HCOR', &
332 ngrid = 1, &
333 ntype = typereal, &
334 ndims = 3, &
335 ltimedep = .true. )
336 CALL io_field_write(tpfile,tzfield,zflx)
337 END IF
338!
339! Storage in the LES configuration (addition to TURB_VER computation)
340!
341 IF (tles%LLES_CALL) THEN
342 CALL second_mnh(ztime1)
343 CALL les_mean_subgrid( zflx, tles%X_LES_SUBGRID_ThlRt, .true. )
344 CALL les_mean_subgrid( mzf(pwm)*zflx, tles%X_LES_RES_W_SBG_ThlRt, .true. )
345 CALL les_mean_subgrid( -xctd*sqrt(ptkem)*zflx/pleps ,tles%X_LES_SUBGRID_DISS_ThlRt, .true. )
346 CALL les_mean_subgrid( za*zflx, tles%X_LES_SUBGRID_RtThv, .true. )
347 CALL les_mean_subgrid( -cst%XG/pthvref/3.*za*zflx, tles%X_LES_SUBGRID_RtPz,.true.)
348 CALL emoist(d,cst,krr,krri,pthlm,prm,plocpexnm,pamoist,psrcm,oocean,za)
349 CALL les_mean_subgrid( za*zflx, tles%X_LES_SUBGRID_ThlThv, .true. )
350 CALL les_mean_subgrid( -cst%XG/pthvref/3.*za*zflx, tles%X_LES_SUBGRID_ThlPz,.true.)
351 CALL second_mnh(ztime2)
352 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
353 END IF
354!
355!* 8.4 <Rnp Rnp>
356!
357 ! Computes the horizontal variance <Rnp Rnp>
358 IF (.NOT. o2d) THEN
359 zflx(:,:,:) = turbn%XCHV * plm(:,:,:) * pleps(:,:,:) * &
360 ( gx_m_m(oflat,prm(:,:,:,1),pdxx,pdzz,pdzx)**2 + &
361 gy_m_m(oflat,prm(:,:,:,1),pdyy,pdzz,pdzy)**2 )
362 ELSE
363 zflx(:,:,:) = turbn%XCHV * plm(:,:,:) * pleps(:,:,:) * &
364 ( gx_m_m(oflat,prm(:,:,:,1),pdxx,pdzz,pdzx)**2 )
365 END IF
366!
367! Compute the flux at the first inner U-point with an uncentred vertical
368! gradient
369 zflx(:,:,ikb) = turbn%XCHV * plm(:,:,ikb) &
370 * pleps(:,:,ikb) * ( &
371 ( mxf(dxm(prm(:,:,ikb,1)) * pinv_pdxx(:,:,ikb)) &
372 - ( zcoeff(:,:,ikb+2)*prm(:,:,ikb+2,1) &
373 +zcoeff(:,:,ikb+1)*prm(:,:,ikb+1,1) &
374 +zcoeff(:,:,ikb)*prm(:,:,ikb,1) &
375 ) * 0.5 * ( pdzx(:,:,ikb+1)+pdzx(:,:,ikb) ) &
376 / mxf(pdxx(:,:,ikb:)) &
377 ) ** 2 + &
378 ( myf(dym(prm(:,:,ikb,1)) * pinv_pdyy(:,:,ikb)) &
379 - ( zcoeff(:,:,ikb+2)*prm(:,:,ikb+2,1) &
380 +zcoeff(:,:,ikb+1)*prm(:,:,ikb+1,1) &
381 +zcoeff(:,:,ikb)*prm(:,:,ikb,1) &
382 ) * 0.5 * ( pdzy(:,:,ikb+1)+pdzy(:,:,ikb) ) &
383 / myf(pdyy(:,:,ikb)) &
384 ) ** 2 )
385!
386!$acc kernels present_cr(ZFLX)
387 !$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
388 zflx(:,:,ikb-1) = zflx(:,:,ikb)
389 !$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
390 !
391 IF ( krrl > 0 ) THEN
392 !$mnh_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
393 zwork(:,:,:) = zwork(:,:,:)+ pamoist(:,:,:) * pamoist(:,:,:) * zflx(:,:,:)
394 !$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
395 END IF
396!$acc end kernels
397 ! stores <Rnp Rnp>
398 IF ( turbn%LTURB_FLX .AND. tpfile%LOPENED ) THEN
399 tzfield = tfieldmetadata( &
400 cmnhname = 'R_HVAR', &
401 cstdname = '', &
402 clongname = 'R_HVAR', &
403 cunits = 'kg2 kg-2', &
404 cdir = 'XY', &
405 ccomment = 'X_Y_Z_R_HVAR', &
406 ngrid = 1, &
407 ntype = typereal, &
408 ndims = 3, &
409 ltimedep = .true. )
410!$acc update self(ZFLX)
411 CALL io_field_write(tpfile,tzfield,zflx)
412 END IF
413 !
414 ! Storage in the LES configuration (addition to TURB_VER computation)
415 !
416 IF (tles%LLES_CALL) THEN
417 CALL second_mnh(ztime1)
418 CALL les_mean_subgrid( zflx, tles%X_LES_SUBGRID_Rt2, .true. )
419 CALL les_mean_subgrid( mzf(pwm)*zflx, tles%X_LES_RES_W_SBG_Rt2, .true. )
420 CALL les_mean_subgrid( za*zflx, tles%X_LES_SUBGRID_RtThv, .true. )
421 CALL les_mean_subgrid( -cst%XG/pthvref/3.*za*zflx, tles%X_LES_SUBGRID_RtPz,.true.)
422 CALL les_mean_subgrid( -2.*xctd*sqrt(ptkem)*zflx/pleps, tles%X_LES_SUBGRID_DISS_Rt2, .true. )
423 CALL second_mnh(ztime2)
424 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
425 END IF
426 !
427 END IF
428!
429! 8.5 Complete the Sigma_s computation:
430!
431 IF ( krrl > 0 ) THEN
432 !
433 !$acc kernels present_cr(PSIGS)
434 psigs(:,:,:)=psigs(:,:,:)*psigs(:,:,:) + zwork(:,:,:)
435 ! Extrapolate PSIGS at the ground and at the top
436 psigs(:,:,ikb-1) = psigs(:,:,ikb)
437 psigs(:,:,ike+1) = psigs(:,:,ike)
438 psigs(:,:,:) = sqrt(max( psigs(:,:,:),1.e-12) )
439 !$acc end kernels
440 END IF
441!
442END IF
443!
444!
445!
446END SUBROUTINE turb_hor_thermo_corr
447END MODULE mode_turb_hor_thermo_corr
