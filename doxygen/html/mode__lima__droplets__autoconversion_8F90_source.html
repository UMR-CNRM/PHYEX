14! ##########################################################################
15!
16!! PURPOSE
17!! -------
18!! Compute the autoconversion of cloud droplets
19!!
20!!

35!* 0. DECLARATIONS
36! ------------
37!
38USE modd_param_lima, ONLY : xrtmin, xctmin, lkhko, lkesslerac, nmom_c
39USE modd_param_lima_warm, ONLY : xlautr, xauto1, xlautr_threshold, &
40 xitautr, xauto2, xitautr_threshold, &
41 xaccr4, xaccr5, xaccr3, xaccr1, xac, xr0
42USE modd_cst, ONLY : xpi, xrholw
43!
44IMPLICIT NONE
45!
46!* 0.1 Declarations of dummy arguments :
47!
48LOGICAL, DIMENSION(:),INTENT(IN) :: LDCOMPUTE
49!
50REAL, DIMENSION(:), INTENT(IN) :: PRHODREF ! Reference Exner function
51!
52REAL, DIMENSION(:), INTENT(IN) :: PRCT ! Cloud water m.r. at t
53REAL, DIMENSION(:), INTENT(IN) :: PCCT ! Cloud water conc. at t
54REAL, DIMENSION(:), INTENT(IN) :: PLBDC !
55REAL, DIMENSION(:), INTENT(IN) :: PLBDR !
56!
57REAL, DIMENSION(:), INTENT(OUT) :: P_RC_AUTO
58REAL, DIMENSION(:), INTENT(OUT) :: P_CC_AUTO
59REAL, DIMENSION(:), INTENT(OUT) :: P_CR_AUTO
60!
61!* 0.2 Declarations of local variables :
62!
63REAL, DIMENSION(SIZE(PRCT)) :: ZW1, ZW2, ZW3 ! work arrays
64!
65!-------------------------------------------------------------------------------
66!
67p_rc_auto(:) = 0.0
68p_cc_auto(:) = 0.0
69p_cr_auto(:) = 0.0
70!
71zw3(:) = 0.0
72zw2(:) = 0.0
73zw1(:) = 0.0
74!
75IF (nmom_c.EQ.1 .AND. lkesslerac) THEN
76 p_rc_auto(:) = - 1.e-3 * max( prct(:) - 0.5e-3 / prhodref(:), 0. )
77ELSE IF (lkhko) THEN
78!
79! 1. Autoconversion of cloud droplets
80! -----------------------------------
81!
82 WHERE ( prct(:)>xrtmin(2) .AND. pcct(:)>xctmin(2) .AND. ldcompute(:) )
83!
84 zw1(:)= 1350.0 * prct(:)**(2.47) * (pcct(:)* prhodref(:)/1.0e6)**(-1.79) ! ZCCT in cm-3
85!
86 p_rc_auto(:) = - zw1(:)
87!
88 zw2(:) = zw1(:) * 3./(4.*xpi*xrholw*(xr0)**(3.))
89 p_cr_auto(:) = zw2(:)
90!
91 zw3(:) = - zw1(:) * pcct(:) / prct(:)
92 p_cc_auto(:) = zw3(:)
93!
94 END WHERE
95!
96ELSE
97!
98! 2. Autoconversion of cloud droplets (Berry-Reinhardt parameterization)
99! ----------------------------------------------------------------------
100!
101 WHERE( prct(:)>xrtmin(2) .AND. pcct(:)>xctmin(2) .AND. plbdc(:)>0. .AND. ldcompute(:) )
102 zw2(:) = max( 0.0, &
103 xlautr*prhodref(:)*prct(:)*(xauto1/min(plbdc(:),1.e9)**4-xlautr_threshold) ) ! L
104!
105 zw3(:) = max( 0.0, &
106 xitautr*zw2(:)*prct(:)*(xauto2/plbdc(:)-xitautr_threshold) ) ! L/tau
107!
108 p_rc_auto(:) = - zw3(:)
109!
110 zw1(:) = min( min( 1.2e4, &
111 (xaccr4/plbdc(:)-xaccr5)/xaccr3 ), &
112 plbdr(:)/xaccr1 ) ! D**-1 threshold diameter for
113 ! switching the autoconversion regimes
114 ! min (80 microns, D_h, D_r)
115 zw3(:) = zw3(:) * max( 0.0,zw1(:) )**3 / xac
116!
117 p_cc_auto(:) = -zw3(:)
118 p_cr_auto(:) = zw3(:)
119!
120 END WHERE
121!
122END IF
123!
124!-------------------------------------------------------------------------------
125!
126END SUBROUTINE lima_droplets_autoconversion
127END MODULE mode_lima_droplets_autoconversion
