14! ##########################################################################
15!
16!! PURPOSE
17!! -------
18!! Compute the autoconversion of cloud droplets
19!!
20!!
21!! AUTHOR
22!! ------
23!! J.-M. Cohard * Laboratoire d'Aerologie*
24!! J.-P. Pinty * Laboratoire d'Aerologie*
25!! S. Berthet * Laboratoire d'Aerologie*
26!! B. Vié * Laboratoire d'Aerologie*
27!!
28!! MODIFICATIONS
29!! -------------
30!! Original 15/03/2018
31!! B. Vie 02/03/2020 : missing CC process
32! Delbeke/Vie 03/2022 : KHKO option
33!-------------------------------------------------------------------------------
34!
35!* 0. DECLARATIONS
36! ------------
37!
38USE modd_param_lima_warm, ONLY:param_lima_warm_t
39USE modd_param_lima, ONLY:param_lima_t
40USE modd_cst, ONLY:cst_t
41USE yomhook, ONLY:lhook, dr_hook, jphook
42!
43IMPLICIT NONE
44!
45!* 0.1 Declarations of dummy arguments :
46!
47TYPE(param_lima_warm_t),INTENT(IN)::LIMAW
48TYPE(param_lima_t),INTENT(IN)::LIMAP
49TYPE(cst_t),INTENT(IN)::CST
50INTEGER, INTENT(IN) :: KSIZE
51LOGICAL, DIMENSION(KSIZE),INTENT(IN) :: ODCOMPUTE
52!
53REAL, DIMENSION(KSIZE), INTENT(IN) :: PRHODREF ! Reference Exner function
54!
55REAL, DIMENSION(KSIZE), INTENT(IN) :: PRCT ! Cloud water m.r. at t
56REAL, DIMENSION(KSIZE), INTENT(IN) :: PCCT ! Cloud water conc. at t
57REAL, DIMENSION(KSIZE), INTENT(IN) :: PLBDC !
58REAL, DIMENSION(KSIZE), INTENT(IN) :: PLBDR !
59!
60REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_RC_AUTO
61REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_CC_AUTO
62REAL, DIMENSION(KSIZE), INTENT(OUT) :: P_CR_AUTO
63!
64!* 0.2 Declarations of local variables :
65!
66REAL(KIND=jphook) :: zhook_handle
67REAL, DIMENSION(SIZE(PRCT)) :: ZW1, ZW2, ZW3 ! work arrays
68!
69!-------------------------------------------------------------------------------
70!
71IF (lhook) CALL dr_hook('LIMA_DROPLETS_AUTOCONVERSION', 0, zhook_handle)
72p_rc_auto(:) = 0.0
73p_cc_auto(:) = 0.0
74p_cr_auto(:) = 0.0
75!
76zw3(:) = 0.0
77zw2(:) = 0.0
78zw1(:) = 0.0
79!
80IF (limap%NMOM_C.EQ.1 .AND. limap%LKESSLERAC) THEN
81 p_rc_auto(:) = - 1.e-3 * max( prct(:) - 0.5e-3 / prhodref(:), 0. )
82ELSE IF (limap%LKHKO) THEN
83!
84! 1. Autoconversion of cloud droplets
85! -----------------------------------
86!
87 WHERE ( prct(:)>limap%XRTMIN(2) .AND. pcct(:)>limap%XCTMIN(2) .AND. odcompute(:) )
88!
89 zw1(:)= 1350.0 * prct(:)**(2.47) * (pcct(:)* prhodref(:)/1.0e6)**(-1.79) ! ZCCT in cm-3
90!
91 p_rc_auto(:) = - zw1(:)
92!
93 zw2(:) = zw1(:) * 3./(4.*cst%XPI*cst%XRHOLW*(limaw%XR0)**(3.))
94 p_cr_auto(:) = zw2(:)
95!
96 zw3(:) = - zw1(:) * pcct(:) / prct(:)
97 p_cc_auto(:) = zw3(:)
98!
99 END WHERE
100!
101ELSE
102!
103! 2. Autoconversion of cloud droplets (Berry-Reinhardt parameterization)
104! ----------------------------------------------------------------------
105!
106 WHERE( prct(:)>limap%XRTMIN(2) .AND. pcct(:)>limap%XCTMIN(2) .AND. plbdc(:)>0. .AND. odcompute(:) )
107 zw2(:) = max( 0.0, &
108 limaw%XLAUTR*prhodref(:)*prct(:)*(limaw%XAUTO1/min(plbdc(:),1.e9)**4-limaw%XLAUTR_THRESHOLD) ) ! L
109!
110 zw3(:) = max( 0.0, &
111 limaw%XITAUTR*zw2(:)*prct(:)*(limaw%XAUTO2/plbdc(:)-limaw%XITAUTR_THRESHOLD) ) ! L/tau
112!
113 p_rc_auto(:) = - zw3(:)
114!
115 zw1(:) = min( min( 1.2e4, &
116 (limaw%XACCR4/plbdc(:)-limaw%XACCR5)/limaw%XACCR3 ), &
117 plbdr(:)/limaw%XACCR1 ) ! D**-1 threshold diameter for
118 ! switching the autoconversion regimes
119 ! min (80 microns, D_h, D_r)
120 zw3(:) = zw3(:) * max( 0.0,zw1(:) )**3 / limaw%XAC
121!
122 p_cc_auto(:) = -zw3(:)
123 p_cr_auto(:) = zw3(:)
124!
125 END WHERE
126!
127END IF
128!
129!-------------------------------------------------------------------------------
130!
131IF (lhook) CALL dr_hook('LIMA_DROPLETS_AUTOCONVERSION', 1, zhook_handle)
132END SUBROUTINE lima_droplets_autoconversion
133END MODULE mode_lima_droplets_autoconversion
