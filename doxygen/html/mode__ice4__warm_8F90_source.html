9SUBROUTINE ice4_warm(CST, ICEP, ICED, KPROMA, KSIZE, LDSOFT, LDCOMPUTE, HSUBG_RC_RR_ACCR, HSUBG_RR_EVAP, &
10 &PRHODREF, PLVFACT, PT, PPRES, PTHT, &
11 &PLBDAR, PLBDAR_RF, PKA, PDV, PCJ, &
12 &PHLC_LCF, PHLC_HCF, PHLC_LRC, PHLC_HRC, &
13 &PCF, PRF, &
14 &PRVT, PRCT, PRRT, &
15 &PRCAUTR, PRCACCR, PRREVAV)
16!!
17!!** PURPOSE
18!! -------
19!! Computes the warm process
20!!
21!! AUTHOR
22!! ------
23!! S. Riette from the plitting of rain_ice source code (nov. 2014)
24!!
25!! MODIFICATIONS
26!! -------------
27!!
28!! R. El Khatib 24-Aug-2021 Optimizations
29!
30!
31!* 0. DECLARATIONS
32! ------------
33!
34USE modd_cst, ONLY: cst_t
35USE modd_rain_ice_descr_n, ONLY: rain_ice_descr_t
36USE modd_rain_ice_param_n, ONLY: rain_ice_param_t
37!
38USE mode_msg
39USE yomhook , ONLY : lhook, dr_hook, jphook
40!
41IMPLICIT NONE
42!
43!* 0.1 Declarations of dummy arguments :
44!
45TYPE(cst_t), INTENT(IN) :: CST
46TYPE(rain_ice_param_t), INTENT(IN) :: ICEP
47TYPE(rain_ice_descr_t), INTENT(IN) :: ICED
48INTEGER, INTENT(IN) :: KPROMA, KSIZE
49LOGICAL, INTENT(IN) :: LDSOFT
50LOGICAL, DIMENSION(KPROMA), INTENT(IN) :: LDCOMPUTE
51CHARACTER(LEN=80), INTENT(IN) :: HSUBG_RC_RR_ACCR ! subgrid rc-rr accretion
52CHARACTER(LEN=80), INTENT(IN) :: HSUBG_RR_EVAP ! subgrid rr evaporation
53REAL, DIMENSION(KPROMA), INTENT(IN) :: PRHODREF ! Reference density
54REAL, DIMENSION(KPROMA), INTENT(IN) :: PLVFACT
55REAL, DIMENSION(KPROMA), INTENT(IN) :: PT ! Temperature
56REAL, DIMENSION(KPROMA), INTENT(IN) :: PPRES ! absolute pressure at t
57REAL, DIMENSION(KPROMA), INTENT(IN) :: PTHT ! Theta at time t
58REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAR ! Slope parameter of the raindrop distribution
59REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAR_RF!like PLBDAR but for the Rain Fraction part
60REAL, DIMENSION(KPROMA), INTENT(IN) :: PKA ! Thermal conductivity of the air
61REAL, DIMENSION(KPROMA), INTENT(IN) :: PDV ! Diffusivity of water vapor in the air
62REAL, DIMENSION(KPROMA), INTENT(IN) :: PCJ ! Function to compute the ventilation coefficient
63REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLC_HCF ! HLCLOUDS : fraction of High Cloud Fraction in grid
64REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLC_LCF ! HLCLOUDS : fraction of Low Cloud Fraction in grid
65REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLC_HRC ! HLCLOUDS : LWC that is High LWC in grid
66REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLC_LRC ! HLCLOUDS : LWC that is Low LWC in grid
67REAL, DIMENSION(KPROMA), INTENT(IN) :: PCF ! Cloud fraction
68REAL, DIMENSION(KPROMA), INTENT(IN) :: PRF ! Rain fraction
69REAL, DIMENSION(KPROMA), INTENT(IN) :: PRVT ! Water vapor m.r. at t
70REAL, DIMENSION(KPROMA), INTENT(IN) :: PRCT ! Cloud water m.r. at t
71REAL, DIMENSION(KPROMA), INTENT(IN) :: PRRT ! Rain water m.r. at t
72REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRCAUTR ! Autoconversion of r_c for r_r production
73REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRCACCR ! Accretion of r_c for r_r production
74REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRREVAV ! Evaporation of r_r
75!
76!* 0.2 declaration of local variables
77!
78REAL :: ZZW2, ZZW3, ZZW4
79REAL, DIMENSION(KPROMA) :: ZUSW ! Undersaturation over water
80REAL, DIMENSION(KPROMA) :: ZTHLT ! Liquid potential temperature
81REAL(KIND=jphook) :: zhook_handle
82INTEGER :: JL
83LOGICAL :: LMASK, LMASK1, LMASK2
84!-------------------------------------------------------------------------------
85!
86IF (lhook) CALL dr_hook('ICE4_WARM', 0, zhook_handle)
87!
88!
89!-------------------------------------------------------------------------------
90!
91!* 4.2 compute the autoconversion of r_c for r_r production: RCAUTR
92!
93!$acc kernels
94!$acc loop independent
95DO jl=1, ksize
96 IF(phlc_hrc(jl)>iced%XRTMIN(2) .AND. phlc_hcf(jl)>0. .AND. ldcompute(jl)) THEN
97 IF(.NOT. ldsoft) THEN
98 !HCF*autoconv(HRC/HCF) with simplification
99 prcautr(jl) = icep%XTIMAUTC*max(phlc_hrc(jl) - phlc_hcf(jl)*icep%XCRIAUTC/prhodref(jl), 0.0)
100 ENDIF
101 ELSE
102 prcautr(jl) = 0.
103 ENDIF
104ENDDO
105!$acc end kernels
106!
107!
108!* 4.3 compute the accretion of r_c for r_r production: RCACCR
109!
110IF (hsubg_rc_rr_accr=='NONE') THEN
111 !CLoud water and rain are diluted over the grid box
112!$acc kernels
113!$acc loop independent
114 DO jl=1, ksize
115 IF(prct(jl)>iced%XRTMIN(2) .AND. prrt(jl)>iced%XRTMIN(3) .AND. ldcompute(jl)) THEN
116 IF(.NOT. ldsoft) THEN
117 prcaccr(jl) = icep%XFCACCR * prct(jl) &
118 & * plbdar(jl)**icep%XEXCACCR &
119 & * prhodref(jl)**(-iced%XCEXVT)
120 ENDIF
121 ELSE
122 prcaccr(jl) = 0.
123 ENDIF
124 ENDDO
125!$acc end kernels
126ELSEIF (hsubg_rc_rr_accr=='PRFR') THEN
127 !Cloud water is concentrated over its fraction with possibly to parts with high and low content as set for autoconversion
128 !Rain is concentrated over its fraction
129 !Rain in high content area fraction: PHLC_HCF
130 !Rain in low content area fraction:
131 ! if PRF<PCF (rain is entirely falling in cloud): PRF-PHLC_HCF
132 ! if PRF>PCF (rain is falling in cloud and in clear sky): PCF-PHLC_HCF
133 ! => min(PCF, PRF)-PHLC_HCF
134!$acc kernels
135!$acc loop independent
136 DO jl=1, ksize
137 lmask = prct(jl)>iced%XRTMIN(2) .AND. prrt(jl)>iced%XRTMIN(3) .AND. ldcompute(jl)
138 lmask1 = lmask .AND. phlc_hrc(jl)>iced%XRTMIN(2) .AND. phlc_hcf(jl)>0.
139 lmask2 = lmask .AND. phlc_lrc(jl)>iced%XRTMIN(2) .AND. phlc_lcf(jl)>1.e-20
140 IF(lmask1 .OR. lmask2) THEN
141 IF(.NOT. ldsoft) THEN
142 IF(lmask1) THEN
143 !Accretion due to rain falling in high cloud content
144 !HCF*accretion(HRC/HCF) with simplification
145 prcaccr(jl) = icep%XFCACCR * phlc_hrc(jl) &
146 &*plbdar_rf(jl)**icep%XEXCACCR &
147 &*prhodref(jl)**(-iced%XCEXVT)
148 ELSE
149 prcaccr(jl)=0.
150 ENDIF
151 IF(lmask2) THEN
152 !We add acrretion due to rain falling in low cloud content
153 prcaccr(jl) = prcaccr(jl) + icep%XFCACCR * ( phlc_lrc(jl)/phlc_lcf(jl) ) &
154 &*plbdar_rf(jl)**icep%XEXCACCR &
155 &*prhodref(jl)**(-iced%XCEXVT) &
156 &*(min(pcf(jl), prf(jl))-phlc_hcf(jl))
157 ENDIF
158 ENDIF
159 ELSE
160 prcaccr(jl)=0.
161 ENDIF
162 ENDDO
163!$acc end kernels
164ELSE
165 CALL print_msg(nverb_fatal,'GEN','ICE4_WARM','wrong HSUBG_RC_RR_ACCR case')
166ENDIF
167!
168!* 4.4 compute the evaporation of r_r: RREVAV
169!
170IF (hsubg_rr_evap=='NONE') THEN
171!$acc kernels
172!$acc loop independent
173 DO jl=1, ksize
174 IF(prrt(jl)>iced%XRTMIN(3) .AND. prct(jl)<=iced%XRTMIN(2) .AND. ldcompute(jl)) THEN
175 IF(.NOT. ldsoft) THEN
176 prrevav(jl) = exp(cst%XALPW - cst%XBETAW/pt(jl) - cst%XGAMW*log(pt(jl))) ! es_w
177 zusw(jl) = 1. - prvt(jl)*(ppres(jl)-prrevav(jl)) / (cst%XEPSILO * prrevav(jl)) ! Undersaturation over water
178 prrevav(jl) = (cst%XLVTT+(cst%XCPV-cst%XCL)*(pt(jl)-cst%XTT) )**2 / (pka(jl)*cst%XRV*pt(jl)**2) &
179 &+(cst%XRV*pt(jl)) / (pdv(jl)*prrevav(jl))
180 prrevav(jl) = (max(0.,zusw(jl) )/(prhodref(jl)*prrevav(jl)) ) * &
181 & (icep%X0EVAR*plbdar(jl)**icep%XEX0EVAR+icep%X1EVAR*pcj(jl)*plbdar(jl)**icep%XEX1EVAR)
182 ENDIF
183 ELSE
184 prrevav(jl)=0.
185 ENDIF
186 ENDDO
187!$acc end kernels
188
189ELSEIF (hsubg_rr_evap=='CLFR' .OR. hsubg_rr_evap=='PRFR') THEN
190 !ATTENTION
191 !Il faudrait recalculer les variables PKA, PDV, PCJ en tenant compte de la température T^u
192 !Ces variables devraient être sorties de rain_ice_slow et on mettrait le calcul de T^u, T^s
193 !et plusieurs versions (comme actuellement, en ciel clair, en ciel nuageux) de PKA, PDV, PCJ dans rain_ice
194 !On utiliserait la bonne version suivant l'option NONE, CLFR... dans l'évaporation et ailleurs
195
196!$acc kernels
197!$acc loop independent
198 DO jl=1, ksize
199 !Evaporation in clear sky part
200 !With CLFR, rain is diluted over the grid box
201 !With PRFR, rain is concentrated in its fraction
202 !Use temperature and humidity in clear sky part like Bechtold et al. (1993)
203 IF (hsubg_rr_evap=='CLFR') THEN
204 zzw4=1. !Precipitation fraction
205 zzw3=plbdar(jl)
206 ELSE
207 zzw4=prf(jl) !Precipitation fraction
208 zzw3=plbdar_rf(jl)
209 ENDIF
210
211 IF(prrt(jl)>iced%XRTMIN(3) .AND. zzw4>pcf(jl) .AND. ldcompute(jl)) THEN
212 IF(.NOT. ldsoft) THEN
213 ! outside the cloud (environment) the use of T^u (unsaturated) instead of T
214 ! Bechtold et al. 1993
215 !
216 ! T_l
217 zthlt(jl) = ptht(jl) - cst%XLVTT*ptht(jl)/cst%XCPD/pt(jl)*prct(jl)
218 !
219 ! T^u = T_l = theta_l * (T/theta)
220 zzw2 = zthlt(jl) * pt(jl) / ptht(jl)
221 !
222 ! es_w with new T^u
223 prrevav(jl) = exp(cst%XALPW - cst%XBETAW/zzw2 - cst%XGAMW*log(zzw2))
224 !
225 ! S, Undersaturation over water (with new theta^u)
226 zusw(jl) = 1.0 - prvt(jl)*(ppres(jl)-prrevav(jl)) / (cst%XEPSILO * prrevav(jl))
227 !
228 prrevav(jl) = (cst%XLVTT+(cst%XCPV-cst%XCL)*(zzw2-cst%XTT))**2 / (pka(jl)*cst%XRV*zzw2**2) &
229 &+(cst%XRV*zzw2) / (pdv(jl)*prrevav(jl))
230 !
231 prrevav(jl) = max(0., zusw(jl))/(prhodref(jl)*prrevav(jl)) * &
232 & (icep%X0EVAR*zzw3**icep%XEX0EVAR+icep%X1EVAR*pcj(jl)*zzw3**icep%XEX1EVAR)
233 !
234 prrevav(jl) = prrevav(jl)*(zzw4-pcf(jl))
235 ENDIF
236 ELSE
237 prrevav(jl)=0.
238 ENDIF
239 ENDDO
240!$acc end kernels
241ELSE
242 CALL print_msg(nverb_fatal,'GEN','ICE4_WARM','wrong HSUBG_RR_EVAP case')
243END IF
244!
245IF (lhook) CALL dr_hook('ICE4_WARM', 1, zhook_handle)
246!
247END SUBROUTINE ice4_warm
248END MODULE mode_ice4_warm
