13  &PCF, PRF, &
14  &PRVT, PRCT, PRRT, &
15  &PRCAUTR, PRCACCR, PRREVAV)
16 !!
17 !!** PURPOSE
18 !! -------
19 !! Computes the warm process
20 !!
21 !! AUTHOR
22 !! ------
23 !! S. Riette from the plitting of rain_ice source code (nov. 2014)
24 !!
25 !! MODIFICATIONS

28 !! R. El Khatib 24-Aug-2021 Optimizations

31 !* 0. DECLARATIONS
32 ! ------------

34 USE modd_cst, ONLY: cst_t

36 USE modd_rain_ice_param_n, ONLY: rain_ice_param_t

38 USE mode_msg
39 USE yomhook , ONLY : lhook, dr_hook, jphook
40 !
41 IMPLICIT NONE
42 !
43 !* 0.1 Declarations of dummy arguments :
44 !
45 TYPE(cst_t), INTENT(IN) :: CST
46 TYPE(rain_ice_param_t), INTENT(IN) :: ICEP
47 TYPE(rain_ice_descr_t), INTENT(IN) :: ICED
48 INTEGER, INTENT(IN) :: KPROMA, KSIZE
49 LOGICAL, INTENT(IN) :: LDSOFT
50 LOGICAL, DIMENSION(KPROMA), INTENT(IN) :: LDCOMPUTE
51 CHARACTER(LEN=80), INTENT(IN) :: HSUBG_RC_RR_ACCR ! subgrid rc-rr accretion
52 CHARACTER(LEN=80), INTENT(IN) :: HSUBG_RR_EVAP ! subgrid rr evaporation
53 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRHODREF ! Reference density
54 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLVFACT
55 REAL, DIMENSION(KPROMA), INTENT(IN) :: PT ! Temperature
56 REAL, DIMENSION(KPROMA), INTENT(IN) :: PPRES ! absolute pressure at t
57 REAL, DIMENSION(KPROMA), INTENT(IN) :: PTHT ! Theta at time t
58 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAR ! Slope parameter of the raindrop distribution
59 REAL, DIMENSION(KPROMA), INTENT(IN) :: PLBDAR_RF!like PLBDAR but for the Rain Fraction part
60 REAL, DIMENSION(KPROMA), INTENT(IN) :: PKA ! Thermal conductivity of the air
61 REAL, DIMENSION(KPROMA), INTENT(IN) :: PDV ! Diffusivity of water vapor in the air
62 REAL, DIMENSION(KPROMA), INTENT(IN) :: PCJ ! Function to compute the ventilation coefficient
63 REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLC_HCF ! HLCLOUDS : fraction of High Cloud Fraction in grid
64 REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLC_LCF ! HLCLOUDS : fraction of Low Cloud Fraction in grid
65 REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLC_HRC ! HLCLOUDS : LWC that is High LWC in grid
66 REAL, DIMENSION(KPROMA), INTENT(IN) :: PHLC_LRC ! HLCLOUDS : LWC that is Low LWC in grid
67 REAL, DIMENSION(KPROMA), INTENT(IN) :: PCF ! Cloud fraction
68 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRF ! Rain fraction
69 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRVT ! Water vapor m.r. at t
70 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRCT ! Cloud water m.r. at t
71 REAL, DIMENSION(KPROMA), INTENT(IN) :: PRRT ! Rain water m.r. at t
72 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRCAUTR ! Autoconversion of r_c for r_r production
73 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRCACCR ! Accretion of r_c for r_r production
74 REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRREVAV ! Evaporation of r_r
75 !
76 !* 0.2 declaration of local variables
77 !
78 REAL :: ZZW2, ZZW3, ZZW4
79 REAL, DIMENSION(KPROMA) :: ZUSW ! Undersaturation over water
80 REAL, DIMENSION(KPROMA) :: ZTHLT ! Liquid potential temperature
81 REAL(KIND=jphook) :: zhook_handle
82 INTEGER :: JL
83 LOGICAL :: LMASK, LMASK1, LMASK2
84 !-------------------------------------------------------------------------------
85 !
86 IF (lhook) CALL dr_hook('ICE4_WARM', 0, zhook_handle)
87 !
88 !
89 !-------------------------------------------------------------------------------
90 !
91 !* 4.2 compute the autoconversion of r_c for r_r production: RCAUTR
92 !
93 DO jl=1, ksize
94  IF(phlc_hrc(jl)>iced%XRTMIN(2) .AND. phlc_hcf(jl)>0. .AND. ldcompute(jl)) THEN
95  IF(.NOT. ldsoft) THEN
96  !HCF*autoconv(HRC/HCF) with simplification
97  prcautr(jl) = icep%XTIMAUTC*max(phlc_hrc(jl) - phlc_hcf(jl)*icep%XCRIAUTC/prhodref(jl), 0.0)
98  ENDIF
99  ELSE
100  prcautr(jl) = 0.
101  ENDIF
102 ENDDO
103 !
104 !
105 !* 4.3 compute the accretion of r_c for r_r production: RCACCR
106 !
107 IF (hsubg_rc_rr_accr=='NONE') THEN
108  !CLoud water and rain are diluted over the grid box
109  DO jl=1, ksize
110  IF(prct(jl)>iced%XRTMIN(2) .AND. prrt(jl)>iced%XRTMIN(3) .AND. ldcompute(jl)) THEN
111  IF(.NOT. ldsoft) THEN
112  prcaccr(jl) = icep%XFCACCR * prct(jl) &
113  & * plbdar(jl)**icep%XEXCACCR &
114  & * prhodref(jl)**(-iced%XCEXVT)
115  ENDIF
116  ELSE
117  prcaccr(jl) = 0.
118  ENDIF
119  ENDDO
120 
121 ELSEIF (hsubg_rc_rr_accr=='PRFR') THEN
122  !Cloud water is concentrated over its fraction with possibly to parts with high and low content as set for autoconversion
123  !Rain is concentrated over its fraction
124  !Rain in high content area fraction: PHLC_HCF
125  !Rain in low content area fraction:
126  ! if PRF<PCF (rain is entirely falling in cloud): PRF-PHLC_HCF
127  ! if PRF>PCF (rain is falling in cloud and in clear sky): PCF-PHLC_HCF
128  ! => min(PCF, PRF)-PHLC_HCF
129  DO jl=1, ksize
130  lmask = prct(jl)>iced%XRTMIN(2) .AND. prrt(jl)>iced%XRTMIN(3) .AND. ldcompute(jl)
131  lmask1 = lmask .AND. phlc_hrc(jl)>iced%XRTMIN(2) .AND. phlc_hcf(jl)>0.
132  lmask2 = lmask .AND. phlc_lrc(jl)>iced%XRTMIN(2) .AND. phlc_lcf(jl)>1.e-20
133  IF(lmask1 .OR. lmask2) THEN
134  IF(.NOT. ldsoft) THEN
135  IF(lmask1) THEN
136  !Accretion due to rain falling in high cloud content
137  !HCF*accretion(HRC/HCF) with simplification
138  prcaccr(jl) = icep%XFCACCR * phlc_hrc(jl) &
139  &*plbdar_rf(jl)**icep%XEXCACCR &
140  &*prhodref(jl)**(-iced%XCEXVT)
141  ELSE
142  prcaccr(jl)=0.
143  ENDIF
144  IF(lmask2) THEN
145  !We add acrretion due to rain falling in low cloud content
146  prcaccr(jl) = prcaccr(jl) + icep%XFCACCR * ( phlc_lrc(jl)/phlc_lcf(jl) ) &
147  &*plbdar_rf(jl)**icep%XEXCACCR &
148  &*prhodref(jl)**(-iced%XCEXVT) &
149  &*(min(pcf(jl), prf(jl))-phlc_hcf(jl))
150  ENDIF
151  ENDIF
152  ELSE
153  prcaccr(jl)=0.
154  ENDIF
155  ENDDO
156 ELSE
157  CALL print_msg(nverb_fatal,'GEN','ICE4_WARM','wrong HSUBG_RC_RR_ACCR case')
158 ENDIF
159 !
160 !* 4.4 compute the evaporation of r_r: RREVAV
161 !
162 IF (hsubg_rr_evap=='NONE') THEN
163  DO jl=1, ksize
164  IF(prrt(jl)>iced%XRTMIN(3) .AND. prct(jl)<=iced%XRTMIN(2) .AND. ldcompute(jl)) THEN
165  IF(.NOT. ldsoft) THEN
166  prrevav(jl) = exp(cst%XALPW - cst%XBETAW/pt(jl) - cst%XGAMW*alog(pt(jl))) ! es_w
167  zusw(jl) = 1. - prvt(jl)*(ppres(jl)-prrevav(jl)) / (cst%XEPSILO * prrevav(jl)) ! Undersaturation over water
168  prrevav(jl) = (cst%XLVTT+(cst%XCPV-cst%XCL)*(pt(jl)-cst%XTT) )**2 / (pka(jl)*cst%XRV*pt(jl)**2) &
169  &+(cst%XRV*pt(jl)) / (pdv(jl)*prrevav(jl))
170  prrevav(jl) = (max(0.,zusw(jl) )/(prhodref(jl)*prrevav(jl)) ) * &
171  & (icep%X0EVAR*plbdar(jl)**icep%XEX0EVAR+icep%X1EVAR*pcj(jl)*plbdar(jl)**icep%XEX1EVAR)
172  ENDIF
173  ELSE
174  prrevav(jl)=0.
175  ENDIF
176  ENDDO
177 
178 ELSEIF (hsubg_rr_evap=='CLFR' .OR. hsubg_rr_evap=='PRFR') THEN
179  !ATTENTION
180  !Il faudrait recalculer les variables PKA, PDV, PCJ en tenant compte de la temp√©rature T^u
181  !Ces variables devraient √™tre sorties de rain_ice_slow et on mettrait le calcul de T^u, T^s
182  !et plusieurs versions (comme actuellement, en ciel clair, en ciel nuageux) de PKA, PDV, PCJ dans rain_ice
183  !On utiliserait la bonne version suivant l'option NONE, CLFR... dans l'√©vaporation et ailleurs
184 
185  DO jl=1, ksize
186  !Evaporation in clear sky part
187  !With CLFR, rain is diluted over the grid box
188  !With PRFR, rain is concentrated in its fraction
189  !Use temperature and humidity in clear sky part like Bechtold et al. (1993)
190  IF (hsubg_rr_evap=='CLFR') THEN
191  zzw4=1. !Precipitation fraction
192  zzw3=plbdar(jl)
193  ELSE
194  zzw4=prf(jl) !Precipitation fraction
195  zzw3=plbdar_rf(jl)
196  ENDIF
197 
198  IF(prrt(jl)>iced%XRTMIN(3) .AND. zzw4>pcf(jl) .AND. ldcompute(jl)) THEN
199  IF(.NOT. ldsoft) THEN
200  ! outside the cloud (environment) the use of T^u (unsaturated) instead of T
201  ! Bechtold et al. 1993
202  !
203  ! T_l
204  zthlt(jl) = ptht(jl) - cst%XLVTT*ptht(jl)/cst%XCPD/pt(jl)*prct(jl)
205  !
206  ! T^u = T_l = theta_l * (T/theta)
207  zzw2 = zthlt(jl) * pt(jl) / ptht(jl)
208  !
209  ! es_w with new T^u
210  prrevav(jl) = exp(cst%XALPW - cst%XBETAW/zzw2 - cst%XGAMW*alog(zzw2))
211  !
212  ! S, Undersaturation over water (with new theta^u)
213  zusw(jl) = 1.0 - prvt(jl)*(ppres(jl)-prrevav(jl)) / (cst%XEPSILO * prrevav(jl))
214  !
215  prrevav(jl) = (cst%XLVTT+(cst%XCPV-cst%XCL)*(zzw2-cst%XTT))**2 / (pka(jl)*cst%XRV*zzw2**2) &
216  &+(cst%XRV*zzw2) / (pdv(jl)*prrevav(jl))
217  !
218  prrevav(jl) = max(0., zusw(jl))/(prhodref(jl)*prrevav(jl)) * &
219  & (icep%X0EVAR*zzw3**icep%XEX0EVAR+icep%X1EVAR*pcj(jl)*zzw3**icep%XEX1EVAR)
220  !
221  prrevav(jl) = prrevav(jl)*(zzw4-pcf(jl))
222  ENDIF
223  ELSE
224  prrevav(jl)=0.
225  ENDIF
226  ENDDO
227 
228 ELSE
229  CALL print_msg(nverb_fatal,'GEN','ICE4_WARM','wrong HSUBG_RR_EVAP case')
230 END IF
231 !
232 IF (lhook) CALL dr_hook('ICE4_WARM', 1, zhook_handle)
233 !
234 END SUBROUTINE ice4_warm
235 END MODULE mode_ice4_warm
