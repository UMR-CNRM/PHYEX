157inegt = countjv( gnegt(:,:),i1(:),i3(:))
158!
159IF (inegt.GT.0) THEN
160
161 ALLOCATE(zrvt(inegt))
162 ALLOCATE(zrct(inegt))
163 ALLOCATE(zrrt(inegt))
164 ALLOCATE(zrit(inegt))
165 ALLOCATE(zrst(inegt))
166 ALLOCATE(zrgt(inegt))

168 ALLOCATE(ztht(inegt))

209 zexnref(il) = pexnref(i1(il),i3(il))

212! PACK : done
213! Prepare computations
214!
215 ALLOCATE( zlsfact(inegt) )
216 ALLOCATE( zlvfact(inegt) )
217 ALLOCATE( zsi(inegt) )
218 ALLOCATE( ztcelsius(inegt) )
219 ALLOCATE( zlbdac(inegt) )

221 ALLOCATE( zzw(inegt) ) ; zzw(:) = 0.0
222 ALLOCATE( zzx(inegt) ) ; zzx(:) = 0.0
223 ALLOCATE( zzy(inegt) ) ; zzy(:) = 0.0
224!
225 ztcelsius(:) = zzt(:)-cst%XTT ! T [°C]
226 zzw(:) = zexnref(:)*( cst%XCPD+cst%XCPV*zrvt(:)+cst%XCL*(zrct(:)+zrrt(:)) &
227 +cst%XCI*(zrit(:)+zrst(:)+zrgt(:)) )
228 zlsfact(:) = (cst%XLSTT+(cst%XCPV-cst%XCI)*ztcelsius(:))/zzw(:) ! L_s/(Pi_ref*C_ph)
229 zlvfact(:) = (cst%XLVTT+(cst%XCPV-cst%XCL)*ztcelsius(:))/zzw(:) ! L_v/(Pi_ref*C_ph)
230!
231 zzw(:) = exp( cst%XALPI - cst%XBETAI/zzt(:) - cst%XGAMI*log(zzt(:) ) ) ! es_i
232 zsi(:) = zrvt(:)*(zpres(:)-zzw(:))/((cst%XMV/cst%XMD)*zzw(:)) ! Saturation over ice

238!* 2. Haze homogeneous freezing
239! ------------------------
240!
241!
242! Compute the haze homogeneous nucleation source: RHHONI
243!

245
246! Sum of the available CCN
247 ALLOCATE( zfreeccn(inegt) )
248 ALLOCATE( zccnfrozen(inegt) )

258 END DO
259!
260 zzw(:) = 0.0
261 zzx(:) = 0.0
262 zeps = cst%XMV / cst%XMD
263 zzy(:) = limac%XCRITSAT1_HONH - & ! Critical Sat.
264 (min( limac%XTMAX_HONH,max( limac%XTMIN_HONH,zzt(:) ) )/limac%XCRITSAT2_HONH)
265!
266 ALLOCATE(zls(inegt))
267 ALLOCATE(zpsi1(inegt))
268 ALLOCATE(zpsi2(inegt))
269 ALLOCATE(ztau(inegt))
270 ALLOCATE(zbfact(inegt))
271!
272 WHERE( (zzt(:)<cst%XTT-35.0) .AND. (zsi(:)>zzy(:)) )
273 zls(:) = cst%XLSTT+(cst%XCPV-cst%XCI)*ztcelsius(:) ! Ls
274!
275 zpsi1(:) = zzy(:) * (cst%XG/(cst%XRD*zzt(:)))*(zeps*zls(:)/(cst%XCPD*zzt(:))-1.)
276! ! Psi1 (a1*Scr in KL01)
277! BV correction PSI2 enlever 1/ZEPS ?
278! ZPSI2(:) = ZSI(:) * (1.0/ZEPS+1.0/ZRVT(:)) + &
279 zpsi2(:) = zsi(:) * (1.0/zrvt(:)) + &
280 zzy(:) * ((zls(:)/zzt(:))**2)/(cst%XCPD*cst%XRV)
281! ! Psi2 (a2+a3*Scr in KL01)
282 ztau(:) = 1.0 / ( max( limac%XC1_HONH,limac%XC1_HONH*(limac%XC2_HONH-limac%XC3_HONH*zzt(:)) ) *&
283 abs( (limac%XDLNJODT1_HONH - limac%XDLNJODT2_HONH*zzt(:)) * &
284 ((zpres(:)/cst%XP00)**(cst%XRD/cst%XCPD))*ztht(:) ) )
285!
286 zbfact(:) = (limac%XRHOI_HONH/zrhodref(:)) * (zsi(:)/(zzy(:)-1.0)) &
287! BV correction ZBFACT enlever 1/ZEPS ?
288! * (1.0/ZRVT(:)+1.0/ZEPS) &
289 * (1.0/zrvt(:)) &
290 / (limac%XCOEF_DIFVAP_HONH*(zzt(:)**limac%XCEXP_DIFVAP_HONH /zpres(:)))
291!
292! BV correction ZZX rho_i{-1} ?
293! ZZX(:) = MAX( MIN( LIMAC%XRHOI_HONH*ZBFACT(:)**1.5 * (ZPSI1(:)/ZPSI2(:)) &
294 zzx(:) = max( min( (1/limac%XRHOI_HONH)*zbfact(:)**1.5 * (zpsi1(:)/zpsi2(:)) &
295 * (zw_nu(:)/sqrt(ztau(:))) , zfreeccn(:) ) , 0.)
296!
297 zzw(:) = min( limac%XRCOEF_HONH*zzx(:)*(ztau(:)/zbfact(:))**1.5 , zrvt(:) )
298 END WHERE
299!
300! Apply the changes
301 DO imod_ccn = 1, limap%NMOD_CCN
302 WHERE(zfreeccn(:)>1.)
303 zccnfrozen(:) = zzx(:) * znft(:,imod_ccn)/zfreeccn(:)
304 END WHERE
305 pnft(:,:,imod_ccn) = pnft(:,:,imod_ccn) - unpack( zccnfrozen(:), mask=gnegt(:,:),field=0.)
306 END DO
307!
308 ptot_rv_honh(:,:) = unpack( zzw(:), mask=gnegt(:,:),field=0.)
309!
310 ptht(:,:) = ptht(:,:) + unpack( zzw(:)*(zlsfact(:)-zlvfact(:)), mask=gnegt(:,:),field=0.)
311 prvt(:,:) = prvt(:,:) - unpack( zzw(:), mask=gnegt(:,:),field=0.)
312 prit(:,:) = prit(:,:) + unpack( zzw(:), mask=gnegt(:,:),field=0.)
313 IF (.NOT. limap%LCRYSTAL_SHAPE) THEN
314 pcit(:,:) = pcit(:,:) + unpack( zzx(:), mask=gnegt(:,:),field=0.)
315 ELSE
316 !--> Hyp : etant donnee la gamme de temperatures, formation de 20% de colonnes et

318 !PCIT_SHAPE(:,:,:,2) = PCIT_SHAPE(:,:,:,2) + &
319 ! 0.2 * UNPACK( ZZX(:), MASK=GNEGT(:,:,:),FIELD=0.)
320 !PCIT_SHAPE(:,:,:,3) = PCIT_SHAPE(:,:,:,3) + &
321 ! 0.8 * UNPACK( ZZX(:), MASK=GNEGT(:,:,:),FIELD=0.)
322 !--> Hyp : la congelation des CCN et des gouttelettes produit des droxtals

325 END IF
326 pnht(:,:) = pnht(:,:) + unpack( zzx(:), mask=gnegt(:,:),field=0.)
