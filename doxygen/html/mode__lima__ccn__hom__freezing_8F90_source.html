142 !

145 iie=SIZE(ptht,1) - jphext
146 ijb=1+jphext

148 ikb=1+jpvext
149 ike=SIZE(ptht,3) - jpvext
150 !
151 ! Temperature
152 zt(:,:,:) = ptht(:,:,:) * ( ppabst(:,:,:)/cst%XP00 ) ** (cst%XRD/cst%XCPD)
153 !
154 znht(:,:,:) = pnht(:,:,:)
155 !
156 ! Computations only where the temperature is below -35°C
157 ! PACK variables
158 !
159 gnegt(:,:,:) = .false.
160 gnegt(iib:iie,ijb:ije,ikb:ike) = zt(iib:iie,ijb:ije,ikb:ike)<cst%XTT-35.0
161 inegt = countjv( gnegt(:,:,:),i1(:),i2(:),i3(:))
162 !
163 IF (inegt.GT.0) THEN
164 
165  ALLOCATE(zrvt(inegt))
166  ALLOCATE(zrct(inegt))
167  ALLOCATE(zrrt(inegt))
168  ALLOCATE(zrit(inegt))
169  ALLOCATE(zrst(inegt))
170  ALLOCATE(zrgt(inegt))

172  ALLOCATE(ztht(inegt))

176  ALLOCATE(zcit(inegt))

179  ALLOCATE(zznht(inegt))
180  !
181  ALLOCATE(zrhodref(inegt))
182  ALLOCATE(zzt(inegt))
183  ALLOCATE(zpres(inegt))
184  ALLOCATE(zexnref(inegt))

186  DO jl=1,inegt
187  zrvt(jl) = prvt(i1(jl),i2(jl),i3(jl))
188  zrct(jl) = prct(i1(jl),i2(jl),i3(jl))
189  zrrt(jl) = prrt(i1(jl),i2(jl),i3(jl))
190  zrit(jl) = prit(i1(jl),i2(jl),i3(jl))
191  zrst(jl) = prst(i1(jl),i2(jl),i3(jl))
192  zrgt(jl) = prgt(i1(jl),i2(jl),i3(jl))
193  !
194  ztht(jl) = ptht(i1(jl),i2(jl),i3(jl))
195  !
196  zcct(jl) = pcct(i1(jl),i2(jl),i3(jl))
197  zcrt(jl) = pcrt(i1(jl),i2(jl),i3(jl))
198  zcit(jl) = pcit(i1(jl),i2(jl),i3(jl))
199  !
200  DO jmod_ccn = 1, nmod_ccn
201  znft(jl,jmod_ccn) = pnft(i1(jl),i2(jl),i3(jl),jmod_ccn)
202  ENDDO
203  zznht(jl) = znht(i1(jl),i2(jl),i3(jl))
204  zrhodref(jl) = prhodref(i1(jl),i2(jl),i3(jl))
205  zzt(jl) = zt(i1(jl),i2(jl),i3(jl))
206  zpres(jl) = ppabst(i1(jl),i2(jl),i3(jl))
207  zexnref(jl) = pexnref(i1(jl),i2(jl),i3(jl))
208  ENDDO
209 !
210 ! PACK : done
211 ! Prepare computations
212 !
213  ALLOCATE( zlsfact(inegt) )
214  ALLOCATE( zlvfact(inegt) )
215  ALLOCATE( zsi(inegt) )
216  ALLOCATE( ztcelsius(inegt) )
217  ALLOCATE( zlbdac(inegt) )
218 !
219  ALLOCATE( zzw(inegt) ) ; zzw(:) = 0.0
220  ALLOCATE( zzx(inegt) ) ; zzx(:) = 0.0
221  ALLOCATE( zzy(inegt) ) ; zzy(:) = 0.0
222 !
223  ztcelsius(:) = zzt(:)-cst%XTT ! T [°C]
224  zzw(:) = zexnref(:)*( cst%XCPD+cst%XCPV*zrvt(:)+cst%XCL*(zrct(:)+zrrt(:)) &
225  +cst%XCI*(zrit(:)+zrst(:)+zrgt(:)) )
226  zlsfact(:) = (cst%XLSTT+(cst%XCPV-cst%XCI)*ztcelsius(:))/zzw(:) ! L_s/(Pi_ref*C_ph)
227  zlvfact(:) = (cst%XLVTT+(cst%XCPV-cst%XCL)*ztcelsius(:))/zzw(:) ! L_v/(Pi_ref*C_ph)
228 !
229  zzw(:) = exp( cst%XALPI - cst%XBETAI/zzt(:) - cst%XGAMI*alog(zzt(:) ) ) ! es_i
230  zsi(:) = zrvt(:)*(zpres(:)-zzw(:))/((cst%XMV/cst%XMD)*zzw(:)) ! Saturation over ice
231 !
232 !
233 !-------------------------------------------------------------------------------
234 !
235 !
236 !* 2. Haze homogeneous freezing
237 ! ------------------------
238 !
239 !
240 ! Compute the haze homogeneous nucleation source: RHHONI
241 !
242  IF( nmod_ccn.GT.0 ) THEN
243 
244 ! Sum of the available CCN
245  ALLOCATE( zfreeccn(inegt) )
246  ALLOCATE( zccnfrozen(inegt) )
247  zfreeccn(:)=0.
248  zccnfrozen(:)=0.
249  DO jmod_ccn = 1, nmod_ccn
250  zfreeccn(:) = zfreeccn(:) + znft(:,jmod_ccn)
251  END DO
252 !
253  ALLOCATE(zw_nu(inegt))
254  DO jl=1,inegt
255  zw_nu(jl) = pw_nu(i1(jl),i2(jl),i3(jl))
256  END DO
257 !
258  zzw(:) = 0.0
259  zzx(:) = 0.0
260  zeps = cst%XMV / cst%XMD
261  zzy(:) = xcritsat1_honh - & ! Critical Sat.
262  (min( xtmax_honh,max( xtmin_honh,zzt(:) ) )/xcritsat2_honh)
263 !
264  ALLOCATE(zls(inegt))
265  ALLOCATE(zpsi1(inegt))
266  ALLOCATE(zpsi2(inegt))
267  ALLOCATE(ztau(inegt))
268  ALLOCATE(zbfact(inegt))
269 !
270  WHERE( (zzt(:)<cst%XTT-35.0) .AND. (zsi(:)>zzy(:)) )
271  zls(:) = cst%XLSTT+(cst%XCPV-cst%XCI)*ztcelsius(:) ! Ls
272 !
273  zpsi1(:) = zzy(:) * (cst%XG/(cst%XRD*zzt(:)))*(zeps*zls(:)/(cst%XCPD*zzt(:))-1.)
274 ! ! Psi1 (a1*Scr in KL01)
275 ! BV correction PSI2 enlever 1/ZEPS ?
276 ! ZPSI2(:) = ZSI(:) * (1.0/ZEPS+1.0/ZRVT(:)) + &
277  zpsi2(:) = zsi(:) * (1.0/zrvt(:)) + &
278  zzy(:) * ((zls(:)/zzt(:))**2)/(cst%XCPD*cst%XRV)
279 ! ! Psi2 (a2+a3*Scr in KL01)
280  ztau(:) = 1.0 / ( max( xc1_honh,xc1_honh*(xc2_honh-xc3_honh*zzt(:)) ) *&
281  abs( (xdlnjodt1_honh - xdlnjodt2_honh*zzt(:)) * &
282  ((zpres(:)/cst%XP00)**(cst%XRD/cst%XCPD))*ztht(:) ) )
283 !
284  zbfact(:) = (xrhoi_honh/zrhodref(:)) * (zsi(:)/(zzy(:)-1.0)) &
285 ! BV correction ZBFACT enlever 1/ZEPS ?
286 ! * (1.0/ZRVT(:)+1.0/ZEPS) &
287  * (1.0/zrvt(:)) &
288  / (xcoef_difvap_honh*(zzt(:)**xcexp_difvap_honh /zpres(:)))
289 !
290 ! BV correction ZZX rho_i{-1} ?
291 ! ZZX(:) = MAX( MIN( XRHOI_HONH*ZBFACT(:)**1.5 * (ZPSI1(:)/ZPSI2(:)) &
292  zzx(:) = max( min( (1/xrhoi_honh)*zbfact(:)**1.5 * (zpsi1(:)/zpsi2(:)) &
293  * (zw_nu(:)/sqrt(ztau(:))) , zfreeccn(:) ) , 0.)
294 !
295  zzw(:) = min( xrcoef_honh*zzx(:)*(ztau(:)/zbfact(:))**1.5 , zrvt(:) )
296  END WHERE
297 !
298 ! Apply the changes
299  DO jmod_ccn = 1, nmod_ccn
300  WHERE(zfreeccn(:)>1.)
301  zccnfrozen(:) = zzx(:) * znft(:,jmod_ccn)/zfreeccn(:)
302  END WHERE
303  pnft(:,:,:,jmod_ccn) = pnft(:,:,:,jmod_ccn) - unpack( zccnfrozen(:), mask=gnegt(:,:,:),field=0.)
304  END DO
305 !
306  ptot_rv_honh(:,:,:) = unpack( zzw(:), mask=gnegt(:,:,:),field=0.)
307 !
308  ptht(:,:,:) = ptht(:,:,:) + unpack( zzw(:)*(zlsfact(:)-zlvfact(:)), mask=gnegt(:,:,:),field=0.)
309  prvt(:,:,:) = prvt(:,:,:) - unpack( zzw(:), mask=gnegt(:,:,:),field=0.)
310  prit(:,:,:) = prit(:,:,:) + unpack( zzw(:), mask=gnegt(:,:,:),field=0.)
311  pcit(:,:,:) = pcit(:,:,:) + unpack( zzx(:), mask=gnegt(:,:,:),field=0.)
312  pnht(:,:,:) = pnht(:,:,:) + unpack( zzx(:), mask=gnegt(:,:,:),field=0.)
313 
314  DEALLOCATE(zfreeccn)
315  DEALLOCATE(zccnfrozen)
316  DEALLOCATE(zls)
317  DEALLOCATE(zpsi1)
318  DEALLOCATE(zpsi2)
319  DEALLOCATE(ztau)
320  DEALLOCATE(zbfact)
321  DEALLOCATE(zw_nu)
322 !
323  END IF
324 !
