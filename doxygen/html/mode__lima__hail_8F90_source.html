146!
147!
148IF (lhook) CALL dr_hook('LIMA_HAIL', 0, zhook_handle)
149p_th_weth(:) = 0.
150p_rc_weth(:) = 0.
151p_cc_weth(:) = 0.
152p_rr_weth(:) = 0.
153p_cr_weth(:) = 0.
154p_ri_weth(:) = 0.
155p_ci_weth(:) = 0.
156p_rs_weth(:) = 0.
157p_cs_weth(:) = 0.
158p_rg_weth(:) = 0.
159p_cg_weth(:) = 0.
160p_rh_weth(:) = 0.
161!
162p_rg_cohg(:) = 0.
163p_cg_cohg(:) = 0.
164!
165p_th_hmlt(:) = 0.
166p_rr_hmlt(:) = 0.
167p_cr_hmlt(:) = 0.
168p_ch_hmlt(:) = 0.
169!
170zzw1(:) = 0. ! RCWETH
171zzw2(:) = 0. ! RIWETH
172zzw3(:) = 0. ! RSWETH
173zzw3n(:) = 0.! NSWETH
174zzw4(:) = 0. ! RGWETH
175zzw4n(:) = 0.! NGWETH
176zzw5(:) = 0. ! RCWETH+RRWETH
177zzw6(:) = 0. ! RSWETH
178!
179zrweth(:) = 0.
180!
181!
182!* 1. Hail growth by collection (wet case only ?)
183! -----------------------------------------------
184!
185! 1.a Collection of rc and ri
186! ---------------------------
187!
188WHERE( prht(:)>limap%XRTMIN(7) .AND. pcht(:)>limap%XCTMIN(7) .AND. odcompute(:) )
189 zzw(:) = pcht(:) * plbdh(:)**(-limam%XDH-2.0) * prhodref(:)**(1-limap%XCEXVT)
190 zzw1(:) = limam%XFWETH * prct(:) * zzw(:) ! RCWETH
191 zzw2(:) = limam%XFWETH * prit(:) * zzw(:) ! RIWETH
192END WHERE
193!
194!* 1.b Collection of rs
195! --------------------
196!
197gwet(:) = prst(:)>limap%XRTMIN(5) .AND. prht(:)>limap%XRTMIN(7) .AND. odcompute(:) .AND. &
198 pcst(:)>limap%XCTMIN(5) .AND. pcht(:)>limap%XCTMIN(7)
199!
200WHERE( gwet )
201!
202!* Select the (ZLBDAG,ZLBDAS) couplet
203!
204 zvec1(:) = plbdh(:)
205 zvec2(:) = plbds(:)
206!
207!* find the next lower indice for the ZLBDAG and for the ZLBDAS
208! in the geometrical set of (Lbda_g,Lbda_s) couplet use to
209! tabulate the SDRYG-kernel
210!
211 zvec1(:) = max( 1.0001, min( real(limam%NWETLBDAH)-0.0001, &
212 limam%XWETINTP1H * log( zvec1(:) ) + limam%XWETINTP2H ) )
213 ivec1(:) = int( zvec1(:) )
214 zvec1(:) = zvec1(:) - real( ivec1(:) )
215!
216 zvec2(:) = max( 1.0001, min( real(limam%NWETLBDAS)-0.0001, &
217 limam%XWETINTP1S * log( zvec2(:) ) + limam%XWETINTP2S ) )
218 ivec2(:) = int( zvec2(:) )
219 zvec2(:) = zvec2(:) - real( ivec2(:) )
220!
221!* perform the bilinear interpolation of the normalized
222! SDRYG-kernel
223 !
224 z1(:) = get_xker_sweth(ksize,ivec1(:)+1,ivec2(:)+1)
225 z2(:) = get_xker_sweth(ksize,ivec1(:)+1,ivec2(:) )
226 z3(:) = get_xker_sweth(ksize,ivec1(:) ,ivec2(:)+1)
227 z4(:) = get_xker_sweth(ksize,ivec1(:) ,ivec2(:) )
228 zvec3(:) = ( z1(:)* zvec2(:) &
229 - z2(:)*(zvec2(:) - 1.0) ) &
230 * zvec1(:) &
231 - ( z3(:)* zvec2(:) &
232 - z4(:)*(zvec2(:) - 1.0) ) &
233 * (zvec1(:) - 1.0)
234 zzw(:) = zvec3(:)
235 zzw3(:) = limam%XFSWETH * zzw(:) & ! RSWETH
236 * prst(:) * pcht(:) &
237 * prhodref(:)**(1-limap%XCEXVT) &
238 *( limam%XLBSWETH1/( plbdh(:)**2 ) + &
239 limam%XLBSWETH2/( plbdh(:) * plbds(:) ) + &
240 limam%XLBSWETH3/( plbds(:)**2) )
241!
242 z1(:) = get_xker_n_sweth(ksize,ivec1(:)+1,ivec2(:)+1)
243 z2(:) = get_xker_n_sweth(ksize,ivec1(:)+1,ivec2(:) )
244 z3(:) = get_xker_n_sweth(ksize,ivec1(:) ,ivec2(:)+1)
245 z4(:) = get_xker_n_sweth(ksize,ivec1(:) ,ivec2(:) )
246 zvec3(:) = ( z1(:)* zvec2(:) &
247 - z2(:)*(zvec2(:) - 1.0) ) &
248 * zvec1(:) &
249 - ( z3(:)* zvec2(:) &
250 - z4(:)*(zvec2(:) - 1.0) ) &
251 * (zvec1(:) - 1.0)
252 zzw(:) = zvec3(:)
253 zzw3n(:) = limam%XFNSWETH * zzw(:) & ! NSWETH
254 * pcst(:) * pcht(:) &
255 * prhodref(:)**(1-limap%XCEXVT) &
256 *( limam%XLBNSWETH1/( plbdh(:)**2 ) + &
257 limam%XLBNSWETH2/( plbdh(:) * plbds(:) ) + &
258 limam%XLBNSWETH3/( plbds(:)**2) )
259END WHERE
260!
261!* 1.c Collection of rg
262! ---------------------
263!
264gwet(:) = prgt(:)>limap%XRTMIN(6) .AND. prht(:)>limap%XRTMIN(7) .AND. odcompute(:) .AND. &
265 pcgt(:)>limap%XCTMIN(6) .AND. pcht(:)>limap%XCTMIN(7)
266!
267WHERE( gwet )
268!
269!* Select the (ZLBDAG,ZLBDAR) couplet
270!
271 zvec1(:) = plbdh(:)
272 zvec2(:) = plbdg(:)
273!
274!* Find the next lower indice for the ZLBDAG and for the ZLBDAR
275! in the geometrical set of (Lbda_g,Lbda_r) couplet use to
276! tabulate the RDRYG-kernel
277!
278 zvec1(:) = max( 1.0001, min( real(limam%NWETLBDAH)-0.0001, &
279 limam%XWETINTP1H * log( zvec1(:) ) + limam%XWETINTP2H ) )
280 ivec1(:) = int( zvec1(:) )
281 zvec1(:) = zvec1(:) - real( ivec1(:) )
282!
283 zvec2(:) = max( 1.0001, min( real(limam%NWETLBDAG)-0.0001, &
284 limam%XWETINTP1G * log( zvec2(:) ) + limam%XWETINTP2G ) )
285 ivec2(:) = int( zvec2(:) )
286 zvec2(:) = zvec2(:) - real( ivec2(:) )
287!
288!* Perform the bilinear interpolation of the normalized
289! RDRYG-kernel
290!
291 z1(:) = get_xker_gweth(ksize,ivec1(:)+1,ivec2(:)+1)
292 z2(:) = get_xker_gweth(ksize,ivec1(:)+1,ivec2(:) )
293 z3(:) = get_xker_gweth(ksize,ivec1(:) ,ivec2(:)+1)
294 z4(:) = get_xker_gweth(ksize,ivec1(:) ,ivec2(:) )
295 zvec3(:) = ( z1(:)* zvec2(:) &
296 - z2(:)*(zvec2(:) - 1.0) ) &
297 * zvec1(:) &
298 - ( z3(:)* zvec2(:) &
299 - z4(:)*(zvec2(:) - 1.0) ) &
300 * (zvec1(:) - 1.0)
301 zzw(:) = zvec3(:)
302 zzw4(:) = limam%XFGWETH * zzw(:) & ! RGWETH
303 * prgt(:) * pcht(:) &
304 * prhodref(:)**(1-limap%XCEXVT) &
305 *( limam%XLBGWETH1/( plbdh(:)**2 ) + &
306 limam%XLBGWETH2/( plbdh(:) * plbdg(:) ) + &
307 limam%XLBGWETH3/( plbdg(:)**2) )
308!
309 z1(:) = get_xker_n_gweth(ksize,ivec1(:)+1,ivec2(:)+1)
310 z2(:) = get_xker_n_gweth(ksize,ivec1(:)+1,ivec2(:) )
311 z3(:) = get_xker_n_gweth(ksize,ivec1(:) ,ivec2(:)+1)
312 z4(:) = get_xker_n_gweth(ksize,ivec1(:) ,ivec2(:) )
313 zvec3(:) = ( z1(:)* zvec2(:) &
314 - z2(:)*(zvec2(:) - 1.0) ) &
315 * zvec1(:) &
316 - ( z3(:)* zvec2(:) &
317 - z4(:)*(zvec2(:) - 1.0) ) &
318 * (zvec1(:) - 1.0)
319 zzw(:) = zvec3(:)
320 zzw4n(:) = limam%XFNGWETH * zzw(:) & ! NGWETH
321 * pcgt(:) * pcht(:) &
322 * prhodref(:)**(1-limap%XCEXVT) &
323 *( limam%XLBNGWETH1/( plbdh(:)**2 ) + &
324 limam%XLBNGWETH2/( plbdh(:) * plbdg(:) ) + &
325 limam%XLBNGWETH3/( plbdg(:)**2) )
326
327END WHERE
328!
329! 1.e Wet growth of hail
330! ----------------------
331!
332zzw(:) = 0.0
333WHERE( prht(:)>limap%XRTMIN(6) .AND. pcht(:) >limap%XCTMIN(6) .AND. odcompute(:) )
334 zzw(:) = prvt(:)*ppres(:)/((cst%XMV/cst%XMD)+prvt(:)) ! Vapor pressure
335 zzw(:) = pka(:)*(cst%XTT-pt(:)) + &
336 ( pdv(:)*(cst%XLVTT + ( cst%XCPV - cst%XCL ) * ( pt(:) - cst%XTT )) &
337 *(cst%XESTT-zzw(:))/(cst%XRV*pt(:)) )
338!
339! Total mass gained by hail in wet mode
340 zrweth(:) = max( 0.0, &
341 ( zzw(:) * pcht(:) * ( limam%X0DEPH* plbdh(:)**limam%XEX0DEPH + &
342 limam%X1DEPH*pcj(:)*plbdh(:)**limam%XEX1DEPH ) + &
343 ( zzw2(:)+zzw3(:)+zzw4(:) ) * ( cst%XLMTT + (cst%XCI-cst%XCL)*(cst%XTT-pt(:)) ) ) &
344 / (cst%XLMTT-cst%XCL*(cst%XTT-pt(:))) )
345 ! We must agregate, at least, the cold species
346 zrweth(:)=max(zrweth(:), zzw2(:)+zzw3(:)+zzw4(:))
347END WHERE
348WHERE( prht(:)>limap%XRTMIN(6) .AND. pcht(:) >limap%XCTMIN(6) .AND. prrt(:)>limap%XRTMIN(3) &
349 .AND. pcrt(:) >limap%XCTMIN(3) .AND. odcompute(:) )
350 ! Mass of rain frozen by hail RRWETH
351 zzw5(:) = zrweth(:) - zzw2(:) - zzw3(:) - zzw4(:) - zzw1(:)
352END WHERE
353!
354zzw(:) = 0.0
355WHERE( odcompute(:) .AND. pt(:)<cst%XTT .AND. zzw5(:)>0.0 )
356 p_rc_weth(:) = - zzw1(:)
357 p_cc_weth(:) = p_rc_weth(:) * pcct(:)/max(prct(:),limap%XRTMIN(2))
358 p_rr_weth(:) = - zzw5(:)
359 p_cr_weth(:) = p_rr_weth(:) * pcrt(:)/max(prrt(:),limap%XRTMIN(3))
360 p_ri_weth(:) = - zzw2(:)
361 p_ci_weth(:) = p_ri_weth(:) * pcit(:)/max(prit(:),limap%XRTMIN(4))
362 p_rs_weth(:) = - zzw3(:)
363 p_cs_weth(:) = - zzw3n(:)
364 p_rg_weth(:) = - zzw4(:)
365 p_cg_weth(:) = - zzw4n(:)
366 p_rh_weth(:) = zrweth(:)
367 !
368 p_th_weth(:) = (zzw5(:)+zzw1(:)) * (plsfact(:)-plvfact(:))
369END WHERE
370!
371!
372!* 2. Hail -> graupel conversion (COHG)
373! -------------------------------------
374!
375zthrh=0.01e-3
376zthrc=0.001e-3
377zzw(:) = 0.0
378gwet(:) = prht(:)<zthrh .AND. prct(:)<zthrc .AND. pt(:)<cst%XTT
379WHERE( gwet(:) )
380 p_rg_cohg(:) = prht(:) * min( 1.0,max( 0.0,1.0-(prct(:)/zthrc) ) )
381 p_cg_cohg(:) = p_rg_cohg(:) * pcht(:)/max(prht(:),limap%XRTMIN(7))
382END WHERE
383!
384!
385!* 3. Hail Melting (HMLT)
386! -----------------------
387!
388zzw(:) = 0.0
389WHERE( prht(:)>limap%XRTMIN(6) .AND. pcht(:)>limap%XCTMIN(7) .AND. pt(:)>cst%XTT .AND. odcompute(:) )
390 zzw(:) = prvt(:)*ppres(:)/((cst%XMV/cst%XMD)+prvt(:)) ! Vapor pressure
391 zzw(:) = pka(:)*(cst%XTT-pt(:)) + &
392 ( pdv(:)*(cst%XLVTT + ( cst%XCPV - cst%XCL ) * ( pt(:) - cst%XTT )) &
393 *(cst%XESTT-zzw(:))/(cst%XRV*pt(:)) )
394!
395! compute RHMLTR
396!
397 zzw(:) = max( 0.0,( -zzw(:) * pcht(:) * &
398 ( limam%X0DEPH* plbdh(:)**limam%XEX0DEPH + &
399 limam%X1DEPH*pcj(:)*plbdh(:)**limam%XEX1DEPH ) - &
400 ( zzw5(:) ) * ( cst%XCL*(cst%XTT-pt(:))) ) &
401 / cst%XLMTT )
402 p_rr_hmlt(:) = zzw(:)
403 p_cr_hmlt(:) = zzw(:) * 5.0e6 ! obtained after averaging, Dshed=1mm and 500 microns
404 p_ch_hmlt(:) = - zzw(:) * pcht(:)/prht(:)
405 !
406 p_th_hmlt(:) = - p_rr_hmlt(:) * (plsfact(:)-plvfact(:))
407END WHERE
408!
409!
410!
411!
412pa_rc(:) = pa_rc(:) + p_rc_weth(:)
413pa_cc(:) = pa_cc(:) + p_cc_weth(:)
414pa_rr(:) = pa_rr(:) + p_rr_weth(:) + p_rr_hmlt(:)
415pa_cr(:) = pa_cr(:) + p_cr_weth(:) + p_cr_hmlt(:)
416pa_ri(:) = pa_ri(:) + p_ri_weth(:)
417pa_ci(:) = pa_ci(:) + p_ci_weth(:)
418pa_rs(:) = pa_rs(:) + p_rs_weth(:)
419pa_cs(:) = pa_cs(:) + p_cs_weth(:)
420pa_rg(:) = pa_rg(:) + p_rg_weth(:) + p_rg_cohg(:)
421pa_cg(:) = pa_cg(:) + p_cg_weth(:) + p_cg_cohg(:)
422pa_rh(:) = pa_rh(:) + p_rh_weth(:) - p_rg_cohg(:) - p_rr_hmlt(:)
423pa_ch(:) = pa_ch(:) - p_cg_cohg(:) + p_ch_hmlt(:)
424pa_th(:) = pa_th(:) + p_th_weth(:) + p_th_hmlt(:)
425!
426!-------------------------------------------------------------------------------
427!
428IF (lhook) CALL dr_hook('LIMA_HAIL', 1, zhook_handle)
429CONTAINS
430 FUNCTION get_xker_sweth(KSIZE,KH,KS) RESULT(RET)

433 INTEGER, DIMENSION(KSIZE), INTENT(IN) :: kh
434 INTEGER, DIMENSION(KSIZE), INTENT(IN) :: ks
435 REAL, DIMENSION(KSIZE) :: ret
436 !
437REAL(kind=jphook) :: zhook_handle
438 INTEGER i
