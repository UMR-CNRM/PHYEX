183 !* 1. Hail growth by collection (wet case only ?)
184 ! -----------------------------------------------
185 !
186 ! 1.a Collection of rc and ri
187 ! ---------------------------
188 !
189 WHERE( prht(:)>xrtmin(7) .AND. pcht(:)>xctmin(7) .AND. ldcompute(:) )
190  zzw(:) = pcht(:) * plbdh(:)**(-xdh-2.0) * prhodref(:)**(1-xcexvt)
191  zzw1(:) = xfweth * prct(:) * zzw(:) ! RCWETH
192  zzw2(:) = xfweth * prit(:) * zzw(:) ! RIWETH
193 END WHERE
194 !
195 !* 1.b Collection of rs
196 ! --------------------
197 !
198 gwet(:) = prst(:)>xrtmin(5) .AND. prht(:)>xrtmin(7) .AND. ldcompute(:) .AND. &
199  pcst(:)>xctmin(5) .AND. pcht(:)>xctmin(7)
200 !
201 WHERE( gwet )
202 !
203 !* Select the (ZLBDAG,ZLBDAS) couplet
204 !
205  zvec1(:) = plbdh(:)
206  zvec2(:) = plbds(:)
207 !
208 !* find the next lower indice for the ZLBDAG and for the ZLBDAS
209 ! in the geometrical set of (Lbda_g,Lbda_s) couplet use to
210 ! tabulate the SDRYG-kernel
211 !
212  zvec1(:) = max( 1.0001, min( real(nwetlbdah)-0.0001, &
213  xwetintp1h * log( zvec1(:) ) + xwetintp2h ) )
214  ivec1(:) = int( zvec1(:) )
215  zvec1(:) = zvec1(:) - real( ivec1(:) )
216 !
217  zvec2(:) = max( 1.0001, min( real(nwetlbdas)-0.0001, &
218  xwetintp1s * log( zvec2(:) ) + xwetintp2s ) )
219  ivec2(:) = int( zvec2(:) )
220  zvec2(:) = zvec2(:) - real( ivec2(:) )
221 !
222 !* perform the bilinear interpolation of the normalized
223 ! SDRYG-kernel
224  !
225  z1(:) = get_xker_sweth(ivec1(:)+1,ivec2(:)+1)
226  z2(:) = get_xker_sweth(ivec1(:)+1,ivec2(:) )
227  z3(:) = get_xker_sweth(ivec1(:) ,ivec2(:)+1)
228  z4(:) = get_xker_sweth(ivec1(:) ,ivec2(:) )
229  zvec3(:) = ( z1(:)* zvec2(:) &
230  - z2(:)*(zvec2(:) - 1.0) ) &
231  * zvec1(:) &
232  - ( z3(:)* zvec2(:) &
233  - z4(:)*(zvec2(:) - 1.0) ) &
234  * (zvec1(:) - 1.0)
235  zzw(:) = zvec3(:)
236  zzw3(:) = xfsweth * zzw(:) & ! RSWETH
237  * prst(:) * pcht(:) &
238  * prhodref(:)**(1-xcexvt) &
239  *( xlbsweth1/( plbdh(:)**2 ) + &
240  xlbsweth2/( plbdh(:) * plbds(:) ) + &
241  xlbsweth3/( plbds(:)**2) )
242 !
243  z1(:) = get_xker_n_sweth(ivec1(:)+1,ivec2(:)+1)
244  z2(:) = get_xker_n_sweth(ivec1(:)+1,ivec2(:) )
245  z3(:) = get_xker_n_sweth(ivec1(:) ,ivec2(:)+1)
246  z4(:) = get_xker_n_sweth(ivec1(:) ,ivec2(:) )
247  zvec3(:) = ( z1(:)* zvec2(:) &
248  - z2(:)*(zvec2(:) - 1.0) ) &
249  * zvec1(:) &
250  - ( z3(:)* zvec2(:) &
251  - z4(:)*(zvec2(:) - 1.0) ) &
252  * (zvec1(:) - 1.0)
253  zzw(:) = zvec3(:)
254  zzw3n(:) = xfnsweth * zzw(:) & ! NSWETH
255  * pcst(:) * pcht(:) &
256  * prhodref(:)**(1-xcexvt) &
257  *( xlbnsweth1/( plbdh(:)**2 ) + &
258  xlbnsweth2/( plbdh(:) * plbds(:) ) + &
259  xlbnsweth3/( plbds(:)**2) )
260 END WHERE
261 !
262 !* 1.c Collection of rg
263 ! ---------------------
264 !
265 gwet(:) = prgt(:)>xrtmin(6) .AND. prht(:)>xrtmin(7) .AND. ldcompute(:) .AND. &
266  pcgt(:)>xctmin(6) .AND. pcht(:)>xctmin(7)
267 !
268 WHERE( gwet )
269 !
270 !* Select the (ZLBDAG,ZLBDAR) couplet
271 !
272  zvec1(:) = plbdh(:)
273  zvec2(:) = plbdg(:)
274 !
275 !* Find the next lower indice for the ZLBDAG and for the ZLBDAR
276 ! in the geometrical set of (Lbda_g,Lbda_r) couplet use to
277 ! tabulate the RDRYG-kernel
278 !
279  zvec1(:) = max( 1.0001, min( real(nwetlbdah)-0.0001, &
280  xwetintp1h * log( zvec1(:) ) + xwetintp2h ) )
281  ivec1(:) = int( zvec1(:) )
282  zvec1(:) = zvec1(:) - real( ivec1(:) )
283 !
284  zvec2(:) = max( 1.0001, min( real(nwetlbdag)-0.0001, &
285  xwetintp1g * log( zvec2(:) ) + xwetintp2g ) )
286  ivec2(:) = int( zvec2(:) )
287  zvec2(:) = zvec2(:) - real( ivec2(:) )
288 !
289 !* Perform the bilinear interpolation of the normalized
290 ! RDRYG-kernel
291 !
292  z1(:) = get_xker_gweth(ivec1(:)+1,ivec2(:)+1)
293  z2(:) = get_xker_gweth(ivec1(:)+1,ivec2(:) )
294  z3(:) = get_xker_gweth(ivec1(:) ,ivec2(:)+1)
295  z4(:) = get_xker_gweth(ivec1(:) ,ivec2(:) )
296  zvec3(:) = ( z1(:)* zvec2(:) &
297  - z2(:)*(zvec2(:) - 1.0) ) &
298  * zvec1(:) &
299  - ( z3(:)* zvec2(:) &
300  - z4(:)*(zvec2(:) - 1.0) ) &
301  * (zvec1(:) - 1.0)
302  zzw(:) = zvec3(:)
303  zzw4(:) = xfgweth * zzw(:) & ! RGWETH
304  * prgt(:) * pcht(:) &
305  * prhodref(:)**(1-xcexvt) &
306  *( xlbgweth1/( plbdh(:)**2 ) + &
307  xlbgweth2/( plbdh(:) * plbdg(:) ) + &
308  xlbgweth3/( plbdg(:)**2) )
309 !
310  z1(:) = get_xker_n_gweth(ivec1(:)+1,ivec2(:)+1)
311  z2(:) = get_xker_n_gweth(ivec1(:)+1,ivec2(:) )
312  z3(:) = get_xker_n_gweth(ivec1(:) ,ivec2(:)+1)
313  z4(:) = get_xker_n_gweth(ivec1(:) ,ivec2(:) )
314  zvec3(:) = ( z1(:)* zvec2(:) &
315  - z2(:)*(zvec2(:) - 1.0) ) &
316  * zvec1(:) &
317  - ( z3(:)* zvec2(:) &
318  - z4(:)*(zvec2(:) - 1.0) ) &
319  * (zvec1(:) - 1.0)
320  zzw(:) = zvec3(:)
321  zzw4n(:) = xfngweth * zzw(:) & ! NGWETH
322  * pcgt(:) * pcht(:) &
323  * prhodref(:)**(1-xcexvt) &
324  *( xlbngweth1/( plbdh(:)**2 ) + &
325  xlbngweth2/( plbdh(:) * plbdg(:) ) + &
326  xlbngweth3/( plbdg(:)**2) )
327 
328 END WHERE
329 !
330 ! 1.e Wet growth of hail
331 ! ----------------------
332 !
333 zzw(:) = 0.0
334 WHERE( prht(:)>xrtmin(6) .AND. pcht(:) >xctmin(6) .AND. ldcompute(:) )
335  zzw(:) = prvt(:)*ppres(:)/((xmv/xmd)+prvt(:)) ! Vapor pressure
336  zzw(:) = pka(:)*(xtt-pt(:)) + &
337  ( pdv(:)*(xlvtt + ( xcpv - xcl ) * ( pt(:) - xtt )) &
338  *(xestt-zzw(:))/(xrv*pt(:)) )
339 !
340 ! Total mass gained by hail in wet mode
341  zrweth(:) = max( 0.0, &
342  ( zzw(:) * pcht(:) * ( x0deph* plbdh(:)**xex0deph + &
343  x1deph*pcj(:)*plbdh(:)**xex1deph ) + &
344  ( zzw2(:)+zzw3(:)+zzw4(:) ) * ( xlmtt + (xci-xcl)*(xtt-pt(:)) ) ) &
345  / (xlmtt-xcl*(xtt-pt(:))) )
346  ! We must agregate, at least, the cold species
347  zrweth(:)=max(zrweth(:), zzw2(:)+zzw3(:)+zzw4(:))
348 END WHERE
349 WHERE( prht(:)>xrtmin(6) .AND. pcht(:) >xctmin(6) .AND. prrt(:)>xrtmin(3) .AND. pcrt(:) >xctmin(3) .AND. ldcompute(:) )
350  ! Mass of rain frozen by hail RRWETH
351  zzw5(:) = zrweth(:) - zzw2(:) - zzw3(:) - zzw4(:) - zzw1(:)
352 END WHERE
353 !
354 zzw(:) = 0.0
355 WHERE( ldcompute(:) .AND. pt(:)<xtt .AND. zzw5(:)>0.0 )
356  p_rc_weth(:) = - zzw1(:)
357  p_cc_weth(:) = p_rc_weth(:) * pcct(:)/max(prct(:),xrtmin(2))
358  p_rr_weth(:) = - zzw5(:)
359  p_cr_weth(:) = p_rr_weth(:) * pcrt(:)/max(prrt(:),xrtmin(3))
360  p_ri_weth(:) = - zzw2(:)
361  p_ci_weth(:) = p_ri_weth(:) * pcit(:)/max(prit(:),xrtmin(4))
362  p_rs_weth(:) = - zzw3(:)
363  p_cs_weth(:) = - zzw3n(:)
364  p_rg_weth(:) = - zzw4(:)
365  p_cg_weth(:) = - zzw4n(:)
366  p_rh_weth(:) = zrweth(:)
367  !
368  p_th_weth(:) = (zzw5(:)+zzw1(:)) * (plsfact(:)-plvfact(:))
369 END WHERE
370 !
371 !
372 !* 2. Hail -> graupel conversion (COHG)
373 ! -------------------------------------
374 !
375 zthrh=0.01e-3
376 zthrc=0.001e-3
377 zzw(:) = 0.0
378 gwet(:) = prht(:)<zthrh .AND. prct(:)<zthrc .AND. pt(:)<xtt
379 WHERE( gwet(:) )
380  p_rg_cohg(:) = prht * min( 1.0,max( 0.0,1.0-(prct(:)/zthrc) ) )
381  p_cg_cohg(:) = p_rg_cohg(:) * pcht(:)/max(prht(:),xrtmin(7))
382 END WHERE
383 !
384 !
385 !* 3. Hail Melting (HMLT)
386 ! -----------------------
387 !
388 zzw(:) = 0.0
389 WHERE( prht(:)>xrtmin(6) .AND. pcht(:)>xctmin(7) .AND. pt(:)>xtt .AND. ldcompute(:) )
390  zzw(:) = prvt(:)*ppres(:)/((xmv/xmd)+prvt(:)) ! Vapor pressure
391  zzw(:) = pka(:)*(xtt-pt(:)) + &
392  ( pdv(:)*(xlvtt + ( xcpv - xcl ) * ( pt(:) - xtt )) &
393  *(xestt-zzw(:))/(xrv*pt(:)) )
394 !
395 ! compute RHMLTR
396 !
397  zzw(:) = max( 0.0,( -zzw(:) * pcht(:) * &
398  ( x0deph* plbdh(:)**xex0deph + &
399  x1deph*pcj(:)*plbdh(:)**xex1deph ) - &
400  ( zzw5(:) ) * ( xcl*(xtt-pt(:))) ) &
401  / xlmtt )
402  p_rr_hmlt(:) = zzw(:)
403  p_cr_hmlt(:) = zzw(:) * 5.0e6 ! obtained after averaging, Dshed=1mm and 500 microns
404  p_ch_hmlt(:) = - zzw(:) * pcht(:)/prht(:)
405  !
406  p_th_hmlt(:) = - p_rr_hmlt(:) * (plsfact(:)-plvfact(:))
407 END WHERE
408 !
409 !
410 !
411 !
412 pa_rc(:) = pa_rc(:) + p_rc_weth(:)
413 pa_cc(:) = pa_cc(:) + p_cc_weth(:)
414 pa_rr(:) = pa_rr(:) + p_rr_weth(:) + p_rr_hmlt(:)
415 pa_cr(:) = pa_cr(:) + p_cr_weth(:) + p_cr_hmlt(:)
416 pa_ri(:) = pa_ri(:) + p_ri_weth(:)
417 pa_ci(:) = pa_ci(:) + p_ci_weth(:)
418 pa_rs(:) = pa_rs(:) + p_rs_weth(:)
419 pa_cs(:) = pa_cs(:) + p_cs_weth(:)
420 pa_rg(:) = pa_rg(:) + p_rg_weth(:) + p_rg_cohg(:)
421 pa_cg(:) = pa_cg(:) + p_cg_weth(:) + p_cg_cohg(:)
422 pa_rh(:) = pa_rh(:) + p_rh_weth(:) - p_rg_cohg(:) - p_rr_hmlt(:)
423 pa_ch(:) = pa_ch(:) - p_cg_cohg(:) + p_ch_hmlt(:)
424 pa_th(:) = pa_th(:) + p_th_weth(:) + p_th_hmlt(:)
425 !
426 !-------------------------------------------------------------------------------
427 !
428 CONTAINS
429  FUNCTION get_xker_sweth(GRAUPEL,SNOW) RESULT(RET)
430  INTEGER, DIMENSION(:), INTENT(IN) :: graupel
431  INTEGER, DIMENSION(:), INTENT(IN) :: snow
