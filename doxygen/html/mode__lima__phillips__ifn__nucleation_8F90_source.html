9 ! #################################################################################
10  SUBROUTINE lima_phillips_ifn_nucleation (CST, PTSTEP, &
11  PRHODREF, PEXNREF, PPABST, &
12  PTHT, PRVT, PRCT, PRRT, PRIT, PRST, PRGT, &
13  PCCT, PCIT, PNAT, PIFT, PINT, PNIT, &
14  P_TH_HIND, P_RI_HIND, P_CI_HIND, &
15  P_TH_HINC, P_RC_HINC, P_CC_HINC, &
16  PICEFR )
17 ! #################################################################################
18 !!
19 !! PURPOSE
20 !! -------
21 !! The purpose of this routine is to compute the heterogeneous nucleation
22 !! following Phillips (2008) for the time-split version of LIMA
23 !!
24 !!
25 !!** METHOD
26 !! ------
27 !! The parameterization of Phillips (2008) is based on observed nucleation
28 !! in the CFDC for a range of T and Si values. Phillips therefore defines a
29 !! reference activity spectrum, that is, for given T and Si values, the
30 !! reference concentration of primary ice crystals.
31 !!
32 !! The activation of IFN is closely related to their total surface. Thus,
33 !! the activable fraction of each IFN specie is determined by an integration
34 !! over the particle size distributions.
35 !!
36 !! Subroutine organisation :
37 !!
38 !! 1- Preliminary computations
39 !! 2- Check where computations are necessary, and pack variables
40 !! 3- Compute the saturation over water and ice
41 !! 4- Compute the reference activity spectrum
42 !! -> CALL LIMA_PHILLIPS_REF_SPECTRUM
43 !! Integrate over the size distributions to compute the IFN activable fraction
44 !! -> CALL LIMA_PHILLIPS_INTEG
45 !! 5- Heterogeneous nucleation of insoluble IFN
46 !! 6- Heterogeneous nucleation of coated IFN
47 !! 7- Unpack variables & deallocations
48 !!
49 !!
50 !! REFERENCE
51 !! ---------
52 !!
53 !! Phillips et al., 2008: An empirical parameterization of heterogeneous
54 !! ice nucleation for multiple chemical species of aerosols, J. Atmos. Sci.
55 !!
56 !! AUTHOR
57 !! ------
58 !! J.-M. Cohard * Laboratoire d'Aerologie*
59 !! J.-P. Pinty * Laboratoire d'Aerologie*
60 !! S. Berthet * Laboratoire d'Aerologie*
61 !! B. Vi√© * Laboratoire d'Aerologie*
62 !!
63 !! MODIFICATIONS
64 !! -------------
65 !! Original 15/03/2018
66 ! P. Wautelet 28/05/2019: move COUNTJV function to tools.f90
67 ! P. Wautelet 27/02/2020: bugfix: P_TH_HIND was not accumulated (will affect budgets) + add P_TH_HINC dummy argument
68 ! + change intent of *_HIND and *_HINC dummy arguments (INOUT->OUT)
69 !-------------------------------------------------------------------------------
70 !
71 !* 0. DECLARATIONS
72 ! ------------
73 !
74 USE modd_cst, ONLY: cst_t
75 USE modd_parameters, ONLY : jphext, jpvext
76 USE modd_param_lima, ONLY : nmod_ifn, nspecie, xfrac, &
77  nmod_ccn, nmod_imm, nind_specie, nindice_ccn_imm, &
78  xdsi0, nphillips
79 USE modd_param_lima_cold, ONLY : xmnu0
80 
81 use mode_tools, only: countjv
82 
83 USE mode_lima_phillips_integ, ONLY : lima_phillips_integ
84 USE mode_lima_phillips_ref_spectrum, ONLY : lima_phillips_ref_spectrum
85 
86 IMPLICIT NONE
87 !
88 !* 0.1 Declarations of dummy arguments :
89 !
90 TYPE(cst_t), INTENT(IN) :: CST
91 REAL, INTENT(IN) :: PTSTEP
92 !
93 REAL, DIMENSION(:,:,:), INTENT(IN) :: PRHODREF! Reference density
94 REAL, DIMENSION(:,:,:), INTENT(IN) :: PEXNREF ! Reference Exner function
95 REAL, DIMENSION(:,:,:), INTENT(IN) :: PPABST ! abs. pressure at time t
96 !
97 REAL, DIMENSION(:,:,:), INTENT(INOUT) :: PTHT ! Theta at time t
98 REAL, DIMENSION(:,:,:), INTENT(INOUT) :: PRVT ! Water vapor m.r. at t
99 REAL, DIMENSION(:,:,:), INTENT(INOUT) :: PRCT ! Cloud water m.r. at t
100 REAL, DIMENSION(:,:,:), INTENT(IN) :: PRRT ! Rain water m.r. at t
101 REAL, DIMENSION(:,:,:), INTENT(INOUT) :: PRIT ! Cloud ice m.r. at t
102 REAL, DIMENSION(:,:,:), INTENT(IN) :: PRST ! Snow/aggregate m.r. at t
103 REAL, DIMENSION(:,:,:), INTENT(IN) :: PRGT ! Graupel m.r. at t
104 !
105 REAL, DIMENSION(:,:,:), INTENT(INOUT) :: PCCT ! Cloud water conc. at t
106 REAL, DIMENSION(:,:,:), INTENT(INOUT) :: PCIT ! Cloud water conc. at t
107 REAL, DIMENSION(:,:,:,:), INTENT(INOUT) :: PNAT ! CCN conc. used for immersion nucl.
108 REAL, DIMENSION(:,:,:,:), INTENT(INOUT) :: PIFT ! Free IFN conc.
109 REAL, DIMENSION(:,:,:,:), INTENT(INOUT) :: PINT ! Nucleated IFN conc.
110 REAL, DIMENSION(:,:,:,:), INTENT(INOUT) :: PNIT ! Nucleated (by immersion) CCN conc.
111 !
112 REAL, DIMENSION(:,:,:), INTENT(OUT) :: P_TH_HIND
113 REAL, DIMENSION(:,:,:), INTENT(OUT) :: P_RI_HIND
114 REAL, DIMENSION(:,:,:), INTENT(OUT) :: P_CI_HIND
115 REAL, DIMENSION(:,:,:), INTENT(OUT) :: P_TH_HINC
116 REAL, DIMENSION(:,:,:), INTENT(OUT) :: P_RC_HINC
117 REAL, DIMENSION(:,:,:), INTENT(OUT) :: P_CC_HINC
118 !
119 REAL, DIMENSION(:,:,:), INTENT(INOUT) :: PICEFR
120 !
121 !
122 !* 0.2 Declarations of local variables :
123 !
124 !
125 INTEGER :: IIB, IIE, IJB, IJE, IKB, IKE ! Physical domain
126 INTEGER :: JL, JMOD_CCN, JMOD_IFN, JSPECIE, JMOD_IMM ! Loop index
127 INTEGER :: INEGT ! Case number of sedimentation, nucleation,
128 !
129 LOGICAL, DIMENSION(SIZE(PRHODREF,1),SIZE(PRHODREF,2),SIZE(PRHODREF,3)) &
130  :: GNEGT ! Test where to compute the nucleation
131 !
132 INTEGER, DIMENSION(SIZE(PRHODREF)) :: I1,I2,I3 ! Indexes for PACK replacement
133 !
134 REAL, DIMENSION(:), ALLOCATABLE :: ZRVT ! Water vapor m.r. at t
135 REAL, DIMENSION(:), ALLOCATABLE :: ZRCT ! Cloud water m.r. at t
136 REAL, DIMENSION(:), ALLOCATABLE :: ZRRT ! Rain water m.r. at t
137 REAL, DIMENSION(:), ALLOCATABLE :: ZRIT ! Pristine ice m.r. at t
138 REAL, DIMENSION(:), ALLOCATABLE :: ZRST ! Snow/aggregate m.r. at t
139 REAL, DIMENSION(:), ALLOCATABLE :: ZRGT ! Graupel/hail m.r. at t
140 !
141 REAL, DIMENSION(:), ALLOCATABLE :: ZCCT ! Cloud water conc. at t
142 !
143 REAL, DIMENSION(:,:), ALLOCATABLE :: ZNAT ! Cloud Cond. nuclei conc. source
144 REAL, DIMENSION(:,:), ALLOCATABLE :: ZIFT ! Free Ice nuclei conc. source
145 REAL, DIMENSION(:,:), ALLOCATABLE :: ZINT ! Nucleated Ice nuclei conc. source
146  !by Deposition/Contact
147 REAL, DIMENSION(:,:), ALLOCATABLE :: ZNIT ! Nucleated Ice nuclei conc. source
148  !by Immersion
149 !
150 REAL, DIMENSION(:), ALLOCATABLE &
151  :: ZRHODREF, & ! RHO Dry REFerence
152  ZZT, & ! Temperature
153  ZPRES, & ! Pressure
154  ZEXNREF, & ! EXNer Pressure REFerence
155  ZZW, & ! Work array
156  ZZX, & ! Work array
157  ZZY, & ! Work array
158  ZLSFACT, & ! L_s/(Pi_ref*C_ph)
159  ZLVFACT, & ! L_v/(Pi_ref*C_ph)
160  ZLBDAC, & ! Slope parameter of the cloud droplet distr.
161  ZSI, &
162  ZSW, &
163  ZSI_W
164 !
165 REAL, DIMENSION(SIZE(PRHODREF,1),SIZE(PRHODREF,2),SIZE(PRHODREF,3)) &
166  :: ZW, ZT ! work arrays
167 !
168 REAL, DIMENSION(:,:), ALLOCATABLE :: ZSI0, & ! Si threshold in H_X for X={DM,BC,O}
169  Z_FRAC_ACT ! Activable frac. of each AP species
170 REAL, DIMENSION(:), ALLOCATABLE :: ZTCELSIUS, ZZT_SI0_BC
171 !
172 !-------------------------------------------------------------------------------
173 !
174 !
175 !* 1. PRELIMINARY COMPUTATIONS
176 ! ------------------------
177 !
178 p_th_hind(:,:,:) = 0.
179 p_ri_hind(:,:,:) = 0.
180 p_ci_hind(:,:,:) = 0.
181 p_th_hinc(:,:,:) = 0.
182 p_rc_hinc(:,:,:) = 0.
183 p_cc_hinc(:,:,:) = 0.
184 !
185 ! Physical domain
186 !
187 iib=1+jphext
188 iie=SIZE(ptht,1) - jphext
189 ijb=1+jphext
190 ije=SIZE(ptht,2) - jphext
191 ikb=1+jpvext
192 ike=SIZE(ptht,3) - jpvext
193 !
194 ! Temperature
195 !
196 zt(:,:,:) = ptht(:,:,:) * ( ppabst(:,:,:)/cst%XP00 ) ** (cst%XRD/cst%XCPD)
197 !
198 ! Saturation over ice
199 !
200 zw(:,:,:) = exp( cst%XALPI - cst%XBETAI/zt(:,:,:) - cst%XGAMI*alog(zt(:,:,:) ) )
201 zw(:,:,:) = prvt(:,:,:)*( ppabst(:,:,:)-zw(:,:,:) ) / ( (cst%XMV/cst%XMD) * zw(:,:,:) )
202 !
203 !
204 !-------------------------------------------------------------------------------
205 !
206 !
207 !* 2. COMPUTATIONS ONLY WHERE NECESSARY : PACK
208 ! ----------------------------------------
209 !
210 !
211 gnegt(:,:,:) = .false.
212 gnegt(iib:iie,ijb:ije,ikb:ike) = zt(iib:iie,ijb:ije,ikb:ike)<cst%XTT-2.0 .AND. &
213  zw(iib:iie,ijb:ije,ikb:ike)>0.95
214 !
215 inegt = countjv( gnegt(:,:,:),i1(:),i2(:),i3(:))
216 !
217 IF (inegt > 0) THEN
218 !
219  ALLOCATE(zrvt(inegt))
220  ALLOCATE(zrct(inegt))
221  ALLOCATE(zrrt(inegt))
222  ALLOCATE(zrit(inegt))
223  ALLOCATE(zrst(inegt))
224  ALLOCATE(zrgt(inegt))
225 !
226  ALLOCATE(zcct(inegt))
227 !
228  ALLOCATE(znat(inegt,nmod_ccn))
229  ALLOCATE(zift(inegt,nmod_ifn))
230  ALLOCATE(zint(inegt,nmod_ifn))
231  ALLOCATE(znit(inegt,nmod_imm))
232 !
233  ALLOCATE(zrhodref(inegt))
234  ALLOCATE(zzt(inegt))
235  ALLOCATE(zpres(inegt))
236  ALLOCATE(zexnref(inegt))
237 !
238  DO jl=1,inegt
239  zrvt(jl) = prvt(i1(jl),i2(jl),i3(jl))
240  zrct(jl) = prct(i1(jl),i2(jl),i3(jl))
241  zrrt(jl) = prrt(i1(jl),i2(jl),i3(jl))
242  zrit(jl) = prit(i1(jl),i2(jl),i3(jl))
243  zrst(jl) = prst(i1(jl),i2(jl),i3(jl))
244  zrgt(jl) = prgt(i1(jl),i2(jl),i3(jl))
245 !
246  zcct(jl) = pcct(i1(jl),i2(jl),i3(jl))
247 !
248  DO jmod_ccn = 1, nmod_ccn
249  znat(jl,jmod_ccn) = pnat(i1(jl),i2(jl),i3(jl),jmod_ccn)
250  ENDDO
251  DO jmod_ifn = 1, nmod_ifn
252  zift(jl,jmod_ifn) = pift(i1(jl),i2(jl),i3(jl),jmod_ifn)
253  zint(jl,jmod_ifn) = pint(i1(jl),i2(jl),i3(jl),jmod_ifn)
254  ENDDO
255  DO jmod_imm = 1, nmod_imm
256  znit(jl,jmod_imm) = pnit(i1(jl),i2(jl),i3(jl),jmod_imm)
257  ENDDO
258  zrhodref(jl) = prhodref(i1(jl),i2(jl),i3(jl))
259  zzt(jl) = zt(i1(jl),i2(jl),i3(jl))
260  zpres(jl) = ppabst(i1(jl),i2(jl),i3(jl))
261  zexnref(jl) = pexnref(i1(jl),i2(jl),i3(jl))
262  ENDDO
263 !
264 ! PACK : done
265 ! Prepare computations
266 !
267  ALLOCATE( zlsfact(inegt) )
268  ALLOCATE( zlvfact(inegt) )
269  ALLOCATE( zsi(inegt) )
270  ALLOCATE( ztcelsius(inegt) )
271  ALLOCATE( zzt_si0_bc(inegt) )
272  ALLOCATE( zlbdac(inegt) )
273  ALLOCATE( zsi0(inegt,nspecie) )
274  ALLOCATE( z_frac_act(inegt,nspecie) ) ; z_frac_act(:,:) = 0.0
275  ALLOCATE( zsw(inegt) )
276  ALLOCATE( zsi_w(inegt) )
277 !
278  ALLOCATE( zzw(inegt) ) ; zzw(:) = 0.0
279  ALLOCATE( zzx(inegt) ) ; zzx(:) = 0.0
280  ALLOCATE( zzy(inegt) ) ; zzy(:) = 0.0
281 !
282 !
283 !-------------------------------------------------------------------------------
284 !
285 !
286 !* 3. COMPUTE THE SATURATION OVER WATER AND ICE
287 ! -----------------------------------------
288 !
289 !
290  ztcelsius(:) = zzt(:)-cst%XTT ! T [¬∞C]
291  zzw(:) = zexnref(:)*( cst%XCPD+cst%XCPV*zrvt(:)+cst%XCL*(zrct(:)+zrrt(:)) &
292  +cst%XCI*(zrit(:)+zrst(:)+zrgt(:)) )
293  zlsfact(:) = (cst%XLSTT+(cst%XCPV-cst%XCI)*ztcelsius(:))/zzw(:) ! L_s/(Pi_ref*C_ph)
294  zlvfact(:) = (cst%XLVTT+(cst%XCPV-cst%XCL)*ztcelsius(:))/zzw(:) ! L_v/(Pi_ref*C_ph)
295 !
296  zzw(:) = exp( cst%XALPI - cst%XBETAI/zzt(:) - cst%XGAMI*alog(zzt(:) ) ) ! es_i
297  zsi(:) = zrvt(:)*(zpres(:)-zzw(:))/((cst%XMV/cst%XMD)*zzw(:)) ! Saturation over ice
298 !
299  zzy(:) = exp( cst%XALPW - cst%XBETAW/zzt(:) - cst%XGAMW*alog(zzt(:) ) ) ! es_w
300  zsw(:) = zrvt(:)*(zpres(:)-zzy(:))/((cst%XMV/cst%XMD)*zzy(:)) ! Saturation over water
301 !
302  zsi_w(:)= zzy(:)/zzw(:) ! Saturation over ice at water saturation: es_w/es_i
303 !
304 ! Saturation parameters for H_X, with X={Dust/Metallic (2 modes), Black Carbon, Organic}
305 !
306  zsi0(:,1) = 1.0 + 10.0**( -1.0261 + 3.1656e-3* ztcelsius(:) &
307  + 5.3938e-4*(ztcelsius(:)**2) &
308  + 8.2584e-6*(ztcelsius(:)**3) ) ! with T [¬∞C]
309  zsi0(:,2) = zsi0(:,1) ! DM2 = DM1
310  zsi0(:,3) = 0.0 ! BC
311  zzt_si0_bc(:) = max( 198.0, min( 239.0,zzt(:) ) )
312  zsi0(:,3) = (-3.118e-5*zzt_si0_bc(:)+1.085e-2)*zzt_si0_bc(:)+0.5652 - xdsi0(3)
313  IF (nphillips == 8) THEN
314  zsi0(:,4) = zsi0(:,3) ! O = BC
315  ELSE IF (nphillips == 13) THEN
316  zsi0(:,4) = 1.15 ! BIO
317  END IF
318 !
319 !
320 !-------------------------------------------------------------------------------
321 !
322 !
323 !* 4. COMPUTE THE ACTIVABLE FRACTION OF EACH IFN SPECIE
324 ! -------------------------------------------------
325 !
326 !
327 ! Computation of the reference activity spectrum ( ZZY = N_{IN,1,*} )
328 !
329  CALL lima_phillips_ref_spectrum(cst, zzt, zsi, zsi_w, zzy)
330 !
331 ! For each aerosol species (DM1, DM2, BC, O), compute the fraction that may be activated
332 ! Z_FRAC_ACT(INEGT,NSPECIE) = fraction of each species that may be activated
333 !
334  CALL lima_phillips_integ(cst, zzt, zsi, zsi0, zsw, zzy, z_frac_act)
335 !
336 !
337 !-------------------------------------------------------------------------------
338 !
339 !
340 !* 5. COMPUTE THE HETEROGENEOUS NUCLEATION OF INSOLUBLE IFN
341 ! -----------------------------------------------------
342 !
343 !
344 !
345  DO jmod_ifn = 1,nmod_ifn ! IFN modes
346  zzx(:)=0.
347  DO jspecie = 1, nspecie ! Each IFN mode is mixed with DM1, DM2, BC, O
348  zzx(:)=zzx(:)+xfrac(jspecie,jmod_ifn)*(zift(:,jmod_ifn)+zint(:,jmod_ifn))* &
349  z_frac_act(:,jspecie)
350  END DO
351 ! Now : ZZX(:) = number conc. of activable AP.
352 ! Activated AP at this time step = activable AP - already activated AP
353  zzx(:) = min( zift(:,jmod_ifn), max( (zzx(:)-zint(:,jmod_ifn)),0.0 ))
354  zzw(:) = min( xmnu0*zzx(:), zrvt(:) )
355 ! Now : ZZX(:) = number conc. of AP activated at this time step (#/kg) from IFN mode JMOD_IFN
356 ! Now : ZZW(:) = mmr of ice nucleated at this time step (kg/kg) from IFN mode JMOD_IFN
357 !
358 ! Update the concentrations and MMR
359 !
360  zw(:,:,:) = unpack( zzx(:), mask=gnegt(:,:,:), field=0. )
361  pift(:,:,:,jmod_ifn) = pift(:,:,:,jmod_ifn) - zw(:,:,:)
362  pint(:,:,:,jmod_ifn) = pint(:,:,:,jmod_ifn) + zw(:,:,:)
363 !
364  p_ci_hind(:,:,:) = p_ci_hind(:,:,:) + zw(:,:,:)
365  pcit(:,:,:) = pcit(:,:,:) + zw(:,:,:)
366 !
367  zw(:,:,:) = unpack( zzw(:), mask=gnegt(:,:,:), field=0. )
368  p_ri_hind(:,:,:) = p_ri_hind(:,:,:) + zw(:,:,:)
369  prvt(:,:,:) = prvt(:,:,:) - zw(:,:,:)
370  prit(:,:,:) = prit(:,:,:) + zw(:,:,:)
371 !
372  zw(:,:,:) = unpack( zzw(:)*zlsfact(:), mask=gnegt(:,:,:), field=0. )
373  p_th_hind(:,:,:) = p_th_hind(:,:,:) + zw(:,:,:)
374  ptht(:,:,:) = ptht(:,:,:) + zw(:,:,:)
375  END DO
376 !
377 !
378 !-------------------------------------------------------------------------------
379 !
380 !
381 !* 6. COMPUTE THE HETEROGENEOUS NUCLEATION OF COATED IFN
382 ! --------------------------------------------------
383 !
384 !
385 ! Heterogeneous nucleation by immersion of the activated CCN
386 ! Currently, we represent coated IFN as a pure aerosol type (NIND_SPECIE)
387 !
388 !
389  DO jmod_imm = 1,nmod_imm ! Coated IFN modes
390  jmod_ccn = nindice_ccn_imm(jmod_imm) ! Corresponding CCN mode
391  IF (jmod_ccn .GT. 0) THEN
392 !
393 ! OLD LIMA : Compute the appropriate mean diameter and sigma
394 ! XMDIAM_IMM = MIN( XMDIAM_IFN(NIND_SPECIE) , XR_MEAN_CCN(JMOD_CCN)*2. )
395 ! XSIGMA_IMM = MIN( XSIGMA_IFN(JSPECIE) , EXP(XLOGSIG_CCN(JMOD_CCN)) )
396 !
397  zzw(:) = min( zcct(:) , znat(:,jmod_ccn) )
398  zzx(:)= ( zzw(:)+znit(:,jmod_imm) ) * z_frac_act(:,nind_specie)
399 ! Now : ZZX(:) = number of activable AP.
400 ! Activated AP at this time step = activable AP - already activated AP
401  zzx(:) = min( zzw(:), max( (zzx(:)-znit(:,jmod_imm)),0.0 ) )
402  zzy(:) = min( xmnu0*zzx(:) , zrvt(:), zrct(:) )
403 !
404 ! Update the concentrations and MMR
405 !
406  zw(:,:,:) = unpack( zzx(:), mask=gnegt(:,:,:), field=0. )
407  pnit(:,:,:,jmod_imm) = pnit(:,:,:,jmod_imm) + zw(:,:,:)
408  pnat(:,:,:,jmod_ccn) = pnat(:,:,:,jmod_ccn) - zw(:,:,:)
409 !
410  p_cc_hinc(:,:,:) = p_cc_hinc(:,:,:) - zw(:,:,:)
411  pcct(:,:,:) = pcct(:,:,:) - zw(:,:,:)
412  pcit(:,:,:) = pcit(:,:,:) + zw(:,:,:)
413 !
414  zw(:,:,:) = unpack( zzy(:), mask=gnegt(:,:,:), field=0. )
415  p_rc_hinc(:,:,:) = p_rc_hinc(:,:,:) - zw(:,:,:)
416  prct(:,:,:) = prct(:,:,:) - zw(:,:,:)
417  prit(:,:,:) = prit(:,:,:) + zw(:,:,:)
418 !
419  zw(:,:,:) = unpack( zzy(:)*zlsfact(:), mask=gnegt(:,:,:), field=0. )
420  p_th_hinc(:,:,:) = p_th_hinc(:,:,:) + zw(:,:,:)
421  ptht(:,:,:) = ptht(:,:,:) + zw(:,:,:)
422  END IF
423  END DO
424 !
425 !-------------------------------------------------------------------------------
426 !
427 !
428 !* 7. CLEAN
429 ! -----
430 !
431 !
432  DEALLOCATE(zrvt)
433  DEALLOCATE(zrct)
434  DEALLOCATE(zrrt)
435  DEALLOCATE(zrit)
436  DEALLOCATE(zrst)
437  DEALLOCATE(zrgt)
438  DEALLOCATE(zcct)
439  DEALLOCATE(znat)
440  DEALLOCATE(zift)
441  DEALLOCATE(zint)
442  DEALLOCATE(znit)
443  DEALLOCATE(zrhodref)
444  DEALLOCATE(zzt)
445  DEALLOCATE(zpres)
446  DEALLOCATE(zexnref)
447  DEALLOCATE(zlsfact)
448  DEALLOCATE(zlvfact)
449  DEALLOCATE(zsi)
450  DEALLOCATE(ztcelsius)
451  DEALLOCATE(zzt_si0_bc)
452  DEALLOCATE(zlbdac)
453  DEALLOCATE(zsi0)
454  DEALLOCATE(z_frac_act)
455  DEALLOCATE(zsw)
456  DEALLOCATE(zzw)
457  DEALLOCATE(zzx)
458  DEALLOCATE(zzy)
459  DEALLOCATE(zsi_w)
460 !
461 END IF ! INEGT > 0
462 !
463 !-------------------------------------------------------------------------------
464 !
