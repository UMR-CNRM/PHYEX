225SUBROUTINE param_mfshalln_init(HPROGRAM, TFILENAM, LDNEEDNAM, KLUOUT, &
226 &LDDEFAULTVAL, LDREADNAM, LDCHECK, KPRINT)
227!!*** *PARAM_MFSHALLN* - Code needed to initialize the MODD_PARAM_MFSHALL_n module
228!!
229!!* PURPOSE
230!! -------
231!! Sets the default values, reads the namelist, performs the checks and prints
232!!
233!!* METHOD
234!! ------
235!! 0. Declarations
236!! 1. Declaration of arguments
237!! 2. Declaration of local variables
238!! 1. Default values
239!! 2. Namelist
240!! 3. Checks
241!! 4. Prints
242!!
243!! AUTHOR
244!! ------
245!! S. Riette
246!!
247!! MODIFICATIONS
248!! -------------
249!! Original Feb 2023
250!-------------------------------------------------------------------------------
251!
252!* 0. DECLARATIONS
253! ---------------
254!
255USE mode_posnam_phy, ONLY: posnam_phy
256USE mode_check_nam_val, ONLY: check_nam_val_char
257USE modd_io, ONLY: tfiledata
258USE mode_msg, ONLY: print_msg, nverb_fatal
259!
260IMPLICIT NONE
261!
262!* 0.1. Declaration of arguments
263! ------------------------
264!
265CHARACTER(LEN=6), INTENT(IN) :: HPROGRAM
266TYPE(tfiledata), INTENT(IN) :: TFILENAM
267LOGICAL, INTENT(IN) :: LDNEEDNAM
268INTEGER, INTENT(IN) :: KLUOUT
269LOGICAL, OPTIONAL, INTENT(IN) :: LDDEFAULTVAL
270LOGICAL, OPTIONAL, INTENT(IN) :: LDREADNAM
271LOGICAL, OPTIONAL, INTENT(IN) :: LDCHECK
272INTEGER, OPTIONAL, INTENT(IN) :: KPRINT
274!
275!* 0.2 Declaration of local variables
276! ------------------------------
277!
278LOGICAL :: LLDEFAULTVAL, LLREADNAM, LLCHECK, LLFOUND
279INTEGER :: IPRINT
280
281lldefaultval=.true.
282llreadnam=.true.
283llcheck=.true.
284iprint=0
285IF(PRESENT(lddefaultval)) lldefaultval=lddefaultval
286IF(PRESENT(ldreadnam )) llreadnam =ldreadnam
287IF(PRESENT(ldcheck )) llcheck =ldcheck
288IF(PRESENT(kprint )) iprint =kprint
289!
290!* 1. DEFAULT VALUES
291! -----------------
292!
293IF(lldefaultval) THEN
294 !NOTES ON GENERAL DEFAULTS AND MODEL-SPECIFIC DEFAULTS :
295 !- General default values *MUST* remain unchanged.
296 !- To change the default value for a given application,
297 ! an "IF(HPROGRAM=='...')" condition must be used.
298
299 ximpl_mf=1.
300 cmf_updraft='EDKF'
301 cmf_cloud='DIRE'
302 cwet_mixing='PKFB'
303 ckic_compute='KFB'
304 cdetr_dry_lup='SURF'
305 lmixuv=.true.
306 lmixtke=.false.
307 lmf_flx=.false.
308 xalp_pert=0.3
309 xabuo=1.
310 xbentr=1.
311 xbdetr=0.
312 xcmf=0.065
313 xentr_mf=0.035
314 xcrad_mf=50.
315 xentr_dry=0.55
316 xdetr_dry=10.
317 xdetr_lup=1.
318 xkcf_mf=2.75
319 xkrc_mf=1.
320 xtausigmf=600.
321 xpres_uv=0.5
322 xalpha_mf=2.
323 xsigma_mf=20.
324 xsigma_env=0.
325 xfrac_up_max=0.33
326 xa1=2./3.
327 xb=0.002
328 xc=0.012
329 xbeta1=0.9
330 xr=2.
331 lthetas_mf=.false.
332 xlambda_mf=0.
333 lgz=.false.
334 xgz=1.83 ! between 1.83 and 1.33
335 lverlimup=.false.
336 lpz_exp_log=.false.
337 xbrio=0.
338 xaadvec=0.
339 lrelax_alpha_mf=.false.
340 IF(hprogram=='MESONH') lverlimup=.true.
341ENDIF
342!
343!* 2. NAMELIST
344! -----------
345!
346IF(llreadnam) THEN
347 CALL posnam_phy(tfilenam, 'NAM_PARAM_MFSHALLN', ldneednam, llfound)
348 IF(llfound) READ(unit=tfilenam%NLU, nml=nam_param_mfshalln)
349ENDIF
350!
351!* 3. CHECKS
352! ---------
353!
354IF(llcheck) THEN
355 CALL check_nam_val_char(kluout, 'CMF_CLOUD', cmf_cloud, 'NONE', 'STAT', 'DIRE', 'BIGA')
356 CALL check_nam_val_char(kluout, 'CWET_MIXING', cwet_mixing, 'PKFB', 'LR01')
357 CALL check_nam_val_char(kluout, 'CKIC_COMPUTE', ckic_compute, 'KFB', 'RS08')
358 CALL check_nam_val_char(kluout, 'CDETR_DRY_LUP', cdetr_dry_lup, 'SURF', 'UPDR')
359 CALL check_nam_val_char(kluout, 'CMF_UPDRAFT', cmf_updraft, 'NONE', 'EDKF', 'RHCJ', 'RAHA')
360 !
361 IF(lmixtke .AND. cmf_updraft /= 'EDKF')THEN
362 CALL print_msg(nverb_fatal, 'GEN', 'PARAM_MFSHALLN_INIT', &
363 &"LMIXTKE only implemented when CMF_UPDRAFT is 'EDKF'")
364 ENDIF
365ENDIF
366!
367!* 3. PRINTS
368! ---------
369!
370IF(iprint>=1) THEN
371 WRITE(unit=kluout,nml=nam_param_mfshalln)
372ENDIF
373!
374END SUBROUTINE param_mfshalln_init
375!
376END MODULE modd_param_mfshall_n
