55REAL, DIMENSION(KPROMA), INTENT(INOUT) :: PRCBERI ! Bergeron-Findeisen effect
56!
57!* 0.2 declaration of local variables
58!
59REAL(KIND=jphook) :: zhook_handle
60INTEGER :: JL
61!
62!-------------------------------------------------------------------------------
63IF (lhook) CALL dr_hook('ICE4_FAST_RI',0,zhook_handle)
64!
65!-------------------------------------------------------------------------------
66!
67!* 7.2 Bergeron-Findeisen effect: RCBERI
68!
69!$acc kernels
70DO jl=1, ksize
71 IF(pssi(jl)>0. .AND. prct(jl)>iced%XRTMIN(2) .AND. prit(jl)>iced%XRTMIN(4) &
72 .AND. pcit(jl)>1.e-20 .AND. ldcompute(jl)) THEN
73 IF(.NOT. ldsoft) THEN
74 prcberi(jl) = min(1.e8, iced%XLBI*(prhodref(jl)*prit(jl)/pcit(jl))**iced%XLBEXI) ! Lbda_i
75 prcberi(jl) = ( pssi(jl) / (prhodref(jl)*pai(jl)) ) * pcit(jl) * &
76 ( icep%X0DEPI/prcberi(jl) + icep%X2DEPI*pcj(jl)*pcj(jl)/prcberi(jl)**(iced%XDI+2.0) )
77 ENDIF
78 ELSE
79 prcberi(jl) = 0.
80 ENDIF
81ENDDO
82!$acc end kernels
83!
84IF (lhook) CALL dr_hook('ICE4_FAST_RI', 1, zhook_handle)
85!
86END SUBROUTINE ice4_fast_ri
