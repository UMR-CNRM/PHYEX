13 !!**** *COMPUTE_FUNCTION_THERMO* routine to compute several thermo functions

25 !-------------------------------------------------------------------------------
26 !
27 !* 0. DECLARATIONS
28 ! ------------
29 !
30 USE yomhook, ONLY: lhook, dr_hook, jphook
31 USE modd_cst, ONLY: cst_t
32 USE modd_dimphyex, ONLY: dimphyex_t
33
34 IMPLICIT NONE
35 !
36 !* 0.1 Declarations of dummy arguments
37 !
38 TYPE(dimphyex_t), INTENT(IN) :: D ! PHYEX variables dimensions structure
39 REAL, INTENT(IN) :: PALP, PBETA, PGAM, PLTT, PC
40 REAL, INTENT(IN), DIMENSION(D%NIJT, D%NKT) :: PT, PEXN, PCP
41 !
42 REAL, INTENT(OUT), DIMENSION(D%NIJT, D%NKT) :: PLOCPEXN
43 REAL, INTENT(OUT), DIMENSION(D%NIJT, D%NKT) :: PAMOIST, PATHETA
44 TYPE(cst_t), INTENT(IN) :: CST
45 REAL, INTENT(IN) :: PPABST(D%NIJT, D%NKT)
46
47 REAL :: ZDRVSATDT(D%NIJT, D%NKT)
48 REAL :: ZRVSAT(D%NIJT, D%NKT)
49 REAL :: ZEPS
50 REAL(KIND=jphook) :: zhook_handle2
51 INTEGER :: JK, JIJ, IIJB, IIJE, IKT
52 !
53 !-------------------------------------------------------------------------------
54 !
55 IF (lhook) CALL dr_hook('TURB:COMPUTE_FUNCTION_THERMO_NEW_STAT', 0, zhook_handle2)
56
57 ikt = d%NKT
58 iije = d%NIJE
59 iijb = d%NIJB
60 zeps = cst%XMV / cst%XMD
61 !
62 !* 1.1 Lv/Cph at t
63 !
64!$acc kernels
65!$mnh_expand_array ( JIJ=IIJB:IIJE,JK=1:IKT )
66 plocpexn(iijb:iije, 1:ikt) = (pltt + (cst%XCPV - pc)*(pt(iijb:iije, 1:ikt) - cst%XTT)) / pcp(iijb:iije, 1:ikt)
67 !
68 !* 1.2 Saturation vapor pressure at t
69 !
70 zrvsat(iijb:iije, 1:ikt) = exp(palp - pbeta / pt(iijb:iije, 1:ikt) - pgam*log(pt(iijb:iije, 1:ikt)))
71 !
72 !* 1.3 saturation mixing ratio at t
73 !
74 zrvsat(iijb:iije, 1:ikt) = zrvsat(iijb:iije, 1:ikt)*zeps / (ppabst(iijb:iije, 1:ikt) - zrvsat(iijb:iije, 1:ikt))
75 !
76 !* 1.4 compute the saturation mixing ratio derivative (rvs')
77 !
78 zdrvsatdt(iijb:iije, 1:ikt) = (pbeta / pt(iijb:iije, 1:ikt) - pgam) / pt(iijb:iije, 1:ikt)*zrvsat(iijb:iije, 1:ikt)*(1. + &
79 & zrvsat(iijb:iije, 1:ikt) / zeps)
80 !
81 !* 1.5 compute Amoist
82 !
83 pamoist(iijb:iije, 1:ikt) = 1.0 / (1.0 + zdrvsatdt(iijb:iije, 1:ikt)*plocpexn(iijb:iije, 1:ikt))
84 !
85 !* 1.6 compute Atheta
86 !
87 patheta(iijb:iije, 1:ikt) = pamoist(iijb:iije, 1:ikt)*pexn(iijb:iije, 1:ikt)*zdrvsatdt(iijb:iije, 1:ikt)
88 !
89 !* 1.7 Lv/Cph/Exner at t-1
90 !
91 plocpexn(iijb:iije, 1:ikt) = plocpexn(iijb:iije, 1:ikt) / pexn(iijb:iije, 1:ikt)
92!$mnh_end_expand_array ( JIJ=IIJB:IIJE,JK=1:IKT )
93!$acc end kernels
94 !
95 IF (lhook) CALL dr_hook('TURB:COMPUTE_FUNCTION_THERMO_NEW_STAT', 1, zhook_handle2)
96 END SUBROUTINE compute_function_thermo_new_stat
