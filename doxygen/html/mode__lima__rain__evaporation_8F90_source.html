107  p_rr_evap(:) = - zzw2(:)
108

115 !

117 !
118 ! Compute the function G(T)
119 !
120  zzw2(:) = 1. / ( xrholw*((((plv(:)/pt(:))**2)/(xthco*xrv)) + (xrv*pt(:))/(xdiva*pevsat(:)))) !G
121 !
122 ! Compute the evaporation tendency
123 !
124  zzw2(:) = zzw2(:) * zzw1(:) * prrt(:) * &
125  (x0evar * plbdr(:)**xex0evar + x1evar * prhodref(:)**xex2evar * plbdr(:)**xex1evar)
126  zzw2(:) = max(zzw2(:),0.0)
127 !
128  p_rr_evap(:) = - zzw2(:)
129 ! P_TH_EVAP(:) = P_RR_EVAP(:) * PLVFACT(:)
130 ! PEVAP3D(:) = - P_RR_EVAP(:)
