104 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PWM ! w at time t
105 REAL, DIMENSION(D%NIJT,D%NKT,KSV), INTENT(IN) :: PSVM ! scalar var. at t-Delta t
106 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTKEM ! TKE at time t
107 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PLM ! Turb. mixing length
108 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PLEPS ! dissipative length
109 REAL, DIMENSION(D%NIJT,D%NKT,KSV), INTENT(IN) :: PPSI_SV ! Inv.Turb.Sch.for scalars
110  ! cumulated sources for the prognostic variables
111 !
112 !
113 !
114 !
115 !* 0.2 declaration of local variables
116 !
117 !
118 REAL, DIMENSION(D%NIJT,D%NKT) :: ZA, ZFLXZ, &
119  ZWORK1,ZWORK2,ZWORK3! working var. for shuman operators (array syntax)
120 !
121 REAL :: ZCSV !constant for the scalar flux
122 !
123 INTEGER :: JIJ,JK,JSV ! loop counters
124 INTEGER :: IIJB, IIJE, IKT
125 !
126 REAL :: ZTIME1, ZTIME2
127 !
128 REAL :: ZCSVD = 1.2 ! constant for scalar variance dissipation
129 REAL :: ZCTSVD = 2.4 ! constant for temperature - scalar covariance dissipation
130 REAL :: ZCQSVD = 2.4 ! constant for humidity - scalar covariance dissipation
131 !----------------------------------------------------------------------------
132 !
133 REAL(KIND=jphook) :: zhook_handle
134 IF (lhook) CALL dr_hook('TURB_VER_SV_CORR',0,zhook_handle)
135 !
136 iije=d%NIJE
137 iijb=d%NIJB
138 ikt=d%NKT
139 !
140 CALL second_mnh(ztime1)
141 !
142 IF(oblowsnow) THEN
143 ! See Vionnet (PhD, 2012) for a complete discussion around the value of the Schmidt number for blowing snow variables
144  zcsv= turbn%XCHF/prsnow
145 ELSE
146  zcsv= turbn%XCHF
147 ENDIF
148 !
149 DO jsv=1,ksv
150  !
151  IF (onomixlg .AND. jsv >= ksv_lgbeg .AND. jsv<= ksv_lgend) cycle
152  !
153  ! variance Sv2
154  !
155  IF (tles%LLES_CALL) THEN
156  ! approximation: diagnosed explicitely (without implicit term)
157  CALL gz_m_w_phy(d,psvm(:,:,jsv),pdzz,zwork1)
158  CALL mzf_phy(d,zflxz,zwork2)
159  CALL mzf_phy(d,pwm,zwork3)
160  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
161  zflxz(iijb:iije,1:ikt) = ppsi_sv(iijb:iije,1:ikt,jsv)*zwork1(iijb:iije,1:ikt)**2
162  zflxz(iijb:iije,1:ikt) = zcsv / zcsvd * plm(iijb:iije,1:ikt) * pleps(iijb:iije,1:ikt) * zwork2(iijb:iije,1:ikt)
163  zwork1(iijb:iije,1:ikt) = -2.*zcsvd*sqrt(ptkem(iijb:iije,1:ikt))*zflxz(iijb:iije,1:ikt)/pleps(iijb:iije,1:ikt)
164  zwork2(iijb:iije,1:ikt) = zwork3(iijb:iije,1:ikt)*zflxz(iijb:iije,1:ikt)
165  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
166  CALL les_mean_subgrid_phy(d,tles,zwork1, tles%X_LES_SUBGRID_DISS_Sv2(:,:,:,jsv) )
167  CALL les_mean_subgrid_phy(d,tles,zwork2, tles%X_LES_RES_W_SBG_Sv2(:,:,:,jsv) )
168  END IF
169  !
170  ! covariance ThvSv
171  !
172  IF (tles%LLES_CALL) THEN
173  ! approximation: diagnosed explicitely (without implicit term)
174  CALL etheta(d,cst,krr,krri,pthlm,prm,plocpexnm,patheta,psrcm,oocean,ocompute_src,za)
175  !
176  CALL gz_m_w_phy(d,pthlm,pdzz,zwork1)
177  CALL gz_m_w_phy(d,psvm(:,:,jsv),pdzz,zwork2)
178  !
179  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
180  zflxz(iijb:iije,1:ikt)= ( turbn%XCSHF * pphi3(iijb:iije,1:ikt) + zcsv * ppsi_sv(iijb:iije,1:ikt,jsv) ) &
181  * zwork1(iijb:iije,1:ikt) * zwork2(iijb:iije,1:ikt)
182  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
183  !
184  CALL mzf_phy(d,zflxz,zwork3)
185  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
186  zflxz(iijb:iije,1:ikt)= plm(iijb:iije,1:ikt) * pleps(iijb:iije,1:ikt) / (2.*zctsvd) * zwork3(iijb:iije,1:ikt)
187  zwork1(iijb:iije,1:ikt) = za(iijb:iije,1:ikt)*zflxz(iijb:iije,1:ikt)
188  zwork2(iijb:iije,1:ikt) = -cst%XG/pthvref(iijb:iije,1:ikt)/3.*za(iijb:iije,1:ikt)*zflxz(iijb:iije,1:ikt)
189  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
190  !
191  CALL les_mean_subgrid_phy(d,tles, zwork1, tles%X_LES_SUBGRID_SvThv(:,:,:,jsv) )
192  CALL les_mean_subgrid_phy(d,tles, zwork2, tles%X_LES_SUBGRID_SvPz(:,:,:,jsv), .true.)
193  !
194  IF (krr>=1) THEN
195  CALL emoist(d,cst,krr,krri,pthlm,prm,plocpexnm,pamoist,psrcm,oocean,za)
196  !
197  CALL gz_m_w_phy(d,prm(:,:,1),pdzz,zwork1)
198  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
199  zflxz(iijb:iije,1:ikt)= ( zcsv * ppsi3(iijb:iije,1:ikt) + zcsv * ppsi_sv(iijb:iije,1:ikt,jsv) ) &
200  * zwork1(iijb:iije,1:ikt) * zwork2(iijb:iije,1:ikt)
201  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
202  CALL mzf_phy(d,zflxz,zwork3)
203  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
204  zflxz(iijb:iije,1:ikt)= plm(iijb:iije,1:ikt) * pleps(iijb:iije,1:ikt) / (2.*zcqsvd) * zwork3(iijb:iije,1:ikt)
205  zwork1(iijb:iije,1:ikt) = za(iijb:iije,1:ikt)*zflxz(iijb:iije,1:ikt)
206  zwork2(iijb:iije,1:ikt) = -cst%XG/pthvref(iijb:iije,1:ikt)/3.*za(iijb:iije,1:ikt)*zflxz(iijb:iije,1:ikt)
207  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
208  CALL les_mean_subgrid_phy(d,tles, zwork1, tles%X_LES_SUBGRID_SvThv(:,:,:,jsv) , .true.)
209  CALL les_mean_subgrid_phy(d,tles, zwork2, tles%X_LES_SUBGRID_SvPz(:,:,:,jsv), .true.)
210  END IF
211  END IF
212  !
213 END DO ! end of scalar loop
214 !
215 CALL second_mnh(ztime2)
216 IF(tles%LLES_CALL) tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
217 !----------------------------------------------------------------------------
218 !
219 IF (lhook) CALL dr_hook('TURB_VER_SV_CORR',1,zhook_handle)
220 END SUBROUTINE turb_ver_sv_corr
