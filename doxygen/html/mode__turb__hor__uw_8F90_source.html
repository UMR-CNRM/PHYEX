104LOGICAL, INTENT(IN) :: OFLAT ! Logical for zero ororography
105TYPE(tfiledata), INTENT(INOUT) :: TPFILE ! Output file

107
108REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PK ! Turbulent diffusion doef.
109 ! PK = PLM * SQRT(PTKEM)
110REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDXX ! 1./PDXX
111REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDZZ ! 1./PDZZ
112REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PMZM_PRHODJ ! MZM(PRHODJ)
113REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PDXX, PDZZ, PDZX
114 ! Metric coefficients
115REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PRHODJ ! density * grid volume
116REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PTHVREF ! ref. state VPT
117!
118! Variables at t-1
119REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PUM,PWM,PTHLM
120REAL, DIMENSION(D%NIT,D%NJT,D%NKT,KRR), INTENT(IN) :: PRM
121REAL, DIMENSION(D%NIT,D%NJT,D%NKT,KSV), INTENT(IN) :: PSVM

123REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PTKEM ! TKE at time t- dt
124REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PLM ! Turb. mixing length
125!
126REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(INOUT) :: PRUS, PRWS
127REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(INOUT) :: PDP ! TKE production terms
128!
129!
130!
131!
132!* 0.2 declaration of local variables
133!
134REAL, DIMENSION(D%NIT,D%NJT,D%NKT) :: ZFLX,ZWORK
135 ! work arrays
136!
137INTEGER :: IKB,IKE,IKU, IIT, IJT, IKT
138 ! Index values for the Beginning and End
139 ! mass points of the domain
140INTEGER :: JSV,JI,JJ,JK ! scalar loop counter
141!
142REAL, DIMENSION(D%NIT,D%NJT,D%NKT) :: GX_W_UW_PWM

144REAL :: ZTIME1, ZTIME2
145TYPE(tfieldmetadata) :: TZFIELD
146! ---------------------------------------------------------------------------
147!
148!* 1. PRELIMINARY COMPUTATIONS
149! ------------------------
150!
151ikb = 1+jpvext
152ike = SIZE(pwm,3)-jpvext
153iku = SIZE(pwm,3)
154iit=d%NIT
155ijt=d%NJT
156ikt=d%NKT
157!
158!
159gx_w_uw_pwm = gx_w_uw(oflat,pwm,pdxx,pdzz,pdzx)
160!
161!
162!* 13. < U'W'>
163! -------
164!
165! residual part of < U'W'> depending on dw/dx
166!
167zflx(:,:,:) = &
168 - xcmfs * mxm(mzm(pk)) * gx_w_uw_pwm
169!! & to be tested
170!! - (2./3.) * XCMFB * MZM( ZVPTU * MXM( PLM / SQRT(PTKEM) * XG / PTHVREF ) )
171!
172!$acc kernels
173!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
174zflx(:,:,ike+1) = 0. ! rigid wall condition => no turbulent flux
175!
176! Nullify the flux at the ground level because it has been fully taken into
177! account in turb_ver and extrapolate the flux under the ground
178zflx(:,:,ikb) = 0.
179zflx(:,:,ikb-1)=2.*zflx(:,:,ikb)- zflx(:,:,ikb+1)
180!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
181!$acc end kernels
182!
183! stores <U W>
184IF ( tpfile%LOPENED .AND. turbn%LTURB_FLX ) THEN
185 tzfield = tfieldmetadata( &
186 cmnhname = 'UW_HFLX', &
187 cstdname = '', &
188 clongname = 'UW_HFLX', &
189 cunits = 'm2 s-2', &
190 cdir = 'XY', &
191 ccomment = 'X_Y_Z_UW_HFLX', &
192 ngrid = 6, &
193 ntype = typereal, &
194 ndims = 3, &
195 ltimedep = .true. )
196!$acc update self(ZFLX)
197 CALL io_field_write(tpfile,tzfield,zflx)
198END IF
199!
200!
201! compute the source for rho*U due to this residual flux ( the other part is
202! taken into account in TURB_VER)
203prus(:,:,:) = prus(:,:,:) - dzf( zflx* mxm( pmzm_prhodj ) / mxm( pdzz ) )
204!
205!computation of the source for rho*W due to this flux
206IF (.NOT. oflat) THEN
207 prws(:,:,:) = prws(:,:,:) &
208 -dxf( mzm( mxm(prhodj) * pinv_pdxx) * zflx) &
209 +dzm( prhodj * mxf( mzf( zflx*pdzx ) * pinv_pdxx ) / mzf(pdzz) )
210ELSE
211 prws(:,:,:) = prws(:,:,:) -dxf( mzm( mxm(prhodj) * pinv_pdxx) * zflx)
212END IF
213!
214IF (ksplt==1) THEN
215 !
216 !Contribution to the dynamic production of TKE:
217 !
218 zwork(:,:,:) =-mzf( mxf( &
219 zflx *( gz_u_uw(pum,pdzz) + gx_w_uw_pwm ) ) )
220 !
221 !
222 ! evaluate the dynamic production at w(IKB+1) in PDP(IKB)
223 zwork(:,:,ikb) = - mxf( &
224 zflx(:,:,ikb+1) * &
225 ( (pum(:,:,ikb+1)-pum(:,:,ikb)) / mxm(pdzz(:,:,ikb+1))&
226 + ( dxm( pwm(:,:,ikb+1) ) &
227 -mxm( (pwm(:,:,ikb+2)-pwm(:,:,ikb+1)) &
228 /(pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) &
229 +(pwm(:,:,ikb+1)-pwm(:,:,ikb)) &
230 /(pdzz(:,:,ikb+1)+pdzz(:,:,ikb)) &
231 ) &
232 * pdzx(:,:,ikb+1) &
233 ) / (0.5*(pdxx(:,:,ikb+1)+pdxx(:,:,ikb))) &
234 ) )
235 !
236 ! dynamic production computation
237 !$acc kernels
238 !$mnh_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
239 pdp(:,:,:) = pdp(:,:,:) + zwork(:,:,:)
240 !$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
241 !$acc end kernels
242 !
243END IF
244!
245! Storage in the LES configuration (addition to TURB_VER computation)
246!
247IF (tles%LLES_CALL .AND. ksplt==1) THEN
248 CALL second_mnh(ztime1)
249 CALL les_mean_subgrid( mzf(mxf(zflx)), tles%X_LES_SUBGRID_WU , .true. )
250 CALL les_mean_subgrid( mzf(mxf(gz_u_uw(pum,pdzz)*zflx)), tles%X_LES_RES_ddxa_U_SBG_UaU , .true.)
251 CALL les_mean_subgrid( mzf(mxf(gx_w_uw_pwm*zflx)), tles%X_LES_RES_ddxa_W_SBG_UaW , .true.)
252 CALL les_mean_subgrid( mxf(gx_m_u(oflat,pthlm,pdxx,pdzz,pdzx)*mzf(zflx)),&
253 tles%X_LES_RES_ddxa_Thl_SBG_UaW , .true.)
254 IF (krr>=1) THEN
255 CALL les_mean_subgrid( mxf(gx_m_u(oflat,prm(:,:,:,1),pdxx,pdzz,pdzx)*mzf(zflx)), &
256 tles%X_LES_RES_ddxa_Rt_SBG_UaW , .true.)
257 END IF
258 DO jsv=1,ksv
259 CALL les_mean_subgrid( mxf(gx_m_u(oflat,psvm(:,:,:,jsv),pdxx,pdzz,pdzx)*mzf(zflx)), &
260 tles%X_LES_RES_ddxa_Sv_SBG_UaW(:,:,:,jsv) , .true.)
261 END DO
262 CALL second_mnh(ztime2)
263 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
264END IF
265
266!
267END SUBROUTINE turb_hor_uw
