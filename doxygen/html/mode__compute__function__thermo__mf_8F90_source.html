54USE modd_dimphyex, ONLY: dimphyex_t
55USE modd_cst, ONLY: cst_t
56USE yomhook , ONLY : lhook, dr_hook, jphook
57!
58IMPLICIT NONE

60!
61!* 0.1 declarations of arguments
62!
63TYPE(dimphyex_t), INTENT(IN) :: D
64TYPE(cst_t), INTENT(IN) :: CST
65LOGICAL, INTENT(IN) :: OSTATNW ! cloud scheme inclues convect. covar. contrib
66INTEGER, INTENT(IN) :: KRR ! number of moist var.
67INTEGER, INTENT(IN) :: KRRL ! number of liquid water var.
68INTEGER, INTENT(IN) :: KRRI ! number of ice water var.
69
70REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTH ! theta
71REAL, DIMENSION(D%NIJT,D%NKT,KRR), INTENT(IN) :: PR ! water species
72REAL, DIMENSION(D%NIJT,D%NKT) , INTENT(IN) :: PPABS,PEXN ! pressure, Exner funct.
73REAL, DIMENSION(D%NIJT,D%NKT) , INTENT(IN) :: PFRAC_ICE ! ice fraction
74
75REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PT ! temperature
76
77REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT) :: PAMOIST,PATHETA
78!
79!-------------------------------------------------------------------------------
80!
81!* 0.2 Declarations of local variables
82!
83REAL :: ZEPS ! XMV / XMD
84REAL, DIMENSION(D%NIJT,D%NKT) :: &
85 ZCP, & ! Cp
86 ZE, & ! Saturation mixing ratio
87 ZDEDT, & ! Saturation mixing ratio derivative
88 ZAMOIST_W, & ! Coefficients for s = f (Thetal,Rnp)
89 ZATHETA_W, & !
90 ZAMOIST_I, & !
91 ZATHETA_I, & !
92 ZLVOCP,ZLSOCP
93
94INTEGER :: JRR, JIJ, JK
95INTEGER :: IIJB,IIJE ! physical horizontal domain indices
96INTEGER :: IKTB,IKTE
97REAL(KIND=jphook) :: zhook_handle
98!
99!-------------------------------------------------------------------------------
100!
101IF (lhook) CALL dr_hook('COMPUTE_FUNCTION_THERMO_MF',0,zhook_handle)
102!
103iije=d%NIJE
104iijb=d%NIJB
105iktb=d%NKTB
106ikte=d%NKTE
107!
108 zeps = cst%XMV / cst%XMD
109
110!
111!* Cph
112!
113zcp=cst%XCPD
114
115IF (krr > 0) THEN
116 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
117 zcp(iijb:iije,iktb:ikte) = zcp(iijb:iije,iktb:ikte) + cst%XCPV * pr(iijb:iije,iktb:ikte,1)
118 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
119ENDIF
120
121DO jrr = 2,1+krrl ! loop on the liquid components
122 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
123 zcp(iijb:iije,iktb:ikte) = zcp(iijb:iije,iktb:ikte) + cst%XCL * pr(iijb:iije,iktb:ikte,jrr)
124 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
125END DO
126
127DO jrr = 2+krrl,1+krrl+krri ! loop on the solid components
128 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
129 zcp(iijb:iije,iktb:ikte) = zcp(iijb:iije,iktb:ikte) + cst%XCI * pr(iijb:iije,iktb:ikte,jrr)
130 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
131
132END DO
133
134!* Temperature
135!
136!$mnh_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
137pt(iijb:iije,iktb:ikte) = pth(iijb:iije,iktb:ikte) * pexn(iijb:iije,iktb:ikte)
138!$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
139!
140!
141!! Liquid water
142!
143IF ( krrl >= 1 ) THEN
144 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
145 !
146 !* Lv/Cph
147 !
148 zlvocp(iijb:iije,iktb:ikte) = (cst%XLVTT + (cst%XCPV-cst%XCL) * (pt(iijb:iije,iktb:ikte)-cst%XTT) ) / &
149 & zcp(iijb:iije,iktb:ikte)
150 !
151 !* Saturation vapor pressure with respect to water
152 !
153 ze(iijb:iije,iktb:ikte) = exp(cst%XALPW - cst%XBETAW/pt(iijb:iije,iktb:ikte) - &
154 &cst%XGAMW*log( pt(iijb:iije,iktb:ikte) ) )
155 !
156 !* Saturation mixing ratio with respect to water
157 !
158 ze(iijb:iije,iktb:ikte) = ze(iijb:iije,iktb:ikte) * zeps / &
159 & ( ppabs(iijb:iije,iktb:ikte) - ze(iijb:iije,iktb:ikte) )
160 !
161 !* Compute the saturation mixing ratio derivative (rvs')
162 !
163 zdedt(iijb:iije,iktb:ikte) = (cst%XBETAW/pt(iijb:iije,iktb:ikte) - cst%XGAMW) / pt(iijb:iije,iktb:ikte)&
164 * ze(iijb:iije,iktb:ikte) * ( 1. + ze(iijb:iije,iktb:ikte) / zeps )
165 !
166 !* Compute Amoist and Atheta
167 !
168 IF (ostatnw) THEN
169 zamoist_w(iijb:iije,iktb:ikte)= 1.0/( 1.0 + zdedt(iijb:iije,iktb:ikte) * zlvocp(iijb:iije,iktb:ikte))
170 zatheta_w(iijb:iije,iktb:ikte)= zamoist_w(iijb:iije,iktb:ikte) * pexn(iijb:iije,iktb:ikte) &
171 * zdedt(iijb:iije,iktb:ikte)
172 ELSE
173 zamoist_w(iijb:iije,iktb:ikte)= 0.5/( 1.0 + zdedt(iijb:iije,iktb:ikte) * zlvocp(iijb:iije,iktb:ikte) )
174 zatheta_w(iijb:iije,iktb:ikte)= zamoist_w(iijb:iije,iktb:ikte) * pexn(iijb:iije,iktb:ikte) * &
175 ( ( ze(iijb:iije,iktb:ikte) - pr(iijb:iije,iktb:ikte,1) ) * zlvocp(iijb:iije,iktb:ikte) / &
176 ( 1. + zdedt(iijb:iije,iktb:ikte) * zlvocp(iijb:iije,iktb:ikte) ) * &
177 ( &
178 ze(iijb:iije,iktb:ikte) * (1. + ze(iijb:iije,iktb:ikte)/zeps) &
179 * ( -2.*cst%XBETAW/pt(iijb:iije,iktb:ikte) + cst%XGAMW ) / pt(iijb:iije,iktb:ikte)**2&
180 +zdedt(iijb:iije,iktb:ikte) * (1. + 2. * ze(iijb:iije,iktb:ikte)/zeps) &
181 * ( cst%XBETAW/pt(iijb:iije,iktb:ikte) - cst%XGAMW ) / pt(iijb:iije,iktb:ikte) &
182 ) &
183 - zdedt(iijb:iije,iktb:ikte) &
184 )
185 END IF
186 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
187 !
188 !! Solid water
189 !
190 IF ( krri >= 1 ) THEN
191 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
192 !
193 !* Ls/Cph
194 !
195 zlsocp(iijb:iije,iktb:ikte) = (cst%XLSTT + (cst%XCPV-cst%XCI) * (pt(iijb:iije,iktb:ikte)-cst%XTT) ) / &
196 & zcp(iijb:iije,iktb:ikte)
197 !
198 !* Saturation vapor pressure with respect to ice
199 !
200 ze(iijb:iije,iktb:ikte) = exp(cst%XALPI - cst%XBETAI/pt(iijb:iije,iktb:ikte) - &
201 &cst%XGAMI*log( pt(iijb:iije,iktb:ikte) ) )
202 !
203 !* Saturation mixing ratio with respect to ice
204 !
205 ze(iijb:iije,iktb:ikte) = ze(iijb:iije,iktb:ikte) * zeps / &
206 & ( ppabs(iijb:iije,iktb:ikte) - ze(iijb:iije,iktb:ikte) )
207 !
208 !* Compute the saturation mixing ratio derivative (rvs')
209 !
210 zdedt(iijb:iije,iktb:ikte) = (cst%XBETAI/pt(iijb:iije,iktb:ikte)-cst%XGAMI) /pt(iijb:iije,iktb:ikte)&
211 * ze(iijb:iije,iktb:ikte) * ( 1. + ze(iijb:iije,iktb:ikte) / zeps )
212 !
213 !* Compute Amoist and Atheta
214 !
215 IF (ostatnw) THEN
216 zamoist_i(iijb:iije,iktb:ikte)= 1.0/( 1.0 + zdedt(iijb:iije,iktb:ikte) *zlvocp(iijb:iije,iktb:ikte))
217 zatheta_i(iijb:iije,iktb:ikte)= zamoist_i(iijb:iije,iktb:ikte) * pexn(iijb:iije,iktb:ikte) &
218 * zdedt(iijb:iije,iktb:ikte)
219 ELSE
220 zamoist_i(iijb:iije,iktb:ikte)= 0.5/(1.0 + zdedt(iijb:iije,iktb:ikte) * zlsocp(iijb:iije,iktb:ikte))
221 zatheta_i(iijb:iije,iktb:ikte)= zamoist_i(iijb:iije,iktb:ikte) * pexn(iijb:iije,iktb:ikte) * &
222 ( ( ze(iijb:iije,iktb:ikte) - pr(iijb:iije,iktb:ikte,1) ) * zlsocp(iijb:iije,iktb:ikte) / &
223 ( 1. + zdedt(iijb:iije,iktb:ikte) * zlsocp(iijb:iije,iktb:ikte) ) * &
224 ( &
225 ze(iijb:iije,iktb:ikte) * (1. + ze(iijb:iije,iktb:ikte)/zeps) &
226 * ( -2.*cst%XBETAI/pt(iijb:iije,iktb:ikte) + cst%XGAMI ) / pt(iijb:iije,iktb:ikte)**2 &
227 +zdedt(iijb:iije,iktb:ikte) * (1. + 2. * ze(iijb:iije,iktb:ikte)/zeps) &
228 * ( cst%XBETAI/pt(iijb:iije,iktb:ikte) - cst%XGAMI ) / pt(iijb:iije,iktb:ikte) &
229 ) &
230 - zdedt(iijb:iije,iktb:ikte) &
231 )
232 END IF
233 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
234
235 ELSE
236 zamoist_i(iijb:iije,iktb:ikte)=0.
237 zatheta_i(iijb:iije,iktb:ikte)=0.
238 ENDIF
239
240 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
241 pamoist(iijb:iije,iktb:ikte) = (1.0-pfrac_ice(iijb:iije,iktb:ikte))*zamoist_w(iijb:iije,iktb:ikte) &
242 +pfrac_ice(iijb:iije,iktb:ikte) *zamoist_i(iijb:iije,iktb:ikte)
243 patheta(iijb:iije,iktb:ikte) = (1.0-pfrac_ice(iijb:iije,iktb:ikte))*zatheta_w(iijb:iije,iktb:ikte) &
244 +pfrac_ice(iijb:iije,iktb:ikte) *zatheta_i(iijb:iije,iktb:ikte)
245 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=IKTB:IKTE)
246 !
247ELSE
248 pamoist(iijb:iije,iktb:ikte) = 0.
249 patheta(iijb:iije,iktb:ikte) = 0.
250ENDIF
251IF (lhook) CALL dr_hook('COMPUTE_FUNCTION_THERMO_MF',1,zhook_handle)
252END SUBROUTINE compute_function_thermo_mf
