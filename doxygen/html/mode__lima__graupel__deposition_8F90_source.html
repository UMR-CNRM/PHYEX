67WHERE ( prgt(:)>xrtmin(6) .AND. pcgt(:)>xctmin(6) .AND. ldcompute(:) )
68 p_rg_depg(:) = pssi(:) / pai(:) * pcgt(:) * &
69 ( x0depg*plbdg(:)**xex0depg + x1depg*pcj(:)*plbdg(:)**xex1depg )
