14 !!** PURPOSE
15 !! -------
16 !! Computes the nucleation
17 !!
18 !! AUTHOR
19 !! ------
20 !! S. Riette from the splitting of rain_ice source code (nov. 2014)
21 !!
22 !! MODIFICATIONS
23 !! -------------
24 !!

26 !
27 !
28 !* 0. DECLARATIONS
29 ! ------------
30 !
31 USE modd_cst, ONLY: cst_t
32 USE yomhook , ONLY : lhook, dr_hook, jphook
33 USE modd_param_lima_cold, ONLY : xalpha1, xbeta1, xalpha2, xbeta2, xnu10, xnu20, xmnu0
34 USE modd_param_lima, ONLY: lfeedbackt, xrtmin
35 !
36 IMPLICIT NONE
37 !
38 !* 0.1 Declarations of dummy arguments :
39 !
40 TYPE(cst_t), INTENT(IN) :: CST
41 INTEGER, INTENT(IN) :: KSIZE
42 REAL, DIMENSION(KSIZE), INTENT(IN) :: PTHT ! Theta at t
43 REAL, DIMENSION(KSIZE), INTENT(IN) :: PPABST ! absolute pressure at t
44 REAL, DIMENSION(KSIZE), INTENT(IN) :: PRHODREF! Reference density
45 REAL, DIMENSION(KSIZE), INTENT(IN) :: PEXN ! Exner function
46 REAL, DIMENSION(KSIZE), INTENT(IN) :: PLSFACT
47 REAL, DIMENSION(KSIZE), INTENT(IN) :: PT ! Temperature at time t
48 REAL, DIMENSION(KSIZE), INTENT(IN) :: PRVT ! Water vapor m.r. at t
49 REAL, DIMENSION(KSIZE), INTENT(INOUT) :: PCIT ! Pristine ice n.c. at t
50 REAL, DIMENSION(KSIZE), INTENT(OUT) :: PRVHENI_MR ! Mixing ratio change due to the heterogeneous nucleation
51 !
52 !* 0.2 declaration of local variables
53 !
54 REAL, DIMENSION(KSIZE) :: ZW ! work array
55 REAL(KIND=jphook) :: zhook_handle
56 LOGICAL, DIMENSION(KSIZE) :: GNEGT ! Test where to compute the HEN process
57 REAL, DIMENSION(KSIZE) :: ZZW, & ! Work array
58  ZUSW, & ! Undersaturation over water
59  ZSSI ! Supersaturation over ice
60 INTEGER :: JI
61 !-------------------------------------------------------------------------------
62 !
63 IF (lhook) CALL dr_hook('LIMA_ICE4_NUCLEATION', 0, zhook_handle)!
64 !
65 !$mnh_expand_where(JI=1:KSIZE)
66 gnegt(:)=pt(:)<cst%XTT .AND. prvt(:)>xrtmin(1)
67 !$mnh_end_expand_where(JI=1:KSIZE)
68 
69 zusw(:)=0.
70 zzw(:)=0.
71 !$mnh_expand_where(JI=1:KSIZE)
72 WHERE(gnegt(:))
73  zzw(:)=alog(pt(:))
74  zusw(:)=exp(cst%XALPW - cst%XBETAW/pt(:) - cst%XGAMW*zzw(:)) ! es_w
75  zzw(:)=exp(cst%XALPI - cst%XBETAI/pt(:) - cst%XGAMI*zzw(:)) ! es_i
76 END WHERE
77 !$mnh_end_expand_where(JI=1:KSIZE)
78 
79 zssi(:)=0.
80 !$mnh_expand_where(JI=1:KSIZE)
81 WHERE(gnegt(:))
82  zzw(:)=min(ppabst(:)/2., zzw(:)) ! safety limitation
83  zssi(:)=prvt(:)*(ppabst(:)-zzw(:)) / (cst%XEPSILO*zzw(:)) - 1.0
84  ! Supersaturation over ice
85  zusw(:)=min(ppabst(:)/2., zusw(:)) ! safety limitation
86  zusw(:)=(zusw(:)/zzw(:))*((ppabst(:)-zzw(:))/(ppabst(:)-zusw(:))) - 1.0
87  ! Supersaturation of saturated water vapor over ice
88  !
89  !* 3.1 compute the heterogeneous nucleation source RVHENI
90  !
91  !* 3.1.1 compute the cloud ice concentration
92  !
93  zssi(:)=min(zssi(:), zusw(:)) ! limitation of SSi according to SSw=0
94 END WHERE
95 !$mnh_end_expand_where(JI=1:KSIZE)
96 
97 zzw(:)=0.
98 DO ji=1,ksize
99  IF(gnegt(ji)) THEN
100  IF(pt(ji)<cst%XTT-5.0 .AND. zssi(ji)>0.0) THEN
101  zzw(ji)=xnu20*exp(xalpha2*zssi(ji)-xbeta2)
102  ELSEIF(pt(ji)<=cst%XTT-2.0 .AND. pt(ji)>=cst%XTT-5.0 .AND. zssi(ji)>0.0) THEN
103  zzw(ji)=max(xnu20*exp(-xbeta2 ), &
104  xnu10*exp(-xbeta1*(pt(ji)-cst%XTT))*(zssi(ji)/zusw(ji))**xalpha1)
105  ENDIF
106  ENDIF
107 ENDDO
108 !$mnh_expand_where(JI=1:KSIZE)
109 WHERE(gnegt(:))
110  zzw(:)=zzw(:)-pcit(:)
111  zzw(:)=min(zzw(:), 50.e3) ! limitation provisoire a 50 l^-1
112 END WHERE
113 !$mnh_end_expand_where(JI=1:KSIZE)
114 
115 prvheni_mr(:)=0.
116 !$mnh_expand_where(JI=1:KSIZE)
117 WHERE(gnegt(:))
118  !
119  !* 3.1.2 update the r_i and r_v mixing ratios
120  !
121  prvheni_mr(:)=max(zzw(:), 0.0)*xmnu0/prhodref(:)
122  prvheni_mr(:)=min(prvt(:), prvheni_mr(:))
123 END WHERE
124 !$mnh_end_expand_where(JI=1:KSIZE)
125 !Limitation due to 0 crossing of temperature
126 IF(lfeedbackt) THEN
127  zw(:)=0.
128  !$mnh_expand_where(JI=1:KSIZE)
129  WHERE(gnegt(:))
130  zw(:)=min(prvheni_mr(:), &
131  max(0., (cst%XTT/pexn(:)-ptht(:))/plsfact(:))) / &
132  max(prvheni_mr(:), 1.e-20)
133  END WHERE
134  prvheni_mr(:)=prvheni_mr(:)*zw(:)
135  zzw(:)=zzw(:)*zw(:)
136  !$mnh_end_expand_where(JI=1:KSIZE)
137 ENDIF
138 !$mnh_expand_where(JI=1:KSIZE)
139 WHERE(gnegt(:))
140  pcit(:)=max(zzw(:)+pcit(:), pcit(:))
141 END WHERE
142 !$mnh_end_expand_where(JI=1:KSIZE)
143 !
144 IF (lhook) CALL dr_hook('LIMA_ICE4_NUCLEATION', 1, zhook_handle)
145 END SUBROUTINE lima_ice4_nucleation
