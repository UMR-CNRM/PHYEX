340SUBROUTINE param_lima_warm_allocate(HNAME, KDIM1, KDIM2, KDIM3, KDIM4)
341USE yomhook, ONLY:lhook, dr_hook, jphook
342 IMPLICIT NONE
343 CHARACTER(LEN=*), INTENT(IN) :: HNAME
344 INTEGER, INTENT(IN) :: KDIM1
345 INTEGER, OPTIONAL, INTENT(IN):: KDIM2
346 INTEGER, OPTIONAL, INTENT(IN):: KDIM3
347 INTEGER, OPTIONAL, INTENT(IN):: KDIM4
348 REAL(KIND=jphook) :: zhook_handle
349
350 IF (lhook) CALL dr_hook('PARAM_LIMA_WARM_ALLOCATE', 0, zhook_handle)
351 SELECT CASE(trim(hname))
352 CASE('XHYPF12')
353 ALLOCATE(param_lima_warm%XHYPF12(kdim1, kdim2))
354 xhypf12 => param_lima_warm%XHYPF12
355 CASE('XHYPF32')
356 ALLOCATE(param_lima_warm%XHYPF32(kdim1, kdim2))
357 xhypf32 => param_lima_warm%XHYPF32
358 CASE('XAHENG')
359 ALLOCATE(param_lima_warm%XAHENG(kdim1))
360 xaheng => param_lima_warm%XAHENG
361 CASE('XAHENG2')
362 ALLOCATE(param_lima_warm%XAHENG2(kdim1))
363 xaheng2 => param_lima_warm%XAHENG2
364 CASE('XAHENG3')
365 ALLOCATE(param_lima_warm%XAHENG3(kdim1))
366 xaheng3 => param_lima_warm%XAHENG3
367 CASE('XPSI1')
368 ALLOCATE(param_lima_warm%XPSI1(kdim1))
369 xpsi1 => param_lima_warm%XPSI1
370 CASE('XPSI3')
371 ALLOCATE(param_lima_warm%XPSI3(kdim1))
372 xpsi3 => param_lima_warm%XPSI3
373 CASE('XAHENF')
374 ALLOCATE(param_lima_warm%XAHENF(kdim1))
375 xahenf => param_lima_warm%XAHENF
376 CASE('XAHENY')
377 ALLOCATE(param_lima_warm%XAHENY(kdim1))
378 xaheny => param_lima_warm%XAHENY
379 CASE('XCONCC_INI')
380 ALLOCATE(param_lima_warm%XCONCC_INI(kdim1,kdim2,kdim3,kdim4))
381 xconcc_ini => param_lima_warm%XCONCC_INI
382 END SELECT
383IF (lhook) CALL dr_hook('PARAM_LIMA_WARM_ALLOCATE', 1, zhook_handle)
384END SUBROUTINE param_lima_warm_allocate
