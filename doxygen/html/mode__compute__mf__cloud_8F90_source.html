152 !Statistical scheme using the bi-gaussian PDF
153 CALL compute_mf_cloud_bigaus(d, cst, parammf, icep, krr, &
154 &prt_up, prv_up, prc_up, pri_up, pth_up, pfrac_up, &
155 &prtm, pthm, prm, &
156 &prhodref, pexnm, ppabsm, &
157 &prc_mf, pri_mf, pcf_mf, psigmf, &
158 &phlc_hrc, phlc_hcf, phli_hri, phli_hcf, pweight_mf_cloud(:,:))
159 !
160ELSEIF (parammf%CMF_CLOUD == 'NONE') THEN
161 ! No CONVECTIVE CLOUD SCHEME
162 ! Nothing to do: PRC_MF, PRI_MF, PCF_MF, PSIGMF are already filled with zero
163ELSE
164 CALL print_msg(nverb_fatal, 'GEN', 'COMPUTE_MF_CLOUD', &
165 'Shallow convection cloud scheme not valid: PARAMMF%CMF_CLOUD='//parammf%CMF_CLOUD)
166ENDIF
167
168IF (lhook) CALL dr_hook('COMPUTE_MF_CLOUD',1,zhook_handle)
