22 !!**** * - Build up a look-up table containing the scaled fall speed
23 !! difference between size distributed particles of specy X and Z
24 !!
25 !!
26 !! PURPOSE
27 !! -------
28 !! The purpose of this routine is to integrate numerically the scaled fall
29 !! speed difference between specy X and specy Z for use in collection
30 !! kernels. A first integral of the form
31 !!
32 !! infty
33 !! / /
34 !! |{| }
35 !! |{| E_xz (Dx+Dz)^2 |cxDx^dx-czDz^dz| Dz^bz g(Dz) dDz} g(Dx) dDx
36 !! |{| }
37 !! / /
38 !! 0
39 !!
40 !! is evaluated and normalised by a second integral of the form
41 !!
42 !! infty
43 !! / /
44 !! |{| }
45 !! |{| (Dx+Dz)^2 Dz^bz g(Dz) dDz} g(Dx) dDx
46 !! |{| }
47 !! / /
48 !! 0
49 !!
50 !! where E_xz is a collection efficiency, g(D) is the generalized Gamma
51 !! distribution law. The 'infty' diameter is defined according to the
52 !! current value of the Lbda that is D_x=PDINFTY/Lbda_x or
53 !! D_z=PINFTY/Lbda_z.
54 !! The result is stored in a two-dimensional array.
55 !!
56 !!** METHOD
57 !! ------
58 !! The free parameters of the size distribution function of specy X and Z
59 !! (slope parameter LAMBDA) are discretized with a geometrical rate in a
60 !! specific range
61 !! LAMBDA = exp( (Log(LAMBDA_max) - Log(LAMBDA_min))/N_interval )
62 !! The two above integrals are performed using the trapezoidal scheme and
63 !! the [0,infty] interval is discretized over KND values of D_x or D_z.
64 !!
65 !! EXTERNAL
66 !! --------
67 !! MODI_GENERAL_GAMMA: Generalized gamma distribution law
68 !!
69 !! IMPLICIT ARGUMENTS
70 !! ------------------
71 !! None
72 !!
73 !! REFERENCE
74 !! ---------
75 !! B.S. Ferrier , 1994 : A Double-Moment Multiple-Phase Four-Class
76 !! Bulk Ice Scheme,JAS,51,249-280.
77 !!
78 !! AUTHOR
79 !! ------
80 !! J.-P. Pinty * Laboratoire d'Aerologie *

167 !* 1 COMPUTE THE SCALED VELOCITZ DIFFERENCE IN THE MASS
168 !* COLLECTION KERNEL,
169 ! -------------------------------------------------
170 !
171 !
172 !
173 !* 1.1 Compute the growth rate of the slope factors LAMBDA
174 !
175 REAL(KIND=jphook) :: zhook_handle
176 IF (lhook) CALL dr_hook('RZCOLX',0,zhook_handle)
177 zdlbdax = exp( log(plbdaxmax/plbdaxmin)/real(SIZE(przcolx(:,:),1)-1) )
178 zdlbdaz = exp( log(plbdazmax/plbdazmin)/real(SIZE(przcolx(:,:),2)-1) )
179 !
180 !* 1.2 Scan the slope factors LAMBDAX and LAMBDAZ
181 !
182 DO jlbdax = 1,SIZE(przcolx(:,:),1)
183  zlbdax = plbdaxmin * zdlbdax ** (jlbdax-1)
184  DO jlbdaz = 1,SIZE(przcolx(:,:),2)
185  zlbdaz = plbdazmin * zdlbdaz ** (jlbdaz-1)
186 !
187 !* 1.3 Initialize the collection integrals
188 !
189  zscalxz = 0.0
190  zcollxz = 0.0
191 !
192 !* 1.4 Compute the diameter steps
193 !
194  zddx = pdinfty / (real(knd) * zlbdax)
195  zddz = pdinfty / (real(knd) * zlbdaz)
196 !
197 !* 1.5 Scan over the diameters DX and DZ
198 !
199  DO jdx = 1,knd-1
200  zdx = zddx * real(jdx)
201 !
202  zscalz = 0.0
203  zcollz = 0.0
204  DO jdz = 1,knd-1
205  zdz = zddz * real(jdz)
206 !
207 !* 1.6 Compute the normalization factor by integration over the
208 ! dimensional spectrum of specy Z
209 !
210  zfunc = (zdx+zdz)**2 * zdz**pexmassz &
211  * general_gamma(palphaz,pnuz,zlbdaz,zdz)
212  zscalz = zscalz + zfunc
213 !
214 !* 1.7 Compute the scaled fall speed difference by integration over
215 ! the dimensional spectrum of specy Z
216 !
217 #ifdef REPRO48
218  zcollz = zcollz + zfunc &
219  * pexz * abs(pfallx*zdx**pexfallx-pfallz*zdz**pexfallz)
220 #else
221  zcollz = zcollz + zfunc * pexz * abs( pfallx*zdx**pexfallx * exp(-(zdx*pfallexpx)**palphax) &
222  - pfallz*zdz**pexfallz * exp(-(zdz*pfallexpz)**palphaz))
223 #endif
224  END DO
225 !
226 !* 1.8 Compute the normalization factor by integration over the
227 ! dimensional spectrum of specy X
228 !
229  zfunc = general_gamma(palphax,pnux,zlbdax,zdx)
230  zscalxz = zscalxz + zscalz * zfunc
231 !
232 !* 1.9 Compute the scaled fall speed difference by integration over
233 ! the dimensional spectrum of specy X
234 !
235  zcollxz = zcollxz + zcollz * zfunc
236  END DO
237 !
238 !* 1.10 Scale the fall speed difference
239 !
240  przcolx(jlbdax,jlbdaz) = zcollxz / zscalxz
241  END DO
242 END DO

244 IF (lhook) CALL dr_hook('RZCOLX',1,zhook_handle)
