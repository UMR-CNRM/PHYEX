11!########################################################################x
12!
13!! PURPOSE
14!! -------
15!! The purpose of this routine is to compute the total number
16!! below-cloud scavenging rate.
17!!
18!!
19!!** METHOD
20!! ------
21!! We assume a generalized gamma distribution law for the raindrop.
22!! The aerosols particles distribution follows a log-normal law.
23!! First, we have to compute the Collision Efficiency, which takes
24!! account of the three most important wet removal mechanism :
25!! Brownian diffusion, interception and inertial impaction.
26!! It is a function of several number (like Reynolds, Schmidt
27!! or Stokes number for instance). Consequently,
28!! we need first to calculate these numbers.
29!!
30!! Then the scavenging coefficient is deduced from the integration
31!! of the droplet size distribution, the falling velocity of
32!! raindrop and aerosol, their diameter, and the collision
33!! (or collection) efficiency, over the spectrum of droplet
34!! diameters.
35!!
36!! The total scavenging rate of aerosol is computed from the
37!! integration, over all the spectrum of particles aerosols
38!! diameters, of the scavenging coefficient.
39!!
40!!
41!! EXTERNAL
42!! --------
43!! Subroutine SCAV_MASS_SEDIMENTATION
44!!
45!! Function COLL_EFFIC : computes the collision efficiency
46!!
47!! Function CONTJV |
48!! Function GAUHER |
49!! Function GAULAG |-> in lima_functions.f90
50!! Function GAMMLN |
51!!
52!!
53!! REFERENCES
54!! ----------

56!! Andronache
57!!

220 ZVOLDR_INV ! INV of Mean volumic Raindrop diameter [m]
221REAL :: ZDENS_RATIO_SQRT

233!------------------------------------------------------------------------------
234IF (lhook) CALL dr_hook('LIMA_PRECIP_SCAVENGING', 0, zhook_handle)
235isv_lima_nr = tnsv%NSV_LIMA_NR - tnsv%NSV_LIMA_BEG + 1
236isv_lima_scavmass = tnsv%NSV_LIMA_SCAVMASS - tnsv%NSV_LIMA_BEG + 1
237
238IF ( buconf%LBUDGET_SV ) then
239 DO il = 1, nmod_ccn
240 idx = tnsv%NSV_LIMA_CCN_FREE - 1 + il
241 CALL budget_store_init_phy(d, tbudgets(nbudget_sv1 - 1 + idx), 'SCAV', prsvs( :, :, idx) )
242 END DO
243 DO il = 1, nmod_ifn
244 idx = tnsv%NSV_LIMA_IFN_FREE - 1 + il
245 CALL budget_store_init_phy(d, tbudgets(nbudget_sv1 - 1 + idx), 'SCAV', prsvs( :, :, idx) )
246 END DO
247 IF ( laero_mass ) then
248 CALL budget_store_init_phy(d, tbudgets(nbudget_sv1 - 1 + tnsv%NSV_LIMA_SCAVMASS), 'SCAV', &
249 prsvs( :, :, tnsv%NSV_LIMA_SCAVMASS) )
250 END IF
251END IF
252!
253!* 1. PRELIMINARY COMPUTATIONS
254! ------------------------
255!
256!
257! ZCRT
258zcrt(:,:)=psvt(:,:,isv_lima_nr)
259!
260! Rain mask
261grain(:,:) = .false.
262grain(d%NIJB:d%NIJE,d%NKB:d%NKE) = (prrt(d%NIJB:d%NIJE,d%NKB:d%NKE)>xrtmin(3) &
263 .AND. zcrt(d%NIJB:d%NIJE,d%NKB:d%NKE)>xctmin(3) )
264!
265! Initialize the total mass scavenging rate if LAERO_MASS=T
266IF (laero_mass) ztot_mass_rate(:,:) = 0.
267!
268! Quadrature method: compute absissae and weights
269CALL gauher(zabscissp,zweightp,ndiamp)
270znum = xnur-1.0e0
271CALL gaulag(zabscissr,zweightr,ndiamr,znum)
272!
273!
274!------------------------------------------------------------------------------
275!
276!
277!* 2. NUMERICAL OPTIMIZATION
278! ----------------------
279!
280!
281! Optimization : compute in advance parameters depending on rain particles and
282! environment conditions only, to avoid multiple identical computations in loops
283!
284!
285zshape_factor = gamma_x0d(xnur+3./xalphar)/gamma_x0d(xnur)
286!
287WHERE ( grain(:,:) )
288 !
289 zt_3d(:,:) = ptht(:,:) * ( ppabst(:,:)/cst%XP00 )**(cst%XRD/cst%XCPD)
290 zconcr_3d(:,:) = zcrt(:,:) * prhodref(:,:) ![/m3]
291 ! Sutherland law for viscosity of air
292 zvisca_3d(:,:) = xmua0*(zt_3d(:,:)/xtref)**1.5*(xtref+xt_suth_a) &
293 /(xt_suth_a+zt_3d(:,:))
294 ! Air mean free path
295 zmfpa_3d(:,:) = xmfpa0*(cst%XP00*zt_3d(:,:))/(ppabst(:,:)*xt0scav)
296 ! Viscosity ratio
297 zvisc_ratio_3d(:,:) = zvisca_3d(:,:)/xviscw !!!!! invers√© par rapport √† orig. !
298 ! Rain drops parameters
299 zlambdar_3d(:,:) = ( ((cst%XPI/6.)*zshape_factor*cst%XRHOLW*zconcr_3d(:,:)) &
300 /(prhodref(:,:)*prrt(:,:)) )**(1./3.) ![/m]
301 zfactor_3d(:,:) = cst%XPI*0.25*zconcr_3d(:,:)*xcr*(xrho00/prhodref(:,:))**(0.4)
302 !
303END WHERE
304!
305DO j2=1,ndiamr
306 WHERE ( grain(:,:) )
307 ! exchange of variables: [m]
308 zvoldr_3d(:,:,j2) = zabscissr(j2)**(1./xalphar)/zlambdar_3d(:,:)
309 zvoldr_3d_inv(:,:,j2) = 1./zvoldr_3d(:,:,j2)
310 zvoldr_3d_pow(:,:,j2) = zvoldr_3d(:,:,j2)**(2.+xdr)
311 ! Raindrop Falling VELocity [m/s]
312 zfvelr_3d(:,:,j2) = xcr*(zvoldr_3d(:,:,j2)**xdr)*(xrho00/prhodref(:,:))**(0.4)
313 ! Reynolds number
314 zre_3d(:,:,j2) = zvoldr_3d(:,:,j2)*zfvelr_3d(:,:,j2) &
315 *prhodref(:,:)/(2.0*zvisca_3d(:,:))
316 zre_3d_sqrt(:,:,j2) = sqrt( zre_3d(:,:,j2) )
317 ! Critical Stokes number
318 zst_star_3d(:,:,j2) = (1.2+(log(1.+zre_3d(:,:,j2)))/12.) &
319 /(1.+log(1.+zre_3d(:,:,j2)))
320 END WHERE
321END DO
322!
323!
324!------------------------------------------------------------------------------
325!
326!
327!* 3. AEROSOL SCAVENGING
328! ------------------
329!
330!
331! Iteration over the aerosol type and mode
332!
333DO isv = 1, nmod_ccn+nmod_ifn
334!
335 IF (isv .LE. nmod_ccn) THEN
336 imod = isv
337 isv_var = tnsv%NSV_LIMA_CCN_FREE - tnsv%NSV_LIMA_BEG + imod ! Variable number in PSVT
338 inm = 1 ! Number of species (for IFN int. mixing)
339 ELSE
340 imod = isv - nmod_ccn
341 isv_var = tnsv%NSV_LIMA_IFN_FREE - tnsv%NSV_LIMA_BEG + imod
342 inm = nspecie
343 END IF
344!
345 zbc_scav_coef(:,:,:) = 0.
346 zmean_scav_coef(:,:) = 0.
347 ztot_scav_rate(:,:) = 0.
348!
349 gscav(:,:) = .false.
350 gscav(d%NIJB:d%NIJE,d%NKB:d%NKE) =grain(d%NIJB:d%NIJE,d%NKB:d%NKE) .AND. &
351 (psvt(d%NIJB:d%NIJE,d%NKB:d%NKE,isv_var)>1.0e-2)
352 iscav = countjv(gscav(:,:),i1(:),i3(:))
353!
354 IF( iscav>=1 ) THEN
355 ALLOCATE(zvisc_ratio(iscav))
356 ALLOCATE(zrhodref(iscav))
357 ALLOCATE(zvisca(iscav))
358 ALLOCATE(zt(iscav))
359 ALLOCATE(zrrt(iscav))
360 ALLOCATE(zconcr(iscav))
361 ALLOCATE(zlambdar(iscav))
362 ALLOCATE(zconcp(iscav))
363 ALLOCATE(zmfpa(iscav))
364 ALLOCATE(ztot_scav_rate1d(iscav))
365 ALLOCATE(ztot_mass_rate1d(iscav))
366 ALLOCATE(zmean_scav_coef1d(iscav))
367 ALLOCATE(zpabst(iscav))
368 ALLOCATE(zknudsen(iscav))
369 ALLOCATE(zfactor(iscav))
370!
371 ALLOCATE(zcunslip(iscav,ndiamp))
372 ALLOCATE(zbc_scav_coef2d(iscav,ndiamp))
373 ALLOCATE(zsc(iscav,ndiamp))
374 ALLOCATE(zsc_inv(iscav,ndiamp))
375 ALLOCATE(zsc_sqrt(iscav,ndiamp))
376 ALLOCATE(zsc_3sqrt(iscav,ndiamp))
377 ALLOCATE(zrelt(iscav,ndiamp))
378 ALLOCATE(zdiff(iscav,ndiamp))
379 ALLOCATE(zvoldr(iscav,ndiamr))
380 ALLOCATE(zvoldr_pow(iscav,ndiamr))
381 ALLOCATE(zvoldr_inv(iscav,ndiamr))
382 ALLOCATE(zre(iscav,ndiamr))
383 ALLOCATE(zre_inv(iscav,ndiamr))
384 ALLOCATE(zre_sqrt(iscav,ndiamr))
385 ALLOCATE(zst_star(iscav,ndiamr))
386 ALLOCATE(zfvelr(iscav,ndiamr))
387 ALLOCATE(zst(iscav,ndiamp,ndiamr))
388 ALLOCATE(zcol_ef(iscav,ndiamp,ndiamr))
389 ALLOCATE(zsize_ratio(iscav,ndiamp,ndiamr))
390!
391 zmean_scav_coef1d(:)=0.
392 ztot_scav_rate1d(:) =0.
393 ztot_mass_rate1d(:) =0.
394 DO il=1,iscav
395 zrhodref(il) = prhodref(i1(il),i3(il))
396 zt(il) = zt_3d(i1(il),i3(il))
397 zrrt(il) = prrt(i1(il),i3(il))
398 zpabst(il) = ppabst(i1(il),i3(il))
399 zconcp(il) = psvt(i1(il),i3(il),isv_var)*zrhodref(il)![/m3]
400 zconcr(il) = zconcr_3d(i1(il),i3(il)) ![/m3]
401 zvisca(il) = zvisca_3d(i1(il),i3(il))
402 zmfpa(il) = zmfpa_3d(i1(il),i3(il))
403 zvisc_ratio(il) = zvisc_ratio_3d(i1(il),i3(il))
404 zlambdar(il) = zlambdar_3d(i1(il),i3(il))
405 zfactor(il) = zfactor_3d(i1(il),i3(il))
406 zvoldr(il,:) = zvoldr_3d(i1(il),i3(il),:)
407 zvoldr_pow(il,:) = zvoldr_3d_pow(i1(il),i3(il),:)
408 zvoldr_inv(il,:) = zvoldr_3d_inv(i1(il),i3(il),:)
409 zfvelr(il,:) = zfvelr_3d(i1(il),i3(il),:)
410 zre(il,:) = zre_3d(i1(il),i3(il),:)
411 zre_sqrt(il,:) = zre_3d_sqrt(i1(il),i3(il),:)
412 zst_star(il,:) = zst_star_3d(i1(il),i3(il),:)
413 ENDDO
414 zre_inv(:,:) = 1./zre(:,:)
415
416 IF (any(zconcr .EQ. 0.)) print *, 'VALEUR NULLE DANS ZLAMBDAR !'
417 IF (any(zlambdar .EQ. 0.)) print *, 'VALEUR NULLE DANS ZLAMBDAR !'
418!
419!------------------------------------------------------------------------------------
420!
421! Loop over the different species (for IFN int. mixing)
422!
423 DO ijm = 1, inm ! species (DM1,DM2,BC,O) for IFN
424 IF ( isv .LE. nmod_ccn ) THEN ! CCN case
425 zrhop = xrho_ccn(imod)
426 zmdiamp = 2*xr_mean_ccn(imod)
427 zsigmap = exp(xlogsig_ccn(imod))
428 zfracp = 1.0
429 ELSE ! IFN case
430 zrhop = xrho_ifn(ijm)
431 zmdiamp = xmdiam_ifn(ijm)
432 zsigmap = xsigma_ifn(ijm)
433 zfracp = xfrac(ijm,imod)
434 END IF
435 !-----------------------------------------------------------------------------
436 ! Loop over the aerosols particles diameters (log normal distribution law) :
437 !
438 DO j1=1,ndiamp
439 ! exchange of variables: [m]
440 zvoldp(j1) = zmdiamp * exp(zabscissp(j1)*sqrt(2.)*log(zsigmap))
441 ! Cunningham slip correction factor (1+alpha*Knudsen)
442 zknudsen(:) = min( 20.,zvoldp(j1)/zmfpa(:) )
443 zcunslip(:,j1) = 1.0+2.0/zknudsen(:)*(1.257+0.4*exp(-0.55*zknudsen(:)))
444 ! Diffusion coefficient
445 zdiff(:,j1) = cst%XBOLTZ*zt(:)*zcunslip(:,j1)/(3.*cst%XPI*zvisca(:)*zvoldp(j1))
446 ! Schmidt number
447 zsc(:,j1) = zvisca(:)/(zrhodref(:)*zdiff(:,j1))
448 zsc_inv(:,j1) = 1./zsc(:,j1)
449 zsc_sqrt(:,j1) = sqrt( zsc(:,j1) )
450 zsc_3sqrt(:,j1) = zsc(:,j1)**(1./3.)
451 ! Characteristic Time Required for reaching terminal velocity
452 zrelt(:,j1) = (zvoldp(j1)**2)*zcunslip(:,j1)*zrhop/(18.*zvisca(:))
453 ! Density number
454 zdens_ratio = zrhop/cst%XRHOLW
455 zdens_ratio_sqrt = sqrt(zdens_ratio)
456 ! Initialisation
457 zbc_scav_coef2d(:,j1)=0.
458 !-------------------------------------------------------------------------
459 ! Loop over the drops diameters (generalized Gamma distribution) :
460 !
461 DO j2=1,ndiamr
462 ! Stokes number
463 zst(:,j1,j2) = 2.*zrelt(:,j1)*(zfvelr(:,j2)-zrelt(1,j1)*cst%XG) &
464 *zvoldr_inv(:,j2)
465 ! Size Ratio
466 zsize_ratio(:,j1,j2) = zvoldp(j1)*zvoldr_inv(:,j2)
467 ! Collision Efficiency
468 zcol_ef(:,j1,j2) = coll_effi(iscav, ndiamp, ndiamr, &
469 zre, zre_inv, zre_sqrt, zsc, zsc_inv, &
470 zsc_sqrt, zsc_3sqrt, zst, zst_star, &
471 zsize_ratio, zvisc_ratio, zdens_ratio_sqrt)
472 ! Below-Cloud Scavenging Coefficient for a fixed ZVOLDP: [/s]
473 zbc_scav_coef2d(:,j1) = zbc_scav_coef2d(:,j1) + &
474 zcol_ef(:,j1,j2) * zweightr(j2) * zfactor(:) * zvoldr_pow(:,j2)
475 END DO
476 ! End of the loop over the drops diameters
477 !--------------------------------------------------------------------------
478
479 ! Total NUMBER Scavenging Rate of aerosol [m**-3.s**-1]
480 ztot_scav_rate1d(:) = ztot_scav_rate1d(:) - &

696 zvtrmax = 10.
697 zdzmin = minval(pzz(d%NIJB:d%NIJE,d%NKB+1:d%NKE+1)-pzz(d%NIJB:d%NIJE,d%NKB:d%NKE))

700 zt = 2.* ptstep / real(isplitr)

703 END DO split
704!

707END IF firstcall

757 END IF
758 DEALLOCATE(zrhodref)

760 DEALLOCATE(zzw)

763 gsedim(d%NIJB:d%NIJE,d%NKB:d%NKE) = zzs(d%NIJB:d%NIJE,d%NKB:d%NKE) > zrtmin3

769 END IF
