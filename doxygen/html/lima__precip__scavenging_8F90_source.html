11 !########################################################################x
12 !
13 !! PURPOSE
14 !! -------
15 !! The purpose of this routine is to compute the total number
16 !! below-cloud scavenging rate.
17 !!
18 !!
19 !!** METHOD
20 !! ------
21 !! We assume a generalized gamma distribution law for the raindrop.
22 !! The aerosols particles distribution follows a log-normal law.
23 !! First, we have to compute the Collision Efficiency, which takes
24 !! account of the three most important wet removal mechanism :
25 !! Brownian diffusion, interception and inertial impaction.
26 !! It is a function of several number (like Reynolds, Schmidt
27 !! or Stokes number for instance). Consequently,
28 !! we need first to calculate these numbers.
29 !!
30 !! Then the scavenging coefficient is deduced from the integration
31 !! of the droplet size distribution, the falling velocity of
32 !! raindrop and aerosol, their diameter, and the collision
33 !! (or collection) efficiency, over the spectrum of droplet
34 !! diameters.
35 !!
36 !! The total scavenging rate of aerosol is computed from the
37 !! integration, over all the spectrum of particles aerosols
38 !! diameters, of the scavenging coefficient.
39 !!
40 !!
41 !! EXTERNAL
42 !! --------
43 !! Subroutine SCAV_MASS_SEDIMENTATION
44 !!
45 !! Function COLL_EFFIC : computes the collision efficiency
46 !!
47 !! Function CONTJV |
48 !! Function GAUHER |
49 !! Function GAULAG |-> in lima_functions.f90
50 !! Function GAMMLN |
51 !!
52 !!

232 isv_lima_nr = nsv_lima_nr - nsv_lima_beg + 1

234

248 !
249 !* 1. PRELIMINARY COMPUTATIONS
250 ! ------------------------
251 !
252 !
253 iib=1+jphext
254 iie=SIZE(prhodref,1) - jphext
255 ijb=1+jphext
256 ije=SIZE(prhodref,2) - jphext
257 ikb=1+jpvext
258 ike=SIZE(prhodref,3) - jpvext
259 !
260 ! PCRT
261 pcrt(:,:,:)=psvt(:,:,:,isv_lima_nr)
262 !
263 ! Rain mask
264 grain(:,:,:) = .false.
265 grain(iib:iie,ijb:ije,ikb:ike) = (prrt(iib:iie,ijb:ije,ikb:ike)>xrtmin(3) &
266  .AND. pcrt(iib:iie,ijb:ije,ikb:ike)>xctmin(3) )
267 !
268 ! Initialize the total mass scavenging rate if LAERO_MASS=T
269 IF (laero_mass) ptot_mass_rate(:,:,:) = 0.
270 !
271 ! Quadrature method: compute absissae and weights
272 CALL gauher(zabscissp,zweightp,ndiamp)
273 znum = xnur-1.0e0
274 CALL gaulag(zabscissr,zweightr,ndiamr,znum)
275 !
276 !
277 !------------------------------------------------------------------------------
278 !
279 !
280 !* 2. NUMERICAL OPTIMIZATION
281 ! ----------------------
282 !
283 !
284 ! Optimization : compute in advance parameters depending on rain particles and
285 ! environment conditions only, to avoid multiple identical computations in loops
286 !
287 !
288 zshape_factor = gamma_x0d(xnur+3./xalphar)/gamma_x0d(xnur)
289 !
290 WHERE ( grain(:,:,:) )
291  !
292  zt_3d(:,:,:) = ptht(:,:,:) * ( ppabst(:,:,:)/cst%XP00 )**(cst%XRD/cst%XCPD)
293  zconcr_3d(:,:,:) = pcrt(:,:,:) * prhodref(:,:,:) ![/m3]
294  ! Sutherland law for viscosity of air
295  zvisca_3d(:,:,:) = xmua0*(zt_3d(:,:,:)/xtref)**1.5*(xtref+xt_suth_a) &
296  /(xt_suth_a+zt_3d(:,:,:))
297  ! Air mean free path
298  zmfpa_3d(:,:,:) = xmfpa0*(cst%XP00*zt_3d(:,:,:))/(ppabst(:,:,:)*xt0scav)
299  ! Viscosity ratio
300  zvisc_ratio_3d(:,:,:) = zvisca_3d(:,:,:)/xviscw !!!!! inversé par rapport à orig. !
301  ! Rain drops parameters
302  zlambdar_3d(:,:,:) = ( ((cst%XPI/6.)*zshape_factor*cst%XRHOLW*zconcr_3d(:,:,:)) &
303  /(prhodref(:,:,:)*prrt(:,:,:)) )**(1./3.) ![/m]
304  factor_3d(:,:,:) = cst%XPI*0.25*zconcr_3d(:,:,:)*xcr*(xrho00/prhodref(:,:,:))**(0.4)
305  !
306 END WHERE
307 !
308 DO j2=1,ndiamr
309  WHERE ( grain(:,:,:) )
310  ! exchange of variables: [m]
311  zvoldr_3d(:,:,:,j2) = zabscissr(j2)**(1./xalphar)/zlambdar_3d(:,:,:)
312  zvoldr_3d_inv(:,:,:,j2) = 1./zvoldr_3d(:,:,:,j2)
313  zvoldr_3d_pow(:,:,:,j2) = zvoldr_3d(:,:,:,j2)**(2.+xdr)
314  ! Raindrop Falling VELocity [m/s]
315  zfvelr_3d(:,:,:,j2) = xcr*(zvoldr_3d(:,:,:,j2)**xdr)*(xrho00/prhodref(:,:,:))**(0.4)
316  ! Reynolds number
317  zre_3d(:,:,:,j2) = zvoldr_3d(:,:,:,j2)*zfvelr_3d(:,:,:,j2) &
318  *prhodref(:,:,:)/(2.0*zvisca_3d(:,:,:))
319  zre_3d_sqrt(:,:,:,j2) = sqrt( zre_3d(:,:,:,j2) )
320  ! Critical Stokes number
321  zst_star_3d(:,:,:,j2) = (1.2+(log(1.+zre_3d(:,:,:,j2)))/12.) &
322  /(1.+log(1.+zre_3d(:,:,:,j2)))
323  END WHERE
324 END DO
325 !
326 !
327 !------------------------------------------------------------------------------
328 !
329 !
330 !* 3. AEROSOL SCAVENGING
331 ! ------------------
332 !
333 !
334 ! Iteration over the aerosol type and mode
335 !
336 DO jsv = 1, nmod_ccn+nmod_ifn
337 !
338  IF (jsv .LE. nmod_ccn) THEN
339  jmod = jsv
340  sv_var = nsv_lima_ccn_free - nsv_lima_beg + jmod ! Variable number in PSVT
341  nm = 1 ! Number of species (for IFN int. mixing)
342  ELSE
343  jmod = jsv - nmod_ccn
344  sv_var = nsv_lima_ifn_free - nsv_lima_beg + jmod
345  nm = nspecie
346  END IF
347 !
348  pbc_scav_coef(:,:,:,:) = 0.
349  pmean_scav_coef(:,:,:) = 0.
350  ptot_scav_rate(:,:,:) = 0.
351 !
352  gscav(:,:,:) = .false.
353  gscav(iib:iie,ijb:ije,ikb:ike) =grain(iib:iie,ijb:ije,ikb:ike) .AND. &
354  (psvt(iib:iie,ijb:ije,ikb:ike,sv_var)>1.0e-2)
355  iscav = countjv(gscav(:,:,:),i1(:),i2(:),i3(:))
356 !
357  IF( iscav>=1 ) THEN
358  ALLOCATE(zvisc_ratio(iscav))
359  ALLOCATE(zrhodref(iscav))
360  ALLOCATE(zvisca(iscav))
361  ALLOCATE(zt(iscav))
362  ALLOCATE(zrrt(iscav))
363  ALLOCATE(zconcr(iscav))
364  ALLOCATE(zlambdar(iscav))
365  ALLOCATE(zconcp(iscav))
366  ALLOCATE(zmfpa(iscav))
367  ALLOCATE(ztot_scav_rate(iscav))
368  ALLOCATE(ztot_mass_rate(iscav))
369  ALLOCATE(zmean_scav_coef(iscav))
370  ALLOCATE(zpabst(iscav))
371  ALLOCATE(zknudsen(iscav))
372  ALLOCATE(factor(iscav))
373 !
374  ALLOCATE(zcunslip(iscav,ndiamp))
375  ALLOCATE(zbc_scav_coef(iscav,ndiamp))
376  ALLOCATE(zsc(iscav,ndiamp))
377  ALLOCATE(zsc_inv(iscav,ndiamp))
378  ALLOCATE(zsc_sqrt(iscav,ndiamp))
379  ALLOCATE(zsc_3sqrt(iscav,ndiamp))
380  ALLOCATE(zrelt(iscav,ndiamp))
381  ALLOCATE(zdiff(iscav,ndiamp))
382  ALLOCATE(zvoldr(iscav,ndiamr))
383  ALLOCATE(zvoldr_pow(iscav,ndiamr))
384  ALLOCATE(zvoldr_inv(iscav,ndiamr))
385  ALLOCATE(zre(iscav,ndiamr))
386  ALLOCATE(zre_inv(iscav,ndiamr))
387  ALLOCATE(zre_sqrt(iscav,ndiamr))
388  ALLOCATE(zst_star(iscav,ndiamr))
389  ALLOCATE(zfvelr(iscav,ndiamr))
390  ALLOCATE(zst(iscav,ndiamp,ndiamr))
391  ALLOCATE(zcol_ef(iscav,ndiamp,ndiamr))
392  ALLOCATE(zsize_ratio(iscav,ndiamp,ndiamr))
393 !
394  zmean_scav_coef(:)=0.
395  ztot_scav_rate(:) =0.
396  ztot_mass_rate(:) =0.
397  DO jl=1,iscav
398  zrhodref(jl) = prhodref(i1(jl),i2(jl),i3(jl))
399  zt(jl) = zt_3d(i1(jl),i2(jl),i3(jl))
400  zrrt(jl) = prrt(i1(jl),i2(jl),i3(jl))
401  zpabst(jl) = ppabst(i1(jl),i2(jl),i3(jl))
402  zconcp(jl) = psvt(i1(jl),i2(jl),i3(jl),sv_var)*zrhodref(jl)![/m3]
403  zconcr(jl) = zconcr_3d(i1(jl),i2(jl),i3(jl)) ![/m3]
404  zvisca(jl) = zvisca_3d(i1(jl),i2(jl),i3(jl))
405  zmfpa(jl) = zmfpa_3d(i1(jl),i2(jl),i3(jl))
406  zvisc_ratio(jl) = zvisc_ratio_3d(i1(jl),i2(jl),i3(jl))
407  zlambdar(jl) = zlambdar_3d(i1(jl),i2(jl),i3(jl))
408  factor(jl) = factor_3d(i1(jl),i2(jl),i3(jl))
409  zvoldr(jl,:) = zvoldr_3d(i1(jl),i2(jl),i3(jl),:)
410  zvoldr_pow(jl,:) = zvoldr_3d_pow(i1(jl),i2(jl),i3(jl),:)
411  zvoldr_inv(jl,:) = zvoldr_3d_inv(i1(jl),i2(jl),i3(jl),:)
412  zfvelr(jl,:) = zfvelr_3d(i1(jl),i2(jl),i3(jl),:)
413  zre(jl,:) = zre_3d(i1(jl),i2(jl),i3(jl),:)
414  zre_sqrt(jl,:) = zre_3d_sqrt(i1(jl),i2(jl),i3(jl),:)
415  zst_star(jl,:) = zst_star_3d(i1(jl),i2(jl),i3(jl),:)
416  ENDDO
417  zre_inv(:,:) = 1./zre(:,:)
418 
419  IF (any(zconcr .eq. 0.)) print *, 'valeur nulle dans ZLAMBDAR !'
420  IF (any(zlambdar .eq. 0.)) print *, 'valeur nulle dans ZLAMBDAR !'
421 !
422 !------------------------------------------------------------------------------------
423 !
424 ! Loop over the different species (for IFN int. mixing)
425 !
426  DO jm = 1, nm ! species (DM1,DM2,BC,O) for IFN
427  IF ( jsv .LE. nmod_ccn ) THEN ! CCN case
428  xrhop = xrho_ccn(jmod)
429  xmdiamp = 2*xr_mean_ccn(jmod)
430  xsigmap = exp(xlogsig_ccn(jmod))
431  xfracp = 1.0
432  ELSE ! IFN case
433  xrhop = xrho_ifn(jm)
434  xmdiamp = xmdiam_ifn(jm)
435  xsigmap = xsigma_ifn(jm)
436  xfracp = xfrac(jm,jmod)
437  END IF
438  !-----------------------------------------------------------------------------
439  ! Loop over the aerosols particles diameters (log normal distribution law) :
440  !
441  DO j1=1,ndiamp
442  ! exchange of variables: [m]
443  zvoldp(j1) = xmdiamp * exp(zabscissp(j1)*sqrt(2.)*log(xsigmap))
444  ! Cunningham slip correction factor (1+alpha*Knudsen)
445  zknudsen(:) = min( 20.,zvoldp(j1)/zmfpa(:) )
446  zcunslip(:,j1) = 1.0+2.0/zknudsen(:)*(1.257+0.4*exp(-0.55*zknudsen(:)))
447  ! Diffusion coefficient
448  zdiff(:,j1) = cst%XBOLTZ*zt(:)*zcunslip(:,j1)/(3.*cst%XPI*zvisca(:)*zvoldp(j1))
449  ! Schmidt number
450  zsc(:,j1) = zvisca(:)/(zrhodref(:)*zdiff(:,j1))
451  zsc_inv(:,j1) = 1./zsc(:,j1)
452  zsc_sqrt(:,j1) = sqrt( zsc(:,j1) )
453  zsc_3sqrt(:,j1) = zsc(:,j1)**(1./3.)
454  ! Characteristic Time Required for reaching terminal velocity

551  ENDIF

688 !-------------------------------------------------------------------------------

692 !
693 !* 2.1 splitting factor for high Courant number C=v_fall*(del_Z/del_T)
694 !
695 firstcall : IF (gsfirstcall) THEN
696  gsfirstcall = .false.
697  zvtrmax = 10.
698  zdzmin = minval(pzz(iib:iie,ijb:ije,ikb+1:ike+1)-pzz(iib:iie,ijb:ije,ikb:ike))

708 END IF firstcall

714  ztstep = ptstep ! Large time step
715  ELSE
716  ztsplitr= 2. * ptstep / real(isplitr)
717  ztstep = 2. * ptstep
718 END IF

721 !
722 ! optimization by looking for locations where
723 ! the precipitating fields are larger than a minimal value only !!!
724 !
725 zrtmin3 = xrtmin(3) / ztstep

729  gsedim(iib:iie,ijb:ije,ikb:ike) = zzs(iib:iie,ijb:ije,ikb:ike) > zrtmin3
730 !
731  isedim = countjv( gsedim(:,:,:),i1(:),i2(:),i3(:))
732  IF( isedim >= 1 ) THEN
733  IF( jn==1 ) THEN
734  zzs(:,:,:) = zzs(:,:,:) * ztstep
735  DO jk = ikb , ike-1
736  zw(:,:,jk) =ztsplitr*2./(prhodref(:,:,jk)*(pzz(:,:,jk+2)-pzz(:,:,jk)))
737  END DO
738  zw(:,:,ike) =ztsplitr/(prhodref(:,:,ike)*(pzz(:,:,ike+1)-pzz(:,:,ike)))
739  END IF
740  ALLOCATE(zrrs(isedim))
741  ALLOCATE(zrhodref(isedim))
742  DO jl=1,isedim
743  zrrs(jl) = zzs(i1(jl),i2(jl),i3(jl))
744  zrhodref(jl) = prhodref(i1(jl),i2(jl),i3(jl))
745  ENDDO
746  ALLOCATE(zzw(isedim)) ; zzw(:) = 0.0
747 !
748 !* 2.2.1 for rain
749 !
750  zzw(:) = xfsedrr * zrrs(:)**(zexsedr) * zrhodref(:)**(zexsedr-xcexvt)
751  zwsed(:,:,:) = unpack( zzw(:),mask=gsedim(:,:,:),field=0.0 )
752  DO jk = ikb , ike
753  zzs(:,:,jk) = zzs(:,:,jk) + zw(:,:,jk)*(zwsed(:,:,jk+1)-zwsed(:,:,jk))
754  END DO
755  IF( jn==1 ) THEN
756  pinpap(:,:) = zwsed(:,:,ikb)* &
757  ( psvt_mass(:,:,ikb)/max(zrtmin3,prrt(:,:,ikb)) )
758  END IF
759  DEALLOCATE(zrhodref)
760  DEALLOCATE(zrrs)
761  DEALLOCATE(zzw)
762  IF( jn==isplitr ) THEN
763  gsedim(:,:,:) = .false.
764  gsedim(iib:iie,ijb:ije,ikb:ike) = zzs(iib:iie,ijb:ije,ikb:ike) > zrtmin3
765  zwsed(:,:,:) = 0.0
766  WHERE( gsedim(:,:,:) )
767  zwsed(:,:,:) = 1.0/ztstep - prain(:,:,:)/zzs(:,:,:)
768  END WHERE
769  END IF
770  END IF
771 END DO
772 !
773 ! Apply the rain sedimentation rate to the WR_xxx aqueous species
774 !
775 prsvs_mass(:,:,:) = prsvs_mass(:,:,:) + zwsed(:,:,:)*psvt_mass(:,:,:)
776 !
777 END SUBROUTINE scav_mass_sedimentation

782  FUNCTION coll_effi (PRE, PRE_INV, PRE_SQRT, PSC, PSC_INV, PSC_SQRT, &
783  PSC_3SQRT, PST, PST_STAR, PSIZE_RATIO, &
784  PVISC_RATIO, PDENS_RATIO_SQRT) RESULT(PCOL_EF)
785 !###################################################################
786 !
787 !Compute the Raindrop-Aerosol Collision Efficiency
788 !
789 !* 0. DECLARATIONS
790 ! ---------------
791 !
792  IMPLICIT NONE
793 !
794  INTEGER :: i
795 !
796  REAL, DIMENSION(:,:), INTENT(IN) :: pre
797  REAL, DIMENSION(:,:), INTENT(IN) :: pre_inv
798  REAL, DIMENSION(:,:), INTENT(IN) :: pre_sqrt

821  pcol_ef(i) = pcol_ef(i) &
822  +(pdens_ratio_sqrt*((pst(i,j1,j2)-pst_star(i,j2)) &
823  /(pst(i,j1,j2)-pst_star(i,j2)+2./3.))**(3./2.))
824  ENDIF
825  ENDDO
826  END FUNCTION coll_effi
