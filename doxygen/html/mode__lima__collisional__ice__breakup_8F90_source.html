106 !
107 p_ri_cibu(:)=0.
108 p_ci_cibu(:)=0.
109 !
110 IF (icibu > 0) THEN
111 !
112 ! 1.3.0 randomization of XNDEBRIS_CIBU values
113 !
114  IF (gfirstcall) THEN
115  CALL random_seed(size=ni_seed) ! get size of seed
116  ALLOCATE(i_seed(ni_seed))
117  i_seed(:) = i_seed_param !
118  CALL random_seed(put=i_seed)
119  gfirstcall = .false.
120  END IF
121 !
122  ALLOCATE(zfragments(icibu))
123 !
124  IF (xndebris_cibu >= 0.0) THEN
125  zfragments(:) = xndebris_cibu
126  ELSE
127 !
128 ! Mantissa gives the mean value (randomization around 10**MANTISSA)
129 ! First digit after the comma provides the full range around 10**MANTISSA
130 !
131  ALLOCATE(zharvest(icibu))
132 !
133  zfact1_xndebris = aint(xndebris_cibu)
134  zfact2_xndebris = abs(anint(10.0*(xndebris_cibu - zfact1_xndebris)))
135 !
136  CALL random_number(zharvest(:))
137 !
138  zfragments(:) = 10.0**(zfact2_xndebris*zharvest(:) + zfact1_xndebris)
139 !
140  DEALLOCATE(zharvest)
141 !
142 ! ZFRAGMENTS is a random variable containing the number of fragments per collision
143 ! For XNDEBRIS_CIBU=-1.2345 => ZFRAGMENTS(:) = 10.0**(2.0*RANDOM_NUMBER(ZHARVEST(:)) - 1.0)
144 ! and ZFRAGMENTS=[0.1, 10.0] centered around 1.0
145 !
146  END IF
147 !
148 !
149 ! 1.3.1 To compute the partial integration of snow gamma function
150 !
151 ! 1.3.1.0 allocations
152 !
153  ALLOCATE(zvec1_s(icibu))
154  ALLOCATE(zvec1_sw(icibu))
155  ALLOCATE(zvec1_s1(icibu))
156  ALLOCATE(zvec1_s2(icibu))
157  ALLOCATE(zvec1_s3(icibu))
158  ALLOCATE(zvec1_s4(icibu))
159  ALLOCATE(zvec1_s11(icibu))
160  ALLOCATE(zvec1_s12(icibu))
161  ALLOCATE(zvec1_s21(icibu))
162  ALLOCATE(zvec1_s22(icibu))
163  ALLOCATE(zvec1_s31(icibu))
164  ALLOCATE(zvec1_s32(icibu))
165  ALLOCATE(zvec1_s41(icibu))
166  ALLOCATE(zvec1_s42(icibu))
167  ALLOCATE(zvec2_s1(icibu))
168  ALLOCATE(ivec2_s1(icibu))
169  ALLOCATE(zvec2_s2(icibu))
170  ALLOCATE(ivec2_s2(icibu))
171 !
172 !
173 ! 1.3.1.1 select the PLBDS
174 !
175  zvec1_s(:) = pack( plbds(:),mask=gcibu(:) )
176  zvec1_sw(:)= ( xfvelos**xalphas + zvec1_s(:)**xalphas ) ** (1./xalphas) ! modified equivalent lambda
177 !
178 !
179 ! 1.3.1.2 find the next lower indice for the PLBDS in the
180 ! geometrical set of Lbda_s used to tabulate some moments of the
181 ! incomplete gamma function, for boundary 1 (0.2 mm)
182 !
183  zvec2_s1(1:icibu) = max( 1.0001, min( float(ngaminc)-0.0001,xcibuintp_s &
184  * log( zvec1_s(1:icibu) ) + xcibuintp1_s ) )
185  ivec2_s1(1:icibu) = int( zvec2_s1(1:icibu) )
186  zvec2_s1(1:icibu) = zvec2_s1(1:icibu) - float( ivec2_s1(1:icibu) )
187 !
188 !
189 ! 1.3.1.3 find the next lower indice for the PLBDS in the
190 ! geometrical set of Lbda_s used to tabulate some moments of the
191 ! incomplete gamma function, for boundary 2 (1 mm)
192 !
193  zvec2_s2(1:icibu) = max( 1.0001, min( float(ngaminc)-0.0001,xcibuintp_s &
194  * log( zvec1_s(1:icibu) ) + xcibuintp2_s ) )
195  ivec2_s2(1:icibu) = int( zvec2_s2(1:icibu) )
196  zvec2_s2(1:icibu) = zvec2_s2(1:icibu) - float( ivec2_s2(1:icibu) )
197 !
198 !
199 ! 1.3.1.4 perform the linear interpolation of the
200 ! normalized "0"-moment of the incomplete gamma function
201 !
202 ! For lower boundary (0.2 mm)
203  zvec1_s11(1:icibu) = xgaminc_cibu_s(1,ivec2_s1(1:icibu)+1) * zvec2_s1(1:icibu) &
204  - xgaminc_cibu_s(1,ivec2_s1(1:icibu)) * (zvec2_s1(1:icibu)-1.0)
205 !
206 ! For upper boundary (1 mm)
207  zvec1_s12(1:icibu) = xgaminc_cibu_s(1,ivec2_s2(1:icibu)+1) * zvec2_s2(1:icibu) &
208  - xgaminc_cibu_s(1,ivec2_s2(1:icibu)) * (zvec2_s2(1:icibu)-1.0)
209 !
210 ! Computation of spectrum from 0.2 mm to 1 mm
211  zvec1_s1(1:icibu) = zvec1_s12(1:icibu) - zvec1_s11(1:icibu)
212 !
213 !
214 ! 1.3.1.5 perform the linear interpolation of the
215 ! normalized "XBS"-moment of the incomplete gamma function
216 !
217 ! For lower boundary (0.2 mm)
218  zvec1_s31(1:icibu) = xgaminc_cibu_s(3,ivec2_s1(1:icibu)+1) * zvec2_s1(1:icibu) &
219  - xgaminc_cibu_s(3,ivec2_s1(1:icibu)) * (zvec2_s1(1:icibu)-1.0)
220 !
221 ! For upper boundary (1 mm)
222  zvec1_s32(1:icibu) = xgaminc_cibu_s(3,ivec2_s2(1:icibu)+1) * zvec2_s2(1:icibu) &
223  - xgaminc_cibu_s(3,ivec2_s2(1:icibu)) * (zvec2_s2(1:icibu)-1.0)
224 !
225 ! From 0.2 mm to 1 mm we need
226  zvec1_s3(1:icibu) = xmomgs_cibu_2 * (zvec1_s32(1:icibu) - zvec1_s31(1:icibu))
227 !
228 !
229 ! 1.3.1.2 find the next lower indice for the PLBDS in the
230 ! geometrical set of Lbda_s used to tabulate some moments of the
231 ! incomplete gamma function, for boundary 1 (0.2 mm) for modified lambda (Wurtz snow fall speed)
232 !
233  zvec2_s1(1:icibu) = max( 1.0001, min( float(ngaminc)-0.0001,xcibuintp_s &
234  * log( zvec1_sw(1:icibu) ) + xcibuintp1_s ) )
235  ivec2_s1(1:icibu) = int( zvec2_s1(1:icibu) )
236  zvec2_s1(1:icibu) = zvec2_s1(1:icibu) - float( ivec2_s1(1:icibu) )
237 !
238 !
239 ! 1.3.1.3 find the next lower indice for the PLBDS in the
240 ! geometrical set of Lbda_s used to tabulate some moments of the
241 ! incomplete gamma function, for boundary 2 (1 mm) for modified lambda (Wurtz snow fall speed)
242 !
243  zvec2_s2(1:icibu) = max( 1.0001, min( float(ngaminc)-0.0001,xcibuintp_s &
244  * log( zvec1_sw(1:icibu) ) + xcibuintp2_s ) )
245  ivec2_s2(1:icibu) = int( zvec2_s2(1:icibu) )
246  zvec2_s2(1:icibu) = zvec2_s2(1:icibu) - float( ivec2_s2(1:icibu) )
247 !
248 !
249 ! 1.3.1.5 perform the linear interpolation of the
250 ! normalized "XDS"-moment of the incomplete gamma function
251 !
252 ! For lower boundary (0.2 mm)
253  zvec1_s21(1:icibu) = xgaminc_cibu_s(2,ivec2_s1(1:icibu)+1) * zvec2_s1(1:icibu) &
254  - xgaminc_cibu_s(2,ivec2_s1(1:icibu)) * (zvec2_s1(1:icibu)-1.0)
255 !
256 ! For upper boundary (1 mm)
257  zvec1_s22(1:icibu) = xgaminc_cibu_s(2,ivec2_s2(1:icibu)+1) * zvec2_s2(1:icibu) &
258  - xgaminc_cibu_s(2,ivec2_s2(1:icibu)) * (zvec2_s2(1:icibu)-1.0)
259 !
260 ! From 0.2 mm to 1 mm we need
261  zvec1_s2(1:icibu) = xmomgs_cibu_1 * (zvec1_s22(1:icibu) - zvec1_s21(1:icibu))
262 !
263 !
264 ! 1.3.1.6 perform the linear interpolation of the
265 ! normalized "XBS+XDS"-moment of the incomplete gamma function
266 !
267 ! For lower boundary (0.2 mm)
268  zvec1_s41(1:icibu) = xgaminc_cibu_s(4,ivec2_s1(1:icibu)+1) * zvec2_s1(1:icibu) &
269  - xgaminc_cibu_s(4,ivec2_s1(1:icibu)) * (zvec2_s1(1:icibu)-1.0)
270 !
271 ! For upper boundary (1 mm)
272  zvec1_s42(1:icibu) = xgaminc_cibu_s(4,ivec2_s2(1:icibu)+1) * zvec2_s2(1:icibu) &
273  - xgaminc_cibu_s(4,ivec2_s2(1:icibu)) * (zvec2_s2(1:icibu)-1.0)
274 !
275 ! From 0.2 mm to 1 mm we need
276  zvec1_s4(1:icibu) = xmomgs_cibu_3 * (zvec1_s42(1:icibu) - zvec1_s41(1:icibu))
277 !
278  zintg_snow_1(:) = unpack( vector=zvec1_s1(:),mask=gcibu,field=0.0 )
279  zintg_snow_2(:) = unpack( vector=zvec1_s2(:),mask=gcibu,field=0.0 )
280  zintg_snow_3(:) = unpack( vector=zvec1_s3(:),mask=gcibu,field=0.0 )
281  zintg_snow_4(:) = unpack( vector=zvec1_s4(:),mask=gcibu,field=0.0 )
282 !
283 !
284 ! 1.3.2 Compute the partial integration of graupel gamma function
285 !
286 ! 1.3.2.0 allocations
287 !
288  ALLOCATE(zvec1_g(icibu))
289  ALLOCATE(zvec1_g1(icibu))
290  ALLOCATE(zvec1_g2(icibu))
291  ALLOCATE(zvec2_g(icibu))
292  ALLOCATE(ivec2_g(icibu))
293 !
294 !
295 ! 1.3.2.1 select the PLBDG
296 !
297  zvec1_g(:) = pack( plbdg(:),mask=gcibu(:) )
298 !
299 !
300 ! 1.3.2.2 find the next lower indice for the PLBDG in the
301 ! geometrical set of Lbda_g used to tabulate some moments of the
302 ! incomplete gamma function, for the "2mm" boundary
303 !
304  zvec2_g(1:icibu) = max( 1.0001, min( float(ngaminc)-0.0001,xcibuintp_g &
305  * log( zvec1_g(1:icibu) ) + xcibuintp1_g ) )
306  ivec2_g(1:icibu) = int( zvec2_g(1:icibu) )
307  zvec2_g(1:icibu) = zvec2_g(1:icibu) - float( ivec2_g(1:icibu) )
308 !
309 !
310 ! 1.3.2.3 perform the linear interpolation of the
311 ! normalized "2+XDG"-moment of the incomplete gamma function
312 !
313  zvec1_g1(1:icibu) = xgaminc_cibu_g(1,ivec2_g(1:icibu)+1) * zvec2_g(1:icibu) &
314  - xgaminc_cibu_g(1,ivec2_g(1:icibu)) * (zvec2_g(1:icibu)-1.0)
315 !
316 ! From 2 mm to infinity we need
317  zvec1_g1(1:icibu) = xmomgg_cibu_1 * (1.0 - zvec1_g1(1:icibu))
318 !
319 !
320 ! 1.3.2.4 perform the linear interpolation of the
321 ! normalized "2.0"-moment of the incomplete gamma function
322 !
323  zvec1_g2(1:icibu) = xgaminc_cibu_g(2,ivec2_g(1:icibu)+1) * zvec2_g(1:icibu) &
324  - xgaminc_cibu_g(2,ivec2_g(1:icibu)) * (zvec2_g(1:icibu)-1.0)
325 !
326 ! From 2 mm to infinity we need
327  zvec1_g2(1:icibu) = xmomgg_cibu_2 * (1.0 - zvec1_g2(1:icibu))
328 !
329 !
330  zintg_graupel_1(:) = unpack( vector=zvec1_g1(:),mask=gcibu,field=0.0 )
331  zintg_graupel_2(:) = unpack( vector=zvec1_g2(:),mask=gcibu,field=0.0 )
332 !
333 !
334 ! 1.3.3 To compute final "CIBU" contributions
335 !
336  zfrag_cibu(:) = unpack( vector=zfragments(:),mask=gcibu,field=0.0 )
337  zni_cibu(:) = zfrag_cibu(:) * (xfactor_cibu_ni * pcst(:) * pcgt(:) / (prhodref(:)**xcexvt)) * &
338  (xcg * zintg_graupel_1(:) * zintg_snow_1(:) * &
339  plbdg(:)**(-(xdg+2.0)) &
340  - xcs * zintg_graupel_2(:) * zintg_snow_2(:) * &
341  plbds(:)**(-xds) * plbdg(:)**(-2.0) * &
342  (1+(xfvelos/plbds(:))**xalphas)**(-xnus-xds/xalphas) )
343 
344  p_ci_cibu(:) = max(zni_cibu(:),0.)
345 !
346  DEALLOCATE(zfragments)
347 !
348 ! Max value of rs removed by CIBU
349  zri_cibu(:) = (xfactor_cibu_ri * pcst(:) * pcgt(:) / (prhodref(:)**xcexvt)) * &
350  (xcg * zintg_graupel_1(:) * zintg_snow_3(:) * &
351  plbds(:)**(-xbs) * plbdg(:)**(-(xdg+2.0)) &
352  - xcs * zintg_graupel_2(:) * zintg_snow_4(:) * &
353  plbds(:)**(-xbs-xds) * plbdg(:)**(-2.0) * &
354  (1+(xfvelos/plbds(:))**xalphas)**(-xnus-(xbs+xds)/xalphas) )
355 !
356 ! The value of rs removed by CIBU is determined by the mean mass of pristine ice
357  WHERE( prit(:)>xrtmin(4) .AND. pcit(:)>xctmin(4) )
358  zri_cibu(:) = min( zri_cibu(:), zni_cibu(:)*prit(:)/pcit(:) )
359  ELSE WHERE
360  zri_cibu(:) = min( zri_cibu(:), max( zni_cibu(:)*xmnu0,xrtmin(4) ) )
361  END WHERE
362 !
363  p_ri_cibu(:) = max(zri_cibu(:), 0.)
364 !
365  DEALLOCATE(zvec1_s)
366  DEALLOCATE(zvec1_sw)
367  DEALLOCATE(zvec1_s1)
368  DEALLOCATE(zvec1_s2)
369  DEALLOCATE(zvec1_s3)
370  DEALLOCATE(zvec1_s4)
371  DEALLOCATE(zvec1_s11)
372  DEALLOCATE(zvec1_s12)
373  DEALLOCATE(zvec1_s21)
374  DEALLOCATE(zvec1_s22)
375  DEALLOCATE(zvec1_s31)
376  DEALLOCATE(zvec1_s32)
377  DEALLOCATE(zvec1_s41)
378  DEALLOCATE(zvec1_s42)
379  DEALLOCATE(zvec2_s1)
380  DEALLOCATE(ivec2_s1)
381  DEALLOCATE(zvec2_s2)
382  DEALLOCATE(ivec2_s2)
383  DEALLOCATE(zvec1_g)
384  DEALLOCATE(zvec1_g1)
385  DEALLOCATE(zvec1_g2)
386  DEALLOCATE(zvec2_g)
387  DEALLOCATE(ivec2_g)
388 END IF
389 !

391 !
392 END SUBROUTINE lima_collisional_ice_breakup
