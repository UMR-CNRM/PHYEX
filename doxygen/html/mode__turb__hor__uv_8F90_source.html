137 !

139  ! sinus of the angle between the vertical and the normal to the orography
140 INTEGER :: IKB,IKE
141  ! Index values for the Beginning and End
142  ! mass points of the domain

144 REAL, DIMENSION(SIZE(PUM,1),SIZE(PUM,2),SIZE(PUM,3)) :: GY_U_UV_PUM

149 ! ---------------------------------------------------------------------------
150 !
151 !* 1. PRELIMINARY COMPUTATIONS
152 ! ------------------------
153 !
154 ikb = 1+jpvext
155 ike = SIZE(pum,3)-jpvext
156 !
157 zdirsinzw(:,:) = sqrt( 1. - pdircoszw(:,:)**2 )
158 !
159 gx_v_uv_pvm = gx_v_uv(pvm,pdxx,pdzz,pdzx)
160 IF (.NOT. o2d) gy_u_uv_pum = gy_u_uv(pum,pdyy,pdzz,pdzy)
161 !
162 !
163 !* 12. < U'V'>
164 ! -------
165 !
166 !
167 IF (.NOT. o2d) THEN
168  zflx(:,:,:)= - xcmfs * mym(mxm(pk)) * &
169  (gy_u_uv_pum + gx_v_uv_pvm)
170 ELSE
171  zflx(:,:,:)= - xcmfs * mym(mxm(pk)) * &
172  (gx_v_uv_pvm)
173 END IF
174 !
175 zflx(:,:,ike+1)= zflx(:,:,ike)
176 !
177 !
178 ! Compute the correlation at the first physical level with the following
179 ! hypothesis du/dz vary in 1/z and w=0 at the ground
180 zflx(:,:,ikb:ikb) = - xcmfs * mym(mxm(pk(:,:,ikb:ikb))) * ( &
181  ( dym( pum(:,:,ikb:ikb) ) &
182  -mym( (pum(:,:,ikb+1:ikb+1)-pum(:,:,ikb:ikb)) &
183  *(1./mxm(pdzz(:,:,ikb+1:ikb+1))+1./mxm(pdzz(:,:,ikb:ikb))))&
184  *0.5*mxm((pdzy(:,:,ikb+1:ikb+1)+pdzy(:,:,ikb:ikb))) &
185  ) / mxm(pdyy(:,:,ikb:ikb)) &
186  +( dxm( pvm(:,:,ikb:ikb) ) &
187  -mxm( (pvm(:,:,ikb+1:ikb+1)-pvm(:,:,ikb:ikb)) &
188  *(1./mym(pdzz(:,:,ikb+1:ikb+1))+1./mym(pdzz(:,:,ikb:ikb))))&
189  *0.5*mym((pdzx(:,:,ikb+1:ikb+1)+pdzx(:,:,ikb:ikb))) &
190  ) / mym(pdxx(:,:,ikb:ikb)) )
191 !
192 ! extrapolates this flux under the ground with the surface flux
193 zflx(:,:,ikb-1) = &
194  ptau11m(:,:) * pcosslope(:,:) * psinslope(:,:) * pdircoszw(:,:)**2 &
195  +ptau12m(:,:) * (pcosslope(:,:)**2 - psinslope(:,:)**2) * &
196  pdircoszw(:,:)**2 &
197  -ptau22m(:,:) * pcosslope(:,:) * psinslope(:,:) &
198  +ptau33m(:,:) * pcosslope(:,:) * psinslope(:,:) * zdirsinzw(:,:)**2 &
199  -pcdueff(:,:) * ( &
200  2. * puslopem(:,:) * pcosslope(:,:) * psinslope(:,:) * &
201  pdircoszw(:,:) * zdirsinzw(:,:) &
202  +pvslopem(:,:) * (pcosslope(:,:)**2 - psinslope(:,:)**2) * zdirsinzw(:,:) &
203  )
204 !
205 zflx(:,:,ikb-1:ikb-1) = 2. * mxm( mym( zflx(:,:,ikb-1:ikb-1) ) ) &
206  - zflx(:,:,ikb:ikb)
207 !
208 ! stores <U V>
209 IF ( tpfile%LOPENED .AND. turbn%LTURB_FLX ) THEN
210  tzfield = tfieldmetadata( &
211  cmnhname = 'UV_FLX', &
212  cstdname = '', &
213  clongname = 'UV_FLX', &
214  cunits = 'm2 s-2', &
215  cdir = 'XY', &
216  ccomment = 'X_Y_Z_UV_FLX', &
217  ngrid = 5, &
218  ntype = typereal, &
219  ndims = 3, &
220  ltimedep = .true. )
221  CALL io_field_write(tpfile,tzfield,zflx)
222 END IF
223 !
224 !
225 !
226 !computation of the source for rho*V due to this flux
227 IF (.NOT. oflat) THEN
228  prus(:,:,:) = prus(:,:,:) &
229  - dyf(zflx * mxm(mym(prhodj) * pinv_pdyy) ) &
230  + dzf( myf( mzm(zflx)*mxm(pdzy/mzm(pdyy))) &
231  * mxm(pmzm_prhodj * pinv_pdzz) )
232 ELSE
233  prus(:,:,:) = prus(:,:,:) - dyf(zflx * mxm(mym(prhodj) * pinv_pdyy) )
234 END IF
235 !
236 !computation of the source for rho*V due to this flux
237 IF (.NOT. oflat) THEN
238  prvs(:,:,:) = prvs(:,:,:) &
239  - dxf(zflx * mym(mxm(prhodj) * pinv_pdxx) ) &
240  + dzf( mxf( mzm(zflx)*mym(pdzx/mzm(pdxx))) &
241  * mym(pmzm_prhodj * pinv_pdzz) )
242 ELSE
243  prvs(:,:,:) = prvs(:,:,:) - dxf(zflx * mym(mxm(prhodj) * pinv_pdxx) )
244 END IF
245 !
246 IF (ksplt==1) THEN
247  !
248  !Contribution to the dynamic production of TKE:
249  !
250  IF (.NOT. o2d) THEN
251  zwork(:,:,:) = - mxf( myf( zflx * &
252  (gy_u_uv_pum + gx_v_uv_pvm) ) )
253  ELSE
254  zwork(:,:,:) = - mxf( myf( zflx * &
255  (gx_v_uv_pvm) ) )
256  ENDIF
257  !
258  ! evaluate the dynamic production at w(IKB+1) in PDP(IKB)
259  !
260  zwork(:,:,ikb:ikb) = - &
261  mxf( myf( 0.5 * (zflx(:,:,ikb+1:ikb+1)+zflx(:,:,ikb:ikb)) ) ) &
262  *(mxf( myf( &
263  dym( 0.5 * (pum(:,:,ikb+1:ikb+1)+pum(:,:,ikb:ikb)) ) &
264  / mxm( 0.5*(pdyy(:,:,ikb:ikb)+pdyy(:,:,ikb+1:ikb+1)) ) &
265  +dxm( 0.5 * (pvm(:,:,ikb+1:ikb+1)+pvm(:,:,ikb:ikb)) ) &
266  / mym( 0.5*(pdxx(:,:,ikb:ikb)+pdxx(:,:,ikb+1:ikb+1)) ) &
267  ) ) &
268  -mxf( (pum(:,:,ikb+1:ikb+1)-pum(:,:,ikb:ikb)) / &
269  mxm(pdzz(:,:,ikb+1:ikb+1)) * pdzy(:,:,ikb+1:ikb+1) &
270  ) / myf(mxm( 0.5*(pdyy(:,:,ikb:ikb)+pdyy(:,:,ikb+1:ikb+1)) ) )&
271  -myf( (pvm(:,:,ikb+1:ikb+1)-pvm(:,:,ikb:ikb)) / &
272  mym(pdzz(:,:,ikb+1:ikb+1)) * pdzx(:,:,ikb+1:ikb+1) &
273  ) / mxf(mym( 0.5*(pdxx(:,:,ikb:ikb)+pdxx(:,:,ikb+1:ikb+1)) ) )&
274  )
275  !
276  ! dynamic production
277  pdp(:,:,:) = pdp(:,:,:) + zwork(:,:,:)
278  !
279 END IF
280 !
281 ! Storage in the LES configuration
282 !
283 IF (tles%LLES_CALL .AND. ksplt==1) THEN
284  CALL second_mnh(ztime1)
285  CALL les_mean_subgrid( mxf(myf(zflx)), tles%X_LES_SUBGRID_UV )
286  CALL les_mean_subgrid( mxf(myf(gy_u_uv(pum,pdyy,pdzz,pdzy)*zflx)), tles%X_LES_RES_ddxa_U_SBG_UaU , .true.)
287  CALL les_mean_subgrid( mxf(myf(gx_v_uv(pvm,pdxx,pdzz,pdzx)*zflx)), tles%X_LES_RES_ddxa_V_SBG_UaV , .true.)
288  CALL second_mnh(ztime2)
289  tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
290 END IF
