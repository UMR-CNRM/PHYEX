97TYPE(tles_t), INTENT(INOUT) :: TLES ! modd_les structure
98INTEGER, INTENT(IN) :: KSPLT ! split process index
99TYPE(tfiledata), INTENT(INOUT) :: TPFILE ! Output file
100LOGICAL, INTENT(IN) :: OFLAT ! Logical for zero ororography
101LOGICAL, INTENT(IN) :: O2D ! Logical for 2D model version (modd_conf)
102!
103REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PK ! Turbulent diffusion doef.
104 ! PK = PLM * SQRT(PTKEM)
105REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDXX ! 1./PDXX
106REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDYY ! 1./PDYY
107REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDZZ ! 1./PDZZ
108REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PMZM_PRHODJ ! MZM(PRHODJ)
109REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PDXX, PDYY, PDZZ, PDZX, PDZY
110 ! Metric coefficients
111REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PDIRCOSZW
112! Director Cosinus along z directions at surface w-point
113REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PCOSSLOPE ! cosinus of the angle
114 ! between i and the slope vector
115REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PSINSLOPE ! sinus of the angle
116 ! between i and the slope vector
117REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PRHODJ ! density * grid volume
118!
119REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PCDUEFF ! Cd * || u || at time t
120REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PTAU11M ! <uu> in the axes linked
121 ! to the maximum slope direction and the surface normal and the binormal
122 ! at time t - dt
123REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PTAU12M ! <uv> in the same axes
124REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PTAU22M ! <vv> in the same axes
125REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PTAU33M ! <ww> in the same axes
126!
127! Variables at t-1
128REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PUM,PVM
129REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PUSLOPEM ! wind component along the
130 ! maximum slope direction
131REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PVSLOPEM ! wind component along the
132 ! direction normal to the maximum slope one
133!
134!
135REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(INOUT) :: PRUS, PRVS ! var. at t+1 -split-
136REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(INOUT) :: PDP ! TKE production terms
137!
138!
139!
140!* 0.2 declaration of local variables
141!
142REAL, DIMENSION(D%NIT,D%NJT,D%NKT) :: ZFLX,ZWORK ! work arrays
143!
144REAL, DIMENSION(D%NIT,D%NJT) :: ZDIRSINZW
145 ! sinus of the angle between the vertical and the normal to the orography
146INTEGER :: IKB,IKE,IIT,IJT,IKT, JI,JJ,JK
147 ! Index values for the Beginning and End
148 ! mass points of the domain
149!
150REAL, DIMENSION(D%NIT,D%NJT,D%NKT) :: GY_U_UV_PUM
151REAL, DIMENSION(D%NIT,D%NJT,D%NKT) :: GX_V_UV_PVM
152!
153REAL :: ZTIME1, ZTIME2
154TYPE(tfieldmetadata) :: TZFIELD
155! ---------------------------------------------------------------------------
156!
157!* 1. PRELIMINARY COMPUTATIONS
158! ------------------------
159!
160ikb = 1+jpvext
161ike = SIZE(pum,3)-jpvext
162iit=d%NIT
163ijt=d%NJT
164ikt=d%NKT
165!
166!$acc kernels
167!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
168zdirsinzw(:,:) = sqrt( 1. - pdircoszw(:,:)**2 )
169!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
170!$acc end kernels
171!
172gx_v_uv_pvm = gx_v_uv(pvm,pdxx,pdzz,pdzx)
173IF (.NOT. o2d) gy_u_uv_pum = gy_u_uv(pum,pdyy,pdzz,pdzy)
174!
175!
176!* 12. < U'V'>
177! -------
178!
179!
180IF (.NOT. o2d) THEN
181 zflx(:,:,:)= - xcmfs * mym(mxm(pk)) * &
182 (gy_u_uv_pum + gx_v_uv_pvm)
183ELSE
184 zflx(:,:,:)= - xcmfs * mym(mxm(pk)) * &
185 (gx_v_uv_pvm)
186END IF
187!
188!$acc kernels
189!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
190zflx(:,:,ike+1)= zflx(:,:,ike)
191!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
192!$acc end kernels
193!
194!
195! Compute the correlation at the first physical level with the following
196! hypothesis du/dz vary in 1/z and w=0 at the ground
197zflx(:,:,ikb) = - xcmfs * mym(mxm(pk(:,:,ikb))) * ( &
198 ( dym( pum(:,:,ikb) ) &
199 -mym( (pum(:,:,ikb+1)-pum(:,:,ikb)) &
200 *(1./mxm(pdzz(:,:,ikb+1))+1./mxm(pdzz(:,:,ikb))))&
201 *0.5*mxm((pdzy(:,:,ikb+1)+pdzy(:,:,ikb))) &
202 ) / mxm(pdyy(:,:,ikb)) &
203 +( dxm( pvm(:,:,ikb) ) &
204 -mxm( (pvm(:,:,ikb+1)-pvm(:,:,ikb)) &
205 *(1./mym(pdzz(:,:,ikb+1))+1./mym(pdzz(:,:,ikb))))&
206 *0.5*mym((pdzx(:,:,ikb+1)+pdzx(:,:,ikb))) &
207 ) / mym(pdxx(:,:,ikb)) )
208!
209! extrapolates this flux under the ground with the surface flux
210!$acc kernels present_cr(ZFLX,ZDIRSINZW)
211!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
212zflx(:,:,ikb-1) = &
213 ptau11m(:,:) * pcosslope(:,:) * psinslope(:,:) * pdircoszw(:,:)**2 &
214 +ptau12m(:,:) * (pcosslope(:,:)**2 - psinslope(:,:)**2) * &
215 pdircoszw(:,:)**2 &
216 -ptau22m(:,:) * pcosslope(:,:) * psinslope(:,:) &
217 +ptau33m(:,:) * pcosslope(:,:) * psinslope(:,:) * zdirsinzw(:,:)**2 &
218 -pcdueff(:,:) * ( &
219 2. * puslopem(:,:) * pcosslope(:,:) * psinslope(:,:) * &
220 pdircoszw(:,:) * zdirsinzw(:,:) &
221 +pvslopem(:,:) * (pcosslope(:,:)**2 - psinslope(:,:)**2) * zdirsinzw(:,:) &
222 )
223!
224!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
225!$acc end kernels
226!
227zflx(:,:,ikb-1) = 2. * mxm( mym( zflx(:,:,ikb-1) ) ) &
228 - zflx(:,:,ikb)
229!
230! stores <U V>
231IF ( tpfile%LOPENED .AND. turbn%LTURB_FLX ) THEN
232 tzfield = tfieldmetadata( &
233 cmnhname = 'UV_FLX', &
234 cstdname = '', &
235 clongname = 'UV_FLX', &
236 cunits = 'm2 s-2', &
237 cdir = 'XY', &
238 ccomment = 'X_Y_Z_UV_FLX', &
239 ngrid = 5, &
240 ntype = typereal, &
241 ndims = 3, &
242 ltimedep = .true. )
243!$acc update self(ZFLX)
244 CALL io_field_write(tpfile,tzfield,zflx)
245END IF
246!
247!
248!
249!computation of the source for rho*V due to this flux
250IF (.NOT. oflat) THEN
251 prus(:,:,:) = prus(:,:,:) &
252 - dyf(zflx * mxm(mym(prhodj) * pinv_pdyy) ) &
253 + dzf( myf( mzm(zflx)*mxm(pdzy/mzm(pdyy))) &
254 * mxm(pmzm_prhodj * pinv_pdzz) )
255ELSE
256 prus(:,:,:) = prus(:,:,:) - dyf(zflx * mxm(mym(prhodj) * pinv_pdyy) )
257END IF
258!
259!computation of the source for rho*V due to this flux
260IF (.NOT. oflat) THEN
261 prvs(:,:,:) = prvs(:,:,:) &
262 - dxf(zflx * mym(mxm(prhodj) * pinv_pdxx) ) &
263 + dzf( mxf( mzm(zflx)*mym(pdzx/mzm(pdxx))) &
264 * mym(pmzm_prhodj * pinv_pdzz) )
265ELSE
266 prvs(:,:,:) = prvs(:,:,:) - dxf(zflx * mym(mxm(prhodj) * pinv_pdxx) )
267END IF
268!
269IF (ksplt==1) THEN
270 !
271 !Contribution to the dynamic production of TKE:
272 !
273 IF (.NOT. o2d) THEN
274 zwork(:,:,:) = - mxf( myf( zflx * &
275 (gy_u_uv_pum + gx_v_uv_pvm) ) )
276 ELSE
277 zwork(:,:,:) = - mxf( myf( zflx * &
278 (gx_v_uv_pvm) ) )
279 ENDIF
280 !
281 ! evaluate the dynamic production at w(IKB+1) in PDP(IKB)
282 !
283 zwork(:,:,ikb) = - &
284 mxf( myf( 0.5 * (zflx(:,:,ikb+1)+zflx(:,:,ikb)) ) ) &
285 *(mxf( myf( &
286 dym( 0.5 * (pum(:,:,ikb+1)+pum(:,:,ikb)) ) &
287 / mxm( 0.5*(pdyy(:,:,ikb)+pdyy(:,:,ikb+1)) ) &
288 +dxm( 0.5 * (pvm(:,:,ikb+1)+pvm(:,:,ikb)) ) &
289 / mym( 0.5*(pdxx(:,:,ikb)+pdxx(:,:,ikb+1)) ) &
290 ) ) &
291 -mxf( (pum(:,:,ikb+1)-pum(:,:,ikb)) / &
292 mxm(pdzz(:,:,ikb+1)) * pdzy(:,:,ikb+1) &
293 ) / myf(mxm( 0.5*(pdyy(:,:,ikb)+pdyy(:,:,ikb+1)) ) )&
294 -myf( (pvm(:,:,ikb+1)-pvm(:,:,ikb)) / &
295 mym(pdzz(:,:,ikb+1)) * pdzx(:,:,ikb+1) &
296 ) / mxf(mym( 0.5*(pdxx(:,:,ikb)+pdxx(:,:,ikb+1)) ) )&
297 )
298 !
299 ! dynamic production
300 !$acc kernels present_crm(PDP)
301 !$mnh_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
302 pdp(:,:,:) = pdp(:,:,:) + zwork(:,:,:)
303 !$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT,JK=1:IKT)
304 !$acc end kernels
305 !
306END IF
307!
308! Storage in the LES configuration
309!
310IF (tles%LLES_CALL .AND. ksplt==1) THEN
311 CALL second_mnh(ztime1)
312 CALL les_mean_subgrid( mxf(myf(zflx)), tles%X_LES_SUBGRID_UV )
313 CALL les_mean_subgrid( mxf(myf(gy_u_uv(pum,pdyy,pdzz,pdzy)*zflx)), tles%X_LES_RES_ddxa_U_SBG_UaU , .true.)
314 CALL les_mean_subgrid( mxf(myf(gx_v_uv(pvm,pdxx,pdzz,pdzx)*zflx)), tles%X_LES_RES_ddxa_V_SBG_UaV , .true.)
315 CALL second_mnh(ztime2)
316 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
317END IF
318!
319!
320END SUBROUTINE turb_hor_uv
