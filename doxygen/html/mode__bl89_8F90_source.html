14!!

16!! -------

18!! formula.
19!!

43!! and optimization

55!* 0. DECLARATIONS

58USE modd_cst, ONLY: cst_t
59USE modd_cturb, ONLY: csturb_t
60USE modd_turb_n, ONLY: turb_t
61USE modd_dimphyex, ONLY: dimphyex_t
62USE modd_precision, ONLY: mnhreal
63!
64!
65IMPLICIT NONE
66!
67!* 0.1 Declaration of arguments
68! ------------------------
69!
70TYPE(dimphyex_t), INTENT(IN) :: D
71TYPE(cst_t), INTENT(IN) :: CST
72TYPE(csturb_t), INTENT(IN) :: CSTURB
73TYPE(turb_t), INTENT(IN) :: TURBN
74REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PZZ
75REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PDZZ
76REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PTHVREF
77REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PTHLM ! conservative pot. temp.
78INTEGER, INTENT(IN) :: KRR
79REAL, DIMENSION(D%NIJT,D%NKT,KRR), INTENT(IN),TARGET :: PRM ! water var.
80REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PTKEM ! TKE
81REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PSHEAR
82REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT),TARGET :: PLM ! Mixing length
83LOGICAL, INTENT(IN) :: OOCEAN ! switch for Ocean model version
84! thermodynamical variables PTHLM=Theta at the begining
85!
86!* 0.2 Declaration of local variables
87! ------------------------------
88!
89INTEGER :: IKT,IKB,IKA,IKU
90
91REAL, DIMENSION(D%NIJT,D%NKT) :: ZVPT ! Virtual Potential Temp at half levels
92REAL, DIMENSION(D%NIJT,D%NKT) :: ZDELTVPT
93 ! Increment of Virtual Potential Temp between two following levels
94REAL, DIMENSION(D%NIJT,D%NKT) :: ZHLVPT
95 ! Virtual Potential Temp at half levels
96REAL, DIMENSION(D%NIJT,D%NKT) :: ZLWORK
97REAL, DIMENSION(D%NIJT) :: ZINTE
98! ! downwards then upwards vertical displacement,
99! ! residual internal energy,
100! ! residual potential energy
101! ! input and output arrays packed according one horizontal coord.
102! ! input array packed according one horizontal coord.
103REAL, DIMENSION(D%NIJT,D%NKT) :: ZSUM ! to replace SUM function
104REAL, DIMENSION(D%NIJT,D%NKT) :: ZG_O_THVREF
105REAL, DIMENSION(D%NIJT,D%NKT) :: ZSQRT_TKE
106REAL, DIMENSION(D%NIJT,D%NKT) :: PLMDN
107!
108INTEGER :: IIJB, IIJE
109INTEGER :: IKTB, IKTE, IKE,IKL
110INTEGER :: JIJ ! horizontal loop counter
111INTEGER :: JK,JKK ! loop counters
112INTEGER :: JRR ! moist loop counter
113REAL :: ZRVORD ! Rv/Rd
114REAL :: ZPOTE,ZLWORK1,ZLWORK2
115REAL :: ZTEST,ZTEST0,ZTESTM ! test for vectorization
116!-------------------------------------------------------------------------------
117!
118REAL(KIND=jphook) :: zhook_handle
119IF (lhook) CALL dr_hook('BL89',0,zhook_handle)
120!
121zrvord = cst%XRV / cst%XRD
122!
123iijb=d%NIJB
124iije=d%NIJE
125iktb=d%NKTB
126ikte=d%NKTE
127ikt=d%NKT
128ikb=d%NKB
129ike=d%NKE
130ika=d%NKA
131iku=d%NKU
132ikl=d%NKL
133!-------------------------------------------------------------------------------
134!
135!* 1. pack the horizontal dimensions into one
136! ---------------------------------------
137!
138! Pointer remapping instead of RESHAPE (contiguous memory)
139! 2D array => 3D array
140!
141!$acc kernels
142IF (oocean) THEN
143!$acc loop independent collapse(2)
144 DO jk=1,ikt
145 DO jij=iijb,iije
146 zg_o_thvref(jij,jk) = cst%XG * cst%XALPHAOC
147 END DO
148 END DO
149ELSE !Atmosphere case
150!$acc loop independent collapse(2)
151 DO jk=1,ikt
152 DO jij=iijb,iije
153 zg_o_thvref(jij,jk) = cst%XG / pthvref(jij,jk)
154 END DO
155 END DO
156END IF
157!$acc end kernels
158!
159!$acc kernels
160!$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
161zsqrt_tke(iijb:iije,1:ikt) = sqrt(ptkem(iijb:iije,1:ikt))
162!$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
163!
164!$acc end kernels
165!-------------------------------------------------------------------------------
166!
167!* 2. Virtual potential temperature on the model grid
168! -----------------------------------------------
169!
170 !$acc kernels
171IF(krr /= 0) THEN
172 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
173 zsum(iijb:iije,1:ikt) = 0.
174 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
175 DO jrr=1,krr
176 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
177 zsum(iijb:iije,1:ikt) = zsum(iijb:iije,1:ikt)+prm(iijb:iije,1:ikt,jrr)
178 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
179 ENDDO
180 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
181 zvpt(iijb:iije,1:ikt)=pthlm(iijb:iije,1:ikt) * ( 1. + zrvord*prm(iijb:iije,1:ikt,1) ) &
182 / ( 1. + zsum(iijb:iije,1:ikt) )
183 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
184ELSE
185 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
186 zvpt(iijb:iije,1:ikt)=pthlm(iijb:iije,1:ikt)
187 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
188END IF
189!$acc end kernels
190!
191!!!!!!!!!!!!
192!!!!!!!!!!!!
193!!!!!!!!!!!!!!!!!! WARNING !!
194!!!!!!!!!!!!
195!!!!!!!!!!!!
196!Any modification done to the following lines and to the sections 4 and
197!6 must be copied in compute_bl89_ml routine.
198!We do not call directly this routine for numerical performance reasons
199!but algorithm must remain the same.
200!!!!!!!!!!!!
201!
202!$acc kernels present_cr(ZLWORK,ZINTE)
203!$acc loop independent collapse(2)
204DO jk=iktb,ikte
205 DO jij=iijb,iije
206 zdeltvpt(jij,jk) = zvpt(jij,jk) - zvpt(jij,jk-ikl)
207 zhlvpt(jij,jk) = 0.5 * ( zvpt(jij,jk) + zvpt(jij,jk-ikl) )
208 END DO
209END DO
210!
211!$acc loop independent
212DO jij=iijb,iije
213 zdeltvpt(jij,iku) = zvpt(jij,iku) - zvpt(jij,iku-ikl)
214 zdeltvpt(jij,ika) = 0.
215 zhlvpt(jij,iku) = 0.5 * ( zvpt(jij,iku) + zvpt(jij,iku-ikl) )
216 zhlvpt(jij,ika) = zvpt(jij,ika)
217END DO
218!$acc end kernels
219!
220!$acc kernels
221!$acc loop independent collapse(2)
222DO jk=1,ikt
223 DO jij=iijb,iije
224 IF(abs(zdeltvpt(jij,jk))<csturb%XLINF) THEN
225 zdeltvpt(jij,jk)=csturb%XLINF
226 END IF
227 END DO
228END DO
229!$acc end kernels
230!
231!-------------------------------------------------------------------------------
232!
233!* 3. loop on model levels
234! --------------------
235!
236DO jk=iktb,ikte
237!Remark create(ZTESTM) to force the NVIDIA compiler 23.5 to manage correctly ZTESTM (implicit reduction)
238!$acc kernels create(ZTESTM)
239!
240!-------------------------------------------------------------------------------
241!
242!* 4. mixing length for a downwards displacement
243! ------------------------------------------
244 zinte(iijb:iije)=ptkem(iijb:iije,jk)
245 zlwork=0.
246 ztestm=1.
247!$acc loop seq
248 DO jkk=jk,ikb,-ikl
249 IF(ztestm > 0.) THEN
250 ztestm=0.
251 !!!$acc loop independent
252 !DO CONCURRENT(JIJ = IIBJ:IIJE)
253 DO jij=iijb,iije
254 ztest0=0.5+sign(0.5,zinte(jij))
255 !--------- SHEAR + STABILITY -----------
256 zpote = ztest0* &
257 (-zg_o_thvref(jij,jk)*(zhlvpt(jij,jkk)-zvpt(jij,jk)) &
258 + csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
259 )*pdzz(jij,jkk)
260
261 ztest =0.5+sign(0.5,zinte(jij)-zpote)
262 ztestm=ztestm+ztest0
263 zlwork1=pdzz(jij,jkk)
264
265 !--------- SHEAR + STABILITY -----------
266 zlwork2 = (zg_o_thvref(jij,jk) *(zvpt(jij,jkk) - zvpt(jij,jk)) &
267 -csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
268 + sqrt(abs( (csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
269 + ( -zg_o_thvref(jij,jk) * (zvpt(jij,jkk) - zvpt(jij,jk)) ))**2.0 + &
270 2. * zinte(jij) * &
271 (zg_o_thvref(jij,jk) * zdeltvpt(jij,jkk)/ pdzz(jij,jkk))))) / &
272 (zg_o_thvref(jij,jk) * zdeltvpt(jij,jkk) / pdzz(jij,jkk))
273 zlwork(jij,jk)=zlwork(jij,jk)+ztest0*(ztest*zlwork1+(1-ztest)*zlwork2)
274 zinte(jij) = zinte(jij) - zpote
275 END DO
276 ENDIF
277 END DO
278!-------------------------------------------------------------------------------
279!
280!* 5. intermediate storage of the final mixing length
281! -----------------------------------------------
282!
283 !$acc loop independent
284 DO jij=iijb,iije
285 plmdn(jij,jk)=min(zlwork(jij,jk),0.5*(pzz(jij,jk)+pzz(jij,jk+ikl))-pzz(jij,ikb))
286 END DO
287!
288!-------------------------------------------------------------------------------
289!
290!* 6. mixing length for an upwards displacement
291! -----------------------------------------
292!
293 zinte(iijb:iije)=ptkem(iijb:iije,jk)
294 zlwork(iijb:iije,jk)=0.
295 ztestm=1.
296!
297!$acc loop seq
298 DO jkk=jk+ikl,ike,ikl
299 IF(ztestm > 0.) THEN
300 ztestm=0.
301 !!!$acc loop independent
302 !DO CONCURRENT(JIJ = IIBJ:IIJE)
303 DO jij=iijb,iije
304 ztest0=0.5+sign(0.5,zinte(jij))
305 !--------- SHEAR + STABILITY -----------
306 zpote = ztest0* &
307 (zg_o_thvref(jij,jk)*(zhlvpt(jij,jkk)-zvpt(jij,jk)) &
308 +csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
309 )*pdzz(jij,jkk)
310 ztest =0.5+sign(0.5,zinte(jij)-zpote)
311 ztestm=ztestm+ztest0
312 zlwork1=pdzz(jij,jkk)
313 !--------- SHEAR + STABILITY -----------
314 zlwork2= ( - zg_o_thvref(jij,jk) *(zvpt(jij,jkk-ikl) - zvpt(jij,jk) ) &
315 - csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
316 + sqrt(abs( &
317 (csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
318 + ( zg_o_thvref(jij,jk) * (zvpt(jij,jkk-ikl) - zvpt(jij,jk))) )**2 &
319 + 2. * zinte(jij) * &
320 (zg_o_thvref(jij,jk)* zdeltvpt(jij,jkk)/pdzz(jij,jkk))))) / &
321 (zg_o_thvref(jij,jk) * zdeltvpt(jij,jkk) / pdzz(jij,jkk))
322 zlwork(jij,jk)=zlwork(jij,jk)+ztest0*(ztest*zlwork1+(1-ztest)*zlwork2)
323 zinte(jij) = zinte(jij) - zpote
324 END DO
325 ENDIF
326 END DO
327!
328!* Maximal length between JK and the levels below (as in ARPEGE)
329!
330 IF (turbn%LBL89TOP) THEN
331 DO jij=iijb,iije
332 zlwork(jij,jk) = max(zlwork(jij,jk),zlwork(jij,jk-ikl)-pdzz(jij,jk))
333 ENDDO
334 END IF
335
336!
337!-------------------------------------------------------------------------------
338!
339!* 7. final mixing length
340!
341!$acc loop independent
342 DO jij=iijb,iije
343 zlwork1=max(plmdn(jij,jk),1.e-10_mnhreal)
344 zlwork2=max(zlwork(jij,jk),1.e-10_mnhreal)
345 zpote = zlwork1 / zlwork2
346 zlwork2=1.d0 + zpote**turbn%XBL89EXP
347 plm(jij,jk) = zlwork1*(2./zlwork2)**turbn%XUSRBL89
348 plm(jij,jk)=max(plm(jij,jk),turbn%XLINI)
349 END DO
350
351!-------------------------------------------------------------------------------
352!* 8. end of the loop on the vertical levels
353! --------------------------------------
354!
355!$acc end kernels
356END DO
357!
358!-------------------------------------------------------------------------------
359!
360!* 9. boundaries
361! ----------
362!
363!$acc kernels
364plm(iijb:iije,ika)=plm(iijb:iije,ikb)
365plm(iijb:iije,ike)=plm(iijb:iije,ike-ikl)
366plm(iijb:iije,iku)=plm(iijb:iije,ike-ikl)
367!
368!$acc end kernels
369!-------------------------------------------------------------------------------
370!
371!* 10. retrieve output array in model coordinates
372! ------------------------------------------
373! Not needed anymore because of the use of Pointer remapping (see 1.)
374! PLM (3D array) is the target of PLM (2D array) in a contiguous way
375!
376IF (lhook) CALL dr_hook('BL89',1,zhook_handle)
377END SUBROUTINE bl89
378END MODULE mode_bl89
