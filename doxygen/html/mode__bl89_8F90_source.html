54 !
55 !* 0. DECLARATIONS
56 ! ------------
57 !
58 USE modd_cst, ONLY: cst_t
59 USE modd_cturb, ONLY: csturb_t
60 USE modd_turb_n, ONLY: turb_t
61 USE modd_dimphyex, ONLY: dimphyex_t
62 USE modd_precision, ONLY: mnhreal
63 !
64 !
65 IMPLICIT NONE
66 !
67 !* 0.1 Declaration of arguments
68 ! ------------------------
69 !
70 TYPE(dimphyex_t), INTENT(IN) :: D
71 TYPE(cst_t), INTENT(IN) :: CST
72 TYPE(csturb_t), INTENT(IN) :: CSTURB
73 TYPE(turb_t), INTENT(IN) :: TURBN
74 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PZZ
75 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PDZZ
76 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PTHVREF
77 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PTHLM ! conservative pot. temp.
78 INTEGER, INTENT(IN) :: KRR
79 REAL, DIMENSION(D%NIJT,D%NKT,KRR), INTENT(IN),TARGET :: PRM ! water var.
80 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PTKEM ! TKE
81 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN),TARGET :: PSHEAR
82 REAL, DIMENSION(D%NIJT,D%NKT), INTENT(OUT),TARGET :: PLM ! Mixing length
83 LOGICAL, INTENT(IN) :: OOCEAN ! switch for Ocean model version
84 ! thermodynamical variables PTHLM=Theta at the begining
85 !
86 !* 0.2 Declaration of local variables
87 ! ------------------------------
88 !
89 INTEGER :: IKT,IKB,IKA,IKU
90 
91 REAL, DIMENSION(D%NIJT,D%NKT) :: ZVPT ! Virtual Potential Temp at half levels
92 REAL, DIMENSION(D%NIJT,D%NKT) :: ZDELTVPT
93  ! Increment of Virtual Potential Temp between two following levels
94 REAL, DIMENSION(D%NIJT,D%NKT) :: ZHLVPT
95  ! Virtual Potential Temp at half levels
96 REAL, DIMENSION(D%NIJT) :: ZLWORK,ZINTE
97 ! ! downwards then upwards vertical displacement,
98 ! ! residual internal energy,
99 ! ! residual potential energy
100 ! ! input and output arrays packed according one horizontal coord.
101 ! ! input array packed according one horizontal coord.
102 REAL, DIMENSION(D%NIJT,D%NKT) :: ZSUM ! to replace SUM function
103 REAL, DIMENSION(D%NIJT,D%NKT) :: ZG_O_THVREF
104 REAL, DIMENSION(D%NIJT,D%NKT) :: ZSQRT_TKE
105 REAL, DIMENSION(D%NIJT,D%NKT) :: PLMDN
106 !
107 INTEGER :: IIJB, IIJE
108 INTEGER :: IKTB, IKTE, IKE,IKL
109 INTEGER :: JIJ ! horizontal loop counter
110 INTEGER :: JK,JKK ! loop counters
111 INTEGER :: JRR ! moist loop counter
112 REAL :: ZRVORD ! Rv/Rd
113 REAL :: ZPOTE,ZLWORK1,ZLWORK2
114 REAL :: ZTEST,ZTEST0,ZTESTM ! test for vectorization
115 REAL :: Z2SQRT2
116 !-------------------------------------------------------------------------------
117 !
118 REAL(KIND=jphook) :: zhook_handle
119 IF (lhook) CALL dr_hook('BL89',0,zhook_handle)
120 z2sqrt2=2.*sqrt(2.)
121 !
122 zrvord = cst%XRV / cst%XRD
123 !
124 iijb=d%NIJB
125 iije=d%NIJE
126 iktb=d%NKTB
127 ikte=d%NKTE
128 ikt=d%NKT
129 ikb=d%NKB
130 ike=d%NKE
131 ika=d%NKA
132 iku=d%NKU
133 ikl=d%NKL
134 !-------------------------------------------------------------------------------
135 !
136 !* 1. pack the horizontal dimensions into one
137 ! ---------------------------------------
138 !
139 ! Pointer remapping instead of RESHAPE (contiguous memory)
140 ! 2D array => 3D array
141 !
142 IF (oocean) THEN
143  DO jk=1,ikt
144  DO jij=iijb,iije
145  zg_o_thvref(jij,jk) = cst%XG * cst%XALPHAOC
146  END DO
147  END DO
148 ELSE !Atmosphere case
149  DO jk=1,ikt
150  DO jij=iijb,iije
151  zg_o_thvref(jij,jk) = cst%XG / pthvref(jij,jk)
152  END DO
153  END DO
154 END IF
155 !
156 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
157 zsqrt_tke(iijb:iije,1:ikt) = sqrt(ptkem(iijb:iije,1:ikt))
158 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
159 !
160 !-------------------------------------------------------------------------------
161 !
162 !* 2. Virtual potential temperature on the model grid
163 ! -----------------------------------------------
164 !
165 IF(krr /= 0) THEN
166  zsum(iijb:iije,1:ikt) = 0.
167  DO jrr=1,krr
168  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
169  zsum(iijb:iije,1:ikt) = zsum(iijb:iije,1:ikt)+prm(iijb:iije,1:ikt,jrr)
170  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
171  ENDDO
172  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
173  zvpt(iijb:iije,1:ikt)=pthlm(iijb:iije,1:ikt) * ( 1. + zrvord*prm(iijb:iije,1:ikt,1) ) &
174  / ( 1. + zsum(iijb:iije,1:ikt) )
175  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
176 ELSE
177  zvpt(iijb:iije,1:ikt)=pthlm(iijb:iije,1:ikt)
178 END IF
179 !
180 !!!!!!!!!!!!
181 !!!!!!!!!!!!
182 !!!!!!!!!!!!!!!!!! WARNING !!
183 !!!!!!!!!!!!
184 !!!!!!!!!!!!
185 !Any modification done to the following lines and to the sections 4 and
186 !6 must be copied in compute_bl89_ml routine.
187 !We do not call directly this routine for numerical performance reasons
188 !but algorithm must remain the same.
189 !!!!!!!!!!!!
190 !
191 DO jk=iktb,ikte
192  DO jij=iijb,iije
193  zdeltvpt(jij,jk) = zvpt(jij,jk) - zvpt(jij,jk-ikl)
194  zhlvpt(jij,jk) = 0.5 * ( zvpt(jij,jk) + zvpt(jij,jk-ikl) )
195  END DO
196 END DO
197 !
198 DO jij=iijb,iije
199  zdeltvpt(jij,iku) = zvpt(jij,iku) - zvpt(jij,iku-ikl)
200  zdeltvpt(jij,ika) = 0.
201  zhlvpt(jij,iku) = 0.5 * ( zvpt(jij,iku) + zvpt(jij,iku-ikl) )
202  zhlvpt(jij,ika) = zvpt(jij,ika)
203 END DO
204 !
205 DO jk=1,ikt
206  DO jij=iijb,iije
207  IF(abs(zdeltvpt(jij,jk))<csturb%XLINF) THEN
208  zdeltvpt(jij,jk)=csturb%XLINF
209  END IF
210  END DO
211 END DO
212 !
213 !-------------------------------------------------------------------------------
214 !
215 !* 3. loop on model levels
216 ! --------------------
217 !
218 DO jk=iktb,ikte
219 !
220 !-------------------------------------------------------------------------------
221 !
222 !* 4. mixing length for a downwards displacement
223 ! ------------------------------------------
224  zinte(iijb:iije)=ptkem(iijb:iije,jk)
225  zlwork=0.
226  ztestm=1.
227  DO jkk=jk,ikb,-ikl
228  IF(ztestm > 0.) THEN
229  ztestm=0.
230  DO jij=iijb,iije
231  ztest0=0.5+sign(0.5,zinte(jij))
232  !--------- SHEAR + STABILITY -----------
233  zpote = ztest0* &
234  (-zg_o_thvref(jij,jk)*(zhlvpt(jij,jkk)-zvpt(jij,jk)) &
235  + csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
236  )*pdzz(jij,jkk)
237 
238  ztest =0.5+sign(0.5,zinte(jij)-zpote)
239  ztestm=ztestm+ztest0
240  zlwork1=pdzz(jij,jkk)
241 
242  !--------- SHEAR + STABILITY -----------
243  zlwork2 = (zg_o_thvref(jij,jk) *(zvpt(jij,jkk) - zvpt(jij,jk)) &
244  -csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
245  + sqrt(abs( (csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
246  + ( -zg_o_thvref(jij,jk) * (zvpt(jij,jkk) - zvpt(jij,jk)) ))**2.0 + &
247  2. * zinte(jij) * &
248  (zg_o_thvref(jij,jk) * zdeltvpt(jij,jkk)/ pdzz(jij,jkk))))) / &
249  (zg_o_thvref(jij,jk) * zdeltvpt(jij,jkk) / pdzz(jij,jkk))
250  zlwork(jij)=zlwork(jij)+ztest0*(ztest*zlwork1+(1-ztest)*zlwork2)
251  zinte(jij) = zinte(jij) - zpote
252  END DO
253  ENDIF
254  END DO
255 !-------------------------------------------------------------------------------
256 !
257 !* 5. intermediate storage of the final mixing length
258 ! -----------------------------------------------
259 !
260  DO jij=iijb,iije
261  plmdn(jij,jk)=min(zlwork(jij),0.5*(pzz(jij,jk)+pzz(jij,jk+ikl))-pzz(jij,ikb))
262  END DO
263 !
264 !-------------------------------------------------------------------------------
265 !
266 !* 6. mixing length for an upwards displacement
267 ! -----------------------------------------
268 !
269  zinte(iijb:iije)=ptkem(iijb:iije,jk)
270  zlwork(iijb:iije)=0.
271  ztestm=1.
272 !
273  DO jkk=jk+ikl,ike,ikl
274  IF(ztestm > 0.) THEN
275  ztestm=0.
276  DO jij=iijb,iije
277  ztest0=0.5+sign(0.5,zinte(jij))
278  !--------- SHEAR + STABILITY -----------
279  zpote = ztest0* &
280  (zg_o_thvref(jij,jk)*(zhlvpt(jij,jkk)-zvpt(jij,jk)) &
281  +csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
282  )*pdzz(jij,jkk)
283  ztest =0.5+sign(0.5,zinte(jij)-zpote)
284  ztestm=ztestm+ztest0
285  zlwork1=pdzz(jij,jkk)
286  !--------- SHEAR + STABILITY -----------
287  zlwork2= ( - zg_o_thvref(jij,jk) *(zvpt(jij,jkk-ikl) - zvpt(jij,jk) ) &
288  - csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
289  + sqrt(abs( &
290  (csturb%XRM17*pshear(jij,jkk)*zsqrt_tke(jij,jk) &
291  + ( zg_o_thvref(jij,jk) * (zvpt(jij,jkk-ikl) - zvpt(jij,jk))) )**2 &
292  + 2. * zinte(jij) * &
293  (zg_o_thvref(jij,jk)* zdeltvpt(jij,jkk)/pdzz(jij,jkk))))) / &
294  (zg_o_thvref(jij,jk) * zdeltvpt(jij,jkk) / pdzz(jij,jkk))
295  zlwork(jij)=zlwork(jij)+ztest0*(ztest*zlwork1+(1-ztest)*zlwork2)
296  zinte(jij) = zinte(jij) - zpote
297  END DO
298  ENDIF
299  END DO
300 !
301 !-------------------------------------------------------------------------------
302 !
303 !* 7. final mixing length
304 !
305  DO jij=iijb,iije
306  zlwork1=max(plmdn(jij,jk),1.e-10_mnhreal)
307  zlwork2=max(zlwork(jij),1.e-10_mnhreal)
308  zpote = zlwork1 / zlwork2
309  zlwork2=1.d0 + zpote**turbn%XBL89EXP
310  plm(jij,jk) = zlwork1*(2./zlwork2)**turbn%XUSRBL89
311  plm(jij,jk)=max(plm(jij,jk),turbn%XLINI)
312  END DO
313 
314 
315 !-------------------------------------------------------------------------------
316 !* 8. end of the loop on the vertical levels
317 ! --------------------------------------
318 !
319 END DO
320 !
321 !-------------------------------------------------------------------------------
322 !
323 !* 9. boundaries
324 ! ----------
325 !
326 plm(iijb:iije,ika)=plm(iijb:iije,ikb)
327 plm(iijb:iije,ike)=plm(iijb:iije,ike-ikl)
328 plm(iijb:iije,iku)=plm(iijb:iije,ike-ikl)
329 !
330 !-------------------------------------------------------------------------------
331 !
332 !* 10. retrieve output array in model coordinates
333 ! ------------------------------------------
334 ! Not needed anymore because of the use of Pointer remapping (see 1.)
335 ! PLM (3D array) is the target of PLM (2D array) in a contiguous way
336 !
337 IF (lhook) CALL dr_hook('BL89',1,zhook_handle)
338 END SUBROUTINE bl89
339 END MODULE mode_bl89
