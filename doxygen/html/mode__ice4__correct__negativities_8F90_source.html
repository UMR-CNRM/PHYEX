48 DO jk = iktb, ikte
49  DO jij = iijb, iije
50  ! 1) deal with negative values for mixing ratio, except for vapor
51  zw =prc(jij,jk)-max(prc(jij,jk), 0.)
52  prv(jij,jk)=prv(jij,jk)+zw
53  pth(jij,jk)=pth(jij,jk)-zw*plvfact(jij,jk)

55 !++cb-- pour l'elec, on peut eventuellement appeler une routine : ca evitera les pb avec xecharge ?
56 ! IF (OELEC .AND. ZW .LT. 0.) THEN
57 ! ZION = PQC(JIJ,JK) / XECHARGE
58 ! ZADD = 0.5 + SIGN(0.5, PQC(JIJ,JK))
59 ! PQPI(JIJ,JK) = PQPI(JIJ,JK) + ZADD * ZION
60 ! PQNI(JIJ,JK) = PQNI(JIJ,JK) + (1. - ZADD) * ZION
61 ! PQC(JIJ,JK) = 0.
62 ! END IF
63 
64  zw =prr(jij,jk)-max(prr(jij,jk), 0.)
65  prv(jij,jk)=prv(jij,jk)+zw
66  pth(jij,jk)=pth(jij,jk)-zw*plvfact(jij,jk)
67  prr(jij,jk)=prr(jij,jk)-zw

69 ! ZION = PQR(JIJ,JK) / XECHARGE
70 ! ZADD = 0.5 + SIGN(0.5, PQR(JIJ,JK))

74 ! END IF
75

77  prv(jij,jk)=prv(jij,jk)+zw

80 ! IF (OELEC .AND. ZW .LT. 0.) THEN
81 ! ZION = PQI(JIJ,JK) / XECHARGE

83 ! PQPI(JIJ,JK) = PQPI(JIJ,JK) + ZADD * ZION

98 ! END IF
99

102  pth(jij,jk)=pth(jij,jk)-zw*plsfact(jij,jk)
103  prg(jij,jk)=prg(jij,jk)-zw

105 ! ZION = PQG(JIJ,JK) / XECHARGE
106 ! ZADD = 0.5 + SIGN(0.5, PQG(JIJ,JK))

108 ! PQNI(JIJ,JK) = PQNI(JIJ,JK) + (1. - ZADD) * ZION
109 ! PQG(JIJ,JK) = 0.
110 ! END IF
111 
112  IF(krr==7) THEN
113  zw =prh(jij,jk)-max(prh(jij,jk), 0.)
114  prv(jij,jk)=prv(jij,jk)+zw
115  pth(jij,jk)=pth(jij,jk)-zw*plsfact(jij,jk)
116  prh(jij,jk)=prh(jij,jk)-zw
117 ! IF (OELEC .AND. ZW .LT. 0.) THEN
118 ! ZION = PQH(JIJ,JK) / XECHARGE

123 ! END IF
124  ENDIF

126  ! 2) deal with negative vapor mixing ratio
127 
128  ! for rc and ri, we keep ice fraction constant
129  zw=min(1., max(iced%XRTMIN(1)-prv(jij,jk), 0.) / &
130  &max(prc(jij,jk)+pri(jij,jk), 1.e-20)) ! Proportion of rc+ri to convert into rv

144 ! ZADD = 0.5 + SIGN(0.5, PQG(JIJ,JK))

165  prv(jij,jk)=prv(jij,jk)+zw
166  prh(jij,jk)=prh(jij,jk)-zw
167  pth(jij,jk)=pth(jij,jk)-zw*plsfact(jij,jk)
168  ENDIF
169  ENDDO
170 ENDDO
