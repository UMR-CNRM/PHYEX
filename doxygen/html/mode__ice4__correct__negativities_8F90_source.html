25TYPE(dimphyex_t), INTENT(IN) :: D

27INTEGER, INTENT(IN) :: KRR
28!LOGICAL, INTENT(IN) :: OELEC
29REAL, DIMENSION(D%NIJT, D%NKT), INTENT(INOUT) :: PRV, PRC, PRR, PRI, PRS, PRG, PTH

33!REAL, DIMENSION(D%NIJT, D%NKT), OPTIONAL, INTENT(INOUT) :: PQH
34!
35REAL :: ZW
36!REAL :: ZION, ZADD
37INTEGER :: JIJ, JK, IKTB, IKTE, IIJB, IIJE
38
39REAL(KIND=jphook) :: zhook_handle
40!
41IF (lhook) CALL dr_hook('ICE4_CORRECT_NEGATIVITIES', 0, zhook_handle)
42!
43iktb=d%NKTB
44ikte=d%NKTE
45iijb=d%NIJB
46iije=d%NIJE
47!
48!We correct negativities with conservation
49!$acc kernels
50!$acc loop independent
51DO jk = iktb, ikte
52 DO jij = iijb, iije
53 ! 1) deal with negative values for mixing ratio, except for vapor
54 zw =prc(jij,jk)-max(prc(jij,jk), 0.)
55 prv(jij,jk)=prv(jij,jk)+zw
56 pth(jij,jk)=pth(jij,jk)-zw*plvfact(jij,jk)
57 prc(jij,jk)=prc(jij,jk)-zw
58!++cb-- pour l'elec, on peut eventuellement appeler une routine : ca evitera les pb avec xecharge ?
59! IF (OELEC .AND. ZW .LT. 0.) THEN
60! ZION = PQC(JIJ,JK) / XECHARGE
61! ZADD = 0.5 + SIGN(0.5, PQC(JIJ,JK))
62! PQPI(JIJ,JK) = PQPI(JIJ,JK) + ZADD * ZION
63! PQNI(JIJ,JK) = PQNI(JIJ,JK) + (1. - ZADD) * ZION
64! PQC(JIJ,JK) = 0.
65! END IF
66
67 zw =prr(jij,jk)-max(prr(jij,jk), 0.)
68 prv(jij,jk)=prv(jij,jk)+zw
69 pth(jij,jk)=pth(jij,jk)-zw*plvfact(jij,jk)
70 prr(jij,jk)=prr(jij,jk)-zw
71! IF (OELEC .AND. ZW .LT. 0.) THEN
72! ZION = PQR(JIJ,JK) / XECHARGE
73! ZADD = 0.5 + SIGN(0.5, PQR(JIJ,JK))
74! PQPI(JIJ,JK) = PQPI(JIJ,JK) + ZADD * ZION
75! PQNI(JIJ,JK) = PQNI(JIJ,JK) + (1. - ZADD) * ZION
76! PQR(JIJ,JK) = 0.
77! END IF
78
79 zw =pri(jij,jk)-max(pri(jij,jk), 0.)
80 prv(jij,jk)=prv(jij,jk)+zw
81 pth(jij,jk)=pth(jij,jk)-zw*plsfact(jij,jk)
82 pri(jij,jk)=pri(jij,jk)-zw
83! IF (OELEC .AND. ZW .LT. 0.) THEN
84! ZION = PQI(JIJ,JK) / XECHARGE
85! ZADD = 0.5 + SIGN(0.5, PQI(JIJ,JK))
86! PQPI(JIJ,JK) = PQPI(JIJ,JK) + ZADD * ZION
87! PQNI(JIJ,JK) = PQNI(JIJ,JK) + (1. - ZADD) * ZION
88! PQI(JIJ,JK) = 0.

94 prs(jij,jk)=prs(jij,jk)-zw
95! IF (OELEC .AND. ZW .LT. 0.) THEN
96! ZION = PQS(JIJ,JK) / XECHARGE
97! ZADD = 0.5 + SIGN(0.5, PQS(JIJ,JK))
98! PQPI(JIJ,JK) = PQPI(JIJ,JK) + ZADD * ZION
99! PQNI(JIJ,JK) = PQNI(JIJ,JK) + (1. - ZADD) * ZION
100! PQS(JIJ,JK) = 0.
101! END IF
102
103 zw =prg(jij,jk)-max(prg(jij,jk), 0.)
104 prv(jij,jk)=prv(jij,jk)+zw
105 pth(jij,jk)=pth(jij,jk)-zw*plsfact(jij,jk)
106 prg(jij,jk)=prg(jij,jk)-zw
107! IF (OELEC .AND. ZW .LT. 0.) THEN
108! ZION = PQG(JIJ,JK) / XECHARGE
109! ZADD = 0.5 + SIGN(0.5, PQG(JIJ,JK))
110! PQPI(JIJ,JK) = PQPI(JIJ,JK) + ZADD * ZION
111! PQNI(JIJ,JK) = PQNI(JIJ,JK) + (1. - ZADD) * ZION
112! PQG(JIJ,JK) = 0.
113! END IF
114
115 IF(krr==7) THEN
116 zw =prh(jij,jk)-max(prh(jij,jk), 0.)
117 prv(jij,jk)=prv(jij,jk)+zw
118 pth(jij,jk)=pth(jij,jk)-zw*plsfact(jij,jk)
119 prh(jij,jk)=prh(jij,jk)-zw
120! IF (OELEC .AND. ZW .LT. 0.) THEN
121! ZION = PQH(JIJ,JK) / XECHARGE
122! ZADD = 0.5 + SIGN(0.5, PQH(JIJ,JK))
123! PQPI(JIJ,JK) = PQPI(JIJ,JK) + ZADD * ZION
124! PQNI(JIJ,JK) = PQNI(JIJ,JK) + (1. - ZADD) * ZION
125! PQH(JIJ,JK) = 0.
126! END IF
127 ENDIF
128
129 ! 2) deal with negative vapor mixing ratio
130
131 ! for rc and ri, we keep ice fraction constant
132 zw=min(1., max(iced%XRTMIN(1)-prv(jij,jk), 0.) / &
133 &max(prc(jij,jk)+pri(jij,jk), 1.e-20)) ! Proportion of rc+ri to convert into rv
134 pth(jij,jk)=pth(jij,jk)-zw* &
135 &(prc(jij,jk)*plvfact(jij,jk)+pri(jij,jk)*plsfact(jij,jk))
136 prv(jij,jk)=prv(jij,jk)+zw*(prc(jij,jk)+pri(jij,jk))
137 prc(jij,jk)=(1.-zw)*prc(jij,jk)
138 pri(jij,jk)=(1.-zw)*pri(jij,jk)
139
140 zw=min(max(prr(jij,jk), 0.), &
141 &max(iced%XRTMIN(1)-prv(jij,jk), 0.)) ! Quantity of rr to convert into rv
142 prv(jij,jk)=prv(jij,jk)+zw
143 prr(jij,jk)=prr(jij,jk)-zw
144 pth(jij,jk)=pth(jij,jk)-zw*plvfact(jij,jk)
145! IF (OELEC .AND. ZW .GT. 0.) THEN
146! ZION = PQR(JIJ,JK) / XECHARGE
147! ZADD = 0.5 + SIGN(0.5, PQG(JIJ,JK))
148! PQPI(JIJ,JK) = PQPI(JIJ,JK) + ZADD * ZION
149! PQNI(JIJ,JK) = PQNI(JIJ,JK) + (1. - ZADD) * ZION
150! PQR(JIJ,JK) = 0.
151! END IF
152
153 zw=min(max(prs(jij,jk), 0.), &
154 &max(iced%XRTMIN(1)-prv(jij,jk), 0.)) ! Quantity of rs to convert into rv
155 prv(jij,jk)=prv(jij,jk)+zw
156 prs(jij,jk)=prs(jij,jk)-zw
157 pth(jij,jk)=pth(jij,jk)-zw*plsfact(jij,jk)
158
159 zw=min(max(prg(jij,jk), 0.), &
160 &max(iced%XRTMIN(1)-prv(jij,jk), 0.)) ! Quantity of rg to convert into rv
161 prv(jij,jk)=prv(jij,jk)+zw
162 prg(jij,jk)=prg(jij,jk)-zw
163 pth(jij,jk)=pth(jij,jk)-zw*plsfact(jij,jk)
164
165 IF(krr==7) THEN
166 zw=min(max(prh(jij,jk), 0.), &
167 &max(iced%XRTMIN(1)-prv(jij,jk), 0.)) ! Quantity of rh to convert into rv
168 prv(jij,jk)=prv(jij,jk)+zw
169 prh(jij,jk)=prh(jij,jk)-zw
170 pth(jij,jk)=pth(jij,jk)-zw*plsfact(jij,jk)
171 ENDIF
172 ENDDO
173ENDDO
174!$acc end kernels
175!
176IF (lhook) CALL dr_hook('ICE4_CORRECT_NEGATIVITIES', 1, zhook_handle)
177!
178END SUBROUTINE ice4_correct_negativities
