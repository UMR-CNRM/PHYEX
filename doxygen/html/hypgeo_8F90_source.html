24 FUNCTION hypgeo(PA,PB,PC,PF,PX) RESULT(PHYPGEO)
25! #############################################

27!
28!!**** *HYPGEO* - hypergeometric function
29!!
30!!
31!! PURPOSE
32!! -------
33! The purpose of this function is to compute the hypergeometric
34!! function of its argument.
35!!
36!!
37!! A*B (A+1)A*(B+1)B X^2
38!! HYPGEO(A,B,C,X)= 1 + ----- * X + ------------- * --- + ... +
39!! C (C+1)C 2
40!!
41!! (A+n)...A*(B+n)...B X^n
42!! --------------------- * ----- + ... ...
43!! (C+n)...C n!
44!!
45!!** METHOD
46!! ------
47!!
48!! EXTERNAL
49!! --------
50!! HYPSER
51!!
52!! IMPLICIT ARGUMENTS
53!! ------------------
54!!
55!! REFERENCE
56!! ---------
57!! Press, Teukolsky, Vetterling and Flannery: Numerical Recipes, 272
58!!
59!!
60!! AUTHOR
61!! ------
62!! Jean-Martial Cohard *LA/OMP*
63!!
64!! MODIFICATIONS
65!! -------------
66!! Original 31/12/96
67!
68!------------------------------------------------------------------------------
69!
70!* 0. DECLARATIONS
71!
72!
73USE modi_gamma
74USE modi_hypser
75!
76IMPLICIT NONE
77!
78!* 0.1 declarations of arguments and result
79!
80REAL, INTENT(IN) :: pa,pb,pc,pf
81REAL, INTENT(IN) :: px
82REAL :: phypgeo
83!
84!* 0.2 declarations of local variables
85!
86!
87REAL :: zeps,ztemp
88REAL :: zxh
89REAL :: zx0,zx1
90!
91!------------------------------------------------------------------------------
92!
93!
94zeps = 4.e-2
95zxh = pf * px**2.0
96IF (zxh.LT.(1-zeps)) THEN
97 CALL hypser(pa,pb,pc,-zxh,phypgeo)
98ELSE IF (zxh.GT.(1.+zeps)) THEN
99 zxh = 1./zxh
100 CALL hypser(pa,pa-pc+1.,pa-pb+1.,-zxh,phypgeo)
101 phypgeo = phypgeo*zxh**(pa)* &
102 (gamma(pc)*gamma(pb-pa)/(gamma(pb)*gamma(pc-pa)))
103 CALL hypser(pb,pb-pc+1.,pb-pa+1.,-zxh,ztemp)
104 phypgeo = phypgeo+ztemp*zxh**(pb)* &
105 (gamma(pc)*gamma(pa-pb)/(gamma(pa)*gamma(pc-pb)))
106ELSE
107 zx0 = (1.-zeps)
108 zx1 = 1./(1.+zeps)
109 CALL hypser(pa,pa-pc+1.,pa-pb+1.,-zx1,phypgeo)
110 phypgeo = phypgeo*zx1**(pa)* &
111 (gamma(pc)*gamma(pb-pa)/(gamma(pb)*gamma(pc-pa)))
112 CALL hypser(pb,pb-pc+1.,pb-pa+1.,-zx1,ztemp)
113 phypgeo = phypgeo+ztemp*zx1**(pb)* &
114 (gamma(pc)*gamma(pa-pb)/(gamma(pa)*gamma(pc-pb)))
115 CALL hypser(pa,pb,pc,-zx0,ztemp)
116 phypgeo = ztemp + (zxh-zx0)*(phypgeo-ztemp)/(2.*zeps)
117ENDIF
118END
