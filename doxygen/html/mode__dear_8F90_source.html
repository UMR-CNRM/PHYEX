34 USE modd_dimphyex, ONLY: dimphyex_t
35 USE modd_turb_n, ONLY: turb_t

54 REAL, INTENT(IN) :: PDIRCOSZW(D%NIJT)
55 REAL, INTENT(IN) :: PDXX(D%NIJT, D%NKT)
56 REAL, INTENT(IN) :: PDYY(D%NIJT, D%NKT)
57 REAL, INTENT(IN) :: PTHVREF(D%NIJT, D%NKT)
58 REAL, INTENT(IN) :: PSRCT(MERGE(D%NIJT, 0, OCOMPUTE_SRC), MERGE(D%NKT, 0, OCOMPUTE_SRC))
59 REAL, INTENT(IN) :: PRT(D%NIJT, D%NKT, KRR)

65 REAL :: ZALPHA,ZD,ZVAR
66 LOGICAL :: GZD

72 REAL :: ZDRTDZ(D%NIJT, D%NKT) !drt_dz used for computing the stablity criterion
73 REAL :: ZDTHLDZ(D%NIJT, D%NKT) !dtheta_l/dz used for computing the stablity criterion
74
75 REAL(KIND=jphook) :: zhook_handle2
76 INTEGER :: IKB, IKE, IKU, IKA, IKT, IKL
77 INTEGER :: IIJB, IKTB, IIJE, IKTE
78 INTEGER :: JK, JIJ
79 !-------------------------------------------------------------------------------
80 !
81 ! initialize the mixing length with the mesh grid
82 IF (lhook) CALL dr_hook('TURB:DEAR', 0, zhook_handle2)
83 ikt = d%NKT
84 iktb = d%NKTB
85 ikte = d%NKTE
86 ikb = d%NKB
87 ike = d%NKE
88 ika = d%NKA
89 iku = d%NKU
90 ikl = d%NKL
91 iije = d%NIJE
92 iijb = d%NIJB
93 IF (turbn%CTURBDIM /= '1DIM') THEN
94 CALL mxf_phy(d, pdxx, pwork1)
95 IF (.not.o2d) THEN
96 CALL myf_phy(d, pdyy, pwork2)
97 END IF
98 END IF
99 ! 1D turbulence scheme
100!$acc kernels present_cr( PLM )
101!$mnh_expand_array ( JIJ=IIJB:IIJE,JK=IKTB:IKTE )
102 plm(iijb:iije, iktb:ikte) = pzz(iijb:iije, iktb + ikl:ikte + ikl) - pzz(iijb:iije, iktb:ikte)
103!$mnh_end_expand_array ( JIJ=IIJB:IIJE,JK=IKTB:IKTE )
104!$mnh_expand_array ( JIJ=IIJB:IIJE )
105 plm(iijb:iije, iku) = plm(iijb:iije, ike)
106 plm(iijb:iije, ika) = pzz(iijb:iije, ikb) - pzz(iijb:iije, ika)
107!$mnh_end_expand_array ( JIJ=IIJB:IIJE )
108!$acc end kernels
109 !
110 IF (turbn%CTURBDIM /= '1DIM') THEN
111 ! 3D turbulence scheme
112 IF (o2d) THEN
113!$acc kernels present_cr( PLM )
114!$mnh_expand_array ( JIJ=IIJB:IIJE,JK=1:IKT )
115 plm(iijb:iije, 1:ikt) = sqrt(plm(iijb:iije, 1:ikt)*pwork1(iijb:iije, 1:ikt))
116!$mnh_end_expand_array ( JIJ=IIJB:IIJE,JK=1:IKT )
117!$acc end kernels
118 ELSE
119!$acc kernels present_cr( PLM )
120!$mnh_expand_array ( JIJ=IIJB:IIJE,JK=1:IKT )
121 plm(iijb:iije, 1:ikt) = (plm(iijb:iije, 1:ikt)*pwork1(iijb:iije, 1:ikt)*pwork2(iijb:iije, 1:ikt))**(1. / 3.)
122!$mnh_end_expand_array ( JIJ=IIJB:IIJE,JK=1:IKT )
123!$acc end kernels
124 END IF
125 END IF
126 ! compute a mixing length limited by the stability
127 !
128 CALL etheta(d, cst, krr, krri, pthlt, prt, plocpexnm, patheta, psrct, oocean, ocompute_src, zetheta)
129 CALL emoist(d, cst, krr, krri, pthlt, prt, plocpexnm, pamoist, psrct, oocean, zemoist)
130 !
131 IF (krr > 0) THEN
132!$acc kernels
133!$acc loop independent collapse( 2 )
134 DO jk=iktb + 1,ikte - 1
135 DO jij=iijb,iije
136 zdthldz(jij, jk) = 0.5*((pthlt(jij, jk + ikl) - pthlt(jij, jk)) / pdzz(jij, jk + ikl) + (pthlt(jij, jk) - &
137 pthlt(jij,jk-ikl)) / pdzz(jij, jk))
138 zdrtdz(jij, jk) = 0.5*((prt(jij, jk + ikl, 1) - prt(jij, jk, 1)) / pdzz(jij, jk + ikl) + (prt(jij, jk, 1) - &
139 prt(jij,jk-ikl, 1)) / pdzz(jij, jk))
140 END DO
141 END DO
142!$acc end kernels
143
144!$acc kernels
145!$acc loop independent collapse( 2 ) private( ZVAR )
146 DO jk=iktb + 1,ikte - 1
147 DO jij=iijb,iije
148 IF (oocean) THEN
149 zvar = cst%XG*(cst%XALPHAOC*zdthldz(jij, jk) - cst%XBETAOC*zdrtdz(jij, jk))
150 ELSE
151 zvar = cst%XG / pthvref(jij, jk)*(zetheta(jij, jk)*zdthldz(jij, jk) + zemoist(jij, jk)*zdrtdz(jij, jk))
152 END IF
153 !
154 IF (zvar > 0.) THEN
155 plm(jij, jk) = max(cst%XMNH_EPSILON, min(plm(jij, jk), 0.76*sqrt(ptket(jij, jk) / zvar)))
156 END IF
157 END DO
158 END DO
159!$acc end kernels
160
161 ELSE
162 ! For dry atmos or unsalted ocean runs
163!$acc kernels
164!$acc loop independent collapse( 2 ) private( ZVAR )
165 DO jk=iktb + 1,ikte - 1
166 DO jij=iijb,iije
167 zdthldz(jij, jk) = 0.5*((pthlt(jij, jk + ikl) - pthlt(jij, jk)) / pdzz(jij, jk + ikl) + (pthlt(jij, jk) - &
168 pthlt(jij,jk-ikl)) / pdzz(jij, jk))
169 IF (oocean) THEN
170 zvar = cst%XG*cst%XALPHAOC*zdthldz(jij, jk)
171 ELSE
172 zvar = cst%XG / pthvref(jij, jk)*zetheta(jij, jk)*zdthldz(jij, jk)
173 END IF
174 !
175 IF (zvar > 0.) THEN
176 plm(jij, jk) = max(cst%XMNH_EPSILON, min(plm(jij, jk), 0.76*sqrt(ptket(jij, jk) / zvar)))
177 END IF
178 END DO
179 END DO
180!$acc end kernels
181 END IF
182!$acc kernels present( PWORK2D, PLM )
183 ! special case near the surface
184!$mnh_expand_array ( JIJ=IIJB:IIJE )
185 zdthldz(iijb:iije, ikb) = (pthlt(iijb:iije, ikb + ikl) - pthlt(iijb:iije, ikb)) / pdzz(iijb:iije, ikb + ikl)
186!$mnh_end_expand_array ( JIJ=IIJB:IIJE )
187 ! For dry simulations
188 IF (krr > 0) THEN
189!$mnh_expand_array ( JIJ=IIJB:IIJE )
190 zdrtdz(iijb:iije, ikb) = (prt(iijb:iije, ikb + ikl, 1) - prt(iijb:iije, ikb, 1)) / pdzz(iijb:iije, ikb + ikl)
191!$mnh_end_expand_array ( JIJ=IIJB:IIJE )
192 ELSE
193 zdrtdz(:, ikb) = 0
194 END IF
195 !
196 IF (oocean) THEN
197!$mnh_expand_array ( JIJ=IIJB:IIJE )
198 pwork2d(iijb:iije) = cst%XG*(cst%XALPHAOC*zdthldz(iijb:iije, ikb) - cst%XBETAOC*zdrtdz(iijb:iije, ikb))
199!$mnh_end_expand_array ( JIJ=IIJB:IIJE )
200 ELSE
201!$mnh_expand_array ( JIJ=IIJB:IIJE )
202 pwork2d(iijb:iije) = cst%XG / pthvref(iijb:iije, ikb)*(zetheta(iijb:iije, ikb)*zdthldz(iijb:iije, ikb) + &
203 zemoist(iijb:iije,ikb)*zdrtdz(iijb:iije, ikb))
204!$mnh_end_expand_array ( JIJ=IIJB:IIJE )
205 END IF
206!$mnh_expand_where ( JIJ=IIJB:IIJE )
207 WHERE (pwork2d(iijb:iije) > 0.)
208 plm(iijb:iije, ikb) = &
209 & max(cst%XMNH_EPSILON, min(plm(iijb:iije, ikb), 0.76*sqrt(ptket(iijb:iije, ikb) / pwork2d(iijb:iije))))
210 END WHERE
211!$mnh_end_expand_where ( JIJ=IIJB:IIJE )
212 !
213 ! mixing length limited by the distance normal to the surface (with the same factor as for BL89)
214 !
215 IF (.NOT. turbn%LRMC01) THEN
216 zalpha = 0.5**(-1.5)
217 !
218!$acc loop independent private( GZD,ZD )
219 DO jij=iijb,iije
220 gzd = .true.
221 IF (oocean) THEN
222!$acc loop seq
223 DO jk=ikte,iktb,-1
224 zd = zalpha*(pzz(jij, ikte + 1) - pzz(jij, jk))
225 IF (plm(jij, jk) > zd .AND. gzd) THEN
226 plm(jij, jk) = zd
227 ELSE
228 gzd = .false.
229 END IF
230 END DO
231 ELSE
232 DO jk=iktb,ikte
233 zd = zalpha*(0.5*(pzz(jij, jk) + pzz(jij, jk + ikl)) - pzz(jij, ikb))*pdircoszw(jij)
234 IF (plm(jij, jk) > zd .AND. gzd) THEN
235 plm(jij, jk) = zd
236 ELSE
237 gzd = .false.
238 END IF
239 END DO
240 END IF
241 END DO
242 END IF
243 !
244!$mnh_expand_array ( JIJ=IIJB:IIJE )
245 plm(iijb:iije, ika) = plm(iijb:iije, ikb)
246 plm(iijb:iije, ike) = plm(iijb:iije, ike - ikl)
247 plm(iijb:iije, iku) = plm(iijb:iije, iku - ikl)
248!$mnh_end_expand_array ( JIJ=IIJB:IIJE )
249 !
250!$acc end kernels
251 IF (lhook) CALL dr_hook('TURB:DEAR', 1, zhook_handle2)
252END SUBROUTINE dear
