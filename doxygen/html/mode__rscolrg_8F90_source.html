21 !!**** * - Build up a look-up table containing the scaled fall speed
22 !! difference between size distributed particles of the aggregates and Z
23 !!
24 !!
25 !! PURPOSE
26 !! -------
27 !! The purpose of this routine is to integrate numerically the scaled fall
28 !! speed difference between aggregates and rain for use in collection
29 !! kernels. A first integral of the form
30 !!
31 !! infty Dz_max
32 !! / /
33 !! |{| }
34 !! |{| E_xz (Dx+Dz)^2 |cxDx^dx-czDz^dz| Dz^bz n(Dz) dDz} n(Dx) dDx
35 !! |{| }
36 !! / /
37 !! 0 Dz_min
38 !!
39 !! is evaluated and normalised by a second integral of the form
40 !!
41 !! infty
42 !! / /
43 !! |{| }
44 !! |{| (Dx+Dz)^2 Dz^bz n(Dz) dDz} n(Dx) dDx
45 !! |{| }
46 !! / /
47 !! 0
48 !!
49 !! The result is stored in a two-dimensional array.
50 !!
51 !!** METHOD
52 !! ------
53 !! The free parameters of the size distribution function of the aggregates
54 !! and Z (slope parameter LAMBDA) are discretized with a geometrical rate
55 !! in a specific range
56 !! LAMBDA = exp( (Log(LAMBDA_max) - Log(LAMBDA_min))/N_interval )
57 !! The two above integrals are performed using the trapezoidal scheme.
58 !!
59 !! EXTERNAL
60 !! --------
61 !! MODI_GENERAL_GAMMA: Generalized gamma distribution law
62 !!
63 !! IMPLICIT ARGUMENTS
64 !! ------------------
65 !! MODD_CST : XPI,XRHOLW
66 !! MODD_RAIN_ICE_DESCR: XAS,XAS,XBS
67 !!
68 !! REFERENCE
69 !! ---------
70 !! B.S. Ferrier , 1994 : A Double-Moment Multiple-Phase Four-Class
71 !! Bulk Ice Scheme,JAS,51,249-280.

127  ! the mass collection kernel as a
128  ! function of LAMBDAX and LAMBDAZ
129 !

131 !* 0.2 Declarations of local variables
132 ! -------------------------------
133 !
134 !
135 INTEGER :: JLBDAS ! Slope index of the size distribution of the aggregates

168 !
169 !* 1 COMPUTE THE SCALED VELOCITY DIFFERENCE IN THE MASS

171 ! -------------------------------------------------
172 !
173 !
174 !* 1.0 Initialization
175 !
176 REAL(KIND=jphook) :: zhook_handle
177 IF (lhook) CALL dr_hook('RSCOLRG',0,zhook_handle)
178 prscolrg(:,:) = 0.0
179 zcst1 = (3.0/xpi)/xrholw
180 !
181 !* 1.1 Compute the growth rate of the slope factors LAMBDA
182 !
183 zdlbdar = exp( log(plbdarmax/plbdarmin)/real(SIZE(prscolrg(:,:),1)-1) )
184 zdlbdas = exp( log(plbdasmax/plbdasmin)/real(SIZE(prscolrg(:,:),2)-1) )
185 !
186 !* 1.2 Scan the slope factors LAMBDAX and LAMBDAZ
187 !
188 DO jlbdar = 1,SIZE(prscolrg(:,:),1)
189  zlbdar = plbdarmin * zdlbdar ** (jlbdar-1)
190  zdrmax = pdinfty / zlbdar
191 !
192 !* 1.3 Compute the diameter steps
193 !
194  zddscalr = pdinfty / (real(knd) * zlbdar)
195  DO jlbdas = 1,SIZE(prscolrg(:,:),2)
196  zlbdas = plbdasmin * zdlbdas ** (jlbdas-1)
197 !
198 !* 1.4 Initialize the collection integrals
199 !
200  zscalsr = 0.0
201  zcollsr = 0.0
202 !
203 !* 1.5 Compute the diameter steps
204 !
205  zdds = pdinfty / (real(knd) * zlbdas)
206 !
207 !* 1.6 Scan over the diameters DS and DR
208 !
209  DO jds = 1,knd-1
210  zds = zdds * real(jds)
211  zscalr = 0.0
212  zcollr = 0.0
213  DO jdr = 1,knd-1
214  zdr = zddscalr * real(jdr)
215 !
216 !* 1.7 Compute the normalization factor by integration over the
217 ! dimensional spectrum of rain
218 !
219  zscalr = zscalr + (zds+zdr)**2 * general_gamma(palphar,pnur,zlbdar,zdr)
220  END DO
221 !
222 !* 1.8 Compute the scaled fall speed difference by partial
223 ! integration over the dimensional spectrum of rain
224 !
225  zfunc = pag - pas*zds**(pbs-3.0) ! approximate limit is Ds=240 microns
226  IF( zfunc>0.0 ) THEN
227  zdrmin = zds*( zcst1*zfunc )**0.3333333
228  ELSE
229  zdrmin = 0.0
230  END IF
231  IF( zds>1.0e-4 ) THEN ! allow computation if Ds>100 microns
232  ! corresponding to a maximal density of the aggregates of XRHOLW
233  IF( (zdrmax-zdrmin) >= 0.5*zddscalr ) THEN
234  inr = ceiling( (zdrmax-zdrmin)/zddscalr )
235  zddcollr = (zdrmax-zdrmin) / real(inr)
236  DO jdr = 1,inr-1
237  zdr = zddcollr * real(jdr) + zdrmin
238 #ifdef REPRO48
239  zcollr = zcollr + (zds+zdr)**2 &
240  * general_gamma(palphar,pnur,zlbdar,zdr) &
241  * pesr * abs(pfalls*zds**pexfalls-pfallr*zdr**pexfallr)
242 #else
243  zcollr = zcollr + (zds+zdr)**2 &
244  * general_gamma(palphar,pnur,zlbdar,zdr) &
245  * pesr * abs(pfalls*zds**pexfalls*exp(-(zds*pfallexps)**palphas)-pfallr*zdr**pexfallr)
246 #endif
247  END DO
248  IF( zdrmin>0.0 ) THEN
249 #ifdef REPRO48
250  zcolldrmin = (zds+zdrmin)**2 &
251  * general_gamma(palphar,pnur,zlbdar,zdrmin) &
252  * pesr * abs(pfalls*zds**pexfalls-pfallr*zdrmin**pexfallr)
253 #else
254  zcolldrmin = (zds+zdrmin)**2 &
255  * general_gamma(palphar,pnur,zlbdar,zdrmin) &
256  * pesr * abs(pfalls*zds**pexfalls*exp(-(zds*pfallexps)**palphas)-pfallr*zdrmin**pexfallr)
257 #endif
258  ELSE
259  zcolldrmin = 0.0
260  END IF
261  zcollr = (zcollr + 0.5*zcolldrmin)*(zddcollr/zddscalr)
262 !
263 !* 1.9 Compute the normalization factor by integration over the
264 ! dimensional spectrum of the aggregates
265 !
266  zfunc = (zds**pexmasss) * general_gamma(palphas,pznus,zlbdas,zds)
267  zscalsr = zscalsr + zscalr * zfunc
268 !
269 !* 1.10 Compute the scaled fall speed difference by integration over
270 ! the dimensional spectrum of the aggregates
271 !
272  zcollsr = zcollsr + zcollr * zfunc
273 !
274 ! Otherwise ZDRMIN>ZDRMAX so PRRCOLSS(JLBDAS,JLBDAR) = 0.0 !
275 !
276  END IF
277 !
278 ! Otherwise ZDRMAX = 0.0 so the density of the graupel cannot be reached
279 ! and so PRRCOLSS(JLBDAS,JLBDAR) = 0.0 !
280 !
281  END IF
282  END DO
283 !
284 !* 1.10 Scale the fall speed difference
285 !
286  IF( zscalsr>0.0 ) prscolrg(jlbdar,jlbdas) = zcollsr / zscalsr
287  END DO
288 END DO
289 !
290 IF (lhook) CALL dr_hook('RSCOLRG',1,zhook_handle)
291 END SUBROUTINE rscolrg
