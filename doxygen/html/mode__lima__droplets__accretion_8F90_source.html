77 !

80 !

86 !
87 !

89 !
90  gaccr(:) = prrt(:)>xrtmin(3) .AND. &
91  pcrt(:)>xctmin(3) .AND. &
92  prct(:)>xrtmin(2) .AND. &
93  pcct(:)>xctmin(2)
94 !
95  WHERE ( gaccr(:) )
96 !
97  zw1(:) = 67.0 * ( prct(:) * prrt(:) )**1.15
98  p_rc_accr(:) = - zw1(:)
99 !
100  zw2(:) = zw1(:) * pcct(:) / prct(:)
101  p_cc_accr(:) = - zw2(:)
102 !
103  END WHERE
104 !
105 ELSE IF (nmom_c.EQ.1 .AND. nmom_r.EQ.1) THEN
106  gaccr(:) = prrt(:)>xrtmin(3) .AND. pcrt(:)>xctmin(3) .AND. &
107  prct(:)>xrtmin(2) .AND. pcct(:)>xctmin(2)
108  WHERE ( gaccr(:) )
109  p_rc_accr(:) = - xfcaccr * prct(:) &
110  * plbdr(:)**xexcaccr &
111  * prhodref(:)**(-xcexvt)
112  END WHERE
113 ELSE
114 !
115  WHERE( prct(:)>xrtmin(2) .AND. pcct(:)>xctmin(2) .AND. prrt(:)>xrtmin(3) .AND. pcrt(:)>xctmin(3) .AND. ldcompute(:) )
116  zw2(:) = max( 0.0,xlautr*prhodref(:)*prct(:)*(xauto1/plbdc(:)**4-xlautr_threshold) ) ! L
117  zw4(:) = xaccr1/plbdr(:)
118  END WHERE
119 !
120  gaccr(:) = ldcompute(:) .AND. &
121  prrt(:)>xrtmin(3) .AND. &
122  pcrt(:)>xctmin(3) .AND. &
123  prct(:)>xrtmin(2) .AND. &
124  pcct(:)>xctmin(2) .AND. &
125  (prrt(:)>1.2*zw2(:)/prhodref(:) .OR. &
126  zw4(:)>=max(xaccr2,xaccr3/(xaccr4/plbdc(:)-xaccr5)) )
127 !
128 ! Accretion for D>100 10-6 m
129  WHERE( gaccr(:).AND.(zw4(:)>1.e-4) )
130  zw3(:) = min(plbdc3(:) / plbdr3(:),1.e15)
131  zw1(:) = ( pcct(:)*pcrt(:) / plbdc3(:) )*prhodref(:)
132  zw2(:) = zw1(:)*(xaccr_clarge1+xaccr_clarge2*zw3(:))
133 !
134  p_cc_accr(:) = - zw2(:)
135 !
136  zw1(:) = ( zw1(:) / plbdc3(:) )
137  zw2(:) = zw1(:)*(xaccr_rlarge1+xaccr_rlarge2*zw3(:))
138 !
139  p_rc_accr(:) = - zw2(:)
140  END WHERE
141 !
142 ! Accretion for D<100 10-6 m
143  WHERE( gaccr(:).AND.(zw4(:)<=1.e-4) )
144  zw3(:) = min(plbdc3(:) / plbdr3(:), 1.e8)
145  zw1(:) = ( pcct(:)*pcrt(:) / plbdc3(:) )*prhodref(:)
146  zw1(:) = zw1(:)/plbdc3(:)
147 
148  zw3(:) = zw3(:)**2
149  zw2(:) = zw1(:)*(xaccr_csmall1+xaccr_csmall2*zw3(:))
150 !
151  p_cc_accr(:) = - zw2(:)
152 !
153  zw1(:) = zw1(:) / plbdc3(:)
154  zw2(:) = zw1(:)*(xaccr_rsmall1+xaccr_rsmall2*zw3(:))
155 !
156  p_rc_accr(:) = - zw2(:)
157  END WHERE
158 !
159 END IF
