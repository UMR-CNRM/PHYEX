15  SUBROUTINE set_conc_lima( kmi, HGETCLOUD, PRHODREF, PRT, PSVT, LDLBC )
16 ! ###########################################################################
17 !
18 !!**** *SET_CONC_LIMA * - initialize droplet, raindrop and ice
19 !! concentration for a RESTArt simulation of the LIMA scheme
20 !!
21 !!
22 !! PURPOSE
23 !! -------
24 !! The purpose of this routine is to initialize cloud droplet and rain drop
25 !! concentrations when the cloud droplet and rain drop mixing ratios are
26 !! only available (generally from a previous run using the Kessler scheme).
27 !! This routine is used to initialize the droplet/drop concentrations
28 !! using the r_c and r_r of a previous REVE or KESS run but also to compute
29 !! the LB tendencies in ONE_WAY$n in case of grid-nesting when the optional
30 !! argument PTIME is set (a LIMA run embedded in a KESS or REVE run).
31 !!
32 !!** METHOD
33 !! ------
34 !! The method assumes a Csk law for the activation of aerososl with "s"
35 !! the supersaturation (here 0.05 % is chosen). A Marshall-Palmer law with
36 !! N_o=10**(-7) m**(-4) is assumed for the rain drop concentration.
37 !! The initialization of the PSVT is straightforward for the cloud droplets
38 !! while N_r=N_0/Lambda_r with Rho*r_r=Pi*Rho_w*N_0/(Lambda_r**4) is used for
39 !! the rain drops. The HGETCLOUD test is used to discriminate between the
40 !! 'REVE' and 'KESS' options for CCLOUD in the previous run (from which
41 !! PRT was calculated).
42 !!
43 !! EXTERNAL
44 !! --------
45 !! None
46 !!
47 !! IMPLICIT ARGUMENTS
48 !! ------------------
49 !! Module MODD_RAIN_C2R2_DESCR, ONLY : XRTMIN, XCTMIN
50 !! Module MODD_RAIN_C2R2_KHKO_PARAM, ONLY : XCONCC_INI, XCONCR_PARAM_INI
51 !! Module MODD_CONF, ONLY : NVERB
52 !!
53 !! REFERENCE
54 !! ---------
55 !! Book2 of documentation ( routine SET_CONC_RAIN_C2R2 )
56 !!
57 !! AUTHOR
58 !! ------
59 !! J.-P. Pinty * Laboratoire d'Aerologie*
60 !! P. Jabouille * CNRM/GMME *
61 !! B. Vi√© * CNRM/GMME *
62 !!
63 !! MODIFICATIONS
64 !! -------------
65 !! Original 15/11/00
66 !! 2014 G.Delautier : remplace MODD_RAIN_C2R2_PARAM par MODD_RAIN_C2R2_KHKO_PARAM *
67 !! Philippe Wautelet: 05/2016-04/2018: new data structures and calls for I/O
68 ! B. Vi√© 03/03/2020: secure physical tests
69 ! P. Wautelet 04/06/2020: correct array start for microphys. concentrations + add kmi dummy argument
70 ! (this subroutine is also called for other models)
71 !-------------------------------------------------------------------------------
72 !
73 !* 0. DECLARATIONS
74 ! ------------
75 !
76 USE modd_param_lima, ONLY : nmod_ccn, nmod_ifn, &
77  nmom_c, nmom_r, nmom_i
78 USE modd_param_lima_cold, ONLY : xas, xbs
79 USE modd_param_lima_mixed,ONLY : xag, xbg, xah, xbh
80 USE modd_nsv, ONLY : nsv_lima_beg, nsv_lima_nc, nsv_lima_nr, nsv_lima_ccn_acti, &
81  nsv_lima_ni, nsv_lima_ns, nsv_lima_ng, nsv_lima_nh, nsv_lima_ifn_nucl
82 USE modd_cst, ONLY : xpi, xrholw
83 !
84 IMPLICIT NONE
85 !
86 !* 0.1 Declarations of dummy arguments :
87 !
88 integer, intent(in) :: kmi ! Model number
89 CHARACTER (LEN=4), INTENT(IN) :: HGETCLOUD ! Get indicator
90 REAL, DIMENSION(:,:,:), INTENT(IN) :: PRHODREF ! Reference density
91 !
92 REAL, DIMENSION(:,:,:,:), INTENT(INOUT) :: PRT ! microphysical mixing ratios
93 !
94 REAL, DIMENSION(:,:,:,:), INTENT(INOUT) :: PSVT ! microphys. concentrations
95 LOGICAL, OPTIONAL, INTENT(IN) :: LDLBC ! T to activate LBC mode
96 !
97 !
98 !* 0.2 Declarations of local variables :
99 !
100 REAL :: ZCONC
101 INTEGER :: ISV_LIMA_NC, ISV_LIMA_NR, ISV_LIMA_CCN_ACTI
102 INTEGER :: ISV_LIMA_NI, ISV_LIMA_NS, ISV_LIMA_NG, ISV_LIMA_NH, ISV_LIMA_IFN_NUCL
103 LOGICAL :: LLLBC
104 REAL :: ZSVTHR
105 !
106 !-------------------------------------------------------------------------------
107 !* 1. RETRIEVE LOGICAL UNIT NUMBER
108 ! ----------------------------
109 !
110 isv_lima_nc = nsv_lima_nc - nsv_lima_beg + 1
111 isv_lima_nr = nsv_lima_nr - nsv_lima_beg + 1
112 isv_lima_ccn_acti = nsv_lima_ccn_acti - nsv_lima_beg + 1
113 isv_lima_ni = nsv_lima_ni - nsv_lima_beg + 1
114 isv_lima_ns = nsv_lima_ns - nsv_lima_beg + 1
115 isv_lima_ng = nsv_lima_ng - nsv_lima_beg + 1
116 isv_lima_nh = nsv_lima_nh - nsv_lima_beg + 1
117 isv_lima_ifn_nucl = nsv_lima_ifn_nucl - nsv_lima_beg + 1
118 !
119 !* 2. INITIALIZATION
120 ! --------------
121 !
122 lllbc=.false.
123 IF(PRESENT(ldlbc)) lllbc=ldlbc
124 IF(lllbc) THEN
125  zsvthr=1.e-11 ! valid value to check
126 ELSE
127  zsvthr=1.e20 ! to deactivate this test
128 ENDIF
129 !
130 IF (nmom_c.GE.2) THEN
131 !
132 ! droplets
133 !
134  zconc = 300.e6 ! droplet concentration set at 300 cm-3
135  WHERE ( prt(:,:,:,2) > 1.e-11 .AND. psvt(:,:,:,isv_lima_nc)<zsvthr)
136  psvt(:,:,:,isv_lima_nc) = zconc
137  END WHERE
138  WHERE ( prt(:,:,:,2) <= 1.e-11 .AND. psvt(:,:,:,isv_lima_nc)<zsvthr)
139  prt(:,:,:,2) = 0.0
140  psvt(:,:,:,isv_lima_nc) = 0.0
141  END WHERE
142 
143  IF (nmod_ccn .GE. 1) THEN
144  WHERE ( prt(:,:,:,2) > 1.e-11 .AND. psvt(:,:,:,isv_lima_nc)<zsvthr)
145  psvt(:,:,:,isv_lima_ccn_acti) = zconc
146  END WHERE
147  WHERE ( prt(:,:,:,2) <= 1.e-11 .AND. psvt(:,:,:,isv_lima_nc)<zsvthr)
148  psvt(:,:,:,isv_lima_ccn_acti) = 0.0
149  END WHERE
150  END IF
151 
152 END IF
153 !
154 IF (nmom_r.GE.2) THEN
155 !
156 ! drops
157 !
158  zconc = (1.e7)**3/(xpi*xrholw) ! cf XCONCR_PARAM_INI in ini_rain_c2r2.f90
159  IF (hgetcloud == 'INI1') THEN ! init from REVE scheme
160  psvt(:,:,:,isv_lima_nr) = 0.0
161  ELSE ! init from KESS, ICE3...
162  WHERE ( prt(:,:,:,3) > 1.e-11 .AND. psvt(:,:,:,isv_lima_nr)<zsvthr )
163  psvt(:,:,:,isv_lima_nr) = max( sqrt(sqrt(prhodref(:,:,:)*prt(:,:,:,3) &
164  *zconc)),1. )
165  END WHERE
166  WHERE ( prt(:,:,:,3) <= 1.e-11 .AND. psvt(:,:,:,isv_lima_nr)<zsvthr )
167  prt(:,:,:,3) = 0.0
168  psvt(:,:,:,isv_lima_nr) = 0.0
169  END WHERE
170  END IF
171 END IF
172 !
173 IF (nmom_i.GE.2) THEN
174 !
175 ! ice crystals
176 !
177  zconc = 100.e3 ! maximum ice concentration set at 100/L
178  WHERE ( prt(:,:,:,4) > 1.e-11 .AND. psvt(:,:,:,isv_lima_ni)<zsvthr )
179 !
180 ! PSVT(:,:,:,NSV_LIMA_NI_A(kmi)) = MIN( PRHODREF(:,:,:) / &
181 ! ( XRHOLI * XAI*(10.E-06)**XBI * PRT(:,:,:,4) ), &
182 ! ZCONC )
183 ! Correction
184  psvt(:,:,:,isv_lima_ni) = min(prt(:,:,:,4)/(0.82*(10.e-06)**2.5),zconc )
185  END WHERE
186  WHERE ( prt(:,:,:,4) <= 1.e-11 .AND. psvt(:,:,:,isv_lima_ni)<zsvthr )
187  prt(:,:,:,4) = 0.0
188  psvt(:,:,:,isv_lima_ni) = 0.0
189  END WHERE
190 
191  IF (nmod_ifn .GE. 1) THEN
192  WHERE ( prt(:,:,:,4) > 1.e-11 .AND. psvt(:,:,:,isv_lima_ni)<zsvthr )
193  psvt(:,:,:,isv_lima_ifn_nucl) = psvt(:,:,:,isv_lima_ni)
194  END WHERE
195  WHERE ( prt(:,:,:,4) <= 1.e-11 .AND. psvt(:,:,:,isv_lima_ni)<zsvthr )
196  psvt(:,:,:,isv_lima_ifn_nucl) = 0.0
197  END WHERE
198  END IF
199 
200 END IF
201 !
202 IF (isv_lima_ns.GE.1) THEN
203 !
204 ! snow
205 !
206  zconc = 1./ (xas*0.001**xbs) ! 1mm particle size
207  WHERE ( prt(:,:,:,5) > 1.e-11 .AND. psvt(:,:,:,isv_lima_ns)<zsvthr )
208  psvt(:,:,:,isv_lima_ns) = prt(:,:,:,5) * zconc
209  END WHERE
210  WHERE ( prt(:,:,:,5) <= 1.e-11 .AND. psvt(:,:,:,isv_lima_ns)<zsvthr )
211  prt(:,:,:,5) = 0.0
212  psvt(:,:,:,isv_lima_ns) = 0.0
213  END WHERE
214 END IF
215 !
216 IF (isv_lima_ng.GE.1) THEN
217 !
218 ! graupel
219 !
220  zconc = 1./ (xag*0.001**xbg) ! 1mm particle size
221  WHERE ( prt(:,:,:,6) > 1.e-11 .AND. psvt(:,:,:,isv_lima_ng)<zsvthr )
222  psvt(:,:,:,isv_lima_ng) = prt(:,:,:,6) * zconc
223  END WHERE
224  WHERE ( prt(:,:,:,6) <= 1.e-11 .AND. psvt(:,:,:,isv_lima_ng)<zsvthr )
225  prt(:,:,:,6) = 0.0
226  psvt(:,:,:,isv_lima_ng) = 0.0
227  END WHERE
228 END IF
229 !
230 IF (isv_lima_nh.GE.1) THEN
231 !
232 ! hail
233 !
234  zconc = 1./ (xah*0.001**xbh) ! 1mm particle size
235  WHERE ( prt(:,:,:,7) > 1.e-11 .AND. psvt(:,:,:,isv_lima_nh)<zsvthr )
236  psvt(:,:,:,isv_lima_nh) = prt(:,:,:,7) * zconc
237  END WHERE
238  WHERE ( prt(:,:,:,7) <= 1.e-11 .AND. psvt(:,:,:,isv_lima_nh)<zsvthr )
239  prt(:,:,:,7) = 0.0
240  psvt(:,:,:,isv_lima_nh) = 0.0
241  END WHERE
242 END IF
243 !
244 END SUBROUTINE set_conc_lima
245 
246 end module mode_set_conc_lima
