15 SUBROUTINE set_conc_lima( TNSV, D, KRR, KMI, HGETCLOUD, PRHODREF, PRT, PSVT, ODLBC )
16! ###########################################################################
17!
18!!**** *SET_CONC_LIMA * - initialize droplet, raindrop and ice
19!! concentration for a RESTArt simulation of the LIMA scheme
20!!
21!!
22!! PURPOSE
23!! -------
24!! The purpose of this routine is to initialize cloud droplet and rain drop
25!! concentrations when the cloud droplet and rain drop mixing ratios are
26!! only available (generally from a previous run using the Kessler scheme).
27!! This routine is used to initialize the droplet/drop concentrations
28!! using the r_c and r_r of a previous REVE or KESS run but also to compute
29!! the LB tendencies in ONE_WAY$n in case of grid-nesting when the optional
30!! argument PTIME is set (a LIMA run embedded in a KESS or REVE run).
31!!
32!!** METHOD
33!! ------
34!! The method assumes a Csk law for the activation of aerososl with "s"
35!! the supersaturation (here 0.05 % is chosen). A Marshall-Palmer law with
36!! N_o=10**(-7) m**(-4) is assumed for the rain drop concentration.
37!! The initialization of the PSVT is straightforward for the cloud droplets
38!! while N_r=N_0/Lambda_r with Rho*r_r=Pi*Rho_w*N_0/(Lambda_r**4) is used for
39!! the rain drops. The HGETCLOUD test is used to discriminate between the
40!! 'REVE' and 'KESS' options for CCLOUD in the previous run (from which
41!! PRT was calculated).
42!!
43!! EXTERNAL
44!! --------
45!! None
46!!
47!! IMPLICIT ARGUMENTS
48!! ------------------
49!! Module MODD_RAIN_C2R2_DESCR, ONLY : XRTMIN, XCTMIN
50!! Module MODD_RAIN_C2R2_KHKO_PARAM, ONLY : XCONCC_INI, XCONCR_PARAM_INI
51!! Module MODD_CONF, ONLY : NVERB
52!!
53!! REFERENCE
54!! ---------
55!! Book2 of documentation ( routine SET_CONC_RAIN_C2R2 )
56!!
57!! AUTHOR
58!! ------
59!! J.-P. Pinty * Laboratoire d'Aerologie*
60!! P. Jabouille * CNRM/GMME *
61!! B. Vi√© * CNRM/GMME *
62!!
63!! MODIFICATIONS
64!! -------------
65!! Original 15/11/00
66!! 2014 G.Delautier : remplace MODD_RAIN_C2R2_PARAM par MODD_RAIN_C2R2_KHKO_PARAM *
67!! Philippe Wautelet: 05/2016-04/2018: new data structures and calls for I/O
68! B. Vi√© 03/03/2020: secure physical tests
69! P. Wautelet 04/06/2020: correct array start for microphys. concentrations + add kmi dummy argument
70! (this subroutine is also called for other models)
71!-------------------------------------------------------------------------------
72!
73!* 0. DECLARATIONS
74! ------------
75!
76USE modd_dimphyex, ONLY: dimphyex_t
77USE modd_param_lima, ONLY : nmod_ccn, nmod_ifn, &
78 nmom_c, nmom_r, nmom_i, &
79 lcrystal_shape, nnb_crystal_shape
80USE modd_param_lima_cold, ONLY : xas, xbs, &
81 xai, xbi, &
82 xai_shape, xbi_shape
83USE modd_param_lima_mixed,ONLY : xag, xbg, xah, xbh
84USE modd_nsv, ONLY : nsv_t
85USE modd_cst, ONLY : xpi, xrholw
86USE yomhook, ONLY:lhook, dr_hook, jphook
87!
88IMPLICIT NONE
89!
90!* 0.1 Declarations of dummy arguments :
91!
92TYPE(nsv_t), INTENT(IN) :: TNSV
93TYPE(dimphyex_t), INTENT(IN) :: D
94INTEGER, INTENT(IN) :: KRR ! Number of moist variables
95INTEGER, INTENT(IN) :: KMI ! Model number
96CHARACTER (LEN=4), INTENT(IN) :: HGETCLOUD ! Get indicator
97REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRHODREF ! Reference density
98!
99REAL, DIMENSION(D%NIJT,D%NKT,KRR), INTENT(INOUT) :: PRT ! microphysical mixing ratios
100!
101REAL, DIMENSION(D%NIJT,D%NKT,TNSV%NSV_LIMA), INTENT(INOUT) :: PSVT ! microphys. concentrations
102LOGICAL, OPTIONAL, INTENT(IN) :: ODLBC ! T to activate LBC mode
103!
104!
105!* 0.2 Declarations of local variables :
106!
107REAL :: ZCONC
108INTEGER :: ISV_LIMA_NC, ISV_LIMA_NR, ISV_LIMA_CCN_ACTI
109INTEGER :: ISV_LIMA_NI, ISV_LIMA_NS, ISV_LIMA_NG, ISV_LIMA_NH, ISV_LIMA_IFN_NUCL
110LOGICAL :: GLLBC
111REAL :: ZSVTHR
112INTEGER :: ISH ! Loop index for ice crystal shapes
113REAL, DIMENSION(D%NIJT,D%NKT) :: ZNI_TOT ! total ice crystal concentration
114REAL(KIND=jphook) :: zhook_handle
115!
116!-------------------------------------------------------------------------------
117!* 1. RETRIEVE LOGICAL UNIT NUMBER
118! ----------------------------
119!
120IF (lhook) CALL dr_hook('SET_CONC_LIMA', 0, zhook_handle)
121isv_lima_nc = tnsv%NSV_LIMA_NC - tnsv%NSV_LIMA_BEG + 1
122isv_lima_nr = tnsv%NSV_LIMA_NR - tnsv%NSV_LIMA_BEG + 1
123isv_lima_ccn_acti = tnsv%NSV_LIMA_CCN_ACTI - tnsv%NSV_LIMA_BEG + 1
124isv_lima_ni = tnsv%NSV_LIMA_NI - tnsv%NSV_LIMA_BEG + 1
125isv_lima_ns = tnsv%NSV_LIMA_NS - tnsv%NSV_LIMA_BEG + 1
126isv_lima_ng = tnsv%NSV_LIMA_NG - tnsv%NSV_LIMA_BEG + 1
127isv_lima_nh = tnsv%NSV_LIMA_NH - tnsv%NSV_LIMA_BEG + 1
128isv_lima_ifn_nucl = tnsv%NSV_LIMA_IFN_NUCL - tnsv%NSV_LIMA_BEG + 1
129!
130!* 2. INITIALIZATION
131! --------------
132!
133gllbc=.false.
134IF(PRESENT(odlbc)) gllbc=odlbc
135IF(gllbc) THEN
136 zsvthr=1.e-11 ! valid value to check
137ELSE
138 zsvthr=1.e20 ! to deactivate this test
139ENDIF
140!
141IF (nmom_c.GE.2) THEN
142!
143! droplets
144!
145 zconc = 300.e6 ! droplet concentration set at 300 cm-3
146 WHERE ( prt(:,:,2) > 1.e-11 .AND. psvt(:,:,isv_lima_nc)<zsvthr)
147 psvt(:,:,isv_lima_nc) = zconc
148 END WHERE
149 WHERE ( prt(:,:,2) <= 1.e-11 .AND. psvt(:,:,isv_lima_nc)<zsvthr)
150 prt(:,:,2) = 0.0
151 psvt(:,:,isv_lima_nc) = 0.0
152 END WHERE
153
154 IF (nmod_ccn .GE. 1) THEN
155 WHERE ( prt(:,:,2) > 1.e-11 .AND. psvt(:,:,isv_lima_nc)<zsvthr)
156 psvt(:,:,isv_lima_ccn_acti) = zconc
157 END WHERE
158 WHERE ( prt(:,:,2) <= 1.e-11 .AND. psvt(:,:,isv_lima_nc)<zsvthr)
159 psvt(:,:,isv_lima_ccn_acti) = 0.0
160 END WHERE
161 END IF
162
163END IF
164!
165IF (nmom_r.GE.2) THEN
166!
167! drops
168!
169 zconc = (1.e7)**3/(xpi*xrholw) ! cf XCONCR_PARAM_INI in ini_rain_c2r2.f90
170 IF (hgetcloud == 'INI1') THEN ! init from REVE scheme
171 psvt(:,:,isv_lima_nr) = 0.0
172 ELSE ! init from KESS, ICE3...
173 WHERE ( prt(:,:,3) > 1.e-11 .AND. psvt(:,:,isv_lima_nr)<zsvthr )
174 psvt(:,:,isv_lima_nr) = max( sqrt(sqrt(prhodref(:,:)*prt(:,:,3) &
175 *zconc)),1. )
176 END WHERE
177 WHERE ( prt(:,:,3) <= 1.e-11 .AND. psvt(:,:,isv_lima_nr)<zsvthr )
178 prt(:,:,3) = 0.0
179 psvt(:,:,isv_lima_nr) = 0.0
180 END WHERE
181 END IF
182END IF
183!
184IF (nmom_i.GE.2) THEN
185!
186! ice crystals
187!
188 zconc = 100.e3 ! maximum ice concentration set at 100/L
189 IF (.NOT. lcrystal_shape) THEN
190 WHERE ( prt(:,:,4) > 1.e-11 .AND. psvt(:,:,isv_lima_ni)<zsvthr )
191!
192! PSVT(:,:,TNSV%NSV_LIMA_NI_A(kmi)) = MIN( PRHODREF(:,:) / &
193! ( XRHOLI * XAI*(10.E-06)**XBI * PRT(:,:,4) ), &
194! ZCONC )
195! Correction
196! PSVT(:,:,ISV_LIMA_NI) = MIN(PRT(:,:,4)/(0.82*(10.E-06)**2.5),ZCONC )
197 psvt(:,:,isv_lima_ni) = min(prt(:,:,4)/(xai*(10.e-06)**xbi),zconc )
198 END WHERE
199 WHERE ( prt(:,:,4) <= 1.e-11 .AND. psvt(:,:,isv_lima_ni)<zsvthr )
200 prt(:,:,4) = 0.0
201 psvt(:,:,isv_lima_ni) = 0.0
202 END WHERE
203 ELSE
204 zni_tot(:,:) = 0.
205 DO ish = 1, nnb_crystal_shape
206 WHERE ( prt(:,:,4) > 1.e-11 .AND. psvt(:,:,isv_lima_ni+ish-1) < zsvthr)
207 psvt(:,:,isv_lima_ni+ish-1) = min(prt(:,:,4)/(xai_shape(ish)*(10.e-06)**xbi_shape(ish)),zconc )
208 END WHERE
209 zni_tot(:,:) = zni_tot(:,:) + psvt(:,:,isv_lima_ni+ish-1)
210 END DO
211 WHERE (prt(:,:,4) <= 1.e-11 .AND. zni_tot(:,:)<zsvthr )
212 prt(:,:,4) = 0.0
213!++cb-- il faudrait trouver une plus jolie solution, mais ca compile
214 !PSVT(:,:,:,ISV_LIMA_NI:ISV_LIMA_NI+NNB_CRYSTAL_SHAPE-1) = 0.0
215 psvt(:,:,isv_lima_ni) = 0.0
216 psvt(:,:,isv_lima_ni+1) = 0.0
217 psvt(:,:,isv_lima_ni+2) = 0.0
218 psvt(:,:,isv_lima_ni+3) = 0.0
219 END WHERE
220 END IF
221
222 IF (nmod_ifn .GE. 1) THEN
223 WHERE ( prt(:,:,4) > 1.e-11 .AND. psvt(:,:,isv_lima_ni)<zsvthr )
224 psvt(:,:,isv_lima_ifn_nucl) = psvt(:,:,isv_lima_ni)
225 END WHERE
226 WHERE ( prt(:,:,4) <= 1.e-11 .AND. psvt(:,:,isv_lima_ni)<zsvthr )
227 psvt(:,:,isv_lima_ifn_nucl) = 0.0
228 END WHERE
229 END IF
230
231END IF
232!
233IF (isv_lima_ns.GE.1) THEN
234!
235! snow
236!
237 zconc = 1./ (xas*0.001**xbs) ! 1mm particle size
238 WHERE ( prt(:,:,5) > 1.e-11 .AND. psvt(:,:,isv_lima_ns)<zsvthr )
239 psvt(:,:,isv_lima_ns) = prt(:,:,5) * zconc
240 END WHERE
241 WHERE ( prt(:,:,5) <= 1.e-11 .AND. psvt(:,:,isv_lima_ns)<zsvthr )
242 prt(:,:,5) = 0.0
243 psvt(:,:,isv_lima_ns) = 0.0
244 END WHERE
245END IF
246!
247IF (isv_lima_ng.GE.1) THEN
248!
249! graupel
250!
251 zconc = 1./ (xag*0.001**xbg) ! 1mm particle size
252 WHERE ( prt(:,:,6) > 1.e-11 .AND. psvt(:,:,isv_lima_ng)<zsvthr )
253 psvt(:,:,isv_lima_ng) = prt(:,:,6) * zconc
254 END WHERE
255 WHERE ( prt(:,:,6) <= 1.e-11 .AND. psvt(:,:,isv_lima_ng)<zsvthr )
256 prt(:,:,6) = 0.0
257 psvt(:,:,isv_lima_ng) = 0.0
258 END WHERE
259END IF
260!
261IF (isv_lima_nh.GE.1) THEN
262!
263! hail
264!
265 zconc = 1./ (xah*0.001**xbh) ! 1mm particle size
266 WHERE ( prt(:,:,7) > 1.e-11 .AND. psvt(:,:,isv_lima_nh)<zsvthr )
267 psvt(:,:,isv_lima_nh) = prt(:,:,7) * zconc
268 END WHERE
269 WHERE ( prt(:,:,7) <= 1.e-11 .AND. psvt(:,:,isv_lima_nh)<zsvthr )
270 prt(:,:,7) = 0.0
271 psvt(:,:,isv_lima_nh) = 0.0
272 END WHERE
273END IF
274!
275IF (lhook) CALL dr_hook('SET_CONC_LIMA', 1, zhook_handle)
276END SUBROUTINE set_conc_lima
