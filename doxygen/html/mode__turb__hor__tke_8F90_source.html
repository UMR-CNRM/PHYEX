101 !
102 !* 1. PRELIMINARY COMPUTATIONS
103 ! ------------------------
104 !
105 ikb = 1.+jpvext
106 iku = SIZE(ptkem,3)
107 !
108 ! compute the coefficients for the uncentred gradient computation near the
109 ! ground
110 !
111 zcoeff(:,:,ikb+2)= - pdzz(:,:,ikb+1) / &
112  ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+2) )
113 zcoeff(:,:,ikb+1)= (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) / &
114  ( pdzz(:,:,ikb+1) * pdzz(:,:,ikb+2) )
115 zcoeff(:,:,ikb)= - (pdzz(:,:,ikb+2)+2.*pdzz(:,:,ikb+1)) / &
116  ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+1) )
117 !
118 !--------------------------------------------------------------------
119 !
120 !* 2. horizontal transport of Tke u'e
121 ! -------------------------------
122 !
123 !
124 zflx = -xcet * mxm(pk) * gx_m_u(1,iku,1,ptkem,pdxx,pdzz,pdzx) ! < u'e >
125 !
126 ! special case near the ground ( uncentred gradient )
127 !
128 zflx(:,:,ikb) = zcoeff(:,:,ikb+2)*ptkem(:,:,ikb+2) &
129  + zcoeff(:,:,ikb+1)*ptkem(:,:,ikb+1) &
130  + zcoeff(:,:,ikb )*ptkem(:,:,ikb )
131 !
132 zflx(:,:,ikb:ikb) = &
133  - xcet * mxm( pk(:,:,ikb:ikb) ) * ( &
134  dxm( ptkem(:,:,ikb:ikb) ) * pinv_pdxx(:,:,ikb:ikb) &
135  -mxm( zflx(:,:,ikb:ikb) ) * pinv_pdxx(:,:,ikb:ikb) &
136  * 0.5 * ( pdzx(:,:,ikb+1:ikb+1) + pdzx(:,:,ikb:ikb) ) )
137 !
138 ! extrapolate the fluxes to obtain < u'e > = 0 at the ground
139 !
140 zflx(:,:,ikb-1) = - zflx(:,:,ikb)
141 !
142 ! let the same flux at IKU-1 and IKU level
143 !
144 zflx(:,:,iku) = zflx(:,:,iku-1)
145 !
146 IF (.NOT. oflat) THEN
147  ptrh =-( dxf( mxm(prhodj) * zflx * pinv_pdxx)&
148  - dzf( pmzm_prhodj * mxf( pdzx * mzm(zflx*pinv_pdxx)) * pinv_pdzz)&
149  ) /prhodj
150 ELSE
151  ptrh =-( dxf( mxm(prhodj) * zflx * pinv_pdxx)&
152  ) /prhodj
153 END IF
154 !
155 IF (tles%LLES_CALL .AND. ksplt==1) THEN
156  CALL second_mnh(ztime1)
157  CALL les_mean_subgrid( mxf(zflx), tles%X_LES_SUBGRID_UTke )
158  CALL second_mnh(ztime2)
159  tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
160 END IF
161 !
162 !
163 !--------------------------------------------------------------------
164 !
165 !* 3. horizontal transport of Tke v'e
166 ! -------------------------------
167 !
168 IF (.NOT. o2d) THEN
169  zflx =-xcet * mym(pk) * gy_m_v(1,iku,1,ptkem,pdyy,pdzz,pdzy) ! < v'e >
170 !
171 ! special case near the ground ( uncentred gradient )
172 !
173  zflx(:,:,ikb) = zcoeff(:,:,ikb+2)*ptkem(:,:,ikb+2) &
174  + zcoeff(:,:,ikb+1)*ptkem(:,:,ikb+1) &
175  + zcoeff(:,:,ikb )*ptkem(:,:,ikb )
176 !
177  zflx(:,:,ikb:ikb) = &
178  - xcet * mym( pk(:,:,ikb:ikb) ) * ( &
179  dym( ptkem(:,:,ikb:ikb) ) * pinv_pdyy(:,:,ikb:ikb) &
180  - mym( zflx(:,:,ikb:ikb) ) * pinv_pdyy(:,:,ikb:ikb) &
181  * 0.5 * ( pdzy(:,:,ikb+1:ikb+1) + pdzy(:,:,ikb:ikb) ) )
182 !
183 ! extrapolate the fluxes to obtain < v'e > = 0 at the ground
184 !
185  zflx(:,:,ikb-1) = - zflx(:,:,ikb)
186 !
187 ! let the same flux at IKU-1 and IKU level
188 !
189  zflx(:,:,iku) = zflx(:,:,iku-1)
190 !
191 ! complete the explicit turbulent transport
192 !
193  IF (.NOT. oflat) THEN
194  ptrh = ptrh - ( dyf( mym(prhodj) * zflx * pinv_pdyy ) &
195  - dzf( pmzm_prhodj * myf( pdzy * mzm(zflx*pinv_pdyy) ) * pinv_pdzz ) &
196  ) /prhodj
197  ELSE
198  ptrh = ptrh - ( dyf( mym(prhodj) * zflx * pinv_pdyy ) &
199  ) /prhodj
200  END IF
201 !
202  IF (tles%LLES_CALL .AND. ksplt==1) THEN
203  CALL second_mnh(ztime1)
204  CALL les_mean_subgrid( myf(zflx), tles%X_LES_SUBGRID_VTke )
205  CALL second_mnh(ztime2)
206  tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
207  END IF
208 !
209 END IF
