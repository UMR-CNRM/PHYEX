74TYPE(tles_t), INTENT(INOUT) :: TLES ! modd_les structure
75INTEGER, INTENT(IN) :: KSPLT ! current split index
76LOGICAL, INTENT(IN) :: OFLAT ! Logical for zero ororography
77LOGICAL, INTENT(IN) :: O2D ! Logical for 2D model version (modd_conf)
78REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PDXX, PDYY, PDZZ, PDZX, PDZY
79 ! Metric coefficients
80REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PK ! Turbulent diffusion doef.
81 ! PK = PLM * SQRT(PTKEM)
82REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDXX ! 1./PDXX
83REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDYY ! 1./PDYY
84REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PINV_PDZZ ! 1./PDZZ
85REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PMZM_PRHODJ ! MZM(PRHODJ)
86REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PRHODJ ! density * grid volume
87!
88!
89REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PTKEM ! TKE at time t- dt
90REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(OUT) :: PTRH ! horizontal transport of Tke
91!
92!
93!
94!* 0.2 declaration of local variables
95!
96INTEGER :: IKB, IKU, IIT, IJT, IKT, JI, JJ, JK
97!
98REAL, DIMENSION(D%NIT,D%NJT,1+JPVEXT:3+JPVEXT) :: ZCOEFF
99 ! coefficients for the uncentred gradient
100 ! computation near the ground
101!
102REAL, DIMENSION(D%NIT,D%NJT,D%NKT):: ZFLX
103!
104REAL :: ZTIME1, ZTIME2
105! ---------------------------------------------------------------------------
106!
107!* 1. PRELIMINARY COMPUTATIONS
108! ------------------------
109!
110ikb = 1.+jpvext
111iku = SIZE(ptkem,3)
112iit=d%NIT
113ijt=d%NJT
114ikt=d%NKT
115!
116! compute the coefficients for the uncentred gradient computation near the
117! ground
118!
119!$acc kernels
120!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
121zcoeff(:,:,ikb+2)= - pdzz(:,:,ikb+1) / &
122 ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+2) )
123zcoeff(:,:,ikb+1)= (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) / &
124 ( pdzz(:,:,ikb+1) * pdzz(:,:,ikb+2) )
125zcoeff(:,:,ikb)= - (pdzz(:,:,ikb+2)+2.*pdzz(:,:,ikb+1)) / &
126 ( (pdzz(:,:,ikb+2)+pdzz(:,:,ikb+1)) * pdzz(:,:,ikb+1) )
127!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
128!$acc end kernels
129!
130!--------------------------------------------------------------------
131!
132!* 2. horizontal transport of Tke u'e
133! -------------------------------
134!
135!
136zflx = -xcet * mxm(pk) * gx_m_u(oflat,ptkem,pdxx,pdzz,pdzx) ! < u'e >
137!
138! special case near the ground ( uncentred gradient )
139!
140!$acc kernels
141!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
142zflx(:,:,ikb) = zcoeff(:,:,ikb+2)*ptkem(:,:,ikb+2) &
143 + zcoeff(:,:,ikb+1)*ptkem(:,:,ikb+1) &
144 + zcoeff(:,:,ikb )*ptkem(:,:,ikb )
145!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
146!$acc end kernels
147!
148zflx(:,:,ikb) = &
149 - xcet * mxm( pk(:,:,ikb) ) * ( &
150 dxm( ptkem(:,:,ikb) ) * pinv_pdxx(:,:,ikb) &
151 -mxm( zflx(:,:,ikb) ) * pinv_pdxx(:,:,ikb) &
152 * 0.5 * ( pdzx(:,:,ikb+1) + pdzx(:,:,ikb) ) )
153!
154! extrapolate the fluxes to obtain < u'e > = 0 at the ground
155!
156!$acc kernels
157!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
158zflx(:,:,ikb-1) = - zflx(:,:,ikb)
159!
160! let the same flux at IKU-1 and IKU level
161!
162zflx(:,:,iku) = zflx(:,:,iku-1)
163!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
164!$acc end kernels
165!
166IF (.NOT. oflat) THEN
167 ptrh =-( dxf( mxm(prhodj) * zflx * pinv_pdxx)&
168 - dzf( pmzm_prhodj * mxf( pdzx * mzm(zflx*pinv_pdxx)) * pinv_pdzz)&
169 ) /prhodj
170ELSE
171 ptrh =-( dxf( mxm(prhodj) * zflx * pinv_pdxx)&
172 ) /prhodj
173END IF
174!
175IF (tles%LLES_CALL .AND. ksplt==1) THEN
176 CALL second_mnh(ztime1)
177 CALL les_mean_subgrid( mxf(zflx), tles%X_LES_SUBGRID_UTke )
178 CALL second_mnh(ztime2)
179 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
180END IF
181!
182!
183!--------------------------------------------------------------------
184!
185!* 3. horizontal transport of Tke v'e
186! -------------------------------
187!
188IF (.NOT. o2d) THEN
189 zflx =-xcet * mym(pk) * gy_m_v(oflat,ptkem,pdyy,pdzz,pdzy) ! < v'e >
190!
191! special case near the ground ( uncentred gradient )
192!
193!$acc kernels
194!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
195 zflx(:,:,ikb) = zcoeff(:,:,ikb+2)*ptkem(:,:,ikb+2) &
196 + zcoeff(:,:,ikb+1)*ptkem(:,:,ikb+1) &
197 + zcoeff(:,:,ikb )*ptkem(:,:,ikb )
198!
199!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
200!$acc end kernels
201 zflx(:,:,ikb) = &
202 - xcet * mym( pk(:,:,ikb) ) * ( &
203 dym( ptkem(:,:,ikb) ) * pinv_pdyy(:,:,ikb) &
204 - mym( zflx(:,:,ikb) ) * pinv_pdyy(:,:,ikb) &
205 * 0.5 * ( pdzy(:,:,ikb+1) + pdzy(:,:,ikb) ) )
206!
207! extrapolate the fluxes to obtain < v'e > = 0 at the ground
208!
209!$acc kernels
210!$mnh_expand_array(JI=1:IIT,JJ=1:IJT)
211 zflx(:,:,ikb-1) = - zflx(:,:,ikb)
212!
213! let the same flux at IKU-1 and IKU level
214!
215 zflx(:,:,iku) = zflx(:,:,iku-1)
216!$mnh_end_expand_array(JI=1:IIT,JJ=1:IJT)
217!$acc end kernels
218!
219! complete the explicit turbulent transport
220!
221 IF (.NOT. oflat) THEN
222 ptrh = ptrh - ( dyf( mym(prhodj) * zflx * pinv_pdyy ) &
223 - dzf( pmzm_prhodj * myf( pdzy * mzm(zflx*pinv_pdyy) ) * pinv_pdzz ) &
224 ) /prhodj
225 ELSE
226 ptrh = ptrh - ( dyf( mym(prhodj) * zflx * pinv_pdyy ) &
227 ) /prhodj
228 END IF
229!
230 IF (tles%LLES_CALL .AND. ksplt==1) THEN
231 CALL second_mnh(ztime1)
232 CALL les_mean_subgrid( myf(zflx), tles%X_LES_SUBGRID_VTke )
233 CALL second_mnh(ztime2)
234 tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
235 END IF
236!
237END IF
238!
239!----------------------------------------------------------------------------
240!
241END SUBROUTINE turb_hor_tke
