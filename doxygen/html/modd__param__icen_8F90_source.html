232 SUBROUTINE param_icen_init(HPROGRAM, TFILENAM, LDNEEDNAM, KLUOUT, &
233  &LDDEFAULTVAL, LDREADNAM, LDCHECK, KPRINT)
234 !!*** *PARAM_ICEN_INIT* - Code needed to initialize the MODD_PARAM_ICE_n module
235 !!
236 !!* PURPOSE
237 !! -------
238 !! Sets the default values, reads the namelist, performs the checks and prints
239 !!
240 !!* METHOD
241 !! ------
242 !! 0. Declarations
243 !! 1. Declaration of arguments
244 !! 2. Declaration of local variables
245 !! 1. Default values
246 !! 2. Namelist
247 !! 3. Checks
248 !! 4. Prints
249 !!
250 !! AUTHOR
251 !! ------
252 !! S. Riette
253 !!
254 !! MODIFICATIONS
255 !! -------------
256 !! Original Feb 2023
257 !-------------------------------------------------------------------------------
258 !
259 !* 0. DECLARATIONS
260 ! ---------------
261 !
262 USE mode_posnam_phy, ONLY: posnam_phy
263 USE mode_msg, ONLY: print_msg, nverb_fatal
264 USE mode_check_nam_val, ONLY: check_nam_val_char, check_nam_val_real, check_nam_val_int
265 USE modd_io, ONLY: tfiledata
266 !
267 IMPLICIT NONE
268 !
269 !* 0.1. Declaration of arguments
270 ! ------------------------
271 !
272 CHARACTER(LEN=6), INTENT(IN) :: HPROGRAM
273 TYPE(tfiledata), INTENT(IN) :: TFILENAM
274 LOGICAL, INTENT(IN) :: LDNEEDNAM
275 INTEGER, INTENT(IN) :: KLUOUT
276 LOGICAL, OPTIONAL, INTENT(IN) :: LDDEFAULTVAL
277 LOGICAL, OPTIONAL, INTENT(IN) :: LDREADNAM
278 LOGICAL, OPTIONAL, INTENT(IN) :: LDCHECK
279 INTEGER, OPTIONAL, INTENT(IN) :: KPRINT
281 !
282 !* 0.2 Declaration of local variables
283 ! ------------------------------
284 !
285 LOGICAL :: LLDEFAULTVAL, LLREADNAM, LLCHECK, LLFOUND
286 INTEGER :: IPRINT
287 
288 lldefaultval=.true.
289 llreadnam=.true.
290 llcheck=.true.
291 iprint=0
292 IF(PRESENT(lddefaultval)) lldefaultval=lddefaultval
293 IF(PRESENT(ldreadnam )) llreadnam =ldreadnam
294 IF(PRESENT(ldcheck )) llcheck =ldcheck
295 IF(PRESENT(kprint )) iprint =kprint
296 !
297 !* 1. DEFAULT VALUES
298 ! -----------------
299 !
300 IF(lldefaultval) THEN
301  !NOTES ON GENERAL DEFAULTS AND MODEL-SPECIFIC DEFAULTS :
302  !- General default values *MUST* remain unchanged.
303  !- To change the default value for a given application,
304  ! an "IF(HPROGRAM=='...')" condition must be used.
305 
306  lwarm=.true.
307  lsedic=.true.
308  ldeposc=.false.
309  xvdeposc= 0.02 ! 2 cm/s
310  cpristine_ice='PLAT'
311  csedim='SPLI'
312  lred=.true.
313  lfeedbackt=.true.
314  levlimit=.true.
315  lnullwetg=.true.
316  lwetgpost=.true.
317  lnullweth=.true.
318  lwethpost=.true.
319  csnowriming='M90'
320  xfracm90=0.1
321  nmaxiter_micro=5
322  xmrstep=0.00005
323  lconvhg=.false.
324  lcrflimit=.true.
325  xtstep_ts=0.
326  csubg_rc_rr_accr='NONE'
327  csubg_rr_evap='NONE'
328  csubg_pr_pdf='SIGM'
329  csubg_aucv_rc='NONE'
330  csubg_aucv_ri='NONE'
331  csubg_mf_pdf='TRIANGLE'
332  ladj_before=.true.
333  ladj_after=.true.
334  lsedim_after=.false.
335  xsplit_maxcfl=0.8
336  lsnow_t=.false.
337  lpack_interp=.true.
338  lpack_micro=.true.
339  npromicro=0
340  lcriauti=.false.
341  !!XCRIAUTIi_NAM = 0.25E-3 ! Critical ice content for the autoconversion to occur
342  xcriauti_nam = 0.2e-4 ! Revised value by Chaboureau et al. (2001)
343  xacriauti_nam=0.06
344  xbcriauti_nam=-3.5
345  xt0criauti_nam=(log10(xcriauti_nam)-xbcriauti_nam)/0.06
346  xcriautc_nam=0.5e-3
347  xrdepsred_nam=1.
348  xrdepgred_nam=1.
349  locnd2=.false.
350  ! Tuning and modication of graupeln etc:
351  xfrmin_nam(1:6)=0.
352  xfrmin_nam(7:9)=1.
353  xfrmin_nam(10) =10.
354  xfrmin_nam(11) =1.
355  xfrmin_nam(12) =0.
356  xfrmin_nam(13) =1.0e-15
357  xfrmin_nam(14) =120.
358  xfrmin_nam(15) =1.0e-4
359  xfrmin_nam(16:20)=0.
360  xfrmin_nam(21:22)=1.
361  xfrmin_nam(23)=0.5
362  xfrmin_nam(24)=1.5
363  xfrmin_nam(25)=30.
364  xfrmin_nam(26:38)=0.
365  xfrmin_nam(39)=0.25
366  xfrmin_nam(40)=0.15
367 
368  IF(hprogram=='AROME') THEN
369  lconvhg=.true.
370  ladj_before=.true.
371  ladj_after=.false.
372  lred=.false.
373  csedim='STAT'
374  lsedic=.false.
375  xmrstep=0.
376  csubg_aucv_rc='PDF'
377  ELSEIF(hprogram=='LMDZ') THEN
378  csubg_aucv_rc='PDF'
379  csedim='STAT'
380  nmaxiter_micro=1
381  lcriauti=.true.
382  xcriautc_nam=0.001
383  xcriauti_nam=0.0002
384  xt0criauti_nam=-5.
385  lred=.true.
386  lconvhg=.true.
387  ladj_before=.true.
388  ladj_after=.false.
389  ENDIF
390 ENDIF
391 !
392 !* 2. NAMELIST
393 ! -----------
394 !
395 IF(llreadnam) THEN
396  CALL posnam_phy(tfilenam, 'NAM_PARAM_ICEN', ldneednam, llfound)
397  IF(llfound) READ(unit=tfilenam%NLU, nml=nam_param_icen)
398 ENDIF
399 !
400 !* 3. CHECKS
401 ! ---------
402 !
403 IF(llcheck) THEN
404  CALL check_nam_val_char(kluout, 'CPRISTINE_ICE', cpristine_ice, 'PLAT', 'COLU', 'BURO')
405  CALL check_nam_val_char(kluout, 'CSEDIM', csedim, 'SPLI', 'STAT', 'NONE')
406  CALL check_nam_val_char(kluout, 'CSUBG_RC_RR_ACCR', csubg_rc_rr_accr, 'NONE', 'PRFR')
407  CALL check_nam_val_char(kluout, 'CSUBG_RR_EVAP', csubg_rr_evap, 'NONE', 'CLFR', 'PRFR')
408  CALL check_nam_val_char(kluout, 'CSUBG_PR_PDF', csubg_pr_pdf, 'SIGM', 'HLCRECTPD', 'HLCTRIANGPDF', &
409  'HLCQUADRAPDF', 'HLCISOTRIPDF')
410  CALL check_nam_val_char(kluout, 'CSUBG_AUCV_RC', csubg_aucv_rc, 'PDF ', 'CLFR', 'NONE', 'ADJU', 'SIGM')
411  CALL check_nam_val_char(kluout, 'CSUBG_AUCV_RI', csubg_aucv_ri, 'NONE', 'CLFR', 'ADJU')
412  CALL check_nam_val_char(kluout, 'CSUBG_MF_PDF', csubg_mf_pdf, 'NONE', 'TRIANGLE')
413  CALL check_nam_val_char(kluout, 'CSNOWRIMING', csnowriming, 'OLD ', 'M90 ')
414  CALL check_nam_val_real(kluout, 'XTSTEP_TS', xtstep_ts, '>=', 0.)
415  CALL check_nam_val_real(kluout, 'XSPLIT_MAXCFL', xsplit_maxcfl, '>', 0., '<=', 1.)
416  CALL check_nam_val_real(kluout, 'XVDEPOSC', xvdeposc, '>=', 0.)
417  CALL check_nam_val_real(kluout, 'XFRACM90', xfracm90, '>=', 0., '<=', 1.)
418  CALL check_nam_val_real(kluout, 'XMRSTEP', xmrstep, '>=', 0.)
419  CALL check_nam_val_int(kluout, 'NPROMICRO', npromicro, '>=', 0)
420 
421  IF (locnd2 .AND. (xrdepsred_nam /= 1 .OR. xrdepgred_nam /= 1)) THEN
422  CALL abor1 ("XRDESRED_NAM and XRDEGRED_NAM must not be activated together with LOCND2")
423  ENDIF
424 
425  IF(hprogram=='AROME' .OR. hprogram=='LMDZ') THEN
426  IF(.NOT. (ladj_before .AND. .NOT. ladj_after)) THEN
427  CALL print_msg(nverb_fatal, 'GEN', 'MODD_PARAM_ICE_n', 'With AROME/LMDZ, LADJ_BEFORE must be .T. and LADJ_AFTER must be .F.')
428  ENDIF
429  ENDIF
430 ENDIF
431 !
432 !* 3. PRINTS
433 ! ---------
434 !
435 IF(iprint>=1) THEN
436  WRITE(unit=kluout,nml=nam_param_icen)
437 ENDIF
438 !
439 END SUBROUTINE param_icen_init
440 !
441 END MODULE modd_param_ice_n
