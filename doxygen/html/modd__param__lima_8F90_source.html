642!

662LOGICAL, OPTIONAL, INTENT(IN) :: ODREADNAM

701 nmom_h = 0

703 xifn_conc(:) = 100.
704 lifn_hom = .true.
705 cifn_species = 'PHILLIPS'
706 cint_mixing = 'DM2'
707 nmod_imm = 0
708 nind_specie = 1
709 lsnow_t = .false.
710 cpristine_ice_lima = 'PLAT'
711 chevrimed_ice_lima = 'GRAU'
712 !XALPHAI=
713 !XNUI=
714 !XALPHAS=
715 !XNUS=
716 !XALPHAG=
717 !XNUG=
718 xfactnuc_dep = 1.0
719 xfactnuc_con = 1.0
720 nphillips=8
721 lcibu = .false.
722 xndebris_cibu = 50.0
723 lrdsf = .false.
724 xpsh_max_rdsf = 0.2
725 lcrystal_shape = .false.
726 nnb_crystal_shape = 1

731 htype_crystal_shape(4) = 'YDRO'
732 lmurakami=.true.
733 lacti = .true.
734 lsedc = .true.
735 lactit = .false.
736 lspro = .false.
737 ladj = .true.
738 lkhko = .false.
739 lkesslerac = .false.
740 nmom_c = 2
741 nmom_r = 2
742 nmod_ccn = 1
743 xccn_conc(:)=300.
744 lccn_hom = .true.
745 cccn_modes = 'COPT'
746 hini_ccn = 'AER'
747 htype_ccn(:) = 'M'
748 xalphac = 3.0
749 xnuc = 1.0
750 xalphar = 1.0
751 xnur = 2.0
752 xfsolub_ccn = 1.0
753 xactemp_ccn = 280.
754 xaerdiff = 0.0
755 xaerheight = 2000.
756 lscav = .false.
757 laero_mass = .false.
758 ldepoc = .true.

760 lacttke = .true.
761 lptsplit = .true.
762 lfeedbackt = .true.
763 nmaxiter = 5
764 xmrstep = 0.005
765 xtstep_ts = 20.
766 lsigmoide_g = .false.
767 lsigmoide_ng = .false.
768 xsigmoide_g = 1.e8
769 xmvdmin_g = 125.e-6

824 CALL print_msg(nverb_fatal, 'GEN', 'MODD_PARAM_LIMA', &
825 &"WHEN LCRYSTAL_SHAPE=T, IT IS POSSIBLE TO USE 3 OR 4 DIFFERENT SHAPES"// &
826 &"YOU HAVE TO SET NNB_CRYSTAL_SHAPE TO 3 OR 4.")
827 END IF
828
829 IF (lcrystal_shape .AND. lice_isc .AND. (nnb_crystal_shape .LE. 3 .OR. nnb_crystal_shape .GT. 4)) THEN
830 CALL print_msg(nverb_fatal, 'GEN', 'MODD_PARAM_LIMA', &
831 &"WHEN LCRYSTAL_SHAPE=T, AND LICE_ISC=T IT IS POSSIBLE TO USE DIFFERENT SHAPES"// &
832 &"YOU HAVE TO SET NNB_CRYSTAL_SHAPE TO 4.")
833 END IF
834
835 IF(hprogram=='AROME' .OR. hprogram=='PHYEX') THEN
836 IF(.NOT. lptsplit) THEN
837 CALL print_msg(nverb_fatal, 'GEN', 'MODD_PARAM_LIMA', &
838 &"LPTSPLIT MUST BE .TRUE. WITH THIS PROGRAM: " // hprogram)
839 ENDIF
840 IF(lspro) THEN
841 CALL print_msg(nverb_fatal, 'GEN', 'MODD_PARAM_LIMA', &
842 &"LSPRO MUST BE .FALSE. WITH THIS PROGRAM: " // hprogram)
843 ENDIF
844 ENDIF
845ENDIF
846!
847!* 3. PRINTS
848! ---------
849!
850IF(iprint>=1) THEN
851 WRITE(unit=kluout, nml=nam_param_lima)
852ENDIF
853!
854IF (lhook) CALL dr_hook('PARAM_LIMA_INIT', 1, zhook_handle)
855END SUBROUTINE param_lima_init
856!
857END MODULE modd_param_lima
