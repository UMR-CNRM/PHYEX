17!!**** Determine thermodynamic properties at new LCL
18!!
19!! PURPOSE
20!! -------
21!! The purpose of this routine is to determine the thermodynamic
22!! properties at the new lifting condensation level LCL
23!!
24!!
25!!
26!!** METHOD
27!! ------
28!! see CONVECT_TRIGGER_FUNCT
29!!
30!!
31!!
32!! EXTERNAL
33!! --------
34!! Routine CONVECT_SATMIXRATIO
35!!
36!!
37!! IMPLICIT ARGUMENTS
38!! ------------------
39!! Module MODD_CST
40!! XG ! gravity constant
41!! XP00 ! Reference pressure
42!! XRD, XRV ! Gaz constants for dry air and water vapor
43!! XCPD ! Cpd (dry air)
44!! XTT ! triple point temperature
45!! XBETAW, XGAMW ! constants for vapor saturation pressure
46!!
47!! Module MODD_CONVPAR
48!! XA25 ! reference grid area
49!! XZLCL ! lowest allowed pressure difference between
50!! ! surface and LCL
51!! XZPBL ! minimum mixed layer depth to sustain convection
52!! XWTRIG ! constant in vertical velocity trigger

78USE modd_cst, ONLY: cst_t
79USE modd_convparext, ONLY : convparext
80USE modd_cst, ONLY: cst_t
81USE modd_dimphyex, ONLY: dimphyex_t

83!
84IMPLICIT NONE
85!
86!* 0.1 Declarations of dummy arguments :
87!
88TYPE(convparext), INTENT(IN) :: CVPEXT
89TYPE(cst_t), INTENT(IN) :: CST
90TYPE(dimphyex_t), INTENT(IN) :: D
91REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PTH ! theta
92REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PRV ! vapor mixing ratio
93REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PPRES ! pressure
94REAL, DIMENSION(D%NIJT,D%NKT), INTENT(IN) :: PZ ! height of grid point (m)
95INTEGER, DIMENSION(D%NIJT), INTENT(IN) :: KDPL ! contains vert. index of DPL
96INTEGER, DIMENSION(D%NIJT), INTENT(IN) :: KPBL ! " vert. index of source layer top
97LOGICAL, DIMENSION(D%NIJT), INTENT(IN) :: OWORK1! logical mask
98!
99REAL, DIMENSION(D%NIJT), INTENT(OUT):: PTHLCL ! theta at LCL
100REAL, DIMENSION(D%NIJT), INTENT(OUT):: PRVLCL ! vapor mixing ratio at LCL
101REAL, DIMENSION(D%NIJT), INTENT(OUT):: PZLCL ! height at LCL (m)
102REAL, DIMENSION(D%NIJT), INTENT(OUT):: PTLCL ! temperature at LCL (m)
103REAL, DIMENSION(D%NIJT), INTENT(OUT):: PTELCL ! environm. temp. at LCL (K)
104INTEGER, DIMENSION(D%NIJT), INTENT(OUT):: KLCL ! contains vert. index of LCL
105!
106!* 0.2 Declarations of local variables :
107!
108INTEGER :: JK, JKM, JKMIN, JKMAX ! vertical loop index
109INTEGER :: JI ! horizontal loop index
110INTEGER :: IKB, IKE ! horizontal + vertical loop bounds
111REAL :: ZEPS ! R_d / R_v
112REAL :: ZCPORD, ZRDOCP ! C_pd / R_d, R_d / C_pd
113!
114REAL, DIMENSION(D%NIJT) :: ZPLCL ! pressure at LCL
115REAL, DIMENSION(D%NIJT) :: ZTMIX ! mixed layer temperature
116REAL, DIMENSION(D%NIJT) :: ZEVMIX ! mixed layer water vapor pressure
117REAL, DIMENSION(D%NIJT) :: ZDPTHMIX, ZPRESMIX ! mixed layer depth and pressure
118REAL, DIMENSION(D%NIJT) :: ZLV, ZCPH! specific heats of vaporisation, dry air
119REAL, DIMENSION(D%NIJT) :: ZDP ! pressure between LCL and model layer
120REAL, DIMENSION(D%NIJT) :: ZWORK1, ZWORK2 ! work arrays
121!
122REAL(KIND=jphook) :: zhook_handle
123!
124!-------------------------------------------------------------------------------
125!
126!* 0.3 Compute array bounds
127! --------------------
128!
129IF (lhook) CALL dr_hook('CONVECT_CLOSURE_THRVLCL',0,zhook_handle)
130ikb = 1 + cvpext%JCVEXB
131ike = d%NKT - cvpext%JCVEXT
132!
133!
134!* 1. Initialize local variables
135! --------------------------
136!
137zeps = cst%XRD / cst%XRV
138zcpord = cst%XCPD / cst%XRD
139zrdocp = cst%XRD / cst%XCPD
140!
141zdpthmix(:) = 0.
142zpresmix(:) = 0.
143pthlcl(:) = 300.
144ptlcl(:) = 300.
145ptelcl(:) = 300.
146prvlcl(:) = 0.
147pzlcl(:) = pz(:,ikb)
148ztmix(:) = 230.
149zplcl(:) = 1.e4
150klcl(:) = ikb + 1
151!
152!
153!* 2. Construct a mixed layer as in TRIGGER_FUNCT
154! -------------------------------------------
155!
156 jkmax=ike
157 jkmin=ikb
158 DO jk = ikb + 1, jkmax
159 jkm = jk + 1
160 DO ji = d%NIJB, d%NIJE
161 IF ( jk >= kdpl(ji) .AND. jk <= kpbl(ji) ) THEN
162 zwork1(ji) = ppres(ji,jk) - ppres(ji,jkm)
163 zdpthmix(ji) = zdpthmix(ji) + zwork1(ji)
164 zpresmix(ji) = zpresmix(ji) + ppres(ji,jk) * zwork1(ji)
165 pthlcl(ji) = pthlcl(ji) + pth(ji,jk) * zwork1(ji)
166 prvlcl(ji) = prvlcl(ji) + prv(ji,jk) * zwork1(ji)
167 END IF
168 END DO
169 END DO
170!
171!
172DO ji=d%NIJB,d%NIJE
173 IF ( owork1(ji) ) THEN
174 !
175 zpresmix(ji) = zpresmix(ji) / zdpthmix(ji)
176 pthlcl(ji) = pthlcl(ji) / zdpthmix(ji)
177 prvlcl(ji) = prvlcl(ji) / zdpthmix(ji)
178 !
179 !* 3.1 Use an empirical direct solution ( Bolton formula )
180 ! to determine temperature and pressure at LCL.
181 ! Nota: the adiabatic saturation temperature is not
182 ! equal to the dewpoint temperature
183 ! --------------------------------------------------
184 !
185 !
186 ztmix(ji) = pthlcl(ji) * ( zpresmix(ji) / cst%XP00 ) ** zrdocp
187 zevmix(ji) = prvlcl(ji) * zpresmix(ji) / ( prvlcl(ji) + zeps )
188 zevmix(ji) = max( 1.e-8, zevmix(ji) )
189 zwork1(ji) = log( zevmix(ji) / 613.3 )
190 ! dewpoint temperature
191 zwork1(ji) = ( 4780.8 - 32.19 * zwork1(ji) ) / ( 17.502 - zwork1(ji) )
192 ! adiabatic saturation temperature
193 ptlcl(ji) = zwork1(ji) - ( .212 + 1.571e-3 * ( zwork1(ji) - cst%XTT ) &
194 - 4.36e-4 * ( ztmix(ji) - cst%XTT ) ) * ( ztmix(ji) - zwork1(ji) )
195 ptlcl(ji) = min( ptlcl(ji), ztmix(ji) )
196 zplcl(ji) = cst%XP00 * ( ptlcl(ji) / pthlcl(ji) ) ** zcpord
197 END IF
198ENDDO
199!
200DO ji = d%NIJB,d%NIJE
201 zplcl(ji) = min( 2.e5, max( 10., zplcl(ji) ) ) ! bound to avoid overflow
202ENDDO
203!
204!
205!* 3.2 Correct PTLCL in order to be completely consistent
206! with MNH saturation formula
207! --------------------------------------------------
208!
209 DO ji=d%NIJB,d%NIJE
210 CALL convect_satmixratio( zplcl(ji), ptlcl(ji), zeps, zwork1(ji), zlv(ji), zwork2(ji), zcph(ji) )
211 ENDDO
212 DO ji=d%NIJB,d%NIJE
213 IF( owork1(ji) ) THEN
214 zwork2(ji) = zwork1(ji) / ptlcl(ji) * ( cst%XBETAW / ptlcl(ji) - cst%XGAMW ) ! dr_sat/dT
215 zwork2(ji) = ( zwork1(ji) - prvlcl(ji) ) / &
216 ( 1. + zlv(ji) / zcph(ji) * zwork2(ji) )
217 ptlcl(ji) = ptlcl(ji) - zlv(ji) / zcph(ji) * zwork2(ji)
218 END IF
219 ENDDO
220!
221!
222!* 3.3 If PRVLCL is oversaturated set humidity and temperature
223! to saturation values.
224! -------------------------------------------------------
225!
226 DO ji=d%NIJB,d%NIJE
227 CALL convect_satmixratio( zpresmix(ji), ztmix(ji), zeps, zwork1(ji), zlv(ji), zwork2(ji), zcph(ji) )
228 ENDDO
229 DO ji=d%NIJB,d%NIJE
230 IF( owork1(ji) .AND. prvlcl(ji) > zwork1(ji) ) THEN
231 zwork2(ji) = zwork1(ji) / ztmix(ji) * ( cst%XBETAW / ztmix(ji) - cst%XGAMW ) ! dr_sat/dT
232 zwork2(ji) = ( zwork1(ji) - prvlcl(ji) ) / &
233 ( 1. + zlv(ji) / zcph(ji) * zwork2(ji) )
234 ptlcl(ji) = ztmix(ji) + zlv(ji) / zcph(ji) * zwork2(ji)
235 prvlcl(ji) = prvlcl(ji) - zwork2(ji)
236 zplcl(ji) = zpresmix(ji)
237 pthlcl(ji) = ptlcl(ji) * ( cst%XP00 / zplcl(ji) ) ** zrdocp
238 END IF
239 ENDDO
240!
241!
242!* 4.1 Determine vertical loop index at the LCL
243! -----------------------------------------
244!
245 DO jk = jkmin, ike - 1
246 DO ji = d%NIJB, d%NIJE
247 IF ( zplcl(ji) <= ppres(ji,jk) .AND. owork1(ji) ) THEN
248 klcl(ji) = jk + 1
249 pzlcl(ji) = pz(ji,jk+1)
250 END IF
251 END DO
252 END DO
253!
254!
255!* 4.2 Estimate height and environmental temperature at LCL
256! ----------------------------------------------------
257!
258 DO ji = d%NIJB, d%NIJE
259 jk = klcl(ji)
260 jkm = jk - 1
261 zdp(ji) = log( zplcl(ji) / ppres(ji,jkm) ) / &
262 log( ppres(ji,jk) / ppres(ji,jkm) )
263 zwork1(ji) = pth(ji,jk) * ( ppres(ji,jk) / cst%XP00 ) ** zrdocp
264 zwork2(ji) = pth(ji,jkm) * ( ppres(ji,jkm) / cst%XP00 ) ** zrdocp
265 zwork1(ji) = zwork2(ji) + ( zwork1(ji) - zwork2(ji) ) * zdp(ji)
266 ! we compute the precise value of the LCL
267 ! The precise height is between the levels KLCL and KLCL-1.
268 zwork2(ji) = pz(ji,jkm) + ( pz(ji,jk) - pz(ji,jkm) ) * zdp(ji)
269 END DO
270 DO ji=d%NIJB,d%NIJE
271 IF( owork1(ji) ) THEN
272 ptelcl(ji) = zwork1(ji)
273 pzlcl(ji) = zwork2(ji)
274 END IF
275 ENDDO
276!
277!
278!
279IF (lhook) CALL dr_hook('CONVECT_CLOSURE_THRVLCL',1,zhook_handle)
280CONTAINS
281include "convect_satmixratio.h"
282!
283END SUBROUTINE convect_closure_thrvlcl
284END MODULE mode_convect_closure_thrvlcl
