104SUBROUTINE nebn_init(HPROGRAM, TFILENAM, LDNEEDNAM, KLUOUT, &
105 &LDDEFAULTVAL, LDREADNAM, LDCHECK, KPRINT)
106!!*** *NEBN_INIT* - Code needed to initialize the MODD_NEB_n module
107!!
108!!* PURPOSE
109!! -------
110!! Sets the default values, reads the namelist, performs the checks and prints
111!!
112!!* METHOD
113!! ------
114!! 0. Declarations
115!! 1. Declaration of arguments
116!! 2. Declaration of local variables
117!! 1. Default values
118!! 2. Namelist
119!! 3. Checks
120!! 4. Prints
121!!
122!! AUTHOR
123!! ------
124!! S. Riette
125!!
126!! MODIFICATIONS
127!! -------------
128!! Original Mar 2023
129!-------------------------------------------------------------------------------
130!
131!* 0. DECLARATIONS
132! ---------------
133!
134USE mode_posnam_phy, ONLY: posnam_phy
135USE mode_check_nam_val, ONLY: check_nam_val_char
136USE modd_io, ONLY: tfiledata
137!
138IMPLICIT NONE
139!
140!* 0.1. Declaration of arguments
141! ------------------------
142!
143CHARACTER(LEN=6), INTENT(IN) :: HPROGRAM
144TYPE(tfiledata), INTENT(IN) :: TFILENAM
145LOGICAL, INTENT(IN) :: LDNEEDNAM
146INTEGER, INTENT(IN) :: KLUOUT
147LOGICAL, OPTIONAL, INTENT(IN) :: LDDEFAULTVAL
148LOGICAL, OPTIONAL, INTENT(IN) :: LDREADNAM
149LOGICAL, OPTIONAL, INTENT(IN) :: LDCHECK
150INTEGER, OPTIONAL, INTENT(IN) :: KPRINT
152!
153!* 0.2 Declaration of local variables
154! ------------------------------
155!
156LOGICAL :: LLDEFAULTVAL, LLREADNAM, LLCHECK, LLFOUND
157INTEGER :: IPRINT
158
159lldefaultval=.true.
160llreadnam=.true.
161llcheck=.true.
162iprint=0
163IF(PRESENT(lddefaultval)) lldefaultval=lddefaultval
164IF(PRESENT(ldreadnam )) llreadnam =ldreadnam
165IF(PRESENT(ldcheck )) llcheck =ldcheck
166IF(PRESENT(kprint )) iprint =kprint
167!
168!* 1. DEFAULT VALUES
169! -----------------
170!
171IF(lldefaultval) THEN
172 !NOTES ON GENERAL DEFAULTS AND MODEL-SPECIFIC DEFAULTS :
173 !- General default values *MUST* remain unchanged.
174 !- To change the default value for a given application,
175 ! an "IF(HPROGRAM=='...')" condition must be used.
176
177 !Freezing between 0 and -20. Other possibilities are 0/-40 or -5/-25
178 xtmaxmix = 273.16
179 xtminmix = 253.16
180 lhgt_qs = .false.
181 cfrac_ice_adjust='S'
182 cfrac_ice_shallow_mf='S'
183 vsigqsat = 0.02
184 ccondens='CB02'
185 clambda3='CB'
186 lsubg_cond=.false.
187 lcondborn=.false.
188 lsigmas =.true.
189 lstatnw=.false.
190
191 IF(hprogram=='AROME') THEN
192 cfrac_ice_adjust='T'
193 cfrac_ice_shallow_mf='T'
194 vsigqsat=0.
195 lsigmas=.false.
196 ELSEIF(hprogram=='LMDZ') THEN
197 lsubg_cond=.true.
198 ENDIF
199ENDIF
200!
201!* 2. NAMELIST
202! -----------
203!
204IF(llreadnam) THEN
205 CALL posnam_phy(tfilenam, 'NAM_NEBN', ldneednam, llfound)
206 IF(llfound) READ(unit=tfilenam%NLU, nml=nam_nebn)
207ENDIF
208!
209!* 3. CHECKS
210! ---------
211!
212IF(llcheck) THEN
213 CALL check_nam_val_char(kluout, 'CFRAC_ICE_ADJUST', cfrac_ice_adjust, 'T', 'O', 'N', 'S')
214 CALL check_nam_val_char(kluout, 'CFRAC_ICE_SHALLOW_MF', cfrac_ice_shallow_mf, 'T', 'O', 'N', 'S')
215 CALL check_nam_val_char(kluout, 'CCONDENS', ccondens, 'CB02', 'GAUS')
216 CALL check_nam_val_char(kluout, 'CLAMBDA3', clambda3, 'CB', 'NONE')
217ENDIF
218!
219!* 3. PRINTS
220! ---------
221!
222IF(iprint>=1) THEN
223 WRITE(unit=kluout,nml=nam_nebn)
224ENDIF
225!
226END SUBROUTINE nebn_init
