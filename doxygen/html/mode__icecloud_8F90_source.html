19! Purpose:
20! calculate subgridscale fraction of supersaturation with respect to ice.
21! Method:
22! Assume a linear distubution of relative humidity and let the variability
23! of humidity be a function of model level thickness.
24! (Also a function of of humidity itself in the boundary layer)
25! Interface: subroutine ICECLOUD is called
26! ------------ from subroutine 'rain_ice'
27!
28! variable type content
29! ======== ==== =======
30!
31! INPUT arguments (arguments d'entree)
32!----------------------------------------------
33! PP : pressure at model level (Pa)
34! PZ : model level height (m)
35! PDZ : model level thickness (m)
36! PT : temperature (K)
37! PR : model level humidity mixing ratio (kg/kg)
38! PTSTEP : timestep
39! PPBLH : plantetary layer height (m) (negative value means unknown)
40! PWCLD : water and / mixed phase cloud cover (negative means unknown)
41! XW2D : quota between ice crystal concentration between dry and wet
42! part of a gridbox
43
44! OUTPUT arguments (arguments d'sortie)
45!---------------------------------------------
46! SIFRC : subgridscale fraction with supersaturation with respect to ice.
47! SSIO : Super-saturation with respect to ice in the
48! supersaturated fraction
49! SSIU : Sub-saturation with respect to ice in the sub-saturated
50! fraction
51! W2D : Factor used to get consistncy between the mean value of
52! the gridbox and parts of the gridbox
53! RSI : Saturation mixing ratio over ice
54
55TYPE(dimphyex_t), INTENT(IN) :: D
56TYPE(cst_t), INTENT(IN) :: CST
57TYPE(rain_ice_param_t), INTENT(IN) :: ICEP

63REAL, INTENT(IN) :: PTSTEP
64REAL, INTENT(IN) :: PPBLH
65REAL, INTENT(IN) :: PWCLD(D%NIJT,D%NKT)
66REAL, INTENT(IN) :: XW2D
67
68! OUTPUT arguments (arguments d'sortie)
69!---------------------------------------------
70REAL, INTENT(OUT) :: SIFRC(D%NIJT,D%NKT)
71REAL, INTENT(OUT) :: SSIO(D%NIJT,D%NKT)

73REAL, INTENT(OUT) :: W2D(D%NIJT,D%NKT)
74REAL, INTENT(OUT) :: RSI(D%NIJT,D%NKT)

76! Working variables:
77REAL :: ZSIGMAX,ZSIGMAY,ZSIGMAZ,ZXDIST,ZYDIST,&
78 & ZRHW,ZRHIN,ZDRHDZ,ZZ,ZRHDIST,ZRHLIM, &
79 & ZRHDIF,ZWCLD,ZI2W,ZRHLIMICE,ZRHLIMINV,ZA,ZRHI,ZR
80INTEGER :: JIJ, JK, IIJB, IIJE, IKTB, IKTE
81
82REAL(KIND=jphook) :: zhook_handle
83IF (lhook) CALL dr_hook('ICECLOUD',0,zhook_handle)
84!
85iijb=d%NIJB
86iije=d%NIJE
87iktb=d%NKTB
88ikte=d%NKTE
89!
90zsigmax=3.e-4 ! assumed rh variation in x axis direction
91zsigmay=zsigmax ! assumed rh variation in y axis direction
92zsigmaz=1.e-2
93
94!ZXDIST=DTHETA*110000.
95zxdist=2500.
96 ! gridsize in x axis (m) Avoid too low
97 ! since the model has a tendency to become
98 ! drier at high horizontal resolution
99 ! due to stronger vertical velocities.
100zydist=zxdist ! gridsize in y axis (m)
101DO jk = iktb, ikte
102 DO jij = iijb, iije
103 zr = max(0.,pr(jij,jk)*ptstep)
104 sifrc(jij,jk) = 0.
105 za = zr*pp(jij,jk)/(cst%XEPSILO + zr)
106 zrhw = za/esatw(icep%TIWMX, pt(jij,jk))
107 rsi(jij,jk) = qsatmx_tab(cst,icep%TIWMX,pp(jij,jk),pt(jij,jk),1.)
108 zrhi = za/esati(icep%TIWMX, pt(jij,jk))
109 zi2w = esatw(icep%TIWMX, pt(jij,jk))/esati(icep%TIWMX, pt(jij,jk))
110
111 ssiu(jij,jk) = min(zi2w,zrhi)
112 ssio(jij,jk) = ssiu(jij,jk)
113 w2d(jij,jk) = 1.
114 IF (pt(jij,jk)>273.1 .OR. zr<=0. .OR. esati(icep%TIWMX, pt(jij,jk)) >= pp(jij,jk)*0.5) THEN
115 ssiu(jij,jk) = ssiu(jij,jk) - 1.
116 ssio(jij,jk) = ssiu(jij,jk)
117 IF(pwcld(jij,jk)>=0.) sifrc(jij,jk) = pwcld(jij,jk)
118 ELSE

120 zrhin = max(0.05, min(1.,zrhw))

122 zdrhdz=zrhin*cst%XG /(pt(jij,jk)*cst%XRD)* &
123 & ( cst%XEPSILO*cst%XLVTT/(cst%XCPD*pt(jij,jk)) - 1.) ! correct
124! & ( ZEPSILO*XLSTT/(CST%XCPD*PT) -1.) ! incorrect
125! more exact
126! assumed rh variation in the z axis (rh/m) in the pbl .
127! Also possible to just use
128! zdrhdz=4.2e-4_jprb ! rh/m !
129
130 zz=0.
131 IF(ppblh < 0. )THEN ! Assume boundary layer height is not available
132 zz = min(1.,max(0.,pz(jij,jk)*0.001))
133 ELSE
134 IF(pz(jij,jk) > 35. .AND. pz(jij,jk) > ppblh) zz = 1.
135 ENDIF
136
137! 1.6e-2 rh/m means variations is of order 0.5 for a 1km dept.
138! sigmaz=4e-2 ! EO 140 lev.
139
140
141! Compute rh-variation is x,y,z direction as approxmately
142! independent, exept for the z variation in the pbl, where rh is
143! assumed to be fairly constantly increasing with height
144
145 zrhdist = sqrt( zxdist*zsigmax**2 + zydist*zsigmay**2 + &
146 & (1.-zz)* (pdz(jij,jk)*zdrhdz)**2 + zz*pdz(jij,jk)*zsigmaz**2)
147! z-variation of rh in the pbl z-variation of rh outside the pbl
148! Safety for very coarse vertical resolution:
149 IF(zz > 0.1) zrhdist = zrhdist/(1.+zrhdist)
150
151!!!! Note ZRHDIST is with respect to water ! !!!!!!!!!!!!
152
153 zrhlim = max(0.5,min(0.99,1.-0.5*zrhdist))
154
155 IF(pwcld(jij,jk) < 0.)THEN
156 ! Assume water/mixed-phase cloud cover from e.g.
157 ! statistical cloud scheme is not available
158 zrhdif = (1. - zrhw)/(1.0-zrhlim)
159 zrhdif = 1. - sqrt(max(0.,zrhdif))
160 zwcld = min(1.,max(zrhdif,0.0))
161 ELSE
162 zwcld = pwcld(jij,jk)
163! possible to backwards compute a critical relative humity consitent with
164! input cloudcover:
165! IF(PWCLD < 0.99 .AND. PWCLD > 0.01) ZRHLIM= 1. - (1.-ZRHW)/(1.-PWCLD)**2
166 ENDIF
167
168 sifrc(jij,jk) = zwcld
169
170! relation rhlim with respect to water to that of ice:
171!ZRHLIMICE = MAX(ZRHDMIN*ZI2W,1.+ ZI2W*( ZRHLIM - 1.))
172 zrhlimice = 1.+ zi2w*( zrhlim - 1.)
173
174 IF(zrhlim <= 0.999)THEN
175
176 ! compute a 1/(1-rhlim) constistant with lstmp(i,k):
177 zrhliminv = 1./(1. - zrhlimice)
178 zrhdif = (zrhi - zrhlimice)*zrhliminv
179
180 IF(zwcld==0.)THEN
181 sifrc(jij,jk) = min(1.,0.5*max(0.,zrhdif))
182 ELSE
183 za = 1. - 1./zi2w
184 sifrc(jij,jk) = min(1.,za*0.5/ (1. - zrhlim))
185 sifrc(jij,jk) = min(1.,zwcld + sifrc(jij,jk))
186 ENDIF
187 ENDIF
188
189 IF(sifrc(jij,jk) > 0.01) THEN
190 ssiu(jij,jk) = sifrc(jij,jk) + zrhlimice*(1.-sifrc(jij,jk))
191 ssio(jij,jk) = (zrhi - (1.- sifrc(jij,jk))*ssiu(jij,jk))/sifrc(jij,jk)
192 ELSE
193 sifrc(jij,jk) = 0.! to aviod mismatch with output variables
194 za = min(0.,zrhi-zrhlimice)
195 ssiu(jij,jk) = max(0.,sifrc(jij,jk) + zrhlimice*(1.-sifrc(jij,jk)) + 2.*za )
196 ENDIF
197 ssio(jij,jk) = min(zi2w,ssio(jij,jk))
198 ssiu(jij,jk) = max(0.,ssiu(jij,jk))
199
200! Transform from relative humidity to degree of saturation:
201 ssiu(jij,jk) = ssiu(jij,jk) - 1.
202 ssio(jij,jk) = ssio(jij,jk) - 1.
203
204 IF (xw2d > 1.) w2d(jij,jk) = 1./(1. - sifrc(jij,jk) + xw2d*sifrc(jij,jk))
205 END IF
206 ENDDO
207ENDDO
208
209IF (lhook) CALL dr_hook('ICECLOUD',1,zhook_handle)
210END SUBROUTINE icecloud
