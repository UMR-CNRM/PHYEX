49  FUNCTION delta(PA,PB,PX,PX1,PX2) RESULT(PDELTA)
50 ! PDELTA takes the value PA if PX<PX1, and PB if PX>=PX2
51 ! PDELTA is a cubic interpolation between PA and PB for PX between PX1 and PX2
52  IMPLICIT NONE
53  REAL, INTENT(IN) :: pa
54  REAL, INTENT(IN) :: pb
55  REAL, DIMENSION(:), INTENT(IN) :: px
56  REAL, INTENT(IN) :: px1
57  REAL, INTENT(IN) :: px2
58  REAL, DIMENSION(SIZE(PX,1)) :: pdelta
59  REAL :: za
60  za = 6.0*(pa-pb)/(px2-px1)**3
61  WHERE (px(:).LT.px1)
62  pdelta(:) = pa
63  ELSEWHERE (px(:).GE.px2)
64  pdelta(:) = pb
65  ELSEWHERE
66  pdelta(:) = pa + za*px1**2*(px1/6.0 - 0.5*px2) &
67  + za*px1*px2* (px(:)) &
68  - (0.5*za*(px1+px2))* (px(:)**2) &
69  + (za/3.0)* (px(:)**3)
70  END WHERE
71  RETURN
72 !
73  END FUNCTION delta

169 !
