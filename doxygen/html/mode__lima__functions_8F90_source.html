57 FUNCTION delta(PA,PB,PX,PX1,PX2) RESULT(PDELTA)
58USE yomhook, ONLY:lhook, dr_hook, jphook
59! PDELTA takes the value PA if PX<PX1, and PB if PX>=PX2
60! PDELTA is a cubic interpolation between PA and PB for PX between PX1 and PX2
61 IMPLICIT NONE
62 REAL, INTENT(IN) :: pa
63 REAL, INTENT(IN) :: pb
64 REAL, DIMENSION(:), INTENT(IN) :: px
65 REAL, INTENT(IN) :: px1
66 REAL, INTENT(IN) :: px2
67 REAL, DIMENSION(SIZE(PX,1)) :: pdelta
68REAL(kind=jphook) :: zhook_handle
69 REAL :: za
70 IF (lhook) CALL dr_hook('DELTA', 0, zhook_handle)
71 za = 6.0*(pa-pb)/(px2-px1)**3
72 WHERE (px(:).LT.px1)
73 pdelta(:) = pa
74 ELSEWHERE (px(:).GE.px2)
75 pdelta(:) = pb
76 ELSEWHERE
77 pdelta(:) = pa + za*px1**2*(px1/6.0 - 0.5*px2) &
78 + za*px1*px2* (px(:)) &
79 - (0.5*za*(px1+px2))* (px(:)**2) &
80 + (za/3.0)* (px(:)**3)
81 END WHERE
82 IF (lhook) CALL dr_hook('DELTA', 1, zhook_handle)
83 RETURN
84!
85 END FUNCTION delta

16213 CONTINUE

178SUBROUTINE gauher(X,W,N)
179 USE modd_precision, only: mnhreal64
180 USE yomhook, ONLY:lhook, dr_hook, jphook
181 INTEGER, INTENT(IN) :: N
182 INTEGER MAXIT
183 REAL, INTENT(OUT) :: W(N),X(N)

188!

191!

21311 CONTINUE

216 z=z1-p1/pp
217 IF(abs(z-z1).LE.eps)GOTO 1
21812 CONTINUE
2191 x(i)=z
220 x(n+1-i)=-z
221 pp=pp/pim4 ! NORMALIZATION
222 w(i)=2.0/(pp*pp)
223 w(n+1-i)=w(i)
22413 CONTINUE

22914 CONTINUE
