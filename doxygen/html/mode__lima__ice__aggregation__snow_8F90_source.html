66!* 0.2 Declarations of local variables :
67!
68REAL, DIMENSION(SIZE(PRIT)) :: ZZW1, ZZW2, ZZW3 ! work arrays
69!
70!-------------------------------------------------------------------------------
71!
72!
73!* 2.4 Aggregation of r_i on r_s: CIAGGS and RIAGGS
74! ---------------------------------------------------
75!
76zzw1(:) = 0.
77zzw2(:) = 0.
78zzw3(:) = 0.
79!
80p_ri_aggs(:) = 0.
81p_ci_aggs(:) = 0.
82!
83!
84IF (nmom_i.EQ.1) THEN
85 WHERE ( prit(:)>xrtmin(4) .AND. prst(:)>xrtmin(5) .AND. ldcompute(:) )
86 zzw1(:) = xfiaggs * exp( xcolexis*(pt(:)-xtt) ) &
87 * platham_iaggs(:) &
88 * prit(:) &
89 * pcst(:) * (1+(xfvelos/plbds(:))**xalphas)**(-xnus+xexiaggs/xalphas) &
90 * prhodref(:)**(-xcexvt+1.) &
91 * plbds(:)**xexiaggs
92!
93 p_ri_aggs(:) = - zzw1(:)
94 END WHERE
95ELSE
96 WHERE ( prit(:)>xrtmin(4) .AND. prst(:)>xrtmin(5) .AND. &
97 pcit(:)>xctmin(4) .AND. pcst(:)>xctmin(5) .AND. ldcompute(:) )
98 zzw1(:) = (plbdi(:) / plbds(:))**3
99 zzw2(:) = pcit(:)*pcst(:)*exp(xcolexis*(pt(:)-xtt))*prhodref(:) / (plbdi(:)**3)
100 zzw3(:) = zzw2(:)*(xaggs_clarge1+xaggs_clarge2*zzw1(:))
101!
102 p_ci_aggs(:) = - zzw3(:)
103!
104 zzw2(:) = zzw2(:) / plbdi(:)**xbi
105 zzw2(:) = zzw2(:)*(xaggs_rlarge1+xaggs_rlarge2*zzw1(:))
106!
107 p_ri_aggs(:) = - zzw2(:)
108 END WHERE
109END IF
110!
111!
