2759!
2760!
2761! Purpose : compute dQ(RAR,T) in the parameterization of Saunders and Peck (1998)

2765! ------------

2800! #################################################################
2801!
2802!
2803! Purpose : interpolate dQ from a lookup table at each gridpoint
2804!
2805!
2806!* 0. DECLARATIONS
2807! ------------
2808!
2809IMPLICIT NONE
2810!
2811INTEGER, INTENT(IN) :: KMICRO
2812INTEGER, INTENT(IN) :: KIND_TEMP, KIND_LWC
2813LOGICAL, DIMENSION(KMICRO), INTENT(IN) :: OMASK
2814REAL, DIMENSION(KMICRO), INTENT(IN) :: PLIQ ! effective water content or rime accretion rate
2815REAL, DIMENSION(KMICRO), INTENT(IN) :: PTEMP ! temperature

2817REAL, DIMENSION(KMICRO), INTENT(INOUT) :: PDQ ! interpolated dQ
2818!

2824!
2825!
2826!* 1. FIND THE INDEXES FOR RAR/EW AND T
2827! ---------------------------------
2828!
2829pdq(:) = 0.
2830!
2831igaux = 0
2832DO ii = 1, kmicro
2833 IF (omask(ii)) THEN
2834 igaux = igaux + 1
2835 i1(igaux) = ii
2836 END IF
2837END DO

2852 zdq_interp(ii) = pdq(i1(ii))

2859 zvect1(1:igaux) = zvect1(1:igaux) - real(ivect1(1:igaux))

2874 ENDWHERE
2875!

2878 zvect2(:) + 18. ) )

2885! ---------------------------
2886!

2890 DO ii = 1, igaux
2891 pdq(i1(ii)) = zdq_interp(ii)
2892 END DO
2893END IF
2894!
2895DEALLOCATE(zdq_interp)
2896DEALLOCATE(zvect1)
2897DEALLOCATE(zvect2)
