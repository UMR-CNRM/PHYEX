16! ###########################################################
17!
18!
19!!**** *ROTATE_WIND* - computes the wind components along the maximum slope
20!! direction and its normal direction in the first mass level.
21!!
22!! PURPOSE
23!! -------
24!!****
25! The purpose of this routine is to compute the wind component parallel
26! to the orography at the first mass level. The exact location where these
27! components are computed is the point of intersection between the normal
28! to the orography and the first mass-level hyper-plane at PDZZ(:,:,IKB)/2
29!
30!!** METHOD
31!! ------
32!! The values of the 3 cartesian components of the wind are determined
33!! by a bilinear interpolation between the 4 nearest points in the first
34!! mass-level hyper-plane. These points are found according to the signs of
35!! the slopes' sinus and cosinus. For each direction of interpolation, the
36!! two different localizations (mass or flux grids) are used to avoid
37!! lateral boundary problems.
38!! Then, the rotation is performed for the wind components. The rotation
39!! angle is the angle between the x axe and the maximum slope direction
40!! defined by the slope vector (dZs/dx , dZs/dy).
41!! Finally, the horizontal components are set at the marginal points
42!! according to cyclic boundary conditions because this is the only case
43!! where these points can be considered.
44!!
45!! EXTERNAL
46!! --------
47!! NONE
48!!
49!! IMPLICIT ARGUMENTS
50!! ------------------
51!!
52!! MODD_CONF : L2D switch for 2D model version
53!!
54!!
55!! REFERENCE
56!! ---------
57!! Book 1 of documentation (Chapter: Turbulence)
58!!
59!! AUTHOR
60!! ------
61!! Joel Stein * Meteo-France *
62!!
63!! MODIFICATIONS
64!! -------------
65!! Original 14/11/95
66!! Modifications: 15/05/96, (N. wood)
67!! take into account no slip conditions
68!! at the surface
69!! 14/02/01 (V. Masson)
70!! Slip condition at the surface restored
71!!

76USE modd_parameters, ONLY: jpvext
77USE modd_dimphyex, ONLY: dimphyex_t
78!
79! These macro are handled by pft_tool.py --craybyPassDOCONCURRENT applied on Cray Rules
80#ifdef MNH_COMPILER_CCE
81!$mnh_undef(LOOP)
82!$mnh_undef(OPENACC)
83#endif

85IMPLICIT NONE
86!
87!
88!* 0.1 declarations of arguments
89!
90TYPE(dimphyex_t), INTENT(IN) :: D
91REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PU,PV,PW ! cartesian components
92 ! of the wind
93REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PDIRCOSXW, PDIRCOSYW, PDIRCOSZW
94! Director Cosinus along x, y and z directions at surface w-point
95REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PCOSSLOPE ! cosinus of the angle
96 ! between i and the slope vector
97REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PSINSLOPE ! sinus of the angle
98 ! between i and the slope vector
99REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PDXX, PDYY, PDZZ
100 ! Metric coefficients
101REAL, DIMENSION(D%NIT,D%NJT), INTENT(OUT) :: PUSLOPE ! wind component along
102 ! the maximum slope direction
103REAL, DIMENSION(D%NIT,D%NJT), INTENT(OUT) :: PVSLOPE ! wind component along
104 ! the direction normal to the maximum slope one
105!
106!-------------------------------------------------------------------------------
107!
108! 0.2 declaration of local variables
109!
110INTEGER, DIMENSION(D%NIT,D%NJT) :: ILOC,JLOC
111 ! shift index to find the 4 nearest points in x and y directions
112REAL, DIMENSION(D%NIT,D%NJT) :: ZCOEFF,ZCOEFM, &
113 ! interpolation weigths for flux and mass locations
114 zuint,zvint,zwint, &
115 ! intermediate values of the cartesian components after x interp.
116 zufin,zvfin,zwfin, &
117 ! final values of the cartesian components after the 2 interp.
118 zwground
119 ! vertical velocity at the surface
120INTEGER :: IIB,IIE,IJB,IJE,IKB
121 ! index values for the Beginning or the End of the physical
122 ! domain in x,y and z directions
123INTEGER :: IIU,IJU
124 ! arrays' sizes for i and j indices
125INTEGER :: JI,JJ
126!
127!----------------------------------------------------------------------------
128!
129!* 1. PRELIMINARIES
130! -------------
131!
132!
133iib = 2
134ijb = 2
135iiu = SIZE(pu,1)
136iju = SIZE(pu,2)
137iie = iiu - 1
138ije = iju - 1
139ikb = 1+jpvext
140!
141!
142!$acc kernels
143!$mnh_expand_array(JI=1:IIU,JJ=1:IJU)
144puslope(:,:)=0.
145pvslope(:,:)=0.
146!
147zwground(:,:) = pw(:,:,ikb)
148!
149!* 2. INTERPOLATE THE CARTESIAN COMPONENTS
150! ------------------------------------
151!
152iloc(:,:)=nint(sign(1.,-pcosslope(:,:)))
153jloc(:,:)=nint(sign(1.,-psinslope(:,:)))
154!$mnh_end_expand_array(JI=1:IIU,JJ=1:IJU)
155!
156! interpolation in x direction
157!
158DO jj=1,iju
159 DO ji=iib,iie
160 zcoeff(ji,jj) = &
161 (0.5*pdxx(ji,jj,ikb) + 0.5*pdzz(ji,jj,ikb)*pdircosxw(ji,jj) ) &
162 * 2. / (pdxx(ji,jj,ikb)+pdxx(ji+1,jj,ikb))
163 zuint(ji,jj) = zcoeff(ji,jj) * pu(ji+1,jj,ikb) + &
164 (1.-zcoeff(ji,jj)) * pu(ji,jj,ikb)
165 !
166 zcoefm(ji,jj) = 1. - 0.5 * pdzz(ji,jj,ikb) * abs(pdircosxw(ji,jj)) &
167 / pdxx(ji+(iloc(ji,jj)+1)/2,jj,ikb)
168 zvint(ji,jj) = zcoefm(ji,jj) * pv(ji,jj,ikb) + &
169 (1.-zcoefm(ji,jj)) * pv(ji+iloc(ji,jj),jj,ikb)
170 !
171 zwint(ji,jj) = zcoefm(ji,jj) * (pw(ji,jj,ikb+1)+zwground(ji,jj)) * 0.5 &
172 + (1.-zcoefm(ji,jj)) &
173 *(pw(ji+iloc(ji,jj),jj,ikb+1)+zwground(ji+iloc(ji,jj),jj)) * 0.5
174 END DO
175END DO
176!
177! interpolation in y direction
178!
179DO jj=ijb,ije
180 DO ji=iib,iie
181 zcoeff(ji,jj) = &
182 (0.5*pdyy(ji,jj,ikb) + 0.5*pdzz(ji,jj,ikb)*pdircosyw(ji,jj) ) &
183 * 2. / (pdyy(ji,jj,ikb)+pdyy(ji+1,jj,ikb))
184 zvfin(ji,jj) = zcoeff(ji,jj) * zvint(ji,jj+1) + &
185 (1.-zcoeff(ji,jj)) * zvint(ji,jj)
186 !
187 zcoefm(ji,jj) = 1. - 0.5 * pdzz(ji,jj,ikb) * abs(pdircosyw(ji,jj)) &
188 / pdyy(ji,jj+(jloc(ji,jj)+1)/2,ikb)
189 zufin(ji,jj) = zcoefm(ji,jj) * zuint(ji,jj) + &
190 (1.-zcoefm(ji,jj)) * zuint(ji,jj+jloc(ji,jj))
191 zwfin(ji,jj) = zcoefm(ji,jj) * zwint(ji,jj) + &
192 (1.-zcoefm(ji,jj)) * zwint(ji,jj+jloc(ji,jj))
193 END DO
194END DO
195!
196!* 3. ROTATE THE WIND
197! ---------------
198!
199!
200!$mnh_expand_array(JI=IIB:IIE,JJ=IJB:IJE)
201 puslope(iib:iie,ijb:ije) = pcosslope(iib:iie,ijb:ije) * pdircoszw(iib:iie,ijb:ije) * zufin(iib:iie,ijb:ije) + &
202 psinslope(iib:iie,ijb:ije) * pdircoszw(iib:iie,ijb:ije) * zvfin(iib:iie,ijb:ije) + &
203 sqrt(1.-pdircoszw(iib:iie,ijb:ije)**2) * zwfin(iib:iie,ijb:ije)
204 !
205 pvslope(iib:iie,ijb:ije) =-psinslope(iib:iie,ijb:ije) * zufin(iib:iie,ijb:ije) + &
206 pcosslope(iib:iie,ijb:ije) * zvfin(iib:iie,ijb:ije)
207 !
208!$mnh_end_expand_array(JI=IIB:IIE,JJ=IJB:IJE)
209!
210!$acc end kernels
211!
212!
213!----------------------------------------------------------------------------
214!
215END SUBROUTINE rotate_wind

252!
253TYPE(list_ll), POINTER :: TZFIELDS_ll ! list of fields to exchange
254INTEGER :: IINFO_ll ! return code of parallel routine
255logical :: gwest, geast, gnorth, gsouth

257!

259!
260!* 1 PROLOGUE

267ije = d%NJE

269!
270!!$IF (NHALO == 1) THEN
271#ifndef MNH_OPENACC
272 CALL add2dfield_ll( tzfields_ll, puslope, 'UPDATE_ROTATE_WIND::PUSLOPE' )

290gnorth = hlbcy(2) /= "CYCL" .AND. lnorth_ll()
291
292!$acc kernels present_cr(PUSLOPE,PVSLOPE)
293IF ( gwest ) THEN
294 puslope(iib-1,:)=puslope(iib,:)
295 pvslope(iib-1,:)=pvslope(iib,:)
296END IF
297IF ( geast ) THEN
298 puslope(iie+1,:)=puslope(iie,:)
299 pvslope(iie+1,:)=pvslope(iie,:)
300END IF
301IF ( gsouth ) THEN
302 puslope(:,ijb-1)=puslope(:,ijb)
303 pvslope(:,ijb-1)=pvslope(:,ijb)
304END IF
305IF( gnorth ) THEN
306 puslope(:,ije+1)=puslope(:,ije)
307 pvslope(:,ije+1)=pvslope(:,ije)
308END IF
309!$acc end kernels
310!
311END SUBROUTINE update_rotate_wind
