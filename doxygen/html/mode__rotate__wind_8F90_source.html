19 !!**** *ROTATE_WIND* - computes the wind components along the maximum slope
20 !! direction and its normal direction in the first mass level.
21 !!
22 !! PURPOSE
23 !! -------
24 !!****
25 ! The purpose of this routine is to compute the wind component parallel
26 ! to the orography at the first mass level. The exact location where these
27 ! components are computed is the point of intersection between the normal
28 ! to the orography and the first mass-level hyper-plane at PDZZ(:,:,IKB)/2
29 !
30 !!** METHOD
31 !! ------
32 !! The values of the 3 cartesian components of the wind are determined
33 !! by a bilinear interpolation between the 4 nearest points in the first
34 !! mass-level hyper-plane. These points are found according to the signs of
35 !! the slopes' sinus and cosinus. For each direction of interpolation, the
36 !! two different localizations (mass or flux grids) are used to avoid
37 !! lateral boundary problems.
38 !! Then, the rotation is performed for the wind components. The rotation
39 !! angle is the angle between the x axe and the maximum slope direction
40 !! defined by the slope vector (dZs/dx , dZs/dy).
41 !! Finally, the horizontal components are set at the marginal points
42 !! according to cyclic boundary conditions because this is the only case
43 !! where these points can be considered.
44 !!
45 !! EXTERNAL
46 !! --------
47 !! NONE
48 !!
49 !! IMPLICIT ARGUMENTS
50 !! ------------------
51 !!
52 !! MODD_CONF : L2D switch for 2D model version
53 !!
54 !!
55 !! REFERENCE
56 !! ---------
57 !! Book 1 of documentation (Chapter: Turbulence)
58 !!
59 !! AUTHOR
60 !! ------
61 !! Joel Stein * Meteo-France *
62 !!
63 !! MODIFICATIONS
64 !! -------------
65 !! Original 14/11/95
66 !! Modifications: 15/05/96, (N. wood)
67 !! take into account no slip conditions
68 !! at the surface
69 !! 14/02/01 (V. Masson)
70 !! Slip condition at the surface restored

74 !* 0. DECLARATIONS
75 ! ------------
76 USE modd_parameters, ONLY: jpvext
77 USE modd_dimphyex, ONLY: dimphyex_t
78 !
79 IMPLICIT NONE
80 !
81 !
82 !* 0.1 declarations of arguments
83 !
84 TYPE(dimphyex_t), INTENT(IN) :: D
85 REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PU,PV,PW ! cartesian components
86  ! of the wind
87 REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PDIRCOSXW, PDIRCOSYW, PDIRCOSZW
88 ! Director Cosinus along x, y and z directions at surface w-point
89 REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PCOSSLOPE ! cosinus of the angle
90  ! between i and the slope vector
91 REAL, DIMENSION(D%NIT,D%NJT), INTENT(IN) :: PSINSLOPE ! sinus of the angle
92  ! between i and the slope vector
93 REAL, DIMENSION(D%NIT,D%NJT,D%NKT), INTENT(IN) :: PDXX, PDYY, PDZZ
94  ! Metric coefficients
95 REAL, DIMENSION(D%NIT,D%NJT), INTENT(OUT) :: PUSLOPE ! wind component along
96  ! the maximum slope direction
97 REAL, DIMENSION(D%NIT,D%NJT), INTENT(OUT) :: PVSLOPE ! wind component along
98  ! the direction normal to the maximum slope one
99 !
100 !-------------------------------------------------------------------------------
101 !
102 ! 0.2 declaration of local variables
103 !
104 INTEGER, DIMENSION(D%NIT,D%NJT) :: ILOC,JLOC
105  ! shift index to find the 4 nearest points in x and y directions
106 REAL, DIMENSION(D%NIT,D%NJT) :: ZCOEFF,ZCOEFM, &
107  ! interpolation weigths for flux and mass locations
108  zuint,zvint,zwint, &
109  ! intermediate values of the cartesian components after x interp.
110  zufin,zvfin,zwfin, &
111  ! final values of the cartesian components after the 2 interp.
112  zwground
113  ! vertical velocity at the surface
114 INTEGER :: IIB,IIE,IJB,IJE,IKB
115  ! index values for the Beginning or the End of the physical
116  ! domain in x,y and z directions
117 INTEGER :: IIU,IJU
118  ! arrays' sizes for i and j indices
119 INTEGER :: JI,JJ
120 !
121 !----------------------------------------------------------------------------
122 !
123 !* 1. PRELIMINARIES
124 ! -------------
125 !
126 puslope(:,:)=0.
127 pvslope(:,:)=0.
128 !
129 iib = 2
130 ijb = 2
131 iiu = SIZE(pu,1)
132 iju = SIZE(pu,2)
133 iie = iiu - 1
134 ije = iju - 1
135 ikb = 1+jpvext
136 !
137 zwground(:,:) = pw(:,:,ikb)
138 !
139 !* 2. INTERPOLATE THE CARTESIAN COMPONENTS
140 ! ------------------------------------
141 !
142 iloc(:,:)=nint(sign(1.,-pcosslope(:,:)))
143 jloc(:,:)=nint(sign(1.,-psinslope(:,:)))
144 !
145 ! interpolation in x direction
146 !
147 DO jj = 1,iju
148  DO ji = iib,iie
149  zcoeff(ji,jj) = &
150  (0.5*pdxx(ji,jj,ikb) + 0.5*pdzz(ji,jj,ikb)*pdircosxw(ji,jj) ) &
151  * 2. / (pdxx(ji,jj,ikb)+pdxx(ji+1,jj,ikb))
152  zuint(ji,jj) = zcoeff(ji,jj) * pu(ji+1,jj,ikb) + &
153  (1.-zcoeff(ji,jj)) * pu(ji,jj,ikb)
154  !
155  zcoefm(ji,jj) = 1. - 0.5 * pdzz(ji,jj,ikb) * abs(pdircosxw(ji,jj)) &
156  / pdxx(ji+(iloc(ji,jj)+1)/2,jj,ikb)
157  zvint(ji,jj) = zcoefm(ji,jj) * pv(ji,jj,ikb) + &
158  (1.-zcoefm(ji,jj)) * pv(ji+iloc(ji,jj),jj,ikb)
159  !
160  zwint(ji,jj) = zcoefm(ji,jj) * (pw(ji,jj,ikb+1)+zwground(ji,jj)) * 0.5 &
161  + (1.-zcoefm(ji,jj)) &
162  *(pw(ji+iloc(ji,jj),jj,ikb+1)+zwground(ji+iloc(ji,jj),jj)) * 0.5
163  END DO
164 END DO
165 !
166 ! interpolation in y direction
167 !
168 DO jj = ijb,ije
169  DO ji = iib,iie
170  zcoeff(ji,jj) = &
171  (0.5*pdyy(ji,jj,ikb) + 0.5*pdzz(ji,jj,ikb)*pdircosyw(ji,jj) ) &
172  * 2. / (pdyy(ji,jj,ikb)+pdyy(ji+1,jj,ikb))
173  zvfin(ji,jj) = zcoeff(ji,jj) * zvint(ji,jj+1) + &
174  (1.-zcoeff(ji,jj)) * zvint(ji,jj)
175  !
176  zcoefm(ji,jj) = 1. - 0.5 * pdzz(ji,jj,ikb) * abs(pdircosyw(ji,jj)) &
177  / pdyy(ji,jj+(jloc(ji,jj)+1)/2,ikb)
178  zufin(ji,jj) = zcoefm(ji,jj) * zuint(ji,jj) + &
179  (1.-zcoefm(ji,jj)) * zuint(ji,jj+jloc(ji,jj))
180  zwfin(ji,jj) = zcoefm(ji,jj) * zwint(ji,jj) + &
181  (1.-zcoefm(ji,jj)) * zwint(ji,jj+jloc(ji,jj))
182  END DO
183 END DO
184 !
185 !* 3. ROTATE THE WIND
186 ! ---------------
187 !
188 !
189 DO jj = ijb,ije
190  DO ji = iib,iie
191  puslope(ji,jj) = pcosslope(ji,jj) * pdircoszw(ji,jj) * zufin(ji,jj) + &
192  psinslope(ji,jj) * pdircoszw(ji,jj) * zvfin(ji,jj) + &
193  sqrt(1.-pdircoszw(ji,jj)**2) * zwfin(ji,jj)
194  !
195  pvslope(ji,jj) =-psinslope(ji,jj) * zufin(ji,jj) + &
196  pcosslope(ji,jj) * zvfin(ji,jj)
197  !
198  END DO
199 END DO
200 !
201 !
202 !
203 !----------------------------------------------------------------------------
204 !
205 END SUBROUTINE rotate_wind
