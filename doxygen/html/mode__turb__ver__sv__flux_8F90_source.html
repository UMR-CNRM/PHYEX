22 !!**** *TURB_VER_SV_FLUX* -compute the source terms due to the vertical turbulent
23 !! fluxes.
24 !!
25 !! PURPOSE
26 !! -------
27 ! The purpose of this routine is to compute the vertical turbulent
28 ! fluxes of the evolutive variables and give back the source
29 ! terms to the main program. In the case of large horizontal meshes,
30 ! the divergence of these vertical turbulent fluxes represent the whole
31 ! effect of the turbulence but when the three-dimensionnal version of
32 ! the turbulence scheme is activated (CTURBDIM="3DIM"), these divergences
33 ! are completed in the next routine TURB_HOR.
34 ! An arbitrary degree of implicitness has been implemented for the
35 ! temporal treatment of these diffusion terms.
36 ! The vertical boundary conditions are as follows:
37 ! * at the bottom, the surface fluxes are prescribed at the same
38 ! as the other turbulent fluxes
39 ! * at the top, the turbulent fluxes are set to 0.
40 ! It should be noted that the condensation has been implicitely included
41 ! in this turbulence scheme by using conservative variables and computing
42 ! the subgrid variance of a statistical variable s indicating the presence
43 ! or not of condensation in a given mesh.
44 !
45 !!** METHOD
46 !! ------
47 !! 1D type calculations are made;
48 !! The vertical turbulent fluxes are computed in an off-centered
49 !! implicit scheme (a Crank-Nicholson type with coefficients different
50 !! than 0.5), which allows to vary the degree of implicitness of the
51 !! formulation.
52 !! The different prognostic variables are treated one by one.
53 !! The contributions of each turbulent fluxes are cumulated into the
54 !! tendency PRvarS, and into the dynamic and thermal production of
55 !! TKE if necessary.
56 !!
57 !! In section 2 and 3, the thermodynamical fields are considered.
58 !! Only the turbulent fluxes of the conservative variables
59 !! (Thetal and Rnp stored in PRx(:,:,:,1)) are computed.
60 !! Note that the turbulent fluxes at the vertical
61 !! boundaries are given either by the soil scheme for the surface one
62 !! ( at the same instant as the others fluxes) and equal to 0 at the
63 !! top of the model. The thermal production is computed by vertically
64 !! averaging the turbulent flux and multiply this flux at the mass point by
65 !! a function ETHETA or EMOIST, which preform the transformation from the
66 !! conservative variables to the virtual potential temperature.
67 !!
68 !! In section 4, the variance of the statistical variable
69 !! s indicating presence or not of condensation, is determined in function
70 !! of the turbulent moments of the conservative variables and its
71 !! squarred root is stored in PSIGS. This information will be completed in
72 !! the horizontal turbulence if the turbulence dimensionality is not
73 !! equal to "1DIM".
74 !!
75 !! In section 5, the x component of the stress tensor is computed.
76 !! The surface flux <u'w'> is computed from the value of the surface
77 !! fluxes computed in axes linked to the orography ( i", j" , k"):
78 !! i" is parallel to the surface and in the direction of the maximum
79 !! slope
80 !! j" is also parallel to the surface and in the normal direction of
81 !! the maximum slope
82 !! k" is the normal to the surface
83 !! In order to prevent numerical instability, the implicit scheme has
84 !! been extended to the surface flux regarding to its dependence in
85 !! function of U. The dependence in function of the other components
86 !! introduced by the different rotations is only explicit.
87 !! The turbulent fluxes are used to compute the dynamic production of
88 !! TKE. For the last TKE level ( located at PDZZ(:,:,IKB)/2 from the
89 !! ground), an harmonic extrapolation from the dynamic production at
90 !! PDZZ(:,:,IKB) is used to avoid an evaluation of the gradient of U
91 !! in the surface layer.
92 !!
93 !! In section 6, the same steps are repeated but for the y direction
94 !! and in section 7, a diagnostic computation of the W variance is
95 !! performed.
96 !!
97 !! In section 8, the turbulent fluxes for the scalar variables are
98 !! computed by the same way as the conservative thermodynamical variables
99 !!
100 !!
101 !! EXTERNAL
102 !! --------
103 !! GX_U_M, GY_V_M, GZ_W_M : cartesian gradient operators
104 !! GX_U_UW,GY_V_VW (X,Y,Z) represent the direction of the gradient
105 !! _(M,U,...)_ represent the localization of the
106 !! field to be derivated
107 !! _(M,UW,...) represent the localization of the
108 !! field derivated
109 !!
110 !!
111 !! MXM,MXF,MYM,MYF,MZM,MZF
112 !! : Shuman functions (mean operators)
113 !! DXF,DYF,DZF,DZM
114 !! : Shuman functions (difference operators)
115 !!
116 !! SUBROUTINE TRIDIAG : to compute the split implicit evolution
117 !! of a variable located at a mass point
118 !!
119 !! SUBROUTINE TRIDIAG_WIND: to compute the split implicit evolution
120 !! of a variable located at a wind point
121 !!
122 !! FUNCTIONs ETHETA and EMOIST :
123 !! allows to compute:
124 !! - the coefficients for the turbulent correlation between
125 !! any variable and the virtual potential temperature, of its
126 !! correlations with the conservative potential temperature and
127 !! the humidity conservative variable:
128 !! ------- ------- -------
129 !! A' Thv' = ETHETA A' Thl' + EMOIST A' Rnp'
130 !!
131 !!

347 CALL mzm_phy(d,prhodj,zmzmrhodj)
348 DO jsv=1,ksv
349 !
350  IF (onomixlg .AND. jsv >= ksv_lgbeg .AND. jsv<= ksv_lgend) cycle
351 !
352 ! Preparation of the arguments for TRIDIAG
353  IF (turbn%LHARAT) THEN
354  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
355  za(iijb:iije,1:ikt) = -ptstep * zkeff(iijb:iije,1:ikt) * zmzmrhodj(iijb:iije,1:ikt) &
356  / pdzz(iijb:iije,1:ikt)**2
357  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
358  ELSE

362  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
363  ENDIF

366 ! Compute the sources for the JSVth scalar variable
367 
368 !* in 3DIM case, a part of the flux goes vertically, and another goes horizontally

377  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
378  ELSE
379  !$mnh_expand_array(JIJ=IIJB:IIJE)
380  zsource(iijb:iije,ikb) = (turbn%XIMPL*psfsvp(iijb:iije,jsv) + pexpl*psfsvm(iijb:iije,jsv)) / &
381  pdzz(iijb:iije,ikb) / pdircoszw(iijb:iije) &
382  * 0.5 * (1. + prhodj(iijb:iije,ika) / prhodj(iijb:iije,ikb))

384  END IF

392  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
393  prsvs(iijb:iije,1:ikt,jsv)= prsvs(iijb:iije,1:ikt,jsv)+ &
394  prhodj(iijb:iije,1:ikt)*(zres(iijb:iije,1:ikt)-psvm(iijb:iije,1:ikt,jsv))/ptstep
395  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
396 !
397  IF ( (turbn%LTURB_FLX .AND. tpfile%LOPENED) .OR. tles%LLES_CALL .OR. oflyer ) THEN
398  ! Diagnostic of the cartesian vertical flux
399  !
400  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
401  zwork1(iijb:iije,1:ikt) = plm(iijb:iije,1:ikt)*sqrt(ptkem(iijb:iije,1:ikt))
402  zwork2(iijb:iije,1:ikt) = turbn%XIMPL*zres(iijb:iije,1:ikt) + pexpl*psvm(iijb:iije,1:ikt,jsv)
403  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
404  CALL mzm_phy(d,zwork1,zwork3)
405  CALL dzm_phy(d,zwork2,zwork4)
406  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
407  zflxz(iijb:iije,1:ikt) = -zcsv * ppsi_sv(iijb:iije,1:ikt,jsv) * zwork3(iijb:iije,1:ikt) &
408  / pdzz(iijb:iije,1:ikt) * &
409  zwork4(iijb:iije,1:ikt)
410  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
411  ! surface flux
412  !* in 3DIM case, a part of the flux goes vertically, and another goes horizontally
413  ! (in presence of slopes)
414  !* in 1DIM case, the part of energy released in horizontal flux
415  ! is taken into account in the vertical part
416  IF (turbn%CTURBDIM=='3DIM') THEN
417  !$mnh_expand_array(JIJ=IIJB:IIJE)
418  zflxz(iijb:iije,ikb) = (turbn%XIMPL*psfsvp(iijb:iije,jsv) + pexpl*psfsvm(iijb:iije,jsv)) &
419  * pdircoszw(iijb:iije)
420  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
421  ELSE
422  !$mnh_expand_array(JIJ=IIJB:IIJE)
423  zflxz(iijb:iije,ikb) = (turbn%XIMPL*psfsvp(iijb:iije,jsv) + pexpl*psfsvm(iijb:iije,jsv)) &
424  / pdircoszw(iijb:iije)
425  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
426  END IF
427  ! extrapolates the flux under the ground so that the vertical average with
428  ! the IKB flux gives the ground value
429  !
430  !$mnh_expand_array(JIJ=IIJB:IIJE)
431  zflxz(iijb:iije,ika) = zflxz(iijb:iije,ikb)
432  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
433 
434  IF ( oflyer ) THEN
435  DO jk=iktb+1,ikte-1
436  !$mnh_expand_array(JIJ=IIJB:IIJE)
437  pwsv(iijb:iije,jk,jsv)=0.5*(zflxz(iijb:iije,jk)+zflxz(iijb:iije,jk+ikl))
438  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
439  END DO
440  !$mnh_expand_array(JIJ=IIJB:IIJE)
441  pwsv(iijb:iije,ikb,jsv)=0.5*(zflxz(iijb:iije,ikb)+zflxz(iijb:iije,ikb+ikl))
442  pwsv(iijb:iije,ike,jsv)=pwsv(iijb:iije,ike-ikl,jsv)
443  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
444  END IF
445  END IF

468  CALL second_mnh(ztime1)
