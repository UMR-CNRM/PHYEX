23 !!**** *TURB_VER_DYN_FLUX* -compute the source terms due to the vertical turbulent
24 !! fluxes.
25 !!
26 !! PURPOSE
27 !! -------
28 ! The purpose of this routine is to compute the vertical turbulent
29 ! fluxes of the evolutive variables and give back the source
30 ! terms to the main program. In the case of large horizontal meshes,
31 ! the divergence of these vertical turbulent fluxes represent the whole
32 ! effect of the turbulence but when the three-dimensionnal version of
33 ! the turbulence scheme is activated (CTURBDIM="3DIM"), these divergences
34 ! are completed in the next routine TURB_HOR.
35 ! An arbitrary degree of implicitness has been implemented for the
36 ! temporal treatment of these diffusion terms.
37 ! The vertical boundary conditions are as follows:
38 ! * at the bottom, the surface fluxes are prescribed at the same
39 ! as the other turbulent fluxes
40 ! * at the top, the turbulent fluxes are set to 0.
41 ! It should be noted that the condensation has been implicitely included
42 ! in this turbulence scheme by using conservative variables and computing
43 ! the subgrid variance of a statistical variable s indicating the presence
44 ! or not of condensation in a given mesh.
45 !
46 !!** METHOD
47 !! ------
48 !! 1D type calculations are made;
49 !! The vertical turbulent fluxes are computed in an off-centered
50 !! implicit scheme (a Crank-Nicholson type with coefficients different
51 !! than 0.5), which allows to vary the degree of implicitness of the
52 !! formulation.
53 !! The different prognostic variables are treated one by one.
54 !! The contributions of each turbulent fluxes are cumulated into the
55 !! tendency PRvarS, and into the dynamic and thermal production of
56 !! TKE if necessary.
57 !!
58 !! In section 2 and 3, the thermodynamical fields are considered.
59 !! Only the turbulent fluxes of the conservative variables
60 !! (Thetal and Rnp stored in PRx(:,:,:,1)) are computed.
61 !! Note that the turbulent fluxes at the vertical
62 !! boundaries are given either by the soil scheme for the surface one
63 !! ( at the same instant as the others fluxes) and equal to 0 at the
64 !! top of the model. The thermal production is computed by vertically
65 !! averaging the turbulent flux and multiply this flux at the mass point by
66 !! a function ETHETA or EMOIST, which preform the transformation from the
67 !! conservative variables to the virtual potential temperature.
68 !!
69 !! In section 4, the variance of the statistical variable
70 !! s indicating presence or not of condensation, is determined in function
71 !! of the turbulent moments of the conservative variables and its
72 !! squarred root is stored in PSIGS. This information will be completed in
73 !! the horizontal turbulence if the turbulence dimensionality is not
74 !! equal to "1DIM".
75 !!
76 !! In section 5, the x component of the stress tensor is computed.
77 !! The surface flux <u'w'> is computed from the value of the surface
78 !! fluxes computed in axes linked to the orography ( i", j" , k"):
79 !! i" is parallel to the surface and in the direction of the maximum
80 !! slope
81 !! j" is also parallel to the surface and in the normal direction of
82 !! the maximum slope
83 !! k" is the normal to the surface
84 !! In order to prevent numerical instability, the implicit scheme has
85 !! been extended to the surface flux regarding to its dependence in
86 !! function of U. The dependence in function of the other components
87 !! introduced by the different rotations is only explicit.
88 !! The turbulent fluxes are used to compute the dynamic production of
89 !! TKE. For the last TKE level ( located at PDZZ(:,:,IKB)/2 from the
90 !! ground), an harmonic extrapolation from the dynamic production at
91 !! PDZZ(:,:,IKB) is used to avoid an evaluation of the gradient of U
92 !! in the surface layer.
93 !!
94 !! In section 6, the same steps are repeated but for the y direction
95 !! and in section 7, a diagnostic computation of the W variance is
96 !! performed.
97 !!
98 !! In section 8, the turbulent fluxes for the scalar variables are
99 !! computed by the same way as the conservative thermodynamical variables
100 !!
101 !!
102 !! EXTERNAL
103 !! --------
104 !! GX_U_M, GY_V_M, GZ_W_M : cartesian gradient operators
105 !! GX_U_UW,GY_V_VW (X,Y,Z) represent the direction of the gradient
106 !! _(M,U,...)_ represent the localization of the
107 !! field to be derivated
108 !! _(M,UW,...) represent the localization of the
109 !! field derivated
110 !!
111 !!
112 !! MXM,MXF,MYM,MYF,MZM,MZF
113 !! : Shuman functions (mean operators)
114 !! DXF,DYF,DZF,DZM
115 !! : Shuman functions (difference operators)
116 !!
117 !! SUBROUTINE TRIDIAG_WIND: to compute the split implicit evolution
118 !! of a variable located at a wind point
119 !!
120 !! FUNCTIONs ETHETA and EMOIST :
121 !! allows to compute:
122 !! - the coefficients for the turbulent correlation between
123 !! any variable and the virtual potential temperature, of its
124 !! correlations with the conservative potential temperature and
125 !! the humidity conservative variable:
126 !! ------- ------- -------
127 !! A' Thv' = ETHETA A' Thl' + EMOIST A' Rnp'

462  zsource(iijb:iije,ikb) = &
463  ( zwork41d(iijb:iije) &
464  + zwork61d(iijb:iije) &
465  - zcoefs(iijb:iije) * pum(iijb:iije,ikb) * turbn%XIMPL &
466  ) * 0.5 * ( 1. + zwork1(iijb:iije,ika) / zwork1(iijb:iije,ikb) )
467  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
468 !
469  zsource(iijb:iije,ike) = 0.
470 ENDIF
471 !
472 ! Obtention of the split U at t+ deltat
473 !
474 CALL tridiag_wind(d,pum,za,zcoefs,ptstep,pexpl,turbn%XIMPL, &
475  zwork1,zsource,zres)
476 !
477 ! Compute the equivalent tendency for the U wind component
478 !
479 CALL mxm_phy(d,prhodj,zwork1)
480 CALL mxm_phy(d,zkeff,zwork2)
481 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
482 zwork3(iijb:iije,1:ikt)=turbn%XIMPL*zres(iijb:iije,1:ikt) + pexpl*pum(iijb:iije,1:ikt)
483 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
484 CALL dzm_phy(d,zwork3,zwork4)
485 CALL mxm_phy(d,pdzz,zwork5)
486 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
487 prus(iijb:iije,1:ikt)= prus(iijb:iije,1:ikt)+zwork1(iijb:iije,1:ikt)*(zres(iijb:iije,1:ikt) &
488  - pum(iijb:iije,1:ikt))/ptstep
489 !
490 !* 5.2 Partial TKE Dynamic Production
491 !
492 ! vertical flux of the U wind component
493 !
494 zflxz(iijb:iije,1:ikt) = -zcmfs * zwork2(iijb:iije,1:ikt) * zwork4(iijb:iije,1:ikt) &
495  / zwork5(iijb:iije,1:ikt)
496 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
497 !
498 IF (oocean) THEN
499  zflxz(iijb:iije,ike+1) = zfluxsfcu(iijb:iije)
500 ELSE
501  ! surface flux
502  CALL mxm_phy(d,pdzz,zwork1)
503  CALL mxm_phy(d,prhodj,zwork2)
504  !$mnh_expand_array(JIJ=IIJB:IIJE)
505  zflxz(iijb:iije,ikb) = zwork1(iijb:iije,ikb) * &
506  ( zsource(iijb:iije,ikb) &
507  +zcoefs(iijb:iije) * zres(iijb:iije,ikb) * turbn%XIMPL &
508  ) / 0.5 / ( 1. + zwork2(iijb:iije,ika)/ zwork2(iijb:iije,ikb) )
509  !
510  zflxz(iijb:iije,ika) = zflxz(iijb:iije,ikb)
511  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
512 END IF
513 !
514 IF ( turbn%LTURB_FLX .AND. tpfile%LOPENED ) THEN
515  ! stores the U wind component vertical flux
516  tzfield = tfieldmetadata( &
517  cmnhname = 'UW_VFLX', &
518  cstdname = '', &
519  clongname = 'UW_VFLX', &
520  cunits = 'm2 s-2', &
521  cdir = 'XY', &
522  ccomment = 'U wind component vertical flux', &
523  ngrid = 4, &
524  ntype = typereal, &
525  ndims = 3, &
526  ltimedep = .true. )
527  CALL io_field_write_phy(d,tpfile,tzfield,zflxz)
528 END IF
529 !
530 ! first part of total momentum flux
531 !
532 pwu(iijb:iije,1:ikt) = zflxz(iijb:iije,1:ikt)
533 !
534 ! Contribution to the TKE dynamic production of TKE
535 ! (computed at mass point)
536 !
537 CALL gz_u_uw_phy(d,pum,pdzz,zwork1)
538 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
539 zwork2(iijb:iije,1:ikt) = zflxz(iijb:iije,1:ikt) * zwork1(iijb:iije,1:ikt)
540 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
541 CALL mxf_phy(d,zwork2,zwork3)
542 CALL mzf_phy(d,zwork3,zwork4)
543 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
544 pdp(iijb:iije,1:ikt) = -zwork4(iijb:iije,1:ikt)
545 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
546 !
547 ! Special cases near surface
548 CALL mxm_phy(d,pdzz,zwork1)
549 IF (oocean) THEN
550  ! evaluate the dynamic production at w(IKE) and store in PDP(IKE)
551  ! before to be extrapolated in tke_eps routine
552  !$mnh_expand_array(JIJ=IIJB:IIJE)
553  zwork2(iijb:iije,ike) = zflxz(iijb:iije,ike) * (pum(iijb:iije,ike)-pum(iijb:iije,ike-ikl)) &
554  / zwork1(iijb:iije,ike-ikl)
555  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
556  CALL mxf_phy(d,zwork2,zwork3)
557  !$mnh_expand_array(JIJ=IIJB:IIJE)
558  pdp(iijb:iije,ike) = -zwork3(iijb:iije,ike)
559  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
560 ELSE ! Atmosphere
561  ! evaluate the dynamic production at w(IKB+KKL) in PDP(IKB)
562  !$mnh_expand_array(JIJ=IIJB:IIJE)
563  zwork2(iijb:iije,ikb) = zflxz(iijb:iije,ikb+ikl) * (pum(iijb:iije,ikb+ikl)-pum(iijb:iije,ikb)) &
564  / zwork1(iijb:iije,ikb+ikl)
565  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
566  CALL mxf_phy(d,zwork2,zwork3)
567  !$mnh_expand_array(JIJ=IIJB:IIJE)
568  pdp(iijb:iije,ikb) = -zwork3(iijb:iije,ikb)
569  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
570 !
571 END IF
572 !
573 ! Storage in the LES configuration
574 !
575 IF (tles%LLES_CALL) THEN
576  CALL second_mnh(ztime1)
577  !
578  CALL mxf_phy(d,zflxz,zwork1)
579  CALL mzf_phy(d,zwork1,zwork2)
580  CALL les_mean_subgrid_phy(d,tles,zwork2, tles%X_LES_SUBGRID_WU )
581  !
582  CALL gz_u_uw_phy(d,pum,pdzz,zwork1)
583  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
584  zwork1(iijb:iije,1:ikt) = zwork1(iijb:iije,1:ikt) * zflxz(iijb:iije,1:ikt)
585  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
586  CALL mxf_phy(d,zwork1,zwork2)
587  CALL mzf_phy(d,zwork2,zwork3)
588  CALL les_mean_subgrid_phy(d,tles,zwork3, tles%X_LES_RES_ddxa_U_SBG_UaU )
589  !
590  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
591  zwork1(iijb:iije,1:ikt) = zcmfs * zkeff(iijb:iije,1:ikt)
592  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
593  CALL les_mean_subgrid_phy(d,tles, zwork1, tles%X_LES_SUBGRID_Km )
594  !
595  CALL second_mnh(ztime2)
596  tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
597 END IF
598 !
599 !* 5.3 Source of W wind component
600 !
601 !
602 IF(turbn%CTURBDIM=='3DIM') THEN
603  ! Compute the source for the W wind component
604  ! used to compute the W source at the ground
605  !$mnh_expand_array(JIJ=IIJB:IIJE)
606  zflxz(iijb:iije,ika) = 2 * zflxz(iijb:iije,ikb) - zflxz(iijb:iije,ikb+ikl) ! extrapolation
607  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
608  IF (oocean) THEN
609  !$mnh_expand_array(JIJ=IIJB:IIJE)
610  zflxz(iijb:iije,iku) = 2 * zflxz(iijb:iije,ike) - zflxz(iijb:iije,ike-ikl) ! extrapolation
611  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
612  END IF
613  !
614  CALL mxm_phy(d,prhodj,zwork1)
615  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
616  zwork1(iijb:iije,1:ikt) = zwork1(iijb:iije,1:ikt) / pdxx(iijb:iije,1:ikt)
617  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
618  CALL mzm_phy(d,zwork1,zwork2)
619  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
620  zwork2(iijb:iije,1:ikt) = zwork2(iijb:iije,1:ikt) * zflxz(iijb:iije,1:ikt)
621  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
622  CALL dxf_phy(d,zwork2,zwork1)
623  !
624  IF (.NOT. oflat) THEN
625  !
626  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
627  zwork2(iijb:iije,1:ikt) = zflxz(iijb:iije,1:ikt)*pdzx(iijb:iije,1:ikt)
628  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
629  CALL mzf_phy(d,zwork2,zwork3)
630  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
631  zwork3(iijb:iije,1:ikt) = zwork3(iijb:iije,1:ikt) / pdxx(iijb:iije,1:ikt)
632  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
633  CALL mxf_phy(d,zwork3,zwork2)
634  CALL mzf_phy(d,pdzz,zwork3)
635  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
636  zwork3(iijb:iije,1:ikt) = prhodj(iijb:iije,1:ikt) &
637  / zwork3(iijb:iije,1:ikt) * zwork2(iijb:iije,1:ikt)
638  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
639  CALL dzm_phy(d,zwork3,zwork2)
640  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
641  prws(iijb:iije,1:ikt) = prws(iijb:iije,1:ikt) - zwork1(iijb:iije,1:ikt) &
642  + zwork2(iijb:iije,1:ikt)
643  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
644  ELSE
645  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
646  prws(iijb:iije,1:ikt)= prws(iijb:iije,1:ikt) - zwork1(iijb:iije,1:ikt)
647  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
648  END IF
649  !
650  ! Complete the TKE dynamical production with the W wind contribution
651  !
652  CALL gx_w_uw_phy(d,oflat,pwm,pdxx,pdzz,pdzx, zwork1)
653  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
654  zwork1(iijb:iije,1:ikt) = zwork1(iijb:iije,1:ikt) * zflxz(iijb:iije,1:ikt)
655  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
656  CALL mxf_phy(d,zwork1,zwork2)
657  CALL mzf_phy(d,zwork2,zwork3)
658  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
659  za(iijb:iije,1:ikt) = -zwork3(iijb:iije,1:ikt)
660  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
661  !
662  ! Special cases near surface
663  CALL dxm_phy(d,pwm,zwork1)
664  IF (oocean) THEN
665  ! evaluate the dynamic production at w(IKE) in PDP(IKE)
666  !
667  !$mnh_expand_array(JIJ=IIJB:IIJE)
668  zwork31d(iijb:iije) = - zflxz(iijb:iije,ike) * zwork1(iijb:iije,ike) &
669  / (0.5*(pdxx(iijb:iije,ike-ikl)+pdxx(iijb:iije,ike)))
670  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
671  CALL mxf2d_phy(d,zwork31d,zwork41d)
672  za(iijb:iije,ike) = zwork41d(iijb:iije)
673  !
674  ELSE !Atmosphere
675  ! evaluate the dynamic production at w(IKB+IKL) in PDP(IKB)
676  !
677  !$mnh_expand_array(JIJ=IIJB:IIJE)
678  zwork21d(iijb:iije) = (pwm(iijb:iije,ikb+2*ikl)-pwm(iijb:iije,ikb+ikl)) &
679  / (pdzz(iijb:iije,ikb+2*ikl)+pdzz(iijb:iije,ikb+ikl)) &
680  + (pwm(iijb:iije,ikb+ikl)-pwm(iijb:iije,ikb)) &
681  / (pdzz(iijb:iije,ikb+ikl)+pdzz(iijb:iije,ikb))
682  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
683  !
684  CALL mxm2d_phy(d,zwork21d,zwork51d)
685  !$mnh_expand_array(JIJ=IIJB:IIJE)
686  zwork31d(iijb:iije) = - zflxz(iijb:iije,ikb+ikl) &
687  * ( zwork1(iijb:iije,ikb+ikl) - zwork51d(iijb:iije) &
688  * pdzx(iijb:iije,ikb+ikl) ) &
689  / (0.5*(pdxx(iijb:iije,ikb+ikl)+pdxx(iijb:iije,ikb)))
690  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
691  CALL mxf2d_phy(d,zwork31d,zwork41d)
692  za(iijb:iije,ikb) = zwork41d(iijb:iije)
693  !
694  END IF
695  !
696  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
697  pdp(iijb:iije,1:ikt)=pdp(iijb:iije,1:ikt)+za(iijb:iije,1:ikt)
698  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
699  !
700  ! Storage in the LES configuration
701  !
702  IF (tles%LLES_CALL) THEN
703  CALL second_mnh(ztime1)
704  !
705  CALL gx_w_uw_phy(d,oflat,pwm,pdxx,pdzz,pdzx,zwork1)
706  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
707  zwork1(iijb:iije,1:ikt) = zwork1(iijb:iije,1:ikt)*zflxz(iijb:iije,1:ikt)
708  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
709  CALL mxf_phy(d,zwork1,zwork2)
710  CALL mzf_phy(d,zwork2,zwork1)
711  CALL les_mean_subgrid_phy(d,tles,zwork1, tles%X_LES_RES_ddxa_W_SBG_UaW )
712  !
713  CALL gx_m_u_phy(d,oflat,pthlm,pdxx,pdzz,pdzx,zwork1)
714  CALL mzf_phy(d,zflxz,zwork2)
715  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
716  zwork2(iijb:iije,1:ikt) = zwork2(iijb:iije,1:ikt) * zwork1(iijb:iije,1:ikt)
717  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
718  CALL mxf_phy(d,zwork2,zwork1)
719  CALL les_mean_subgrid_phy(d,tles,zwork1, tles%X_LES_RES_ddxa_Thl_SBG_UaW )
720  !
721  IF (krr>=1) THEN
722  CALL gx_u_m_phy(d,oflat,prm(:,:,1),pdxx,pdzz,pdzx,zwork1)
723  CALL mzf_phy(d,zflxz,zwork2)
724  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
725  zwork1(iijb:iije,1:ikt) = zwork1(iijb:iije,1:ikt) * zwork2(iijb:iije,1:ikt)
726  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
727  CALL mxf_phy(d,zwork1,zwork2)
728  CALL les_mean_subgrid_phy(d,tles,zwork2,tles%X_LES_RES_ddxa_Rt_SBG_UaW )
729  END IF
730  DO jsv=1,ksv
731  CALL gx_u_m_phy(d,oflat,psvm(:,:,jsv),pdxx,pdzz,pdzx,zwork1)
732  CALL mzf_phy(d,zflxz,zwork2)
733  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
734  zwork1(iijb:iije,1:ikt) = zwork1(iijb:iije,1:ikt) * zwork2(iijb:iije,1:ikt)
735  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
736  CALL mxf_phy(d,zwork1,zwork2)
737  CALL les_mean_subgrid_phy(d,tles,zwork2,tles%X_LES_RES_ddxa_Sv_SBG_UaW(:,:,:,jsv) )
738  END DO
739  CALL second_mnh(ztime2)
740  tles%XTIME_LES = tles%XTIME_LES + ztime2 - ztime1
741  END IF
742 END IF

745 !
746 !
747 !* 6. SOURCES OF V,W WIND COMPONENTS AND COMPLETE 1D TKE DYNAMIC PRODUCTION
748 ! -----------------------------------------------------------------
749 !
750 !* 6.1 Source of V wind component
751 !
752 ! Preparation of the arguments for TRIDIAG_WIND
753 !!
754 CALL mym_phy(d,zkeff,zwork1)
755 CALL mym_phy(d,pdzz,zwork2)
756 CALL mzm_phy(d,prhodj,zwork3)
757 CALL mym_phy(d,zwork3,zwork4)
758 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
759 za(iijb:iije,1:ikt) = -ptstep * zcmfs * zwork1(iijb:iije,1:ikt)* zwork4(iijb:iije,1:ikt) &
760  / zwork2(iijb:iije,1:ikt)**2
761 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
762 !
763 !
764 !
765 ! Compute the source of V wind component
766 ! compute the coefficient between the vertical flux and the 2 components of the
767 ! wind following the slope
768 !$mnh_expand_array(JIJ=IIJB:IIJE)
769 zcoefflxu(iijb:iije) = pcdueff(iijb:iije) * (pdircoszw(iijb:iije)**2 - zdirsinzw(iijb:iije)**2) &
770  * psinslope(iijb:iije)

772 
773 ! prepare the implicit scheme coefficients for the surface flux
774 zcoefs(iijb:iije)= zcoefflxu(iijb:iije) * psinslope(iijb:iije) * pdircoszw(iijb:iije) &
775  +zcoefflxv(iijb:iije) * pcosslope(iijb:iije)
776 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
777 !
778 ! average this flux to be located at the V,W vorticity point
779 !$mnh_expand_array(JIJ=IIJB:IIJE)
780 zwork11d(iijb:iije)=zcoefs(iijb:iije) / pdzz(iijb:iije,ikb)
781 !$mnh_end_expand_array(JIJ=IIJB:IIJE)
782 CALL mym2d_phy(d,zwork11d,zcoefs)
783 !
784 ! No flux in SOURCE TERM NULL OUTSIDE BC
785 zsource(iijb:iije,ikb+1:ike-1) = 0.
786 ! Surface case
787 CALL mym_phy(d,prhodj,zwork1)
788 IF (oocean) THEN ! Ocean case
789  zcoefflxu(iijb:iije) = pcdueff(iijb:iije)
790  zcoefflxv(iijb:iije) = pcdueff(iijb:iije)
791  zcoefs(iijb:iije)=zcoefflxu(iijb:iije)
792  ! average this flux to be located at the U,W vorticity point
793  !$mnh_expand_array(JIJ=IIJB:IIJE)
794  zwork11d(iijb:iije) = zcoefs(iijb:iije) / pdzz(iijb:iije,ike)
795  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
796  CALL mym2d_phy(d,zwork11d,zcoefs)
797  !
798  zwork11d(iijb:iije) = zfluxsfcv(iijb:iije) / pdzz(iijb:iije,ike)
799  CALL mym2d_phy(d,zwork11d,zwork21d)
800  !
801  !$mnh_expand_array(JIJ=IIJB:IIJE)
802  zsource(iijb:iije,ike) = zwork21d(iijb:iije) &
803  *0.5 * ( 1. + zwork1(iijb:iije,iku) / zwork1(iijb:iije,ike))
804  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
805  !No flux at the ocean domain bottom
806  zsource(iijb:iije,ikb) = 0.
807 !
808 ELSE ! Atmos case
809 !
810  !$mnh_expand_array(JIJ=IIJB:IIJE)
811  zwork31d(iijb:iije) = zcoefflxu(iijb:iije) / pdzz(iijb:iije,ikb) &
812  *zuslopem(iijb:iije) &
813  +zcoefflxv(iijb:iije) / pdzz(iijb:iije,ikb) &
814  *zvslopem(iijb:iije)
815  !$mnh_end_expand_array(JIJ=IIJB:IIJE)
816  CALL mym2d_phy(d,zwork31d,zwork61d)
817  !
818  ! compute the explicit tangential flux at the W point
819  !$mnh_expand_array(JIJ=IIJB:IIJE)
820  zsource(iijb:iije,ikb) = &
