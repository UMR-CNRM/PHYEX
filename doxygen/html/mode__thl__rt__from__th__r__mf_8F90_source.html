72 !-------------------------------------------------------------------------------

75 !
76 
77 !----------------------------------------------------------------------------
78 REAL, DIMENSION(D%NIJT,D%NKT) :: ZCP, ZT
79 REAL, DIMENSION(D%NIJT,D%NKT) :: ZLVOCPEXN, ZLSOCPEXN
80 INTEGER :: JRR, JIJ, JK
81 INTEGER :: IIJB,IIJE ! physical horizontal domain indices
82 INTEGER :: IKT
83 REAL(KIND=jphook) :: zhook_handle
84 !----------------------------------------------------------------------------
85 !
86 !
87 IF (lhook) CALL dr_hook('THL_RT_FRM_TH_R_MF',0,zhook_handle)
88 !
89 iije=d%NIJE
90 iijb=d%NIJB
91 ikt=d%NKT
92 !
93 !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
94 !temperature
95 zt(iijb:iije,:) = pth(iijb:iije,:) * pexn(iijb:iije,:)
96 
97 !Cp
98 zcp(iijb:iije,:)=cst%XCPD
99 IF (krr > 0) zcp(iijb:iije,:) = zcp(iijb:iije,:) + cst%XCPV * pr(iijb:iije,:,1)
100 !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
101 DO jrr = 2,1+krrl ! loop on the liquid components
102  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
103  zcp(iijb:iije,:) = zcp(iijb:iije,:) + cst%XCL * pr(iijb:iije,:,jrr)
104  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
105 END DO
106 DO jrr = 2+krrl,1+krrl+krri ! loop on the solid components
107  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
108  zcp(iijb:iije,:) = zcp(iijb:iije,:) + cst%XCI * pr(iijb:iije,:,jrr)
109  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
110 END DO
111 
112 IF ( krrl >= 1 ) THEN
113  IF ( krri >= 1 ) THEN
114  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
115  !ZLVOCPEXN and ZLSOCPEXN
116  zlvocpexn(iijb:iije,:)=(cst%XLVTT + (cst%XCPV-cst%XCL) * (zt(iijb:iije,:)-cst%XTT) ) &
117  &/ zcp(iijb:iije,:) / pexn(iijb:iije,:)
118  zlsocpexn(iijb:iije,:)=(cst%XLSTT + (cst%XCPV-cst%XCI) * (zt(iijb:iije,:)-cst%XTT) ) &
119  &/ zcp(iijb:iije,:) / pexn(iijb:iije,:)
120  ! Rnp
121  prt(iijb:iije,:) = pr(iijb:iije,:,1) + pr(iijb:iije,:,2) + pr(iijb:iije,:,4)
122  ! Theta_l
123  pthl(iijb:iije,:) = pth(iijb:iije,:) - zlvocpexn(iijb:iije,:) * pr(iijb:iije,:,2) &
124  - zlsocpexn(iijb:iije,:) * pr(iijb:iije,:,4)
125  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
126  ELSE
127  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
128  !ZLVOCPEXN
129  zlvocpexn(iijb:iije,:)=(cst%XLVTT + (cst%XCPV-cst%XCL) * (zt(iijb:iije,:)-cst%XTT) ) &
130  &/ zcp(iijb:iije,:) / pexn(iijb:iije,:)
131  ! Rnp
132  prt(iijb:iije,:) = pr(iijb:iije,:,1) + pr(iijb:iije,:,2)
133  ! Theta_l
134  pthl(iijb:iije,:) = pth(iijb:iije,:) - zlvocpexn(iijb:iije,:) * pr(iijb:iije,:,2)
135  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
136  END IF
137 ELSE
138  !$mnh_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
139  ! Rnp = rv
140  prt(iijb:iije,:) = pr(iijb:iije,:,1)
141  ! Theta_l = Theta
142  pthl(iijb:iije,:) = pth(iijb:iije,:)
143  !$mnh_end_expand_array(JIJ=IIJB:IIJE,JK=1:IKT)
144 END IF
145 IF (lhook) CALL dr_hook('THL_RT_FRM_TH_R_MF',1,zhook_handle)
