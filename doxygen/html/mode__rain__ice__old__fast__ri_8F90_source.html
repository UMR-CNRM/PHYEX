116 IF (buconf%LBUDGET_TH) CALL budget_store_init_phy(d, tbudgets(nbudget_th), 'IMLT', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
117 IF (buconf%LBUDGET_RC) CALL budget_store_init_phy(d, tbudgets(nbudget_rc), 'IMLT', unpack(prcs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))

120 zzw(:) = 0.0
121 DO jk = 1, ksize
122 IF ((pris(jk)>0.0) .AND. (pzt(jk)>cst%XTT)) THEN
123 zzw(jk) = pris(jk)
124 prcs(jk) = prcs(jk) + pris(jk)
125 pzths(jk) = pzths(jk) - pris(jk)*(plsfact(jk)-plvfact(jk)) ! f(L_f*(-RIMLTC))
126 pris(jk) = 0.0

132 IF (buconf%LBUDGET_RC) CALL budget_store_end_phy(d, tbudgets(nbudget_rc), 'IMLT', unpack(prcs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
133 IF (buconf%LBUDGET_RI) CALL budget_store_end_phy(d, tbudgets(nbudget_ri), 'IMLT', unpack(pris(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
134
135!* 7.2 Bergeron-Findeisen effect: RCBERI
136
137 IF (buconf%LBUDGET_TH) CALL budget_store_init_phy(d, tbudgets(nbudget_th), 'BERFI', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
138 IF (buconf%LBUDGET_RC) CALL budget_store_init_phy(d, tbudgets(nbudget_rc), 'BERFI', unpack(prcs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
139 IF (buconf%LBUDGET_RI) CALL budget_store_init_phy(d, tbudgets(nbudget_ri), 'BERFI', unpack(pris(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
140
141 zzw(:) = 0.0
142 IF(ocnd2)THEN
143
144 ! Sub gridscale decomposition into a supersaturation part of the gridbox,
145 ! PSIFRC with a superaturation PSSIO and a subsaturated part (1.- PSIFRC)
146 ! with a (negative) superaturation of PSSIU
147
148 IF (lmodicedep) THEN
149
150 DO jl=1,ksize
151 zzw2(jl) = max(pcit(jl),icenumber2(pris(jl)*ptstep,pzt(jl))* &
152 prhodref(jl))
153 ENDDO
154
155 DO jk = 1, ksize
156 IF (zzw2(jk)>0.0 .AND. pesi(jk) < pres(jk)*0.5) THEN
157 zzw(jk)= icep%X0DEPI/(iced%XLBI*pai(jk)) *(zzw2(jk)/prhodref(jk))**(1.+iced%XLBEXI) * &
158 & (ptstep*max(iced%XRTMIN(4)/ptstep,pris(jk))*pw2d(jk) )**(-iced%XLBEXI)
159 zzw(jk)= max(-pris(jk)*pw2d(jk)*(1.-psifrc(jk))+zzw(jk)*pssio(jk)* psifrc(jk)* pxw2d13(jk), &
160 & zzw(jk)* ( pssio(jk)* psifrc(jk)* pxw2d13(jk) + pcitred23*pssiu(jk)* (1.-psifrc(jk)) ))
161
162 pris(jk) = pris(jk) + zzw(jk)
163 prvs(jk) = prvs(jk) - zzw(jk) ! Budget here: ! cloud ice + vapor = const
164 pzths(jk) = pzths(jk) + zzw(jk)*plsfact(jk) ! f(L_f*(RCBERI))
165 END IF
166 END DO
167 ELSE
168
169 DO jk=1,ksize
170
171 ztc = max(-18.,min(-1.,pzt(jk)-cst%XTT))
172 zhu = min(0.15,max(0.,pssi(jk)))
173 zcrysha(jk)=1.1+ 3.*zhu*(1.+ sin(0.64*ztc -1.3))
174! icedensity*4/3 *pi /8. =366.5 ; icedensity=700 kg/m3
175 zqimax = 366.5 * pdicrit**3 * pcit(jk)*pcitred/prhodref(jk)
176 zci2s(jk) = 0.
177 IF(pris(jk)*ptstep > 1.0e-12)THEN
178 zci2s(jk) = pris(jk)*(1. - min(1., 0.5*zqimax /pris(jk)/ptstep))* &
179 & (1.-psifrc(jk))*pw2d(jk)
180 ENDIF
181 ENDDO
182
183 DO jk = 1, ksize
184 IF (pcit(jk)>0.0 .AND. pesi(jk) < pres(jk)*0.5) THEN
185 zzwc(jk)=zcrysha(jk)*0.878/pai(jk)*(pcit(jk)/prhodref(jk))**0.667 &
186 &*(max(iced%XRTMIN(4)/ptstep,pris(jk))*ptstep*pw2d(jk))**0.333
187! Ice supersaturated part of grid box:
188 IF (pssio(jk)>0. .AND. psifrc(jk) > 0.02_jprb) THEN
189 zzw(jk) = zzwc(jk)*pxw2d13(jk)*pssio(jk)
190 pris(jk) = pris(jk) + zzw(jk)*psifrc(jk)
191 prvs(jk) = prvs(jk) - zzw(jk)*psifrc(jk) ! Budget here: ! cloud ice + vapor = const
192 pzths(jk) = pzths(jk) + zzw(jk)*plsfact(jk)*psifrc(jk) ! f(L_f*(RCBERI))
193 END IF
194! Ice subsaturated part of grid box:
195 IF (pssiu(jk)<0. .AND. psifrc(jk) <0.98_jprb) THEN
196 pris(jk) = pris(jk) - zci2s(jk)
197 prss(jk) = prss(jk) + zci2s(jk)
198 zzw(jk) = zzwc(jk)*pcitred23*pssiu(jk)
199 pris(jk) = pris(jk) + zzw(jk)*(1.-psifrc(jk))
200 prvs(jk) = prvs(jk) - zzw(jk)*(1.-psifrc(jk))
201 pzths(jk) = pzths(jk) + zzw(jk)*plsfact(jk)*(1.-psifrc(jk))
202 END IF
203 END IF
204 END DO
205 ENDIF
206
207 ELSE ! End OCND2
208
209 DO jk = 1, ksize
210 IF ((prcs(jk) > 0.0) .AND. &
211 (pssi(jk) > 0.0) .AND. &
212 (prit(jk) > iced%XRTMIN(4)) .AND. &
213 (pcit(jk) > 0.0)) THEN
214 zzw(jk) = min(1.e8,iced%XLBI*( prhodref(jk)*prit(jk)/pcit(jk) )**iced%XLBEXI) ! Lbda_i
215 zzw(jk) = min( prcs(jk),( pssi(jk) / (prhodref(jk)*pai(jk)) ) * pcit(jk) * &
216 ( icep%X0DEPI/zzw(jk) + icep%X2DEPI*pcj(jk)*pcj(jk)/zzw(jk)**(iced%XDI+2.0) ) )
217 prcs(jk) = prcs(jk) - zzw(jk)
218 pris(jk) = pris(jk) + zzw(jk)
219 pzths(jk) = pzths(jk) + zzw(jk)*(plsfact(jk)-plvfact(jk))
220 END IF
221 END DO
222 ENDIF
223
224 IF (buconf%LBUDGET_TH) CALL budget_store_end_phy(d, tbudgets(nbudget_th), 'BERFI', unpack(pzths(:),mask=gmicro(:,:),field=pths)*prhodj(:,:))
225 IF (buconf%LBUDGET_RC) CALL budget_store_end_phy(d, tbudgets(nbudget_rc), 'BERFI', unpack(prcs(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
226 IF (buconf%LBUDGET_RI) CALL budget_store_end_phy(d, tbudgets(nbudget_ri), 'BERFI', unpack(pris(:)*pzrhodj(:),mask=gmicro(:,:),field=0.0))
227
228 IF (lhook) CALL dr_hook('RAIN_ICE_OLD:RAIN_ICE_FAST_RI',1,zhook_handle)
