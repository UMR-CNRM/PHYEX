72 !NOTES ON SIZES
73 !If we pack:
74 ! - KSIZE is the number of relevant point (with mixing ratio different from 0)
75 ! - KPROMA is the size of bloc of points
76 ! - ZSIZE2 has the same value as KPROMA
77 !If we do not pack:
78 ! - KSIZE is the total number of points
79 ! - KPROMA is null for memory saving
80 ! - KSIZE2 has the same value as KSIZE
81 !
82 !When we do not pack, we can transmit directly the 3D arrays to the ice4_stepping subroutine, we do not need
83 !to copy the values. It is why KPROMA is null because we do not need these arrays.
84 !But some arrays must me manipulated before being transmitted and we need temporary arrays for this.
85 !KSIZE2 is used for those arrays that must be dimensioned KPROMA if we pack or with the total size if not.

136 TYPE(tbudgetdata), DIMENSION(KBUDGETS), INTENT(INOUT) :: TBUDGETS
137 INTEGER, INTENT(IN) :: KBUDGETS

145 !

147 !
148 REAL(KIND=jphook) :: zhook_handle
149 !
150 INTEGER :: JIJ, JK
151 INTEGER :: IKT, IKTB, IKTE, IIJT, IIJB, IIJE
152 INTEGER :: ISTIJ, ISTK
153 !
154 LOGICAL :: GEXT_TEND
155 !
156 !Output packed total mixing ratio change (for budgets only)
157 REAL, DIMENSION(KSIZE, IBUNUM-IBUNUM_EXTRA) :: ZBU_PACK
158 !
159 !For packing
160 INTEGER :: IMICRO ! Case r_x>0 locations
161 INTEGER :: JL, JV
162 REAL, DIMENSION(KPROMA) :: &
163  & ZCIT, & ! Pristine ice conc. at t
164  & ZRHODREF, & ! RHO Dry REFerence
165  & ZPRES, & ! Pressure
166  & ZEXN, & ! EXNer Pressure
167  & ZCF, & ! Cloud fraction
168  & ZHLC_HCF, & ! HLCLOUDS : fraction of High Cloud Fraction in grid
169  & ZHLC_HRC, & ! HLCLOUDS : LWC that is High LWC in grid
170  & ZHLI_HCF, &
171  & ZHLI_HRI, &
172  & ZRAINFR, &
173  & ZRREVAV
174 REAL, DIMENSION(KSIZE2) :: ZSIGMA_RC ! Standard deviation of rc at time t
175 LOGICAL, DIMENSION(KPROMA) :: LLMICRO
176 !
177 !Output packed tendencies (for budgets only)
178 REAL, DIMENSION(KPROMA, IBUNUM-IBUNUM_EXTRA) :: ZBU_SUM
179 !
180 !For mixing-ratio-splitting
181 REAL, DIMENSION(KPROMA,0:7) :: ZVART !Packed variables
182 REAL, DIMENSION(KSIZE2,0:7) :: ZEXTPK !To take into acount external tendencies inside the splitting
183 !
184 !For retroaction of E on IAGGS
185 REAL, DIMENSION(MERGE(KPROMA,0,OELEC)) :: ZLATHAM_IAGGS
186 !
187 INTEGER, DIMENSION(KPROMA) :: I1,I2 ! Used to replace the COUNT and PACK intrinsics on variables
188 INTEGER, DIMENSION(KSIZE) :: I1TOT, I2TOT ! Used to replace the COUNT and PACK intrinsics
189 !
190 INTEGER :: IC, JMICRO
191 LOGICAL :: LLSIGMA_RC, LL_AUCV_ADJU
192 !
193 !-------------------------------------------------------------------------------
194 IF (lhook) CALL dr_hook('ICE4_PACK', 0, zhook_handle)
195 !
196 !* 1. GENERALITIES
197 ! ------------
198 !
199 ikt=d%NKT
200 iktb=d%NKTB
201 ikte=d%NKTE
202 iijt=d%NIJT
203 iijb=d%NIJB
204 iije=d%NIJE
205 gext_tend=.true.
206 llsigma_rc=(parami%CSUBG_AUCV_RC=='PDF ' .AND. parami%CSUBG_PR_PDF=='SIGM')
207 ll_aucv_adju=(parami%CSUBG_AUCV_RC=='ADJU' .OR. parami%CSUBG_AUCV_RI=='ADJU')
208 !
209 IF(parami%LPACK_MICRO) THEN
210  IF(buconf%LBU_ENABLE .OR. osave_micro) THEN
211  DO jv=1, ibunum-ibunum_extra
212  zbu_pack(:, jv)=0.
213  ENDDO
214  ENDIF
215  !
216  !* 2. POINT SELECTION
217  ! ---------------
218  !
219  ! optimization by looking for locations where
220  ! the microphysical fields are larger than a minimal value only !!!
221  !
222  IF (ksize /= count(odmicro(iijb:iije,iktb:ikte))) THEN
223  CALL print_msg(nverb_fatal, 'GEN', 'ICE4_PACK', 'ICE4_PACK : KSIZE /= COUNT(ODMICRO)')
224  ENDIF
225 
226  IF (ksize > 0) THEN
227  !
228  !* 3. CACHE-BLOCKING LOOP
229  ! -------------------
230  !
231 
232  ! starting indexes :
233  ic=0
234  istk=iktb
235  istij=iijb
236 
237  DO jmicro=1,ksize,kproma
238 
239  imicro=min(kproma,ksize-jmicro+1)
240  !
241  !* 4. PACKING
242  ! -------
243  !
244 
245  ! Setup packing parameters
246  outer_loop: DO jk = istk, ikte
247  IF (any(odmicro(iijb:iije,jk))) THEN
248  DO jij = istij, iije
249  IF (odmicro(jij,jk)) THEN
250  ic=ic+1
251  llmicro(ic)=.true.
252  ! Initialization of variables in packed format :
253  zvart(ic, ith)=pwr(jij, jk, ith)
254  zvart(ic, irv)=pwr(jij, jk, irv)
255  zvart(ic, irc)=pwr(jij, jk, irc)
256  zvart(ic, irr)=pwr(jij, jk, irr)
257  zvart(ic, iri)=pwr(jij, jk, iri)
258  zvart(ic, irs)=pwr(jij, jk, irs)
259  zvart(ic, irg)=pwr(jij, jk, irg)
260  IF (krr==7) THEN
261  zvart(ic, irh)=pwr(jij, jk, irh)
262  ENDIF
263  IF (gext_tend) THEN
264  !The th tendency is not related to a mixing ratio change, there is no exn/exnref issue here
265  zextpk(ic, ith)=pths(jij, jk)
266  zextpk(ic, irv)=prvs(jij, jk)
267  zextpk(ic, irc)=prcs(jij, jk)
268  zextpk(ic, irr)=prrs(jij, jk)
269  zextpk(ic, iri)=pris(jij, jk)
270  zextpk(ic, irs)=prss(jij, jk)
271  zextpk(ic, irg)=prgs(jij, jk)
272  IF (krr==7) THEN
273  zextpk(ic, irh)=prhs(jij, jk)
274  ENDIF
275  ENDIF
276  zcit(ic)=pcit(jij, jk)
277  zcf(ic)=pcldfr(jij, jk)
278  zrhodref(ic)=prhodref(jij, jk)
279  zpres(ic)=ppabst(jij, jk)
280  zexn(ic)=pexn(jij, jk)
281  IF(llsigma_rc) THEN
282  zsigma_rc(ic)=psigs(jij, jk)
283  ENDIF
284  IF (ll_aucv_adju) THEN
285  zhlc_hcf(ic) = phlc_hcf(jij, jk)
286  zhlc_hrc(ic) = phlc_hrc(jij, jk)
287  zhli_hcf(ic) = phli_hcf(jij, jk)
288  zhli_hri(ic) = phli_hri(jij, jk)
289  ENDIF
290  zrainfr(ic)=prainfr(jij, jk)
291  IF (oelec) zlatham_iaggs(ic) = platham_iaggs(jij, jk)
292  ! Save indices for later usages:
293  i1(ic) = jij
294  i2(ic) = jk
295  IF(buconf%LBU_ENABLE .OR. osave_micro) THEN
296  i1tot(jmicro+ic-1)=jij
297  i2tot(jmicro+ic-1)=jk
298  ENDIF
299  IF (ic==imicro) THEN
300  ! the end of the chunk has been reached, then reset the starting index :
301  istij=jij+1
302  IF (istij <= iije) THEN
303  istk=jk
304  ELSE
305  ! end of line, restart from 1 and increment upper loop
306  istij=d%NIJB
307  istk=jk+1
308  IF (istk > ikte) THEN
309  ! end of line, restart from 1
310  istk=iktb
311  ENDIF
312  ENDIF
313  ic=0
314  EXIT outer_loop
315  ENDIF
316  ENDIF
317  ENDDO
318  ENDIF
319  ! restart inner loop on JIJ :
320  istij=iijb
321  ENDDO outer_loop
322 
323  !
324  !* 5. TENDENCIES COMPUTATION
325  ! ----------------------
326  !
327  CALL ice4_stepping(d, cst, parami, icep, iced, buconf, &
328  &llsigma_rc, ll_aucv_adju, gext_tend, &
329  &kproma, imicro, llmicro, ptstep, &
330  &krr, osave_micro, oelec, &
331  &zexn, zrhodref, i1, i2, &
332  &zpres, zcf, zsigma_rc, &
333  &zcit, &
334  &zvart, &
335  &zhlc_hcf, zhlc_hrc, &
336  &zhli_hcf, zhli_hri, zrainfr, &
337  &zextpk, zbu_sum, zrrevav, &
338  &zlatham_iaggs)
339  !
340  !* 6. UNPACKING
341  ! ---------
342  !
343  DO jl=1, imicro
344  pcit(i1(jl),i2(jl))=zcit(jl)
345  IF(parami%LWARM) THEN
346  pevap3d(i1(jl),i2(jl))=zrrevav(jl)
347  ENDIF
348  pwr(i1(jl),i2(jl),irv)=zvart(jl, irv)
349  pwr(i1(jl),i2(jl),irc)=zvart(jl, irc)
350  pwr(i1(jl),i2(jl),irr)=zvart(jl, irr)
351  pwr(i1(jl),i2(jl),iri)=zvart(jl, iri)
352  pwr(i1(jl),i2(jl),irs)=zvart(jl, irs)
353  pwr(i1(jl),i2(jl),irg)=zvart(jl, irg)
354  IF (krr==7) THEN
355  pwr(i1(jl),i2(jl),irh)=zvart(jl, irh)
356  ENDIF
357  prainfr(i1(jl),i2(jl))=zrainfr(jl)
358  ENDDO
359  IF(buconf%LBU_ENABLE .OR. osave_micro) THEN
360  DO jv=1, ibunum-ibunum_extra
361  DO jl=1, imicro
362  zbu_pack(jmicro+jl-1, jv) = zbu_sum(jl, jv)
363  ENDDO
364  ENDDO
365  ENDIF
366 
367 
368  ENDDO ! JMICRO
369  ENDIF ! KSIZE > 0
370 
371 ELSE ! PARAMI%LPACK_MICRO
372  !We assume, here, that points outside the physical domain of the model (extral levels,
373  !horizontal points in the halo) contain valid values, sufficiently valid to be used in tests
374  !such as "PTHT(JL)>ZTHRESHOLD .AND. LLMICRO(JL)". In these tests, LLMICRO(JL) will be evaluated
375  !to .FALSE. on these kind of points but valid values for PTHT are needed to prevent crash.
376  !
377  IF (ksize /= d%NIJT*d%NKT) THEN
378  CALL print_msg(nverb_fatal, 'GEN', 'ICE4_PACK', 'ICE4_PACK : KSIZE /= NIJT*NKT')
379  ENDIF
380 
381  !Some arrays must be copied. In order not to waste memory, we re-use temporary arrays
382  !declared for the pack case.
383  ic=0
384  DO jk = 1, ikt
385  DO jij = 1, iijt
386  ic=ic+1
387  i1tot(ic)=jij
388  i2tot(ic)=jk
389  IF (gext_tend) THEN
390  !The th tendency is not related to a mixing ratio change, there is no exn/exnref issue here
391  zextpk(ic, ith)=pths(jij, jk)
392  zextpk(ic, irv)=prvs(jij, jk)
393  zextpk(ic, irc)=prcs(jij, jk)
394  zextpk(ic, irr)=prrs(jij, jk)
395  zextpk(ic, iri)=pris(jij, jk)
396  zextpk(ic, irs)=prss(jij, jk)
397  zextpk(ic, irg)=prgs(jij, jk)
398  IF (krr==7) THEN
399  zextpk(ic, irh)=prhs(jij, jk)
400  ENDIF
401  ENDIF
402  IF(llsigma_rc) THEN
403  !Copy needed because sigma is modified in ice4_stepping
404  zsigma_rc(ic)=psigs(jij, jk)
405  ENDIF
406  ENDDO
407  ENDDO
408  !
409  !When PARAMI%LPACK_MICRO=T, values on the extra levels are not given to ice4_stepping,
410  !so there was not filled in rain_ice.
411  !When PARAMI%LPACK_MICRO=F, we need to complement the work done in rain_ice to provide
412  !valid values on these levels.
413  !The same applies for the first points and last points on the horizontal dimension.
414  IF (iktb /= 1) THEN
415  DO jk=1, iktb-1
416  pwr(:, jk, :)=pwr(:, iktb, :)
417  ENDDO
418  ENDIF
419  IF (ikte /= ikt) THEN
420  DO jk=ikte+1, ikt
421  pwr(:, jk, :)=pwr(:, ikte, :)
422  ENDDO
423  ENDIF
424  IF (iijb /= 1) THEN
425  DO jij=1, iijb-1
426  pwr(jij, :, :)=pwr(iijb, :, :)
427  ENDDO
428  ENDIF
429  IF (iije /= iijt) THEN
430  DO jij=iije+1, iijt
431  pwr(jij, :, :)=pwr(iije, :, :)
432  ENDDO
433  ENDIF
434  !
435  !* 5bis. TENDENCIES COMPUTATION
436  ! ----------------------
437  !
438  CALL ice4_stepping(d, cst, parami, icep, iced, buconf, &
439  &llsigma_rc, ll_aucv_adju, gext_tend, &
440  &ksize, ksize, odmicro, ptstep, &
441  &krr, osave_micro, oelec, &
442  &pexn, prhodref, i1tot, i2tot, &
443  &ppabst, pcldfr, zsigma_rc, &
444  &pcit, &
445  &pwr, &
446  &phlc_hcf, phlc_hrc, &
447  &phli_hcf, phli_hri, prainfr, &
448  &zextpk, zbu_pack, pevap3d, &
449  &zlatham_iaggs)
450 
451 ENDIF ! PARAMI%LPACK_MICRO
452 !
453 !
454 !* 6. SAVE MICROPHYSICAL TENDENCIES USED BY OTHER PHYSICAL PARAMETERIZATIONS
455 ! ----------------------------------------------------------------------
456 !
457 IF (osave_micro) THEN
458  DO jv = 1, ibunum-ibunum_extra
459  DO jl = 1, ksize
460  pmicro_tend(i1tot(jl),i2tot(jl),jv) = zbu_pack(jl,jv)
461  ENDDO
462  ENDDO
463 END IF
464 !
465 !
466 !* 7. BUDGETS
467 ! -------
468 !
469 IF(buconf%LBU_ENABLE) THEN
470  !Budgets for the different processes
471  CALL ice4_budgets(d, parami, buconf, ksize, ptstep, krr, i1tot, i2tot, &
472  plvfact, plsfact, prhodj, pexnref, &
473  prvheni, zbu_pack, &
474  tbudgets, kbudgets)
475 ENDIF
476 IF (lhook) CALL dr_hook('ICE4_PACK', 1, zhook_handle)
477 END SUBROUTINE ice4_pack
478 END MODULE mode_ice4_pack
