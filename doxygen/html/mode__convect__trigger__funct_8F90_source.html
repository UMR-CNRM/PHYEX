14! #########################################################################
15!
16!!**** Determine convective columns as well as the cloudy values of theta,
17!! and qv at the lifting condensation level (LCL)
18!!
19!! PURPOSE
20!! -------
21!! The purpose of this routine is to determine convective columns
22!!
23!!
24!!
25!!** METHOD
26!! ------
27!! Computations are done at every model level starting from bottom.
28!! The use of masks allows to optimise the inner loops (horizontal loops).
29!! What we look for is the undermost unstable level at each grid point.
30!!
31!!
32!!
33!! EXTERNAL
34!! --------
35!! Routine CONVECT_SATMIXRATIO
36!!
37!!
38!! IMPLICIT ARGUMENTS
39!! ------------------
40!! Module MODD_CST
41!! XG ! gravity constant
42!! XP00 ! Reference pressure
43!! XRD, XRV ! Gaz constants for dry air and water vapor
44!! XCPD ! Cpd (dry air)
45!! XTT ! triple point temperature
46!! XBETAW, XGAMW ! constants for vapor saturation pressure
47!!
48!! Module MODD_CONVPAR
49!! XA25 ! reference grid area
50!! XZLCL ! maximum height difference between
51!! ! the surface and the DPL
52!! XZPBL ! minimum mixed layer depth to sustain convection
53!! XWTRIG ! constant in vertical velocity trigger
54!! XCDEPTH ! minimum necessary cloud depth
55!! XNHGAM ! coefficient for buoyancy term in w eq.
56!! ! accounting for nh-pressure
57!!
58!! Module MODD_CONVPAREXT
59!! JCVEXB, JCVEXT ! extra levels on the vertical boundaries
60!!
61!! REFERENCE
62!! ---------
63!!
64!! Book2 of documentation ( routine TRIGGER_FUNCT)
65!! Fritsch and Chappell (1980), J. Atm. Sci., Vol. 37, 1722-1761.
66!!
67!! AUTHOR
68!! ------
69!! P. BECHTOLD * Laboratoire d'Aerologie *
70!!
71!! MODIFICATIONS

74!! Last modified 20/03/97 Select first departure level
75!! that produces a cloud thicker than XCDEPTH
76!! Last modified 12/12/97 add small perturbation

91!

93INTEGER, INTENT(IN) :: KLEV ! vertical loop index
94REAL, DIMENSION(KLON), INTENT(IN) :: PDXDY ! grid area
95REAL, DIMENSION(KLON,KLEV),INTENT(IN) :: PTH, PTHV ! theta, theta_v

97REAL, DIMENSION(KLON,KLEV),INTENT(IN) :: PRV ! vapor mixing ratio

99REAL, DIMENSION(KLON,KLEV),INTENT(IN) :: PZ ! height of grid point (m)
100REAL, DIMENSION(KLON,KLEV),INTENT(IN) :: PW ! vertical velocity
101!
102REAL, DIMENSION(KLON), INTENT(OUT):: PTHLCL ! theta at LCL
103REAL, DIMENSION(KLON), INTENT(OUT):: PTLCL ! temp. at LCL
104REAL, DIMENSION(KLON), INTENT(OUT):: PRVLCL ! vapor mixing ratio at LCL
105REAL, DIMENSION(KLON), INTENT(OUT):: PWLCL ! parcel velocity at LCL
106REAL, DIMENSION(KLON), INTENT(OUT):: PZLCL ! height at LCL (m)
107REAL, DIMENSION(KLON), INTENT(OUT):: PTHVELCL ! environm. theta_v at LCL (K)
108LOGICAL, DIMENSION(KLON), INTENT(OUT):: OTRIG ! logical mask for convection
109INTEGER, DIMENSION(KLON), INTENT(INOUT):: KLCL ! contains vert. index of LCL
110INTEGER, DIMENSION(KLON), INTENT(INOUT):: KDPL ! contains vert. index of DPL
111INTEGER, DIMENSION(KLON), INTENT(INOUT):: KPBL ! contains index of source layer top
112REAL, DIMENSION(KLON), INTENT(OUT):: PCAPE ! CAPE (J/kg) for diagnostics
113!
114!* 0.2 Declarations of local variables :
115!
116INTEGER :: JKK, JK, JKP, JKM, JKDL, JL, JKT, JT! vertical loop index
117INTEGER :: JI ! horizontal loop index
118INTEGER :: IIE, IKB, IKE ! horizontal + vertical loop bounds
119REAL :: ZEPS, ZEPSA ! R_d / R_v, R_v / R_d
120REAL :: ZCPORD, ZRDOCP ! C_pd / R_d, R_d / C_pd
121!
122REAL, DIMENSION(KLON) :: ZTHLCL, ZTLCL, ZRVLCL, & ! locals for PTHLCL,PTLCL
123 ZWLCL, ZZLCL, ZTHVELCL ! PRVLCL, ....
124INTEGER, DIMENSION(KLON) :: IDPL, IPBL, ILCL ! locals for KDPL, ...
125REAL, DIMENSION(KLON) :: ZPLCL ! pressure at LCL
126REAL, DIMENSION(KLON) :: ZZDPL ! height of DPL
127REAL, DIMENSION(KLON) :: ZTHVLCL ! theta_v at LCL = mixed layer value
128REAL, DIMENSION(KLON) :: ZTMIX ! mixed layer temperature
129REAL, DIMENSION(KLON) :: ZEVMIX ! mixed layer water vapor pressure
130REAL, DIMENSION(KLON) :: ZDPTHMIX, ZPRESMIX ! mixed layer depth and pressure
131REAL, DIMENSION(KLON) :: ZCAPE ! convective available energy (m^2/s^2/g)
132REAL, DIMENSION(KLON) :: ZTHEUL ! updraft equiv. pot. temperature (K)
133REAL, DIMENSION(KLON) :: ZLV, ZCPH! specific heats of vaporisation, dry air
134REAL, DIMENSION(KLON) :: ZDP ! pressure between LCL and model layer
135REAL, DIMENSION(KLON) :: ZTOP ! estimated cloud top (m)
136REAL, DIMENSION(KLON,KLEV):: ZCAP ! CAPE at every level for diagnostics
137REAL, DIMENSION(KLON) :: ZWORK1, ZWORK2, ZWORK3 ! work arrays
138LOGICAL, DIMENSION(KLON) :: GTRIG, GTRIG2 ! local arrays for OTRIG
139LOGICAL, DIMENSION(KLON) :: GWORK1 ! work array
140!
141REAL(KIND=jphook) :: zhook_handle
142!
143!-------------------------------------------------------------------------------
144!
145!* 0.3 Compute array bounds
146! --------------------
147!
148IF (lhook) CALL dr_hook('CONVECT_TRIGGER_FUNCT',0,zhook_handle)
149iie = klon
150ikb = 1 + jcvexb
151ike = klev - jcvext
152!
153!
154!* 1. Initialize local variables
155! --------------------------
156!
157zeps = cst%XRD / cst%XRV
158zepsa = cst%XRV / cst%XRD
159zcpord = cst%XCPD / cst%XRD
160zrdocp = cst%XRD / cst%XCPD
161otrig(:) = .false.
162idpl(:) = kdpl(:)
163ipbl(:) = kpbl(:)
164ilcl(:) = klcl(:)
165pwlcl(:) = 0.
166zwlcl(:) = 0.
167pthlcl(:) = 1.
168pthvelcl(:)= 1.
169ptlcl(:) = 1.
170prvlcl(:) = 0.
171pwlcl(:) = 0.
172pzlcl(:) = pz(:,ikb)
173zzdpl(:) = pz(:,ikb)
174gtrig2(:) = .true.
175zcap(:,:) = 0.
176!
177!
178!
179! 1. Determine highest necessary loop test layer
180! -------------------------------------------
181!
182jt = ike - 2
183DO jk = ikb + 1, ike - 2
184 IF ( pz(1,jk) - pz(1,ikb) < 12.e3 ) jt = jk
185END DO
186!
187!
188!* 2. Enter loop for convection test
189! ------------------------------
190!
191jkp = minval( idpl(:) ) + 1
192jkt = jt
193DO jkk = jkp, jkt
194!
195 gwork1(:) = zzdpl(:) - pz(:,ikb) < xzlcl
196 ! we exit the trigger test when the center of the mixed layer is more
197 ! than 3500 m above soil level.
198 WHERE ( gwork1(:) )
199 zdpthmix(:) = 0.
200 zpresmix(:) = 0.
201 zthlcl(:) = 0.
202 zrvlcl(:) = 0.
203 zzdpl(:) = pz(:,jkk)
204 idpl(:) = jkk
205 END WHERE
206!
207!
208!* 3. Construct a mixed layer of at least 60 hPa (XZPBL)
209! ------------------------------------------
210!
211 DO jk = jkk, ike - 1
212 jkm = jk + 1
213 DO ji = 1, iie
214 IF ( gwork1(ji) .AND. zdpthmix(ji) < xzpbl ) THEN
215 ipbl(ji) = jk
216 zwork1(ji) = ppres(ji,jk) - ppres(ji,jkm)
217 zdpthmix(ji) = zdpthmix(ji) + zwork1(ji)
218 zpresmix(ji) = zpresmix(ji) + ppres(ji,jk) * zwork1(ji)
219 zthlcl(ji) = zthlcl(ji) + pth(ji,jk) * zwork1(ji)
220 zrvlcl(ji) = zrvlcl(ji) + prv(ji,jk) * zwork1(ji)
221 END IF
222 END DO
223 IF ( minval( zdpthmix(:) ) >= xzpbl ) EXIT
224 END DO
225!
226!
227 WHERE ( gwork1(:) )
228!
229 zpresmix(:) = zpresmix(:) / zdpthmix(:)
230 zthlcl(:) = zthlcl(:) / zdpthmix(:) +.3 ! add small perturbation
231 zrvlcl(:) = zrvlcl(:) / zdpthmix(:) +1.e-4
232 zthvlcl(:) = zthlcl(:) * ( 1. + zepsa * zrvlcl(:) ) &
233 / ( 1. + zrvlcl(:) )
234!
235!* 4.1 Use an empirical direct solution ( Bolton formula )
236! to determine temperature and pressure at LCL.
237! Nota: the adiabatic saturation temperature is not
238! equal to the dewpoint temperature
239! ----------------------------------------------------
240!
241!
242 ztmix(:) = zthlcl(:) * ( zpresmix(:) / cst%XP00 ) ** zrdocp
243 zevmix(:) = zrvlcl(:) * zpresmix(:) / ( zrvlcl(:) + zeps )
244 zevmix(:) = max( 1.e-8, zevmix(:) )
245 zwork1(:) = log( zevmix(:) / 613.3 )
246 ! dewpoint temperature
247 zwork1(:) = ( 4780.8 - 32.19 * zwork1(:) ) / ( 17.502 - zwork1(:) )
248 ! adiabatic saturation temperature
249 ztlcl(:) = zwork1(:) - ( .212 + 1.571e-3 * ( zwork1(:) - cst%XTT ) &
250 - 4.36e-4 * ( ztmix(:) - cst%XTT ) ) * ( ztmix(:) - zwork1(:) )
251 ztlcl(:) = min( ztlcl(:), ztmix(:) )
252 zplcl(:) = cst%XP00 * ( ztlcl(:) / zthlcl(:) ) ** zcpord
253!
254 END WHERE
255!
256!
257!* 4.2 Correct ZTLCL in order to be completely consistent
258! with MNH saturation formula
259! ---------------------------------------------
260!
261 DO ji = 1, iie
262 CALL convect_satmixratio( zplcl(ji), ztlcl(ji), zeps, zwork1(ji), zlv(ji), zwork2(ji), zcph(ji) )
263 END DO
264 WHERE( gwork1(:) )
265 zwork2(:) = zwork1(:) / ztlcl(:) * ( cst%XBETAW / ztlcl(:) - cst%XGAMW ) ! dr_sat/dT
266 zwork2(:) = ( zwork1(:) - zrvlcl(:) ) / &
267 ( 1. + zlv(:) / zcph(:) * zwork2(:) )
268 ztlcl(:) = ztlcl(:) - zlv(:) / zcph(:) * zwork2(:)
269!
270 END WHERE
271!
272!
273!* 4.3 If ZRVLCL = PRVMIX is oversaturated set humidity
274! and temperature to saturation values.
275! ---------------------------------------------
276!
277 DO ji = 1, iie
278 CALL convect_satmixratio( zpresmix(ji), ztmix(ji), zeps, zwork1(ji), zlv(ji), zwork2(ji), zcph(ji) )
279 END DO
280 WHERE( gwork1(:) .AND. zrvlcl(:) > zwork1(:) )
281 zwork2(:) = zwork1(:) / ztmix(:) * ( cst%XBETAW / ztmix(:) - cst%XGAMW ) ! dr_sat/dT
282 zwork2(:) = ( zwork1(:) - zrvlcl(:) ) / &
283 ( 1. + zlv(:) / zcph(:) * zwork2(:) )
284 ztlcl(:) = ztmix(:) - zlv(:) / zcph(:) * zwork2(:)
285 zrvlcl(:) = zrvlcl(:) - zwork2(:)
286 zplcl(:) = zpresmix(:)
287 zthlcl(:) = ztlcl(:) * ( cst%XP00 / zplcl(:) ) ** zrdocp
288 zthvlcl(:)= zthlcl(:) * ( 1. + zepsa * zrvlcl(:) ) &
289 / ( 1. + zrvlcl(:) )
290 END WHERE
291!
292!
293!* 5.1 Determine vertical loop index at the LCL and DPL
294! --------------------------------------------------
295!
296 DO jk = jkk, ike - 1
297 DO ji = 1, iie
298 IF ( zplcl(ji) <= ppres(ji,jk) .AND. gwork1(ji) ) ilcl(ji) = jk + 1
299 END DO
300 END DO
301!
302!
303!* 5.2 Estimate height and environm. theta_v at LCL
304! --------------------------------------------------
305!
306 DO ji = 1, iie
307 jk = ilcl(ji)
308 jkm = jk - 1
309 zdp(ji) = log( zplcl(ji) / ppres(ji,jkm) ) / &
310 log( ppres(ji,jk) / ppres(ji,jkm) )
311 zwork1(ji) = pthv(ji,jkm) + ( pthv(ji,jk) - pthv(ji,jkm) ) * zdp(ji)
312 ! we compute the precise value of the LCL
313 ! The precise height is between the levels ILCL and ILCL-1.
314 zwork2(ji) = pz(ji,jkm) + ( pz(ji,jk) - pz(ji,jkm) ) * zdp(ji)
315 END DO
316 WHERE( gwork1(:) )
317 zthvelcl(:) = zwork1(:)
318 zzlcl(:) = zwork2(:)
319 END WHERE
320!
321!
322!* 6. Check to see if cloud is bouyant
323! --------------------------------
324!
325!* 6.1 Compute grid scale vertical velocity perturbation term ZWORK1
326! -------------------------------------------------------------
327!
328 ! normalize w grid scale to a 25 km refer. grid
329 DO ji = 1, iie
330 jk = ilcl(ji)
331 jkm = jk - 1
332 jkdl= idpl(ji)
333 !ZWORK1(JI) = ( PW(JI,JKM) + ( PW(JI,JK) - PW(JI,JKM) ) * ZDP(JI) ) &
334 zwork1(ji) = ( pw(ji,jk) + pw(ji,jkdl)*zzlcl(ji)/pz(ji,jkdl) ) * .5 &
335 * sqrt( pdxdy(ji) / xa25 )
336! - 0.02 * ZZLCL(JI) / XZLCL ! avoid spurious convection
337 END DO
338 ! compute sign of normalized grid scale w
339 zwork2(:) = sign( 1., zwork1(:) )
340 zwork1(:) = xwtrig * zwork2(:) * abs( zwork1(:) ) ** 0.333 &
341 * ( cst%XP00 / zplcl(:) ) ** zrdocp
342!
343!* 6.2 Compute parcel vertical velocity at LCL
344! ---------------------------------------
345!
346 DO ji = 1, iie
347 jkdl = idpl(ji)
348 zwork3(ji) = cst%XG * zwork1(ji) * ( zzlcl(ji) - pz(ji,jkdl) ) &
349 / ( pthv(ji,jkdl) + zthvelcl(ji) )
350 END DO
351 WHERE( gwork1(:) )
352 zwlcl(:) = 1. + .5 * zwork2(:) * sqrt( abs( zwork3(:) ) )
353 gtrig(:) = zthvlcl(:) - zthvelcl(:) + zwork1(:) > 0. .AND. &
354 zwlcl(:) > 0.
355 END WHERE
356!
357!
358!* 6.3 Look for parcel that produces sufficient cloud depth.
359! The cloud top is estimated as the level where the CAPE
360! is smaller than a given value (based on vertical velocity eq.)
361! --------------------------------------------------------------
362!
363 ztheul(:) = ztlcl(:) * ( zthlcl(:) / ztlcl(:) ) &
364 ** ( 1. - 0.28 * zrvlcl(:) ) &
365 * exp( ( 3374.6525 / ztlcl(:) - 2.5403 ) * &
366 zrvlcl(:) * ( 1. + 0.81 * zrvlcl(:) ) )
367!
368 zcape(:) = 0.
369 ztop(:) = 0.
370 zwork3(:)= 0.
371 jkm = minval( ilcl(:) )
372 DO jl = jkm, jt
373 jk = jl + 1
374 DO ji = 1, iie
375 zwork1(ji) = ( 2. * ztheul(ji) / &
376 ( pthes(ji,jk) + pthes(ji,jl) ) - 1. ) * ( pz(ji,jk) - pz(ji,jl) )
377 IF ( jl < ilcl(ji) ) zwork1(ji) = 0.
378 ! IF ( JL <= ILCL(JI) ) ZWORK1(JI) = 0.
379 zcape(ji) = zcape(ji) + zwork1(ji)
380 zcap(ji,jkk) = zcap(ji,jkk) + cst%XG * max( 0., zwork1(ji) ) ! actual CAPE
381 zwork2(ji) = xnhgam * cst%XG * zcape(ji) + 1.05 * zwlcl(ji) * zwlcl(ji)
382 ! the factor 1.05 takes entrainment into account
383 zwork2(ji) = sign( 1., zwork2(ji) )
384 zwork3(ji) = zwork3(ji) + min(0., zwork2(ji) )
385 zwork3(ji) = max( -1., zwork3(ji) )
386 ! Nota, the factors ZWORK2 and ZWORK3 are only used to avoid
387 ! if and goto statements, the difficulty is to extract only
388 ! the level where the criterium is first fullfilled
389 ztop(ji) = pz(ji,jl) * .5 * ( 1. + zwork2(ji) ) * ( 1. + zwork3(ji) ) + &
390 ztop(ji) * .5 * ( 1. - zwork2(ji) )
391 END DO
392 END DO
393!
394!
395 WHERE( ztop(:) - zzlcl(:) .GE. xcdepth .AND. gtrig(:) .AND. gtrig2(:) )
396 gtrig2(:) = .false.
397 otrig(:) = gtrig(:) ! we select the first departure level
398 pthlcl(:) = zthlcl(:) ! that gives sufficient cloud depth
399 prvlcl(:) = zrvlcl(:)
400 ptlcl(:) = ztlcl(:)
401 pwlcl(:) = zwlcl(:)
402 pzlcl(:) = zzlcl(:)
403 pthvelcl(:) = zthvelcl(:)
404 kdpl(:) = idpl(:)
405 kpbl(:) = ipbl(:)
406 klcl(:) = ilcl(:)
407 END WHERE
408!
409END DO
410!
411 DO ji = 1, iie
412 pcape(ji) = maxval( zcap(ji,:) ) ! maximum CAPE for diagnostics
413 END DO
414!
415!
416IF (lhook) CALL dr_hook('CONVECT_TRIGGER_FUNCT',1,zhook_handle)
417CONTAINS
418include "convect_satmixratio.h"
419!
420END SUBROUTINE convect_trigger_funct
421END MODULE mode_convect_trigger_funct
